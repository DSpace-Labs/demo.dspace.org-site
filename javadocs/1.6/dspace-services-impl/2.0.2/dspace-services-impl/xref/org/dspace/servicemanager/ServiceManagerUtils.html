8   package org.dspace.servicemanager;

10  import java.io.Serializable;
11  import java.util.Comparator;
12  
13  class ServiceManagerUtils {
14      /***
15       * Compares objects by class name.
16       */
17      public static class ServiceComparator implements Comparator<Object>, Serializable {
18          public static final long serialVersionUID = 1l;
19          public int compare(Object o1, Object o2) {
20              if (o1 != null && o2 != null) {
21                  return o1.getClass().getName().compareTo(o2.getClass().getName());
22              }
23              return 0;
24          }
25      }
26  }
