80      protected Map<String, MapCache> getRequestCaches() {
81          if (requestService == null || requestService.getCurrentRequestId() == null ) {
82              return null;
83          }
84  
85          Map<String, MapCache> requestCaches = requestCachesMap.get(requestService.getCurrentRequestId());
86          if (requestCaches == null) {
87              requestCaches = new HashMap<String, MapCache>();
88              requestCachesMap.put(requestService.getCurrentRequestId(), requestCaches);
89          }
90  
91          return requestCaches;
92      }

219     /* (non-Javadoc)
220      * @see org.dspace.kernel.mixins.InitializedService#init()
221      */
222     public void init() {
223         log.info("init()");
224         // get settings
225         reloadConfig();
226 
227         // make sure we have a cache manager
228         if (cacheManager == null) {
229             // not injected so we need to create one
230             ClassLoader cl = Thread.currentThread().getContextClassLoader();
231             InputStream is = cl.getResourceAsStream(DEFAULT_CONFIG);
232             try {
233                 if (is == null) {
234                     throw new IllegalStateException("Could not init the cache manager, no config file found as a resource in the classloader: " + DEFAULT_CONFIG);
235                 }
236                 cacheManager = new CacheManager(is);
237             } finally {
238                 if (is != null) {
239                     try {
240                         is.close();
241                     } catch (IOException e) {
242                         log.debug("Error closing config stream", e);
243                     }
244                 }
245             }
246         }
247 
248         // get all caches out of the cachemanager and load them into the cache list
249         List<Ehcache> ehcaches = getAllEhCaches(false);
250         for (Ehcache ehcache : ehcaches) {
251             EhcacheCache cache = new EhcacheCache(ehcache, null);
252             cacheRecord.put(cache.getName(), cache);
253         }
254 
255         // load provider
256         reloadProvider();
257 
258         if (requestService != null) {
259             requestService.registerRequestInterceptor(new CachingServiceRequestInterceptor());
260         }
261 
262         log.info("Caching service initialized:\n" + getStatus(null));
263     }

297      * @see org.dspace.services.CachingService#destroyCache(java.lang.String)
298      */
299     public void destroyCache(String cacheName) {
300         if (cacheName == null || "".equals(cacheName)) {
301             throw new IllegalArgumentException("cacheName cannot be null or empty string");
302         }
303 
304         // handle provider first
305         if (getCacheProvider() != null) {
306             try {
307                 getCacheProvider().destroyCache(cacheName);
308             } catch (Exception e) {
309                 log.warn("Failure in provider ("+getCacheProvider()+"): " + e.getMessage());
310             }
311         }
312 
313         EhcacheCache cache = cacheRecord.get(cacheName);
314         if (cache != null) {
315             cacheManager.removeCache(cacheName);
316             cacheRecord.remove(cacheName);
317         } else {
318             Map<String, MapCache> caches = getRequestCaches();
319             if (caches != null) {
320                 caches.remove(cacheName);
321             }
322         }
323     }

325     /* (non-Javadoc)
326      * @see org.dspace.services.CachingService#getCache(java.lang.String, org.dspace.services.model.CacheConfig)
327      */
328     public Cache getCache(String cacheName, CacheConfig cacheConfig) {
329         Cache cache = null;
330 
331         if (cacheName == null || "".equals(cacheName)) {
332             throw new IllegalArgumentException("cacheName cannot be null or empty string");
333         }
334 
335         if (cacheConfig != null && CacheScope.REQUEST.equals(cacheConfig.getCacheScope()) ) {
336             Map<String, MapCache> caches = getRequestCaches();
337             if (caches != null) {
338                 cache = caches.get(cacheName);
339             }
340 
341             if (cache == null) {
342                 cache = instantiateMapCache(cacheName, cacheConfig);
343             }
344         } else {
345             // find the cache in the records if possible
346             cache = this.cacheRecord.get(cacheName);
347 
348             // handle provider
349             if (cache == null && getCacheProvider() != null) {
350                 try {
351                     cache = getCacheProvider().getCache(cacheName, cacheConfig);
352                 } catch (Exception e) {
353                     log.warn("Failure in provider ("+getCacheProvider()+"): " + e.getMessage());
354                 }
355             }
356 
357             if (cache == null) {
358                 cache = instantiateEhCache(cacheName, cacheConfig);
359             }
360         }
361 
362         return cache;
363     }

386     public String getStatus(String cacheName) {
387         final StringBuilder sb = new StringBuilder();
388 
389         if (cacheName == null || "".equals(cacheName)) {
390             // add in overall memory stats
391             sb.append("** Memory report\n");
392             sb.append(" freeMemory: ").append(Runtime.getRuntime().freeMemory());
393             sb.append("\n");
394             sb.append(" totalMemory: ").append(Runtime.getRuntime().totalMemory());
395             sb.append("\n");
396             sb.append(" maxMemory: ").append(Runtime.getRuntime().maxMemory());
397             sb.append("\n");
398 
399             // caches summary report
400             List<Cache> allCaches = getCaches();
401             sb.append("\n** Full report of all known caches (").append(allCaches.size()).append("):\n");
402             for (Cache cache : allCaches) {
403                 sb.append(" * ");
404                 sb.append(cache.toString());
405                 sb.append("\n");
406                 if (cache instanceof EhcacheCache) {
407                     Ehcache ehcache = ((EhcacheCache)cache).getCache();
408                     sb.append(generateCacheStats(ehcache));
409                     sb.append("\n");
410                 }
411             }
412         } else {
413             // report for a single cache
414             sb.append("\n** Report for cache (").append(cacheName).append("):\n");
415             Cache cache = this.cacheRecord.get(cacheName);
416             if (cache == null) {
417                 Map<String, MapCache> caches = getRequestCaches();
418                 if (caches != null) {
419                     cache = caches.get(cacheName);
420                 }
421             }
422             if (cache == null) {
423                 sb.append(" * Could not find cache by this name: ").append(cacheName);
424             } else {
425                 sb.append(" * ");
426                 sb.append(cache.toString());
427                 sb.append("\n");
428                 if (cache instanceof EhcacheCache) {
429                     Ehcache ehcache = ((EhcacheCache)cache).getCache();
430                     sb.append(generateCacheStats(ehcache));
431                     sb.append("\n");
432                 }
433             }
434         }
435 
436         return sb.toString();
437     }

484     /***
485      * Create an EhcacheCache (and the associated EhCache) using the 
486      * supplied name (with default settings), or get the cache out of
487      * Spring or the current configured cache.
488      * <p>
489      * This expects that the cacheRecord has already been checked and 
490      * will not check it again.
491      * <p>
492      * Will proceed in this order:
493      * <ol>
494      *  <li>Attempt to load a bean with the name of the cache</li>
495      *  <li>Attempt to load cache from caching system</li>
496      *  <li>Create a new cache by this name</li>
497      *  <li>Put the cache in the cache record</li>
498      * </ol>
499      * 
500      * @param cacheName the name of the cache
501      * @param cacheConfig the config for this cache
502      * @return a cache instance
503      */
504     protected EhcacheCache instantiateEhCache(String cacheName, CacheConfig cacheConfig) {
505         if (log.isDebugEnabled()) {
506             log.debug("instantiateEhCache(String " + cacheName + ")");
507         }
508 
509         if (cacheName == null || "".equals(cacheName)) {
510             throw new IllegalArgumentException("String cacheName must not be null or empty!");
511         }
512 
513         // try to locate a named cache in the service manager
514         Ehcache ehcache = serviceManager.getServiceByName(cacheName, Ehcache.class);
515 
516         // try to locate the cache in the cacheManager by name
517         if (ehcache == null) {
518             // if this cache name is created or already in use then we just get it
519             if (!cacheManager.cacheExists(cacheName)) {
520                 // did not find the cache
521                 if (cacheConfig == null) {
522                     cacheManager.addCache(cacheName); // create a new cache using ehcache defaults
523                 } else {
524                     if (useClustering) {
525                         // TODO
526                         throw new UnsupportedOperationException("Still need to do this");
527                     } else {
528                         cacheManager.addCache(cacheName); // create a new cache using ehcache defaults
529                     }
530                 }
531                 log.info("Created new Cache (from default settings): " + cacheName);
532             }
533             ehcache = cacheManager.getEhcache(cacheName);
534         }
535         // wrap the ehcache in the cache impl
536         EhcacheCache cache = new EhcacheCache(ehcache, cacheConfig);
537         cacheRecord.put(cacheName, cache);
538         return cache;
539     }

541     /***
542      * Create a thread map cache using the supplied name with supplied 
543      * settings.
544      * <p>
545      * This expects that the cacheRecord has already been checked and 
546      * will not check it again.  It also places the cache into the
547      * request map.
548      * 
549      * @param cacheName the name of the cache
550      * @param cacheConfig the config for this cache
551      * @return a cache instance
552      */
553     protected MapCache instantiateMapCache(String cacheName, CacheConfig cacheConfig) {
554         if (log.isDebugEnabled()) {
555             log.debug("instantiateMapCache(String " + cacheName + ")");
556         }
557 
558         if (cacheName == null || "".equals(cacheName)) {
559             throw new IllegalArgumentException("String cacheName must not be null or empty!");
560         }
561 
562         // check for existing cache
563         MapCache cache = null;
564         CacheScope scope = CacheScope.REQUEST;
565         if (cacheConfig != null) {
566             scope = cacheConfig.getCacheScope();
567         }
568 
569         Map<String, MapCache> caches = getRequestCaches();
570         if (caches != null) {
571             if (CacheScope.REQUEST.equals(scope)) {
572                 cache = caches.get(cacheName);
573             }
574 
575             if (cache == null) {
576                 cache = new MapCache(cacheName, cacheConfig);
577                 // place cache into the right TL
578                 if (CacheScope.REQUEST.equals(scope)) {
579                     caches.put(cacheName, cache);
580                 }
581             }
582         }
583 
584         return cache;
585     }

587     /***
588      * Generate some stats for this cache.
589      * Note that this is not cheap so do not use it very often.
590      *
591      * @param cache an Ehcache
592      * @return the stats of this cache as a string
593      */
594     protected static String generateCacheStats(Ehcache cache) {
595         StringBuilder sb = new StringBuilder();
596         sb.append(cache.getName()).append(":");
597         // this will make this costly but it is important to get accurate settings
598         cache.setStatisticsAccuracy(Statistics.STATISTICS_ACCURACY_GUARANTEED);
599         Statistics stats = cache.getStatistics();
600         final long memSize = cache.getMemoryStoreSize();
601         final long diskSize = cache.getDiskStoreSize();
602         final long size = memSize + diskSize;
603         final long hits = stats.getCacheHits();
604         final long misses = stats.getCacheMisses();
605         final String hitPercentage = ((hits+misses) > 0) ? ((100l * hits) / (hits + misses)) + "%" : "N/A";
606         final String missPercentage = ((hits+misses) > 0) ? ((100l * misses) / (hits + misses)) + "%" : "N/A";
607         sb.append("  Size: ").append(size).append(" [memory:").append(memSize).append(", disk:").append(diskSize).append("]");
608         sb.append(",  Hits: ").append(hits).append(" [memory:").append(stats.getInMemoryHits()).append(", disk:").append(stats.getOnDiskHits()).append("] (").append(hitPercentage).append(")");
609         sb.append(",  Misses: ").append(misses).append(" (").append(missPercentage).append(")");
610         return sb.toString();
611     }
