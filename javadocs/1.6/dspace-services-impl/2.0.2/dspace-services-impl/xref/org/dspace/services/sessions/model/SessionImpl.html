51      /***
52       * Make a session that is associated with the current HTTP request.
53       *
54       * @param request
55       */
56      public SessionImpl(HttpServletRequest request) {
57          if (request == null) {
58              throw new IllegalArgumentException("Cannot create a session without an http request");
59          }
60          this.httpSession = request.getSession(); // establish the session
61          setKeyAttribute(HOST_IP, request.getRemoteAddr());
62          setKeyAttribute(HOST_NAME, request.getRemoteHost());
63      }

65      /***
66       * Make a session which is not associated with the current HTTP request.
67       */
68      public SessionImpl() {
69          // creates a new internal http session that is not cached anywhere
70          this.httpSession = new InternalHttpSession();
71          try {
72              InetAddress i4 = Inet4Address.getLocalHost();
73              setKeyAttribute(HOST_IP, i4.getHostAddress()); // IP address
74              setKeyAttribute(HOST_NAME, i4.getHostName());
75          } catch (UnknownHostException e) {
76              // could not get address
77              setKeyAttribute(HOST_IP, "10.0.0.1"); // set a fake one I guess
78          }

80      }

83       * Set the sessionId.  Normally this should not probably happen much.
84       * @param sessionId

86      public void setSessionId(String sessionId) {
87          if (! isAttributeSet(SESSION_ID)) {

150     /***
151      * @return true if this session is invalidated
152      */
153     public boolean isInvalidated() {
154         boolean invalid = true;
155         if (this.httpSession != null) {
156             try {
157                 this.httpSession.getCreationTime();
158                 invalid = false;
159             } catch (IllegalStateException e) {
160                 invalid = true;
161             }
162         } else {
163             // no httpsession
164             invalid = false;
165         }
166         return invalid;
167     }

169     /***
170      * Handles the general setting of things in the session.
171      * Use this to build other set methods.
172      * Handles checking the session is still valid, and
173      * the checking for null values in the value and key.
174      *
175      * @param key the key to use
176      * @param value the value to set
177      * @return true if the value was set, false if cleared or failure
178      * @throws IllegalArgumentException if the key is null
179      */
180     protected boolean setKeyAttribute(String key, String value) {
181         if (key == null) {
182             throw new IllegalArgumentException("session attribute key cannot be null");
183         }
184         boolean wasSet = false;
185         if (! isInvalidated()) {
186             if (isBlank(value)) {
187                 this.httpSession.removeAttribute(key);
188             } else {
189                 this.httpSession.setAttribute(key, value);
190                 wasSet = true;
191             }
192         }
193         return wasSet;
194     }
195 
196     /***
197      * Handles the general getting of things from the session.
198      * Use this to build other set methods.
199      * Checks that the session is still valid.
200      *
201      * @param key the key to use
202      * @return the value OR null if not found
203      * @throws IllegalArgumentException if the key is null
204      */
205     protected String getKeyAtribute(String key) {
206         if (key == null) {
207             throw new IllegalArgumentException("session attribute key cannot be null");
208         }
209         String value = null;
210         if (! isInvalidated()) {
211             value = (String) this.httpSession.getAttribute(key);
212         }
213         return value;
214     }

216     /***
217      * Handles removal of attributes and related checks.
218      *
219      * @param key the key to use
220      * @throws IllegalArgumentException if the key is null
221      */
222     protected void removeKeyAttribute(String key) {
223         if (key == null) {
224             throw new IllegalArgumentException("session attribute key cannot be null");
225         }
226         if (! isInvalidated()) {
227             this.httpSession.removeAttribute(key);
228         }
229     }

231     @Override
232     public boolean equals(Object obj) {
233         // sessions are equal if the ids are the same, allows comparison across reloaded items
234         if (null == obj) {
235             return false;
236         }
237         if (!(obj instanceof SessionImpl)) {
238             return false;
239         } else {
240             SessionImpl castObj = (SessionImpl) obj;
241             boolean eq;
242             try {
243                 eq = this.getId().equals(castObj.getId());
244             } catch (IllegalStateException e) {
245                 eq = false;
246             }
247             return eq;
248         }
249     }

257     @Override
258     public String toString() {
259         String str;
260         if (isInvalidated()) {
261             str = "invalidated:" + this.httpSession.toString() + ":" + super.toString();
262         } else {
263             str = "active:"+getId()+":user="+getUserId()+"("+getUserEID()+"):sid="+getSessionId()+":server="+getServerId()+":created="+getCreationTime()+":accessed="+getLastAccessedTime()+":maxInactiveSecs="+getMaxInactiveInterval()+":hostIP="+getOriginatingHostIP()+":hostName="+getOriginatingHostName()+":"+super.toString();
264         }
265         return "Session:" + str;
266     }

302     /***
303      * @return a copy of the attributes in this session.
304      * Modifying it has no effect on the session attributes.
305      */
306     @SuppressWarnings("unchecked")
307     public Map<String, String> getAttributes() {
308         Map<String, String> map = new HashMap<String, String>();
309         if (! isInvalidated()) {
310             Enumeration<String> names = this.httpSession.getAttributeNames();
311             while (names.hasMoreElements()) {
312                 String key = names.nextElement();
313                 String value = (String) this.httpSession.getAttribute(key);
314                 map.put(key, value);
315             }
316         }
317         return map;
318     }

471     /***
472      * Compares sessions by the last time they were accessed, with more 
473      * recent first.
474      */
475     public static final class SessionLastAccessedComparator implements Comparator<Session>, Serializable {
476         public static final long serialVersionUID = 1l;
477         public int compare(Session o1, Session o2) {
478             try {
479                 Long lat1 = Long.valueOf(o1.getLastAccessedTime());
480                 Long lat2 = Long.valueOf(o2.getLastAccessedTime());
481                 return lat2.compareTo(lat1); // reverse
482             } catch (Exception e) {
483                 return 0;
484             }
485         }
486     }
