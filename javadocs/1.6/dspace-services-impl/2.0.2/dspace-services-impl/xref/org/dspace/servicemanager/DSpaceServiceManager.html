90      /***
91       * Registers the activators using this service manager.
92       */
93      private void registerActivators() {
94  
95          for (Activator activator : this.getServicesByType(Activator.class))
96          {
97               // succeeded creating the activator
98              try {
99                  activator.start(this);
100                 activators.add(activator);
101                 log.info("Started and registered activator: " + activator.getClass().getName());
102             } catch (Exception e1) {
103                 log.error("ERROR: Failed to start activator ("+ activator.getClass().getName() +"): " + e1, e1);
104             }
105         }
106     }
107 
108     /***
109      * Unregisters all registered activators using this service manager.
110      */
111     private void unregisterActivators() {
112         for (Activator activator : activators) {
113             if (activator != null) {
114                 String activatorClassName = activator.getClass().getName();
115                 // succeeded creating the activator
116                 try {
117                     activator.stop(this);
118                     log.info("Stopped and unregistered activator: " + activatorClassName);
119                 } catch (Exception e1) {
120                     log.error("ERROR: Failed to stop activator ("+activatorClassName+"): " + e1,e1);
121                 }
122             }
123         }
124         activators.clear();
125     }

143     /***
144      * Checks to see if a listener should be notified
145      * @param implementedTypes the types implemented by the service changing
146      * @param serviceChangeListener the listener
147      * @return true if it should be notified, false otherwise
148      */
149     private boolean checkNotifyServiceChange(List<Class<?>> implementedTypes,
150             ServiceChangeListener serviceChangeListener) {
151         boolean notify = false;
152         Class<?>[] notifyTypes = serviceChangeListener.notifyForTypes();
153         if (notifyTypes == null || notifyTypes.length == 0) {
154             notify = true;
155         } else {
156             for (Class<?> notifyType : notifyTypes) {
157                 for (Class<?> implementedType : implementedTypes) {
158                     if (notifyType.equals(implementedType)) {
159                         notify = true;
160                         break;
161                     }
162                 }
163             }
164         }
165         return notify;
166     }

168     /***
169      * Shut down all service managers, including this one.
170      */
171     public void shutdown() {
172         unregisterActivators();
173         for (ServiceManagerSystem sms : serviceManagers) {
174             try {
175                 sms.shutdown();
176             } catch (Exception e) {
177                 // shutdown failures are not great but should NOT cause an interruption of processing
178                 System.err.println("Failure shutting down service manager ("+sms+"): " + e.getMessage());
179             }
180         }
181         this.running = false; // wait til the end
182         this.serviceManagers.clear();
183         this.primaryServiceManager = null;
184         log.info("Shutdown DSpace core service manager");
185     }

187     public void startup() {
188         if (!testing) {
189             // try to load up extra config files for spring
190             String[] extraConfigs = configurationService.getPropertyAsType("service.manager.spring.configs", String[].class);
191             if (extraConfigs != null) {
192                 if (springXmlConfigFiles == null) {
193                     springXmlConfigFiles = extraConfigs;
194                 } else {
195                     springXmlConfigFiles = (String[])ArrayUtils.addAll(springXmlConfigFiles, extraConfigs);
196                 }
197             }
198         }
199         try {
200             // have to put this at the top because otherwise initializing beans will die when they try to use the SMS
201             this.running = true;
202             // create the primary SMS and start it
203             SpringServiceManager springSMS = new SpringServiceManager(this, configurationService, testing, developing, springXmlConfigFiles);
204             try {
205                 springSMS.startup();
206             } catch (Exception e) {
207                 // startup failures are deadly
208                 throw new IllegalStateException("failure starting up spring service manager: " + e.getMessage(), e);
209             }
210             // add it to the list of service managers
211             this.serviceManagers.add(springSMS);
212             this.primaryServiceManager = springSMS;
213 
214             // now startup the activators
215             registerActivators();
216 
217             // now we call the ready mixins
218             notifyServiceManagerReady();
219             
220         } catch (Exception e) {
221             shutdown(); // execute the shutdown
222             String message = "Failed to startup the DSpace Service Manager: " + e.getMessage();
223             System.err.println(message);
224             throw new RuntimeException(message, e);
225         }
226     }

257     public <T> T getServiceByName(String name, Class<T> type) {
258         checkRunning();
259         if (type == null) {
260             throw new IllegalArgumentException("type cannot be null");
261         }
262         T service = null;
263         for (ServiceManagerSystem sms : serviceManagers) {
264             try {
265                 service = sms.getServiceByName(name, type);
266                 if (service != null) {
267                     break;
268                 }
269             } catch (Exception e) {
270                 // keep going
271             }
272         }
273         // need to check the service mixin manager if not found
274         if (service == null 
275                 && name != null) {
276             for (ServiceManagerSystem sms : serviceManagers) {
277                 if (service == null) {
278                     service = sms.getServiceByName(name, type);
279                 }
280             }
281         }
282         return service;
283     }

285     public <T> List<T> getServicesByType(Class<T> type) {
286         checkRunning();
287         if (type == null) {
288             throw new IllegalArgumentException("type cannot be null");
289         }
290         HashSet<T> set = new HashSet<T>();
291         for (ServiceManagerSystem sms : serviceManagers) {
292             try {
293                 set.addAll( sms.getServicesByType(type) );
294             } catch (Exception e) {
295                 // keep going
296             }
297         }
298         // put the set into a list for easier access and sort it
299         List<T> services = new ArrayList<T>(set);
300         Collections.sort(services, new ServiceManagerUtils.ServiceComparator());
301         return services;
302     }

337     public Map<String, Object> getServices() {
338         checkRunning();
339         Map<String, Object> services = new HashMap<String, Object>();
340         for (ServiceManagerSystem sms : serviceManagers) {
341             try {
342                 for (Entry<String, Object> entry : sms.getServices().entrySet()) {
343                     if (! services.containsKey(entry.getKey())) {
344                         services.put(entry.getKey(), entry.getValue());
345                     }
346                 }
347             } catch (Exception e) {
348                 // keep going if it fails for one
349                 System.err.println("Failed to get list of services from service manager ("+sms.getClass()+"): " + e);
350             }
351         }
352         return services;
353     }

355     /*
356      * Handles the configuration push for all services.
357      * Every service gets called to notify them of the config change
358      * depending on the the listener they are using.
359      */
360     public void pushConfig(Map<String, String> properties) {
361         checkRunning();
362         if (properties != null && !properties.isEmpty()) {
363             // load in the new settings to the config service
364             String[] changedNames = configurationService.loadConfiguration(properties, false);
365             if (changedNames.length > 0) {
366                 // some configs changed so push the changes to the listeners in all known services and providers
367                 // make the list of changed setting names and map of changed settings
368                 ArrayList<String> changedSettingNames = new ArrayList<String>();
369                 Map<String, String> changedSettings = new LinkedHashMap<String, String>();
370                 for (String configName : changedNames) {
371                     changedSettingNames.add(configName);
372                     changedSettings.put( getSimplerName(configName), configurationService.getProperty(configName) );
373                 }
374                 // notify the services that implement the mixin
375                 for (ServiceManagerSystem sms : serviceManagers) {
376                     List<ConfigChangeListener> configChangeListeners = sms.getServicesByType(ConfigChangeListener.class);
377                     for (ConfigChangeListener configChangeListener : configChangeListeners) {
378                         String serviceImplName = configChangeListener.getClass().getName();
379                         // notify this service
380                         try {
381                             boolean notify = false;
382                             String[] notifyNames = configChangeListener.notifyForConfigNames();
383                             if (notifyNames == null || notifyNames.length == 0) {
384                                 notify = true;
385                             } else {
386                                 for (String notifyName : notifyNames) {
387                                     // check to see if the change was one of the bean properties for our service
388                                     String simplerName = getSimplerName(notifyName);
389                                     String notifyBeanName = DSpaceConfig.getBeanName(notifyName);
390                                     if (notifyBeanName != null && notifyBeanName.equals(serviceImplName)) {
391                                         // this is a bean key
392                                         notify = true;
393                                         break;
394                                     }
395                                     // check to see if the name matches one of those the listener cares about
396                                     for (String changedName : changedNames) {
397                                         if (simplerName != null && simplerName.equals(changedName)) {
398                                             notify = true;
399                                             break;
400                                         }
401                                     }
402                                 }
403                             }
404                             // do the notify if we should at this point
405                             if (notify) {
406                                 configChangeListener.configurationChanged(changedSettingNames, changedSettings);
407                             }
408                         } catch (Exception e) {
409                             System.err.println("Failure occurred while trying to notify service of config change: " + e.getMessage());
410                         }
411                     }
412                 }
413             }
414         }
415     }

417     /***
418      * @param key a DSpace config key
419      * @return the simpler name (property name or key without property)
420      */
421     protected static String getSimplerName(String key) {
422         String simpleName = key;
423         if (key != null) {
424             String propertyName = DSpaceConfig.getBeanProperty(key);
425             if (propertyName != null) {
426                 simpleName = propertyName;
427             }
428         }
429         return simpleName;
430     }

434     /***
435      * Adds configuration settings into services if possible.
436      * Skips any that are invalid.
437      * 
438      * @param serviceName the name of the service
439      * @param service the service object
440      * @param serviceNameConfigs all known service configuration settings (from the DS config service impl)
441      */
442     public static void configureService(String serviceName, Object service, Map<String, Map<String, ServiceConfig>> serviceNameConfigs) {
443         // stuff the config settings into the bean if there are any
444         if (serviceNameConfigs.containsKey(serviceName)) {
445             BeanWrapper beanWrapper = PropertyAccessorFactory.forBeanPropertyAccess(service);
446 
447             Map<String, ServiceConfig> configs = serviceNameConfigs.get(serviceName);
448             for (ServiceConfig config : configs.values()) {
449                 try {
450                     beanWrapper.setPropertyValue(config.getParamName(), config.getValue());
451 
452                     log.info("Set param ("+config.getParamName()+") on service bean ("+serviceName+") to: " + config.getValue());
453                 } catch (RuntimeException e) {
454                     log.error("Unable to set param ("+config.getParamName()+") on service bean ("+serviceName+"): " + e.getMessage(), e);
455                 }
456             }
457         }
458     }
