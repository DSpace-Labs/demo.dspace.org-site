121      * @see org.dspace.services.ConfigurationService#getPropertyAsType(java.lang.String, java.lang.Object, boolean)

123     @SuppressWarnings("unchecked")
124     public <T> T getPropertyAsType(String name, T defaultValue, boolean setDefaultIfNotFound) {
125         String value = getProperty(name);
126         T property = null;
127         if (defaultValue == null) {
128             property = null; // just return null when default value is null
129         } else if (value == null) {
130             property = defaultValue; // just return the default value if nothing is currently set
131             // also set the default value as the current stored value
132             if (setDefaultIfNotFound) {
133                 setProperty(name, defaultValue);
134             }
135         } else {
136             // something is already set so we convert the stored value to match the type
137             property = (T)convert(value, defaultValue.getClass());
138         }
139         return property;
140     }

145      * @see org.dspace.services.ConfigurationService#setProperty(java.lang.String, java.lang.Object)
146      */
147     public boolean setProperty(String name, Object value) {
148         if (name == null) {
149             throw new IllegalArgumentException("name cannot be null for setting configuration");
150         }
151         boolean changed = false;
152         if (value == null) {
153             changed = this.configuration.remove(name) != null;
154             log.info("Cleared the configuration setting for name ("+name+")");
155         } else {
156             SimpleTypeConverter converter = new SimpleTypeConverter();
157             String sVal = (String)converter.convertIfNecessary(value, String.class);
158             changed = loadConfig(name, sVal);
159         }
160         return changed;
161     }

204     /***
205      * Load a series of properties into the configuration.
206      * Checks to see if the settings exist or are changed and only loads 
207      * changes.  Clears out existing ones depending on the setting.
208      * 
209      * @param properties a map of key -> value strings 
210      * @param clear if true then clears the existing configuration settings first
211      * @return the list of changed configuration names
212      */
213     public String[] loadConfiguration(Map<String, String> properties, boolean clear) {
214         if (properties == null) {
215             throw new IllegalArgumentException("properties cannot be null");
216         }
217         // transform to configs and call load
218         ArrayList<DSpaceConfig> dspaceConfigs = new ArrayList<DSpaceConfig>();
219         for (Entry<String, String> entry : properties.entrySet()) {
220             String key = entry.getKey();
221             if (key != null &&  ! "".equals(key)) {
222                 String val = entry.getValue();
223                 if (val != null &&  ! "".equals(val)) {
224                     dspaceConfigs.add( new DSpaceConfig(entry.getKey(), entry.getValue()) );
225                 }
226             }
227         }
228         return loadConfiguration(dspaceConfigs, clear);
229     }

231     /***
232      * Load up a bunch of {@link DSpaceConfig}s into the configuration.
233      * Checks to see if the settings exist or are changed and only
234      * loads changes.  Clears out existing ones depending on the setting.
235      * 
236      * @param dspaceConfigs a list of {@link DSpaceConfig} objects
237      * @param clear if true then clears the existing configuration settings first
238      * @return the list of changed configuration names
239      */
240     public String[] loadConfiguration(List<DSpaceConfig> dspaceConfigs, boolean clear) {
241         ArrayList<String> changed = new ArrayList<String>();
242         if (clear) {
243             this.configuration.clear();
244         }
245         for (DSpaceConfig config : dspaceConfigs) {
246             String key = config.getKey();
247             boolean same = true;
248             if (clear) {
249                 // all are new
250                 same = false;
251             } else {
252                 if (this.configuration.containsKey(key)) {
253                     if (this.configuration.get(key).equals(config)) {
254                         // this one has changed
255                         same = false;
256                     }
257                 } else {
258                     // this one is new
259                     same = false;
260                 }
261             }
262             if (!same) {
263                 changed.add(key);
264                 this.configuration.put(key, config);
265             }
266         }
267         if (changed.size() > 0) {
268             replaceVariables(this.configuration);
269             // refresh the configs
270             serviceNameConfigs = makeServiceNameConfigs();
271         }
272         return changed.toArray(new String[changed.size()]);
273     }

275     /***
276      * Loads an additional config setting into the system.
277      * @param key
278      * @param value
279      * @return true if the config is new or changed
280      */
281     public boolean loadConfig(String key, String value) {
282         if (key == null) {
283             throw new IllegalArgumentException("key cannot be null");
284         }
285         // update replacements and add
286         boolean changed = replaceAndAddConfig( new DSpaceConfig(key, value) );
287         if (changed) {
288             // refresh the configs
289             serviceNameConfigs = makeServiceNameConfigs();
290         }
291         return changed;
292     }

305     /***
306      * Loads up the default initial configuration from the dspace config 
307      * files in the file home and on the classpath.
308      */
309     public void loadInitialConfig(String providedHome) {
310         Map<String, String> configMap = new LinkedHashMap<String, String>();
311         // load default settings
312         try {
313             String defaultServerId = InetAddress.getLocalHost().getHostName();
314             configMap.put("serverId", defaultServerId);
315         } catch (UnknownHostException e) {
316             // oh well
317         }
318         // default is testing mode off
319         configMap.put(Constants.DSPACE_TESTING_MODE, "false");
320 
321         // now we load the settings from properties files
322         String homePath = System.getProperty(DSPACE_HOME);
323         
324 		// now we load from the provided parameter if its not null
325         if (providedHome != null && homePath == null) {
326 			homePath = providedHome;
327 		}
328         
329         if (homePath == null) {
330             String catalina = getCatalina();
331             if (catalina != null) {
332                 homePath = catalina + File.separatorChar + DSPACE + File.separatorChar;
333             }
334         }
335         if (homePath == null) {
336             homePath = System.getProperty("user.home");
337         }
338         if (homePath == null) {
339             homePath = "/";
340         }
341 
342         // make sure it's set properly
343         //System.setProperty(DSPACE_HOME, homePath);
344         configMap.put(DSPACE_HOME, homePath);
345 
346         // LOAD the internal defaults
347         Properties defaultProps = readPropertyResource(DEFAULT_DSPACE_CONFIG_PATH);
348         if (defaultProps.size() <= 0) {
349             // failed to load defaults!
350             throw new RuntimeException("Failed to load default dspace config properties: " + DEFAULT_DSPACE_CONFIG_PATH);
351         }
352         pushPropsToMap(configMap, defaultProps);
353 
354         // load all properties from the system which begin with the prefix
355         Properties systemProps = System.getProperties();
356         for (Object o : systemProps.keySet()) {
357             String key = (String) o;
358             if (key != null
359                     && ! key.equals(DSPACE_HOME)) {
360                 try {
361                     if (key.startsWith(DSPACE_PREFIX)) {
362                         String propName = key.substring(DSPACE_PREFIX.length());
363                         String propVal = systemProps.getProperty(key);
364                         log.info("Loading system property as config: "+propName+"=>"+propVal);
365                         configMap.put(propName, propVal);
366                     }
367                 } catch (RuntimeException e) {
368                     log.error("Failed to properly get config value from system property: " + o, e);
369                 }
370             }
371         }
372         
373         // Collect values from all the properties files: the later ones loaded override settings from prior.
374 
375         // read all the known files from the home path that are properties files
376         pushPropsToMap(configMap, readPropertyFile(homePath + File.separatorChar + LEGACY_DSPACE_CONFIG_PATH));
377         pushPropsToMap(configMap, readPropertyFile(homePath + File.separatorChar + DSPACE_CONFIG_PATH));
378         pushPropsToMap(configMap, readPropertyFile(homePath + "local" + DOT_PROPERTIES));
379 
380         // attempt to load from the current classloader also (works for commandline config sitting on classpath
381         pushPropsToMap(configMap, readPropertyResource(DSPACE + DOT_CONFIG));
382         pushPropsToMap(configMap, readPropertyResource(DSPACE + DOT_PROPERTIES));
383         pushPropsToMap(configMap, readPropertyResource("local" + DOT_PROPERTIES));
384         pushPropsToMap(configMap, readPropertyResource("webapp" + DOT_PROPERTIES));
385 
386         // now push all of these into the config service store
387         loadConfiguration(configMap, true);
388         log.info("Started up configuration service and loaded "+configMap.size()+" settings");
389     }

392     /***
393      * Adds in this DSConfig and then updates the config by checking for 
394      * replacements everywhere else.
395      * @param dsConfig a DSConfig to update the value of and then add in to the main config
396      * @return true if the config changed or is new
397      */
398     protected boolean replaceAndAddConfig(DSpaceConfig dsConfig) {
399         DSpaceConfig newConfig = null;
400         String key = dsConfig.getKey();
401         if (dsConfig.getValue().contains("${")) {
402             String value = dsConfig.getValue();
403             int start = -1;
404             while ((start = value.indexOf("${")) > -1) {
405                 int end = value.indexOf('}', start);
406                 if (end > -1) {
407                     String newKey = value.substring(start+2, end);
408                     if (newKey.equals(key)) {
409                         log.warn("Found circular reference for key ("+newKey+") in config value: " + value);
410                         break;
411                     }
412                     DSpaceConfig dsc = this.configuration.get(newKey);
413                     if (dsc == null) {
414                         log.warn("Could not find key ("+newKey+") for replacement in value: " + value);
415                         break;
416                     }
417                     String newVal = dsc.getValue();
418                     value = value.replace("${"+newKey+"}", newVal);
419                     newConfig = new DSpaceConfig(key, value);
420                 } else {
421                     log.warn("Found '${' but could not find a closing '}' in the value: " + value);
422                     break;
423                 }
424             }
425         }
426 
427         // add the config
428         if (this.configuration.containsKey(key) && this.configuration.get(key).equals(dsConfig)) {
429             return false; // SHORT CIRCUIT
430         }
431 
432         // config changed or new
433         this.configuration.put(key, newConfig != null ? newConfig : dsConfig);
434         // update replacements
435         replaceVariables(this.configuration);
436         return true;
437     }

439     /***
440      * This will replace the ${key} with the value from the matching key 
441      * if it exists.  Logs a warning if the key does not exist.
442      * Goes through and updates the replacements for the the entire 
443      * configuration and updates any replaced values.
444      */
445     protected void replaceVariables(Map<String, DSpaceConfig> dsConfiguration) {
446         for (Entry<String, DSpaceConfig> entry : dsConfiguration.entrySet()) {
447             if (entry.getValue().getValue().contains("${")) {
448                 String value = entry.getValue().getValue();
449                 int start = -1;
450                 while ((start = value.indexOf("${")) > -1) {
451                     int end = value.indexOf('}', start);
452                     if (end > -1) {
453                         String newKey = value.substring(start+2, end);
454                         if (newKey.equals(entry.getKey())) {
455                             log.warn("Found circular reference for key ("+newKey+") in config value: " + value);
456                             break;
457                         }
458                         DSpaceConfig dsc = dsConfiguration.get(newKey);
459                         if (dsc == null) {
460                             log.warn("Could not find key ("+newKey+") for replacement in value: " + value);
461                             break;
462                         }
463                         String newVal = dsc.getValue();
464                         value = value.replace("${"+newKey+"}", newVal);
465                         entry.setValue( new DSpaceConfig(entry.getValue().getKey(), value) );
466                     } else {
467                         log.warn("Found '${' but could not find a closing '}' in the value: " + value);
468                         break;
469                     }
470                 }
471             }
472         }
473     }

475     protected Properties readPropertyFile(String filePathName) {
476         Properties props = new Properties();
477         InputStream is = null;
478         try {
479             File f = new File(filePathName);
480             if (f.exists()) {
481                 is = new FileInputStream(f);
482                 props.load(is);
483                 log.info("Loaded "+props.size()+" config properties from file: " + f);
484             }
485             else
486             {
487             	log.info("Failed to load config properties from file ("+filePathName+"): Does not exist");
488                 	
489             }
490         } catch (Exception e) {
491             log.warn("Failed to load config properties from file ("+filePathName+"): " + e.getMessage(), e);
492         } finally {
493             if (is != null) {
494                 try {
495                     is.close();
496                 } catch (IOException ioe) {
497                     // Ignore exception on close
498                 }
499             }
500         }
501         return props;
502     }

542     /***
543      * Constructs service name configs map for fast lookup of service 
544      * configurations.
545      * @return the map of config service settings
546      */
547     public Map<String, Map<String, ServiceConfig>> makeServiceNameConfigs() {
548         Map<String, Map<String, ServiceConfig>> serviceNameConfigs = new HashMap<String, Map<String,ServiceConfig>>();
549         for (DSpaceConfig dsConfig : getConfiguration()) {
550             String beanName = dsConfig.getBeanName();
551             if (beanName != null) {
552                 Map<String, ServiceConfig> map = null;
553                 if (serviceNameConfigs.containsKey(beanName)) {
554                     map = serviceNameConfigs.get(beanName);
555                 } else {
556                     map = new HashMap<String, ServiceConfig>();
557                     serviceNameConfigs.put(beanName, map);
558                 }
559                 map.put(beanName, new ServiceConfig(dsConfig));
560             }
561         }
562         return serviceNameConfigs;
563     }

565     private <T> T convert(String value, Class<T> type) {
566         SimpleTypeConverter converter = new SimpleTypeConverter();
567 
568         if (value != null) {
569             if (type.isArray()) {
570                 String[] values = value.split(",");
571                 return (T)converter.convertIfNecessary(values, type);
572             }
573 
574             if (type.isAssignableFrom(String.class)) {
575                 return (T)value;
576             }
577         } else {
578             if (boolean.class.equals(type)) {
579                 return (T)Boolean.FALSE;
580             } else if (int.class.equals(type) || long.class.equals(type)) {
581                 return (T)converter.convertIfNecessary(0, type);
582             }
583         }
584 
585         return (T)converter.convertIfNecessary(value, type);
586     }
