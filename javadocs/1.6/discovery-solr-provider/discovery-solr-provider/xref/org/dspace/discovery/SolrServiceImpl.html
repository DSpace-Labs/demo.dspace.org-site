109     protected CommonsHttpSolrServer getSolr() throws java.net.MalformedURLException, org.apache.solr.client.solrj.SolrServerException
110     {
111 
112 
113         if ( solr == null)
114         {
115             String solrService = configurationService.getProperty("solr.search.server") ;
116 
117             if(solrService == null)
118             {
119                 solrService = "http://localhost:8080/solr/search";
120             }
121 
122             log.debug("Solr URL: " + solrService);
123                     solr = new CommonsHttpSolrServer(solrService);
124 
125             solr.setBaseURL(solrService);
126 
127             SolrQuery solrQuery = new SolrQuery()
128                         .setQuery("search.resourcetype:2 AND search.resourceid:1");
129 
130             solr.query(solrQuery);
131         }
132 
133         return solr;
134     }

151     /***
152      * If the handle for the "dso" already exists in the index, and the "dso"
153      * has a lastModified timestamp that is newer than the document in the index
154      * then it is updated, otherwise a new document is added.
155      *
156      * @param context Users Context
157      * @param dso     DSpace Object (Item, Collection or Community
158      * @param force   Force update even if not stale.
159      * @throws SQLException
160      * @throws IOException
161      */
162     public void indexContent(Context context, DSpaceObject dso,
163                              boolean force) throws SQLException {
164 
165         String handle = dso.getHandle();
166 
167         if (handle == null) {
168             handle = HandleManager.findHandle(context, dso);
169         }
170 
171         try {
172             switch (dso.getType()) {
173                 case Constants.ITEM:
174                     Item item = (Item) dso;
175                     if (item.isArchived() && !item.isWithdrawn()) {
176                         /***
177                          * If the item is in the repository now, add it to the index
178                          */
179                         if (requiresIndexing(handle, ((Item) dso).getLastModified())
180                                 || force) {
181                             unIndexContent(context, handle);
182                             buildDocument(context, (Item) dso);
183                         }
184                     } else {
185                         /***
186                          * Make sure the item is not in the index if it is not in
187                          * archive. TODO: Someday DSIndexer should block withdrawn
188                          * content on search/retrieval and allow admins the ablitity
189                          * to still search for withdrawn Items.
190                          */
191                         unIndexContent(context, handle);
192                         log.info("Removed Item: " + handle + " from Index");
193                     }
194                     break;
195 
196                 case Constants.COLLECTION:
197                     buildDocument(context, (Collection) dso);
198                     log.info("Wrote Collection: " + handle + " to Index");
199                     break;
200 
201                 case Constants.COMMUNITY:
202                     buildDocument(context, (Community) dso);
203                     log.info("Wrote Community: " + handle + " to Index");
204                     break;
205 
206                 default:
207                     log
208                             .error("Only Items, Collections and Communities can be Indexed");
209             }
210 
211         } catch (Exception e) {
212             log.error(e.getMessage(), e);
213         }
214         try {
215             getSolr().commit();
216         } catch (SolrServerException e) {
217             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
218         } catch (IOException e) {
219             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
220         }
221     }

300     /***
301      * Iterates over all Items, Collections and Communities. And updates them in
302      * the index. Uses decaching to control memory footprint. Uses indexContent
303      * and isStale ot check state of item in index.
304      * <p/>
305      * At first it may appear counterintuitive to have an IndexWriter/Reader
306      * opened and closed on each DSO. But this allows the UI processes to step
307      * in and attain a lock and write to the index even if other processes/jvms
308      * are running a reindex.
309      *
310      * @param context
311      * @param force
312      */
313     public void updateIndex(Context context, boolean force) {
314         try {
315             ItemIterator items = null;
316             try {
317                 for (items = Item.findAll(context); items.hasNext();) {
318                     Item item = (Item) items.next();
319                     indexContent(context, item, force);
320                     item.decache();
321                 }
322             } finally {
323                 if (items != null)
324                     items.close();
325             }
326 
327             Collection[] collections = Collection.findAll(context);
328             for (int i = 0; i < collections.length; i++) {
329                 indexContent(context, collections[i], force);
330                 context.removeCached(collections[i], collections[i].getID());
331 
332             }
333 
334             Community[] communities = Community.findAll(context);
335             for (int i = 0; i < communities.length; i++) {
336                 indexContent(context, communities[i], force);
337                 context.removeCached(communities[i], communities[i].getID());
338             }
339 
340 
341         } catch (Exception e) {
342             log.error(e.getMessage(), e);
343         }
344     }

346     /***
347      * Iterates over all documents in the Lucene index and verifies they are in
348      * database, if not, they are removed.
349      *
350      * @param force
351      * @throws IOException
352      * @throws SQLException
353      * @throws SolrServerException
354      */
355     public void cleanIndex(boolean force) throws IOException,
356             SQLException, SearchServiceException {
357 
358         Context context = new Context();
359         context.turnOffAuthorisationSystem();
360 
361         try
362         {
363             if (force) {
364                 getSolr().deleteByQuery("*:*");
365             } else {
366                 SolrQuery query = new SolrQuery();
367                 query.setQuery("*:*");
368                 QueryResponse rsp = getSolr().query(query);
369                 SolrDocumentList docs = rsp.getResults();
370 
371                 Iterator iter = docs.iterator();
372                 while (iter.hasNext()) {
373 
374                  SolrDocument doc = (SolrDocument) iter.next();
375 
376                 String handle = (String) doc.getFieldValue("handle");
377 
378                 DSpaceObject o = HandleManager.resolveToObject(context, handle);
379 
380                 if (o == null) {
381                     log.info("Deleting: " + handle);
382                     /*
383                           * Use IndexWriter to delete, its easier to manage
384                           * write.lock
385                           */
386                     unIndexContent(context, handle);
387                 } else {
388                     context.removeCached(o, o.getID());
389                     log.debug("Keeping: " + handle);
390                 }
391             }
392             }
393         } catch(Exception e){
394 
395             throw new SearchServiceException(e.getMessage(), e);
396         } finally
397         {
398             context.abort();
399         }

410     private void emailException(Exception exception) {
411         // Also email an alert, system admin may need to check for stale lock
412         try {
413             String recipient = ConfigurationManager
414                     .getProperty("alert.recipient");
415 
416             if (recipient != null) {
417                 Email email = ConfigurationManager
418                         .getEmail(I18nUtil.getEmailFilename(
419                                 Locale.getDefault(), "internal_error"));
420                 email.addRecipient(recipient);
421                 email.addArgument(ConfigurationManager
422                         .getProperty("dspace.url"));
423                 email.addArgument(new Date());
424 
425                 String stackTrace;
426 
427                 if (exception != null) {
428                     StringWriter sw = new StringWriter();
429                     PrintWriter pw = new PrintWriter(sw);
430                     exception.printStackTrace(pw);
431                     pw.flush();
432                     stackTrace = sw.toString();
433                 } else {
434                     stackTrace = "No exception";
435                 }
436 
437                 email.addArgument(stackTrace);
438                 email.send();
439             }
440         } catch (Exception e) {
441             // Not much we can do here!
442             log.warn("Unable to send email alert", e);
443         }
444 
445     }

448     /***
449      * Is stale checks the lastModified time stamp in the database and the index
450      * to determine if the index is stale.
451      *
452      * @param handle
453      * @param lastModified
454      * @return
455      * @throws SQLException
456      * @throws IOException
457      * @throws SolrServerException
458      */
459     private boolean requiresIndexing(String handle, Date lastModified)
460             throws SQLException, IOException, SearchServiceException {
461 
462         boolean reindexItem = false;
463         boolean inIndex = false;
464 
465         SolrQuery query = new SolrQuery();
466         query.setQuery("handle:" + handle);
467         QueryResponse rsp = null;
468 
469         try {
470             rsp = getSolr().query(query);
471         } catch (SolrServerException e) {
472             throw new SearchServiceException(e.getMessage(),e);
473         }
474 
475         for (SolrDocument doc : rsp.getResults()) {
476 
477             inIndex = true;
478 
479             Object value = doc.getFieldValue(LAST_INDEXED_FIELD);
480 
481             if(value instanceof Date)
482             {
483                 Date lastIndexed = (Date) value;
484 
485                 if (lastIndexed == null
486                     || lastIndexed.before(lastModified)) {
487 
488                     reindexItem = true;
489                 }
490             }
491         }
492 
493         return reindexItem || !inIndex;
494     }

606     /***
607      * Build a Lucene document for a DSpace Item and write the index
608      *
609      * @param context Users Context
610      * @param item    The DSpace Item to be indexed
611      * @throws SQLException
612      * @throws IOException
613      */
614     private void buildDocument(Context context, Item item)
615             throws SQLException, IOException {
616         String handle = item.getHandle();
617 
618         if (handle == null) {
619             handle = HandleManager.findHandle(context, item);
620         }
621 
622         // get the location string (for searching by collection & community)
623         List<String> locations = getItemLocations(context, item);
624 
625         SolrInputDocument doc = buildDocument(Constants.ITEM, item.getID(), handle,
626                 locations);
627 
628         log.debug("Building Item: " + handle);
629 
630 
631         try {
632 
633             DCValue[] mydc = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
634 
635             for(DCValue meta : mydc)
636             {
637                 String field = meta.schema + "." + meta.element;
638 
639                 String value = meta.value;
640 
641                 if (meta.element.equals("date") && meta.schema.equals("dc"))
642                 {
643                     try{
644                         value = DateFormatUtils.formatUTC(this.toDate(value), "yyyy-MM-dd'T'HH:mm:ss'Z'");
645                     } catch (Exception e)  {
646                         log.error(e.getMessage(), e);
647                     }
648                 }
649                 
650                 if(meta.qualifier != null && !meta.qualifier.trim().equals("")) {
651                     field += "." + meta.qualifier;
652                 }
653 
654                 doc.addField(field, value);
655 
656                 if(meta.language != null && !meta.language.trim().equals("")) {
657                     field += "." + meta.language;
658                     doc.addField(field, value);
659                 }
660             }
661 
662         } catch (Exception e)  {
663             log.error(e.getMessage(), e);
664         }
665 
666 
667         log.debug("  Added Metadata");
668 
669         try {
670 
671             DCValue[] values = item.getMetadata("dc.relation.ispartof");
672 
673             if(values != null && values.length > 0 && values[0] != null && values[0].value != null)
674             {
675                 // group on parent
676                 doc.addField("publication_grp",values[0].value.replaceFirst("http://hdl.handle.net/","") );
677                 
678             }
679             else
680             {
681                 // group on self
682                 doc.addField("publication_grp", item.getHandle());
683             }
684 
685         } catch (Exception e){
686             log.error(e.getMessage(),e);
687         }
688 
689 
690         log.debug("  Added Grouping");
691 
692 
693         Vector<InputStreamReader> readers = new Vector<InputStreamReader>();
694 
695         try {
696             // now get full text of any bitstreams in the TEXT bundle
697             // trundle through the bundles
698             Bundle[] myBundles = item.getBundles();
699 
700             for (int i = 0; i < myBundles.length; i++) {
701                 if ((myBundles[i].getName() != null)
702                         && myBundles[i].getName().equals("TEXT")) {
703                     // a-ha! grab the text out of the bitstreams
704                     Bitstream[] myBitstreams = myBundles[i].getBitstreams();
705 
706                     for (int j = 0; j < myBitstreams.length; j++) {
707                         try {
708                             InputStreamReader is = new InputStreamReader(
709                                     myBitstreams[j].retrieve()); // get input
710                             readers.add(is);
711 
712                             // Add each InputStream to the Indexed Document
713                             // (Acts like an Append)
714 //							doc.addField("default", is);
715                             //doc.add(new Field("default", is));
716 
717                             log.debug("  Added BitStream: "
718                                     + myBitstreams[j].getStoreNumber() + "	"
719                                     + myBitstreams[j].getSequenceID() + "   "
720                                     + myBitstreams[j].getName());
721 
722                         } catch (Exception e) {
723                             // this will never happen, but compiler is now
724                             // happy.
725                             log.error(e.getMessage(), e);
726                         }
727                     }
728                 }
729             }
730         } catch (Exception e) {
731             log.error(e.getMessage(), e);
732         }
733 
734         // write the index and close the inputstreamreaders
735         try {
736             writeDocument(doc);
737             log.info("Wrote Item: " + handle + " to Index");
738         } catch (Exception e) {
739             log.error(e.getMessage(), e);
740         } finally {
741             Iterator<InputStreamReader> itr = readers.iterator();
742             while (itr.hasNext()) {
743                 InputStreamReader reader = itr.next();
744                 if (reader != null) {
745                     reader.close();
746                 }
747             }
748             log.debug("closed " + readers.size() + " readers");
749         }
750     }

752     /***
753      * Create Lucene document with all the shared fields initialized.
754      *
755      * @param type      Type of DSpace Object
756      * @param id
757      * @param handle
758      * @param locations @return
759      */
760     private SolrInputDocument buildDocument(int type, int id, String handle,
761                                             List<String> locations) {
762         SolrInputDocument doc = new SolrInputDocument();
763 
764         // want to be able to check when last updated
765         // (not tokenized, but it is indexed)
766         doc.addField(LAST_INDEXED_FIELD, new Date());
767 
768         // New fields to weaken the dependence on handles, and allow for faster
769         // list display
770         doc.addField("search.resourcetype", Integer.toString(type));
771 
772         doc.addField("search.resourceid", Integer.toString(id));
773 
774         // want to be able to search for handle, so use keyword
775         // (not tokenized, but it is indexed)
776         if (handle != null) {
777             // want to be able to search for handle, so use keyword
778             // (not tokenized, but it is indexed)
779             doc.addField("handle", handle);
780         }

802     public static Date toDate(String t) {
803         SimpleDateFormat[] dfArr;
804 
805         // Choose the likely date formats based on string length
806         switch (t.length()) {
807             case 4:
808                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat("yyyy")};
809                 break;
810             case 6:
811                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat("yyyyMM")};
812                 break;
813             case 7:
814                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat("yyyy-MM")};
815                 break;
816             case 8:
817                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat("yyyyMMdd"),
818                         new SimpleDateFormat("yyyy MMM")};
819                 break;
820             case 10:
821                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat("yyyy-MM-dd")};
822                 break;
823             case 11:
824                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat("yyyy MMM dd")};
825                 break;
826             case 20:
827                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat(
828                         "yyyy-MM-dd'T'HH:mm:ss'Z'")};
829                 break;
830             default:
831                 dfArr = new SimpleDateFormat[]{new SimpleDateFormat(
832                         "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'")};
833                 break;
834         }
835 
836         for (SimpleDateFormat df : dfArr) {
837             try {
838                 // Parse the date
839                 df.setCalendar(Calendar
840                         .getInstance(TimeZone.getTimeZone("UTC")));
841                 df.setLenient(false);
842                 return df.parse(t);
843             } catch (ParseException pe) {
844                 log.error("Unable to parse date format", pe);
845             }
846         }
847 
848         return null;
849     }
