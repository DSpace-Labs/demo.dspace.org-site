114     /***
115      * Generate the unique caching key.
116      * This key must be unique inside the space of this component.
117      */
118     public Serializable getKey() {
119         try {
120             String key = "";
121 
122             // Page Parameter
123             Request request = ObjectModelHelper.getRequest(objectModel);
124             key += "-" + getParameterPage();
125             key += "-" + getParameterRpp();
126             key += "-" + getParameterSortBy();
127             key += "-" + getParameterOrder();
128             key += "-" + getParameterEtAl();
129 
130             // What scope the search is at
131             DSpaceObject scope = getScope();
132             if (scope != null)
133                 key += "-" + scope.getHandle();
134 
135             // The actual search query.
136             key += "-" + getQuery();
137 
138             return HashUtil.hash(key);
139         }
140         catch (Exception e) {
141             // Ignore all errors and just don't cache.
142             return "0";
143         }
144     }

146     /***
147      * Generate the cache validity object.
148      * <p/>
149      * This validity object should never "over cache" because it will
150      * perform the search, and serialize the results using the
151      * DSpaceValidity object.
152      */
153     public SourceValidity getValidity() {
154         if (this.validity == null) {
155             try {
156                 DSpaceValidity validity = new DSpaceValidity();
157 
158                 DSpaceObject scope = getScope();
159                 validity.add(scope);
160 
161                 performSearch(scope);
162 
163                 SolrDocumentList results = this.queryResults.getResults();
164 
165                 if (results != null) {
166                     validity.add("size:" + results.size());
167 
168                     for (SolrDocument result : results) {
169                         validity.add(result.toString());
170                     }
171                 }
172 
173                 this.validity = validity.complete();
174             }
175             catch (Exception e) {
176                 // Just ignore all errors and return an invalid cache.
177             }
178 
179             // add log message that we are viewing the item
180             // done here, as the serialization may not occur if the cache is valid
181             logSearch();
182         }
183         return this.validity;
184     }

193     /***
194      * Attach a division to the given search division named "search-results"
195      * which contains results for this search query.
196      *
197      * @param search The search division to contain the search-results division.
198      */
199     protected void buildSearchResultsDivision(Division search)
200             throws IOException, SQLException, WingException, SearchServiceException {
201 
202         try {
203             if (queryResults == null || queryResults.getResults() == null) {
204 
205                 DSpaceObject scope = getScope();
206                 this.performSearch(scope);
207             }
208         }
209         catch (Throwable t) {
210             log.error(t.getMessage(), t);
211             queryResults = null;
212         }
213 
214         if (queryResults != null) {
215             search.addPara("result-query", "result-query")
216                     .addContent(T_result_query.parameterize(getQuery(), queryResults.getResults().getNumFound()));
217         }
218 
219         Division results = search.addDivision("search-results", "primary");
220 
221         DSpaceObject searchScope = getScope();
222 
223         if (searchScope instanceof Community) {
224             Community community = (Community) searchScope;
225             String communityName = community.getMetadata("name");
226             results.setHead(T_head1_community.parameterize(communityName));
227         } else if (searchScope instanceof Collection) {
228             Collection collection = (Collection) searchScope;
229             String collectionName = collection.getMetadata("name");
230             results.setHead(T_head1_collection.parameterize(collectionName));
231         } else {
232             results.setHead(T_head1_none);
233         }
234 
235         if (queryResults != null &&
236                 queryResults.getResults().getNumFound() > 0) {
237 
238             SolrDocumentList solrResults = queryResults.getResults();
239 
240             // Pagination variables.
241             int itemsTotal = (int) solrResults.getNumFound();
242             int firstItemIndex = (int) solrResults.getStart() + 1;
243             int lastItemIndex = (int) solrResults.getStart() + solrResults.size();
244 
245             //if (itemsTotal < lastItemIndex)
246             //    lastItemIndex = itemsTotal;
247             int currentPage = (int) (solrResults.getStart() / this.queryArgs.getRows()) + 1;
248             int pagesTotal = (int) ((solrResults.getNumFound() - 1) / this.queryArgs.getRows()) + 1;
249             Map<String, String> parameters = new HashMap<String, String>();
250             parameters.put("page", "{pageNum}");
251             String pageURLMask = generateURL(parameters);
252             //Check for facet queries ? If we have any add them
253             String[] fqs = getParameterFacetQueries();
254             if(fqs != null){
255                 for (String fq : fqs) {
256                     pageURLMask += "&fq=" + fq;
257                 }
258             }
259 
260             results.setMaskedPagination(itemsTotal, firstItemIndex,
261                     lastItemIndex, currentPage, pagesTotal, pageURLMask);
262 
263             // Look for any communities or collections in the mix
264             ReferenceSet referenceSet = null;
265 
266 
267             boolean resultsContainsBothContainersAndItems = false;
268 
269             for (SolrDocument doc : solrResults) {
270 
271                 DSpaceObject resultDSO =
272                         SearchUtils.findDSpaceObject(context, doc);
273 
274                 if (resultDSO instanceof Community
275                         || resultDSO instanceof Collection) {
276                     if (referenceSet == null) {
277                         referenceSet = results.addReferenceSet("search-results-repository",
278                                 ReferenceSet.TYPE_SUMMARY_LIST, null, "repository-search-results");
279                         // Set a heading showing that we will be listing containers that matched:
280                         referenceSet.setHead(T_head2);
281                         resultsContainsBothContainersAndItems = true;
282 
283                     }
284                     referenceSet.addReference(resultDSO);
285                 }
286             }
287 
288 
289             // Put in palce top level referenceset
290             referenceSet = results.addReferenceSet("search-results-repository",
291                     ReferenceSet.TYPE_SUMMARY_LIST, null, "repository-search-results");
292 
293 
294             for (SolrDocument doc : solrResults) {
295 
296                 DSpaceObject resultDSO = SearchUtils.findDSpaceObject(context, doc);
297 
298                 if (resultDSO instanceof Item) {
299 
300 
301                     String group_by = this.getParameterGroup();
302 
303 
304                     // If we are grouping, attempt to acquire the dc.isPartOf parent of the Item and group on it.
305                     // Otherwise, Group on the current Item.
306 
307                     // TODO: this is a hack to always make sure any subItem is grouped under its parent
308                     if (!group_by.equals("none")) {
309 
310                         Item parent = getParent((Item) resultDSO);
311 
312                         // if parent not null, use parent otherwise use existing item.
313                         if (parent != null) {
314                             Reference parentRef = referenceSet.addReference(parent);
315                             addCollapsedDocuments(parentRef, parent, doc);
316                         }
317                         else
318                         {
319                             referenceSet.addReference(resultDSO);
320                         }
321 
322                     } else {
323                         referenceSet.addReference(resultDSO);
324 
325                     }
326                 }
327             }
328 
329         } else {
330             results.addPara(T_no_results);
331         }
332         //}// Empty query
333     }

408     private FieldCollapseResponse.CollapseGroup getCollapseGroup(String field) {
409 
410 
411         FieldCollapseResponse cResp = this.queryResults.getFieldCollapseResponse();
412         /*** make lookup map */
413         for (FieldCollapseResponse.CollapseGroup grp : cResp.getCollapseGroups()) {
414 
415             String key = grp.getCollapseGroupId().replaceFirst("http://hdl.handle.net/", "");
416 
417             if (key.equals(field)) {
418                 return grp;
419             }
420 
421         }
422 
423         return null;
424     }

441 
442         }

444         return null;

458     /***
459      * Add options to the search scope field. This field determines in what
460      * communities or collections to search for the query.
461      * <p/>
462      * The scope list will depend upon the current search scope. There are three
463      * cases:
464      * <p/>
465      * No current scope: All top level communities are listed.
466      * <p/>
467      * The current scope is a community: All collections contained within the
468      * community are listed.
469      * <p/>
470      * The current scope is a collection: All parent communities are listed.
471      *
472      * @param scope The current scope field.
473      */
474     protected void buildScopeList(Select scope) throws SQLException,
475             WingException {
476 
477         DSpaceObject scopeDSO = getScope();
478         if (scopeDSO == null) {
479             // No scope, display all root level communities
480             scope.addOption("/", T_all_of_dspace);
481             scope.setOptionSelected("/");
482             for (Community community : Community.findAllTop(context)) {
483                 scope.addOption(community.getHandle(), community.getMetadata("name"));
484             }
485         } else if (scopeDSO instanceof Community) {
486             // The scope is a community, display all collections contained
487             // within
488             Community community = (Community) scopeDSO;
489             scope.addOption("/", T_all_of_dspace);
490             scope.addOption(community.getHandle(), community.getMetadata("name"));
491             scope.setOptionSelected(community.getHandle());
492 
493             for (Collection collection : community.getCollections()) {
494                 scope.addOption(collection.getHandle(), collection.getMetadata("name"));
495             }
496         } else if (scopeDSO instanceof Collection) {
497             // The scope is a collection, display all parent collections.
498             Collection collection = (Collection) scopeDSO;
499             scope.addOption("/", T_all_of_dspace);
500             scope.addOption(collection.getHandle(), collection.getMetadata("name"));
501             scope.setOptionSelected(collection.getHandle());
502 
503             Community[] communities = collection.getCommunities()[0]
504                     .getAllParents();
505             for (Community community : communities) {
506                 scope.addOption(community.getHandle(), community.getMetadata("name"));
507             }
508         }
509     }

515      * @return The associated query results.
516      */
517     public void performSearch(DSpaceObject scope) throws UIException, SearchServiceException {
518 
519         if (queryResults != null)
520             return;
521         
522 
523         String query = getQuery();
524 
525         //DSpaceObject scope = getScope();
526 
527         int page = getParameterPage();
528 
529         queryArgs = this.prepareDefaultFilters("search");
530 
531 
532         queryArgs.setRows(getParameterRpp());
533 
534         String sortBy = ObjectModelHelper.getRequest(objectModel).getParameter("sort_by");
535 
536         String sortOrder = ObjectModelHelper.getRequest(objectModel).getParameter("order");
537 
538 
539         //webui.itemlist.sort-option.1 = title:dc.title:title
540         //webui.itemlist.sort-option.2 = dateissued:dc.date.issued:date
541         //webui.itemlist.sort-option.3 = dateaccessioned:dc.date.accessioned:date
542         //webui.itemlist.sort-option.4 = ispartof:dc.relation.ispartof:text
543 
544         if (sortBy != null) {
545             if (sortOrder == null || sortOrder.equals("DESC"))
546                 queryArgs.addSortField(sortBy, SolrQuery.ORDER.desc);
547             else
548                 queryArgs.addSortField(sortBy, SolrQuery.ORDER.asc);
549         } else {
550             queryArgs.addSortField("score", SolrQuery.ORDER.asc);
551         }
552 
553 
554         String groupBy = ObjectModelHelper.getRequest(objectModel).getParameter("group_by");
555 
556 
557         // Enable groupBy collapsing if designated
558         if (groupBy != null && !groupBy.equalsIgnoreCase("none")) {
559             /*** Construct a Collapse Field Query */
560             queryArgs.add("collapse.field", groupBy);
561             queryArgs.add("collapse.threshold", "1");
562             queryArgs.add("collapse.includeCollapsedDocs.fl", "handle");
563             queryArgs.add("collapse.facet", "before");
564 
565             //queryArgs.a  type:Article^2
566

571 
572         }
573 
574 
575         queryArgs.setQuery(query != null && !query.trim().equals("") ? query : "*:*");
576 
577         if (page > 1)
578             queryArgs.setStart((page - 1) * queryArgs.getRows());
579         else
580             queryArgs.setStart(0);
581 
582 
583         List<String> list = new ArrayList<String>();
584

586             list.add("location:m" + scope.getID());
587 
588         } else if (scope instanceof Collection) {

590 
591         }
592 
593         String[] fqs = getParameterFacetQueries();
594 
595         if (fqs != null) {
596             for (String fq : getParameterFacetQueries()) {
597                 list.add(fq);
598             }
599         }
600 
601 
602         if (list.size() > 0) {
603             queryArgs.setFilterQueries(list.toArray(new String[0]));
604         }
605 
606 
607         // Use mlt
608         // queryArgs.add("mlt", "true");
609 
610         // The fields to use for similarity. NOTE: if possible, these should have a stored TermVector
611         // queryArgs.add("mlt.fl", "author");
612 
613         // Minimum Term Frequency - the frequency below which terms will be ignored in the source doc.
614         // queryArgs.add("mlt.mintf", "1");
615 
616         // Minimum Document Frequency - the frequency at which words will be ignored which do not occur in at least this many docs.
617         // queryArgs.add("mlt.mindf", "1");
618 
619         //queryArgs.add("mlt.q", "");
620 
621         // mlt.minwl
622         // minimum word length below which words will be ignored.
623 
624         // mlt.maxwl
625         // maximum word length above which words will be ignored.
626 
627         // mlt.maxqt
628         // maximum number of query terms that will be included in any generated query.
629 
630         // mlt.maxntp
631         // maximum number of tokens to parse in each example doc field that is not stored with TermVector support.
632 
633         // mlt.boost
634         // [true/false] set if the query will be boosted by the interesting term relevance.
635 
636         // mlt.qf
637         // Query fields and their boosts using the same format as that used in DisMaxRequestHandler. These fields must also be specified in mlt.fl.
638 
639 
640         //filePost.addParameter("fl", "handle, "search.resourcetype")");
641         //filePost.addParameter("field", "search.resourcetype");
642 
643         //Set the default limit to 11
644         /*
645         ClientUtils.escapeQueryChars(location)
646         //f.category.facet.limit=5
647 
648         for(Enumeration en = request.getParameterNames(); en.hasMoreElements();)
649         {
650         	String key = (String)en.nextElement();
651         	if(key.endsWith(".facet.limit"))
652         	{
653         		filePost.addParameter(key, request.getParameter(key));
654         	}
655         }

668     protected DSpaceObject getScope() throws SQLException {
669         Request request = ObjectModelHelper.getRequest(objectModel);
670         String scopeString = request.getParameter("scope");
671 
672         // Are we in a community or collection?
673         DSpaceObject dso;
674         if (scopeString == null || "".equals(scopeString))
675             // get the search scope from the url handle
676             dso = HandleUtil.obtainHandle(objectModel);
677         else
678             // Get the search scope from the location parameter
679             dso = HandleManager.resolveToObject(context, scopeString);
680 
681         return dso;
682     }
