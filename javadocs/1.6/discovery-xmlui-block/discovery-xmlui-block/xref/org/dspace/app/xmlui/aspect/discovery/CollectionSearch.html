78      /***
79       * Generate the unique caching key.
80       * This key must be unique inside the space of this component.
81       */
82      public Serializable getKey()
83      {
84          try
85          {
86              DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
87  
88              if (dso == null)
89                  return "0";
90  
91              return HashUtil.hash(dso.getHandle());
92          }
93          catch (SQLException sqle)
94          {
95              // Ignore all errors and just return that the component is not
96              // cachable.
97              return "0";
98          }
99      }

101     /***
102      * Generate the cache validity object.
103      *
104      * The validity object will include the collection being viewed and
105      * all recently submitted items. This does not include the community / collection
106      * hierarch, when this changes they will not be reflected in the cache.
107      */
108     public SourceValidity getValidity()
109     {
110     	if (this.validity == null)
111     	{
112             Collection collection = null;
113 	        try
114 	        {
115 	            DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
116 
117 	            if (dso == null)
118 	                return null;
119 
120 	            if (!(dso instanceof Collection))
121 	                return null;
122 
123 	            collection = (Collection) dso;
124 
125 	            DSpaceValidity validity = new DSpaceValidity();
126 
127 	            // Add the actual collection;
128 	            validity.add(collection);
129 
130 	            this.validity = validity.complete();
131 	        }
132 	        catch (Exception e)
133 	        {
134 	            // Just ignore all errors and return an invalid cache.
135 	        }
136             
137     	}
138     	return this.validity;
139     }

142     /***
143      * Add a page title and trail links.
144      */
145     public void addPageMeta(PageMeta pageMeta) throws SAXException,
146             WingException, UIException, SQLException, IOException,
147             AuthorizeException
148     {
149         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
150         if (!(dso instanceof Collection))
151             return;
152 
153         Collection collection = (Collection) dso;
154 
155         // Set the page title
156         String name = collection.getMetadata("name");
157         if (name == null || name.length() == 0)
158         	pageMeta.addMetadata("title").addContent(T_untitled);
159         else
160         	pageMeta.addMetadata("title").addContent(name);
161 
162         pageMeta.addTrailLink(contextPath + "/",T_dspace_home);
163         HandleUtil.buildHandleTrail(collection,pageMeta,contextPath);
164 
165         // Add RSS links if available
166         String formats = ConfigurationManager.getProperty("webui.feed.formats");
167 		if ( formats != null )
168 		{
169 			for (String format : formats.split(","))
170 			{
171 				// Remove the protocol number, i.e. just list 'rss' or' atom'
172 				String[] parts = format.split("_");
173 				if (parts.length < 1)
174 					continue;
175 
176 				String feedFormat = parts[0].trim()+"+xml";
177 
178 				String feedURL = contextPath+"/feed/"+format.trim()+"/"+collection.getHandle();
179 				pageMeta.addMetadata("feed", feedFormat).addContent(feedURL);
180 			}
181 		}
182     }
