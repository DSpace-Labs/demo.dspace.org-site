114     /***
115      * Generate the cache validity object.
116      * <p/>
117      * The validity object will include the collection being viewed and
118      * all recently submitted items. This does not include the community / collection
119      * hierarch, when this changes they will not be reflected in the cache.
120      */
121     public SourceValidity getValidity() {
122         if (this.validity == null) {
123 
124             try {
125                 DSpaceValidity validity = new DSpaceValidity();
126 
127                 DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
128 
129                 if (dso != null) {
130                     // Add the actual collection;
131                     validity.add(dso);
132                 }
133 
134                 // add reciently submitted items, serialize solr query contents.
135                 QueryResponse response = getQueryResponse(dso);
136 
137                 validity.add("numFound:" + response.getResults().getNumFound());
138 
139                 for (SolrDocument doc : response.getResults()) {
140                     validity.add(doc.toString());
141                 }
142 
143                 for (SolrDocument doc : response.getResults()) {
144                     validity.add(doc.toString());
145                 }
146 
147                 for (FacetField field : response.getFacetFields()) {
148                     validity.add(field.getName());
149 
150                     for (FacetField.Count count : field.getValues()) {
151                         validity.add(count.getName() + count.getCount());
152                     }
153                 }
154 
155 
156                 this.validity = validity.complete();
157             }
158             catch (Exception e) {
159                 // Just ignore all errors and return an invalid cache.
160             }
161 
162             //TODO: dependent on tags as well :)
163         }
164         return this.validity;
165     }

167     /***
168      * Get the recently submitted items for the given community or collection.
169      *
170      * @param scope The collection.
171      */
172     protected QueryResponse getQueryResponse(DSpaceObject scope) {
173 
174 
175         Request request = ObjectModelHelper.getRequest(objectModel);
176 
177         if (queryResults != null)
178             return queryResults;
179 
180         queryArgs = new SolrQuery();
181 
182         queryArgs.setQuery("search.resourcetype:" + Constants.ITEM);
183 
184         queryArgs.setRows(0);
185 
186 
187         queryArgs.setSortField(
188                 ConfigurationManager.getProperty("recent.submissions.sort-option"),
189                 SolrQuery.ORDER.asc
190         );
191         queryArgs.setFilterQueries(getParameterFacetQueries());
192 
193 
194         //Set the default limit to 11
195         //query.setFacetLimit(11);
196         queryArgs.setFacetMinCount(1);
197 
198         //sort
199         queryArgs.setFacetSort(config.getPropertyAsType("solr.browse.sort","lex"));
200 
201         queryArgs.setFacet(true);
202 
203         int offset = RequestUtils.getIntParameter(request, OFFSET);
204         if (offset == -1)
205             offset = 0;
206         queryArgs.setParam(FacetParams.FACET_OFFSET, String.valueOf(offset));
207 
208         //We add +1 so we can use the extra one to make sure that we need to show the next page
209         queryArgs.setParam(FacetParams.FACET_LIMIT, String.valueOf(DEFAULT_PAGE_SIZE + 1));
210 
211 
212         if (scope != null) /* top level search / community */ {
213             if (scope instanceof Community) {
214                 queryArgs.setFilterQueries("location:m" + scope.getID());
215             } else if (scope instanceof Collection) {
216                 queryArgs.setFilterQueries("location:l" + scope.getID());
217             }
218         }
219 
220 
221         String facetField = request.getParameter(FACET_FIELD);
222         boolean isDate = false;
223         if(facetField.endsWith("_dt")){
224             facetField = facetField.split("_")[0];
225             isDate = true;
226         }
227 
228         if (isDate) {
229 
230             queryArgs.setParam(FacetParams.FACET_DATE,new String[]{facetField});
231             queryArgs.setParam(FacetParams.FACET_DATE_GAP,"+1YEAR");
232 
233             Date lowestDate = getLowestDateValue(queryArgs.getQuery(), facetField, queryArgs.getFilterQueries());
234             int thisYear = Calendar.getInstance().get(Calendar.YEAR);
235 
236             DateFormat formatter = new SimpleDateFormat("yyyy");
237             int maxEndYear = Integer.parseInt(formatter.format(lowestDate));
238 
239             //Since we have a date, we need to find the last year
240             String startDate = "NOW/YEAR-" + SearchUtils.getConfig().getString("solr.date.gap", "10") + "YEARS";
241             String endDate =  "NOW";
242             int startYear =  thisYear - (offset + DEFAULT_PAGE_SIZE);
243             // We shouldn't go lower then our max bottom year
244             // Make sure to substract one so the bottom year is also counted !
245             if(startYear < maxEndYear)
246                 startYear = maxEndYear - 1;
247 
248             if(0 < offset){
249                 //Say that we have an offset of 10 years
250                 //we need to go back 10 years (2010 - (2010 - 10))
251                 //(add one to compensate for the NOW in the start)
252                 int endYear = thisYear - offset + 1;
253 
254                 endDate = "NOW/YEAR-" + (thisYear - endYear) + "YEARS";
255                 //Add one to the startyear to get one more result
256                 //When we select NOW, the current year is also used (so auto+1)
257             }
258             startDate = "NOW/YEAR-" + (thisYear - startYear) + "YEARS";
259 
260             queryArgs.setParam(FacetParams.FACET_DATE_START, startDate);
261             queryArgs.setParam(FacetParams.FACET_DATE_END, endDate);
262 
263             System.out.println(startDate);
264             System.out.println(endDate);
265 
266 
267 
268         } else {
269             queryArgs.addFacetField(new String[]{facetField});
270         }
271         
272         try {
273             queryResults = searchService.search(queryArgs);
274         } catch (SearchServiceException e) {
275             log.error(e.getMessage(), e);
276         }
277 
278         return queryResults;
279     }

281         /***
282      * Retrieves the lowest date value in the given field
283      * @param query a solr query
284      * @param dateField the field for which we want to retrieve our date
285      * @param filterquery the filterqueries
286      * @return the lowest date found, in a date object
287      */
288     private Date getLowestDateValue(String query, String dateField, String... filterquery){
289 
290 
291         try {
292             SolrQuery solrQuery = new SolrQuery();
293             solrQuery.setQuery(query);
294             solrQuery.setFields(dateField);
295             solrQuery.setRows(1);
296             solrQuery.setSortField(dateField, SolrQuery.ORDER.asc);
297             solrQuery.setFilterQueries(filterquery);
298 
299             QueryResponse rsp = searchService.search(solrQuery);
300             if(0 < rsp.getResults().getNumFound()){
301                 return (Date) rsp.getResults().get(0).getFieldValue(dateField);
302             }
303         }catch (Exception e){
304             e.printStackTrace();
305         }
306         return null;
307     }

440     private String getNextPageURL(Request request) {
441         Map<String, String> parameters = new HashMap<String, String>();
442         parameters.put(FACET_FIELD, request.getParameter(FACET_FIELD));
443         if (queryArgs.get(FacetParams.FACET_OFFSET) != null)
444             parameters.put(OFFSET, String.valueOf(Integer.parseInt(queryArgs.get(FacetParams.FACET_OFFSET)) + DEFAULT_PAGE_SIZE));
445 
446         // Add the filter queries
447         String url = generateURL("browse", parameters);
448         String[] fqs = getParameterFacetQueries();
449         if (fqs != null) {
450             for (String fq : fqs) {
451                 url += "&fq=" + fq;
452             }
453         }
454 
455         return url;
456     }

458     private String getPreviousPageURL(Request request) {
459         //If our offset should be 0 then we shouldn't be able to view a previous page url
460         if ("0".equals(queryArgs.get(FacetParams.FACET_OFFSET)))
461             return null;
462 
463         Map<String, String> parameters = new HashMap<String, String>();
464         parameters.put(FACET_FIELD, request.getParameter(FACET_FIELD));
465         if (queryArgs.get(FacetParams.FACET_OFFSET) != null)
466             parameters.put(OFFSET, String.valueOf(Integer.parseInt(queryArgs.get(FacetParams.FACET_OFFSET)) - DEFAULT_PAGE_SIZE));
467 
468         // Add the filter queries
469         String url = generateURL("browse", parameters);
470         String[] fqs = getParameterFacetQueries();
471         if (fqs != null) {
472             for (String fq : fqs) {
473                 url += "&fq=" + fq;
474             }
475         }
476 
477         return url;
478     }
