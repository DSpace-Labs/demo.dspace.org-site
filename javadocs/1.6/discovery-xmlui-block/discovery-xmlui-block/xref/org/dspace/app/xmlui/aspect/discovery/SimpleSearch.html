111     public void addPageMeta(PageMeta pageMeta) throws WingException, SQLException {

115         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);

131         String queryString = getQuery();

218     /***
219      * Get the search query from the URL parameter, if none is found the empty
220      * string is returned.
221      */
222     protected String getQuery() throws UIException {
223         Request request = ObjectModelHelper.getRequest(objectModel);
224         String query = URLDecode(request.getParameter("query"));
225         if (query == null)
226             return "";
227         return query.trim();
228     }
229 
230     /***
231      * Generate a url to the simple search url.
232      */
233     protected String generateURL(Map<String, String> parameters)
234             throws UIException {
235         String query = getQuery();
236         if (!"".equals(query))
237             parameters.put("query", URLEncode(query));
238 
239         if (parameters.get("page") == null)
240             parameters.put("page", String.valueOf(getParameterPage()));
241 
242         if (parameters.get("rpp") == null)
243             parameters.put("rpp", String.valueOf(getParameterRpp()));
244 
245 
246         if (parameters.get("group_by") == null)
247             parameters.put("group_by", String.valueOf(this.getParameterGroup()));
248 
249         if (parameters.get("sort_by") == null)
250             parameters.put("sort_by", String.valueOf(getParameterSortBy()));
251 
252         if (parameters.get("order") == null)
253             parameters.put("order", getParameterOrder());
254 
255         if (parameters.get("etal") == null)
256             parameters.put("etal", String.valueOf(getParameterEtAl()));
257 
258         return super.generateURL("search", parameters);
259     }
