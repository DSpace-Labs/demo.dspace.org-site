34  /***
35   * Display a list of recent submissions.
36   *
37   * @author Mark Diggory
38   */
39  public class SiteRecentSubmissions extends AbstractFiltersTransformer {
40  
41      private static final Logger log = Logger.getLogger(SiteRecentSubmissions.class);
42  
43      private static final Message T_head_recent_submissions =
44              message("xmlui.ArtifactBrowser.SiteViewer.head_recent_submissions");
45  
46  
47      /***
48       * Display a single community (and refrence any sub communites or
49       * collections)
50       */
51      public void addBody(Body body) throws SAXException, WingException,
52              UIException, SQLException, IOException, AuthorizeException {
53  
54          try {
55              performSearch(null);
56          } catch (SearchServiceException e) {
57              log.error(e.getMessage(), e);
58          }
59  
60          Division home = body.addDivision("site-home", "primary repository");
61  
62          Division lastSubmittedDiv = home
63                  .addDivision("site-recent-submission", "secondary recent-submission");
64  
65          lastSubmittedDiv.setHead(T_head_recent_submissions);
66  
67          ReferenceSet lastSubmitted = lastSubmittedDiv.addReferenceSet(
68                  "site-last-submitted", ReferenceSet.TYPE_SUMMARY_LIST,
69                  null, "recent-submissions");
70  
71          if (queryResults != null)
72          for (SolrDocument doc : queryResults.getResults()) {
73              lastSubmitted.addReference(SearchUtils.findDSpaceObject(context, doc));
74          }
75      }
76  
77  
78      /***
79       * facet.limit=11&wt=javabin&rows=5&sort=dateaccessioned+asc&facet=true&facet.mincount=1&q=search.resourcetype:2&version=1
80       *
81       * @param object
82       */
83      @Override
84      public void performSearch(DSpaceObject object) throws SearchServiceException, UIException {
85  
86          if (queryResults != null)
87              return;// queryResults;
88  
89  
90          queryArgs = prepareDefaultFilters("site");
91  
92          
93  
94          queryArgs.setQuery("search.resourcetype:" + Constants.ITEM);
95  
96          queryArgs.setRows(SearchUtils.getConfig().getInt("solr.recent-submissions.size", 5));
97  
98          queryArgs.setSortField(
99                  ConfigurationManager.getProperty("recent.submissions.sort-option"),
100                 SolrQuery.ORDER.asc
101         );
102 
103         SearchService service = getSearchService();
104 
105         queryResults = service.search(queryArgs);
106 
107     }
