191   /***
192    * Used to determine if a run of text has been deleted.
193    *
194    * @param grpprl The list of sprms for a particular run of text.
195    * @return true if this run of text has been deleted.
196    */
197   protected boolean isDeleted(byte[] grpprl)
198   {
199     SprmIterator iterator = new SprmIterator(grpprl);
200     while (iterator.hasNext())
201     {
202       SprmOperation op = iterator.next();
203       // 0 is the operation that signals a FDelRMark operation
204       if (op.getOperation() == 0 && op.getOperand() != 0)
205       {
206         return true;
207       }
208     }
209     return false;
210   }
