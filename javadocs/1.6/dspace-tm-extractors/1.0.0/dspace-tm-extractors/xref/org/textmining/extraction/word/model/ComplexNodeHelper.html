21  package org.textmining.extraction.word.model;

23  import java.util.Collections;
24  import java.util.List;
25  
26  public class ComplexNodeHelper extends NodeHelper
27  {
28  
29    public ComplexNodeHelper(TextPieceTable tpt)
30    {
31      super(tpt);
32    }
33  
34    public void addChpNodes(List nodes, byte[] grpprl, int fcStart, int fcEnd)
35    {    
36      for (int x = 0; x < _textPieces.size(); x++)
37      {
38        TextPiece tp = (TextPiece)_textPieces.get(x);
39        
40        int pieceFcStart = tp.getFcStart();
41        int pieceFcEnd = tp.getFcEnd();
42        
43        
44        if (!(fcStart < pieceFcStart && fcEnd <= pieceFcStart) && 
45            !(fcStart >= pieceFcEnd && fcEnd > pieceFcEnd))
46        {
47          int textStart = Math.max(fcStart, pieceFcStart);
48          int textEnd = Math.min(fcEnd, pieceFcEnd);
49          
50          int divisor = (tp.unicode() ? 2 : 1);
51          int cpStart = tp.getStart() + (textStart - pieceFcStart)/divisor;
52          int cpEnd = cpStart + (textEnd - textStart)/divisor ;
53          
54          CHPX chpx = new CHPX(cpStart, cpEnd, grpprl);    
55          nodes.add(chpx);
56        }      
57      }   
58    }
59    
60    
61    
62    public void sortNodes(List nodes, boolean paragraphs)
63    {
64      Collections.sort(nodes);
65      
66      if (paragraphs)
67      {
68        int len = nodes.size();
69        int lastEnd = 0;
70        for (int x = 0; x < len; x++)
71        {
72          PropertyNode node = (PropertyNode)nodes.get(x);
73          node.setStart(lastEnd);
74          lastEnd = node.getEnd();
75        }
76      }
77      
78    }  
79  }
