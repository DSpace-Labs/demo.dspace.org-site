8   package org.dspace.utils.servicemanager;

10  import java.util.Comparator;
11  import java.io.Serializable;
12  
13  
14  /***
15   * A Comparator for provider stacks.  This is specially designed for
16   * sorting a list of ProviderHolders so it will unpack them and then do 
17   * the typical sorting on them while properly handling the null cases.
18   * 
19   * @author Aaron Zeckoski (azeckoski @ gmail.com)
20   */
21  public final class ProviderStackComparator implements Comparator<ProviderHolder<?>>, Serializable {
22      public static final long serialVersionUID = 1l;
23      public int compare(ProviderHolder<?> ph0, ProviderHolder<?> ph1) {
24          /* a negative integer, zero, or a positive integer as the first argument 
25           * is less than, equal to, or greater than the second.
26           */
27          int comparison = 0;
28          Object arg0 = ph0.getProvider();
29          Object arg1 = ph1.getProvider();
30          if (arg0 == null && arg1 == null) {
31              comparison = 0;
32          } else if (arg0 == null) {
33              comparison = 1; // null is last
34          } else if (arg1 == null) {
35              comparison = -1; // null is last
36          } else {
37              comparison = new OrderedServiceComparator().compare(arg0, arg1);
38          }
39          return comparison;
40      }
41  }
