99  	 * Initialise the servlet
100 	 * 
101 	 * @throws ServletException
102 	 */
103 	public void init() throws ServletException {
104 		// Instantiate the correct SWORD Server class
105 		String className = getServletContext().getInitParameter("sword-server-class");
106 		if (className == null) {
107 			log.fatal("Unable to read value of 'sword-server-class' from Servlet context");
108 		} else {
109 			try {
110 				myRepository = (SWORDServer) Class.forName(className)
111 						.newInstance();
112 				log.info("Using " + className + " as the SWORDServer");
113 			} catch (Exception e) {
114 				log
115 						.fatal("Unable to instantiate class from 'sword-server-class': "
116 								+ className);
117 				throw new ServletException(
118 						"Unable to instantiate class from 'sword-server-class': "
119 								+ className);
120 			}
121 		}
122 
123 		authN = getServletContext().getInitParameter("authentication-method");
124 		if ((authN == null) || (authN.equals(""))) {
125 			authN = "None";
126 		}
127 		log.info("Authentication type set to: " + authN);
128 
129 		String maxUploadSizeStr = getServletContext().getInitParameter("maxUploadSize");
130 		if ((maxUploadSizeStr == null) || 
131 		    (maxUploadSizeStr.equals("")) || 
132 		    (maxUploadSizeStr.equals("-1"))) {
133 			maxUploadSize = -1;
134 			log.warn("No maxUploadSize set, so setting max file upload size to unlimited.");
135 		} else {
136 			try {
137 				maxUploadSize = Integer.parseInt(maxUploadSizeStr);
138 				log.info("Setting max file upload size to " + maxUploadSize);
139 			} catch (NumberFormatException nfe) {
140 				maxUploadSize = -1;
141 				log.warn("maxUploadSize not a number, so setting max file upload size to unlimited.");
142 			}
143 		}
144 
145 		tempDirectory = getServletContext().getInitParameter(
146 				"upload-temp-directory");
147 		if ((tempDirectory == null) || (tempDirectory.equals(""))) {
148 			tempDirectory = System.getProperty("java.io.tmpdir");
149 		}
150         if (!tempDirectory.endsWith(System.getProperty("file.separator")))
151         {
152             tempDirectory += System.getProperty("file.separator");
153         }
154 		File tempDir = new File(tempDirectory);
155 		log.info("Upload temporary directory set to: " + tempDir);
156 		if (!tempDir.exists()) {
157 			if (!tempDir.mkdirs()) {
158 				throw new ServletException(
159 						"Upload directory did not exist and I can't create it. "
160 								+ tempDir);
161 			}
162 		}
163 		if (!tempDir.isDirectory()) {
164 			log.fatal("Upload temporary directory is not a directory: "
165 					+ tempDir);
166 			throw new ServletException(
167 					"Upload temporary directory is not a directory: " + tempDir);
168 		}
169 		if (!tempDir.canWrite()) {
170 			log.fatal("Upload temporary directory cannot be written to: "
171 					+ tempDir);
172 			throw new ServletException(
173 					"Upload temporary directory cannot be written to: "
174 							+ tempDir);
175 		}
176 	}

186 	/**
187 	 * Process a post request.
188 	 */
189 	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
190 		// Create the Deposit request
191 		Deposit d = new Deposit();
192 		Date date = new Date();
193 		log.debug("Starting deposit processing at " + date.toString() + " by "
194 				+ request.getRemoteAddr());
195 
196 		// Are there any authentication details?
197 		String usernamePassword = getUsernamePassword(request);
198 		if ((usernamePassword != null) && (!usernamePassword.equals(""))) {
199 			int p = usernamePassword.indexOf(":");
200 			if (p != -1) {
201 				d.setUsername(usernamePassword.substring(0, p));
202 				d.setPassword(usernamePassword.substring(p + 1));
203 			}
204 		} else if (authenticateWithBasic()) {
205 			String s = "Basic realm=\"SWORD\"";
206 			response.setHeader("WWW-Authenticate", s);
207 			response.setStatus(401);
208 			return;
209 		}
210 		
211 		// Set up some variables
212 		String filename = null;
213 		File f = null;
214 		FileInputStream fis = null;
215 
216 		// Do the processing
217 		try {
218 			// Write the file to the temp directory
219 			filename = tempDirectory + "SWORD-"
220 					+ request.getRemoteAddr() + "-" + counter.addAndGet(1);
221             log.debug("Package temporarily stored as: " + filename);
222 			InputStream inputstream = request.getInputStream();
223 			OutputStream outputstream = new FileOutputStream(new File(filename));
224 			try
225 			{
226 			    byte[] buf = new byte[1024];
227 			    int len;
228 			    while ((len = inputstream.read(buf)) > 0)
229 			    {
230 			        outputstream.write(buf, 0, len);
231 			    }
232 			}
233 			finally
234 			{
235 			    inputstream.close();
236 			    outputstream.close();
237 			}
238 
239 			// Check the size is OK
240 			File file = new File(filename);
241 		    long fLength = file.length() / 1024;
242 		    if ((maxUploadSize != -1) && (fLength > maxUploadSize)) {
243 		    	this.makeErrorDocument(ErrorCodes.MAX_UPLOAD_SIZE_EXCEEDED, 
244 		    			               HttpServletResponse.SC_REQUEST_ENTITY_TOO_LARGE, 
245 		    			               "The uploaded file exceeded the maximum file size this server will accept (the file is " + 
246 		    			               fLength + "kB but the server will only accept files as large as " + 
247 		    			               maxUploadSize + "kB)",
248 		    			               request,
249 		    			               response);
250 		    	return;
251 		    }
252 		    
253 			// Check the MD5 hash
254 			String receivedMD5 = ChecksumUtils.generateMD5(filename);
255 			log.debug("Received filechecksum: " + receivedMD5);
256 			d.setMd5(receivedMD5);
257 			String md5 = request.getHeader("Content-MD5");
258 			log.debug("Received file checksum header: " + md5);
259 			if ((md5 != null) && (!md5.equals(receivedMD5))) {
260 				// Return an error document
261 				this.makeErrorDocument(ErrorCodes.ERROR_CHECKSUM_MISMATCH, 
262 						               HttpServletResponse.SC_PRECONDITION_FAILED,
263 						               "The received MD5 checksum for the deposited file did not match the checksum sent by the deposit client",
264 						               request,
265 						               response);
266 				log.debug("Bad MD5 for file. Aborting with appropriate error message");
267 				return;
268 			} else {
269 				// Set the file
270 				f = new File(filename);
271 				fis = new FileInputStream(f);
272 				d.setFile(fis);
273 
274 				// Set the X-On-Behalf-Of header
275                 String onBehalfOf = request.getHeader(HttpHeaders.X_ON_BEHALF_OF.toString());
276 				if ((onBehalfOf != null) && (onBehalfOf.equals("reject"))) {
277                     // user name is "reject", so throw a not know error to allow the client to be tested
278                     throw new SWORDErrorException(ErrorCodes.TARGET_OWNER_UKNOWN,"unknown user \"reject\"");
279                 } else {
280                     d.setOnBehalfOf(onBehalfOf);
281                 }
282 
283 				// Set the X-Packaging header
284 				d.setPackaging(request.getHeader(HttpHeaders.X_PACKAGING));
285 
286 				// Set the X-No-Op header
287 				String noop = request.getHeader(HttpHeaders.X_NO_OP);
288                 log.error("X_NO_OP value is " + noop);
289 				if ((noop != null) && (noop.equals("true"))) {
290 					d.setNoOp(true);
291 				} else if ((noop != null) && (noop.equals("false"))) {
292 					d.setNoOp(false);
293                 }else if (noop == null) {
294                     d.setNoOp(false);
295 				} else {
296                     throw new SWORDErrorException(ErrorCodes.ERROR_BAD_REQUEST,"Bad no-op");
297                 }
298 
299 				// Set the X-Verbose header
300 				String verbose = request.getHeader(HttpHeaders.X_VERBOSE);
301 				if ((verbose != null) && (verbose.equals("true"))) {
302 					d.setVerbose(true);
303 				} else if ((verbose != null) && (verbose.equals("false"))) {
304 					d.setVerbose(false);
305                 }else if (verbose == null) {
306                     d.setVerbose(false);
307 				} else {
308                     throw new SWORDErrorException(ErrorCodes.ERROR_BAD_REQUEST,"Bad verbose");
309                 }
310 
311 				// Set the slug
312 				String slug = request.getHeader(HttpHeaders.SLUG);
313 				if (slug != null) {
314 					d.setSlug(slug);
315 				}
316 
317 				// Set the content disposition
318 				d.setContentDisposition(request.getHeader(HttpHeaders.CONTENT_DISPOSITION));
319 
320 				// Set the IP address
321 				d.setIPAddress(request.getRemoteAddr());
322 
323 				// Set the deposit location
324 				d.setLocation(getUrl(request));
325 
326 				// Set the content type
327 				d.setContentType(request.getContentType());
328 
329 				// Set the content length
330 				String cl = request.getHeader(HttpHeaders.CONTENT_LENGTH);
331 				if ((cl != null) && (!cl.equals(""))) {
332 					d.setContentLength(Integer.parseInt(cl));
333 				}
334 
335 				// Get the DepositResponse
336 				DepositResponse dr = myRepository.doDeposit(d);
337 				
338 				// Echo back the user agent
339 				if (request.getHeader(HttpHeaders.USER_AGENT.toString()) != null) {
340 					dr.getEntry().setUserAgent(request.getHeader(HttpHeaders.USER_AGENT.toString()));
341 				}
342 				
343 				// Echo back the packaging format
344 				if (request.getHeader(HttpHeaders.X_PACKAGING.toString()) != null) {
345 					dr.getEntry().setPackaging(request.getHeader(HttpHeaders.X_PACKAGING.toString()));
346 				}
347 				
348 				// Print out the Deposit Response
349 				response.setStatus(dr.getHttpResponse());
350 				if ((dr.getLocation() != null) && (!dr.getLocation().equals("")))
351 				{
352 					response.setHeader("Location", dr.getLocation());
353 				}
354 				response.setContentType("application/atom+xml; charset=UTF-8");
355 				PrintWriter out = response.getWriter();
356 				out.write(dr.marshall());
357 				out.flush();
358 			}
359 		} catch (SWORDAuthenticationException sae) {
360 			// Ask for credentials again
361 			if (authN.equals("Basic")) {
362 				String s = "Basic realm=\"SWORD\"";
363 				response.setHeader("WWW-Authenticate", s);
364 				response.setStatus(401);
365 			}
366 		} catch (SWORDErrorException see) {
367 			// Get the details and send the right SWORD error document
368 			log.error(see.toString());
369 			this.makeErrorDocument(see.getErrorURI(), 
370 		               			   see.getStatus(),
371 		               			   see.getDescription(),
372 		                           request,
373 		                           response);
374 			return;
375 		} catch (SWORDException se) {
376 			response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
377 			log.error(se.toString());
378 		} catch (NoSuchAlgorithmException nsae) {
379 			response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
380 			log.error(nsae.toString());
381 		}
382 		
383 		finally {
384 			// Close the input stream if it still open
385 			if (fis != null) {
386 				fis.close();
387 			}
388 
389 			// Try deleting the temp file
390 			if (filename != null) {
391 				f = new File(filename);
392 				f.delete();
393 			}
394 		}
395 	}

397 	/**
398 	 * Utility method to construct a SWORDErrorDocumentTest
399 	 * 
400 	 * @param errorURI The error URI to pass
401 	 * @param status The HTTP status to return
402 	 * @param summary The textual description to give the user
403 	 * @param request The HttpServletRequest object
404 	 * @param response The HttpServletResponse to send the error document to
405 	 * @throws IOException 
406 	 */
407 	protected void makeErrorDocument(String errorURI, int status, String summary, 
408 			                       HttpServletRequest request, HttpServletResponse response) throws IOException
409 	{
410 		SWORDErrorDocument sed = new SWORDErrorDocument(errorURI);
411 		Title title = new Title();
412 		title.setContent("ERROR");
413 		sed.setTitle(title);
414 		Calendar calendar = Calendar.getInstance();
415 		String utcformat = "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'";
416 		SimpleDateFormat zulu = new SimpleDateFormat(utcformat);
417 		String serializeddate = zulu.format(calendar.getTime());
418 		sed.setUpdated(serializeddate);
419 		Summary sum = new Summary();
420 		sum.setContent(summary);
421 		sed.setSummary(sum);
422 		if (request.getHeader(HttpHeaders.USER_AGENT.toString()) != null) {
423 			sed.setUserAgent(request.getHeader(HttpHeaders.USER_AGENT.toString()));
424 		}
425 		response.setStatus(status);
426     	response.setContentType("application/atom+xml; charset=UTF-8");
427 		PrintWriter out = response.getWriter();
428     	out.write(sed.marshall().toXML());
429 		out.flush();
430 	}

432 	/**
433 	 * Utility method to return the username and password (separated by a colon
434 	 * ':')
435 	 * 
436 	 * @param request
437 	 * @return The username and password combination
438 	 */
439 	protected String getUsernamePassword(HttpServletRequest request) {
440 		try {
441 			String authHeader = request.getHeader("Authorization");
442 			if (authHeader != null) {
443 				StringTokenizer st = new StringTokenizer(authHeader);
444 				if (st.hasMoreTokens()) {
445 					String basic = st.nextToken();
446 					if (basic.equalsIgnoreCase("Basic")) {
447 						String credentials = st.nextToken();
448 						String userPass = new String(Base64
449 								.decodeBase64(credentials.getBytes()));
450 						return userPass;
451 					}
452 				}
453 			}
454 		} catch (Exception e) {
455 			log.debug(e.toString());
456 		}
457 		return null;
458 	}
