74  /**
75   * A 'dummy server' which acts as dumb repository which implements the
76   * SWORD ServerInterface. It accepts any type of deposit, and tries to
77   * return appropriate responses.
78   * 
79   * It supports authentication: if the username and password match
80   * (case sensitive) it authenticates the user, if not, the authentication 
81   * fails.
82   * 
83   * @author Stuart Lewis
84   */
85  public class DummyServer implements SWORDServer {

106 		String password = sdr.getPassword();
107 		if ((username != null) && (password != null) && 
108 			(((username.equals("")) && (password.equals(""))) || 
109 		     (!username.equalsIgnoreCase(password))) ) {
110 				// User not authenticated
111 				throw new SWORDAuthenticationException("Bad credentials");
112 		}
113 		
114 		// Allow users to force the throwing of a SWORD error exception by setting
115 		// the OBO user to 'error'
116 		if ((sdr.getOnBehalfOf() != null) && (sdr.getOnBehalfOf().equals("error"))) {
117 			// Throw the error exception
118 			throw new SWORDErrorException(ErrorCodes.MEDIATION_NOT_ALLOWED, "Mediated deposits not allowed");

121 		// Create and return a dummy ServiceDocument

224 	public DepositResponse doDeposit(Deposit deposit) 
225 	             throws SWORDAuthenticationException, SWORDErrorException, SWORDException {
226 		// Authenticate the user
227 		String username = deposit.getUsername();
228 		String password = deposit.getPassword();
229 		if ((username != null) && (password != null) && 
230 			(((username.equals("")) && (password.equals(""))) || 
231 			 (!username.equalsIgnoreCase(password))) ) {
232 			// User not authenticated
233 			throw new SWORDAuthenticationException("Bad credentials");
234 		}
235 		
236 		// Check this is a collection that takes obo deposits, else thrown an error
237 		if (((deposit.getOnBehalfOf() != null) && (!deposit.getOnBehalfOf().equals(""))) && 
238 	        (!deposit.getLocation().contains("deposit?user="))) {
239 			throw new SWORDErrorException(ErrorCodes.MEDIATION_NOT_ALLOWED,
240 					                      "Mediated deposit not allowed to this collection");
241 		}
242 		
243 		// Get the filenames
244 		StringBuffer filenames = new StringBuffer("Deposit file contained: ");
245 		if (deposit.getFilename() != null) {
246 			filenames.append("(filename = " + deposit.getFilename() + ") ");
247 		}
248 		if (deposit.getSlug() != null) {
249 			filenames.append("(slug = " + deposit.getSlug() + ") ");
250 		}
251 		try {
252 			ZipInputStream zip = new ZipInputStream(deposit.getFile());
253 			ZipEntry ze;
254 			while ((ze = zip.getNextEntry()) != null) {
255 				filenames.append(" " + ze.toString());
256 			}
257 		} catch (IOException ioe) {
258 			throw new SWORDException("Failed to open deposited zip file", null, ErrorCodes.ERROR_CONTENT);
259 		}
260 		
261 		// Handle the deposit
262 		if (!deposit.isNoOp()) {
263 			counter++;
264 		}
265 		DepositResponse dr = new DepositResponse(Deposit.CREATED);
266 		SWORDEntry se = new SWORDEntry();
267 		
268 		Title t = new Title();
269 		t.setContent("DummyServer Deposit: #" + counter);
270 		se.setTitle(t);
271 		
272 		se.addCategory("Category");
273 		
274 		if (deposit.getSlug() != null) {
275 			se.setId(deposit.getSlug() + " - ID: " + counter);
276 		} else {
277 			se.setId("ID: " + counter);
278 		}
279 		 
280 		SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
281 		TimeZone utc = TimeZone.getTimeZone("UTC");
282 		sdf.setTimeZone (utc);
283 		String milliFormat = sdf.format(new Date());
284 		se.setUpdated(milliFormat);
285 			
286 	    Summary s = new Summary();
287 		s.setContent(filenames.toString());
288 		se.setSummary(s);
289 		Author a = new Author();
290 		if (username != null) {
291 			a.setName(username);
292 		} else {
293 			a.setName("unknown");
294 		}
295 		se.addAuthors(a);
296 		
297 		Link em = new Link();
298 		em.setRel("edit-media");
299 		em.setHref("http://www.myrepository.ac.uk/sdl/workflow/my deposit");
300 		se.addLink(em);
301 		
302 		Link e = new Link();
303 		e.setRel("edit");
304 		e.setHref("http://www.myrepository.ac.uk/sdl/workflow/my deposit.atom");
305 		se.addLink(e);
306 		
307 		if (deposit.getOnBehalfOf() != null) {
308 			Contributor c = new Contributor();
309 			c.setName(deposit.getOnBehalfOf());
310 			c.setEmail(deposit.getOnBehalfOf() + "@myrepository.ac.uk");
311 			se.addContributor(c);
312 		}
313 		
314 		Generator generator = new Generator();
315 		generator.setContent("Stuart's Dummy SWORD Server");
316 		generator.setUri("http://dummy-sword-server.example.com/");
317 		generator.setVersion("1.3");
318 		se.setGenerator(generator);
319 		
320 		Content content = new Content();
321 		try {
322 			content.setType("application/zip");
323 		} catch (InvalidMediaTypeException ex) {
324 			ex.printStackTrace();
325 		}
326 		content.setSource("http://www.myrepository.ac.uk/sdl/uploads/upload-" + counter + ".zip");
327 		se.setContent(content);
328 		
329 		se.setTreatment("Short back and sides");
330 		
331 		if (deposit.isVerbose()) {
332 			se.setVerboseDescription("I've done a lot of hard work to get this far!");
333 		}
334 		
335 		se.setNoOp(deposit.isNoOp());
336 		
337 		dr.setEntry(se);
338 		
339 		dr.setLocation("http://www.myrepository.ac.uk/atom/" + counter);
340 		
341 		return dr;
342 	}
