40  import java.io.IOException;
41  import java.io.PrintWriter;
42  
43  import javax.servlet.ServletException;
44  import javax.servlet.http.HttpServletRequest;
45  import javax.servlet.http.HttpServletResponse;
46  
47  import org.apache.log4j.Logger;
48  import org.purl.sword.base.AtomDocumentRequest;
49  import org.purl.sword.base.AtomDocumentResponse;
50  import org.purl.sword.base.SWORDAuthenticationException;
51  import org.purl.sword.base.SWORDErrorException;
52  import org.purl.sword.base.SWORDException;
53  
54  /**
55   * EntryDocumentServlet
56   * 
57   * @author Glen Robson
58   * @author Stuart Lewis
59   */
60  public class AtomDocumentServlet extends DepositServlet {
61  
62  	/** Logger */
63  	private static Logger log = Logger.getLogger(AtomDocumentServlet.class);
64  
65  	/**
66  	 * Initialise the servlet.
67  	 * 
68  	 * @throws ServletException
69  	 */
70  	public void init() throws ServletException {
71  		super.init();
72  	}
73  
74  	/**
75  	 * Process the get request.
76  	 */
77  	protected void doGet(HttpServletRequest request,
78  			HttpServletResponse response) throws ServletException, IOException {
79  		try {
80  			// Create the atom document request object
81  			AtomDocumentRequest adr = new AtomDocumentRequest();
82  			
83  			// Are there any authentication details?
84  			String usernamePassword = getUsernamePassword(request);
85  			if ((usernamePassword != null) && (!usernamePassword.equals(""))) {
86  				int p = usernamePassword.indexOf(":");
87  				if (p != -1) {
88  					adr.setUsername(usernamePassword.substring(0, p));
89  					adr.setPassword(usernamePassword.substring(p + 1));
90  				}
91  			} else if (authenticateWithBasic()) {
92  				String s = "Basic realm=\"SWORD\"";
93  				response.setHeader("WWW-Authenticate", s);
94  				response.setStatus(401);
95  				return;
96  			}
97  			
98  			// Set the IP address
99  			adr.setIPAddress(request.getRemoteAddr());
100 
101 			// Set the deposit location
102 			adr.setLocation(getUrl(request));
103 			
104 			// Generate the response
105 			AtomDocumentResponse dr = myRepository.doAtomDocument(adr);
106 	
107 			// Print out the Deposit Response
108 			response.setStatus(dr.getHttpResponse());
109 			response.setContentType("application/atom+xml; charset=UTF-8");
110 			PrintWriter out = response.getWriter();
111 			out.write(dr.marshall());
112 			out.flush();
113 		} catch (SWORDAuthenticationException sae) {
114 			// Ask for credentials again
115 			String s = "Basic realm=\"SWORD\"";
116 			response.setHeader("WWW-Authenticate", s);
117 			response.setStatus(401);
118 		} catch (SWORDException se) {
119 			response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
120 		} catch (SWORDErrorException see) {
121 			// Get the details and send the right SWORD error document
122 			super.makeErrorDocument(see.getErrorURI(), 
123 		               			   see.getStatus(),
124 		               			   see.getDescription(),
125 		                           request,
126 		                           response);
127 		}	
128 	}
129 }
