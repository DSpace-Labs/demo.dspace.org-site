311    /**
312     * Overrides the unmarshall method in the parent Entry. This will 
313     * call the parent method to parse the general Atom elements and
314     * attributes. This method will then parse the remaining sword
315     * extensions that exist in the element. 
316     * 
317     * @param entry The entry to parse. 
318     * 
319     * @throws UnmarshallException If the entry is not an atom:entry 
320     *              or if there is an exception extracting the data. 
321     */
322    public SwordValidationInfo unmarshallWithValidation(Element entry, Properties validationProperties)
323    throws UnmarshallException
324    {
325       SwordValidationInfo result = super.unmarshallWithoutValidate(entry, validationProperties);
326 
327       processUnexpectedAttributes(entry, result);
328 
329       // retrieve all of the sub-elements
330       Elements elements = entry.getChildElements();
331       Element element = null; 
332       int length = elements.size();
333 
334       for(int i = 0; i < length; i++ )
335       {
336     	  element = elements.get(i);
337 
338     	  if (isInstanceOf(element, SwordTreatment.elementName()))
339     	  {
340              if( swordTreatment == null )
341              {
342                  swordTreatment = new SwordTreatment();
343                  result.addUnmarshallElementInfo(
344                          swordTreatment.unmarshall(element, validationProperties));
345              }
346              else if( validationProperties != null )
347              {
348                  SwordValidationInfo info = new SwordValidationInfo(SwordTreatment.elementName(),
349                            SwordValidationInfo.DUPLICATE_ELEMENT,
350                            SwordValidationInfoType.WARNING);
351                  info.setContentDescription(element.getValue());
352                  result.addUnmarshallElementInfo(info);
353              }
354     	  }
355     	  else if (isInstanceOf(element, SwordNoOp.elementName()))
356     	  {
357     		 if( swordNoOp == null )
358              {
359                  swordNoOp = new SwordNoOp();
360                  result.addUnmarshallElementInfo(swordNoOp.unmarshall(element, validationProperties));
361              }
362              else if( validationProperties != null )
363              {
364                  SwordValidationInfo info = new SwordValidationInfo(SwordNoOp.elementName(),
365                            SwordValidationInfo.DUPLICATE_ELEMENT,
366                            SwordValidationInfoType.WARNING);
367                  info.setContentDescription(element.getValue());
368                  result.addUnmarshallElementInfo(info);
369              }
370     	  }
371     	  else if (isInstanceOf(element, SwordVerboseDescription.elementName()))
372     	  {
373     		 if( swordVerboseDescription == null )
374              {
375                  swordVerboseDescription = new SwordVerboseDescription();
376                  result.addUnmarshallElementInfo(swordVerboseDescription.unmarshall(element, validationProperties));
377              }
378              else if( validationProperties != null )
379              {
380                  SwordValidationInfo info = new SwordValidationInfo(SwordVerboseDescription.elementName(),
381                            SwordValidationInfo.DUPLICATE_ELEMENT,
382                            SwordValidationInfoType.WARNING);
383                  info.setContentDescription(element.getValue());
384                  result.addUnmarshallElementInfo(info);
385              }
386     	  }
387     	  else if (isInstanceOf(element, SwordUserAgent.elementName()))
388     	  {
389     		 if( swordUserAgent == null )
390              {
391                  swordUserAgent = new SwordUserAgent();
392                  result.addUnmarshallElementInfo(swordUserAgent.unmarshall(element, validationProperties));
393              }
394              else if( validationProperties != null )
395              {
396                  SwordValidationInfo info = new SwordValidationInfo(SwordUserAgent.elementName(),
397                            SwordValidationInfo.DUPLICATE_ELEMENT,
398                            SwordValidationInfoType.WARNING);
399                  info.setContentDescription(element.getValue());
400                  result.addUnmarshallElementInfo(info);
401              }
402     	  }
403     	  else if (isInstanceOf(element, SwordPackaging.elementName()))
404     	  {
405     		  if( swordPackaging == null )
406              {
407                  swordPackaging = new SwordPackaging();
408                  result.addUnmarshallElementInfo(swordPackaging.unmarshall(element, validationProperties));
409              }
410              else if( validationProperties != null )
411              {
412                  SwordValidationInfo info = new SwordValidationInfo(SwordPackaging.elementName(),
413                            SwordValidationInfo.DUPLICATE_ELEMENT,
414                            SwordValidationInfoType.WARNING);
415                  info.setContentDescription(element.getValue());
416                  result.addUnmarshallElementInfo(info);
417              }
418     	  }
419           else if (validationProperties != null )
420           {
421               XmlName name = new XmlName(element);
422               if( ! isElementChecked(name) )
423               {
424                  SwordValidationInfo info = new SwordValidationInfo(name,
425                            SwordValidationInfo.UNKNOWN_ELEMENT,
426                            SwordValidationInfoType.INFO);
427                  info.setContentDescription(element.getValue());
428                  result.addUnmarshallElementInfo(info);
429               }
430           }
431 
432       } // for
433       return result;
434    }

436    public SwordValidationInfo unmarshall(Element entry, Properties validationProperties)
437    throws UnmarshallException
438    {
439 
440       SwordValidationInfo result = unmarshallWithValidation(entry, validationProperties);
441       if( validationProperties != null )
442       {
443           result = validate(result, validationProperties);
444       }
445       return result;
446    }

448    /**
449     * 
450     * @param elementName
451     * @return
452     */
453    protected boolean isElementChecked(XmlName elementName)
454    {
455        if( elementName == null )
456        {
457            return false;
458        }
459 
460        return elementName.equals(SwordNoOp.elementName()) |
461               elementName.equals(SwordUserAgent.elementName()) | 
462               elementName.equals(SwordTreatment.elementName()) |
463               elementName.equals(SwordVerboseDescription.elementName()) |
464               elementName.equals(SwordPackaging.elementName()) |
465               super.isElementChecked(elementName);
466    }

473    protected SwordValidationInfo validate(SwordValidationInfo info, 
474            Properties validationContext)
475    {
476       boolean validateAll = (info == null);
477 
478       SwordValidationInfo swordEntry = super.validate(info, validationContext);
479       
480       if( swordUserAgent == null )
481       {
482           String agent = validationContext.getProperty(HttpHeaders.USER_AGENT);
483 
484           if( agent != null )
485           {
486              swordEntry.addValidationInfo(new SwordValidationInfo(SwordUserAgent.elementName(),
487                   SwordValidationInfo.MISSING_ELEMENT_WARNING +
488                      " Clients SHOULD provide a User-Agent request-header (as described in [HTTP1.1] section 14.43). If provided, servers SHOULD store the value in the sword:userAgent element.",
489                   SwordValidationInfoType.WARNING));
490           }
491       }
492       else if( swordUserAgent != null && validateAll )
493       {
494          info.addValidationInfo(swordUserAgent.validate(validationContext));
495       }
496 
497       // additional rules for sword elements
498       if( swordTreatment == null )
499       {
500           swordEntry.addValidationInfo(new SwordValidationInfo(SwordTreatment.elementName(),
501                   SwordValidationInfo.MISSING_ELEMENT_ERROR + " MUST be present and contain either a human-readable statement describing treatment the deposited resource has received or a URI that dereferences to such a description.",
502                   SwordValidationInfoType.ERROR));
503       }
504       else if( swordTreatment != null && validateAll )
505       {
506          info.addValidationInfo(swordTreatment.validate(validationContext));
507       }
508 
509       // additional rules for sword elements
510       if( swordVerboseDescription == null )
511       {
512           String verbose = validationContext.getProperty(HttpHeaders.X_VERBOSE);
513           if( verbose != null )
514           {
515              swordEntry.addValidationInfo(new SwordValidationInfo(SwordVerboseDescription.elementName(),
516                   SwordValidationInfo.MISSING_ELEMENT_WARNING + " If the client made the POST request with an X-Verbose:true header, the server SHOULD supply a verbose description of the deposit process.",
517                   SwordValidationInfoType.WARNING));
518           }
519       }
520       else if( swordVerboseDescription != null && validateAll )
521       {
522          info.addValidationInfo(swordVerboseDescription.validate(validationContext));
523       }
524 
525       if( swordNoOp == null )
526       {
527           String noOp = validationContext.getProperty(HttpHeaders.X_NO_OP);
528           if( noOp != null )
529           {
530              swordEntry.addValidationInfo(new SwordValidationInfo(SwordNoOp.elementName(),
531                   SwordValidationInfo.MISSING_ELEMENT_WARNING + " If the client made the POST request with an X-No-Op:true header, the server SHOULD reflect this by including a sword:noOp element with a value of 'true' in the response. See Part A Section 3.1. Servers MAY use a value of 'false' to indicate that the deposit proceeded but MUST NOT use this element to signify an error.",
532                   SwordValidationInfoType.WARNING));
533           }
534       }
535       else if( swordNoOp != null && validateAll )
536       {
537          info.addValidationInfo(swordNoOp.validate(validationContext));
538       }
539 
540       if( swordPackaging == null )
541       {
542           swordEntry.addValidationInfo(new SwordValidationInfo(SwordPackaging.elementName(),
543                   SwordValidationInfo.MISSING_ELEMENT_WARNING + " If the POST request results in the creation of packaged resource, the server MAY use this element to declare the packaging type. If used it SHOULD take a value from [SWORD-TYPES].",
544                   SwordValidationInfoType.INFO));
545       }
546       else if( swordPackaging != null && validateAll )
547       {
548          info.addValidationInfo(swordPackaging.validate(validationContext));
549       }
550 
551       return swordEntry;
552    }
