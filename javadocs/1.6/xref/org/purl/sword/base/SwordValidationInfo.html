309     protected void resetType()
310     {
311         type = SwordValidationInfoType.VALID;
312         
313         resetType(getValidationElementInfoIterator());
314         resetType(getValidationAttributeInfoIterator());
315         resetType(getUnmarshallElementInfoIterator());
316         resetType(getUnmarshallAttributeInfoIterator());
317     }
318 
319     protected void resetType(Iterator<SwordValidationInfo> iterator)
320     {
321         SwordValidationInfo item = null;
322         while( iterator.hasNext() )
323         {
324             item = iterator.next();
325             if( item != null )
326             {
327                 if( type.compareTo(item.getType()) < 0 )
328                 {
329                     type = item.getType(); 
330                 }
331             }
332         }
333     }

387     /**
388      * 
389      * @param elementItems
390      * @param attributeItems
391      */
392     public void addUnmarshallValidationInfo(
393             ArrayList<SwordValidationInfo> elementItems,
394             ArrayList<SwordValidationInfo> attributeItems)
395     {
396        if( elementItems != null )
397        {
398            Iterator<SwordValidationInfo> items = elementItems.iterator();
399 
400            while( items.hasNext() )
401            {
402               addUnmarshallElementInfo(items.next());
403            }
404        }
405 
406        if( attributeItems != null )
407        {
408           Iterator<SwordValidationInfo> attributes = attributeItems.iterator();
409 
410            while( attributes.hasNext() )
411            {
412               addUnmarshallAttributeInfo(attributes.next());
413            }
414        }
415    }

428    /**
429      * Utility method that will recursively print out the list of items
430      * for the specified validation info object.
431      *
432      * @param info   The validation info object to display.
433      * @param indent The level of indent, expressed as a number of space characters.
434      */
435     public void createString(SwordValidationInfo info, StringBuffer buffer, String indent)
436     {
437        String prefix = info.getElement().getPrefix();
438        buffer.append(indent);
439 
440        buffer.append("[");
441        buffer.append(info.getType());
442        buffer.append("]");
443 
444        if( prefix != null && prefix.trim().length() > 0 )
445        {
446           buffer.append(prefix);
447           buffer.append(":");
448        }
449 
450        buffer.append(info.getElement().getLocalName());
451        buffer.append("  ");
452        if (info.getAttribute() != null) {
453           buffer.append(info.getAttribute().getLocalName());
454           buffer.append("=\"");
455           if( info.getContentDescription() != null )
456           {
457              buffer.append(info.getContentDescription());
458           }
459           buffer.append("\"");
460        }
461        else
462        {
463           if( info.getContentDescription() != null )
464           {
465                  buffer.append(" Value: '");
466                  buffer.append(info.getContentDescription());
467                  buffer.append("'");
468           }
469        }
470        buffer.append("\n" + indent + "message: " );
471        buffer.append(info.getMessage());
472        buffer.append("\n");
473        
474        // process the list of attributes first
475        Iterator<SwordValidationInfo> iterator = info.getValidationAttributeInfoIterator();
476        while( iterator.hasNext())
477        {
478           createString(iterator.next(), buffer, "   " + indent);
479        }
480 
481        iterator = info.getUnmarshallAttributeInfoIterator();
482        while( iterator.hasNext())
483        {
484           createString(iterator.next(), buffer, "   " + indent);
485        }
486 
487        // next, process the element messages
488        iterator = info.getValidationElementInfoIterator();
489        while( iterator.hasNext())
490        {
491           createString(iterator.next(), buffer, "   " + indent);
492        }
493 
494        iterator = info.getUnmarshallElementInfoIterator();
495        while( iterator.hasNext())
496        {
497           createString(iterator.next(), buffer, "   " + indent);
498        }
499     }
