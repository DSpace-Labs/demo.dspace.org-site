37  package org.purl.sword.base;
38  
39  import nu.xom.Attribute;
40  import nu.xom.Element;
41  
42  /**
43   *
44   * @author Neil Taylor (nst@aber.ac.uk)
45   */
46  public class XmlName {
47  
48      /** Prefix for the name */
49      private String prefix;
50  
51      /** Local name */
52      private String localName;
53  
54      /** The namespace for the element */ 
55      private String namespace;
56  
57      
58      /**
59       * Create a new instance with the specified prefix and local name.
60       * @param prefix     The namespace prefix.
61       * @param localName  The element's local name. 
62       */
63      public XmlName(String prefix, String localName, String namespace )
64      {
65          this.prefix = prefix;
66          this.localName = localName;
67          this.namespace = namespace;
68      }
69  
70      public XmlName(Element element)
71      {
72          this.prefix = element.getNamespacePrefix();
73          this.localName = element.getLocalName();
74          this.namespace = element.getNamespaceURI();
75      }
76  
77      public XmlName(Attribute attribute)
78      {
79          this.prefix = attribute.getNamespacePrefix();
80          this.localName = attribute.getLocalName();
81          this.namespace = attribute.getNamespaceURI();
82      }
83  
84      /**
85       * Get the prefix.
86       *
87       * @return the prefix
88       */
89      public String getPrefix() {
90          return prefix;
91      }
92  
93      /**
94       * Set the prefix.
95       *
96       * @param prefix the prefix to set
97       */
98      public void setPrefix(String prefix) {
99          this.prefix = prefix;
100     }
101 
102     /**
103      * Get the local name.
104      *
105      * @return the localName
106      */
107     public String getLocalName() {
108         return localName;
109     }
110 
111     /**
112      * Set the local name.
113      * 
114      * @param localName the localName to set
115      */
116     public void setLocalName(String localName) {
117         this.localName = localName;
118     }
119 
120     /**
121      * Get the current namespace value.
122      * 
123      * @return the namespace
124      */
125     public String getNamespace() {
126         return namespace;
127     }
128 
129     /**
130      * Set the namespace value.
131      *
132      * @param namespace the namespace to set
133      */
134     public void setNamespace(String namespace) {
135         this.namespace = namespace;
136     }
137 
138     public String getQualifiedName()
139     {
140         String qName = "";
141         if( prefix != null && prefix.trim().length() > 0 )
142         {
143            qName = prefix + ":";
144         }
145         qName += localName;
146         return qName;
147     }
148 
149     @Override
150     public boolean equals(Object other)
151     {
152 
153         if( other instanceof XmlName )
154         {
155             XmlName otherName = (XmlName) other;
156             try
157             {
158                return (this.namespace.equals(otherName.namespace) &&
159                        this.localName.equals(otherName.localName));
160 
161             }
162             catch(Exception ex) 
163             {
164                 // fall through to the default return case
165             }
166         }
167 
168         return false;
169     }
