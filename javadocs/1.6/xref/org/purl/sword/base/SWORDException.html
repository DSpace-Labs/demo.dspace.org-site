37  package org.purl.sword.base;
38  
39  /**
40   * Represents a generic SWORD exception. If this thrown by a repository,
41   * it would result in a HTTP 500 message being returned to the user.
42   * 
43   * @author Stuart Lewis
44   * @author Neil Taylor
45   */
46  public class SWORDException extends Exception 
47  {
48  	private String errorCode;
49  	
50  	/**
51       * Create a new instance and store the specified message and source data. 
52       * 
53       * @param message The message for the exception. 
54       * @param source  The original exception that lead to this exception. This
55       *                can be <code>null</code>.
56       */
57      public SWORDException(String message, Exception source)
58      {  
59         super(message, source); 
60      }
61      
62      /**
63       * Create a new instance and store the specified message and source data. 
64       * 
65       * @param message The message for the exception. 
66       * @param source  The original exception that lead to this exception. This
67       *                can be <code>null</code>.
68       * @param errorCode The error code to sed back with the request
69       */
70      public SWORDException(String message, Exception source, String errorCode)
71      {  
72         super(message, source); 
73         this.errorCode = errorCode;
74      }
75      
76      /**
77       * Create a new instance and store the specified message. 
78       * 
79       * @param message The message for the exception. 
80       */
81      public SWORDException(String message)
82      {
83         super(message);
84      }
85      
86      /**
87       * Get the error code
88       * 
89       * @return The error code
90       */
91      public String getErrorCode() {
92      	return errorCode;
93      }
94  }
