179    /**
180     * Marshall the data in this element to an Element. 
181     * 
182     * @return An element that contains the data in this object. 
183     */
184    public Element marshall( ) 
185    {
186       // convert data into XOM elements and return the 'root', i.e. the one 
187       // that represents the collection. 
188       Element workspace = new Element(xmlName.getQualifiedName(), xmlName.getNamespace());
189 
190       if( title != null )
191       {
192          workspace.appendChild(title.marshall());
193       }
194       
195       for( Collection item : collections )
196       {
197          workspace.appendChild(item.marshall());
198       }
199 
200       return workspace;   
201    }

216    /**
217     *
218     * @param workspace
219     * @param validate
220     * @return
221     * @throws org.purl.sword.base.UnmarshallException
222     */
223    public SwordValidationInfo unmarshall( Element workspace, Properties validationProperties )
224    throws UnmarshallException
225    {
226       if( ! isInstanceOf(workspace, xmlName))
227       {
228          return handleIncorrectElement(workspace, validationProperties);
229       }
230 
231       ArrayList<SwordValidationInfo> validationItems = new ArrayList<SwordValidationInfo>();
232 
233       try
234       {
235          initialise();
236 
237          // FIXME - process the attributes 
238 
239          // retrieve all of the sub-elements
240          Elements elements = workspace.getChildElements();
241          Element element = null;
242          int length = elements.size();
243 
244          for(int i = 0; i < length; i++ )
245          {
246             element = elements.get(i);
247             if( isInstanceOf(element, Title.elementName() ) )
248             {
249                if( title == null )
250                {
251                   title = new Title();
252                   validationItems.add(title.unmarshall(element, validationProperties));
253                }
254                else
255                {
256                   SwordValidationInfo info =
257                           new SwordValidationInfo(Title.elementName(),
258                              SwordValidationInfo.DUPLICATE_ELEMENT,
259                              SwordValidationInfoType.WARNING);
260                   info.setContentDescription(element.getValue());
261                   validationItems.add(info);
262                }
263             }
264             else if( isInstanceOf(element, Collection.elementName() ))
265             {
266                Collection collection = new Collection( );
267                validationItems.add(collection.unmarshall(element, validationProperties));
268                collections.add(collection); 
269             }
270             else if( validationProperties != null )
271             {
272                 validationItems.add(new SwordValidationInfo(new XmlName(element),
273                         SwordValidationInfo.UNKNOWN_ELEMENT,
274                         SwordValidationInfoType.INFO));
275             }
276          }
277       }
278       catch( Exception ex )
279       {
280          log.error("Unable to parse an element in workspace: " + ex.getMessage());
281          throw new UnmarshallException("Unable to parse element in workspace.", ex);
282       }
283 
284       SwordValidationInfo result = null;
285       if( validationProperties != null )
286       {
287           result = validate(validationItems, validationProperties);
288       }
289       return result; 
290    }

302    /**
303     * 
304     * @param existing
305     * @return
306     */
307    protected SwordValidationInfo validate(ArrayList<SwordValidationInfo> existing,
308            Properties validationContext)
309    {
310       boolean validateAll = (existing == null );
311 
312       SwordValidationInfo result = new SwordValidationInfo(xmlName);
313 
314       if( collections == null || collections.size() == 0 )
315       {
316           result.addValidationInfo(new SwordValidationInfo(Collection.elementName(),
317                     SwordValidationInfo.MISSING_ELEMENT_WARNING,
318                     SwordValidationInfoType.WARNING ));
319       }
320 
321       if( validateAll )
322       {
323           if( title != null )
324           {
325               result.addValidationInfo(title.validate(validationContext));
326           }
327 
328           if( collections.size() > 0 )
329           {
330              Iterator<Collection> iterator = collections.iterator();
331              while( iterator.hasNext() )
332              {
333                  result.addValidationInfo(iterator.next().validate(validationContext));
334              }
335           }
336       }
337 
338       result.addUnmarshallValidationInfo(existing, null);
339       return result; 
340    }
