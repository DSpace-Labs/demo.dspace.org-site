37  package org.purl.sword.base;

40  import java.io.InputStream;
41  import java.util.Enumeration;
42  import java.util.Properties;
43  
44  import org.apache.log4j.Logger;
45  
46  /**
47   *
48   * @author Neil Taylor (nst@aber.ac.uk)
49   */
50  public class SwordContentPackageTypes {
51  
52      private static Logger log = Logger.getLogger(SwordContentPackageTypes.class);
53  
54      private static Properties types;
55  
56      public SwordContentPackageTypes()
57      {
58          // static constructor to attempt to load the properties file
59          try
60          {
61  
62              types = new Properties();
63              InputStream stream = SwordContentPackageTypes.class.getClassLoader().getResourceAsStream("swordContentPackageTypes.properties");
64              if( stream != null ) 
65              {
66                  types.loadFromXML(stream);
67              }
68          }
69          catch(Exception ex)
70          {
71              log.error("Unable to load sword types property file: " + ex.getMessage());
72          }
73      }
74  
75      private static SwordContentPackageTypes instance;
76  
77      public static SwordContentPackageTypes instance()
78      {
79         if( instance == null )
80         {
81             instance = new SwordContentPackageTypes();
82         }
83         return instance;
84      }
85  
86      public boolean isValidType(String uri)
87      {
88          return types.containsKey(uri);
89      }
90  
91      public boolean isEmpty()
92      {
93          return types.isEmpty();
94      }
95  
96      /**
97       *
98       * @return
99       */
100     public Enumeration elements()
101     {
102 
103         return types.elements();
104     }
105     
106     public Enumeration keys()
107     {
108         return types.keys();
109     }
