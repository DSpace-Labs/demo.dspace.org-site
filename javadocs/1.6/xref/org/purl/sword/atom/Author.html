188    /**
189     * Unmarshall the author details from the specified element. The element
190     * is a XOM element.
191     *
192     * @param author The element to unmarshall.
193     */
194    public SwordValidationInfo unmarshall(Element author, Properties validationProperties)
195    throws UnmarshallException
196    {
197       if( ! isInstanceOf( author, xmlName) )
198       {
199          handleIncorrectElement(author, validationProperties);
200       }
201 
202       ArrayList<SwordValidationInfo> validationItems = new ArrayList<SwordValidationInfo>();
203       ArrayList<SwordValidationInfo> attributeItems = new ArrayList<SwordValidationInfo>();
204 
205       processUnexpectedAttributes(author, attributeItems);
206       
207       // retrieve all of the sub-elements
208       Elements elements = author.getChildElements();
209       Element element = null;
210       int length = elements.size();
211 
212       for(int i = 0; i < length; i++ )
213       {
214          element = elements.get(i);
215 
216          if( isInstanceOf(element, Name.elementName() ))
217          {
218             name = new Name();
219             validationItems.add(name.unmarshall(element, validationProperties));
220          }
221          else if( isInstanceOf(element, Uri.elementName()))
222          {
223             uri = new Uri();
224             validationItems.add(uri.unmarshall(element, validationProperties));
225             
226          }
227          else if( isInstanceOf(element, Email.elementName() ))
228          {
229             email = new Email();
230             validationItems.add(email.unmarshall(element, validationProperties));
231          }
232          else if( validationProperties != null )
233          {
234             SwordValidationInfo info = new SwordValidationInfo(new XmlName(element),
235                     SwordValidationInfo.UNKNOWN_ELEMENT,
236                     SwordValidationInfoType.INFO);
237             info.setContentDescription(element.getValue());
238             validationItems.add(info);
239          }
240 
241       } // for
242 
243       SwordValidationInfo result = null;
244       if( validationProperties != null )
245       {
246           result = validate(validationItems, attributeItems, validationProperties);
247       }
248       return result;
249    }

256    public SwordValidationInfo validate(ArrayList<SwordValidationInfo> elements,
257            ArrayList<SwordValidationInfo> attributes,
258            Properties validationContext)
259    {
260        SwordValidationInfo result = new SwordValidationInfo(xmlName);
261 
262        if( name == null )
263        {
264            SwordValidationInfo info = new SwordValidationInfo(Name.elementName(),
265                    SwordValidationInfo.MISSING_ELEMENT_ERROR,
266                    SwordValidationInfoType.ERROR);
267            result.addValidationInfo(info);
268        }
269        else if( elements == null && name != null)
270        {
271           result.addValidationInfo(name.validate(validationContext));
272        }
273 
274        if( elements == null && uri != null )
275        {
276            result.addValidationInfo(uri.validate(validationContext));
277        }
278 
279        if( elements == null && email != null )
280        {
281            result.addValidationInfo(email.validate(validationContext));
282        }
283 
284        result.addUnmarshallValidationInfo(elements, attributes);
285        return result;
286    }
