233    protected boolean isElementChecked(XmlName elementName)
234    {
235        if( elementName == null )
236        {
237            return false;
238        }
239 
240        return elementName.equals(Author.elementName()) |
241               elementName.equals(Category.elementName()) |
242               elementName.equals(Content.elementName()) |
243               elementName.equals(Generator.elementName()) |
244               elementName.equals(Contributor.elementName()) |
245               elementName.equals(Id.elementName()) |
246               elementName.equals(Link.elementName()) |
247               elementName.equals(Published.elementName()) |
248               elementName.equals(Rights.elementName()) |
249               elementName.equals(Source.elementName()) |
250               elementName.equals(Summary.elementName()) |
251               elementName.equals(Title.elementName()) |
252               elementName.equals(Updated.elementName());
253    }

266    /**
267     * Mashall the data stored in this object into Element objects. 
268     * 
269     * @return An element that holds the data associated with this object. 
270     */
271    public Element marshall()
272    {
273       Element entry = new Element(getQualifiedName(), Namespaces.NS_ATOM);
274       entry.addNamespaceDeclaration(Namespaces.PREFIX_SWORD, Namespaces.NS_SWORD);
275       entry.addNamespaceDeclaration(Namespaces.PREFIX_ATOM, Namespaces.NS_ATOM);
276       this.marshallElements(entry);
277       return entry;  
278    }

280    protected void marshallElements(Element entry)
281    {
282 	      if (id != null)
283 	      {
284              entry.appendChild(id.marshall());
285 	      }
286 
287 	      for (Author author : authors)
288 	      {
289 	         entry.appendChild(author.marshall());
290 	      }
291 
292 	      if (content != null)
293 	      {

295 	      }
296 
297 	      if (generator != null)
298 	      {
299 	         entry.appendChild(generator.marshall());
300 	      }
301 
302 	      for (Author contributor : contributors)
303 	      {
304 	         entry.appendChild(contributor.marshall());
305 	      }
306 
307 	      for (Link link : links)
308 	      {
309 	         entry.appendChild(link.marshall());
310 	      }
311 
312 	      if (published != null)
313 	      {
314              entry.appendChild(published.marshall());
315 	      }
316 
317 	      if (rights != null)
318 	      {
319 	         entry.appendChild(rights.marshall());
320 	      }
321 
322 	      if (summary != null)
323 	      {
324 	         entry.appendChild(summary.marshall());
325 	      }
326 
327 	      if (title != null)
328 	      {
329 	         entry.appendChild(title.marshall());
330 	      }
331 
332 	      if (source != null)

334 	         entry.appendChild(source.marshall());
335 	      }
336

339              entry.appendChild(updated.marshall());

364    public SwordValidationInfo unmarshallWithoutValidate(Element entry, Properties validationProperties)
365    throws UnmarshallException
366    {
367 	  if (! isInstanceOf(entry, xmlName) )
368       {
369          return handleIncorrectElement(entry, validationProperties);
370       }
371 
372       // used to hold the element and attribute unmarshall info results
373       SwordValidationInfo result = new SwordValidationInfo(xmlName);
374 
375       try
376       {
377          initialise();
378 
379          // FIXME - attributes? 
380 
381          // retrieve all of the sub-elements
382          Elements elements = entry.getChildElements();
383          Element element = null; 
384          int length = elements.size();
385 
386          for (int i = 0; i < length; i++)
387          {
388             element = elements.get(i);
389 
390             if (isInstanceOf(element, Author.elementName()))
391             {
392                Author author = new Author(); 
393                result.addUnmarshallElementInfo(author.unmarshall(element, validationProperties));
394                authors.add(author);
395             }
396             else if (isInstanceOf(element, Category.elementName()))
397             {
398                Category category = new Category();
399                result.addUnmarshallElementInfo(category.unmarshall(element, validationProperties));
400                categories.add(category); 
401             }
402             else if (isInstanceOf(element, Content.elementName()))
403             {
404                if( content == null )
405                {
406                   content = new Content();
407                   result.addUnmarshallElementInfo(content.unmarshall(element, validationProperties));
408                }
409                else if( validationProperties != null )
410                {
411                    SwordValidationInfo info = new SwordValidationInfo(Content.elementName(),
412                           SwordValidationInfo.DUPLICATE_ELEMENT,
413                           SwordValidationInfoType.WARNING);
414                   info.setContentDescription(element.getValue());
415                   result.addUnmarshallElementInfo(info);
416                }
417 
418             }
419             else if (isInstanceOf(element, Generator.elementName()))
420             {
421                if( generator == null )
422                {
423                   generator = new Generator();
424                   result.addUnmarshallElementInfo(generator.unmarshall(element, validationProperties));
425                }
426                else if( validationProperties != null )
427                {
428                   SwordValidationInfo info = new SwordValidationInfo(Generator.elementName(),
429                           SwordValidationInfo.DUPLICATE_ELEMENT,
430                           SwordValidationInfoType.WARNING);
431                   info.setContentDescription(element.getValue());
432                   result.addUnmarshallElementInfo(info);
433                }
434             }
435             else if (isInstanceOf(element, Contributor.elementName()))
436             {
437                Contributor contributor = new Contributor(); 
438                result.addUnmarshallElementInfo(contributor.unmarshall(element, validationProperties));
439                contributors.add(contributor);
440             }
441             else if (isInstanceOf(element, Id.elementName()))
442             {
443                if( id == null )
444                {
445                  id = new Id();
446                  result.addUnmarshallElementInfo(id.unmarshall(element, validationProperties));
447                }
448                else if( validationProperties != null )
449                {
450                   SwordValidationInfo info = new SwordValidationInfo(Id.elementName(),
451                           SwordValidationInfo.DUPLICATE_ELEMENT,
452                           SwordValidationInfoType.WARNING);
453                   info.setContentDescription(element.getValue());
454                   result.addUnmarshallElementInfo(info);
455                }
456 
457             }
458             else if (isInstanceOf(element, Link.elementName()))
459             {
460                Link link = new Link(); 
461                result.addUnmarshallElementInfo(link.unmarshall(element, validationProperties));
462                links.add(link);
463             }
464             else if (isInstanceOf(element, Published.elementName()))
465             {
466                if( published == null )
467                {
468                   published = new Published();
469                   result.addUnmarshallElementInfo(published.unmarshall(element, validationProperties));
470                }
471                else if( validationProperties != null )
472                {
473                   SwordValidationInfo info = new SwordValidationInfo(Published.elementName(),
474                           SwordValidationInfo.DUPLICATE_ELEMENT,
475                           SwordValidationInfoType.WARNING);
476                   info.setContentDescription(element.getValue());
477                   result.addUnmarshallElementInfo(info);
478                }
479             }
480             else if (isInstanceOf(element, Rights.elementName()))
481             {
482                if( rights == null )
483                {
484                   rights = new Rights();
485                   result.addUnmarshallElementInfo(rights.unmarshall(element, validationProperties));
486                }
487                else if( validationProperties != null )
488                {
489                   SwordValidationInfo info = new SwordValidationInfo(Rights.elementName(),
490                           SwordValidationInfo.DUPLICATE_ELEMENT,
491                           SwordValidationInfoType.WARNING);
492                   info.setContentDescription(element.getValue());
493                   result.addUnmarshallElementInfo(info);
494                }
495             }
496             else if (isInstanceOf(element, Summary.elementName()))
497             {
498                if( summary == null )
499                {
500                   summary = new Summary();
501                   result.addUnmarshallElementInfo(summary.unmarshall(element, validationProperties));
502                }
503                else if( validationProperties != null )
504                {
505                   SwordValidationInfo info = new SwordValidationInfo(Summary.elementName(),
506                           SwordValidationInfo.DUPLICATE_ELEMENT,
507                           SwordValidationInfoType.WARNING);

509                   result.addUnmarshallElementInfo(info);
510                }
511             }
512             else if (isInstanceOf(element, Title.elementName()))
513             {
514                if( title == null )
515                {
516                   title = new Title();
517                   result.addUnmarshallElementInfo(title.unmarshall(element, validationProperties));
518                }

521                   SwordValidationInfo info = new SwordValidationInfo(Title.elementName(),
522                           SwordValidationInfo.DUPLICATE_ELEMENT,
523                           SwordValidationInfoType.WARNING);
524                   info.setContentDescription(element.getValue());
525                   result.addUnmarshallElementInfo(info);
526                }
527             }
528             else if (isInstanceOf(element, Updated.elementName()))
529             {
530                if( updated == null )
531                {
532                   updated = new Updated();
533                   result.addUnmarshallElementInfo(updated.unmarshall(element, validationProperties));
534                }
535                else if( validationProperties != null )
536                {
537                   SwordValidationInfo info = new SwordValidationInfo(Updated.elementName(),
538                           SwordValidationInfo.DUPLICATE_ELEMENT,
539                           SwordValidationInfoType.WARNING);
540                   info.setContentDescription(element.getValue());
541                   result.addUnmarshallElementInfo(info);
542                }
543             }
544             else if (isInstanceOf(element, Source.elementName()))
545             {
546                if( source == null )
547                {
548                   source = new Source();
549                   result.addUnmarshallElementInfo(source.unmarshall(element, validationProperties));
550                }
551                else if( validationProperties != null )
552                {
553                   SwordValidationInfo info = new SwordValidationInfo(Source.elementName(),
554                           SwordValidationInfo.DUPLICATE_ELEMENT,
555                           SwordValidationInfoType.WARNING);
556                   info.setContentDescription(element.getValue());
557                   result.addUnmarshallElementInfo(info);
558                }
559             }
560             else if( validationProperties != null )
561             {
562                 XmlName name = new XmlName(element);
563                 if( ! isElementChecked(name) )
564                 {
565                    SwordValidationInfo info = new SwordValidationInfo(name,
566                           SwordValidationInfo.UNKNOWN_ELEMENT,
567                           SwordValidationInfoType.INFO);
568                    info.setContentDescription(element.getValue());
569                    result.addUnmarshallElementInfo(info);
570                 }
571             }
572 
573          } // for 
574       }
575       catch (Exception ex)
576       {
577          log.error("Unable to parse an element in Entry: " + ex.getMessage());
578          ex.printStackTrace();
579          throw new UnmarshallException("Unable to parse an element in " + getQualifiedName(), ex);
580       }
581 
582       return result; 
583    }

585    public SwordValidationInfo unmarshall(Element entry, Properties validationProperties)
586    throws UnmarshallException
587    {
588 
589       SwordValidationInfo result = unmarshallWithoutValidate(entry, validationProperties);
590       if( validationProperties != null )
591       {
592           result = validate(result, validationProperties);
593       }
594       return result; 
595    }

606    /**
607     *
608     * @param elementsInfo
609     * @param attributeInfo
610     * @return
611     */
612    protected SwordValidationInfo validate(SwordValidationInfo info, Properties validationContext)
613    {
614        // determine if a full validation is required 
615        boolean validateAll = (info == null);
616 
617        SwordValidationInfo result = info;
618        if( result == null )
619        {
620           result = new SwordValidationInfo(xmlName);
621        }
622        
623        // id, title an updated are required
624        if( id == null )
625        {
626           result.addValidationInfo(new SwordValidationInfo(Id.elementName(),
627                   SwordValidationInfo.MISSING_ELEMENT_ERROR,
628                   SwordValidationInfoType.ERROR));
629        }
630        else if( id != null && validateAll )
631        {
632            result.addValidationInfo(id.validate(validationContext));
633        }
634 
635        if( title == null )
636        {
637           result.addValidationInfo(new SwordValidationInfo(Title.elementName(),
638                   SwordValidationInfo.MISSING_ELEMENT_ERROR,
639                   SwordValidationInfoType.ERROR));
640        }
641        else if( title != null && validateAll )
642        {
643            result.addValidationInfo(title.validate(validationContext));
644        }
645 
646        if( updated == null )
647        {
648           result.addValidationInfo(new SwordValidationInfo(Updated.elementName(),
649                   SwordValidationInfo.MISSING_ELEMENT_ERROR,
650                   SwordValidationInfoType.ERROR));
651        }
652        else if( updated != null && validateAll )
653        {
654            result.addValidationInfo(updated.validate(validationContext));
655        }
656 
657        // additional sword requirements on the element
658        if( contributors.isEmpty() )
659        {
660           String contributor = validationContext.getProperty(HttpHeaders.X_ON_BEHALF_OF);
661           if( contributor != null )
662           {
663              result.addValidationInfo(new SwordValidationInfo(Contributor.elementName(),
664                   SwordValidationInfo.MISSING_ELEMENT_ERROR +
665                     " This item SHOULD contain the value of the X-On-Behalf-Of header, if one was present in the POST request.",
666                   SwordValidationInfoType.ERROR));
667           }
668        }
669        else if( (! contributors.isEmpty()) && validateAll )
670        {
671            Iterator<Contributor> iterator = contributors.iterator();
672            while( iterator.hasNext() )
673            {
674               Contributor contributor = iterator.next();
675               result.addValidationInfo(contributor.validate(validationContext));
676            }
677        }
678 
679        if( generator == null )
680        {
681            result.addValidationInfo(new SwordValidationInfo(Generator.elementName(),
682                    SwordValidationInfo.MISSING_ELEMENT_ERROR +
683                     " SHOULD contain the URI and version of the server software.",
684                    SwordValidationInfoType.ERROR));
685        }
686        else if( generator != null && validateAll )
687        {
688            result.addValidationInfo(generator.validate(validationContext));
689        }
690 
691        if( validateAll )
692        {
693            // process the remaining items
694            Iterator<Link> linksIterator = links.iterator();
695            while( linksIterator.hasNext() )
696            {
697               Link link = linksIterator.next();
698               result.addValidationInfo(link.validate(validationContext));
699            }
700 
701            Iterator<Author> authorIterator = authors.iterator();
702            while( authorIterator.hasNext() )
703            {
704               Author author = authorIterator.next();
705               result.addValidationInfo(author.validate(validationContext));
706            }
707 
708            if( content != null )
709            {
710                result.addValidationInfo(content.validate(validationContext));
711            }
712 
713            if( published != null )
714            {
715                result.addValidationInfo(published.validate(validationContext));
716            }
717 
718            if( rights != null )
719            {
720                result.addValidationInfo(rights.validate(validationContext));
721            }
722 
723            if( summary != null )
724            {
725                result.addValidationInfo(summary.validate(validationContext));
726            }
727 
728            Iterator<Category> categoryIterator = categories.iterator();
729            while( categoryIterator.hasNext() )
730            {
731               Category category = categoryIterator.next();
732               result.addValidationInfo(category.validate(validationContext));
733            }
734            
735        }
736 
737        return result;
738    }
