137    /**
138     * Set the type for the content. This should match the pattern 
139     * ".* /.*" [Note, there is no space before the /, this has been added
140     * to allow this text to be written in a Java comment.]. 
141     * 
142     * An example of the type is <code>application/zip</code>. 
143     * 
144     * @param type The specified type. 
145     * @throws InvalidMediaTypeException If the specified type is null or
146     * it does not match the specified pattern. 
147     */
148    public void setType(String type)
149    throws InvalidMediaTypeException
150    {
151       if( type == null || ! type.matches(".*/.*") )
152       {
153          throw new InvalidMediaTypeException("Type: '" + type + "' does not match .*/.*");
154       }
155       
156       this.type = type;
157    }

164    public Element marshall()
165    {
166       Element content = new Element(getQualifiedName(), Namespaces.NS_ATOM);
167          
168       if( type != null )
169       {
170          Attribute typeAttribute = new Attribute(ATTRIBUTE_TYPE, type);
171          content.addAttribute(typeAttribute);
172       }
173        
174       if( source != null )
175       {
176          Attribute typeAttribute = new Attribute(ATTRIBUTE_SRC, source);
177          content.addAttribute(typeAttribute);
178       }
179       
180       return content;

189    /**
190     * Unmarshall the content element into the data in this object. 
191     * 
192     * @throws UnmarshallException If the element does not contain a
193     *                             content element or if there are problems
194     *                             accessing the data. 
195     */
196    public SwordValidationInfo unmarshall(Element content, Properties validationProperties)
197    throws UnmarshallException 
198    {
199 
200       if( ! isInstanceOf( content, xmlName.getLocalName(), Namespaces.NS_ATOM))
201       {
202          return handleIncorrectElement(content, validationProperties);
203       }
204 
205       ArrayList<SwordValidationInfo> elements = new ArrayList<SwordValidationInfo>();
206       ArrayList<SwordValidationInfo> attributes = new ArrayList<SwordValidationInfo>();
207       
208       try
209       {
210          // get the attributes
211          int attributeCount = content.getAttributeCount();
212          Attribute attribute = null; 
213          for( int i = 0; i < attributeCount; i++ )
214          {
215             attribute = content.getAttribute(i);
216             String name = attribute.getQualifiedName();
217             if( ATTRIBUTE_TYPE.equals(name))
218             {
219                 type = attribute.getValue();
220                 if( validationProperties != null )
221                 {
222                    attributes.add(createValidAttributeInfo(ATTRIBUTE_TYPE, type));
223                 }
224             }
225             else if( ATTRIBUTE_SRC.equals(name) )
226             {
227                source = attribute.getValue();
228                if( validationProperties != null )
229                {
230                   attributes.add(createValidAttributeInfo(ATTRIBUTE_SRC, source));
231                }
232             }
233             else
234             {
235                SwordValidationInfo info = new SwordValidationInfo(xmlName,
236                        new XmlName(attribute),
237                        SwordValidationInfo.UNKNOWN_ATTRIBUTE,
238                        SwordValidationInfoType.INFO );
239                 info.setContentDescription(attribute.getValue());
240                 attributes.add(info);
241             }
242          }
243 
244          // check if there is any content. If there is, add a simple message to
245          // say that there are sub elements that are not used in this profile
246          if( content.getChildCount() > 0 )
247          {
248             elements.add(new SwordValidationInfo(xmlName,
249                     "This element has child elements. These are not expected as part of the SWORD profile",
250                     SwordValidationInfoType.INFO));
251          }
252 
253       }
254       catch( Exception ex )
255       {
256          log.error("Unable to parse an element in Content: " + ex.getMessage());
257          throw new UnmarshallException("Error parsing Content", ex);
258       }
259 
260       SwordValidationInfo result = null;
261       if( validationProperties != null )
262       {
263           result = validate(elements, attributes, validationProperties);
264       }
265       return result;
266    }

273    /**
274     * 
275     * @param elements
276     * @param attributes
277     * @return
278     */
279    protected SwordValidationInfo validate(ArrayList<SwordValidationInfo> elements,
280            ArrayList<SwordValidationInfo> attributes,
281            Properties validationContext)
282    {
283        SwordValidationInfo info = new SwordValidationInfo(xmlName);
284 
285        if( source == null )
286        {
287            XmlName attributeName = new XmlName(xmlName.getPrefix(),
288                        ATTRIBUTE_SRC,
289                        xmlName.getNamespace());
290 
291           SwordValidationInfo item = new SwordValidationInfo(xmlName, attributeName,
292                   SwordValidationInfo.MISSING_ATTRIBUTE_WARNING,
293                   SwordValidationInfoType.ERROR);
294           info.addValidationInfo(item);
295        }
296 
297        info.addUnmarshallValidationInfo(elements, attributes);
298        return info; 
299    }
