131         // create the CC bundle if it doesn't exist
132         // If it does, remove it and create a new one.
133     private static Bundle getCcBundle(Item item)
134         throws SQLException, AuthorizeException, IOException
135     {
136         Bundle[] bundles = item.getBundles(CC_BUNDLE_NAME);
137 
138         if ((bundles.length > 0) && (bundles[0] != null))
139         {
140             item.removeBundle(bundles[0]);
141         }
142         return item.createBundle(CC_BUNDLE_NAME);
143     }

146     /**
147      * This is a bit of the "do-the-right-thing" method for CC stuff in an item
148      */
149     public static void setLicense(Context context, Item item,
150             String cc_license_url) throws SQLException, IOException,
151             AuthorizeException
152     {
153         Bundle bundle = getCcBundle(item);
154 
155         // get some more information
156         String license_text = fetchLicenseText(cc_license_url);
157         String license_rdf = fetchLicenseRDF(cc_license_url);
158         
159         // set the format
160         BitstreamFormat bs_format = BitstreamFormat.findByShortDescription(
161                 context, "License");
162 
163         // set the URL bitstream
164         setBitstreamFromBytes(item, bundle, BSN_LICENSE_URL, bs_format,
165                 cc_license_url.getBytes());
166 
167         // set the license text bitstream
168         setBitstreamFromBytes(item, bundle, BSN_LICENSE_TEXT, bs_format,
169                 license_text.getBytes());
170 
171         // set the RDF bitstream
172         setBitstreamFromBytes(item, bundle, BSN_LICENSE_RDF, bs_format,
173                 license_rdf.getBytes());
174     }

176     public static void setLicense(Context context, Item item,
177                                   InputStream licenseStm, String mimeType)
178             throws SQLException, IOException, AuthorizeException
179     {
180         Bundle bundle = getCcBundle(item);
181 
182         // generic "License" format -- change for CC?
183         BitstreamFormat bs_format = BitstreamFormat.findByShortDescription(
184                 context, "License");
185 
186         Bitstream bs = bundle.createBitstream(licenseStm);
187         bs.setSource(CC_BS_SOURCE);
188         bs.setName((mimeType != null &&
189                     (mimeType.equalsIgnoreCase("text/xml") ||
190                      mimeType.equalsIgnoreCase("text/rdf"))) ?
191                    BSN_LICENSE_RDF : BSN_LICENSE_TEXT);
192         bs.setFormat(bs_format);
193         bs.update();
194     }

208     public static boolean hasLicense(Context context, Item item)
209             throws SQLException, IOException
210     {
211         // try to find CC license bundle
212         Bundle[] bundles = item.getBundles(CC_BUNDLE_NAME);
213 
214         if (bundles.length == 0)
215         {
216             return false;
217         }
218 
219         // verify it has correct contents
220         try
221         {
222             if ((getLicenseURL(item) == null) || (getLicenseText(item) == null)
223                     || (getLicenseRDF(item) == null))
224             {
225                 return false;
226             }
227         }
228         catch (AuthorizeException ae)
229         {
230             return false;
231         }
232 
233         return true;
234     }

276      * Get a few license-specific properties. We expect these to be cached at
277      * least per server run.

279     public static String fetchLicenseText(String license_url)
280     {
281         String text_url = license_url;

284         return (urlBytes != null) ? new String(urlBytes) : "";

288     {
289         StringWriter result = new StringWriter();

309     /**
310      * This helper method takes some bytes and stores them as a bitstream for an
311      * item, under the CC bundle, with the given bitstream name
312      */
313     private static void setBitstreamFromBytes(Item item, Bundle bundle,
314             String bitstream_name, BitstreamFormat format, byte[] bytes)
315             throws SQLException, IOException, AuthorizeException
316     {
317         ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
318         Bitstream bs = bundle.createBitstream(bais);
319 
320         bs.setName(bitstream_name);
321         bs.setSource(CC_BS_SOURCE);
322 
323         bs.setFormat(format);
324 
325         // commit everything
326         bs.update();
327     }

329     /**
330      * This helper method wraps a String around a byte array returned from the
331      * bitstream method further down

333     private static String getStringFromBitstream(Item item,
334             String bitstream_name) throws SQLException, IOException,
335             AuthorizeException
336     {

341             return null;
342         }

344         return new String(bytes);

347     /**
348      * This helper method retrieves the bytes of a bitstream for an item under
349      * the CC bundle, with the given bitstream name
350      */
351     private static Bitstream getBitstream(Item item, String bitstream_name)
352             throws SQLException, IOException, AuthorizeException
353     {
354         Bundle cc_bundle = null;
355 
356         // look for the CC bundle
357         try
358         {
359             Bundle[] bundles = item.getBundles(CC_BUNDLE_NAME);
360 
361             if ((bundles != null) && (bundles.length > 0))
362             {
363                 cc_bundle = bundles[0];
364             }
365             else
366             {
367                 return null;
368             }
369         }
370         catch (Exception exc)
371         {
372             // this exception catching is a bit generic,
373             // but basically it happens if there is no CC bundle
374             return null;
375         }
376 
377         return cc_bundle.getBitstreamByName(bitstream_name);
378     }
379 
380     private static byte[] getBytesFromBitstream(Item item, String bitstream_name)
381             throws SQLException, IOException, AuthorizeException
382     {
383         Bitstream bs = getBitstream(item, bitstream_name);
384 
385         // no such bitstream
386         if (bs == null)
387         {
388             return null;
389         }
390 
391         // create a ByteArrayOutputStream
392         ByteArrayOutputStream baos = new ByteArrayOutputStream();
393         Utils.copy(bs.retrieve(), baos);
394 
395         return baos.toByteArray();
396     }

398     /**
399      * Fetch the contents of a URL
400      */
401     private static byte[] fetchURL(String url_string)
402     {
403         try
404         {
405             URL url = new URL(url_string);
406             URLConnection connection = url.openConnection();
407             byte[] bytes = new byte[connection.getContentLength()];
408 
409             // loop and read the data until it's done
410             int offset = 0;
411 
412             while (true)
413             {
414                 int len = connection.getInputStream().read(bytes, offset,
415                         bytes.length - offset);
416 
417                 if (len == -1)
418                 {
419                     break;
420                 }
421 
422                 offset += len;
423             }
424 
425             return bytes;
426         }
427         catch (Exception exc)
428         {
429             return null;
430         }
431     }
