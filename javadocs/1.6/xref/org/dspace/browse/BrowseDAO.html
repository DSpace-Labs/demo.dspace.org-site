42  /**
43   * Interface for any class wishing to interact with the Browse storage layer for
44   * Read Only operations.  If you wish to modify the contents of the browse indices
45   * or create and destroy index tables you should look at implementations for
46   * BrowseCreateDAO.
47   *
48   * If you implement this class, and you wish it to be loaded via the BrowseDAOFactory
49   * you must supply a constructor of the form:
50   *
51   * public BrowseDAOImpl(Context context) {}
52   *
53   * Where Context is the DSpace Context object
54   *
55   * Where tables are referred to in this class, they can be obtained from the BrowseIndex
56   * class, which will answer queries given the context of the request on which table
57   * is the relevant target.
58   *
59   * @author Richard Jones
60   *
61   */
62  public interface BrowseDAO
63  {
64      // Objects implementing this interface should also include
65      // a constructor which takes the DSpace Context as an argument
66      //
67      // public BrowseDAOImpl(Context context) ...

69      /**
70       * This executes a query which will count the number of results for the
71       * parameters you set.
72       *
73       * @return      the integer value of the number of results found
74       * @throws BrowseException
75       */
76      public int doCountQuery() throws BrowseException;
77  
78      /**
79       * This executes a query which returns a List object containing String
80       * values which represent the results of a single value browse (for
81       * example, the list of all subject headings).  This is most
82       * commonly used with a Distinct browse type.
83       *
84       * @return  List of Strings representing the single value query results
85       * @throws BrowseException
86       */
87      public List doValueQuery() throws BrowseException;
88  
89      /**
90       * This executes a query which returns a List object containing BrowseItem objects
91       * represening the results of a full item browse.
92       *
93       * @return  List of BrowseItem objects
94       * @throws BrowseException
95       */
96      public List doQuery() throws BrowseException;
97  
98      /**
99       * This executes a query which returns the value of the "highest" (max) value
100      * in the given table's column for the given item id.
101      *
102      * @param column    the column to interrogate
103      * @param table     the table to query
104      * @param itemID    the item id
105      * @return          String representing the max value in the given column
106      * @throws BrowseException
107      */
108     public String doMaxQuery(String column, String table, int itemID) throws BrowseException;
109 
110     /**
111      * This executes a query which returns the offset where the value (or nearest greater
112      * equivalent) can be found in the specified table ordered by the column.
113      *
114      * @param column    the column to interrogate
115      * @param value     the item id
116      * @param isAscending browsing in ascending or descending order
117      * @return          the offset into the table
118      * @throws BrowseException
119      */
120     public int doOffsetQuery(String column, String value, boolean isAscending) throws BrowseException;
121 
122     /**
123      * This executes a query which returns the offset where the value (or nearest greater
124      * equivalent) can be found in the specified table ordered by the column.
125      *
126      * @param column    the column to interrogate
127      * @param value     the item id
128      * @param isAscending browsing in ascending or descending order
129      * @return          the offset into the table
130      * @throws BrowseException
131      */
132     public int doDistinctOffsetQuery(String column, String value, boolean isAscending) throws BrowseException;
133 
134     /**
135      * Does the query use the equals comparator when doing less than or greater than
136      * comparisons.  @see setEqualsComparator
137      *
138      * Default value is true
139      *
140      * @return  true if using it, false if not
141      */
142     public boolean useEqualsComparator();
143 
144     /**
145      * Set whether the query should use an equals comparator when doing less than or
146      * greater than comparisons.  That is, if true then comparisons will be made
147      * using the equivalent of "<=" and ">=", while if false it will use the
148      * equivalent of "<" and ">"
149      *
150      * @param equalsComparator  true to use, false to not.
151      */
152     public void setEqualsComparator(boolean equalsComparator);
153

155      * Is the sort order ascending or descending?
156      *
157      * Default value is true
158      *
159      * @return  true for ascending, false for descending
160      */
161     public boolean isAscending();
162 
163     /**
164      * Set whether the results should be sorted in ascending order (on the given sort column)
165      * or descending order.
166      *
167      * @param ascending     true to ascend, false to descend

169     public void setAscending(boolean ascending);

177     public int getContainerID();

195     public String getContainerIDField();

202      */
203     public void setContainerIDField(String containerIDField);

211      */
212     public String getJumpToField();
213

219      * param focusField     the name of the focus field
220      */
221     public void setJumpToField(String focusField);
222

226      *

228      */
229     public String getJumpToValue();
230 
231     /**

233      * will be the top result on the page of results
234      *
235      * @param focusValue    the value in the focus field on which to start browsing
236      */
237     public void setJumpToValue(String focusValue);

239     /**
240      * get the integer number which is the limit of the results that will be returned

246 
247     /**

256     public void setLimit(int limit);

272      * means do not offset.
