54  /**
55   * This class does most of the actual grunt work of preparing a browse
56   * result.  It takes in to a couple of available methods (depending on your
57   * desired browse type) a BrowserScope object, and uses this to produce a
58   * BrowseInfo object which is sufficient to describe to the User Interface
59   * the results of the requested browse
60   *
61   * @author Richard Jones
62   *
63   */

99      /**
100      * Perform a standard browse, which will return a BrowseInfo
101      * object that represents the results for the current page, the
102      * total number of results, the range, and information to construct
103      * previous and next links on any web page
104      *
105      * @param bs    the scope of the browse
106      * @return      the results of the browse
107      * @throws BrowseException
108      */
109     public BrowseInfo browse(BrowserScope bs)
110         throws BrowseException
111     {
112         log.debug(LogManager.getHeader(context, "browse", ""));
113 
114         // first, load the browse scope into the object
115         this.scope = bs;
116 
117         // since we use it so much, get the browse index out of the
118         // scope and store as a member
119         browseIndex = scope.getBrowseIndex();
120 
121         // now make the decision as to how to browse
122         if (browseIndex.isMetadataIndex() && !scope.isSecondLevel())
123         {
124             // this is a single value browse type that has not gone to
125             // the second level (i.e. authors, not items by a given author)
126             return browseByValue(scope);
127         }
128         else
129         {
130             // this is the full browse type or a browse that has gone to
131             // the second level
132             return browseByItem(scope);
133         }
134     }

136     /**
137      * Perform a limited browse, which only returns the results requested,
138      * without any extraneous information.  To perform a full browse, use
139      * BrowseEngine.browse() above.  This supports Item browse only, and does
140      * not currently support focus or values.  This method is used, for example,
141      * to generate the Recently Submitted Items results.
142      *
143      * @param bs    the scope of the browse
144      * @return      the results of the browse
145      */
146     public BrowseInfo browseMini(BrowserScope bs)
147         throws BrowseException
148     {
149         log.info(LogManager.getHeader(context, "browse_mini", ""));
150 
151         // load the scope into the object
152         this.scope = bs;
153 
154         // since we use it so much, get the browse index out of the
155         // scope and store as a member
156         browseIndex = scope.getBrowseIndex();
157 
158         // get the table name that we are going to be getting our data from
159         dao.setTable(browseIndex.getTableName());
160 
161         // tell the browse query whether we are ascending or descending on the value
162         dao.setAscending(scope.isAscending());
163 
164         // define a clause for the WHERE clause which will allow us to constraine
165         // our browse to a specified community or collection
166         if (scope.inCollection() || scope.inCommunity())
167         {
168             if (scope.inCollection())
169             {
170                 Collection col = (Collection) scope.getBrowseContainer();
171                 dao.setContainerTable("collection2item");
172                 dao.setContainerIDField("collection_id");
173                 dao.setContainerID(col.getID());
174             }
175             else if (scope.inCommunity())
176             {
177                 Community com = (Community) scope.getBrowseContainer();
178                 dao.setContainerTable("communities2item");
179                 dao.setContainerIDField("community_id");
180                 dao.setContainerID(com.getID());
181             }
182         }
183 
184         dao.setOffset(scope.getOffset());
185         dao.setLimit(scope.getResultsPerPage());
186 
187         // assemble the ORDER BY clause
188         String orderBy = browseIndex.getSortField(scope.isSecondLevel());
189         if (scope.getSortBy() > 0)
190         {
191             orderBy = "sort_" + Integer.toString(scope.getSortBy());
192         }
193         dao.setOrderField(orderBy);
194 
195         // now run the query
196         List results = dao.doQuery();
197 
198         // construct the mostly empty BrowseInfo object to pass back
199         BrowseInfo browseInfo = new BrowseInfo(results, 0, scope.getResultsPerPage(), 0);
200 
201         // add the browse index to the Browse Info
202         browseInfo.setBrowseIndex(browseIndex);
203 
204         // set the sort option for the Browse Info
205         browseInfo.setSortOption(scope.getSortOption());
206 
207         // tell the Browse Info which way we are sorting
208         browseInfo.setAscending(scope.isAscending());
209 
210         // tell the browse info what the container for the browse was
211         if (scope.inCollection() || scope.inCommunity())
212         {
213             browseInfo.setBrowseContainer(scope.getBrowseContainer());
214         }
215 
216         browseInfo.setResultsPerPage(scope.getResultsPerPage());
217 
218         browseInfo.setEtAl(scope.getEtAl());
219 
220         return browseInfo;
221     }

223     /**
224      * Browse the archive by the full item browse mechanism.  This produces a
225      * BrowseInfo object which contains full BrowseItem objects as its result
226      * set.
227      *
228      * @param bs        the scope of the browse
229      * @return          the results of the browse
230      * @throws BrowseException
231      */
232     private BrowseInfo browseByItem(BrowserScope bs)
233         throws BrowseException
234     {
235         log.info(LogManager.getHeader(context, "browse_by_item", ""));
236         try
237         {
238             // get the table name that we are going to be getting our data from
239             dao.setTable(browseIndex.getTableName());
240 
241             // tell the browse query whether we are ascending or descending on the value
242             dao.setAscending(scope.isAscending());
243 
244             // assemble the value clause
245             String rawValue = null;
246             if (scope.hasFilterValue() && scope.isSecondLevel())
247             {
248                 String value = scope.getFilterValue();
249                 rawValue = value;
250 
251                 // make sure the incoming value is normalised
252                 value = OrderFormat.makeSortString(value, scope.getFilterValueLang(),
253                             scope.getBrowseIndex().getDataType());
254 
255                 dao.setAuthorityValue(scope.getAuthorityValue());
256 
257                 // set the values in the Browse Query
258                 if (scope.isSecondLevel())
259                 {
260                     dao.setFilterValueField("value");
261                     dao.setFilterValue(rawValue);    
262                 }
263                 else
264                 {
265 	                dao.setFilterValueField("sort_value");
266 	                dao.setFilterValue(value);
267                 }
268                 dao.setFilterValuePartial(scope.getFilterValuePartial());
269 
270                 // to apply the filtering, we need the distinct and map tables for the index
271                 dao.setFilterMappingTables(browseIndex.getDistinctTableName(),
272                                            browseIndex.getMapTableName());
273             }
274 
275             // define a clause for the WHERE clause which will allow us to constraine
276             // our browse to a specified community or collection
277             if (scope.inCollection() || scope.inCommunity())
278             {
279                 if (scope.inCollection())
280                 {
281                     Collection col = (Collection) scope.getBrowseContainer();
282                     dao.setContainerTable("collection2item");
283                     dao.setContainerIDField("collection_id");
284                     dao.setContainerID(col.getID());
285                 }
286                 else if (scope.inCommunity())
287                 {
288                     Community com = (Community) scope.getBrowseContainer();
289                     dao.setContainerTable("communities2item");
290                     dao.setContainerIDField("community_id");
291                     dao.setContainerID(com.getID());
292                 }
293             }
294 
295             // this is the total number of results in answer to the query
296             int total = getTotalResults();
297 
298             // assemble the ORDER BY clause
299             String orderBy = browseIndex.getSortField(scope.isSecondLevel());
300             if (scope.getSortBy() > 0)
301             {
302                 orderBy = "sort_" + Integer.toString(scope.getSortBy());
303             }
304             dao.setOrderField(orderBy);
305 
306             int offset = scope.getOffset();
307             String rawFocusValue = null;
308             if (offset < 1 && (scope.hasJumpToItem() || scope.hasJumpToValue() || scope.hasStartsWith()))
309             {
310                 // We need to convert these to an offset for the actual browse query.
311                 // First, get a value that we can look up in the ordering field
312                 rawFocusValue = getJumpToValue();
313 
314                 // make sure the incoming value is normalised
315                 String focusValue = normalizeJumpToValue(rawFocusValue);
316 
317                 log.debug("browsing using focus: " + focusValue);
318 
319                 // Now we have a value to focus on, we need to find out where it is
320                 String focusField = browseIndex.getSortField(scope.isSecondLevel());
321                 if (scope.getSortBy() > 0)
322                 {
323                     focusField = "sort_" + Integer.toString(scope.getSortBy());
324                 }
325 
326                 // Convert the focus value into an offset
327                 offset = getOffsetForValue(focusValue);
328             }
329 
330             dao.setOffset(offset);
331 
332             // assemble the LIMIT clause
333             dao.setLimit(scope.getResultsPerPage());
334 
335             // Holder for the results
336             List results = null;
337 
338             // Does this browse have any contents?
339             if (total > 0)
340             {
341                 // now run the query
342                 results = dao.doQuery();
343 
344                 // now, if we don't have any results, we are at the end of the browse.  This will
345                 // be because a starts_with value has been supplied for which we don't have
346                 // any items.
347                 if (results.size() == 0)
348                 {
349                     // In this case, we will calculate a new offset for the last page of results
350                     offset = total - scope.getResultsPerPage();
351                     if (offset < 0)
352                         offset = 0;
353 
354                     // And rerun the query
355                     dao.setOffset(offset);
356                     results = dao.doQuery();
357                 }
358             }
359             else
360             {
361                 // No records, so make an empty list
362                 results = new ArrayList();
363             }
364 
365             // construct the BrowseInfo object to pass back
366 //            BrowseInfo browseInfo = new BrowseInfo(results, position, total, offset);
367             BrowseInfo browseInfo = new BrowseInfo(results, offset, total, offset);
368 
369             if (offset + scope.getResultsPerPage() < total)
370             {
371                 browseInfo.setNextOffset(offset + scope.getResultsPerPage());
372             }
373 
374             if (offset - scope.getResultsPerPage() > -1)
375             {
376                 browseInfo.setPrevOffset(offset - scope.getResultsPerPage());
377             }
378 
379             // add the browse index to the Browse Info
380             browseInfo.setBrowseIndex(browseIndex);
381 
382             // set the sort option for the Browse Info
383             browseInfo.setSortOption(scope.getSortOption());
384 
385             // tell the Browse Info which way we are sorting
386             browseInfo.setAscending(scope.isAscending());
387 
388             // tell the Browse Info which level of browse we are at
389             browseInfo.setBrowseLevel(scope.getBrowseLevel());
390 
391             // set the browse value if there is one
392             browseInfo.setValue(rawValue);
393 
394             // set the browse authority key if there is one
395             browseInfo.setAuthority(scope.getAuthorityValue());
396 
397             // set the focus value if there is one
398             browseInfo.setFocus(rawFocusValue);
399 
400             if (scope.hasJumpToItem())
401             {
402                 browseInfo.setFocusItem(scope.getJumpToItem());
403             }
404 
405             // tell the browse info if it is working from a starts with parameter
406             browseInfo.setStartsWith(scope.hasStartsWith());
407 
408             // tell the browse info what the container for the browse was
409             if (scope.inCollection() || scope.inCommunity())
410             {
411                 browseInfo.setBrowseContainer(scope.getBrowseContainer());
412             }
413 
414             browseInfo.setResultsPerPage(scope.getResultsPerPage());
415 
416             browseInfo.setEtAl(scope.getEtAl());
417 
418             return browseInfo;
419         }
420         catch (SQLException e)
421         {
422             log.error("caught exception: ", e);
423             throw new BrowseException(e);
424         }
425     }

427     /**
428      * Browse the archive by single values (such as the name of an author).  This
429      * produces a BrowseInfo object that contains Strings as the results of
430      * the browse
431      *
432      * @param bs        the scope of the browse
433      * @return          the results of the browse
434      * @throws BrowseException
435      */
436     private BrowseInfo browseByValue(BrowserScope bs)
437         throws BrowseException
438     {
439         log.info(LogManager.getHeader(context, "browse_by_value", "focus=" + bs.getJumpToValue()));
440 
441         try
442         {
443             // get the table name that we are going to be getting our data from
444             // this is the distinct table constrained to either community or collection
445             dao.setTable(browseIndex.getDistinctTableName());
446 
447             // remind the DAO that this is a distinct value browse, so it knows what sort
448             // of query to build
449             dao.setDistinct(true);
450 
451             // tell the browse query whether we are ascending or descending on the value
452             dao.setAscending(scope.isAscending());
453 
454             // set our constraints on community or collection
455             if (scope.inCollection() || scope.inCommunity())
456             {
457                 // Scoped browsing of distinct metadata requires the mapping
458                 // table to be specified.
459                 dao.setFilterMappingTables(null, browseIndex.getMapTableName());
460 
461                 if (scope.inCollection())
462                 {
463                     Collection col = (Collection) scope.getBrowseContainer();
464                     dao.setContainerTable("collection2item");
465                     dao.setContainerIDField("collection_id");
466                     dao.setContainerID(col.getID());
467                 }
468                 else if (scope.inCommunity())
469                 {
470                     Community com = (Community) scope.getBrowseContainer();
471                     dao.setContainerTable("communities2item");
472                     dao.setContainerIDField("community_id");
473                     dao.setContainerID(com.getID());
474                 }
475             }
476 
477             // this is the total number of results in answer to the query
478             int total = getTotalResults(true);
479 
480             // set the ordering field (there is only one option)
481             dao.setOrderField("sort_value");
482 
483             // assemble the focus clase if we are to have one
484             // it will look like one of the following
485             // - sort_value < myvalue
486             // = sort_1 > myvalue
487             dao.setJumpToField("sort_value");
488             int offset = scope.getOffset();
489             String rawFocusValue = null;
490             if (offset < 1 && scope.hasJumpToValue() || scope.hasStartsWith())
491             {
492                 String focusValue = getJumpToValue();
493 
494                 // store the value to tell the Browse Info object which value we are browsing on
495                 rawFocusValue = focusValue;
496 
497                 // make sure the incoming value is normalised
498                 focusValue = normalizeJumpToValue(focusValue);
499 
500                 offset = getOffsetForDistinctValue(focusValue);
501             }
502 
503 
504             // assemble the offset and limit
505             dao.setOffset(offset);
506             dao.setLimit(scope.getResultsPerPage());
507 
508             // Holder for the results
509             List results = null;
510 
511             // Does this browse have any contents?
512             if (total > 0)
513             {
514                 // now run the query
515                 results = dao.doValueQuery();
516 
517                 // now, if we don't have any results, we are at the end of the browse.  This will
518                 // be because a starts_with value has been supplied for which we don't have
519                 // any items.
520                 if (results.size() == 0)
521                 {
522                     // In this case, we will calculate a new offset for the last page of results
523                     offset = total - scope.getResultsPerPage();
524                     if (offset < 0)
525                         offset = 0;
526 
527                     // And rerun the query
528                     dao.setOffset(offset);
529                     results = dao.doValueQuery();
530                 }
531             }
532             else
533             {
534                 // No records, so make an empty list
535                 results = new ArrayList();
536             }
537 
538             // construct the BrowseInfo object to pass back
539             BrowseInfo browseInfo = new BrowseInfo(results, offset, total, offset);
540 
541             if (offset + scope.getResultsPerPage() < total)
542             {
543                 browseInfo.setNextOffset(offset + scope.getResultsPerPage());
544             }
545 
546             if (offset - scope.getResultsPerPage() > -1)
547             {
548                 browseInfo.setPrevOffset(offset - scope.getResultsPerPage());
549             }
550 
551             // add the browse index to the Browse Info
552             browseInfo.setBrowseIndex(browseIndex);
553 
554             // set the sort option for the Browse Info
555             browseInfo.setSortOption(scope.getSortOption());
556 
557             // tell the Browse Info which way we are sorting
558             browseInfo.setAscending(scope.isAscending());
559 
560             // tell the Browse Info which level of browse we are at
561             browseInfo.setBrowseLevel(scope.getBrowseLevel());
562 
563             // set the browse value if there is one
564             browseInfo.setFocus(rawFocusValue);
565 
566             // tell the browse info if it is working from a starts with parameter
567             browseInfo.setStartsWith(scope.hasStartsWith());
568 
569             // tell the browse info what the container for the browse was
570             if (scope.inCollection() || scope.inCommunity())
571             {
572                 browseInfo.setBrowseContainer(scope.getBrowseContainer());
573             }
574 
575             browseInfo.setResultsPerPage(scope.getResultsPerPage());
576 
577             return browseInfo;
578         }
579         catch (SQLException e)
580         {
581             log.error("caught exception: ", e);
582             throw new BrowseException(e);
583         }
584     }

586     /**
587      * Return the focus value.
588      *
589      * @return  the focus value to use
590      * @throws BrowseException
591      */
592     private String getJumpToValue()
593         throws BrowseException
594     {
595         log.debug(LogManager.getHeader(context, "get_focus_value", ""));
596 
597         // if the focus is by value, just return it
598         if (scope.hasJumpToValue())
599         {
600             log.debug(LogManager.getHeader(context, "get_focus_value_return", "return=" + scope.getJumpToValue()));
601             return scope.getJumpToValue();
602         }
603 
604         // if the focus is to start with, then we need to return the value of the starts with
605         if (scope.hasStartsWith())
606         {
607             log.debug(LogManager.getHeader(context, "get_focus_value_return", "return=" + scope.getStartsWith()));
608             return scope.getStartsWith();
609         }
610 
611         // since the focus is not by value, we need to obtain it
612 
613         // get the id of the item to focus on
614         int id = scope.getJumpToItem();
615 
616         // get the table name.  We don't really need to care about whether we are in a
617         // community or collection at this point.  This is only for full or second
618         // level browse, so there is no need to worry about distinct value browsing
619         String tableName = browseIndex.getTableName();
620 
621         // we need to make sure that we select from the correct column.  If the sort option
622         // is the 0th option then we use sort_value, but if it is one of the others we have
623         // to select from that column instead.  Otherwise, we end up missing the focus value
624         // to do comparisons in other columns.  The use of the focus value needs to be consistent
625         // across the browse
626         SortOption so = scope.getSortOption();
627         if (so == null || so.getNumber() == 0)
628         {
629             if (browseIndex.getSortOption() != null)
630                 so = browseIndex.getSortOption();
631         }
632 
633         String col = "sort_1";
634         if (so.getNumber() > 0)
635         {
636             col = "sort_" + Integer.toString(so.getNumber());
637         }
638 
639 
640         // now get the DAO to do the query for us, returning the highest
641         // string value in the given column in the given table for the
642         // item (I think)
643         String max = dao.doMaxQuery(col, tableName, id);
644 
645         log.debug(LogManager.getHeader(context, "get_focus_value_return", "return=" + max));
646 
647         return max;
648     }

707         // item (I think)

711     /**
712      * Return a normalized focus value. If there is no normalization that can be performed,

872      * if the caller wishes the actual results from the previous page can also be returned
873      * (this is useful, for example, if you are on the very last page of results, and need
874      * some previous results to pad out the full number of results per page).  If the

878      * @return          the database id of the top of the previous page
879      * @throws SQLException
880      * @throws BrowseException
881      */
882     private int getPreviousPageID(List callback)
883         throws SQLException, BrowseException
884     {
885         log.debug(LogManager.getHeader(context, "get_previous_page_id", ""));
886 
887         // do we want to capture the results?
888         boolean capture = false;
889         if (callback != null)
890         {
891             capture = true;
892         }
893 
894         // the only thing we need to do is reverse the direction
895         // of the query, and set it to not use the "=" part of the
896         // comparator (i.e. < and > not <= and >=).
897         boolean isAscending = dao.isAscending();
898         dao.setAscending(!isAscending);
899 
900         boolean useEquals = dao.useEqualsComparator();
901         dao.setEqualsComparator(false);
902 
903         // store in local scope the things that we are going to change
904         // during this method
905         int resultLimit = dao.getLimit();
906 
907         // if the database supports it (postgres does), we use limit
908         // and offset to minimise the work it has to do.
909 
910         // the limit will be the size of a page, or double the size of the
911         // page if we are capturing the result set (because the first page's worth
912         // will be the results, so the previous link will need to come from the
913         // page *before* that)
914         int limit = scope.getResultsPerPage();
915         if (capture)
916         {
917             limit *= 2;
918         }
919         dao.setLimit(limit);
920 
921         // now we have a query which is exactly the opposite of the
922         // original query.  So lets execute it:
923         List results = dao.doQuery();
924 
925         // before we continue, put back the variables we messed with
926         dao.setAscending(isAscending);
927         dao.setEqualsComparator(useEquals);
928         dao.setLimit(resultLimit);
929 
930         Iterator itr = results.iterator();
931 
932         // work our way through the list, capturing if necessary, and finally
933         // having the last result, which will be the top of the previous page
934         int i = 0;
935         BrowseItem prev = null;
936         while (itr.hasNext())
937         {
938             BrowseItem browseItem = (BrowseItem) itr.next();
939 
940             // we need to copy this, because of the scoping vs by-reference issue
941             prev = browseItem;
942 
943             // if we need to capture the results in the call back, do it here.
944             // Note that since the results will come in backwards, we place
945             // each one at the start of the array so that it displaces the others
946             // in the right direction.  By the end, they should be sorted correctly
947             // we use the index i to be sure that we only grab one page's worth
948             // of results (see note above about limit)
949             if (capture && i < scope.getResultsPerPage())
950             {
951                 callback.add(0, browseItem);
952                 i++;
953             }
954         }
955 
956         if (prev != null)
957         {
958             return prev.getID();
959         }
960         else
961         {
962             return -1;
963         }
