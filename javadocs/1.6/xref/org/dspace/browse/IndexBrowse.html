135      * Create a new IndexBrowse object.  This will ignore any authorisations
136      * applied to the Context
137      * 
138      * @param context
139      * @throws SQLException
140      * @throws BrowseException
141      */
142     public IndexBrowse(Context context)
143     	throws SQLException, BrowseException
144     {
145     	this.context = context;
146     	
147     	// get the browse indices, and ensure that
148     	// we have all the relevant tables prepped
149         this.bis = BrowseIndex.getBrowseIndices();
150         checkConfig();
151         
152         // get the DAO for the create operations
153         dao = BrowseDAOFactory.getCreateInstance(context);
154         
155         // set the outputter
156         output = new BrowseOutput();
157         
158         // then generate all the metadata bits that we
159         // are going to use
160         for (int k = 0; k < bis.length; k++)
161     	{
162     		bis[k].generateMdBits();
163     	}
164     }

304     /**
305      * Prune indexes - called from the public interfaces or at the end of a batch indexing process
306      */
307     private void pruneIndexes() throws BrowseException
308     {
309         // go over the indices and prune
310         for (int i = 0; i < bis.length; i++)
311         {
312             if (bis[i].isMetadataIndex())
313             {
314                 log.debug("Pruning metadata index: " + bis[i].getTableName());
315                 dao.pruneExcess(bis[i].getTableName(), bis[i].getMapTableName(), false);
316                 dao.pruneDistinct(bis[i].getDistinctTableName(), bis[i].getMapTableName());
317             }
318         }
319 
320         dao.pruneExcess(BrowseIndex.getItemBrowseIndex().getTableName(), null, false);
321         dao.pruneExcess(BrowseIndex.getWithdrawnBrowseIndex().getTableName(), null, true);
322     }

324     /**
325      * Index the given item
326      * 
327      * @param item	the item to index
328      * @throws BrowseException
329      */
330     public void indexItem(Item item)
331     	throws BrowseException
332     {
333         // If the item is not archived AND has not been withdrawn
334         // we can assume that it has *never* been archived - in that case,
335         // there won't be anything in the browse index, so we can just skip processing.
336         // If it is either archived or withdrawn, then there may be something in the browse
337         // tables, so we *must* process it.
338         // Caveat: an Item.update() that changes isArchived() from TRUE to FALSE, whilst leaving
339         // isWithdrawn() as FALSE, may result in stale data in the browse tables.
340         // Such an update should never occur though, and if it does, probably indicates a major
341         // problem with the code updating the Item.
342         if (item.isArchived() || item.isWithdrawn())
343         {
344             indexItem(new ItemMetadataProxy(item));
345 
346             // Ensure that we remove any invalid entries
347             pruneIndexes();
348         }
349     }

351        /**
352          * Index the given item
353          * 
354          * @param item  the item to index
355          * @throws BrowseException
356          */
357     private void indexItem(ItemMetadataProxy item)
358         throws BrowseException
359     {
360         // Map to store the metadata from the Item
361         // so that we don't grab it multiple times
362         Map<String, String> itemMDMap = new HashMap<String, String>();
363         
364         try
365         {
366             boolean reqCommunityMappings = false;
367             Map<Integer, String> sortMap = getSortValues(item, itemMDMap);
368             if (item.isArchived() && !item.isWithdrawn())
369             {
370                 // Try to update an existing record in the item index
371                 if (!dao.updateIndex(BrowseIndex.getItemBrowseIndex().getTableName(), item.getID(), sortMap))
372                 {
373                     // Record doesn't exist - ensure that it doesn't exist in the withdrawn index,
374                     // and add it to the archived item index
375                     removeIndex(item.getID(), BrowseIndex.getWithdrawnBrowseIndex().getTableName());
376                     dao.insertIndex(BrowseIndex.getItemBrowseIndex().getTableName(), item.getID(), sortMap);
377                 }
378 
379                 reqCommunityMappings = true;
380             }
381             else if (item.isWithdrawn())
382             {
383                 // Try to update an existing record in the withdrawn index
384                 if (!dao.updateIndex(BrowseIndex.getWithdrawnBrowseIndex().getTableName(), item.getID(), sortMap))
385                 {
386                     // Record doesn't exist - ensure that it doesn't exist in the item index,
387                     // and add it to the withdrawn item index
388                     removeIndex(item.getID(), BrowseIndex.getItemBrowseIndex().getTableName());
389                     dao.insertIndex(BrowseIndex.getWithdrawnBrowseIndex().getTableName(), item.getID(), sortMap);
390                 }
391             }
392             else
393             {
394                 // This item shouldn't exist in either index - ensure that it is removed
395                 removeIndex(item.getID(), BrowseIndex.getItemBrowseIndex().getTableName());
396                 removeIndex(item.getID(), BrowseIndex.getWithdrawnBrowseIndex().getTableName());
397             }
398 
399             // Update the community mappings if they are required, or remove them if they aren't
400             if (reqCommunityMappings)
401             {
402                 dao.updateCommunityMappings(item.getID());
403             }
404             else
405             {
406                 dao.deleteCommunityMappings(item.getID());
407             }
408 
409             // Now update the metadata indexes
410             for (int i = 0; i < bis.length; i++)
411             {
412                 log.debug("Indexing for item " + item.getID() + ", for index: " + bis[i].getTableName());
413                 
414                 if (bis[i].isMetadataIndex())
415                 {
416                     Set<Integer> distIDSet = new HashSet<Integer>();
417 
418                     // now index the new details - but only if it's archived and not withdrawn
419                     if (item.isArchived() && !item.isWithdrawn())
420                     {
421                         // get the metadata from the item
422                         for (int mdIdx = 0; mdIdx < bis[i].getMetadataCount(); mdIdx++)
423                         {
424                             String[] md = bis[i].getMdBits(mdIdx);
425                             DCValue[] values = item.getMetadata(md[0], md[1], md[2], Item.ANY);
426 
427                             // if we have values to index on, then do so
428                             if (values != null && values.length > 0)
429                             {
430                                 int minConfidence = MetadataAuthorityManager.getManager()
431                                         .getMinConfidence(values[0].schema, values[0].element, values[0].qualifier);
432 
433                                 for (int x = 0; x < values.length; x++)
434                                 {
435                                     // Ensure that there is a value to index before inserting it
436                                     if (StringUtils.isEmpty(values[x].value))
437                                     {
438                                         log.error("Null metadata value for item " + item.getID() + ", field: " +
439                                                 values[x].schema + "." +
440                                                 values[x].element +
441                                                 (values[x].qualifier == null ? "" : "." + values[x].qualifier));
442                                     }
443                                     else
444                                     {                                        
445                                         if (bis[i].isAuthorityIndex() && 
446                                                 (values[x].authority == null || values[x].confidence < minConfidence))
447                                         {
448                                             // if we have an authority index only authored metadata will go here!
449                                             log.debug("Skipping item="+item.getID()+", field="+values[x].schema+"."+values[x].element+"."+values[x].qualifier+", value="+values[x].value+", authority="+values[x].authority+", confidence="+values[x].confidence+" (BAD AUTHORITY)");
450                                             break;
451                                         }
452                                         
453                                         // is there any valid (with appropriate confidence) authority key?
454                                         if (values[x].authority != null
455                                                 && values[x].confidence >= minConfidence)
456                                         {
457                                             boolean isValueVariants = false;
458                                             List<String> variants = ChoiceAuthorityManager.getManager()
459                                                                         .getVariants(values[x].schema, values[x].element, values[x].qualifier,
460                                                                                         values[x].authority, values[x].language);
461                                             if (variants != null)
462                                             {
463                                                 for (String var : variants)
464                                                 {
465                                                     String nVal = OrderFormat.makeSortString(var, values[x].language, bis[i].getDataType());
466                                                     distIDSet.add(dao.getDistinctID(bis[i].getDistinctTableName(), var, values[x].authority, nVal));
467                                                     if (var.equals(values[x].value))
468                                                     {
469                                                         isValueVariants = true;
470                                                     }
471                                                 }
472                                             }
473 
474                                             if (!isValueVariants)
475                                             {
476                                                 // get the normalised version of the value
477                                                 String nVal = OrderFormat.makeSortString(values[x].value, values[x].language, bis[i].getDataType());
478                                                 distIDSet.add(dao.getDistinctID(bis[i].getDistinctTableName(), values[x].value, values[x].authority, nVal));
479                                             }
480                                         }
481                                         else // put it in the browse index as if it hasn't have an authority key
482                                         {
483                                         // get the normalised version of the value
484                                         String nVal = OrderFormat.makeSortString(values[x].value, values[x].language, bis[i].getDataType());
485                                             distIDSet.add(dao.getDistinctID(bis[i].getDistinctTableName(), values[x].value, null, nVal));
486                                     }
487                                 }
488                             }
489                         }
490                     }
491                     }
492 
493                     // Do we have any mappings?
494                     if (distIDSet.isEmpty())
495                     {
496                         // remove any old mappings
497                         removeIndex(item.getID(), bis[i].getMapTableName());
498                     }
499                     else
500                     {
501                         // Update the existing mappings
502                         int[] distIDarr = new int[distIDSet.size()];
503                         int didx = 0;
504                         for (Integer distID : distIDSet)
505                         {
506                             distIDarr[didx++] = distID;
507                         }
508                         dao.updateDistinctMappings(bis[i].getMapTableName(), item.getID(), distIDarr);
509                     }
510                 }
511             }
512         }
513         catch (SQLException e)
514         {
515             log.error("caught exception: ", e);
516             throw new BrowseException(e);
517         }
518     }

520     /**
521      * Get the normalised values for each of the sort columns
522      * @param item
523      * @param itemMDMap
524      * @return
525      * @throws BrowseException
526      * @throws SQLException
527      */
528     private Map<Integer, String> getSortValues(ItemMetadataProxy item, Map itemMDMap)
529             throws BrowseException, SQLException
530     {
531         try
532         {
533             // now obtain the sort order values that we will use
534             Map<Integer, String> sortMap = new HashMap<Integer, String>();
535             for (SortOption so : SortOption.getSortOptions())
536             {
537                 Integer key = new Integer(so.getNumber());
538                 String metadata = so.getMetadata();
539 
540                 // If we've already used the metadata for this Item
541                 // it will be cached in the map
542                 DCValue value = null;
543 
544                 if (itemMDMap != null)
545                     value = (DCValue) itemMDMap.get(metadata);
546 
547                 // We haven't used this metadata before, so grab it from the item
548                 if (value == null)
549                 {
550                     String[] somd = so.getMdBits();
551                     DCValue[] dcv = item.getMetadata(somd[0], somd[1], somd[2], Item.ANY);
552 
553                     if (dcv == null)
554                     {
555                         continue;
556                     }
557 
558                     // we only use the first dc value
559                     if (dcv.length > 0)
560                     {
561                         // Set it as the current metadata value to use
562                         // and add it to the map
563                         value = dcv[0];
564 
565                         if (itemMDMap != null)
566                             itemMDMap.put(metadata, dcv[0]);
567                     }
568                 }
569 
570                 // normalise the values as we insert into the sort map
571                 if (value != null && value.value != null)
572                 {
573                     String nValue = OrderFormat.makeSortString(value.value, value.language, so.getType());
574                     sortMap.put(key, nValue);
575                 }
576             }
577             
578             return sortMap;
579         }
580         catch (SortException se)
581         {
582             throw new BrowseException("Error in SortOptions", se);
583         }
584     }

625     public boolean itemRemoved(int itemID)
626             throws BrowseException
627 	{
628 		// go over the indices and index the item
629 		for (int i = 0; i < bis.length; i++)
630 		{
631 		    if (bis[i].isMetadataIndex())
632 		    {
633     			log.debug("Removing indexing for removed item " + itemID + ", for index: " + bis[i].getTableName());
634     			removeIndex(itemID, bis[i].getMapTableName());
635 		    }
636 	    }
637 
638         // Remove from the item indexes (archive and withdrawn)
639         removeIndex(itemID, BrowseIndex.getItemBrowseIndex().getTableName());
640         removeIndex(itemID, BrowseIndex.getWithdrawnBrowseIndex().getTableName());
641         dao.deleteCommunityMappings(itemID);
642 
643         // Ensure that we remove any invalid entries
644         pruneIndexes();
645 
646         return true;
647 	}

649 	/**
650 	 * Creates Browse indexes, destroying the old ones.
651 	 * 
652 	 * @param argv
653 	 *            Command-line arguments
654 	 */
655 	public static void main(String[] argv)
656 		throws SQLException, BrowseException, ParseException
657 	{
658         Context context = new Context();
659         context.turnOffAuthorisationSystem();
660         IndexBrowse indexer = new IndexBrowse(context);
661 	    
662 	    // create an options object and populate it
663 	    CommandLineParser parser = new PosixParser();
664 	    Options options = new Options();
665 	   
666 	    // these are mutually exclusive, and represent the primary actions
667 	    options.addOption("t", "tables", false, "create the tables only, do not attempt to index.  Mutually exclusive with -f and -i");
668 	    options.addOption("i", "index", false, "actually do the indexing.  Mutually exclusive with -t and -f");
669 	    options.addOption("f", "full", false, "make the tables, and do the indexing.  This forces -x.  Mutually exclusive with -t and -i");
670 	    
671 	    // these options can be specified only with the -f option
672 	    options.addOption("r", "rebuild", false, "should we rebuild all the indices, which removes old index tables and creates new ones.  For use with -f. Mutually exclusive with -d");
673 	    options.addOption("d", "delete", false, "delete all the indices, but don't create new ones.  For use with -f. This is mutually exclusive with -r");
674 	    
675 	    // these options can be specified only with the -t and -f options
676 	    options.addOption("o", "out", true, "[-o <filename>] write the remove and create SQL to the given file. For use with -t and -f");  // FIXME: not currently working
677 	    options.addOption("p", "print", false, "write the remove and create SQL to the stdout. For use with -t and -f");
678 	    options.addOption("x", "execute", false, "execute all the remove and create SQL against the database. For use with -t and -f");
679 	    options.addOption("s", "start", true, "[-s <int>] start from this index number and work upward (mostly only useful for debugging). For use with -t and -f");
680 	    
681 	    // this option can be used with any argument
682 	    options.addOption("v", "verbose", false, "print extra information to the stdout.  If used in conjunction with -p, you cannot use the stdout to generate your database structure");
683 	    
684 	    // display the help.  If this is spefified, it trumps all other arguments
685 	    options.addOption("h", "help", false, "show this help documentation.  Overrides all other arguments");
686 	    
687 	    CommandLine line = parser.parse(options, argv);
688 	    
689 	    // display the help
690 	    if (line.hasOption("h"))
691 	    {
692 	    	indexer.usage(options);
693 	    	return;
694 	    }
695 	    
696 	    if (line.hasOption("v"))
697 	    {
698 	    	indexer.setVerbose(true);
699 	    }
700 	    
701 	    if (line.hasOption("i"))
702 	    {
703 	    	indexer.createIndex();
704 	    	return;
705 	    }
706 	    
707 	    if (line.hasOption("f"))
708 	    {
709 	    	if (line.hasOption('r'))
710 		    {
711 		        indexer.setRebuild(true);
712 		    }
713 	    	else if (line.hasOption("d"))
714 		    {
715 		    	indexer.setDelete(true);
716 		    }
717 	    }
718
