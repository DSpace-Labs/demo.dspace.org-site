34  package org.dspace.browse;
35  
36  import java.io.IOException;
37  import java.util.HashMap;
38  import java.util.Map;
39  import java.util.StringTokenizer;
40  
41  import org.dspace.core.ConfigurationManager;
42  
43  /**
44   * Class to mediate with the item list configuration
45   * 
46   * @author Richard Jones
47   *
48   */
49  public class ItemListConfig
50  {
51  	/** a map of column number to metadata value */
52  	private Map metadata = new HashMap();
53  	
54  	/** a map of column number to data type */
55  	private Map types = new HashMap();
56  	
57  	/** constant for a DATE column */
58  	private static final int DATE = 1;
59  	
60  	/** constant for a TEXT column */
61  	private static final int TEXT = 2;
62  	
63  	/**
64  	 * Create a new instance of the Item list configuration.  This loads
65  	 * all the required information from configuration
66  	 * 
67  	 * @throws BrowseException
68  	 */
69  	public ItemListConfig()
70  		throws BrowseException
71  	{
72  		try
73  		{
74  			String configLine = ConfigurationManager.getProperty("webui.itemlist.columns");
75  			
76  			if (configLine == null || "".equals(configLine))
77  			{
78  				throw new BrowseException("There is no configuration for webui.itemlist.columns");
79  			}
80  			
81  			// parse the config
82  			StringTokenizer st = new StringTokenizer(configLine, ",");
83  			int i = 1;
84  			while (st.hasMoreTokens())
85  			{
86  				Integer key = new Integer(i);
87  				String token = st.nextToken();
88  				
89  				// find out if the field is a date
90  				if (token.indexOf("(date)") > 0)
91  				{
92  					token = token.replaceAll("\\(date\\)", "");
93  					types.put(key, new Integer(ItemListConfig.DATE));
94  				}
95  				else
96  				{
97  					types.put(key, new Integer(ItemListConfig.TEXT));
98  				}
99  				
100 				String[] mdBits = interpretField(token.trim(), null);
101 				metadata.put(key, mdBits);
102 				
103 				// don't forget to increment the key counter
104 				i++;
105 			}
106 		}
107 		catch (IOException e)
108 		{
109 			throw new BrowseException(e);
110 		}
111 	}
112 
113 	/**
114 	 * how many columns are there?
115 	 * 
116 	 * @return	the number of columns
117 	 */
118 	public int numCols()
119 	{
120 		return metadata.size();
121 	}
122 	
123 	/**
124 	 * what metadata is to go in the given column number
125 	 * 
126 	 * @param col
127 	 * @return
128 	 */
129 	public String[] getMetadata(int col)
130 	{
131 		return (String[]) metadata.get(new Integer(col));
132 	}
133 	
134 	/**
135      * Take a string representation of a metadata field, and return it as an array.
136      * This is just a convenient utility method to basically break the metadata 
137      * representation up by its delimiter (.), and stick it in an array, inserting
138      * the value of the init parameter when there is no metadata field part.
139      * 
140      * @param mfield	the string representation of the metadata
141      * @param init	the default value of the array elements
142      * @return	a three element array with schema, element and qualifier respectively
143      */
144     public String[] interpretField(String mfield, String init)
145     	throws IOException
146     {
147     	StringTokenizer sta = new StringTokenizer(mfield, ".");
148     	String[] field = {init, init, init};
149     	
150     	int i = 0;
151     	while (sta.hasMoreTokens())
152     	{
153     		field[i++] = sta.nextToken();
154     	}
155     	
156     	// error checks to make sure we have at least a schema and qualifier for both
157     	if (field[0] == null || field[1] == null)
158     	{
159     		throw new IOException("at least a schema and element be " +
160     				"specified in configuration.  You supplied: " + mfield);
161     	}
162     	
163     	return field;
164     }
