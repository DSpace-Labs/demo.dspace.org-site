420     /**
421      * Obtain the sort option
422      *
423      * @return  the sort option
424      * @throws BrowseException
425      */
426     public SortOption getSortOption()
427         throws BrowseException
428     {
429         try
430         {
431             // If a sortOption hasn't been set, work out the default
432             if (sortOption == null)
433             {
434                 // We need a browse index first though
435                 if (browseIndex != null)
436                 {
437                     // If a sorting hasn't been specified, and it's a metadata browse
438                     if (sortBy <= 0 && browseIndex.isMetadataIndex())
439                     {
440                         // Create a dummy sortOption for the metadata sort
441                         String dataType = browseIndex.getDataType();
442                         String type = ("date".equals(dataType) ? "date" : "text");
443                         sortOption = new SortOption(0, browseIndex.getName(), browseIndex.getMetadata(0), type);
444                     }
445                     else
446                     {
447                         // If a sorting hasn't been specified
448                         if (sortBy <= 0)
449                         {
450                             // Get the sort option from the index
451                             sortOption = browseIndex.getSortOption();
452 
453                             if (sortOption == null)
454                             {
455                                 // No sort option, so default to the first one defined in the config
456                                 for (SortOption so : SortOption.getSortOptions())
457                                 {
458                                     sortOption = so;
459                                     break;
460                                 }
461                             }
462                         }
463                         else
464                         {
465                             // A sorting has been specified, so get it from the configured sort columns
466                             for (SortOption so : SortOption.getSortOptions())
467                             {
468                                 if (so.getNumber() == sortBy)
469                                     sortOption = so;
470                             }
471                         }
472                     }
473                 }
474             }
475 
476             return sortOption;
477         }
478         catch (SortException se)
479         {
480             throw new BrowseException("Error in SortOptions", se);
481         }
482     }

578     /**
579      * @return  true if ascending, false if not - or not set
580      */
581     public boolean isAscending()
582     {
583         if (SortOption.ASCENDING.equalsIgnoreCase(order))
584         {
585             return true;
586         }
587 
588         if (SortOption.DESCENDING.equalsIgnoreCase(order))
589         {
590             return false;
591         }
592 
593         BrowseIndex bi = getBrowseIndex();
594 
595         if (bi != null && SortOption.DESCENDING.equalsIgnoreCase(bi.getDefaultOrder()))
596             return false;
597 
598         return true;
599     }
