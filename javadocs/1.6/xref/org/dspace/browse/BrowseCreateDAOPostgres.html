154             ArrayList<String> array = new ArrayList<String>();
155

157 
158             if (value)
159                 array.add("CREATE INDEX " + table + "_value_idx ON " + table + "(sort_value);");
160 
161             for (Integer i : sortCols)
162             {
163                 array.add("CREATE INDEX " + table + "_s" + i + "_idx ON " + table + "(sort_" + i + ");");
164             }
165             
166             if (execute)
167             {
168                 for (String query : array)
169                 {
170                     DatabaseManager.updateQuery(context, query);
171                 }
172             }
173             
174             String[] arr = new String[array.size()];
175             return array.toArray(arr);
176         }
177         catch (SQLException e)
178         {
179             log.error("caught exception: ", e);
180             throw new BrowseException(e);
181         }

242     /* (non-Javadoc)
243      * @see org.dspace.browse.BrowseCreateDAO#updateDistinctMapping(java.lang.String, int, int)
244      */
245     public boolean updateDistinctMappings(String table, int itemID, int[] distinctIDs) throws BrowseException
246     {
247         try
248         {
249             // Remove (set to -1) any duplicate distinctIDs
250             for (int i = 0; i < distinctIDs.length; i++)
251             {
252                 if (!isFirstOccurrence(distinctIDs, i))
253                     distinctIDs[i] = -1;
254             }
255 
256             // Find all existing mappings for this item
257             TableRowIterator tri = DatabaseManager.queryTable(context, table, "SELECT * FROM " + table + " WHERE item_id=?", itemID);
258             if (tri != null)
259             {
260                 try
261                 {
262                     while (tri.hasNext())
263                     {
264                         TableRow tr = tri.next();
265 
266                         // Check the item mappings to see if it contains this mapping
267                         boolean itemIsMapped = false;
268                         int trDistinctID = tr.getIntColumn("distinct_id");
269                         for (int i = 0; i < distinctIDs.length; i++)
270                         {
271                             // Found this mapping
272                             if (distinctIDs[i] == trDistinctID)
273                             {
274                                 // Flag it, and remove (-1) from the item mappings
275                                 itemIsMapped = true;
276                                 distinctIDs[i] = -1;
277                             }
278                         }
279 
280                         // The item is no longer mapped to this community, so remove the database record
281                         if (!itemIsMapped)
282                             DatabaseManager.delete(context, tr);
283                     }
284                 }
285                 finally
286                 {
287                     tri.close();
288                 }
289             }
290 
291             // Any remaining mappings need to be added to the database
292             for (int i = 0; i < distinctIDs.length; i++)
293             {
294                 if (distinctIDs[i] > -1)
295                 {
296                     TableRow row = DatabaseManager.create(context, table);
297                     row.setColumn("item_id", itemID);
298                     row.setColumn("distinct_id", distinctIDs[i]);
299                     DatabaseManager.update(context, row);
300                 }
301             }
302         }
303         catch (SQLException e)
304         {
305             log.error("caught exception: ", e);
306             String msg = "problem updating distinct mappings: table=" + table + ",item-id=" + itemID;
307             throw new BrowseException(msg, e);
308         }
309 
310         return false;
311     }

512     /* (non-Javadoc)
513      * @see org.dspace.browse.BrowseCreateDAO#getDistinctID(java.lang.String, java.lang.String, java.lang.String)
514      */
515     public int getDistinctID(String table, String value, String authority, String sortValue)
516         throws BrowseException
517     {
518         TableRowIterator tri = null;
519         
520         if (log.isDebugEnabled())
521         {
522             log.debug("getDistinctID: table=" + table + ",value=" + value + ",sortValue=" + sortValue);
523         }
524         
525         try
526         {
527             Object[] params;
528             String select;
529 			if (authority != null)
530 			{
531 				params = new Object[]{ value, authority };
532 			}
533 			else
534 			{
535 				params = new Object[]{ value };
536 			}
537 
538             if (ConfigurationManager.getBooleanProperty("webui.browse.metadata.case-insensitive", false))
539             {
540 				if (authority != null)
541 	            {	             
542 	                select = "SELECT distinct_id FROM " + table + " WHERE UPPER(value) = UPPER(?) and authority = ?";
543 	            }
544 	            else
545 	            {	                
546 	                select = "SELECT distinct_id FROM " + table + " WHERE UPPER(value) = UPPER(?) and authority IS NULL";
547 	            }
548             }
549             else
550             {
551                 select = "SELECT id FROM " + table + " WHERE value = ?";
552 				if (authority != null)
553 	            {	             
554 	                select = "SELECT distinct_id FROM " + table + " WHERE value = ? and authority = ?";
555 	            }
556 	            else
557 	            {	                
558 	                select = "SELECT distinct_id FROM " + table + " WHERE value = ? and authority IS NULL";
559 	            }
560             }
561 
562             tri = DatabaseManager.query(context, select, params);
563             int distinctID = -1;
564             if (!tri.hasNext())
565             {
566                 distinctID = insertDistinctRecord(table, value, authority, sortValue);
567             }
568             else
569             {
570                 distinctID = tri.next().getIntColumn("distinct_id");
571             }
572 
573             if (log.isDebugEnabled())
574             {
575                 log.debug("getDistinctID: return=" + distinctID);
576             }
577 
578             return distinctID;
579         }
580         catch (SQLException e)
581         {
582             log.error("caught exception: ", e);
583             throw new BrowseException(e);
584         }
585         finally
586         {
587             if (tri != null)
588             {
589                 tri.close();
590             }
591         }
592     }

620 
621                         // Check the item mappings to see if it contains this community mapping
622                         boolean itemIsMapped = false;
623                         int trCommID = tr.getIntColumn("community_id");
624                         for (int i = 0; i < commID.length; i++)
625                         {
626                             // Found this community
627                             if (commID[i] == trCommID)
628                             {
629                                 // Flag it, and remove (-1) from the item mappings

697      */
698     public void insertIndex(String table, int itemID, Map sortCols)
699         throws BrowseException
700     {
701         try
702         {
703             // create us a row in the index
704             TableRow row = DatabaseManager.create(context, table);
705             
706             // set the primary information for the index
707             row.setColumn("item_id", itemID);
708             
709             // now set the columns for the other sort values
710             Iterator itra = sortCols.keySet().iterator();
711             while (itra.hasNext())
712             {
713                 Integer key = (Integer) itra.next();
714                 String nValue = (String) sortCols.get(key);
715                 row.setColumn("sort_" + key.toString(), utils.truncateSortValue(nValue));
716             }
717             
718             DatabaseManager.update(context, row);
719         }
720         catch (SQLException e)
721         {
722             log.error("caught exception: ", e);
723             throw new BrowseException(e);
724         }

727     /* (non-Javadoc)
728      * @see org.dspace.browse.BrowseCreateDAO#updateIndex(java.lang.String, int, java.util.Map)
729      */
730     public boolean updateIndex(String table, int itemID, Map sortCols)
731             throws BrowseException
732     {
733         try
734         {
735             boolean rowUpdated = false;
736             TableRow row = DatabaseManager.findByUnique(context, table, "item_id", itemID);
737 
738             // If the item does not exist in the table, return that it couldn't be found
739             if (row == null)
740                 return false;
741 
742             // Iterate through all the sort values
