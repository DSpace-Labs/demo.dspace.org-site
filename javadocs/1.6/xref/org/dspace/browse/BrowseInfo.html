134     /**
135      * Constructor

139      * @param overallPosition

157     }

534     /**
535      * Return the results of the Browse as an Item array.
536      *
537      * @return The results of the Browse as an Item array.
538      */
539     public Item[] getItemResults(Context context)
540     	throws BrowseException
541     {
542     	try
543     	{
544     		BrowseItem[] bis = getBrowseItemResults();
545     		Item[] items = new Item[bis.length];
546     		for (int i = 0; i < bis.length; i++)
547     		{
548     			items[i] = Item.find(context, bis[i].getID());
549     		}
550     		return items;
551     	}
552     	catch (SQLException e)
553     	{
554     		throw new BrowseException(e);
555     	}
556     }

737 	/**
738 	 * Utility method for obtaining a string representation of the browse.  This is
739 	 * useful only for debug
740 	 */
741     public String toString()
742     {
743     	try
744     	{
745     		StringBuffer sb = new StringBuffer();
746 
747     		// calculate the range for display
748     		String from = Integer.toString(overallPosition + 1);
749     		String to = Integer.toString(overallPosition + results.size());
750     		String of = Integer.toString(total);
751 
752     		// report on the positional information of the browse
753     		sb.append("BrowseInfo String Representation: ");
754     		sb.append("Browsing " + from + " to " + to + " of " + of + " ");
755 
756     		// insert the information about which index
757     		sb.append("in index: " + browseIndex.getName() +
758     				" (data type: " + browseIndex.getDataType() +
759     				", display type: " + browseIndex.getDisplayType() + ") ");
760 
761     		sb.append("||");
762 
763     		// report on the browse scope container
764     		String container = "all of DSpace";
765     		DSpaceObject theContainer = null;
766     		if (inCollection())
767     		{
768     			container = "collection";
769     			theContainer = this.collection;
770     		}
771     		else if (inCommunity())
772     		{
773     			container = "community";
774     			theContainer = this.community;
775     		}
776 
777     		String containerID = "no id available/necessary";
778     		if (theContainer != null)
779     		{
780     			containerID = Integer.toString(theContainer.getID()) + " (" + theContainer.getHandle() + ")";
781     		}
782 
783     		sb.append("Browsing in " + container + ": " + containerID);
784     		sb.append("||");
785 
786     		// load the item list display configuration
787     		ItemListConfig config = new ItemListConfig();
788 
789     		// some information about the columns to be displayed
790     		if (browseIndex.isItemIndex())
791     		{
792     			sb.append("Listing over " + Integer.toString(config.numCols()) + " columns: ");
793     			for (int k = 1; k <= config.numCols(); k++)
794     			{
795     				if (k > 1)
796     				{
797     					sb.append(",");
798     				}
799     				String[] meta = config.getMetadata(k);
800     				sb.append(meta[0] + "." + meta[1] + "." + meta[2]);
801     			}
802 
803     			if (value != null)
804     			{
805     				sb.append(" on value: " + value);
806     			}
807 
808     			if (isStartsWith())
809     			{
810     				sb.append(" sort column starting with: " + focus);
811     			}
812     			else if (hasFocus())
813     			{
814     				sb.append(" sort column focus: " + focus);
815     			}
816     		}
817     		else if (browseIndex.isMetadataIndex())
818     		{
819     			sb.append("Listing single column: " + browseIndex.getMetadata());
820     			if (isStartsWith())
821     			{
822     				sb.append(" sort column starting with: " + focus);
823     			}
824     			else if (hasFocus())
825     			{
826     				sb.append(" sort column focus: " + focus);
827     			}
828     		}
829 
830     		sb.append("||");
831 
832     		// some information about how the data is sorted
833     		String direction = (ascending ? "ASC" : "DESC");
834     		sb.append("Sorting by: " + sortOption.getMetadata() + " " + direction +
835     				" (option " + Integer.toString(sortOption.getNumber()) + ")");
836     		sb.append("||");
837 
838     		// output the results
839     		if (browseIndex.isMetadataIndex() && !isSecondLevel())
840     		{
841     			sb.append(valueListingString());
842     		}
843     		else if (browseIndex.isItemIndex() || isSecondLevel())
844     		{
845     			sb.append(fullListingString(config));
846     		}
847 
848     		sb.append("||");
849 
850     		// tell us what the next and previous values are going to be
851     		sb.append("Top of next page: ");
852     		if (hasNextPage())
853     		{
854 				sb.append("offset: " + Integer.toString(this.nextOffset));
855     		}
856     		else
857     		{
858     			sb.append("n/a");
859     		}
860     		sb.append(";");
861 
862     		sb.append("Top of previous page: ");
863     		if (hasPrevPage())
864     		{
865 				sb.append("offset: " + Integer.toString(this.prevOffset));
866     		}
867     		else
868     		{
