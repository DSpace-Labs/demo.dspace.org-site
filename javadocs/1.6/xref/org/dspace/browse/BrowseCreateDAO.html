132     /**
133 	 * Get the browse index's internal id for the location of the given string
134 	 * and sort value in the given table.  This method should always return a 
135 	 * positive integer, as if no existing ID is available for the given value
136 	 * then one should be inserted using the data supplied, and the ID returned.
137 	 * 
138 	 * Generally this method is used in conjunction with createDistinctMapping thus:
139 	 * 
140 	 * <code>
141 	 * BrowseCreateDAO dao = BrowseDAOFactory.getCreateInstance();
142 	 * dao.createDistinctMapping("index_1_distinct_map", 21, 
143 	 *           dao.getDistinctID("index_1_distinct", "Human Readable", "human readable"));
144 	 * </code>
145 	 * 
146 	 * When it creates a distinct record, it would usually do so through insertDistinctRecord
147 	 * defined below.
148 	 * 
149 	 * @param table		the table in which to look for/create the id
150 	 * @param value		the value on which to search
151 	 * @param sortValue	the sort value to use in case of the need to create
152 	 * @return			the database id of the distinct record
153 	 * @throws BrowseException
154 	 */
155 	public int getDistinctID(String table, String value, String authority, String sortValue) throws BrowseException;

162 	 * <code>
163 	 * BrowseCreateDAO dao = BrowseDAOFactory.getCreateInstance();
