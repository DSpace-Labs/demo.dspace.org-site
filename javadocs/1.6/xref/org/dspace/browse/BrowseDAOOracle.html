157     /* (non-Javadoc)
158      * @see org.dspace.browse.BrowseDAO#doCountQuery()
159      */
160     public int doCountQuery() throws BrowseException
161     {
162         String   query  = getQuery();
163         Object[] params = getQueryParams();
164 
165         if (log.isDebugEnabled())
166         {
167             log.debug(LogManager.getHeader(context, "executing_count_query", "query=" + query));
168         }
169 
170         TableRowIterator tri = null;
171 
172         try
173         {
174             // now run the query
175             tri = DatabaseManager.query(context, query, params);
176 
177             if (tri.hasNext())
178             {
179                 TableRow row = tri.next();
180                 return (int) row.getLongColumn("num");
181             }
182             else
183             {
184                 return 0;
185             }
186         }
187         catch (SQLException e)
188         {
189             log.error("caught exception: ", e);
190             throw new BrowseException(e);
191         }
192         finally
193         {
194             if (tri != null)
195             {
196                 tri.close();
197             }
198         }
199     }

240     /* (non-Javadoc)
241      * @see org.dspace.browse.BrowseDAO#doOffsetQuery(java.lang.String, java.lang.String, java.lang.String)
242      */
243     public int doOffsetQuery(String column, String value, boolean isAscending)
244             throws BrowseException
245     {
246         TableRowIterator tri = null;
247 
248         if (column == null || value == null)
249             return 0;
250 
251         try
252         {
253             List paramsList = new ArrayList();
254             StringBuffer queryBuf = new StringBuffer();
255 
256             queryBuf.append("COUNT(").append(column).append(") AS offset ");
257 
258             buildSelectStatement(queryBuf, paramsList);
259             if (isAscending)
260             {
261                 queryBuf.append(" WHERE ").append(column).append("<?");
262                 paramsList.add(value);
263             }
264             else
265             {
266                 queryBuf.append(" WHERE ").append(column).append(">?");
267                 paramsList.add(value + Character.MAX_VALUE);
268             }
269 
270             if (containerTable != null || (value != null && valueField != null && tableDis != null && tableMap != null))
271             {
272                 queryBuf.append(" AND ").append("mappings.item_id=");
273                 queryBuf.append(table).append(".item_id");
274             }
275 
276             tri = DatabaseManager.query(context, queryBuf.toString(), paramsList.toArray());
277 
278             TableRow row;
279             if (tri.hasNext())
280             {
281                 row = tri.next();
282                 return row.getIntColumn("offset");
283             }
284             else
285             {
286                 return 0;
287             }
288         }
289         catch (SQLException e)
290         {
291             throw new BrowseException(e);
292         }
293         finally
294         {
295             if (tri != null)
296             {
297                 tri.close();
298             }
299         }
300     }

361     /* (non-Javadoc)
362      * @see org.dspace.browse.BrowseDAO#doQuery()
363      */
364     public List doQuery() throws BrowseException
365     {
366         String query = getQuery();
367         Object[] params = getQueryParams();
368 
369         if (log.isDebugEnabled())
370         {
371             log.debug(LogManager.getHeader(context, "executing_full_query", "query=" + query));
372         }
373 
374         TableRowIterator tri = null;
375         try
376         {
377             // now run the query
378             tri = DatabaseManager.query(context, query, params);
379 
380             // go over the query results and process
381             List results = new ArrayList();
382             while (tri.hasNext())
383             {
384                 TableRow row = tri.next();
385                 BrowseItem browseItem = new BrowseItem(context, row.getIntColumn("item_id"),
386                                                   itemsInArchive,
387                                                   itemsWithdrawn);
388                 results.add(browseItem);
389             }
390 
391             return results;
392         }
393         catch (SQLException e)
394         {
395             log.error("caught exception: ", e);
396             throw new BrowseException("problem with query: " + query, e);
397         }
398         finally
399         {
400             if (tri != null)
401             {
402                 tri.close();
403             }
404         }
405     }

758     /**
759      * Build the query that will be used for a distinct select.  This incorporates
760      * only the parts of the parameters that are actually useful for this type
761      * of browse
762      *
763      * @return      the query to be executed
764      * @throws BrowseException
765      */
766     private String buildDistinctQuery(List params) throws BrowseException
767     {
768         StringBuffer queryBuf = new StringBuffer();
769 
770         if (!buildSelectListCount(queryBuf))
771         {
772             if (!buildSelectListValues(queryBuf))
773             {
774                 throw new BrowseException("No arguments for SELECT statement");
775             }
776         }
777 
778         buildSelectStatementDistinct(queryBuf, params);
779         buildWhereClauseOpReset();
780 
781         // assemble the focus clase if we are to have one
782         // it will look like one of the following, for example
783         //     sort_value <= myvalue
784         //     sort_1 >= myvalue
785         buildWhereClauseJumpTo(queryBuf, params);
786 
787         // assemble the where clause out of the two possible value clauses
788         // and include container support
789         buildWhereClauseDistinctConstraints(queryBuf, params);
790 
791         // assemble the order by field
792         buildOrderBy(queryBuf);
793 
794         // prepare the limit and offset clauses
795         buildRowLimitAndOffset(queryBuf, params);
796 
797         return queryBuf.toString();
798     }

800     /**
801      * Build the query that will be used for a full browse.
802      *
803      * @return      the query to be executed
804      * @throws BrowseException
805      */
806     private String buildQuery(List params) throws BrowseException
807     {
808         StringBuffer queryBuf = new StringBuffer();
809 
810         if (!buildSelectListCount(queryBuf))
811         {
812             if (!buildSelectListValues(queryBuf))
813             {
814                 throw new BrowseException("No arguments for SELECT statement");
815             }
816         }
817 
818         buildSelectStatement(queryBuf, params);
819         buildWhereClauseOpReset();
820 
821         // assemble the focus clase if we are to have one
822         // it will look like one of the following, for example
