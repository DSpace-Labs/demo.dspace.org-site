58  /**
59   * AuthorizeManager handles all authorization checks for DSpace. For better
60   * security, DSpace assumes that you do not have the right to do something
61   * unless that permission is spelled out somewhere. That "somewhere" is the
62   * ResourcePolicy table. The AuthorizeManager is given a user, an object, and an
63   * action, and it then does a lookup in the ResourcePolicy table to see if there
64   * are any policies giving the user permission to do that action.
65   * <p>
66   * ResourcePolicies now apply to single objects (such as submit (ADD) permission
67   * to a collection.)
68   * <p>
69   * Note: If an eperson is a member of the administrator group (id 1), then they
70   * are automatically given permission for all requests another special group is
71   * group 0, which is anonymous - all EPeople are members of group 0.
72   */
73  public class AuthorizeManager
74  {
75      /**
76       * Utility method, checks that the current user of the given context can
77       * perform all of the specified actions on the given object. An
78       * <code>AuthorizeException</code> if all the authorizations fail.
79       * 
80       * @param c
81       *            context with the current user
82       * @param o
83       *            DSpace object user is attempting to perform action on
84       * @param actions
85       *            array of action IDs from
86       *            <code>org.dspace.core.Constants</code>
87       * @throws AuthorizeException
88       *             if any one of the specified actions cannot be performed by
89       *             the current user on the given object.
90       * @throws SQLException
91       *             if there's a database problem
92       */
93      public static void authorizeAnyOf(Context c, DSpaceObject o, int[] actions)
94              throws AuthorizeException, SQLException
95      {
96          AuthorizeException ex = null;
97  
98          for (int i = 0; i < actions.length; i++)
99          {
100             try
101             {
102                 authorizeAction(c, o, actions[i]);
103 
104                 return;
105             }
106             catch (AuthorizeException e)
107             {
108                 if (ex == null)
109                 {
110                     ex = e;
111                 }
112             }
113         }
114 
115         throw ex;
116     }

139     /**
140      * Checks that the context's current user can perform the given action on
141      * the given object. Throws an exception if the user is not authorized,
142      * otherwise the method call does nothing.
143      * 
144      * @param c
145      *            context
146      * @param o
147      *            a DSpaceObject
148      * @param useInheritance
149      *            flag to say if ADMIN action on the current object or parent
150      *            object can be used
151      * @param action
152      *            action to perform from <code>org.dspace.core.Constants</code>
153      * 
154      * @throws AuthorizeException
155      *             if the user is denied
156      */
157     public static void authorizeAction(Context c, DSpaceObject o, int action, boolean useInheritance)
158             throws AuthorizeException, SQLException
159     {
160         if (o == null)
161         {
162             // action can be -1 due to a null entry
163             String actionText;
164 
165             if (action == -1)
166             {
167                 actionText = "null";
168             }
169             else
170             {
171                 actionText = Constants.actionText[action];
172             }
173 
174             EPerson e = c.getCurrentUser();
175             int userid;
176 
177             if (e == null)
178             {
179                 userid = 0;
180             }
181             else
182             {
183                 userid = e.getID();
184             }
185 
186             throw new AuthorizeException(
187                     "Authorization attempted on null DSpace object "
188                             + actionText + " by user " + userid);
189         }
190 
191         if (!authorize(c, o, action, c.getCurrentUser(), useInheritance))
192         {
193             // denied, assemble and throw exception
194             int otype = o.getType();
195             int oid = o.getID();
196             int userid;
197             EPerson e = c.getCurrentUser();
198 
199             if (e == null)
200             {
201                 userid = 0;
202             }
203             else
204             {
205                 userid = e.getID();
206             }
207 
208             //            AuthorizeException j = new AuthorizeException("Denied");
209             //            j.printStackTrace();
210             // action can be -1 due to a null entry
211             String actionText;
212 
213             if (action == -1)
214             {
215                 actionText = "null";
216             }
217             else
218             {
219                 actionText = Constants.actionText[action];
220             }
221 
222             throw new AuthorizeException("Authorization denied for action "
223                     + actionText + " on " + Constants.typeText[otype] + ":"
224                     + oid + " by user " + userid, o, action);
225         }
226     }

252      *

254      *            DSpace context, containing current user
255      * @param o
256      *            DSpaceObject
257      * @param a
258      *            action being attempted, from
259      *            <code>org.dspace.core.Constants</code>
260      * @param useInheritance
261      *            flag to say if ADMIN action on the current object or parent

263      *

268             int a, boolean useInheritance) throws SQLException
269     {
270         boolean isAuthorized = true;
271 
272         if (o == null)
273         {
274             return false;
275         }
276 
277         try
278         {
279             authorizeAction(c, o, a, useInheritance);
280         }
281         catch (AuthorizeException e)
282         {
283             isAuthorized = false;
284         }
285 
286         return isAuthorized;

289     /**
290      * Check to see if the given user can perform the given action on the given
291      * object. Always returns true if the ignore authorization flat is set in
292      * the current context.
293      * 
294      * @param c
295      *            current context. User is irrelevant; "ignore authorization"
296      *            flag is relevant
297      * @param o
298      *            object action is being attempted on
299      * @param action
300      *            ID of action being attempted, from
301      *            <code>org.dspace.core.Constants</code>
302      * @param e
303      *            user attempting action
304      * @param useInheritance
305      *            flag to say if ADMIN action on the current object or parent
306      *            object can be used
307      * @return <code>true</code> if user is authorized to perform the given
308      *         action, <code>false</code> otherwise
309      * @throws SQLException
310      */
311     private static boolean authorize(Context c, DSpaceObject o, int action,
312             EPerson e, boolean useInheritance) throws SQLException
313     {
314         int userid;
315 
316         // return FALSE if there is no DSpaceObject
317         if (o == null)
318         {
319             return false;
320         }
321 
322         // is authorization disabled for this context?
323         if (c.ignoreAuthorization())
324         {
325             return true;
326         }
327 
328         // is eperson set? if not, userid = 0 (anonymous)
329         if (e == null)
330         {
331             userid = 0;
332         }
333         else
334         {
335             userid = e.getID();
336 
337             // perform isAdmin check to see
338             // if user is an Admin on this object
339             DSpaceObject testObject = useInheritance?o.getAdminObject(action):null;
340             
341             if (isAdmin(c, testObject))
342             {
343                 return true;
344             }
345         }
346 
347         for (ResourcePolicy rp : getPoliciesActionFilter(c, o, action))
348         {
349             // check policies for date validity
350             if (rp.isDateValid())
351             {
352                 if ((rp.getEPersonID() != -1) && (rp.getEPersonID() == userid))
353                 {
354                     return true; // match
355                 }
356 
357                 if ((rp.getGroupID() != -1)
358                         && (Group.isMember(c, rp.getGroupID())))
359                 {
360                     // group was set, and eperson is a member
361                     // of that group
362                     return true;
363                 }
364             }
365         }
366 
367         // default authorization is denial
368         return false;
369     }

375     /**
376      * Check to see if the current user is an Administrator of a given object
377      * within DSpace. Always return <code>true</code> if the user is a System
378      * Admin
379      * 
380      * @param c
381      *            current context
382      * @param o
383      *            current DSpace Object, if <code>null</code> the call will be
384      *            equivalent to a call to the <code>isAdmin(Context c)</code>
385      *            method
386      * 
387      * @return <code>true</code> if user has administrative privileges on the
388      *         given DSpace object
389      */
390     public static boolean isAdmin(Context c, DSpaceObject o) throws SQLException {
391 		if (isAdmin(c))
392 		{
393 			return true;
394 		}
395 		
396 		if (o == null)
397 		{
398 		    return false;
399 		}
400 
401         //
402         // First, check all Resource Policies directly on this object
403         //
404 		List<ResourcePolicy> policies = getPoliciesActionFilter(c, o, Constants.ADMIN);
405         int userid = c.getCurrentUser().getID();
406         
407         for (ResourcePolicy rp : policies)
408         {
409             // check policies for date validity
410             if (rp.isDateValid())
411             {
412                 if ((rp.getEPersonID() != -1) && (rp.getEPersonID() == userid))
413                 {
414                     return true; // match
415                 }
416 
417                 if ((rp.getGroupID() != -1)
418                         && (Group.isMember(c, rp.getGroupID())))
419                 {
420                     // group was set, and eperson is a member
421                     // of that group
422                     return true;
423                 }
424             }
425         }
426 
427         // If user doesn't have specific Admin permissions on this object,
428         // check the *parent* objects of this object.  This allows Admin
429         // permissions to be inherited automatically (e.g. Admin on Community
430         // is also an Admin of all Collections/Items in that Community)
431         DSpaceObject parent = o.getParentObject();
432         if (parent != null)
433         {
434             return isAdmin(c, parent);
435         }
436 	
437 		return false;
438 	}

441     /**
442      * Check to see if the current user is a System Admin. Always return
443      * <code>true</code> if c.ignoreAuthorization is set. Anonymous users
444      * can't be Admins (EPerson set to NULL)
445      * 
446      * @param c
447      *            current context
448      * 
449      * @return <code>true</code> if user is an admin or ignore authorization
450      *         flag set
451      */
452     public static boolean isAdmin(Context c) throws SQLException
453     {
454         // if we're ignoring authorization, user is member of admin
455         if (c.ignoreAuthorization())
456         {
457             return true;
458         }
459 
460         EPerson e = c.getCurrentUser();
461 
462         if (e == null)
463         {
464             return false; // anonymous users can't be admins....
465         }
466         else
467         {
468             return Group.isMember(c, 1);
469         }
470     }

531     /**
532      * Return a List of the policies for an object
533      * 
534      * @param c  current context
535      * @param o  object to retrieve policies for
536      * 
537      * @return List of <code>ResourcePolicy</code> objects
538      */
539     public static List<ResourcePolicy> getPolicies(Context c, DSpaceObject o)
540             throws SQLException
541     {
542     	TableRowIterator tri = DatabaseManager.queryTable(c, "resourcepolicy",
543                 "SELECT * FROM resourcepolicy WHERE resource_type_id= ? AND resource_id= ? ",
544                 o.getType(),o.getID());
545 
546         List<ResourcePolicy> policies = new ArrayList();
547 
548         try
549         {
550             while (tri.hasNext())
551             {
552                 TableRow row = tri.next();
553 
554                 // first check the cache (FIXME: is this right?)
555                 ResourcePolicy cachepolicy = (ResourcePolicy) c.fromCache(
556                         ResourcePolicy.class, row.getIntColumn("policy_id"));
557 
558                 if (cachepolicy != null)
559                 {
560                     policies.add(cachepolicy);
561                 }
562                 else
563                 {
564                     policies.add(new ResourcePolicy(c, row));
565                 }
566             }
567         }
568         finally
569         {
570             if (tri != null)
571                 tri.close();
572         }
573 
574         return policies;
575     }

674     /**
675      * Add policies to an object to match those from a previous object
676      * 
677      * @param c  context
678      * @param src
679      *            source of policies
680      * @param dest
681      *            destination of inherited policies
682      * @throws SQLException
683      *             if there's a database problem
684      * @throws AuthorizeException
685      *             if the current user is not authorized to add these policies
686      */
687     public static void inheritPolicies(Context c, DSpaceObject src,
688             DSpaceObject dest) throws SQLException, AuthorizeException
689     {
690         // find all policies for the source object
691         List<ResourcePolicy> policies = getPolicies(c, src);
692 
693         //Only inherit non-ADMIN policies (since ADMIN policies are automatically inherited)
694         List<ResourcePolicy> nonAdminPolicies = new ArrayList<ResourcePolicy>();
695         for (ResourcePolicy rp : policies)
696         {
697         	if (rp.getAction() != Constants.ADMIN)
698         	{
699         		nonAdminPolicies.add(rp);
700             }
701         }
702         addPolicies(c, nonAdminPolicies, dest);
703     }

705     /**
706      * Copies policies from a list of resource policies to a given DSpaceObject
707      * 
708      * @param c
709      *            DSpace context
710      * @param policies
711      *            List of ResourcePolicy objects
712      * @param dest
713      *            object to have policies added
714      * @throws SQLException
715      *             if there's a database problem
716      * @throws AuthorizeException
717      *             if the current user is not authorized to add these policies
718      */
719     public static void addPolicies(Context c, List<ResourcePolicy> policies, DSpaceObject dest)
720             throws SQLException, AuthorizeException
721     {
722         // now add them to the destination object
723         for (ResourcePolicy srp : policies)
724         {
725             ResourcePolicy drp = ResourcePolicy.create(c);
726 
727             // copy over values
728             drp.setResource(dest);
729             drp.setAction(srp.getAction());
730             drp.setEPerson(srp.getEPerson());
731             drp.setGroup(srp.getGroup());
732             drp.setStartDate(srp.getStartDate());
733             drp.setEndDate(srp.getEndDate());
734 
735             // and write out new policy
736             drp.update();
737         }
738     }
