39  package org.dspace.sword;
40  
41  import org.dspace.core.Context;
42  import org.dspace.core.PluginManager;
43  import org.dspace.content.DSpaceObject;
44  import org.dspace.content.Collection;
45  import org.dspace.content.Item;
46  
47  import org.purl.sword.base.Deposit;
48  import org.purl.sword.base.SWORDErrorException;
49  import org.purl.sword.base.ErrorCodes;
50  import org.apache.log4j.Logger;
51  
52  /**
53   * Factory class which will mint objects conforming to the
54   * SWORDIngester interface.
55   * 
56   * @author Richard Jones
57   *
58   */
59  public class SWORDIngesterFactory
60  {
61  	private static Logger log = Logger.getLogger(SWORDIngesterFactory.class);
62  
63  	/**
64  	 * Generate an object which conforms to the SWORDIngester interface.
65  	 * This Factory method may use the given DSpace context and the given
66  	 * SWORD Deposit request to decide on the most appropriate implementation
67  	 * of the interface to return.
68  	 * 
69  	 * To configure how this method will respond, configure the package ingester
70  	 * for the appropriate media types and defaults.  See the sword configuration
71  	 * documentation for more details.
72  	 * 
73  	 * @param context
74  	 * @param deposit
75  	 * @return
76  	 * @throws DSpaceSWORDException
77  	 */
78  	public static SWORDIngester getInstance(Context context, Deposit deposit, DSpaceObject dso)
79              throws DSpaceSWORDException, SWORDErrorException
80      {
81  		if (dso instanceof Collection)
82  		{
83  			SWORDIngester ingester = (SWORDIngester) PluginManager.getNamedPlugin(SWORDIngester.class, deposit.getPackaging());
84  			if (ingester == null)
85  			{
86  				throw new SWORDErrorException(ErrorCodes.ERROR_CONTENT, "No ingester configured for this package type");
87  			}
88  			return ingester;
89  		}
90  		else if (dso instanceof Item)
91  		{
92  			SWORDIngester ingester = (SWORDIngester) PluginManager.getNamedPlugin(SWORDIngester.class, "SimpleFileIngester");
93  			if (ingester == null)
94  			{
95  				throw new DSpaceSWORDException("SimpleFileIngester is not configured in plugin manager");
96  			}
97  			return ingester;
98  		}
99  
100 		throw new DSpaceSWORDException("No ingester could be found which works for this DSpace Object");
101 	}
102 }
