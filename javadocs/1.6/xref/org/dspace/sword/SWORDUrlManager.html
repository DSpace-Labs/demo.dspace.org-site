143 	 * Obtain the collection which is represented by the given
144 	 * URL
145 	 *
146 	 * @param context	the DSpace context
147 	 * @param location	the URL to resolve to a collection
148 	 * @return		The collection to which the url resolves
149 	 * @throws DSpaceSWORDException
150 	 */
151 	// FIXME: we need to generalise this to DSpaceObjects, so that we can support
152 	// Communities, Collections and Items separately
153 	public Collection getCollection(Context context, String location)
154 		throws DSpaceSWORDException, SWORDErrorException
155 	{
156 		try
157 		{
158 			String baseUrl = this.getBaseDepositUrl();
159 			if (baseUrl.length() == location.length())
160 			{
161 				throw new SWORDErrorException(DSpaceSWORDErrorCodes.BAD_URL, "The deposit URL is incomplete");
162 			}
163 			String handle = location.substring(baseUrl.length());
164 			if (handle.startsWith("/"))
165 			{
166 				handle = handle.substring(1);
167 			}
168 			if ("".equals(handle))
169 			{
170 				throw new SWORDErrorException(DSpaceSWORDErrorCodes.BAD_URL, "The deposit URL is incomplete");
171 			}
172 
173 			DSpaceObject dso = HandleManager.resolveToObject(context, handle);
174 
175 			if (!(dso instanceof Collection))
176 			{
177 				throw new SWORDErrorException(DSpaceSWORDErrorCodes.BAD_URL, "The deposit URL does not resolve to a valid collection");
178 			}
179 
180 			return (Collection) dso;
181 		}
182 		catch (SQLException e)
183 		{
184 			// log.error("Caught exception:", e);
185 			throw new DSpaceSWORDException("There was a problem resolving the collection", e);
186 		}
187 	}

266 	/**
267 	 * Extract a DSpaceObject from the given url.  If this method is unable to
268 	 * locate a meaningful and appropriate dspace object it will throw the
269 	 * appropriate sword error
270 	 * @param url
271 	 * @return
272 	 * @throws DSpaceSWORDException
273 	 * @throws SWORDErrorException
274 	 */
275 	public DSpaceObject extractDSpaceObject(String url)
276 			throws DSpaceSWORDException, SWORDErrorException
277 	{
278 		try
279 		{
280 			String sdBase = this.getBaseServiceDocumentUrl();
281 			String mlBase = this.getBaseMediaLinkUrl();
282 
283 			if (url.startsWith(sdBase))
284 			{
285 				// we are dealing with a service document request
286 				
287 				// first, let's find the beginning of the handle
288 				url = url.substring(sdBase.length());
289 				if (url.startsWith("/"))
290 				{
291 					url = url.substring(1);
292 				}
293 				if (url.endsWith("/"))
294 				{
295 					url = url.substring(0, url.length() - 1);
296 				}
297 
298 				DSpaceObject dso = HandleManager.resolveToObject(context, url);
299 				if (dso instanceof Collection || dso instanceof Community)
300 				{
301 					return dso;
302 				}
303 				else
304 				{
305 					throw new SWORDErrorException(DSpaceSWORDErrorCodes.BAD_URL,
306 							"Service Document request does not refer to a DSpace Collection or Community");
307 				}
308 			}
309 			else if (url.startsWith(mlBase))
310 			{
311 				// we are dealing with a bitstream media link
312 
313 				// find the index of the "/bitstream/" segment of the url
314 				int bsi = url.indexOf("/bitstream/");
315 
316 				// subtsring the url from the end of this "/bitstream/" string, to get the bitstream id
317 				String bsid = url.substring(bsi + 11);
318 
319 				// strip off extraneous slashes
320 				if (bsid.endsWith("/"))
321 				{
322 					bsid = bsid.substring(0, url.length() - 1);
323 				}
324 
325 				Bitstream bitstream = Bitstream.find(context, Integer.parseInt(bsid));
326 				return bitstream;
327 			}
328 			else
329 			{
330 				throw new SWORDErrorException(DSpaceSWORDErrorCodes.BAD_URL,
331 						"Unable to recognise URL as a valid service document: " + url);
332 			}
333 		}
334 		catch (SQLException e)
335 		{
336 			throw new DSpaceSWORDException(e);
337 		}
338 	}

340 	/**
341 	 * get the base url for service document requests
342 	 *
343 	 * @return
344 	 * @throws DSpaceSWORDException
345 	 */
346 	public String getBaseServiceDocumentUrl()
347 			throws DSpaceSWORDException
348 	{
349 		String depositUrl = ConfigurationManager.getProperty("sword.servicedocument.url");
350 		if (depositUrl == null || "".equals(depositUrl))
351 		{
352 			String dspaceUrl = ConfigurationManager.getProperty("dspace.url");
353 			if (dspaceUrl == null || "".equals(dspaceUrl))
354 			{
355 				throw new DSpaceSWORDException("Unable to construct service document urls, due to missing/invalid " +
356 						"config in sword.servicedocument.url and/or dspace.url");
357 			}
358 
359             try
360             {
361                 URL url = new URL(dspaceUrl);
362                 depositUrl = new URL(url.getProtocol(), url.getHost(), url.getPort(), "/sword/servicedocument").toString();
363             }
364             catch (MalformedURLException e)
365             {
366                 throw new DSpaceSWORDException("Unable to construct service document urls, due to invalid dspace.url " +
367 						e.getMessage(),e);
368             }
369 
370 
371 		}
372 		return depositUrl;
373 	}

446 	/**
447 	 * Central location for constructing usable urls for dspace bitstreams.  There
448 	 * is no place in the main DSpace code base for doing this.
449 	 *
450 	 * @param bitstream
451 	 * @return
452 	 * @throws DSpaceSWORDException
453 	 */
454 	public String getBitstreamUrl(Bitstream bitstream)
455 			throws DSpaceSWORDException
456 	{
457 		try
458 		{
459 			Bundle[] bundles = bitstream.getBundles();
460 			Bundle parent = null;
461 			if (bundles.length > 0)
462 			{
463 				parent = bundles[0];
464 			}
465 			else
466 			{
467 				throw new DSpaceSWORDException("Encountered orphaned bitstream");
468 			}
469 
470 			Item[] items = parent.getItems();
471 			Item item;
472 			if (items.length > 0)
473 			{
474 				item = items[0];
475 			}
476 			else
477 			{
478 				throw new DSpaceSWORDException("Encountered orphaned bundle");
479 			}
480 
481 			String handle = item.getHandle();
482 			String bsLink = ConfigurationManager.getProperty("dspace.url");
483 
484 			if (handle != null && !"".equals(handle))
485 			{
486 				bsLink = bsLink + "/bitstream/" + handle + "/" + bitstream.getSequenceID() + "/" + bitstream.getName();
487 			}
488 			else
489 			{
490 				bsLink = bsLink + "/retrieve/" + bitstream.getID();
491 			}
492 
493 			return bsLink;
494 		}
495 		catch (SQLException e)
496 		{
497 			throw new DSpaceSWORDException(e);
498 		}
499 	}

556 	 * get the media link url for the given bitstream
557 	 * 
558 	 * @param bitstream
559 	 * @return
560 	 * @throws DSpaceSWORDException
561 	 */
562 	public String getMediaLink(Bitstream bitstream)
563 			throws DSpaceSWORDException
564 	{
565 		try
566 		{
567 			Bundle[] bundles = bitstream.getBundles();
568 			Bundle parent = null;
569 			if (bundles.length > 0)
570 			{
571 				parent = bundles[0];
572 			}
573 			else
574 			{
575 				throw new DSpaceSWORDException("Encountered orphaned bitstream");
576 			}
577 
578 			Item[] items = parent.getItems();
579 			Item item;
580 			if (items.length > 0)
581 			{
582 				item = items[0];
583 			}
584 			else
585 			{
586 				throw new DSpaceSWORDException("Encountered orphaned bundle");
587 			}
588 
589 			String itemUrl = this.getMediaLink(item);
590 			if (itemUrl.equals(this.getBaseMediaLinkUrl()))
591 			{
592 				return itemUrl;
593 			}
594 
595 			String bsUrl = itemUrl + "/bitstream/" + bitstream.getID();
596 
597 			return bsUrl;
598 		}
599 		catch (SQLException e)
600 		{
601 			throw new DSpaceSWORDException(e);
