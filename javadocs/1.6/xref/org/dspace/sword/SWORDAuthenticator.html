72  	/**
73  	 * Does the given username and password authenticate for the
74  	 * given DSpace Context?
75  	 * 
76  	 * @param context
77  	 * @param un
78  	 * @param pw
79  	 * @return	true if yes, false if not
80  	 */
81  	public boolean authenticates(Context context, String un, String pw)
82  	{
83  		int auth = AuthenticationManager.authenticate(context, un, pw, null, null);
84  		if (auth == AuthenticationMethod.SUCCESS)
85  		{
86  			return true;
87  		}
88  		return false;
89  	}

216 	/**
217 	 * Authenticate the given username/password pair, in conjunction with
218 	 * the onBehalfOf user.  The rules are that the username/password pair
219 	 * must successfully authenticate the user, and the onBehalfOf user
220 	 * must exist in the user database.
221 	 *
222 	 * @param un
223 	 * @param pw
224 	 * @param obo
225 	 * @return	a SWORD context holding the various user information
226 	 * @throws SWORDAuthenticationException
227 	 * @throws SWORDException
228 	 */
229 	private SWORDContext authenticate(Context context, String un, String pw, String obo, String ip)
230 		throws SWORDAuthenticationException, SWORDException, SWORDErrorException
231 	{
232 		// smooth out the OnBehalfOf request, so that empty strings are
233 		// treated as null
234 		if ("".equals(obo))
235 		{
236 			obo = null;
237 		}
238 
239 		// first find out if we support on-behalf-of deposit
240 		boolean mediated = ConfigurationManager.getBooleanProperty("sword.on-behalf-of.enable");
241 		if (!mediated && obo != null)
242 		{
243 			// user is trying to do a mediated deposit on a repository which does not support it
244 			log.error("Attempted mediated deposit on service not configured to do so");
245 			throw new SWORDErrorException(ErrorCodes.MEDIATION_NOT_ALLOWED, "Mediated deposit to this service is not permitted");
246 		}
247 
248 		log.info(LogManager.getHeader(context, "sword_authenticate", "username=" + un + ",on_behalf_of=" + obo));
249 		
250 		try
251 		{
252 			// attempt to authenticate the primary user
253 			SWORDContext sc = new SWORDContext();
254 			EPerson ep = null;
255 			boolean authenticated = false;
256 			if (this.authenticates(context, un, pw))
257 			{
258 				// if authenticated, obtain the eperson object
259 				ep = context.getCurrentUser();
260 
261 				if (ep != null)
262 				{
263 					authenticated = true;
264 					sc.setAuthenticated(ep);
265 					sc.setAuthenticatorContext(context);
266 					sc.setContext(context);
267 				}
268 
269 				// if there is an onBehalfOfuser, then find their eperson
270 				// record, and if it exists set it.  If not, then the
271 				// authentication process fails
272 				EPerson epObo = null;
273 				if (obo != null)
274 				{
275 					epObo = EPerson.findByEmail(context, obo);
276 					if (epObo == null)
277 					{
278 						epObo = EPerson.findByNetid(context, obo);
279 					}
280 
281 					if (epObo != null)
282 					{
283 						sc.setOnBehalfOf(epObo);
284 						Context oboContext = this.constructContext(ip);
285 						oboContext.setCurrentUser(epObo);
286 						sc.setContext(oboContext);
287 					}
288 					else
289 					{
290 						authenticated = false;
291 						throw new SWORDErrorException(ErrorCodes.TARGET_OWNER_UKNOWN, "unable to identify on-behalf-of user: " + obo);
292 					}
293 				}
294 			}
295 
296 			if (!authenticated)
297 			{
298 				// decide what kind of error to throw
299 				if (ep != null)
300 				{
301 					log.info(LogManager.getHeader(context, "sword_unable_to_set_user", "username=" + un));
302 					throw new SWORDAuthenticationException("Unable to authenticate the supplied used");
303 				}
304 				else
305 				{
306 					// FIXME: this shouldn't ever happen now, but may as well leave it in just in case
307 					// there's a bug elsewhere
308 					log.info(LogManager.getHeader(context, "sword_unable_to_set_on_behalf_of", "username=" + un + ",on_behalf_of=" + obo));
309 					throw new SWORDAuthenticationException("Unable to authenticate the onBehalfOf account");
310 				}
311 			}
312 			
313 			return sc;
314 		}
315 		catch (SQLException e)
316 		{
317 			log.error("caught exception: ", e);
318 			throw new SWORDException("There was a problem accessing the repository user database", e);
319 		}
320 		catch (AuthorizeException e)
321 		{
322 			log.error("caught exception: ", e);
323 			throw new SWORDAuthenticationException("There was a problem authenticating or authorising the user", e);
324 		}
325 	}

328 	 * Can the users contained in this object's member SWORDContext
329 	 * make a successful submission to the selected collection.
330 	 *
331 	 * See javadocs for individual canSubmitTo methods to see the conditions
332 	 * which are applied in each situation
333 	 *
334 	 * @return	true if yes, false if not
335 	 * @throws DSpaceSWORDException
336 	 */
337 	public boolean canSubmit(SWORDService swordService, Deposit deposit, DSpaceObject dso)
338 		throws DSpaceSWORDException, SWORDErrorException
339 	{
340 		// get the things we need out of the service
341 		SWORDContext swordContext = swordService.getSwordContext();
342 
343 		// determine if we can submit
344 		boolean submit = this.canSubmitTo(swordContext, dso);
345 
346 		if (submit)
347 		{
348 			swordService.message("User is authorised to submit to collection");
349 		}
350 		else
351 		{
352 			swordService.message("User is not authorised to submit to collection");
353 		}
354 
355 		return submit;
356 	}

450 	/**
451 	 * Is the given eperson in the given group, or any of the groups
452 	 * that are also members of that group.  This method recurses
453 	 * until it has exhausted the tree of groups or finds the given
454 	 * eperson
455 	 *
456 	 * @param group
457 	 * @param eperson
458 	 * @return	true if in group, false if not
459 	 */
460 	public boolean isInGroup(Group group, EPerson eperson)
461 	{
462 		EPerson[] eps = group.getMembers();
463 		Group[] groups = group.getMemberGroups();
464 
465 		// is the user in the current group
466 		for (int i = 0; i < eps.length; i++)
467 		{
468 			if (eperson.getID() == eps[i].getID())
469 			{
470 				return true;
471 			}
472 		}
473 
474 		// is the eperson in the sub-groups (recurse)
475 		if (groups != null && groups.length > 0)
476 		{
477 			for (int j = 0; j < groups.length; j++)
478 			{
479 				if (isInGroup(groups[j], eperson))
480 				{
481 					return true;
482 				}
483 			}
484 		}
485 
486 		// ok, we didn't find you
487 		return false;
488 	}

490 	/**
491 	 * Get an array of all the communities that the current SWORD
492 	 * context will allow deposit onto in the given DSpace context
493 	 *
494 	 * The user may submit to a community if the following conditions
495 	 * are met:
496 	 *
497 	 * IF: the authenticated user is an administrator
498 	 *   AND:
499 	 *      (the on-behalf-of user is an administrator
500 	 *	 	OR the on-behalf-of user is authorised to READ
501 	 *	 	OR the on-behalf-of user is null)
502 	 * OR IF: the authenticated user is authorised to READ
503 	 *   AND:
504 	 *	     (the on-behalf-of user is an administrator
505 	 *		  OR the on-behalf-of user is authorised to READ
506 	 *		  OR the on-behalf-of user is null)
507 	 *
508 	 * @param swordContext
509 	 * @return	the array of allowed collections
510 	 * @throws DSpaceSWORDException
511 	 */
512 	public List<Community> getAllowedCommunities(SWORDContext swordContext)
513 		throws DSpaceSWORDException
514 	{
515 		// a community is allowed if the following conditions are met
516 		//
517 		// - the authenticated user is an administrator
518 		// -- the on-behalf-of user is an administrator
519 		// -- the on-behalf-of user is authorised to READ
520 		// -- the on-behalf-of user is null
521 		// - the authenticated user is authorised to READ
522 		// -- the on-behalf-of user is an administrator
523 		// -- the on-behalf-of user is authorised to READ
524 		// -- the on-behalf-of user is null
525 		try
526 		{
527 			// locate all the top level communities
528 			Context context = swordContext.getContext();
529 			List<Community> allowed = new ArrayList<Community>();
530 			Community[] comms = Community.findAllTop(context);
531 			for (int i = 0; i < comms.length; i++)
532 			{
533 				boolean authAllowed = false;
534 				boolean oboAllowed = false;
535 
536 				// check for obo null
537 				if (swordContext.getOnBehalfOf() == null)
538 				{
539 					oboAllowed = true;
540 				}
541 
542 				// look up the READ policy on the community.  This will include determining if the user is an administrator
543 				// so we do not need to check that separately
544 				if (!authAllowed)
545 				{
546 					authAllowed = AuthorizeManager.authorizeActionBoolean(swordContext.getAuthenticatorContext(), comms[i], Constants.READ);
547 				}
548 
549 				// if we have not already determined that the obo user is ok to submit, look up the READ policy on the
550 				// community.  THis will include determining if the user is an administrator.
551 				if (!oboAllowed)
552 				{
553 					oboAllowed = AuthorizeManager.authorizeActionBoolean(swordContext.getOnBehalfOfContext(), comms[i], Constants.READ);
554 				}
555 
556 				// final check to see if we are allowed to READ
557 				if (authAllowed && oboAllowed)
558 				{
559 					allowed.add(comms[i]);
560 				}
561 			}
562 			return allowed;
563 		}
564 		catch (SQLException e)
565 		{
566 			log.error("Caught exception: ", e);
567 			throw new DSpaceSWORDException(e);
568 		}
569 	}
