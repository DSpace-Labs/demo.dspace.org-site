57  /**
58   * This class is responsible for initiating the process of 
59   * deposit of SWORD Deposit objects into the DSpace repository
60   * 
61   * @author Richard Jones
62   *
63   */
64  public class DepositManager
65  {
66  	/** Log4j logger */
67  	public static Logger log = Logger.getLogger(DepositManager.class);
68  
69  	/** The SWORD service implementation */
70  	private SWORDService swordService;
71  
72  	/**
73  	 * Construct a new DepositManager using the given instantiation of
74  	 * the SWORD service implementation
75  	 *
76  	 * @param service
77  	 */
78  	public DepositManager(SWORDService service)
79  	{
80  		this.swordService = service;
81  		log.debug("Created instance of DepositManager");
82  	}
83  
84  	public DSpaceObject getDepositTarget(Deposit deposit)
85  			throws DSpaceSWORDException, SWORDErrorException
86  	{
87  		SWORDUrlManager urlManager = swordService.getUrlManager();
88  		Context context = swordService.getContext();
89  
90  		// get the target collection
91  		String loc = deposit.getLocation();
92  		DSpaceObject dso = urlManager.getDSpaceObject(context, loc);
93  
94  		swordService.message("Performing deposit using location: " + loc);
95  
96  		if (dso instanceof Collection)
97  		{
98  			swordService.message("Location resolves to collection with handle: " + dso.getHandle() +
99  				" and name: " + ((Collection) dso).getMetadata("name"));
100 		}
101 		else if (dso instanceof Item)
102 		{
103 			swordService.message("Location resolves to item with handle: " + dso.getHandle());
104 		}
105 
106 		return dso;
107 	}
108 
109 	/**
110 	 * Once this object is fully prepared, this method will execute
111 	 * the deposit process.  The returned DepositRequest can be
112 	 * used then to assembel the SWORD response.
113 	 * 
114 	 * @return	the response to the deposit request
115 	 * @throws DSpaceSWORDException
116 	 */
117 	public DepositResponse deposit(Deposit deposit)
118 		throws DSpaceSWORDException, SWORDErrorException, SWORDAuthenticationException
119 	{
120 		// start the timer, and initialise the verboseness of the request
121 		Date start = new Date();
122 		swordService.message("Initialising verbose deposit");
123 
124 		// get the things out of the service that we need
125 		SWORDContext swordContext = swordService.getSwordContext();
126 		Context context = swordService.getContext();
127 
128 		// get the deposit target
129 		DSpaceObject dso = this.getDepositTarget(deposit);
130 
131 		// find out if the supplied SWORDContext can submit to the given
132 		// dspace object
133 		SWORDAuthenticator auth = new SWORDAuthenticator();
134 		if (!auth.canSubmit(swordService, deposit, dso))
135 		{
136 			// throw an exception if the deposit can't be made
137 			String oboEmail = "none";
138 			if (swordContext.getOnBehalfOf() != null)
139 			{
140 				oboEmail = swordContext.getOnBehalfOf().getEmail();
141 			}
142 			log.info(LogManager.getHeader(context, "deposit_failed_authorisation", "user=" +
143 					swordContext.getAuthenticated().getEmail() + ",on_behalf_of=" + oboEmail));
144 			throw new SWORDAuthenticationException("Cannot submit to the given collection with this context");
145 		}
146 
147 		// make a note of the authentication in the verbose string
148 		swordService.message("Authenticated user: " + swordContext.getAuthenticated().getEmail());
149 		if (swordContext.getOnBehalfOf() != null)
150 		{
151 			swordService.message("Depositing on behalf of: " + swordContext.getOnBehalfOf().getEmail());
152 		}
153 
154 		// determine which deposit engine we initialise
155 		Depositor dep = null;
156 		if (dso instanceof Collection)
157 		{
158 			swordService.message("Initialising depositor for an Item in a Collection");
159 			dep = new CollectionDepositor(swordService, dso);
160 		}
161 		else if (dso instanceof Item)
162 		{
163 			swordService.message("Initialising depositor for a Bitstream in an Item");
164 			dep = new ItemDepositor(swordService, dso);
165 		}
166 
167 		if (dep == null)
168 		{
169 			log.error("The specified deposit target does not exist, or is not a collection or an item");
170 			throw new DSpaceSWORDException("Deposit target is not a collection or an item");
171 		}
172 
173 		DepositResult result = dep.doDeposit(deposit);
174 
175 		// now construct the deposit response.  The response will be
176 		// CREATED if the deposit is in the archive, or ACCEPTED if
177 		// the deposit is in the workflow.  We use a separate record
178 		// for the handle because DSpace will not supply the Item with
179 		// a record of the handle straight away.
180 		String handle = result.getHandle();
181 		int state = Deposit.CREATED;
182 		if (handle == null || "".equals(handle))
183 		{
184 			state = Deposit.ACCEPTED;
185 		}
186 		
187 		DepositResponse response = new DepositResponse(state);
188 		response.setLocation(result.getMediaLink());
189 
190 		DSpaceATOMEntry dsatom = null;
191 		if (result.getItem() != null)
192 		{
193 			swordService.message("Initialising ATOM entry generator for an Item");
194 			dsatom = new ItemEntryGenerator(swordService);
195 		}
196 		else if (result.getBitstream() != null)
197 		{
198 			swordService.message("Initialising ATOM entry generator for a Bitstream");
199 			dsatom = new BitstreamEntryGenerator(swordService);
200 		}
201 		if (dsatom == null)
202 		{
203 			log.error("The deposit failed, see exceptions for explanation");
204 			throw new DSpaceSWORDException("Result of deposit did not yield an Item or a Bitstream");
205 		}
206 		SWORDEntry entry = dsatom.getSWORDEntry(result, deposit);
207 
208 		// if this was a no-op, we need to remove the files we just
209 		// deposited, and abort the transaction
210 		if (deposit.isNoOp())
211 		{
212 			dep.undoDeposit(result);
213 			swordService.message("NoOp Requested: Removed all traces of submission");
214 		}
215 		
216 		entry.setNoOp(deposit.isNoOp());
217 
218 		Date finish = new Date();
219 		long delta = finish.getTime() - start.getTime();
220 		swordService.message("Total time for deposit processing: " + delta + " ms");
221 		entry.setVerboseDescription(swordService.getVerboseDescription().toString());
222 
223 		response.setEntry(entry);
224 		
225 		return response;
226 	}
