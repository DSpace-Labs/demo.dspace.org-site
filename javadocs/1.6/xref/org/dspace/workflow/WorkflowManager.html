73  /**
74   * Workflow state machine
75   * 
76   * Notes:
77   * 
78   * Determining item status from the database:
79   * 
80   * When an item has not been submitted yet, it is in the user's personal
81   * workspace (there is a row in PersonalWorkspace pointing to it.)
82   * 
83   * When an item is submitted and is somewhere in a workflow, it has a row in the
84   * WorkflowItem table pointing to it. The state of the workflow can be
85   * determined by looking at WorkflowItem.getState()
86   * 
87   * When a submission is complete, the WorkflowItem pointing to the item is
88   * destroyed and the archive() method is called, which hooks the item up to the
89   * archive.
90   * 
91   * Notification: When an item enters a state that requires notification,
92   * (WFSTATE_STEP1POOL, WFSTATE_STEP2POOL, WFSTATE_STEP3POOL,) the workflow needs
93   * to notify the appropriate groups that they have a pending task to claim.
94   * 
95   * Revealing lists of approvers, editors, and reviewers. A method could be added
96   * to do this, but it isn't strictly necessary. (say public List
97   * getStateEPeople( WorkflowItem wi, int state ) could return people affected by
98   * the item's current state.
99   */
100 public class WorkflowManager

144     /**
145      * Translate symbolic name of workflow state into number.
146      * The name is case-insensitive.  Returns -1 when name cannot
147      * be matched.
148      * @param state symbolic name of workflow state, must be one of
149      *        the elements of workflowText array.
150      * @return numeric workflow state or -1 for error.
151      */
152     public static int getWorkflowID(String state)
153     {
154         for (int i = 0; i < workflowText.length; ++i)
155             if (state.equalsIgnoreCase(workflowText[i]))
156                 return i;
157         return -1;
158     }

160     /**
161      * startWorkflow() begins a workflow - in a single transaction do away with
162      * the PersonalWorkspace entry and turn it into a WorkflowItem.
163      * 
164      * @param c
165      *            Context
166      * @param wsi
167      *            The WorkspaceItem to convert to a workflow item
168      * @return The resulting workflow item
169      */
170     public static WorkflowItem start(Context c, WorkspaceItem wsi)
171             throws SQLException, AuthorizeException, IOException
172     {
173         // FIXME Check auth
174         Item myitem = wsi.getItem();
175         Collection collection = wsi.getCollection();
176 
177         log.info(LogManager.getHeader(c, "start_workflow", "workspace_item_id="
178                 + wsi.getID() + "item_id=" + myitem.getID() + "collection_id="
179                 + collection.getID()));
180 
181         // record the start of the workflow w/provenance message
182         recordStart(c, myitem);
183 
184         // create the WorkflowItem
185         TableRow row = DatabaseManager.create(c, "workflowitem");
186         row.setColumn("item_id", myitem.getID());
187         row.setColumn("collection_id", wsi.getCollection().getID());
188 
189         WorkflowItem wfi = new WorkflowItem(c, row);
190 
191         wfi.setMultipleFiles(wsi.hasMultipleFiles());
192         wfi.setMultipleTitles(wsi.hasMultipleTitles());
193         wfi.setPublishedBefore(wsi.isPublishedBefore());
194 
195         // remove the WorkspaceItem
196         wsi.deleteWrapper();
197 
198         // now get the worflow started
199         doState(c, wfi, WFSTATE_STEP1POOL, null);
200 
201         // Return the workflow item
202         return wfi;
203     }

288     /**
289      * claim() claims a workflow task for an EPerson
290      * 
291      * @param wi
292      *            WorkflowItem to do the claim on
293      * @param e
294      *            The EPerson doing the claim
295      */
296     public static void claim(Context c, WorkflowItem wi, EPerson e)
297             throws SQLException, IOException, AuthorizeException
298     {
299         int taskstate = wi.getState();
300 
301         switch (taskstate)
302         {
303         case WFSTATE_STEP1POOL:
304 
305             // authorize DSpaceActions.SUBMIT_REVIEW
306             doState(c, wi, WFSTATE_STEP1, e);
307 
308             break;
309 
310         case WFSTATE_STEP2POOL:
311 
312             // authorize DSpaceActions.SUBMIT_STEP2
313             doState(c, wi, WFSTATE_STEP2, e);
314 
315             break;
316 
317         case WFSTATE_STEP3POOL:
318 
319             // authorize DSpaceActions.SUBMIT_STEP3
320             doState(c, wi, WFSTATE_STEP3, e);
321 
322             break;
323 
324         // if we got here, we weren't pooled... error?
325         // FIXME - log the error?
326         }
327 
328         log.info(LogManager.getHeader(c, "claim_task", "workflow_item_id="
329                 + wi.getID() + "item_id=" + wi.getItem().getID()
330                 + "collection_id=" + wi.getCollection().getID()
331                 + "newowner_id=" + wi.getOwner().getID() + "old_state="
332                 + taskstate + "new_state=" + wi.getState()));
333     }

335     /**
336      * approveAction() sends an item forward in the workflow (reviewers,
337      * approvers, and editors all do an 'approve' to move the item forward) if
338      * the item arrives at the submit state, then remove the WorkflowItem and
339      * call the archive() method to put it in the archive, and email notify the
340      * submitter of a successful submission
341      * 
342      * @param c
343      *            Context
344      * @param wi
345      *            WorkflowItem do do the approval on
346      * @param e
347      *            EPerson doing the approval
348      */
349     public static void advance(Context c, WorkflowItem wi, EPerson e)
350             throws SQLException, IOException, AuthorizeException
351     {
352         int taskstate = wi.getState();
353 
354         switch (taskstate)
355         {
356         case WFSTATE_STEP1:
357 
358             // authorize DSpaceActions.SUBMIT_REVIEW
359             // Record provenance
360             recordApproval(c, wi, e);
361             doState(c, wi, WFSTATE_STEP2POOL, e);
362 
363             break;
364 
365         case WFSTATE_STEP2:
366 
367             // authorize DSpaceActions.SUBMIT_STEP2
368             // Record provenance
369             recordApproval(c, wi, e);
370             doState(c, wi, WFSTATE_STEP3POOL, e);
371 
372             break;
373 
374         case WFSTATE_STEP3:
375 
376             // authorize DSpaceActions.SUBMIT_STEP3
377             // We don't record approval for editors, since they can't reject,
378             // and thus didn't actually make a decision
379             doState(c, wi, WFSTATE_ARCHIVE, e);
380 
381             break;
382 
383         // error handling? shouldn't get here
384         }
385 
386         log.info(LogManager.getHeader(c, "advance_workflow",
387                 "workflow_item_id=" + wi.getID() + ",item_id="
388                         + wi.getItem().getID() + ",collection_id="
389                         + wi.getCollection().getID() + ",old_state="
390                         + taskstate + ",new_state=" + wi.getState()));
391     }

442     /**
443      * abort() aborts a workflow, completely deleting it (administrator do this)
444      * (it will basically do a reject from any state - the item ends up back in
445      * the user's PersonalWorkspace
446      * 
447      * @param c
448      *            Context
449      * @param wi
450      *            WorkflowItem to operate on
451      * @param e
452      *            EPerson doing the operation
453      */
454     public static void abort(Context c, WorkflowItem wi, EPerson e)
455             throws SQLException, AuthorizeException, IOException
456     {
457         // authorize a DSpaceActions.ABORT
458         if (!AuthorizeManager.isAdmin(c))
459         {
460             throw new AuthorizeException(
461                     "You must be an admin to abort a workflow");
462         }
463 
464         // stop workflow regardless of its state
465         deleteTasks(c, wi);
466 
467         log.info(LogManager.getHeader(c, "abort_workflow", "workflow_item_id="
468                 + wi.getID() + "item_id=" + wi.getItem().getID()
469                 + "collection_id=" + wi.getCollection().getID() + "eperson_id="
470                 + e.getID()));
471 
472         // convert into personal workspace
473         WorkspaceItem wsi = returnToWorkspace(c, wi);
474     }

476     // returns true if archived
477     private static boolean doState(Context c, WorkflowItem wi, int newstate,
478             EPerson newowner) throws SQLException, IOException,
479             AuthorizeException
480     {
481         Collection mycollection = wi.getCollection();
482         Group mygroup = null;
483         boolean archived = false;
484 
485         wi.setState(newstate);
486 
487         switch (newstate)
488         {
489         case WFSTATE_STEP1POOL:
490 
491             // any reviewers?
492             // if so, add them to the tasklist
493             wi.setOwner(null);
494 
495             // get reviewers (group 1 )
496             mygroup = mycollection.getWorkflowGroup(1);
497 
498             if ((mygroup != null) && !(mygroup.isEmpty()))
499             {
500                 // get a list of all epeople in group (or any subgroups)
501                 EPerson[] epa = Group.allMembers(c, mygroup);
502                 
503                 // there were reviewers, change the state
504                 //  and add them to the list
505                 createTasks(c, wi, epa);
506                 wi.update();
507 
508                 // email notification
509                 notifyGroupOfTask(c, wi, mygroup, epa);
510             }
511             else
512             {
513                 // no reviewers, skip ahead
514                 archived = doState(c, wi, WFSTATE_STEP2POOL, null);
515             }
516 
517             break;
518 
519         case WFSTATE_STEP1:
520 
521             // remove reviewers from tasklist
522             // assign owner
523             deleteTasks(c, wi);
524             wi.setOwner(newowner);
525 
526             break;
527 
528         case WFSTATE_STEP2POOL:
529 
530             // clear owner
531             // any approvers?
532             // if so, add them to tasklist
533             // if not, skip to next state
534             wi.setOwner(null);
535 
536             // get approvers (group 2)
537             mygroup = mycollection.getWorkflowGroup(2);
538 
539             if ((mygroup != null) && !(mygroup.isEmpty()))
540             {
541                 //get a list of all epeople in group (or any subgroups)
542                 EPerson[] epa = Group.allMembers(c, mygroup);
543                 
544                 // there were approvers, change the state
545                 //  timestamp, and add them to the list
546                 createTasks(c, wi, epa);
547 
548                 // email notification
549                 notifyGroupOfTask(c, wi, mygroup, epa);
550             }
551             else
552             {
553                 // no reviewers, skip ahead
554                 archived = doState(c, wi, WFSTATE_STEP3POOL, null);
555             }
556 
557             break;
558 
559         case WFSTATE_STEP2:
560 
561             // remove admins from tasklist
562             // assign owner
563             deleteTasks(c, wi);
564             wi.setOwner(newowner);
565 
566             break;
567 
568         case WFSTATE_STEP3POOL:
569 
570             // any editors?
571             // if so, add them to tasklist
572             wi.setOwner(null);
573             mygroup = mycollection.getWorkflowGroup(3);
574 
575             if ((mygroup != null) && !(mygroup.isEmpty()))
576             {
577                 // get a list of all epeople in group (or any subgroups)
578                 EPerson[] epa = Group.allMembers(c, mygroup);
579                 
580                 // there were editors, change the state
581                 //  timestamp, and add them to the list
582                 createTasks(c, wi, epa);
583 
584                 // email notification
585                 notifyGroupOfTask(c, wi, mygroup, epa);
586             }
587             else
588             {
589                 // no editors, skip ahead
590                 archived = doState(c, wi, WFSTATE_ARCHIVE, newowner);
591             }
592 
593             break;
594 
595         case WFSTATE_STEP3:
596 
597             // remove editors from tasklist
598             // assign owner
599             deleteTasks(c, wi);
600             wi.setOwner(newowner);
601 
602             break;
603 
604         case WFSTATE_ARCHIVE:
605 
606             // put in archive in one transaction
607             try
608             {
609                 // remove workflow tasks
610                 deleteTasks(c, wi);
611 
612                 mycollection = wi.getCollection();
613 
614                 Item myitem = archive(c, wi);
615 
616                 // now email notification
617                 notifyOfArchive(c, myitem, mycollection);
618                 archived = true;
619             }
620             catch (IOException e)
621             {
622                 // indexer causes this
623                 throw e;
624             }
625             catch (SQLException e)
626             {
627                 // problem starting workflow
628                 throw e;
629             }
630 
631             break;
632         }
633 
634         if ((wi != null) && !archived)
635         {
636             wi.update();
637         }
638 
639         return archived;
640     }

669     /**
670      * notify the submitter that the item is archived
671      */
672     private static void notifyOfArchive(Context c, Item i, Collection coll)
673             throws SQLException, IOException
674     {
675         try
676         {
677             // Get submitter
678             EPerson ep = i.getSubmitter();
679             // Get the Locale
680             Locale supportedLocale = I18nUtil.getEPersonLocale(ep);
681             Email email = ConfigurationManager.getEmail(I18nUtil.getEmailFilename(supportedLocale, "submit_archive"));
682             
683             // Get the item handle to email to user
684             String handle = HandleManager.findHandle(c, i);
685 
686             // Get title
687             DCValue[] titles = i.getDC("title", null, Item.ANY);
688             String title = "";
689             try
690             {
691                 title = I18nUtil.getMessage("org.dspace.workflow.WorkflowManager.untitled");
692             }
693             catch (MissingResourceException e)
694             {
695                 title = "Untitled";
696             }
697             if (titles.length > 0)
698             {
699                 title = titles[0].value;
700             }
701 
702             email.addRecipient(ep.getEmail());
703             email.addArgument(title);
704             email.addArgument(coll.getMetadata("name"));
705             email.addArgument(HandleManager.getCanonicalForm(handle));
706 
707             email.send();
708         }
709         catch (MessagingException e)
710         {
711             log.warn(LogManager.getHeader(c, "notifyOfArchive",
712                     "cannot email user" + " item_id=" + i.getID()));
713         }
714     }

724      * @return the workspace item
725      */
726     private static WorkspaceItem returnToWorkspace(Context c, WorkflowItem wfi)
727             throws SQLException, IOException, AuthorizeException
728     {
729         Item myitem = wfi.getItem();
730         Collection mycollection = wfi.getCollection();
731 
732         // FIXME: How should this interact with the workflow system?
733         // FIXME: Remove license
734         // FIXME: Provenance statement?
735         // Create the new workspace item row
736         TableRow row = DatabaseManager.create(c, "workspaceitem");
737         row.setColumn("item_id", myitem.getID());
738         row.setColumn("collection_id", mycollection.getID());
739         DatabaseManager.update(c, row);
740 
741         int wsi_id = row.getIntColumn("workspace_item_id");
742         WorkspaceItem wi = WorkspaceItem.find(c, wsi_id);
743         wi.setMultipleFiles(wfi.hasMultipleFiles());
744         wi.setMultipleTitles(wfi.hasMultipleTitles());
745         wi.setPublishedBefore(wfi.isPublishedBefore());
