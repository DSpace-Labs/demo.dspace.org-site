83      /**
84       * Construct a workspace item corresponding to the given database row
85       * 
86       * @param context
87       *            the context this object exists in
88       * @param row
89       *            the database row
90       */
91      WorkflowItem(Context context, TableRow row) throws SQLException
92      {
93          ourContext = context;
94          wfRow = row;
95  
96          item = Item.find(context, wfRow.getIntColumn("item_id"));
97          collection = Collection.find(context, wfRow
98                  .getIntColumn("collection_id"));
99  
100         if (wfRow.isColumnNull("owner"))
101         {
102             owner = null;
103         }
104         else
105         {
106             owner = EPerson.find(context, wfRow.getIntColumn("owner"));
107         }
108 
109         // Cache ourselves
110         context.cache(this, row.getIntColumn("workflow_id"));
111     }

113     /**
114      * Get a workflow item from the database. The item, collection and submitter
115      * are loaded into memory.
116      * 
117      * @param context
118      *            DSpace context object
119      * @param id
120      *            ID of the workspace item
121      * 
122      * @return the workflow item, or null if the ID is invalid.
123      */
124     public static WorkflowItem find(Context context, int id)
125             throws SQLException
126     {
127         // First check the cache
128         WorkflowItem fromCache = (WorkflowItem) context.fromCache(
129                 WorkflowItem.class, id);
130 
131         if (fromCache != null)
132         {
133             return fromCache;
134         }
135 
136         TableRow row = DatabaseManager.find(context, "workflowitem", id);
137 
138         if (row == null)
139         {
140             if (log.isDebugEnabled())
141             {
142                 log.debug(LogManager.getHeader(context, "find_workflow_item",
143                         "not_found,workflow_id=" + id));
144             }
145 
146             return null;
147         }
148         else
149         {
150             if (log.isDebugEnabled())
151             {
152                 log.debug(LogManager.getHeader(context, "find_workflow_item",
153                         "workflow_id=" + id));
154             }
155 
156             return new WorkflowItem(context, row);
157         }
158     }

160     /**
161      * return all workflowitems
162      * 
163      * @param c  active context
164      * @return WorkflowItem [] of all workflows in system
165      */
166     public static WorkflowItem[] findAll(Context c) throws SQLException
167     {
168         List wfItems = new ArrayList();
169         TableRowIterator tri = DatabaseManager.queryTable(c, "workflowitem",
170                 "SELECT * FROM workflowitem");
171 
172         try
173         {
174             // make a list of workflow items
175             while (tri.hasNext())
176             {
177                 TableRow row = tri.next();
178                 WorkflowItem wi = new WorkflowItem(c, row);
179                 wfItems.add(wi);
180             }
181         }
182         finally
183         {
184             if (tri != null)
185                 tri.close();
186         }
187 
188         WorkflowItem[] wfArray = new WorkflowItem[wfItems.size()];
189         wfArray = (WorkflowItem[]) wfItems.toArray(wfArray);
190 
191         return wfArray;
192     }

194     /**
195      * Get all workflow items that were original submissions by a particular
196      * e-person. These are ordered by workflow ID, since this should likely keep
197      * them in the order in which they were created.
198      * 
199      * @param context
200      *            the context object
201      * @param ep
202      *            the eperson
203      * 
204      * @return the corresponding workflow items
205      */
206     public static WorkflowItem[] findByEPerson(Context context, EPerson ep)
207             throws SQLException
208     {
209         List wfItems = new ArrayList();
210 
211         TableRowIterator tri = DatabaseManager.queryTable(context, "workflowitem",
212                 "SELECT workflowitem.* FROM workflowitem, item WHERE " +
213                 "workflowitem.item_id=item.item_id AND " +
214                 "item.submitter_id= ? " + 
215                 "ORDER BY workflowitem.workflow_id",
216                 ep.getID());
217 
218         try
219         {
220             while (tri.hasNext())
221             {
222                 TableRow row = tri.next();
223 
224                 // Check the cache
225                 WorkflowItem wi = (WorkflowItem) context.fromCache(
226                         WorkflowItem.class, row.getIntColumn("workflow_id"));
227 
228                 if (wi == null)
229                 {
230                     wi = new WorkflowItem(context, row);
231                 }
232 
233                 wfItems.add(wi);
234             }
235         }
236         finally
237         {
238             if (tri != null)
239                 tri.close();
240         }
241 
242         WorkflowItem[] wfArray = new WorkflowItem[wfItems.size()];
243         wfArray = (WorkflowItem[]) wfItems.toArray(wfArray);
244 
245         return wfArray;
246     }

248     /**
249      * Get all workflow items for a particular collection.
250      * 
251      * @param context
252      *            the context object
253      * @param c
254      *            the collection
255      * 
256      * @return array of the corresponding workflow items
257      */
258     public static WorkflowItem[] findByCollection(Context context, Collection c)
259             throws SQLException
260     {
261         List wsItems = new ArrayList();
262 
263         TableRowIterator tri = DatabaseManager.queryTable(context, "workflowitem",
264                 "SELECT workflowitem.* FROM workflowitem WHERE " +
265                 "workflowitem.collection_id= ? ",
266                 c.getID());
267 
268         try
269         {
270             while (tri.hasNext())
271             {
272                 TableRow row = tri.next();
273 
274                 // Check the cache
275                 WorkflowItem wi = (WorkflowItem) context.fromCache(
276                         WorkflowItem.class, row.getIntColumn("workflow_id"));
277 
278                 // not in cache? turn row into workflowitem
279                 if (wi == null)
280                 {
281                     wi = new WorkflowItem(context, row);
282                 }
283 
284                 wsItems.add(wi);
285             }
286         }
287         finally
288         {
289             if (tri != null)
290                 tri.close();
291         }
292 
293         WorkflowItem[] wsArray = new WorkflowItem[wsItems.size()];
294         wsArray = (WorkflowItem[]) wsItems.toArray(wsArray);
295 
296         return wsArray;

379     public void deleteWrapper() throws SQLException, IOException,

381     {
