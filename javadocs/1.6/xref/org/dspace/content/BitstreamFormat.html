95      /**
96       * Class constructor for creating a BitstreamFormat object based on the
97       * contents of a DB table row.
98       * 
99       * @param context
100      *            the context this object exists in
101      * @param row
102      *            the corresponding row in the table
103      * @throws SQLException
104      */
105     BitstreamFormat(Context context, TableRow row) throws SQLException
106     {
107         bfContext = context;
108         bfRow = row;
109         extensions = new ArrayList();
110 
111         TableRowIterator tri = DatabaseManager.query(context,
112                 "SELECT * FROM fileextension WHERE bitstream_format_id= ? ",
113                  getID());
114 
115         try
116         {
117             while (tri.hasNext())
118             {
119                 extensions.add(tri.next().getStringColumn("extension"));
120             }
121         }
122         finally
123         {
124             // close the TableRowIterator to free up resources
125             if (tri != null)
126                 tri.close();
127         }
128 
129         // Cache ourselves
130         context.cache(this, row.getIntColumn("bitstream_format_id"));
131     }

133     /**
134      * Get a bitstream format from the database.
135      * 
136      * @param context
137      *            DSpace context object
138      * @param id
139      *            ID of the bitstream format
140      * 
141      * @return the bitstream format, or null if the ID is invalid.
142      * @throws SQLException
143      */
144     public static BitstreamFormat find(Context context, int id)
145             throws SQLException
146     {
147         // First check the cache
148         BitstreamFormat fromCache = (BitstreamFormat) context.fromCache(
149                 BitstreamFormat.class, id);
150 
151         if (fromCache != null)
152         {
153             return fromCache;
154         }
155 
156         TableRow row = DatabaseManager.find(context, "bitstreamformatregistry",
157                 id);
158 
159         if (row == null)
160         {
161             if (log.isDebugEnabled())
162             {
163                 log.debug(LogManager.getHeader(context,
164                         "find_bitstream_format",
165                         "not_found,bitstream_format_id=" + id));
166             }
167 
168             return null;
169         }
170 
171         // not null, return format object
172         if (log.isDebugEnabled())
173         {
174             log.debug(LogManager.getHeader(context, "find_bitstream_format",
175                     "bitstream_format_id=" + id));
176         }
177 
178         return new BitstreamFormat(context, row);
179     }

181     /**
182      * Find a bitstream format by its (unique) MIME type.
183      * If more than one bitstream format has the same MIME type, the
184      * one returned is unpredictable.
185      *
186      * @param context
187      *            DSpace context object
188      * @param mimeType
189      *            MIME type value
190      *
191      * @return the corresponding bitstream format, or <code>null</code> if
192      *         there's no bitstream format with the given MIMEtype.
193      * @throws SQLException
194      */
195     public static BitstreamFormat findByMIMEType(Context context,
196             String mimeType) throws SQLException
197     {
198         // NOTE: Avoid internal formats since e.g. "License" also has
199         // a MIMEtype of text/plain.
200         TableRow formatRow = DatabaseManager.querySingle(context,
201             "SELECT * FROM bitstreamformatregistry "+
202             "WHERE mimetype LIKE ? AND internal = '0' ",
203             mimeType);
204 
205         if (formatRow == null)
206             return null;
207         return findByFinish(context, formatRow);
208     }

275     public static BitstreamFormat findUnknown(Context context)
276             throws SQLException
277     {
278         BitstreamFormat bf = findByShortDescription(context, "Unknown");
279 
280         if (bf == null)

289     /**
290      * Retrieve all bitstream formats from the registry, ordered by ID
291      * 
292      * @param context
293      *            DSpace context object
294      * 
295      * @return the bitstream formats.
296      * @throws SQLException
297      */
298     public static BitstreamFormat[] findAll(Context context)
299             throws SQLException
300     {
301         List formats = new ArrayList();
302 
303         TableRowIterator tri = DatabaseManager.queryTable(context, "bitstreamformatregistry",
304                         "SELECT * FROM bitstreamformatregistry ORDER BY bitstream_format_id");
305 
306         try
307         {
308             while (tri.hasNext())
309             {
310                 TableRow row = tri.next();
311 
312                 // From cache?
313                 BitstreamFormat fromCache = (BitstreamFormat) context.fromCache(
314                         BitstreamFormat.class, row
315                                 .getIntColumn("bitstream_format_id"));
316 
317                 if (fromCache != null)
318                 {
319                     formats.add(fromCache);
320                 }
321                 else
322                 {
323                     formats.add(new BitstreamFormat(context, row));
324                 }
325             }
326         }
327         finally
328         {
329             // close the TableRowIterator to free up resources
330             if (tri != null)
331                 tri.close();
332         }
333 
334         // Return the formats as an array
335         BitstreamFormat[] formatArray = new BitstreamFormat[formats.size()];
336         formatArray = (BitstreamFormat[]) formats.toArray(formatArray);
337 
338         return formatArray;
339     }

341     /**
342      * Retrieve all non-internal bitstream formats from the registry. The
343      * "unknown" format is not included, and the formats are ordered by support
344      * level (highest first) first then short description.
345      * 
346      * @param context
347      *            DSpace context object
348      * 
349      * @return the bitstream formats.
350      * @throws SQLException
351      */
352     public static BitstreamFormat[] findNonInternal(Context context)
353             throws SQLException
354     {
355         List formats = new ArrayList();
356 
357         String myQuery = "SELECT * FROM bitstreamformatregistry WHERE internal='0' "
358                 + "AND short_description NOT LIKE 'Unknown' "
359                 + "ORDER BY support_level DESC, short_description";
360 
361         TableRowIterator tri = DatabaseManager.queryTable(context,
362                 "bitstreamformatregistry", myQuery);
363 
364         try
365         {
366             while (tri.hasNext())
367             {
368                 TableRow row = tri.next();
369 
370                 // From cache?
371                 BitstreamFormat fromCache = (BitstreamFormat) context.fromCache(
372                         BitstreamFormat.class, row
373                                 .getIntColumn("bitstream_format_id"));
374 
375                 if (fromCache != null)
376                 {
377                     formats.add(fromCache);
378                 }
379                 else
380                 {
381                     formats.add(new BitstreamFormat(context, row));
382                 }
383             }
384         }
385         finally
386         {
387             // close the TableRowIterator to free up resources
388             if (tri != null)
389                 tri.close();
390         }
391 
392         // Return the formats as an array
393         BitstreamFormat[] formatArray = new BitstreamFormat[formats.size()];

449     /**
450      * Set the short description of the bitstream format
451      * 
452      * @param s
453      *            the new short description
454      */
455     public void setShortDescription(String s)
456        throws SQLException
457     {
458         // You can not reset the unknown's registry's name
459         BitstreamFormat unknown = null;;
460 		try {
461 			unknown = findUnknown(bfContext);
462 		} catch (IllegalStateException e) {
463 			// No short_description='Unknown' found in bitstreamformatregistry
464 			// table. On first load of registries this is expected because it
465 			// hasn't been inserted yet! So, catch but ignore this runtime 
466 			// exception thrown by method findUnknown.
467 		}
468 		
469 		// If the exception was thrown, unknown will == null so goahead and 
470 		// load s. If not, check that the unknown's registry's name is not
471 		// being reset.
472 		if (unknown == null || unknown.getID() != getID()) {
473             bfRow.setColumn("short_description", s);
474 		}
475     }

574     /**
575      * Update the bitstream format metadata
576      * 
577      * @throws SQLException
578      * @throws AuthorizeException
579      */
580     public void update() throws SQLException, AuthorizeException
581     {
582         // Check authorisation - only administrators can change formats
583         if (!AuthorizeManager.isAdmin(bfContext))
584         {
585             throw new AuthorizeException(
586                     "Only administrators can modify bitstream formats");
587         }
588 
589         log.info(LogManager.getHeader(bfContext, "update_bitstream_format",
590                 "bitstream_format_id=" + getID()));
591 
592         // Delete extensions
593         DatabaseManager.updateQuery(bfContext,
594                 "DELETE FROM fileextension WHERE bitstream_format_id= ? ",
595                 getID());
596 
597         // Rewrite extensions
598         for (int i = 0; i < extensions.size(); i++)
599         {
600             String s = (String) extensions.get(i);
601             TableRow r = DatabaseManager.create(bfContext, "fileextension");
602             r.setColumn("bitstream_format_id", getID());
603             r.setColumn("extension", s);
604             DatabaseManager.update(bfContext, r);
605         }
606 
607         DatabaseManager.update(bfContext, bfRow);
608     }

610     /**
611      * Delete this bitstream format. This converts the types of any bitstreams
612      * that may have this type to "unknown". Use this with care!
613      * 
614      * @throws SQLException
615      * @throws AuthorizeException
616      */
617     public void delete() throws SQLException, AuthorizeException
618     {
619         // Check authorisation - only administrators can delete formats
620         if (!AuthorizeManager.isAdmin(bfContext))
621         {
622             throw new AuthorizeException(
623                     "Only administrators can delete bitstream formats");
624         }
625 
626         // Find "unknown" type
627         BitstreamFormat unknown = findUnknown(bfContext);
628 
629         if (unknown.getID() == getID())
630 	     throw new IllegalArgumentException("The Unknown bitstream format may not be deleted."); 
631 
632         // Remove from cache
633         bfContext.removeCached(this, getID());
634 
635         // Set bitstreams with this format to "unknown"
636         int numberChanged = DatabaseManager.updateQuery(bfContext,
637                 "UPDATE bitstream SET bitstream_format_id= ? " + 
638                 " WHERE bitstream_format_id= ? ", 
639                 unknown.getID(),getID());
640 
641         // Delete extensions
642         DatabaseManager.updateQuery(bfContext,
643                 "DELETE FROM fileextension WHERE bitstream_format_id= ? ",
644                 getID());
645 
646         // Delete this format from database
647         DatabaseManager.delete(bfContext, bfRow);
648 
649         log.info(LogManager.getHeader(bfContext, "delete_bitstream_format",
650                 "bitstream_format_id=" + getID() + ",bitstreams_changed="
651                         + numberChanged));
652     }
