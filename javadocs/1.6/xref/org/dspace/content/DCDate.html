55  /**
56   * Dublin Core date utility class
57   * <P>
58   * Dates in the DSpace database are held in the ISO 8601 format. They are always
59   * stored in UTC, converting to and from the current time zone.
60   * <P>
61   * <code>YYYY-MM-DDThh:mm:ss</code>
62   * <P>
63   * There are four levels of granularity, depending on how much date information
64   * is available: year, month, day, time.
65   * <P>
66   * Examples: <code>1994-05-03T15:30:24</code>,<code>1995-10-04</code>,
67   * <code>2001-10</code>,<code>1975</code>
68   *
69   * The main() method is a simple test program: run it with an optional
70   * first argument that is a date string to decode, and it prints the
71   * results of all the accessor methods.
72   *
73   * @author Robert Tansley
74   * @author Larry Stone
75   * @version $Revision: 4277 $
76   */

137     /**
138      * Construct a date object from a Java <code>Date</code> object.
139      *
140      * @param date
141      *            the Java <code>Date</code> object.
142      */
143     public DCDate(Date date)
144     {
145         super();
146         setTime(date);
147         if (!(calendar.get(Calendar.HOUR_OF_DAY) == 0 &&
148               calendar.get(Calendar.MINUTE) == 0 &&
149               calendar.get(Calendar.SECOND) == 0))
150             granularity = DateGran.TIME;
151 
152         // if date is 1-jan, assume it's because this was set for year
153         else if (calendar.get(Calendar.DAY_OF_MONTH) == 1 && calendar.get(Calendar.MONTH) == 0)
154             granularity = DateGran.YEAR;
155 
156         // otherwise day
157         else
158             granularity = DateGran.DAY;
159     }

161     /**
162      * Construct a date from a Dublin Core value
163      *
164      * @param fromDC
165      *            the date string, in ISO 8601 (no timezone, always use UTC/GMT)
166      */
167     public DCDate(String fromDC)
168     {
169         super();
170 
171         // An empty date is OK
172         if ((fromDC == null) || fromDC.equals(""))
173         {
174             return;
175         }
176         // default granularity
177         granularity = DateGran.TIME;
178         Date date = tryParse(fullIso, fromDC);
179         if (date == null)
180             date = tryParse(fullIso2, fromDC);
181         if (date == null)
182             date = tryParse(fullIso3, fromDC);
183         if (date == null)
184         {
185             date = tryParse(dateIso, fromDC);
186             granularity = DateGran.DAY;
187         }
188         if (date == null)
189         {
190             date = tryParse(yearMonthIso, fromDC);
191             granularity = DateGran.MONTH;
192         }
193         if (date == null)
194         {
195             date = tryParse(yearIso, fromDC);
196             granularity = DateGran.YEAR;
197         }
198 
199         if (date == null)
200             log.warn("Mangled date: " + fromDC + "  ..failed all attempts to parse as date.");
201         else
202             setTime(date);
203     }

205     // Attempt to parse, swallowing errors; return null for failure.
206     private synchronized Date tryParse(SimpleDateFormat sdf,  String source)
207     {
208         try
209         {
210             return sdf.parse(source);
211         }
212         catch (ParseException pe)
213         {
214             return null;
215         }
216     }

252     private synchronized String toStringInternal()
253     {
254         if (granularity == DateGran.YEAR)
255             return yearIso.format(calendar.getTime());
256         else if (granularity == DateGran.MONTH)
257             return yearMonthIso.format(calendar.getTime());
258         else if (granularity == DateGran.DAY)
259             return dateIso.format(calendar.getTime());
260         else
261             return fullIso.format(calendar.getTime());
262     }

277     /**
278      * Set the date. The date passed in is assumed to be in the current time
279      * zone, and is adjusting to fit the current time zone. Unknown values
280      * should be given as -1.
281      *
282      * @param yyyy
283      *            the year
284      * @param mm
285      *            the month
286      * @param dd
287      *            the day
288      * @param hh
289      *            the hours
290      * @param mn
291      *            the minutes
292      * @param ss
293      *            the seconds
294      */
295     public void setDateLocal(int yyyy, int mm, int dd, int hh, int mn, int ss)
296     {
297         // default values
298         int lyear = 0;
299         int lhours = 0;
300         int lminutes = 0;
301         int lseconds = 0;
302         int lmonth = 1;
303         int lday = 1;
304 
305         if (yyyy > 0)
306         {
307             lyear = yyyy;
308             granularity = DateGran.YEAR;
309         }
310         if (mm > 0)
311         {
312             lmonth = mm;
313             granularity = DateGran.MONTH;
314         }
315         if (dd > 0)
316         {
317             lday = dd;
318             granularity = DateGran.DAY;
319         }
320         if (hh >= 0)
321         {
322             lhours = hh;
323             granularity = DateGran.TIME;
324         }
325         if (mn >= 0)
326         {
327             lminutes = mn;
328             granularity = DateGran.TIME;
329         }
330         if (ss >= 0)
331         {
332             lseconds = ss;
333             granularity = DateGran.TIME;
334         }
335 
336         // do the timezone adjustment: get Date and put it in UTC zone.
337         GregorianCalendar localGC = new GregorianCalendar(lyear, lmonth - 1, lday,
338                                         lhours, lminutes, lseconds);
339         setTime(localGC.getTime());
340     }

342     // get cached calendar in local timezone
343     private GregorianCalendar getLocalCalendar()
344     {
345         if (localCalendar == null)
346         {
347             if (calendar == null)
348                 return null;
349             localCalendar = new GregorianCalendar();
350             localCalendar.setTime(calendar.getTime());
351         }
352         return localCalendar;
353     }

475     /**
476      * Get a month's name for a month between 1 and 12. Any invalid month value
477      * (e.g. 0 or -1) will return a value of "Unspecified".
478      *
479      * @param m
480      *            the month number
481      *
482      * @return the month name.
483      */
484     public static String getMonthName(int m, Locale locale)
485     {
486         if ((m > 0) && (m < 13))
487         {
488             if (dfs == null || !langMonth.equals(locale))
489                 {
490                         dfs = new DateFormatSymbols(locale);
491                         langMonth = locale;
492                 }
493            return dfs.getMonths()[m-1];
494         }
495         else
496         {
497             return "Unspecified";
498         }
499 
500     }

502     /**
503      * Simple test program
504      * Usage:  java org.dspace.content.DCdate [DCDate | -l yyyy [mm [dd ..]]] ]
505      *  (-l form tests local time parsing)
506      * Default is to use current time.
507      */
508     public static void main(String args[])
509         throws Exception
510     {
511         DCDate d;
512 
513         // if there's an arg, parse it for the date, otherwise use now
514         if (args.length > 0)
515         {
516             if (args[0].equals("-l"))
517             {
518                 int val[] = { -1, -1, -1, -1, -1, -1 };
519                 for (int i = 1; i < 7 && i < args.length; ++i)
520                     val[i-1] = Integer.parseInt(args[i]);
521                 d = new DCDate();
522                 d.setDateLocal(val[0], val[1], val[2], val[3], val[4], val[5]);
523             }
524             else
525                 d = new DCDate(args[0]);
526         }
527         else
528             d = DCDate.getCurrent();
529 
530         // display results:
531         System.out.println("toString() = \""+d.toString()+"\"");
532         System.out.println("toDate().toString() = \""+d.toDate().toString()+"\"");
533 
534         System.out.println("By component:");
535         System.out.println("granularity   = "+d.granularity);
536         System.out.println("getYear(),    = "+d.getYear());
537         System.out.println("getMonth(),   = "+d.getMonth());
538         System.out.println("getDay(),     = "+d.getDay());
539         System.out.println("getHour(),    = "+d.getHour());
540         System.out.println("getMinute(),  = "+d.getMinute());
541         System.out.println("getSecond()); = "+d.getSecond());
542         System.out.println("By GMT component:");
543         System.out.println("getYearGMT(),    = "+d.getYearGMT());
544         System.out.println("getMonthGMT(),   = "+d.getMonthGMT());
545         System.out.println("getDayGMT(),     = "+d.getDayGMT());
546         System.out.println("getHourGMT(),    = "+d.getHourGMT());
547         System.out.println("getMinuteGMT(),  = "+d.getMinuteGMT());
548         System.out.println("getSecondGMT()); = "+d.getSecondGMT());
549 
550         // convert it the hard way:
551         DCDate hw = new DCDate();
552         hw.setDateLocal(d.getYear(),d.getMonth(),d.getDay(),
553                         d.getHour(),d.getMinute(),d.getSecond());
554         System.out.println("hardway.toString() = \""+hw.toString()+"\"");
555 
556         // month str
557         System.out.println("Month Name   = \""+DCDate.getMonthName(d.getMonth(), Locale.getDefault())+"\"");
558     }
