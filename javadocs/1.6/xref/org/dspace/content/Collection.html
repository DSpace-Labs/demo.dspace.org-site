72  /**
73   * Class representing a collection.
74   * <P>
75   * The collection's metadata (name, introductory text etc), workflow groups, and
76   * default group of submitters are loaded into memory. Changes to metadata are
77   * not written to the database until <code>update</code> is called. If you
78   * create or remove a workflow group, the change is only reflected in the
79   * database after calling <code>update</code>. The default group of
80   * submitters is slightly different - creating or removing this has instant
81   * effect.
82   * 
83   * @author Robert Tansley
84   * @version $Revision: 4309 $
85   */

124     /**
125      * Construct a collection with the given table row
126      * 
127      * @param context
128      *            the context this object exists in
129      * @param row
130      *            the corresponding row in the table
131      * @throws SQLException
132      */
133     Collection(Context context, TableRow row) throws SQLException
134     {
135         ourContext = context;
136         collectionRow = row;
137 
138         // Get the logo bitstream
139         if (collectionRow.isColumnNull("logo_bitstream_id"))
140         {
141             logo = null;
142         }
143         else
144         {
145             logo = Bitstream.find(ourContext, collectionRow
146                     .getIntColumn("logo_bitstream_id"));
147         }
148 
149         // Get the template item
150         if (collectionRow.isColumnNull("template_item_id"))
151         {
152             template = null;
153         }
154         else
155         {
156             template = Item.find(ourContext, collectionRow
157                     .getIntColumn("template_item_id"));
158         }
159 
160         // Get the relevant groups
161         workflowGroup = new Group[3];
162 
163         workflowGroup[0] = groupFromColumn("workflow_step_1");
164         workflowGroup[1] = groupFromColumn("workflow_step_2");
165         workflowGroup[2] = groupFromColumn("workflow_step_3");
166 
167         submitters = groupFromColumn("submitter");
168         admins = groupFromColumn("admin");
169         
170         // Get our Handle if any
171         handle = HandleManager.findHandle(context, this);
172 
173         // Cache ourselves
174         context.cache(this, row.getIntColumn("collection_id"));
175 
176         modified = modifiedMetadata = false;
177         clearDetails();
178     }

180     /**
181      * Get a collection from the database. Loads in the metadata
182      * 
183      * @param context
184      *            DSpace context object
185      * @param id
186      *            ID of the collection
187      * 
188      * @return the collection, or null if the ID is invalid.
189      * @throws SQLException
190      */
191     public static Collection find(Context context, int id) throws SQLException
192     {
193         // First check the cache
194         Collection fromCache = (Collection) context.fromCache(Collection.class,
195                 id);
196 
197         if (fromCache != null)
198         {
199             return fromCache;
200         }
201 
202         TableRow row = DatabaseManager.find(context, "collection", id);
203 
204         if (row == null)
205         {
206             if (log.isDebugEnabled())
207             {
208                 log.debug(LogManager.getHeader(context, "find_collection",
209                         "not_found,collection_id=" + id));
210             }
211 
212             return null;
213         }
214 
215         // not null, return Collection
216         if (log.isDebugEnabled())
217         {
218             log.debug(LogManager.getHeader(context, "find_collection",
219                     "collection_id=" + id));
220         }
221 
222         return new Collection(context, row);
223     }

225     /**
226      * Create a new collection, with a new ID. This method is not public, and
227      * does not check authorisation.
228      * 
229      * @param context
230      *            DSpace context object
231      * 
232      * @return the newly created collection
233      * @throws SQLException
234      * @throws AuthorizeException
235      */
236     static Collection create(Context context) throws SQLException,
237             AuthorizeException
238     {
239         TableRow row = DatabaseManager.create(context, "collection");
240         Collection c = new Collection(context, row);
241         c.handle = HandleManager.createHandle(context, c);
242 
243         // create the default authorization policy for collections
244         // of 'anonymous' READ
245         Group anonymousGroup = Group.find(context, 0);
246 
247         ResourcePolicy myPolicy = ResourcePolicy.create(context);
248         myPolicy.setResource(c);
249         myPolicy.setAction(Constants.READ);
250         myPolicy.setGroup(anonymousGroup);
251         myPolicy.update();
252 
253         // now create the default policies for submitted items
254         myPolicy = ResourcePolicy.create(context);
255         myPolicy.setResource(c);
256         myPolicy.setAction(Constants.DEFAULT_ITEM_READ);
257         myPolicy.setGroup(anonymousGroup);
258         myPolicy.update();
259 
260         myPolicy = ResourcePolicy.create(context);
261         myPolicy.setResource(c);
262         myPolicy.setAction(Constants.DEFAULT_BITSTREAM_READ);
263         myPolicy.setGroup(anonymousGroup);
264         myPolicy.update();
265 
266         context.addEvent(new Event(Event.CREATE, Constants.COLLECTION, c.getID(), c.handle));
267 
268         log.info(LogManager.getHeader(context, "create_collection",
269                 "collection_id=" + row.getIntColumn("collection_id"))
270                 + ",handle=" + c.handle);
271 
272         return c;
273     }

275     /**
276      * Get all collections in the system. These are alphabetically sorted by
277      * collection name.
278      * 
279      * @param context
280      *            DSpace context object
281      * 
282      * @return the collections in the system
283      * @throws SQLException
284      */
285     public static Collection[] findAll(Context context) throws SQLException
286     {
287         TableRowIterator tri = DatabaseManager.queryTable(context, "collection",
288                 "SELECT * FROM collection ORDER BY name");
289 
290         List<Collection> collections = new ArrayList<Collection>();
291 
292         try
293         {
294             while (tri.hasNext())
295             {
296                 TableRow row = tri.next();
297 
298                 // First check the cache
299                 Collection fromCache = (Collection) context.fromCache(
300                         Collection.class, row.getIntColumn("collection_id"));
301 
302                 if (fromCache != null)
303                 {
304                     collections.add(fromCache);
305                 }
306                 else
307                 {
308                     collections.add(new Collection(context, row));
309                 }
310             }
311         }
312         finally
313         {
314             // close the TableRowIterator to free up resources
315             if (tri != null)
316                 tri.close();
317         }
318 
319         Collection[] collectionArray = new Collection[collections.size()];
320         collectionArray = (Collection[]) collections.toArray(collectionArray);
321 
322         return collectionArray;
323     }

428                 value = "Untitled";
429             }
430         }
431         
432         /* 
433          * Set metadata field to null if null 
434          * and trim strings to eliminate excess
435          * whitespace.
436          */
437 		if(value == null)
438         {
439             collectionRow.setColumnNull(field);
440         }
441         else
442         {
443             collectionRow.setColumn(field, value.trim());
444         }
445         
446         modifiedMetadata = true;
447         addDetails(field);

466     /**
467      * Give the collection a logo. Passing in <code>null</code> removes any
468      * existing logo. You will need to set the format of the new logo bitstream
469      * before it will work, for example to "JPEG". Note that
470      * <code>update(/code> will need to be called for the change to take
471      * effect.  Setting a logo and not calling <code>update</code> later may
472      * result in a previous logo lying around as an "orphaned" bitstream.
473      *
474      * @param  is   the stream to use as the new logo
475      *
476      * @return   the new logo bitstream, or <code>null</code> if there is no
477      *           logo (<code>null</code> was passed in)
478      * @throws AuthorizeException
479      * @throws IOException
480      * @throws SQLException
481      */
482     public Bitstream setLogo(InputStream is) throws AuthorizeException,
483             IOException, SQLException
484     {
485         // Check authorisation
486         // authorized to remove the logo when DELETE rights
487         // authorized when canEdit
488         if (!((is == null) && AuthorizeManager.authorizeActionBoolean(
489                 ourContext, this, Constants.DELETE)))
490         {
491             canEdit(true);
492         }
493 
494         // First, delete any existing logo
495         if (!collectionRow.isColumnNull("logo_bitstream_id"))
496         {
497             logo.delete();
498         }
499 
500         if (is == null)
501         {
502             collectionRow.setColumnNull("logo_bitstream_id");
503             logo = null;
504 
505             log.info(LogManager.getHeader(ourContext, "remove_logo",
506                     "collection_id=" + getID()));
507         }
508         else
509         {
510             Bitstream newLogo = Bitstream.create(ourContext, is);
511             collectionRow.setColumn("logo_bitstream_id", newLogo.getID());
512             logo = newLogo;
513 
514             // now create policy for logo bitstream
515             // to match our READ policy
516             List policies = AuthorizeManager.getPoliciesActionFilter(
517                     ourContext, this, Constants.READ);
518             AuthorizeManager.addPolicies(ourContext, policies, newLogo);
519 
520             log.info(LogManager.getHeader(ourContext, "set_logo",
521                     "collection_id=" + getID() + "logo_bitstream_id="
522                             + newLogo.getID()));
523         }
524 
525         modified = true;
526         return logo;
527     }

529     /**
530      * Create a workflow group for the given step if one does not already exist.
531      * Returns either the newly created group or the previously existing one.
532      * Note that while the new group is created in the database, the association
533      * between the group and the collection is not written until
534      * <code>update</code> is called.
535      * 
536      * @param step
537      *            the step (1-3) of the workflow to create or get the group for
538      * 
539      * @return the workflow group associated with this collection
540      * @throws SQLException
541      * @throws AuthorizeException
542      */
543     public Group createWorkflowGroup(int step) throws SQLException,
544             AuthorizeException
545     {
546         // Check authorisation - Must be an Admin to create Workflow Group
547         AuthorizeUtil.authorizeManageWorkflowsGroup(ourContext, this);
548 
549         if (workflowGroup[step - 1] == null)
550         {
551             //turn off authorization so that Collection Admins can create Collection Workflow Groups
552             ourContext.turnOffAuthorisationSystem();
553             Group g = Group.create(ourContext);
554             ourContext.restoreAuthSystemState();
555 
556             g.setName("COLLECTION_" + getID() + "_WORKFLOW_STEP_" + step);
557             g.update();
558             setWorkflowGroup(step, g);
559 
560             AuthorizeManager.addPolicy(ourContext, this, Constants.ADD, g);
561         }
562 
563         return workflowGroup[step - 1];
564     }

633         collectionRow.setColumn("submitter", submitters.getID());
634         
635         AuthorizeManager.addPolicy(ourContext, this, Constants.ADD, submitters);
636 
637         modified = true;
