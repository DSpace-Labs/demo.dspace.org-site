38  package org.dspace.content.authority;
39  
40  /**
41   * Plugin interface that supplies an authority control mechanism for
42   * one metadata field.
43   *
44   * @author Larry Stone
45   * @see ChoiceAuthorityManager, MetadataAuthorityManager
46   */
47  public interface ChoiceAuthority
48  {
49      /**
50       * Get all values from the authority that match the proferred value.
51       * Note that the offering was entered by the user and may contain
52       * mixed/incorrect case, whitespace, etc so the plugin should be careful
53       * to clean up user data before making comparisons.
54       *
55       * Value of a "Name" field will be in canonical DSpace person name format,
56       * which is "Lastname, Firstname(s)", e.g. "Smith, John Q.".
57       *
58       * Some authorities with a small set of values may simply return the whole
59       * set for any sample value, although it's a good idea to set the
60       * defaultSelected index in the Choices instance to the choice, if any,
61       * that matches the value.
62       *
63       * @param text user's value to match
64       * @param collection database ID of Collection for context (owner of Item)
65       * @param start choice at which to start, 0 is first.
66       * @param limit maximum number of choices to return, 0 for no limit.
67       * @param locale explicit localization key if available, or null
68       * @return a Choices object (never null).
69       */
70      public Choices getMatches(String text, int collection, int start, int limit, String locale);
71  
72      /**
73       * Get the single "best" match (if any) of a value in the authority
74       * to the given user value.  The "confidence" element of Choices is
75       * expected to be set to a meaningful value about the circumstances of
76       * this match.
77       *
78       * This call is typically used in non-interactive metadata ingest
79       * where there is no interactive agent to choose from among options.
80       *
81       * @param text user's value to match
82       * @param collection database ID of Collection for context (owner of Item)
83       * @param locale explicit localization key if available, or null
84       * @return a Choices object (never null) with 1 or 0 values.
85       */
86      public Choices getBestMatch(String text, int collection, String locale);
87  
88      /**
89       * Get the canonical user-visible "label" (i.e. short descriptive text)
90       * for a key in the authority.  Can be localized given the implicit
91       * or explicit locale specification.
92       *
93       * This may get called many times while populating a Web page so it should
94       * be implemented as efficiently as possible.
95       *
96       * @param key authority key known to this authority.
97       * @param locale explicit localization key if available, or null
98       * @return descriptive label - should always return something, never null.
99       */
100     public String getLabel(String key, String locale);
101 }
