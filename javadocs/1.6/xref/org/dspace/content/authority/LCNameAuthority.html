75  /**
76   * Sample personal name authority based on Library of Congress Name Authority
77   * Also serves as an example of an SRU client as authority.

79   * This is tuned for the data in the LC Name Authority test instance, see

82   * WARNING: This is just a proof-of-concept implementation.  It would need
83   * WARNING: lots of refinement to be used in production, because it is very
84   * WARNING: sloppy about digging through the MARC/XML results.  No doubt

150     /**
151      * Guts of the implementation, returns a complete Choices result, or
152      * null for a failure.
153      */
154     private Choices queryPerson(String text, int start, int limit)
155     {
156         // punt if there is no query text
157         if (text == null || text.trim().length() == 0)
158             return new Choices(true);
159 
160         // 1. build CQL query
161         DCPersonName pn = new DCPersonName(text);
162         StringBuilder query = new StringBuilder();
163         query.append("local.FirstName = \"").append(pn.getFirstNames()).
164           append("\" and local.FamilyName = \"").append(pn.getLastName()).
165           append("\"");
166 
167         // XXX arbitrary default limit - should be configurable?
168         if (limit == 0)
169             limit = 50;
170 
171         NameValuePair args[] = new NameValuePair[6];
172         args[0] = new NameValuePair("operation", "searchRetrieve");
173         args[1] = new NameValuePair("version", "1.1");
174         args[2] = new NameValuePair("recordSchema", "info:srw/schema/1/marcxml-v1.1");
175         args[3] = new NameValuePair("query", query.toString());
176         args[4] = new NameValuePair("maximumRecords", String.valueOf(limit));
177         args[5] = new NameValuePair("startRecord", String.valueOf(start+1));
178         HttpClient hc = new HttpClient();
179         String srUrl = url + "?" + EncodingUtil.formUrlEncode(args, "UTF8");
180         GetMethod get = new GetMethod(srUrl);
181 
182         log.debug("Trying SRU query, URL="+srUrl);
183 
184         // 2. web request
185         try
186         {
187             int status = hc.executeMethod(get);
188             if (status == 200)
189             {
190                 SAXParserFactory spf = SAXParserFactory.newInstance();
191                 SAXParser sp = spf.newSAXParser();
192                 XMLReader xr = sp.getXMLReader();
193                 SRUHandler handler = new SRUHandler();
194 
195                 // XXX FIXME: should turn off validation here explicitly, but
196                 //  it seems to be off by default.
197                 xr.setFeature("http://xml.org/sax/features/namespaces", true);
198                 xr.setContentHandler(handler);
199                 xr.setErrorHandler(handler);
200                 xr.parse(new InputSource(get.getResponseBodyAsStream()));
201 
202                 // this probably just means more results available..
203                 if (handler.hits != handler.result.size())
204                     log.warn("Discrepency in results, result.length="+handler.result.size()+
205                           ", yet expected results="+handler.hits);
206                 boolean more = handler.hits > (start + handler.result.size());
207 
208                 // XXX add non-auth option; perhaps the UI should do this?
209                 // XXX it's really a policy matter if they allow unauth result.
210                    // XXX good, stop it.
211                 // handler.result.add(new Choice("", text, "Non-Authority: \""+text+"\""));
212 
213                 int confidence;
214                 if (handler.hits == 0)
215                     confidence = Choices.CF_NOTFOUND;
216                 else if (handler.hits == 1)
217                     confidence = Choices.CF_UNCERTAIN;
218                 else
219                     confidence = Choices.CF_AMBIGUOUS;
220                 return new Choices(handler.result.toArray(new Choice[handler.result.size()]),
221                                    start, handler.hits, confidence, more);
222             }
223         }
224         catch (HttpException e)
225         {
226             log.error("SRU query failed: ", e);
227             return new Choices(true);
228         }
229         catch (IOException e)
230         {
231             log.error("SRU query failed: ", e);
232             return new Choices(true);
233         }
234         catch (ParserConfigurationException  e)
235         {
236             log.warn("Failed parsing SRU result: ", e);
237             return new Choices(true);
238         }
239         catch (SAXException  e)
240         {
241             log.warn("Failed parsing SRU result: ", e);
242             return new Choices(true);
243         }
244         finally
245         {
246             get.releaseConnection();
247         }
248         return new Choices(true);
249     }

251     /**
252      * XXX FIXME TODO: Very sloppy MARC/XML parser.
253      * This only reads subfields 010.a (for LCCN, to use as key)
254      * and 100.a (for "established personal name")
255      * Maybe look at Indicator on 100 too.
256      * Should probably read other 100 subfields to build a more detailed label.
257      */
258     private static class SRUHandler
259         extends DefaultHandler
260     {
261         private List<Choice> result = new ArrayList<Choice>();
262         private int hits = -1;
263         private String textValue = null;
264         private String name = null;
265         private String lccn = null;
266         private String lastTag = null;
267         private String lastCode = null;
268 
269         // NOTE:  text value MAY be presented in multiple calls, even if
270         // it all one word, so be ready to splice it together.
271         // BEWARE:  subclass's startElement method should call super()
272         // to null out 'value'.  (Don't you miss the method combination
273         // options of a real object system like CLOS?)
274         public void characters(char[] ch, int start, int length)
275             throws SAXException
276         {
277             String newValue = new String(ch, start, length);
278             if (newValue.length() > 0)
279             {
280                 if (textValue == null)
281                     textValue = newValue;
282                 else
283                     textValue += newValue;
284             }
285         }

287         public void endElement(String namespaceURI, String localName,
288                                  String qName)
289             throws SAXException
290         {
291             if (localName.equals("numberOfRecords") &&
292                      namespaceURI.equals(NS_SRU))
293             {
294                 hits = Integer.parseInt(textValue.trim());
295                 if (hits > 0)
296                 {
297                     name = null;
298                     lccn = null;
299                     log.debug("Expecting "+hits+" records in results.");
300                 }
301             }
302 
303             // after record get next hit ready
304             else if (localName.equals("record") &&
305                      namespaceURI.equals(NS_SRU))
306             {
307                 if (name != null && lccn != null)
308                 {
309                     // HACK: many LC name entries end with ',' ...trim it.
310                     if (name.endsWith(","))
311                         name = name.substring(0, name.length()-1);
312 
313                     // XXX DEBUG
314                     // log.debug("Got result, name="+name+", lccn="+lccn);
315                     result.add(new Choice(lccn, name, name));
316                 }
317                 else
318                     log.warn("Got anomalous result, at least one of these null: lccn="+lccn+", name="+name);
319                 name = null;
320                 lccn = null;
321             }
322 
323             else if (localName.equals("subfield") &&
324                      namespaceURI.equals(NS_MX))
325             {
326                 if (lastTag != null && lastCode != null)
327                 {
328                     // 010.a is lccn, "authority code"
329                     if (lastTag.equals("010") && lastCode.equals("a"))
330                         lccn = textValue;
331                      
332                     // 100.a is the personal name
333                     else if (lastTag.equals("100") && lastCode.equals("a"))
334                         name = textValue;
335                 }
336             }
337         }

339         // subclass overriding this MUST call it with super()
340         public void startElement(String namespaceURI, String localName,
341                                  String qName, Attributes atts)
342             throws SAXException
343         {
344             textValue = null;
345 
346             if (localName.equals("datafield") &&
347                      namespaceURI.equals(NS_MX))
348             {
349                 lastTag = atts.getValue("tag");
350                 if (lastTag == null)
351                     log.warn("MARC datafield without tag attribute!");
352             }
353             else if (localName.equals("subfield") &&
354                      namespaceURI.equals(NS_MX))
355             {
356                 lastCode = atts.getValue("code");
357                 if (lastCode == null)
358                     log.warn("MARC subfield without code attribute!");
359             }
360         }
