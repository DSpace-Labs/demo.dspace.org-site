70  /**
71   * Configurable QDC Crosswalk
72   * <p>
73   * This class supports multiple dissemination crosswalks from DSpace
74   * internal data to the Qualified Dublin Core XML format
75   *  (see <a href="http://dublincore.org/">http://dublincore.org/</a>
76   * <p>
77   * It registers multiple Plugin names, which it reads from
78   * the DSpace configuration as follows:
79   *
80   * <h3>Configuration</h3>
81   * Every key starting with <code>"crosswalk.qdc.properties."</code> describes a
82   * QDC crosswalk.  Everything after the last period is the <em>plugin instance</em>,
83   * and the value is the pathname (relative to <code><em>dspace.dir</em>/config</code>)
84   * of the crosswalk configuration file.
85   * <p>
86   * You can have two aliases point to the same crosswalk,
87   * just add two configuration entries with the same value, e.g.
88   * <pre>
89   *    crosswalk.qdc.properties.QDC = xwalk/qdc.properties
90   *    crosswalk.qdc.properties.default = xwalk/qdc.properties
91   * </pre>
92   * The first line creates a plugin with the name <code>"QDC"</code>
93   * which is configured from the file <em>dspace-dir</em><code>/xwalk/qdc.properties</code>.
94   * <p>
95   * Since there is significant overhead in reading the properties file to
96   * configure the crosswalk, and a crosswalk instance may be used any number
97   * of times, we recommend caching one instance of the crosswalk for each
98   * alias and simply reusing those instances. The PluginManager does
99   * this by default.
100  * <p>
101  * Each named crosswalk has two other types of configuration lines:
102  * <p>
103  * XML Namespaces: all XML namespace prefixes used in the XML fragments below
104  * <em>must</em> be defined in the configuration as follows.  Add a line of
105  * the form: <pre>
106  *  crosswalk.qdc.namespace.{NAME}.{prefix} = {namespace-URI}</pre>
107  * e.g. for the namespaces <code>dc</code> and <code>dcterms</code>
108  * in the plugin named <code>QDC</code>, add these lines:
109  * <pre>crosswalk.qdc.namespace.QDC.dc = http://purl.org/dc/elements/1.1/
110  * crosswalk.qdc.namespace.QDC.dcterms = http://purl.org/dc/terms/</pre>
111  *
112  * <p>
113  * Finally, you need to declare an XML Schema URI for the plugin, with
114  * a line of the form <pre>
115  *  crosswalk.qdc.schema.{NAME} = {schema-URI}</pre>
116  * for example,
117  * <pre>crosswalk.qdc.schemaLocation.QDC  = \
118  *  http://purl.org/dc/terms/ \
119  *  http://dublincore.org/schemas/xmls/qdc/2003/04/02/qualifieddc.xsd</pre>
120  *
121  * @author Larry Stone
122  * @version $Revision: 3761 $
123  */

185     // utility: return "fully qualified" name of XML element, for a
186     // hashtable key to use on ingesting elements.
187     // Format is {prefix:}name  where prefix is optional.
188     private String makeQualifiedTagName(Element element)
189     {
190         String prefix = "";
191         Namespace ns = element.getNamespace();
192         if (ns != null)
193             prefix = ns.getPrefix() + ":";
194         
195         String tagName;
196         String nsQualifier = element.getAttributeValue("type", DisseminationCrosswalk.XSI_NS);
197         
198         if (nsQualifier == null || nsQualifier.length() < 1)
199         {
200             String qualifier = element.getAttributeValue("type");
201             if (qualifier == null || qualifier.length() < 1)
202             {
203             	tagName = prefix+element.getName();
204             }
205             else
206             {
207             	tagName = prefix+element.getName()+qualifier;
208             }
209         }
210         else
211         {
212         	tagName = prefix+element.getName()+nsQualifier;
213         }
214         
215         return tagName;
216     }

218     /**
219      * Initialize Crosswalk table from a properties file
220      * which itself is the value of the DSpace configuration property
221      * "crosswalk.qdc.properties.X", where "X" is the alias name of this instance.
222      * Each instance may be configured with a separate mapping table.
223      *
224      * The QDC crosswalk configuration properties follow the format:
225      *
226      *  {qdc-element} = {XML-fragment}
227      *
228      *  1. qualified DC field name is of the form (qualifier is optional)
229      *       {MDschema}.{element}.{qualifier}
230      *
231      *      e.g.  dc.contributor.author
232      *            dc.title
233      *
234      *  2. XML fragment is prototype of metadata element, with empty
235      *     placeholders for value).
236      *
237      * Example properties line:
238      *
239      *  dc.coverage.temporal = <dcterms:temporal />
240      */
241     private void init()
242         throws CrosswalkException, IOException
243     {
244         if (inited)
245             return;
246         inited = true;
247 
248         myName = getPluginInstanceName();
249         if (myName == null)
250             throw new CrosswalkInternalException("Cannot determine plugin name, "+
251                        "You must use PluginManager to instantiate QDCCrosswalk so the instance knows its name.");
252 
253         // grovel DSpace configuration for namespaces
254         List nsList = new ArrayList();
255         Enumeration pe = ConfigurationManager.propertyNames();
256         String propname = CONFIG_PREFIX + ".namespace."+ myName +".";
257         while (pe.hasMoreElements())
258         {
259             String key = (String)pe.nextElement();
260             if (key.startsWith(propname))
261                 nsList.add(Namespace.getNamespace(key.substring(propname.length()),
262                              ConfigurationManager.getProperty(key)));
263         }
264         nsList.add(Namespace.XML_NAMESPACE);
265         namespaces = (Namespace[])nsList.toArray(new Namespace[nsList.size()]);
266 
267         // get XML schemaLocation fragment from config
268         schemaLocation = ConfigurationManager.getProperty(CONFIG_PREFIX + ".schemaLocation."+ myName);
269 
270         // read properties
271         String cmPropName = CONFIG_PREFIX+".properties."+myName;
272         String propsFilename = ConfigurationManager.getProperty(cmPropName);
273         if (propsFilename == null)
274             throw new CrosswalkInternalException("Configuration error: "+
275                 "No properties file configured for QDC crosswalk named \""+myName+"\"");
276 
277         String parent = ConfigurationManager.getProperty("dspace.dir") +
278             File.separator + "config" + File.separator;
279         File propsFile = new File(parent, propsFilename);
280         Properties qdcProps = new Properties();
281         FileInputStream pfs = null;
282         try
283         {
284             pfs = new FileInputStream(propsFile);
285             qdcProps.load(pfs);
286         }
287         finally
288         {
289             if (pfs != null)
290                 try { pfs.close(); } catch (IOException ioe) { }
291         }
292 
293         // grovel properties to initialize qdc->element and element->qdc maps.
294         // evaluate the XML fragment with a wrapper including namespaces.
295         String postlog = "</wrapper>";
296         StringBuffer prologb = new StringBuffer("<wrapper");
297         for (int i = 0; i < namespaces.length; ++i)
298         {
299             prologb.append(" xmlns:");
300             prologb.append(namespaces[i].getPrefix());
301             prologb.append("=\"");
302             prologb.append(namespaces[i].getURI());
303             prologb.append("\"");
304         }
305         prologb.append(">");
306         String prolog = prologb.toString();
307         pe = qdcProps.propertyNames();
308         while (pe.hasMoreElements())
309         {
310             String qdc = (String)pe.nextElement();
311             String val = qdcProps.getProperty(qdc);
312             try
313             {
314                 Document d = builder.build(new StringReader(prolog+val+postlog));
315                 Element element = (Element)d.getRootElement().getContent(0);
316                 qdc2element.put(qdc, element);
317                 element2qdc.put(makeQualifiedTagName(element), qdc);
318                 log.debug("Building Maps: qdc=\""+qdc+"\", element=\""+element.toString()+"\"");
319             }
320             catch (org.jdom.JDOMException je)
321             {
322                 throw new CrosswalkInternalException("Failed parsing XML fragment in properties file: \""+prolog+val+postlog+"\": "+je.toString());
323             }
324         }
325     }

361     private List disseminateListInternal(DSpaceObject dso, boolean addSchema)
362         throws CrosswalkException,
363                IOException, SQLException, AuthorizeException
364     {
365         if (dso.getType() != Constants.ITEM)
366             throw new CrosswalkObjectNotSupported("QDCCrosswalk can only crosswalk an Item.");
367         Item item = (Item)dso;
368         init();
369 
370         DCValue[] dc = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
371         List result = new ArrayList(dc.length);
372         for (int i = 0; i < dc.length; i++)
373         {
374             // Compose qualified DC name - schema.element[.qualifier]
375             // e.g. "dc.title", "dc.subject.lcc", "lom.Classification.Keyword"
376             String qdc = dc[i].schema+"."+
377                          ((dc[i].qualifier == null) ? dc[i].element
378                             : (dc[i].element + "." + dc[i].qualifier));
379 
380             Element elt = (Element)qdc2element.get(qdc);
381 
382             // only complain about missing elements in the DC schema:
383             if (elt == null)
384             {
385                 if (dc[i].schema.equals(MetadataSchema.DC_SCHEMA))
386                     log.warn("WARNING: "+myName+": No QDC mapping for \"" + qdc+"\"");
387             }
388             else
389             {
390                 Element qe = (Element)elt.clone();
391                 qe.setText(dc[i].value);
392                 if (addSchema && schemaLocation != null)
393                     qe.setAttribute("schemaLocation", schemaLocation, XSI_NS);
394                 if (dc[i].language != null)
395                     qe.setAttribute("lang", dc[i].language, Namespace.XML_NAMESPACE);
396                 result.add(qe);
397             }
398         }
399         return result;
400     }

402     public Element disseminateElement(DSpaceObject dso)
403         throws CrosswalkException,
404                IOException, SQLException, AuthorizeException
405     {
406         init();
407         Element root = new Element("qualifieddc", DCTERMS_NS);

419     public void ingest(Context context, DSpaceObject dso, Element root)
420         throws CrosswalkException, IOException, SQLException, AuthorizeException
421     {
422         init();
423 
424         // NOTE:  don't bother comparing namespace on root element
425         // because DCMI doesn't specify one, and every app uses its
426         // own..  just give up in the face of this madness and accept
427         // anything with the right name.
428         if (!(root.getName().equals("qualifieddc")))
429             throw new MetadataValidationException("Wrong root element for Qualified DC: "+root.toString());
430         ingest(context, dso, root.getChildren());
431     }

433     public void ingest(Context context, DSpaceObject dso, List ml)
434         throws CrosswalkException, IOException, SQLException, AuthorizeException
435     {
436         init();
437 
438         // for now, forget about any targets but item.
439         if (dso.getType() != Constants.ITEM)
440             throw new CrosswalkInternalException("Wrong target object type, QDCCrosswalk can only crosswalk to an Item.");
441 
442         Item item = (Item)dso;
443 
444         Iterator mi = ml.iterator();
445         while (mi.hasNext())
446         {
447             Element me = (Element)mi.next();
448             String key = makeQualifiedTagName(me);
449 
450             // if the root element gets passed here, recurse:
451             if (me.getName().equals("qualifieddc"))
452                 ingest(context, dso, me.getChildren());
453 
454             else if (element2qdc.containsKey(key))
455             {
456                 String qdc[] = ((String)element2qdc.get(key)).split("\\.");
457 
458                 // get language - prefer xml:lang, accept lang.
459                 String lang = me.getAttributeValue("lang", Namespace.XML_NAMESPACE);
460                 if (lang == null)
461                     lang = me.getAttributeValue("lang");
462 
463                 if (qdc.length == 3)
464                     item.addMetadata(qdc[0], qdc[1], qdc[2], lang, me.getText());
465                 else if (qdc.length == 2)
466                     item.addMetadata(qdc[0], qdc[1], null,   lang, me.getText());
467                 else
468                     throw new CrosswalkInternalException("Unrecognized format in QDC element identifier for key=\""+key+"\", qdc=\""+(String)element2qdc.get(key)+"\"");
469             }
470             else
471                 log.warn("WARNING: "+myName+": No mapping for Element=\"" + key+"\" to qdc.");
472         }
473     }
