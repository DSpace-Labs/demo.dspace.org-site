72  /**
73   * Configurable MODS Crosswalk
74   * <p>
75   * This class supports multiple dissemination crosswalks from DSpace
76   * internal data to the MODS XML format
77   *  (see <a href="http://www.loc.gov/standards/mods/">http://www.loc.gov/standards/mods/</a>.)
78   * <p>
79   * It registers multiple Plugin names, which it reads from
80   * the DSpace configuration as follows:
81   *
82   * <h3>Configuration</h3>
83   * Every key starting with <code>"crosswalk.mods.properties."</code> describes a
84   * MODS crosswalk.  Everything after the last period is the <em>plugin name</em>,
85   * and the value is the pathname (relative to <code><em>dspace.dir</em>/config</code>)
86   * of the crosswalk configuration file.
87   * <p>
88   * You can have two names point to the same crosswalk,
89   * just add two configuration entries with the same value, e.g.
90   * <pre>
91   *    crosswalk.mods.properties.MODS = crosswalks/mods.properties
92   *    crosswalk.mods.properties.default = crosswalks/mods.properties
93   * </pre>
94   * The first line creates a plugin with the name <code>"MODS"</code>
95   * which is configured from the file <em>dspace-dir</em><code>/config/crosswalks/mods.properties</code>.
96   * <p>
97   * Since there is significant overhead in reading the properties file to
98   * configure the crosswalk, and a crosswalk instance may be used any number
99   * of times, we recommend caching one instance of the crosswalk for each
100  * name and simply reusing those instances.  The PluginManager does this
101  * by default.
102  *
103  * @author Larry Stone
104  * @version $Revision: 3761 $
105  */
106 public class MODSDisseminationCrosswalk extends SelfNamedPlugin
107     implements DisseminationCrosswalk

114     /**
115      * Fill in the plugin alias table from DSpace configuration entries
116      * for configuration files for flavors of MODS crosswalk:
117      */
118     private static String aliases[] = null;
119     static
120     {
121         List aliasList = new ArrayList();
122         Enumeration pe = ConfigurationManager.propertyNames();
123         while (pe.hasMoreElements())
124         {
125             String key = (String)pe.nextElement();
126             if (key.startsWith(CONFIG_PREFIX))
127                 aliasList.add(key.substring(CONFIG_PREFIX.length()));
128         }
129         aliases = (String[])aliasList.toArray(new String[aliasList.size()]);
130     }

148     /**  URL of MODS XML Schema */
149     public static final String MODS_XSD =

152     private static final String schemaLocation =
153         MODS_NS.getURI()+" "+MODS_XSD;

155     private static XMLOutputter outputUgly = new XMLOutputter();
156     private static XMLOutputter outputPretty = new XMLOutputter(Format.getPrettyFormat());
157     private static SAXBuilder builder = new SAXBuilder();

159     private HashMap modsMap = null;

161     /**
162      * Container for crosswalk mapping: expressed as "triple" of:
163      * 1. QDC field name (really field.qualifier).
164      * 2. XML subtree to add to MODS record.
165      * 3. XPath expression showing places to plug in the value.
166      */
167     static class modsTriple
168     {
169         public String qdc = null;
170         public Element xml = null;
171         public XPath xpath = null;
172 
173         /**
174          * Initialize from text versions of QDC, XML and XPath.
175          * The DC stays a string; parse the XML with appropriate
176          * namespaces; "compile" the XPath.
177          */
178         public static modsTriple create(String qdc, String xml, String xpath)
179         {
180             modsTriple result = new modsTriple();
181 
182             final String prolog = "<mods xmlns:"+MODS_NS.getPrefix()+"=\""+MODS_NS.getURI()+"\" "+
183                             "xmlns:"+XLINK_NS.getPrefix()+"=\""+XLINK_NS.getURI()+"\">";
184             final String postlog = "</mods>";
185             try
186             {
187                 result.qdc = qdc;
188                 result.xpath = XPath.newInstance(xpath);
189                 result.xpath.addNamespace(MODS_NS.getPrefix(), MODS_NS.getURI());
190                 result.xpath.addNamespace(XLINK_NS);
191                 Document d = builder.build(new StringReader(prolog+xml+postlog));
192                 result.xml = (Element)d.getRootElement().getContent(0);
193             }
194             catch (JDOMException je)
195             {
196                 log.error("Error initializing modsTriple(\""+qdc+"\",\""+xml+"\",\""+xpath+"\"): got "+je.toString());
197                 return null;
198             }
199             catch (IOException je)
200             {
201                 log.error("Error initializing modsTriple(\""+qdc+"\",\""+xml+"\",\""+xpath+"\"): got "+je.toString());
202                 return null;
203             }
204             return result;
205         }
206     }

208     /**
209      * Initialize Crosswalk table from a properties file
210      * which itself is the value of the DSpace configuration property
211      * "crosswalk.mods.properties.X", where "X" is the alias name of this instance.
212      * Each instance may be configured with a separate mapping table.
213      *
214      * The MODS crosswalk configuration properties follow the format:
215      *
216      *  {field-name} = {XML-fragment} | {XPath}
217      *
218      *  1. qualified DC field name is of the form
219      *       {MDschema}.{element}.{qualifier}
220      *
221      *      e.g.  dc.contributor.author
222      *
223      *  2. XML fragment is prototype of metadata element, with empty or "%s"
224      *     placeholders for value(s).  NOTE: Leave the %s's in becaue
225      *     it's much easier then to see if something is broken.
226      *
227      *  3. XPath expression listing point(s) in the above XML where
228      *     the value is to be inserted.  Context is the element itself.
229      *
230      * Example properties line:
231      *
232      *  dc.description.abstract = <mods:abstract>%s</mods:abstract> | text()
233      *
234      */
235     private void initMap()
236         throws CrosswalkInternalException
237     {
238         if (modsMap != null)
239             return;
240         String myAlias = getPluginInstanceName();
241         if (myAlias == null)
242         {
243             log.error("Must use PluginManager to instantiate MODSDisseminationCrosswalk so the class knows its name.");
244             return;
245         }
246         String cmPropName = CONFIG_PREFIX+myAlias;
247         String propsFilename = ConfigurationManager.getProperty(cmPropName);
248         if (propsFilename == null)
249         {
250             String msg = "MODS crosswalk missing "+
251                 "configuration file for crosswalk named \""+myAlias+"\"";
252             log.error(msg);
253             throw new CrosswalkInternalException(msg);
254         }
255         else
256         {
257             String parent = ConfigurationManager.getProperty("dspace.dir") +
258                 File.separator + "config" + File.separator;
259             File propsFile = new File(parent, propsFilename);
260             Properties modsConfig = new Properties();
261             FileInputStream pfs = null;
262             try
263             {
264                 pfs = new FileInputStream(propsFile);
265                 modsConfig.load(pfs);
266             }
267             catch (IOException e)
268             {
269                 log.error("Error opening or reading MODS properties file: "+propsFile.toString()+": "+e.toString());
270                 throw new CrosswalkInternalException("MODS crosswalk cannot "+
271                     "open config file: "+e.toString());
272             }
273             finally
274             {
275                 if (pfs != null)
276                     try { pfs.close(); } catch (IOException ioe) { }
277             }
278 
279             modsMap = new HashMap();
280             Enumeration pe = modsConfig.propertyNames();
281             while (pe.hasMoreElements())
282             {
283                 String qdc = (String)pe.nextElement();
284                 String val = modsConfig.getProperty(qdc);
285                 String pair[] = val.split("\\s+\\|\\s+", 2);
286                 if (pair.length < 2)
287                     log.warn("Illegal MODS mapping in "+propsFile.toString()+", line = "+
288                             qdc + " = " + val);
289                 else
290                 {
291                     modsTriple trip = modsTriple.create(qdc, pair[0], pair[1]);
292                     if (trip != null)
293                         modsMap.put(qdc, trip);
294                 }
295             }
296         }
297     }

319     public Element disseminateElement(DSpaceObject dso)
320         throws CrosswalkException,
321                IOException, SQLException, AuthorizeException
322     {
323         Element root = new Element("mods", MODS_NS);
324         root.setAttribute("schemaLocation", schemaLocation, XSI_NS);
325         root.addContent(disseminateListInternal(dso,false));
326         return root;

329     private List disseminateListInternal(DSpaceObject dso, boolean addSchema)
330         throws CrosswalkException,
331                IOException, SQLException, AuthorizeException
332     {
333         if (dso.getType() != Constants.ITEM)
334             throw new CrosswalkObjectNotSupported("MODSDisseminationCrosswalk can only crosswalk an Item.");
335         Item item = (Item)dso;
336         initMap();
337 
338         DCValue[] dc = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
339         List result = new ArrayList(dc.length);
340         for (int i = 0; i < dc.length; i++)
341         {
342             // Compose qualified DC name - schema.element[.qualifier]
343             // e.g. "dc.title", "dc.subject.lcc", "lom.Classification.Keyword"
344             String qdc = dc[i].schema+"."+
345                          ((dc[i].qualifier == null) ? dc[i].element
346                             : (dc[i].element + "." + dc[i].qualifier));
347 
348             modsTriple trip = (modsTriple)modsMap.get(qdc);
349             if (trip == null)
350                 log.warn("WARNING: "+getPluginInstanceName()+": No MODS mapping for \"" + qdc+"\"");
351             else
352             {
353                 try
354                 {
355                     Element me = (Element)trip.xml.clone();
356                     if (addSchema)
357                         me.setAttribute("schemaLocation", schemaLocation, XSI_NS);
358                     Iterator ni = trip.xpath.selectNodes(me).iterator();
359                     if (!ni.hasNext())
360                         log.warn("XPath \""+trip.xpath.getXPath()+
361                           "\" found no elements in \""+
362                           outputUgly.outputString(me)+
363                           "\", qdc="+qdc);
364                     while (ni.hasNext())
365                     {
366                         Object what = ni.next();
367                         if (what instanceof Element)
368                             ((Element)what).setText(dc[i].value);
369                         else if (what instanceof Attribute)
370                             ((Attribute)what).setValue(dc[i].value);
371                         else if (what instanceof Text)
372                             ((Text)what).setText(dc[i].value);
373                         else
374                             log.warn("Got unknown object from XPath, class="+what.getClass().getName());
375                     }
376                     result.add(me);
377                 }
378                 catch (JDOMException je)
379                 {
380                     log.error("Error following XPath in modsTriple: context="+
381                         outputUgly.outputString(trip.xml)+
382                         ", xpath="+trip.xpath.getXPath()+", exception="+
383                         je.toString());
384                 }
385             }
386         }
387         return result;
388     }
