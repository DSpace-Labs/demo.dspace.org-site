246     /**
247      * Creates a new metadata field.
248      *
249      * @param context
250      *            DSpace context object
251      * @throws IOException
252      * @throws AuthorizeException
253      * @throws SQLException
254      * @throws NonUniqueMetadataException
255      */
256     public void create(Context context) throws IOException, AuthorizeException,
257             SQLException, NonUniqueMetadataException
258     {
259         // Check authorisation: Only admins may create DC types
260         if (!AuthorizeManager.isAdmin(context))
261         {
262             throw new AuthorizeException(
263                     "Only administrators may modify the metadata registry");
264         }
265 
266         // Ensure the element and qualifier are unique within a given schema.
267         if (!unique(context, schemaID, element, qualifier))
268         {
269             throw new NonUniqueMetadataException("Please make " + element + "."
270                     + qualifier + " unique within schema #" + schemaID);
271         }
272 
273         // Create a table row and update it with the values
274         row = DatabaseManager.row("MetadataFieldRegistry");
275         row.setColumn("metadata_schema_id", schemaID);
276         row.setColumn("element", element);
277         row.setColumn("qualifier", qualifier);
278         row.setColumn("scope_note", scopeNote);
279         DatabaseManager.insert(context, row);
280         decache();
281 
282         // Remember the new row number
283         this.fieldID = row.getIntColumn("metadata_field_id");
284 
285         log.info(LogManager.getHeader(context, "create_metadata_field",
286                 "metadata_field_id=" + row.getIntColumn("metadata_field_id")));
287     }

289     /**
290      * Retrieves the metadata field from the database.
291      *
292      * @param context dspace context
293      * @param schemaID schema by ID
294      * @param element element name
295      * @param qualifier qualifier (may be ANY or null)
296      * @return recalled metadata field
297      * @throws SQLException
298      * @throws AuthorizeException
299      */
300     public static MetadataField findByElement(Context context, int schemaID,
301             String element, String qualifier) throws SQLException,
302             AuthorizeException
303     {
304         // Grab rows from DB
305         TableRowIterator tri;
306         if (qualifier == null)
307         {
308         	tri = DatabaseManager.queryTable(context,"MetadataFieldRegistry",
309                     "SELECT * FROM MetadataFieldRegistry WHERE metadata_schema_id= ? " + 
310                     "AND element= ?  AND qualifier is NULL ",
311                     schemaID, element);
312         } 
313         else
314         {
315         	tri = DatabaseManager.queryTable(context,"MetadataFieldRegistry",
316                     "SELECT * FROM MetadataFieldRegistry WHERE metadata_schema_id= ? " + 
317                     "AND element= ?  AND qualifier= ? ",
318                     schemaID, element, qualifier);
319         }
320         
321         TableRow row = null;
322         try
323         {
324             if (tri.hasNext())
325             {
326                 row = tri.next();
327             }
328         }
329         finally
330         {
331             // close the TableRowIterator to free up resources
332             if (tri != null)
333                 tri.close();
334         }
335 
336         if (row == null)
337         {
338             return null;
339         }
340         else
341         {
342             return new MetadataField(row);
343         }
344     }

419     /**
420      * Update the metadata field in the database.
421      *
422      * @param context dspace context
423      * @throws SQLException
424      * @throws AuthorizeException
425      * @throws NonUniqueMetadataException
426      * @throws IOException
427      */
428     public void update(Context context) throws SQLException,
429             AuthorizeException, NonUniqueMetadataException, IOException
430     {
431         // Check authorisation: Only admins may update the metadata registry
432         if (!AuthorizeManager.isAdmin(context))
433         {
434             throw new AuthorizeException(
435                     "Only administrators may modiffy the Dublin Core registry");
436         }
437 
438         // Check to see if the schema ID was altered. If is was then we will
439         // query to ensure that there is not already a duplicate name field.
440         if (row.getIntColumn("metadata_schema_id") != schemaID)
441         {
442             if (MetadataField.hasElement(context, schemaID, element, qualifier))
443             {
444                 throw new NonUniqueMetadataException(
445                         "Duplcate field name found in target schema");
446             }
447         }
448 
449         // Ensure the element and qualifier are unique within a given schema.
450         if (!unique(context, schemaID, element, qualifier))
451         {
452             throw new NonUniqueMetadataException("Please make " + element + "."
453                     + qualifier);
454         }
455 
456         row.setColumn("metadata_schema_id", schemaID);
457         row.setColumn("element", element);
458         row.setColumn("qualifier", qualifier);
459         row.setColumn("scope_note", scopeNote);
460         DatabaseManager.update(context, row);
461         decache();
462 
463         log.info(LogManager.getHeader(context, "update_metadatafieldregistry",
464                 "metadata_field_id=" + getFieldID() + "element=" + getElement()
465                         + "qualifier=" + getQualifier()));
466     }

511     /**
512      * A sanity check that ensures a given element and qualifier are unique
513      * within a given schema. The check happens in code as we cannot use a
514      * database constraint.
515      *
516      * @param context dspace context
517      * @param schemaID
518      * @param element
519      * @param qualifier
520      * @return true if unique
521      * @throws AuthorizeException
522      * @throws SQLException
523      * @throws IOException
524      */
525     private boolean unique(Context context, int schemaID, String element,
526             String qualifier) throws IOException, SQLException,
527             AuthorizeException
528     {
529         int count = 0;
530         Connection con = null;
531         PreparedStatement statement = null;
532         ResultSet rs = null;
533 
534         try
535         {
536             con = context.getDBConnection();
537             TableRow reg = DatabaseManager.row("MetadataFieldRegistry");
538 
539             String qualifierClause = "";
540 
541             if (qualifier == null)
542             {
543                 qualifierClause = "and qualifier is null";
544             }
545             else
546             {
547                 qualifierClause = "and qualifier = ?";
548             }
549 
550             String query = "SELECT COUNT(*) FROM " + reg.getTable()
551                 + " WHERE metadata_schema_id= ? "
552                 + " and metadata_field_id != ? "
553                 + " and element= ? " + qualifierClause;
554 
555             statement = con.prepareStatement(query);
556             statement.setInt(1,schemaID);
557             statement.setInt(2,fieldID);
558             statement.setString(3,element);
559 
560             if (qualifier != null)
561             {
562                 statement.setString(4,qualifier);
563             }
564 
565             rs = statement.executeQuery();
566 
567             if (rs.next())
568             {
569                 count = rs.getInt(1);
570             }
571         }
572         finally
573         {
574             if (rs != null)
575             {
576                 try { rs.close(); } catch (SQLException sqle) { }
577             }
578 
579             if (statement != null)
580             {
581                 try { statement.close(); } catch (SQLException sqle) { }
582             }
583         }
584 
585         return (count == 0);
586     }

608     /**
609      * Find the field corresponding to the given numeric ID.  The ID is
610      * a database key internal to DSpace.
611      *
612      * @param context
613      *            context, in case we need to read it in from DB
614      * @param id
615      *            the metadata field ID
616      * @return the metadata field object
617      * @throws SQLException
618      */
619     public static MetadataField find(Context context, int id)
620             throws SQLException
621     {
622         initCache(context);
623 
624         // 'sanity check' first.
625         Integer iid = new Integer(id);
626         if (!id2field.containsKey(iid))
627             return null;
628 
629         return (MetadataField) id2field.get(iid);
630     }

632     // invalidate the cache e.g. after something modifies DB state.
633     private static void decache()
634     {
635         id2field = null;
636     }
637 
638     // load caches if necessary
639     private static void initCache(Context context) throws SQLException
640     {
641         if (id2field != null)
642             return;
643 
644         synchronized (MetadataField.class)
645         {
646             if (id2field == null)
647             {
648                 HashMap new_id2field = new HashMap();
649                 log.info("Loading MetadataField elements into cache.");
650 
651                 // Grab rows from DB
652                 TableRowIterator tri = DatabaseManager.queryTable(context,"MetadataFieldRegistry",
653                         "SELECT * from MetadataFieldRegistry");
654 
655                 try
656                 {
657                     while (tri.hasNext())
658                     {
659                         TableRow row = tri.next();
660                         int fieldID = row.getIntColumn("metadata_field_id");
661                         new_id2field.put(new Integer(fieldID), new MetadataField(row));
662                     }
663                 }
664                 finally
665                 {
666                     // close the TableRowIterator to free up resources
667                     if (tri != null)
668                         tri.close();
669                 }
670 
671                 id2field = new_id2field;
672             }
673         }
