52  /**
53   * Database access class representing a Dublin Core metadata value.
54   * It represents a value of a given <code>MetadataField</code> on an Item.
55   * (The Item can have many values of the same field.)  It contains                                           element, qualifier, value and language.
56   * the field (which names the schema, element, and qualifer), language,
57   * and a value.

322     {
323         // Grab rows from DB
324         TableRowIterator tri = DatabaseManager.queryTable(context, "MetadataValue",
325                 "SELECT * FROM MetadataValue where metadata_value_id= ? ",
326                 valueId);
327 
328         TableRow row = null;
329         try
330         {
331             if (tri.hasNext())
332             {
333                 row = tri.next();
334             }
335         }
336         finally
337         {
338             // close the TableRowIterator to free up resources
339             if (tri != null)
340                 tri.close();
341         }
342 
343         if (row == null)
344         {
345             return null;
346         }
347         else
348         {
349             return new MetadataValue(row);
350         }
351     }

353     /**
354      * Retrieves the metadata values for a given field from the database.
355      *
356      * @param context dspace context
357      * @param fieldId field whose values to look for
358      * @return a collection of metadata values
359      * @throws IOException
360      * @throws SQLException
361      * @throws AuthorizeException
362      */
363     public static java.util.Collection findByField(Context context, int fieldId)
364             throws IOException, SQLException, AuthorizeException
365     {
366         // Grab rows from DB
367         TableRowIterator tri = DatabaseManager.queryTable(context, "MetadataValue",
368                 "SELECT * FROM MetadataValue WHERE metadata_field_id= ? ",
369                 fieldId);
370 
371         TableRow row = null;
372         java.util.Collection ret = new ArrayList();
373         try
374         {
375             while (tri.hasNext())
376             {
377                 row = tri.next();
378                 ret.add(new MetadataValue(row));
379             }
380         }
381         finally
382         {
383             // close the TableRowIterator to free up resources
384             if (tri != null)
385                 tri.close();
386         }
387 
388         return ret;
389     }
