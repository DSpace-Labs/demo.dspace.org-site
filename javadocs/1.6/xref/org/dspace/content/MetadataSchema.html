194     /**
195      * Creates a new metadata schema in the database, out of this object.
196      *
197      * @param context
198      *            DSpace context object
199      * @throws SQLException
200      * @throws AuthorizeException
201      * @throws NonUniqueMetadataException
202      */
203     public void create(Context context) throws SQLException,
204             AuthorizeException, NonUniqueMetadataException
205     {
206         // Check authorisation: Only admins may create metadata schemas
207         if (!AuthorizeManager.isAdmin(context))
208         {
209             throw new AuthorizeException(
210                     "Only administrators may modify the metadata registry");
211         }
212 
213         // Ensure the schema name is unique
214         if (!uniqueShortName(context, name))
215         {
216             throw new NonUniqueMetadataException("Please make the name " + name
217                     + " unique");
218         }
219         
220         // Ensure the schema namespace is unique
221         if (!uniqueNamespace(context, namespace))
222         {
223             throw new NonUniqueMetadataException("Please make the namespace " + namespace
224                     + " unique");
225         }
226 
227 
228         // Create a table row and update it with the values
229         row = DatabaseManager.create(context, "MetadataSchemaRegistry");
230         row.setColumn("namespace", namespace);
231         row.setColumn("short_id", name);
232         DatabaseManager.update(context, row);
233 
234         // invalidate our fast-find cache.
235         decache();
236 
237         // Remember the new row number
238         this.schemaID = row.getIntColumn("metadata_schema_id");
239 
240         log
241                 .info(LogManager.getHeader(context, "create_metadata_schema",
242                         "metadata_schema_id="
243                                 + row.getIntColumn("metadata_schema_id")));
244     }

359     public static MetadataSchema[] findAll(Context context) throws SQLException
360     {
361         List schemas = new ArrayList();
362 
363         // Get all the metadataschema rows
364         TableRowIterator tri = DatabaseManager.queryTable(context, "MetadataSchemaRegistry",
365                         "SELECT * FROM MetadataSchemaRegistry ORDER BY metadata_schema_id");
366 
367         try
368         {
369             // Make into DC Type objects
370             while (tri.hasNext())
371             {
372                 schemas.add(new MetadataSchema(tri.next()));
373             }
374         }
375         finally

377             // close the TableRowIterator to free up resources
378             if (tri != null)
379                 tri.close();
380         }
381 
382         // Convert list into an array
383         MetadataSchema[] typeArray = new MetadataSchema[schemas.size()];
384         return (MetadataSchema[]) schemas.toArray(typeArray);

387     /**
388      * Return true if and only if the passed name appears within the allowed
389      * number of times in the current schema.
390      *
391      * @param context DSpace context
392      * @param namespace namespace URI to match
393      * @return true of false
394      * @throws SQLException
395      */
396     private boolean uniqueNamespace(Context context, String namespace)
397             throws SQLException
398     {
399         int count = 0;
400         Connection con = context.getDBConnection();
401         PreparedStatement statement = null;
402         ResultSet rs = null;
403 
404         try
405         {
406             TableRow reg = DatabaseManager.row("MetadataSchemaRegistry");
407 
408             String query = "SELECT COUNT(*) FROM " + reg.getTable() + " " +
409                     "WHERE metadata_schema_id != ? " +
410                     "AND namespace= ? ";
411 
412             statement = con.prepareStatement(query);
413             statement.setInt(1,schemaID);
414             statement.setString(2,namespace);
415 
416             rs = statement.executeQuery();
417 
418             if (rs.next())
419             {
420                 count = rs.getInt(1);
421             }
422         }
423         finally
424         {
425             if (rs != null)
426             {
427                 try { rs.close(); } catch (SQLException sqle) { }
428             }
429 
430             if (statement != null)
431             {
432                 try { statement.close(); } catch (SQLException sqle) { }
433             }
434         }
435 
436         return (count == 0);
437     }

490     /**
491      * Get the schema corresponding with this numeric ID.
492      * The ID is a database key internal to DSpace.
493      *
494      * @param context
495      *            context, in case we need to read it in from DB
496      * @param id
497      *            the schema ID
498      * @return the metadata schema object
499      * @throws SQLException
500      */
501     public static MetadataSchema find(Context context, int id)
502             throws SQLException
503     {
504         initCache(context);
505         Integer iid = new Integer(id);
506 
507         // sanity check
508         if (!id2schema.containsKey(iid))
509             return null;
510 
511         return (MetadataSchema) id2schema.get(iid);
512     }

514     /**
515      * Get the schema corresponding with this short name.
516      *
517      * @param context
518      *            context, in case we need to read it in from DB
519      * @param shortName
520      *            the short name for the schema
521      * @return the metadata schema object
522      * @throws SQLException
523      */
524     public static MetadataSchema find(Context context, String shortName)
525         throws SQLException
526     {
527         // If we are not passed a valid schema name then return
528         if (shortName == null)
529             return null;

532 
533         if (!name2schema.containsKey(shortName))
534             return null;
535 
536         return (MetadataSchema) name2schema.get(shortName);
537     }

539     // invalidate the cache e.g. after something modifies DB state.
540     private static void decache()
541     {
542         id2schema = null;
543         name2schema = null;
544     }
545 
546     // load caches if necessary
547     private static void initCache(Context context) throws SQLException
548     {
549         if (id2schema != null && name2schema != null)
550             return;
551 
552         synchronized (MetadataSchema.class)
553         {
554             if (id2schema == null && name2schema == null)
555             {
556                 log.info("Loading schema cache for fast finds");
557                 HashMap new_id2schema = new HashMap();
558                 HashMap new_name2schema = new HashMap();
559 
560                 TableRowIterator tri = DatabaseManager.queryTable(context,"MetadataSchemaRegistry",
561                         "SELECT * from MetadataSchemaRegistry");
562 
563                 try
564                 {
565                     while (tri.hasNext())
566                     {
567                         TableRow row = tri.next();
568 
569                         MetadataSchema s = new MetadataSchema(row);
570                         new_id2schema.put(new Integer(s.schemaID), s);
571                         new_name2schema.put(s.name, s);
572                     }
573                 }
574                 finally
575                 {
576                     // close the TableRowIterator to free up resources
577                     if (tri != null)
578                         tri.close();
579                 }
580 
581                 id2schema = new_id2schema;
582                 name2schema = new_name2schema;
583             }
584         }
585     }
