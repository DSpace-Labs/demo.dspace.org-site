89      /**
90       * Private constructor for creating a Bitstream object based on the contents
91       * of a DB table row.
92       * 
93       * @param context
94       *            the context this object exists in
95       * @param row
96       *            the corresponding row in the table
97       * @throws SQLException
98       */
99      Bitstream(Context context, TableRow row) throws SQLException
100     {
101         bContext = context;
102         bRow = row;
103 
104         // Get the bitstream format
105         bitstreamFormat = BitstreamFormat.find(context, row
106                 .getIntColumn("bitstream_format_id"));
107 
108         if (bitstreamFormat == null)
109         {
110             // No format: use "Unknown"
111             bitstreamFormat = BitstreamFormat.findUnknown(context);
112 
113             // Panic if we can't find it
114             if (bitstreamFormat == null)
115             {
116                 throw new IllegalStateException("No Unknown bitsream format");
117             }
118         }
119 
120         // Cache ourselves
121         context.cache(this, row.getIntColumn("bitstream_id"));
122 
123         modified = modifiedMetadata = false;
124         clearDetails();
125     }

127     /**
128      * Get a bitstream from the database. The bitstream metadata is loaded into
129      * memory.
130      * 
131      * @param context
132      *            DSpace context object
133      * @param id
134      *            ID of the bitstream
135      * 
136      * @return the bitstream, or null if the ID is invalid.
137      * @throws SQLException
138      */
139     public static Bitstream find(Context context, int id) throws SQLException
140     {
141         // First check the cache
142         Bitstream fromCache = (Bitstream) context
143                 .fromCache(Bitstream.class, id);
144 
145         if (fromCache != null)
146         {
147             return fromCache;
148         }
149 
150         TableRow row = DatabaseManager.find(context, "bitstream", id);
151 
152         if (row == null)
153         {
154             if (log.isDebugEnabled())
155             {
156                 log.debug(LogManager.getHeader(context, "find_bitstream",
157                         "not_found,bitstream_id=" + id));
158             }
159 
160             return null;
161         }
162 
163         // not null, return Bitstream
164         if (log.isDebugEnabled())
165         {
166             log.debug(LogManager.getHeader(context, "find_bitstream",
167                     "bitstream_id=" + id));
168         }
169 
170         return new Bitstream(context, row);
171     }

173     /**
174      * Create a new bitstream, with a new ID. The checksum and file size are
175      * calculated. This method is not public, and does not check authorisation;
176      * other methods such as Bundle.createBitstream() will check authorisation.
177      * The newly created bitstream has the "unknown" format.
178      * 
179      * @param context
180      *            DSpace context object
181      * @param is
182      *            the bits to put in the bitstream
183      * 
184      * @return the newly created bitstream
185      * @throws IOException
186      * @throws SQLException
187      */
188     static Bitstream create(Context context, InputStream is)
189             throws IOException, SQLException
190     {
191         // Store the bits
192         int bitstreamID = BitstreamStorageManager.store(context, is);
193 
194         log.info(LogManager.getHeader(context, "create_bitstream",
195                 "bitstream_id=" + bitstreamID));
196 
197         // Set the format to "unknown"
198         Bitstream bitstream = find(context, bitstreamID);
199         bitstream.setFormat(null);
200 
201         context.addEvent(new Event(Event.CREATE, Constants.BITSTREAM, bitstreamID, null));
202 
203         return bitstream;
204     }

207      * Register a new bitstream, with a new ID.  The checksum and file size
208      * are calculated.  This method is not public, and does not check
209      * authorisation; other methods such as Bundle.createBitstream() will
210      * check authorisation.  The newly created bitstream has the "unknown"
211      * format.
212      *
213      * @param  context DSpace context object
214      * @param assetstore corresponds to an assetstore in dspace.cfg
215      * @param bitstreamPath the path and filename relative to the assetstore 
216      * @return  the newly registered bitstream
217      * @throws IOException
218      * @throws SQLException
219      */
220     static Bitstream register(Context context, 
221     		int assetstore, String bitstreamPath)
222         	throws IOException, SQLException
223     {
224         // Store the bits
225         int bitstreamID = BitstreamStorageManager.register(
226         		context, assetstore, bitstreamPath);
227 
228         log.info(LogManager.getHeader(context,
229             "create_bitstream",
230             "bitstream_id=" + bitstreamID));
231 
232         // Set the format to "unknown"
233         Bitstream bitstream = find(context, bitstreamID);
234         bitstream.setFormat(null);
235 
236         context.addEvent(new Event(Event.CREATE, Constants.BITSTREAM, bitstreamID, "REGISTER"));
237 
238         return bitstream;
239     }

476         modified = true;

479     /**
480      * Update the bitstream metadata. Note that the content of the bitstream
481      * cannot be changed - for that you need to create a new bitstream.
482      * 
483      * @throws SQLException
484      * @throws AuthorizeException
485      */
486     public void update() throws SQLException, AuthorizeException
487     {
488         // Check authorisation
489         AuthorizeManager.authorizeAction(bContext, this, Constants.WRITE);
490 
491         log.info(LogManager.getHeader(bContext, "update_bitstream",
492                 "bitstream_id=" + getID()));
493 
494         if (modified)
495         {
496             bContext.addEvent(new Event(Event.MODIFY, Constants.BITSTREAM, getID(), null));
497             modified = false;
498         }
499         if (modifiedMetadata)
500         {
501             bContext.addEvent(new Event(Event.MODIFY_METADATA, Constants.BITSTREAM, getID(), getDetails()));
502             modifiedMetadata = false;
503             clearDetails();
504         }
505 
506         DatabaseManager.update(bContext, bRow);
507     }

509     /**
510      * Delete the bitstream, including any mappings to bundles
511      * 
512      * @throws SQLException
513      */
514     void delete() throws SQLException
515     {
516         boolean oracle = false;
517         if ("oracle".equals(ConfigurationManager.getProperty("db.name")))
518         {
519             oracle = true;
520         }
521 
522         // changed to a check on remove
523         // Check authorisation
524         //AuthorizeManager.authorizeAction(bContext, this, Constants.DELETE);
525         log.info(LogManager.getHeader(bContext, "delete_bitstream",
526                 "bitstream_id=" + getID()));
527 
528         bContext.addEvent(new Event(Event.DELETE, Constants.BITSTREAM, getID(), String.valueOf(getSequenceID())));
529 
530         // Remove from cache
531         bContext.removeCached(this, getID());
532 
533         // Remove policies
534         AuthorizeManager.removeAllPolicies(bContext, this);
535 
536         // Remove references to primary bitstreams in bundle
537         String query = "update bundle set primary_bitstream_id = ";
538         query += (oracle ? "''" : "Null") + " where primary_bitstream_id = ? ";
539         DatabaseManager.updateQuery(bContext,
540                 query, bRow.getIntColumn("bitstream_id"));
541 
542         // Remove bitstream itself
543         BitstreamStorageManager.delete(bContext, bRow
544                 .getIntColumn("bitstream_id"));
545     }

565     /**
566      * Get the bundles this bitstream appears in
567      * 
568      * @return array of <code>Bundle</code> s this bitstream appears in
569      * @throws SQLException
570      */
571     public Bundle[] getBundles() throws SQLException
572     {
573         // Get the bundle table rows
574         TableRowIterator tri = DatabaseManager.queryTable(bContext, "bundle",
575                 "SELECT bundle.* FROM bundle, bundle2bitstream WHERE " + 
576                 "bundle.bundle_id=bundle2bitstream.bundle_id AND " +
577                 "bundle2bitstream.bitstream_id= ? ",
578                  bRow.getIntColumn("bitstream_id"));
579 
580         // Build a list of Bundle objects
581         List<Bundle> bundles = new ArrayList<Bundle>();
582         try
583         {
584             while (tri.hasNext())
585             {
586                 TableRow r = tri.next();
587 
588                 // First check the cache
589                 Bundle fromCache = (Bundle) bContext.fromCache(Bundle.class, r
590                         .getIntColumn("bundle_id"));
591 
592                 if (fromCache != null)
593                 {
594                     bundles.add(fromCache);
595                 }
596                 else
597                 {
598                     bundles.add(new Bundle(bContext, r));
599                 }
600             }
601         }
602         finally
603         {
604             // close the TableRowIterator to free up resources
605             if (tri != null)
606                 tri.close();
607         }
608 
609         Bundle[] bundleArray = new Bundle[bundles.size()];
610         bundleArray = (Bundle[]) bundles.toArray(bundleArray);
611 
612         return bundleArray;
613     }

644     public DSpaceObject getParentObject() throws SQLException
645     {
646         Bundle[] bundles = getBundles();
647         if (bundles != null && (bundles.length > 0 && bundles[0] != null))
648         {
649             // the ADMIN action is not allowed on Bundle object so skip to the item
650             Item[] items = bundles[0].getItems();
651             if (items != null && items.length > 0)
652             {
653                 return items[0];
654             }
655             else
656             {
657                 return null;
658             }
659         }
660         else
661         {
662             // is the bitstream a logo for a community or a collection?
663             TableRow qResult = DatabaseManager.querySingle(bContext,
664                        "SELECT collection_id FROM collection " +
665                        "WHERE logo_bitstream_id = ?",getID());
666             if (qResult != null) 
667             {
668                 Collection collection = Collection.find(bContext,qResult.getIntColumn("collection_id"));
669                 return collection;
670             }
671             else
672             {   
673                 // is the bitstream related to a community?
674                 qResult = DatabaseManager.querySingle(bContext,
675                         "SELECT community_id FROM community " +
676                         "WHERE logo_bitstream_id = ?",getID());
677     
678                 if (qResult != null)
679                 {
680                     Community community = Community.find(bContext,qResult.getIntColumn("community_id"));
681                     return community;
682                 }
683                 else
684                 {
685                     return null;
686                 }
687             }                                   
688         }
