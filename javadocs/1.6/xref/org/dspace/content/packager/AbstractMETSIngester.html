72  /**
73   * Base class for package ingester of
74   * METS (Metadata Encoding & Transmission Standard) Package.<br>
75   *   See <a href="http://www.loc.gov/standards/mets/">http://www.loc.gov/standards/mets/</a>
76   * <p>
77   * This is a generic packager framework intended to be subclassed to create
78   * ingesters for more specific METS "profiles".   METS is an
79   * abstract and flexible framework that can encompass many
80   * different kinds of metadata and inner package structures.
81   * <p>
82   * <b>Configuration:</b>
83   * If the property <code>mets.submission.preserveManifest</code> is <em>true</em>,
84   * the METS manifest itself is preserved in a bitstream named
85   * <code>mets.xml</code> in the <code>METADATA</code> bundle.  If it is
86   * <em>false</em> (the default), the manifest is discarded after ingestion.
87   *
88   * @author Larry Stone
89   * @version $Revision: 3761 $
90   * @see org.dspace.content.packager.METSManifest
91   */
92  public abstract class AbstractMETSIngester
93         implements PackageIngester
94  {

96      private static Logger log = Logger.getLogger(AbstractMETSIngester.class);

99      public static final String MANIFEST_FILE = "mets.xml";

102     private static final String MANIFEST_BITSTREAM_FORMAT =
103             "DSpace METS SIP";
104

106     private static final boolean preserveManifest =
107         ConfigurationManager.getBooleanProperty("mets.submission.preserveManifest", false);
108

110     private static final boolean useTemplate =
111         ConfigurationManager.getBooleanProperty("mets.submission.useCollectionTemplate", false);

113     /**
114      * An instance of MdrefManager holds the state needed to
115      * retrieve the contents (or bitstream corresponding to) an
116      * external metadata stream referenced by an <code>mdRef</code>
117      * element in the METS manifest.
118      * <p>
119      * Initialize it with the DSpace Bundle containing all of the
120      * metadata bitstreams.  Match an mdRef by finding the bitstream
121      * with the same name.
122      */
123     protected class MdrefManager
124         implements METSManifest.Mdref
125     {
126     private Bundle mdBundle = null;
127 
128         // constructor initializes metadata bundle.
129         private MdrefManager(Bundle mdBundle)
130         {
131             super();
132             this.mdBundle = mdBundle;
133         }
134 
135         /**
136          * Find the local Bitstream referenced in
137          * an <code>mdRef</code> element.
138          * @param mdref the METS mdRef element to locate the bitstream for.
139          * @return bitstream or null if none found.
140          */
141         public Bitstream getBitstreamForMdRef(Element mdref)
142             throws MetadataValidationException, IOException, SQLException, AuthorizeException
143         {
144             String path = METSManifest.getFileName(mdref);
145             if (mdBundle == null)
146                 throw new MetadataValidationException("Failed referencing mdRef element, because there were no metadata files.");
147             return mdBundle.getBitstreamByName(path);
148         }
149          
150         /**
151          * Make the contents of an external resource mentioned in
152          * an <code>mdRef</code> element available as an <code>InputStream</code>.
153          * See the <code>METSManifest.MdRef</code> interface for details.
154          * @param mdref the METS mdRef element to locate the input for.
155          * @return the input stream of its content.
156          */
157         public InputStream getInputStream(Element mdref)
158             throws MetadataValidationException, IOException, SQLException, AuthorizeException
159         {
160             Bitstream mdbs = getBitstreamForMdRef(mdref);
161             if (mdbs == null)
162                 throw new MetadataValidationException("Failed dereferencing bitstream for mdRef element="+mdref.toString());
163             return mdbs.retrieve();
164         }
165     }

167     /**
168      * Create a new DSpace item out of a METS content package.
169      * All contents are dictated by the METS manifest.
170      * Package is a ZIP archive, all files relative to top level
171      * and the manifest (as per spec) in mets.xml.
172      *
173      * @param context - DSpace context.
174      * @param collection - collection under which to create new item.
175      * @param pkg - input stream containing package to ingest.
176      * @param license - may be null, which takes default license.
177      * @return workspace item created by ingest.
178      * @throws PackageValidationException if package is unacceptable or there is
179      *  a fatal error turning it into an Item.
180      */
181     public WorkspaceItem ingest(Context context, Collection collection,
182                                 InputStream pkg, PackageParameters params,
183                                 String license)
184         throws PackageValidationException, CrosswalkException,
185                AuthorizeException, SQLException, IOException
186     {
187         ZipInputStream zip = new ZipInputStream(pkg);
188         HashMap fileIdToBitstream = new HashMap();
189         WorkspaceItem wi = null;
190         boolean success = false;
191         HashSet packageFiles = new HashSet();
192 
193         boolean validate = params.getBooleanProperty("validate", true);
194 
195         try
196         {
197             /* 1. Read all the files in the Zip into bitstreams first,
198              *  because we only get to take one pass through a Zip input
199              *  stream.  Give them temporary bitstream names corresponding
200              *  to the same names they had in the Zip, since those MUST
201              *  match the URL references in <Flocat> and <mdRef> elements.
202              */
203             METSManifest manifest = null;
204             wi = WorkspaceItem.create(context, collection, useTemplate);
205             Item item = wi.getItem();
206             Bundle contentBundle = item.createBundle(Constants.CONTENT_BUNDLE_NAME);
207             Bundle mdBundle = null;
208             ZipEntry ze;
209             while ((ze = zip.getNextEntry()) != null)
210             {
211                 if (ze.isDirectory())
212                     continue;
213                 Bitstream bs = null;
214                 String fname = ze.getName();
215                 if (fname.equals(MANIFEST_FILE))
216                 {
217                     if (preserveManifest)
218                     {
219                         mdBundle = item.createBundle(Constants.METADATA_BUNDLE_NAME);
220                         bs = mdBundle.createBitstream(new PackageUtils.UnclosableInputStream(zip));
221                         bs.setName(fname);
222                         bs.setSource(fname);
223 
224                         // Get magic bitstream format to identify manifest.
225                         BitstreamFormat manifestFormat = null;
226                         manifestFormat = PackageUtils.findOrCreateBitstreamFormat(context,
227                              MANIFEST_BITSTREAM_FORMAT, "application/xml",
228                              MANIFEST_BITSTREAM_FORMAT+" package manifest");
229                         bs.setFormat(manifestFormat);
230 
231                         manifest = METSManifest.create(bs.retrieve(), validate);
232                     }
233                     else
234                     {
235                         manifest = METSManifest.create(new PackageUtils.UnclosableInputStream(zip), validate);
236                         continue;
237                     }
238                 }
239                 else
240                 {
241                     bs = contentBundle.createBitstream(new PackageUtils.UnclosableInputStream(zip));
242                     bs.setSource(fname);
243                     bs.setName(fname);
244                 }
245                 packageFiles.add(fname);
246                 bs.setSource(fname);
247                 bs.update();
248             }
249             zip.close();
250 
251             if (manifest == null)
252                 throw new PackageValidationException("No METS Manifest found (filename="+MANIFEST_FILE+").  Package is unacceptable.");
253 
254             // initial sanity checks on manifest (in subclass)
255             checkManifest(manifest);
256 
257             /* 2. Grovel a file list out of METS Manifest and compare
258              *  it to the files in package, as an integrity test.
259              */
260             List manifestContentFiles = manifest.getContentFiles();
261 
262             // Compare manifest files with the ones found in package:
263             //  a. Start with content files (mentioned in <fileGrp>s)
264             HashSet missingFiles = new HashSet();
265             for (Iterator mi = manifestContentFiles.iterator(); mi.hasNext(); )
266             {
267                 // First locate corresponding Bitstream and make
268                 // map of Bitstream to <file> ID.
269                 Element mfile = (Element)mi.next();
270                 String mfileId = mfile.getAttributeValue("ID");
271                 if (mfileId == null)
272                     throw new PackageValidationException("Invalid METS Manifest: file element without ID attribute.");
273                 String path = METSManifest.getFileName(mfile);
274                 Bitstream bs = contentBundle.getBitstreamByName(path);
275                 if (bs == null)
276                 {
277                     log.warn("Cannot find bitstream for filename=\""+path+
278                              "\", skipping it..may cause problems later.");
279                     missingFiles.add(path);
280                 }
281                 else
282                 {
283                     fileIdToBitstream.put(mfileId, bs);
284 
285                     // Now that we're done using Name to match to <file>,
286                     // set default bitstream Name to last path element;
287                     // Zip entries all have '/' pathname separators
288                     // NOTE: set default here, hopefully crosswalk of
289                     // a bitstream techMD section will override it.
290                     String fname = bs.getName();
291                     int lastSlash = fname.lastIndexOf('/');
292                     if (lastSlash >= 0  && lastSlash+1 < fname.length())
293                         bs.setName(fname.substring(lastSlash+1));
294 
295                     // Set Default bitstream format:
296                     //  1. attempt to guess from MIME type
297                     //  2. if that fails, guess from "name" extension.
298                     String mimeType = mfile.getAttributeValue("MIMETYPE");
299                     BitstreamFormat bf = (mimeType == null) ? null :
300                             BitstreamFormat.findByMIMEType(context, mimeType);
301                     if (bf == null)
302                         bf = FormatIdentifier.guessFormat(context, bs);
303                     bs.setFormat(bf);
304 
305                     // if this bitstream belongs in another Bundle, move it:
306                     String bundleName = manifest.getBundleName(mfile);
307                     if (!bundleName.equals(Constants.CONTENT_BUNDLE_NAME))
308                     {
309                         Bundle bn;
310                         Bundle bns[] = item.getBundles(bundleName);
311                         if (bns != null && bns.length > 0)
312                             bn = bns[0];
313                         else
314                             bn = item.createBundle(bundleName);
315                         bn.addBitstream(bs);
316                         contentBundle.removeBitstream(bs);
317                     }
318 
319                     // finally, build compare lists by deleting matches.
320                     if (packageFiles.contains(path))
321                         packageFiles.remove(path);
322                     else
323                         missingFiles.add(path);
324                 }
325             }
326 
327             //  b. Process files mentioned in <mdRef>s - check and move
328             //     to METADATA bundle.
329             for (Iterator mi = manifest.getMdFiles().iterator(); mi.hasNext(); )
330             {
331                 Element mdref = (Element)mi.next();
332                 String path = METSManifest.getFileName(mdref);
333 
334                 // finally, build compare lists by deleting matches.
335                 if (packageFiles.contains(path))
336                     packageFiles.remove(path);
337                 else
338                     missingFiles.add(path);
339 
340                 // if there is a bitstream with that name in Content, move
341                 // it to the Metadata bundle:
342                 Bitstream mdbs = contentBundle.getBitstreamByName(path);
343                 if (mdbs != null)
344                 {
345                     if (mdBundle == null)
346                         mdBundle = item.createBundle(Constants.METADATA_BUNDLE_NAME);
347                     mdBundle.addBitstream(mdbs);
348                     contentBundle.removeBitstream(mdbs);
349                 }
350             }
351 
352             // KLUDGE: make sure Manifest file doesn't get flagged as missing
353             // or extra, since it won't be mentioned in the manifest.
354             if (packageFiles.contains(MANIFEST_FILE))
355                 packageFiles.remove(MANIFEST_FILE);
356 
357             // Give subclass a chance to refine the lists of in-package
358             // and missing files, delete extraneous files, etc.
359             checkPackageFiles(packageFiles, missingFiles, manifest);
360 
361             // Any discrepency in file lists is a fatal error:
362             if (!(packageFiles.isEmpty() && missingFiles.isEmpty()))
363             {
364                 StringBuffer msg = new StringBuffer("Package is unacceptable: contents do not match manifest.");
365                 if (!missingFiles.isEmpty())
366                 {
367                     msg.append("\nPackage is missing these files listed in Manifest:");
368                     for (Iterator mi = missingFiles.iterator(); mi.hasNext(); )
369                         msg.append("\n\t"+(String)mi.next());
370                 }
371                 if (!packageFiles.isEmpty())
372                 {
373                     msg.append("\nPackage contains extra files NOT in manifest:");
374                     for (Iterator mi = packageFiles.iterator(); mi.hasNext(); )
375                         msg.append("\n\t"+(String)mi.next());
376                 }
377                 throw new PackageValidationException(msg.toString());
378             }
379 
380             /* 3. crosswalk the metadata
381              */
382             // get mdref'd streams from "callback" object.
383             MdrefManager callback = new MdrefManager(mdBundle);
384 
385             chooseItemDmd(context, item, manifest, callback, manifest.getItemDmds());
386 
387             // crosswalk content bitstreams too.
388             for (Iterator ei = fileIdToBitstream.entrySet().iterator();
389                  ei.hasNext();)
390             {
391                 Map.Entry ee = (Map.Entry)ei.next();
392                 manifest.crosswalkBitstream(context, (Bitstream)ee.getValue(),
393                                         (String)ee.getKey(), callback);
394             }
395 
396             // Take a second pass over files to correct names of derived files
397             // (e.g. thumbnails, extracted text) to what DSpace expects:
398             for (Iterator mi = manifestContentFiles.iterator(); mi.hasNext(); )
399             {
400                 Element mfile = (Element)mi.next();
401                 String bundleName = manifest.getBundleName(mfile);
402                 if (!bundleName.equals(Constants.CONTENT_BUNDLE_NAME))
403                 {
404                     Element origFile = manifest.getOriginalFile(mfile);
405                     if (origFile != null)
406                     {
407                         String ofileId = origFile.getAttributeValue("ID");
408                         Bitstream obs = (Bitstream)fileIdToBitstream.get(ofileId);
409                         String newName = makeDerivedFilename(bundleName, obs.getName());
410                         if (newName != null)
411                         {
412                             String mfileId = mfile.getAttributeValue("ID");
413                             Bitstream bs = (Bitstream)fileIdToBitstream.get(mfileId);
414                             bs.setName(newName);
415                             bs.update();
416                         }
417                     }
418                 }
419             }
420 
421             // Sanity-check the resulting metadata on the Item:
422             PackageUtils.checkMetadata(item);
423 
424             /* 4. Set primary bitstream; same Bundle
425              */
426             Element pbsFile = manifest.getPrimaryBitstream();
427             if (pbsFile != null)
428             {
429                 Bitstream pbs = (Bitstream)fileIdToBitstream.get(pbsFile.getAttributeValue("ID"));
430                 if (pbs == null)
431                     log.error("Got Primary Bitstream file ID="+pbsFile.getAttributeValue("ID")+
432                              ", but found no corresponding bitstream.");
433                 else
434                 {
435                     Bundle bn[] = pbs.getBundles();
436                     if (bn.length > 0)
437                         bn[0].setPrimaryBitstreamID(pbs.getID());
438                     else
439                         log.error("Sanity check, got primary bitstream without any parent bundle.");
440                 }
441             }
442 
443             // have subclass manage license since it may be extra package file.
444             addLicense(context, collection, item, manifest, callback, license );
445 
446             // subclass hook for final checks and rearrangements
447             finishItem(context, item);
448 
449             // commit any changes to bundles
450             Bundle allBn[] = item.getBundles();
451             for (int i = 0; i < allBn.length; ++i)
452             {
453                 allBn[i].update();
454             }
455 
456             wi.update();
457             success = true;
458             log.info(LogManager.getHeader(context, "ingest",
459                 "Created new Item, db ID="+String.valueOf(item.getID())+
460                 ", WorkspaceItem ID="+String.valueOf(wi.getID())));
461             return wi;
462         }
463         catch (SQLException se)
464         {
465             // disable attempt to delete the workspace object, since
466             // database may have suffered a fatal error and the
467             // transaction rollback will get rid of it anyway.
468             wi = null;
469 
470             // Pass this exception on to the next handler.
471             throw se;
472         }
473         finally
474         {
475             // kill item (which also deletes bundles, bitstreams) if ingest fails
476             if (!success && wi != null)
477                 wi.deleteAll();
478         }
479     }

492     // return name of derived file as if MediaFilter created it, or null
493     private String makeDerivedFilename(String bundleName, String origName)
494     {
495         // get the MediaFilter that would create this bundle:
496         String mfNames[] = PluginManager.getAllPluginNames(MediaFilter.class);
497 
498         for (int i = 0; i < mfNames.length; ++i)
499         {
500             MediaFilter mf = (MediaFilter)PluginManager.getNamedPlugin(MediaFilter.class, mfNames[i]);
501             if (bundleName.equals(mf.getBundleName()))
502                 return mf.getFilteredName(origName);
503         }
504         return null;
505     }

516     /**
517      * Hook for subclass to modify the test of the package's
518      * integrity, and add other tests. E.g. evaluate a PGP signature of
519      * the manifest in a separate file.
520      * <p>
521      * The <code>packageFiles</code> contains "extra" files that were in
522      * the package but were not referenced by the METS manifest (either as
523      * content or metadata (mdRefs)).
524      * The implementation of this method should look for any "extra" files
525      * uses (e.g. a checksum or cryptographic signature for the manifest
526      * itself) and remove them from the Set.
527      * <p>
528      * The <code>missingFiles</code> set is for
529      * any files
530      * referenced by the manifest but not found in the package.
531      * The implementation can check it for "false positives", or add
532      * other missing files it knows of.
533      * <p>
534      * If either  of the Sets <code>missingFiles</code>
535      * or <code>packageFiles</code>
536      * is not empty, the ingest will fail.
537      *
538      * @param packageFiles files in package but not referenced by METS
539      * @param missingFiles files referenced by manifest but not in package
540      *
541      */
542     abstract public void checkPackageFiles(Set packageFiles, Set missingFiles,
543                                            METSManifest manifest)
544         throws PackageValidationException, CrosswalkException;

546     /**
547      * Select the <code>dmdSec</code> element(s) to apply to the
548      * Item.  The implementation is responsible for choosing which
549      * (if any) of the metadata sections to crosswalk to get the
550      * descriptive metadata for the item being ingested.  It is
551      * responsible for calling the crosswalk, using the manifest's helper
552      * i.e. <code>manifest.crosswalkItem(context,item,dmdElement,callback);</code>
553      * (The final argument is a reference to itself since the
554      * class also implements the <code>METSManifest.MdRef</code> interface
555      * to fetch package files referenced by mdRef elements.)
556      * <p>
557      * Note that <code>item</code> and <code>manifest</code> are available
558      * as protected fields from the superclass.
559      *
560      * @param context the DSpace context
561      * @param dmds array of Elements, each a METS dmdSec that applies to the Item as a whole.
562      *
563      */
564     abstract public void chooseItemDmd(Context context, Item item,
565                                        METSManifest manifest, MdrefManager cb,
566                                        Element dmds[])
