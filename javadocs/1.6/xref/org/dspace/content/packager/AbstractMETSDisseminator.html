99  /**
100  * Base class for disseminator of
101  * METS (Metadata Encoding & Transmission Standard) Package.<br>
102  *   See <a href="http://www.loc.gov/standards/mets/">http://www.loc.gov/standards/mets/</a>
103  * <p>
104  * This is a generic packager framework intended to be subclassed to create
105  * packagers for more specific METS "profiles".   METS is an
106  * abstract and flexible framework that can encompass many
107  * different kinds of metadata and inner package structures.
108  * <p>
109  * <b>Package Parameters:</b><br>
110  * <code>manifestOnly</code> -- if true, generate a standalone XML
111  * document of the METS manifest instead of a complete package.  Any
112  * other metadata (such as licenses) will be encoded inline.
113  * Default is <code>false</code>.
114  *
115  *   <code>unauthorized</code> -- this determines what is done when the
116  *   packager encounters a Bundle or Bitstream it is not authorized to
117  *   read.  By default, it just quits with an AuthorizeException.
118  *   If this option is present, it must be one of the following values:
119  *     <code>skip</code> -- simply exclude unreadable content from package.
120  *     <code>zero</code> -- include unreadable bitstreams as 0-length files;
121  *       unreadable Bundles will still cause authorize errors.
122  *
123  * @author Larry Stone
124  * @author Robert Tansley
125  * @version $Revision: 3761 $
126  */
127 public abstract class AbstractMETSDisseminator
128     implements PackageDisseminator
129 {
130     /** log4j category */
131     private static Logger log = Logger.getLogger(AbstractMETSDisseminator.class);

150     /**
151      * Make a new unique ID with specified prefix.
152      * @param prefix the prefix of the identifier, constrained to XML ID schema
153      * @return a new string identifier unique in this session (instance).
154      */
155     protected String gensym(String prefix)
156     {
157         return prefix + "_" + String.valueOf(idCounter++);
158     }

166     /**
167      * Export the object (Item, Collection, or Community) to a
168      * package file on the indicated OutputStream.
169      * Gets an exception of the object cannot be packaged or there is
170      * a failure creating the package.
171      *
172      * @param context - DSpace context.
173      * @param dso - DSpace object (item, collection, etc)
174      * @param pkg - output stream on which to write package
175      * @throws PackageException if package cannot be created or there is
176      *  a fatal error in creating it.
177      */
178     public void disseminate(Context context, DSpaceObject dso,
179                             PackageParameters params, OutputStream pkg)
180         throws PackageValidationException, CrosswalkException, AuthorizeException, SQLException, IOException
181     {
182         if (dso.getType() == Constants.ITEM)
183         {
184             Item item = (Item)dso;
185             long lmTime = item.getLastModified().getTime();
186 
187             // how to handle unauthorized bundle/bitstream:
188             String unauth = (params == null) ? null : params.getProperty("unauthorized");
189 
190             if (params != null && params.getProperty("manifestOnly") != null)
191             {
192                 extraFiles = null;
193                 writeManifest(context, item, params, pkg);
194             }
195             else
196             {
197                 extraFiles = new HashMap();
198                 ZipOutputStream zip = new ZipOutputStream(pkg);
199                 zip.setComment("METS archive created by DSpace METSDisseminationCrosswalk");
200 
201                 // write manifest first.
202                 ZipEntry me = new ZipEntry(MANIFEST_FILE);
203                 me.setTime(lmTime);
204                 zip.putNextEntry(me);
205                 writeManifest(context, item, params, zip);
206                 zip.closeEntry();
207                 
208                 // copy extra (meta?) bitstreams into zip
209                 Iterator fi = extraFiles.keySet().iterator();
210                 while (fi.hasNext())
211                 {
212                     String fname = (String)fi.next();
213                     ZipEntry ze = new ZipEntry(fname);
214                     ze.setTime(lmTime);
215                     zip.putNextEntry(ze);
216                     Utils.copy((InputStream)extraFiles.get(fname), zip);
217                     zip.closeEntry();
218                 }
219 
220                 // copy all non-meta bitstreams into zip
221                 Bundle bundles[] = item.getBundles();
222                 for (int i = 0; i < bundles.length; i++)
223                 {
224                     if (!PackageUtils.isMetaInfoBundle(bundles[i]))
225                     {
226                         // unauthorized bundle?
227                         if (!AuthorizeManager.authorizeActionBoolean(context,
228                                     bundles[i], Constants.READ))
229                         {
230                             if (unauth != null &&
231                                 (unauth.equalsIgnoreCase("skip")))
232                             {
233                                 log.warn("Skipping Bundle[\""+bundles[i].getName()+"\"] because you are not authorized to read it.");
234                                 continue;
235                             }
236                             else
237                                 throw new AuthorizeException("Not authorized to read Bundle named \""+bundles[i].getName()+"\"");
238                         }
239                         Bitstream[] bitstreams = bundles[i].getBitstreams();
240                         for (int k = 0; k < bitstreams.length; k++)
241                         {
242                             boolean auth = AuthorizeManager.authorizeActionBoolean(context,
243                                     bitstreams[k], Constants.READ);
244                             if (auth ||
245                                 (unauth != null && unauth.equalsIgnoreCase("zero")))
246                             {
247                                 ZipEntry ze = new ZipEntry(
248                                     makeBitstreamName(bitstreams[k]));
249                                 ze.setTime(lmTime);
250                                 ze.setSize(auth ? bitstreams[k].getSize() : 0);
251                                 zip.putNextEntry(ze);
252                                 if (auth)
253                                 Utils.copy(bitstreams[k].retrieve(), zip);
254                                 else
255                                     log.warn("Adding zero-length file for Bitstream, SID="+String.valueOf(bitstreams[k].getSequenceID())+", not authorized for READ.");
256                                 zip.closeEntry();
257                             }
258                             else if (unauth != null &&
259                                      unauth.equalsIgnoreCase("skip"))
260                             {
261                                 log.warn("Skipping Bitstream, SID="+String.valueOf(bitstreams[k].getSequenceID())+", not authorized for READ.");
262                             }
263                             else
264                             {
265                                 throw new AuthorizeException("Not authorized to read Bitstream, SID="+String.valueOf(bitstreams[k].getSequenceID()));
266                             }
267                         }
268                     }
269                 }
270                 zip.close();
271                 extraFiles = null;
272             }
273 
274         }
275         else
276             throw new PackageValidationException("Can only disseminate an Item now.");
277     }

279     /**
280      * Create name that bitstream will have in archive.  Name must
281      * be unique and relative to archive top level, e.g. "bitstream_<id>.ext"
282      */
283     private String makeBitstreamName(Bitstream bitstream)
284     {
285         String base = "bitstream_"+String.valueOf(bitstream.getID());
286         String ext[] = bitstream.getFormat().getExtensions();
287         return (ext.length > 0) ? base+"."+ext[0] : base;
288     }

291     // set metadata type - if Mdtype.parse() gets exception,
292     // that means it's not in the MDTYPE vocabulary, so use OTHER.
293     private void setMdType(MdWrap mdWrap, String mdtype)
294     {
295         try
296         {
297             mdWrap.setMDTYPE(Mdtype.parse(mdtype));
298         }
299         catch (MetsException e)
300         {
301             mdWrap.setMDTYPE(Mdtype.OTHER);
302             mdWrap.setOTHERMDTYPE(mdtype);
303         }
304     }

306     /**
307      * Write out a METS manifest.
308      * Mostly lifted from Rob Tansley's METS exporter.
309      */
310     private void writeManifest(Context context, Item item,
311                                PackageParameters params, OutputStream out)
312         throws PackageValidationException, CrosswalkException, AuthorizeException, SQLException, IOException
313 
314     {
315         try
316         {
317             // Create the METS file
318             Mets mets = new Mets();
319          
320             // Top-level stuff
321             mets.setID(gensym("mets"));
322             mets.setOBJID("hdl:" + item.getHandle());
323             mets.setLABEL("DSpace Item");
324             mets.setPROFILE(getProfile());
325          
326             // MetsHdr
327             MetsHdr metsHdr = new MetsHdr();
328             metsHdr.setCREATEDATE(new Date()); // FIXME: CREATEDATE is now:
329                                                // maybe should be item create
330             // date?
331 
332             // Agent
333             Agent agent = new Agent();
334             agent.setROLE(Role.CUSTODIAN);
335             agent.setTYPE(Type.ORGANIZATION);
336             Name name = new Name();
337             name.getContent()
338                     .add(new PCData(ConfigurationManager
339                                     .getProperty("dspace.name")));
340             agent.getContent().add(name);
341             metsHdr.getContent().add(agent);
342             mets.getContent().add(metsHdr);
343          
344             // add DMD sections
345             // Each type element MAY be either just a MODS-and-crosswalk name, OR
346             // a combination "MODS-name:crosswalk-name" (e.g. "DC:qDC").
347             String dmdTypes[] = getDmdTypes(params);
348 
349             // record of ID of each dmdsec to make DMDID in structmap.
350             String dmdGroup = gensym("dmd_group");
351             String dmdId[] = new String[dmdTypes.length];
352             for (int i = 0; i < dmdTypes.length; ++i)
353             {
354                 dmdId[i] = gensym("dmd");
355                 XmlData xmlData = new XmlData();
356                 String xwalkName, metsName;
357                 String parts[] = dmdTypes[i].split(":", 2);
358                 if (parts.length > 1)
359                 {
360                     metsName = parts[0];
361                     xwalkName = parts[1];
362                 }
363                 else
364                     xwalkName = metsName = dmdTypes[i];
365 
366                 DisseminationCrosswalk xwalk = (DisseminationCrosswalk)
367                   PluginManager.getNamedPlugin(DisseminationCrosswalk.class, xwalkName);
368                 if (xwalk == null)
369                     throw new PackageValidationException("Cannot find "+dmdTypes[i]+" crosswalk plugin!");
370                 else
371                     crosswalkToMets(xwalk, item, xmlData);
372 
373                 DmdSec dmdSec = new DmdSec();
374                 dmdSec.setID(dmdId[i]);
375                 dmdSec.setGROUPID(dmdGroup);
376                 MdWrap mdWrap = new MdWrap();
377                 setMdType(mdWrap, metsName);
378                 mdWrap.getContent().add(xmlData);
379                 dmdSec.getContent().add(mdWrap);
380                 mets.getContent().add(dmdSec);
381             }
382          
383             // Only add license AMD section if there are any licenses.
384             // Catch authorization failures accessing license bitstreams
385             // only if we are skipping unauthorized bitstreams.
386             String licenseID = null;
387             try
388             {
389             AmdSec amdSec = new AmdSec();
390             addRightsMd(context, item, amdSec);
391             if (amdSec.getContent().size() > 0)
392             {
393                 licenseID = gensym("license");
394                 amdSec.setID(licenseID);
395                 mets.getContent().add(amdSec);
396             }
397             }
398             catch (AuthorizeException e)
399             {
400                 String unauth = (params == null) ? null : params.getProperty("unauthorized");
401                 if (!(unauth != null && unauth.equalsIgnoreCase("skip")))
402                     throw e;
403                 else
404                     log.warn("Skipping license metadata because of access failure: "+e.toString());
405             }
406 
407             // FIXME: History data???? Nooooo!!!!
408 
409             // fileSec - all non-metadata bundles go into fileGrp,
410             // and each bitstream therein into a file.
411             // Create the bitstream-level techMd and div's for structmap
412             // at the same time so we can connec the IDREFs to IDs.
413             FileSec fileSec = new FileSec();
414          
415             String techMdType = getTechMdType(params);
416             String parts[] = techMdType.split(":", 2);
417             String xwalkName, metsName;
418             if (parts.length > 1)
419             {
420                 metsName = parts[0];
421                 xwalkName = parts[1];
422             }
423             else
424                 xwalkName = metsName = techMdType;
425 
426             DisseminationCrosswalk xwalk = (DisseminationCrosswalk)
427               PluginManager.getNamedPlugin(DisseminationCrosswalk.class, xwalkName);
428             if (xwalk == null)
429                 throw new PackageValidationException("Cannot find "+xwalkName+" crosswalk plugin!");
430 
431             // log the primary bitstream for structmap
432             String primaryBitstreamFileID = null;
433 
434             // accumulate content DIV items to put in structMap later.
435             List contentDivs = new ArrayList();
436 
437             // how to handle unauthorized bundle/bitstream:
438             String unauth = (params == null) ? null : params.getProperty("unauthorized");
439 
440             Bundle[] bundles = item.getBundles();
441             for (int i = 0; i < bundles.length; i++)
442             {
443                 if (PackageUtils.isMetaInfoBundle(bundles[i]))
444                     continue;
445 
446                 // unauthorized bundle?
447                 // NOTE: This must match the logic in disseminate()
448                 if (!AuthorizeManager.authorizeActionBoolean(context,
449                             bundles[i], Constants.READ))
450                 {
451                     if (unauth != null &&
452                         (unauth.equalsIgnoreCase("skip")))
453                         continue;
454                     else
455                         throw new AuthorizeException("Not authorized to read Bundle named \""+bundles[i].getName()+"\"");
456                 }
457 
458                 Bitstream[] bitstreams = bundles[i].getBitstreams();
459 
460                 // Create a fileGrp
461                 FileGrp fileGrp = new FileGrp();
462          
463                 // Bundle name for USE attribute
464                 String bName = bundles[i].getName();
465                 if ((bName != null) && !bName.equals(""))
466                     fileGrp.setUSE(bundleToFileGrp(bName));
467          
468                 // watch for primary bitstream
469                 int primaryBitstreamID = -1;
470                 boolean isContentBundle = false;
471                 if ((bName != null) && bName.equals("ORIGINAL"))
472                 {
473                     isContentBundle = true;
474                     primaryBitstreamID = bundles[i].getPrimaryBitstreamID();
475                 }
476 
477                 for (int bits = 0; bits < bitstreams.length; bits++)
478                 {
479                     // Check for authorization.  Handle unauthorized
480                     // bitstreams to match the logic in disseminate(),
481                     // i.e. "unauth=zero" means include a 0-length bitstream,
482                     // "unauth=skip" means to ignore it (and exclude from
483                     // manifest).
484                     boolean auth = AuthorizeManager.authorizeActionBoolean(context,
485                             bitstreams[bits], Constants.READ);
486                     if (!auth)
487                     {
488                         if (unauth != null && unauth.equalsIgnoreCase("skip"))
489                             continue;
490                         else if (!(unauth != null && unauth.equalsIgnoreCase("zero")))
491                             throw new AuthorizeException("Not authorized to read Bitstream, SID="+String.valueOf(bitstreams[bits].getSequenceID()));
492                     }
493 
494                     String sid = String.valueOf(bitstreams[bits].getSequenceID());
495          
496                     edu.harvard.hul.ois.mets.File file = new edu.harvard.hul.ois.mets.File();
497          
498                     String xmlIDstart = "bitstream_";
499                     String fileID = xmlIDstart + sid;
500 
501                     file.setID(fileID);
502 
503                     // log primary bitstream for later (structMap)
504                     if (bitstreams[bits].getID() == primaryBitstreamID)
505                         primaryBitstreamFileID = fileID;
506 
507                     // if this is content, add to structmap too:
508                     if (isContentBundle)
509                     {
510                         Div div = new Div();
511                         div.setID(gensym("div"));
512                         div.setTYPE("DSpace Content Bitstream");
513                         Fptr fptr = new Fptr();
514                         fptr.setFILEID(fileID);
515                         div.getContent().add(fptr);
516                         contentDivs.add(div);
517                     }
518 
519                     file.setSEQ(bitstreams[bits].getSequenceID());
520          
521                     String groupID = "GROUP_" + xmlIDstart + sid;
522          
523                     /*
524                      * If we're in THUMBNAIL or TEXT bundles, the bitstream is
525                      * extracted text or a thumbnail, so we use the name to work
526                      * out which bitstream to be in the same group as
527                      */
528                     if ((bundles[i].getName() != null)
529                             && (bundles[i].getName().equals("THUMBNAIL") ||
530                                 bundles[i].getName().startsWith("TEXT")))
531                     {
532                         // Try and find the original bitstream, and chuck the
533                         // derived bitstream in the same group
534                         Bitstream original = findOriginalBitstream(item,
535                                 bitstreams[bits]);
536          
537                         if (original != null)
538                         {
539                             groupID = "GROUP_" + xmlIDstart
540                                     + original.getSequenceID();
541                         }
542                     }
543          
544                     file.setGROUPID(groupID);
545                     file.setMIMETYPE(bitstreams[bits].getFormat().getMIMEType());
546          
547                     // FIXME: CREATED: no date
548 
549                     file.setSIZE(auth ? bitstreams[bits].getSize() : 0);
550 
551                     // translate checksum and type to METS, if available.
552                     String csType = bitstreams[bits].getChecksumAlgorithm();
553                     String cs = bitstreams[bits].getChecksum();
554                     if (auth && cs != null && csType != null)
555                     {
556                         try
557                         {
558                             file.setCHECKSUMTYPE(Checksumtype.parse(csType));
559                             file.setCHECKSUM(cs);
560                         }
561                         catch (MetsException e)
562                         {
563                             log.warn("Cannot set bitstream checksum type="+csType+" in METS.");
564                         }
565                     }
566          
567                     // FLocat: filename is MD5 checksum
568                     FLocat flocat = new FLocat();
569                     flocat.setLOCTYPE(Loctype.URL);
570                     flocat.setXlinkHref(makeBitstreamName(bitstreams[bits]));
571 
572                     // Make bitstream techMD metadata, add to file.
573                     String techID = "techMd_for_bitstream_"+bitstreams[bits].getSequenceID();
574                     AmdSec fAmdSec = new AmdSec();
575                     fAmdSec.setID(techID);
576                     TechMD techMd = new TechMD();
577                     techMd.setID(gensym("tech"));
578                     MdWrap mdWrap = new MdWrap();
579                     setMdType(mdWrap, metsName);
580                     XmlData xmlData = new XmlData();
581                     mdWrap.getContent().add(xmlData);
582                     techMd.getContent().add(mdWrap);
583                     fAmdSec.getContent().add(techMd);
584                     mets.getContent().add(fAmdSec);
585                     crosswalkToMets(xwalk, bitstreams[bits], xmlData);
586                     file.setADMID(techID);
587 
588                     // Add FLocat to File, and File to FileGrp
589                     file.getContent().add(flocat);
590                     fileGrp.getContent().add(file);
591                 }
592          
593                 // Add fileGrp to fileSec
594                 fileSec.getContent().add(fileGrp);
595             }
596          
597             // Add fileSec to document
598             mets.getContent().add(fileSec);
599          
600             // Create simple structMap: initial div represents the Item,
601             // and user-visible content bitstreams are in its child divs.
602             StringBuffer dmdIds = new StringBuffer();
603             for (int i = 0; i < dmdId.length; ++i)
604                 dmdIds.append(" "+dmdId[i]);
605             StructMap structMap = new StructMap();
606             structMap.setID(gensym("struct"));
607             structMap.setTYPE("LOGICAL");
608             structMap.setLABEL("DSpace");
609             Div div0 = new Div();
610             div0.setID(gensym("div"));
611             div0.setTYPE("DSpace Item");
612             div0.setDMDID(dmdIds.substring(1));
613             if (licenseID != null)
614                 div0.setADMID(licenseID);
615 
616             // if there is a primary bitstream, add FPTR to it.
617             if (primaryBitstreamFileID != null)
618             {
619                 Fptr fptr = new Fptr();
620                 fptr.setFILEID(primaryBitstreamFileID);
621                 div0.getContent().add(fptr);
622             }
623 
624             // add DIV for each content bitstream
625             div0.getContent().addAll(contentDivs);
626 
627             structMap.getContent().add(div0);
628 
629             // Does subclass have something to add to structMap?
630             addStructMap(context, item, params, mets);
631 
632             mets.getContent().add(structMap);
633

648     /**
649      * For a bitstream that's a thumbnail or extracted text, find the
650      * corresponding bitstream it was derived from, in the ORIGINAL bundle.
651      *
652      * @param item
653      *            the item we're dealing with
654      * @param derived
655      *            the derived bitstream
656      *
657      * @return the corresponding original bitstream (or null)
658      */
659     protected static Bitstream findOriginalBitstream(Item item, Bitstream derived)
660         throws SQLException
661     {
662         Bundle[] bundles = item.getBundles();
663 
664         // Filename of original will be filename of the derived bitstream
665         // minus the extension (last 4 chars - .jpg or .txt)
666         String originalFilename = derived.getName().substring(0,
667                 derived.getName().length() - 4);
668 
669         // First find "original" bundle
670         for (int i = 0; i < bundles.length; i++)
671         {
672             if ((bundles[i].getName() != null)
673                     && bundles[i].getName().equals("ORIGINAL"))
674             {
675                 // Now find the corresponding bitstream
676                 Bitstream[] bitstreams = bundles[i].getBitstreams();
677 
678                 for (int bsnum = 0; bsnum < bitstreams.length; bsnum++)
679                 {
680                     if (bitstreams[bsnum].getName().equals(originalFilename))
681                     {
682                         return bitstreams[bsnum];
683                     }
684                 }
685             }
686         }
687 
688         // Didn't find it
689         return null;
690     }

692     // Get result from crosswalk plugin and add it to the document,
693     // including namespaces and schema.
694     private void crosswalkToMets(DisseminationCrosswalk xwalk,
695                                  DSpaceObject dso, MetsElement me)
696         throws CrosswalkException,
697                IOException, SQLException, AuthorizeException
698     {
699         // add crosswalk's namespaces and schemaLocation to this element:
700         String raw = xwalk.getSchemaLocation();
701         String sloc[] = raw == null ? null : raw.split("\\s+");
702         Namespace ns[] = xwalk.getNamespaces();
703         for (int i = 0; i < ns.length; ++i)
704         {
705             String uri = ns[i].getURI();
706             if (sloc != null && sloc.length > 1 && uri.equals(sloc[0]))
707                 me.setSchema(ns[i].getPrefix(), uri, sloc[1]);
708             else
709                 me.setSchema(ns[i].getPrefix(), uri);
710         }
711 
712         // add result of crosswalk
713         PreformedXML pXML =
714           new PreformedXML(
715             xwalk.preferList() ?
716               outputter.outputString(xwalk.disseminateList(dso)) :
717               outputter.outputString(xwalk.disseminateElement(dso)));
718         me.getContent().add(pXML);
719     }
