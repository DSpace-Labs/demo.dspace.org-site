74  /**
75   * <P>
76   * Manage the METS manifest document for METS importer classes,
77   * such as the package importer <code>org.dspace.content.packager.MetsSubmission</code>
78   * and the federated importer <code>org.dspace.app.mets.FederatedMETSImport</code>
79   * </P>
80   * <P>
81   * It can parse the METS document, build an internal model, and give the importers
82   * access to that model.  It also crosswalks
83   * all of the descriptive and administrative metadata in the METS
84   * manifest into the target DSpace Item, under control of the importer.
85   * </P>
86   *
87   * <P>
88   * It reads the following DSpace Configuration entries:
89   * </P>
90   * <UL>
91   * <LI>Local XML schema (XSD) declarations, in the general format:
92   * <br><code>mets.xsd.<em>identifier</em> = <em>namespace</em> <em>xsd-URL</em></code>
93   * <br> eg. <code>mets.xsd.dc =  http://purl.org/dc/elements/1.1/ dc.xsd</code>
94   * <br>Add a separate config entry for each schema.
95   * </LI>
96   * <p><LI>Crosswalk plugin mappings:
97   * These tell it the name of the crosswalk plugin to invoke for metadata sections
98   * with a particular value of <code>MDTYPE</code> (or <code>OTHERMDTYPE</code>)
99   * By default, the crosswalk mechanism will look for a plugin with the
100  * same name as the metadata type (e.g.  <code>"MODS"</code>,
101  * <code>"DC"</code>).  This example line invokes the <code>QDC</code>
102  * plugin when <code>MDTYPE="DC"</code>
103  * <br><code>mets.submission.crosswalk.DC = QDC </code>
104  * <br> general format is:
105  * <br><code>mets.submission.crosswalk.<em>mdType</em> = <em>pluginName</em> </code>
106  * </LI>
107  * </UL>
108  *
109  *
110  * @author Robert Tansley
111  * @author WeiHua Huang
112  * @author Rita Lee
113  * @author Larry Stone
114  * @see org.dspace.content.packager.MetsSubmission
115  * @see org.dspace.app.mets.FederatedMETSImport
116  */
117 public class METSManifest

119     /**
120      * Callback interface to retrieve data streams in mdRef elements.
121      * "Package" or file reader returns an input stream for the
122      * given relative path, e.g. to dereference <code>mdRef</code> elements.
123      */
124     public interface Mdref
125     {
126         /**
127          * Make the contents of an external resource mentioned in
128          * an <code>mdRef</code> element available as an <code>InputStream</code>.
129          * The implementation must use the information in the
130          * <code>mdRef</code> element, and the state in the object that
131          * implements this interface, to find the actual metadata content.
132          * <p>
133          * For example, an implementation that ingests a directory of
134          * files on the local filesystem would get a relative pathname
135          * out of the <code>mdRef</code> and open that file.
136          *
137          * @param mdRef JDOM element of mdRef in the METS manifest.
138          * @return stream containing the metadata mentioned in mdRef.
139          * @throw MetadataValidationException if the mdRef is unacceptable or missing required information.
140          * @throw IOException if it is returned by services called by this method.
141          * @throw SQLException if it is returned by services called by this method.
142          * @throw AuthorizeException if it is returned by services called by this method.
143          */
144         public InputStream getInputStream(Element mdRef)
145             throws MetadataValidationException, IOException, SQLException, AuthorizeException;
146     }

190     // Create list of local schemas at load time, since it depends only
191     // on the DSpace configuration.
192     private static String localSchemas;
193     static
194     {
195         String dspace_dir = ConfigurationManager.getProperty("dspace.dir");
196         File xsdPath1 = new File(dspace_dir+"/config/schemas/");
197         File xsdPath2 = new File(dspace_dir+"/config/");
198 
199         Enumeration pe = ConfigurationManager.propertyNames();
200         StringBuffer result = new StringBuffer();
201         while (pe.hasMoreElements())
202         {
203             // config lines have the format:
204             //  mets.xsd.{identifier} = {namespace} {xsd-URL}
205             // e.g.
206             //  mets.xsd.dc =  http://purl.org/dc/elements/1.1/ dc.xsd
207             // (filename is relative to {dspace_dir}/config/schemas/)
208             String key = (String)pe.nextElement();
209             if (key.startsWith(CONFIG_XSD_PREFIX))
210             {
211                 String spec  = ConfigurationManager.getProperty(key);
212                 String val[] = spec.trim().split("\\s+");
213                 if (val.length == 2)
214                 {
215                     File xsd = new File(xsdPath1, val[1]);
216                     if (!xsd.exists())
217                          xsd = new File(xsdPath2, val[1]);
218                     if (!xsd.exists())
219                         log.warn("Schema file not found for config entry=\""+spec+"\"");
220                     else
221                     {
222                         try
223                         {
224                             String u = xsd.toURL().toString();
225                             if (result.length() > 0)
226                                 result.append(" ");
227                             result.append(val[0]).append(" ").append(u);
228                         }
229                         catch (java.net.MalformedURLException e)
230                         {
231                             log.warn("Skipping badly formed XSD URL: "+e.toString());
232                         }
233                     }
234                 }
235                 else
236                     log.warn("Schema config entry has wrong format, entry=\""+spec+"\"");
237             }
238         }
239         localSchemas = result.toString();
240         log.debug("Got local schemas = \""+localSchemas+"\"");
241     }

255     /**
256      * Create a new manifest object from a serialized METS XML document.
257      * Parse document read from the input stream, optionally validating.
258      * @param is input stream containing serialized XML
259      * @param validate if true, enable XML validation using schemas
260      *   in document.  Also validates any sub-documents.
261      * @throws MetadataValidationException if there is any error parsing
262      *          or validating the METS.
263      * @return new METSManifest object.
264      */
265     public static METSManifest create(InputStream is, boolean validate)
266             throws IOException,
267             MetadataValidationException
268     {
269         SAXBuilder builder = new SAXBuilder(validate);
270 
271         // Set validation feature
272         if (validate)
273             builder.setFeature("http://apache.org/xml/features/validation/schema",
274                     true);
275 
276         // Tell the parser where local copies of schemas are, to speed up
277         // validation.  Local XSDs are identified in the configuration file.
278         if (localSchemas.length() > 0)
279             builder.setProperty(
280                     "http://apache.org/xml/properties/schema/external-schemaLocation",
281                     localSchemas);
282 
283         // Parse the METS file
284         Document metsDocument;
285 
286         try
287         {
288             metsDocument = builder.build(is);
289 
290             // XXX for temporary debugging
291             /*
292             XMLOutputter outputPretty = new XMLOutputter(Format.getPrettyFormat());
293             log.debug("Got METS DOCUMENT:");
294             log.debug(outputPretty.outputString(metsDocument));
295               */
296         }
297         catch (JDOMException je)
298         {
299             throw new MetadataValidationException("Error validating METS in "
300                     + is.toString(),  je);
301         }
302 
303         return new METSManifest(builder, metsDocument.getRootElement());
304     }

315     /**
316      * Gets all <code>file</code> elements which make up
317      *   the item's content.
318      * @return a List of <code>Element</code>s.
319      */
320     public List getContentFiles()
321         throws MetadataValidationException
322     {
323         if (contentFiles != null)
324             return contentFiles;
325 
326         Element fileSec = mets.getChild("fileSec", metsNS);
327         if (fileSec == null)
328             throw new MetadataValidationException("Invalid METS Manifest: DSpace requires a fileSec element, but it is missing.");
329 
330         contentFiles = new ArrayList();
331         Iterator fgi = fileSec.getChildren("fileGrp", metsNS).iterator();
332         while (fgi.hasNext())
333         {
334             Element fg = (Element)fgi.next();
335             Iterator fi = fg.getChildren("file", metsNS).iterator();
336             while (fi.hasNext())
337             {
338                 Element f = (Element)fi.next();
339                 contentFiles.add(f);
340             }
341         }
342         return contentFiles;
343     }

345     /**
346      * Gets list of all <code>mdRef</code> elements in the METS
347      *   document.  Used by ingester to e.g. check that all
348      *   required files are present.
349      * @return a List of <code>Element</code>s.
350      */
351     public List getMdFiles()
352         throws MetadataValidationException
353     {
354         if (mdFiles == null)
355         {
356             try
357             {
358                 // Use a special namespace with known prefix
359                 // so we get the right prefix.
360                 XPath xpath = XPath.newInstance("descendant::mets:mdRef");
361                 xpath.addNamespace(metsNS);
362                 mdFiles = xpath.selectNodes(mets);
363             }
364             catch (JDOMException je)
365             {
366                 throw new MetadataValidationException("Failed while searching for mdRef elements in manifest: ", je);
367             }
368         }
369         return mdFiles;
370     }

372     /**
373      * Get the "original" file element for a derived file.
374      * Finds the original from which this was derived by matching the GROUPID
375      * attribute that binds it to its original.  For instance, the file for
376      * a thumbnail image would have the same GROUPID as its full-size version.
377      * <p>
378      * NOTE: This pattern of relating derived files through the GROUPID
379      * attribute is peculiar to the DSpace METS SIP profile, and may not be
380      * generally useful with other sorts of METS documents.
381      * @param file METS file element of derived file
382      * @return file Element of original or null if none found.
383      */
384     public Element getOriginalFile(Element file)
385     {
386         String groupID = file.getAttributeValue("GROUPID");
387         if (groupID == null || groupID.equals(""))
388             return null;
389 
390         try
391         {
392             XPath xpath = XPath.newInstance(
393 "mets:fileSec/mets:fileGrp[@USE=\"CONTENT\"]/mets:file[@GROUPID=\""+groupID+"\"]");
394             xpath.addNamespace(metsNS);
395             List oFiles = xpath.selectNodes(mets);
396             if (oFiles.size() > 0)
397             {
398                 log.debug("Got ORIGINAL file for derived="+file.toString());
399                 return (Element)oFiles.get(0);
400             }
401             else
402                 return null;
403         }
404         catch (JDOMException je)
405         {
406             log.warn("Got exception on XPATH looking for Original file, "+je.toString());
407             return null;
408         }
409     }

437     /**
438      * Get the "local" file name of this <code>file</code> or <code>mdRef</code> element.
439      * By "local" we mean the reference to the actual resource containing
440      * the data for this file, e.g. a relative path within a Zip or tar archive
441      * if the METS is serving as a manifest for that sort of package.
442      * @return "local" file name (i.e.  relative to package or content
443      *  directory) corresponding to this <code>file</code> or <code>mdRef</code> element.
444      * @throws MetadataValidationException when there is not enough information to find a resource identifier.
445      */
446     public static String getFileName(Element file)
447         throws MetadataValidationException
448     {
449         Element ref;
450         if (file.getName().equals("file"))
451         {
452             ref = file.getChild("FLocat", metsNS);
453             if (ref == null)
454             {
455                 // check for forbidden FContent child first:
456                 if (file.getChild("FContent", metsNS) == null)
457                     throw new MetadataValidationException("Invalid METS Manifest: Every file element must have FLocat child.");
458                 else
459                     throw new MetadataValidationException("Invalid METS Manifest: file element has forbidden FContent child, only FLocat is allowed.");
460             }
461         }
462         else if (file.getName().equals("mdRef"))
463             ref = file;
464         else
465             throw new MetadataValidationException("getFileName() called with recognized element type: "+file.toString());
466         String loctype = ref.getAttributeValue("LOCTYPE");
467         if (loctype != null && loctype.equals("URL"))
468         {
469             String result = ref.getAttributeValue("href", xlinkNS);
470             if (result == null)
471                 throw new MetadataValidationException("Invalid METS Manifest: FLocat/mdRef is missing the required xlink:href attribute.");
472             return result;
473         }
474         throw new MetadataValidationException("Invalid METS Manifest: FLocat/mdRef does not have LOCTYPE=\"URL\" attribute.");
475     }

477     /**
478      * Returns file element corresponding to primary bitstream.
479      * There is <i>ONLY</i> a primary bitstream if the first <code>div</code> under
480      * first </code>structMap</code> has an </code>fptr</code>.
481      *
482      * @return file element of Item's primary bitstream, or null if there is none.
483      */
484     public Element getPrimaryBitstream()
485         throws MetadataValidationException
486     {
487         Element firstDiv = getFirstDiv();
488         Element fptr = firstDiv.getChild("fptr", metsNS);
489         if (fptr == null)
490             return null;
491         String id = fptr.getAttributeValue("FILEID");
492         if (id == null)
493             throw new MetadataValidationException("fptr for Primary Bitstream is missing the required FILEID attribute.");
494         Element result = getElementByXPath("descendant::mets:file[@ID=\""+id+"\"]", false);
495         if (result == null)
496             throw new MetadataValidationException("Cannot find file element for Primary Bitstream: looking for ID="+id);
497         return result;
498     }

500     /** Get the metadata type from within a *mdSec element.
501      * @return metadata type name.
502      */
503     public String getMdType(Element mdSec)
504         throws MetadataValidationException
505     {
506         Element md = mdSec.getChild("mdRef", metsNS);
507         if (md == null)
508             md = mdSec.getChild("mdWrap", metsNS);
509         if (md == null)
510             throw new MetadataValidationException("Invalid METS Manifest: ?mdSec element has neither mdRef nor mdWrap child.");
511         String result = md.getAttributeValue("MDTYPE");
512         if (result != null && result.equals("OTHER"))
513             result = md.getAttributeValue("OTHERMDTYPE");
514         if (result == null)
515             throw new MetadataValidationException("Invalid METS Manifest: "+md.getName()+" has no MDTYPE or OTHERMDTYPE attribute.");
516         return result;
517     }

519     /**
520      *  Returns MIME type of metadata content, if available.
521      *  @return MIMEtype word, or null if none is available.
522      */
523     public String getMdContentMimeType(Element mdSec)
524         throws MetadataValidationException
525     {
526         Element mdWrap = mdSec.getChild("mdWrap", metsNS);
527         if (mdWrap != null)
528         {
529             String mimeType = mdWrap.getAttributeValue("MIMETYPE");
530             if (mimeType == null && mdWrap.getChild("xmlData", metsNS) != null)
531             mimeType = "text/xml";
532             return mimeType;
533         }
534         Element mdRef = mdSec.getChild("mdRef", metsNS);
535         if (mdRef != null)
536             return mdRef.getAttributeValue("MIMETYPE");
537         return null;
538     }

540     /**
541      * Return contents of *md element as List of XML Element objects.
542      * Gets content, dereferecing mdRef if necessary, or decoding and parsing
543      * a binData that contains XML.
544      * @return contents of metadata section, or empty list if no XML content is available.
545      * @throws MetadataValidationException if METS is invalid, or there is an error parsing the XML.
546      */
547     public List getMdContentAsXml(Element mdSec, Mdref callback)
548         throws MetadataValidationException, IOException, SQLException, AuthorizeException
549     {
550         try
551         {
552             Element mdRef = null;
553             Element mdWrap = mdSec.getChild("mdWrap", metsNS);
554             if (mdWrap != null)
555             {
556                 Element xmlData = mdWrap.getChild("xmlData", metsNS);
557                 if (xmlData == null)
558                 {
559                     Element bin = mdWrap.getChild("binData", metsNS);
560                     if (bin == null)
561                         throw new MetadataValidationException("Invalid METS Manifest: mdWrap element with neither xmlData nor binData child.");
562 
563                     // if binData is actually XML, return it; otherwise ignore.
564                     else
565                     {
566                         String mimeType = mdWrap.getAttributeValue("MIMETYPE");
567                         if (mimeType != null && mimeType.equalsIgnoreCase("text/xml"))
568                         {
569                             byte value[] = Base64.decodeBase64(bin.getText().getBytes());
570                             Document mdd = parser.build(new ByteArrayInputStream(value));
571                             List result = new ArrayList(1);
572                             result.add(mdd.getRootElement());
573                             return result;
574                         }
575                         else
576                         {
577                             log.warn("Ignoring binData section because MIMETYPE is not XML, but: "+mimeType);
578                             return new ArrayList(0);
579                         }
580                    }
581                 }
582                 else
583                 {
584                     return xmlData.getChildren();
585                 }
586             }
587             else if ((mdRef = mdSec.getChild("mdRef", metsNS)) != null)
588             {
589                 String mimeType = mdRef.getAttributeValue("MIMETYPE");
590                 if (mimeType != null && mimeType.equalsIgnoreCase("text/xml"))
591                 {
592                     Document mdd = parser.build(callback.getInputStream(mdRef));
593                     List result = new ArrayList(1);
594                     result.add(mdd.getRootElement());
595                     return result;
596                 }
597                 else
598                 {
599                     log.warn("Ignoring mdRef section because MIMETYPE is not XML, but: "+mimeType);
600                     return new ArrayList(0);
601                 }
602             }
603             else
604                 throw new MetadataValidationException("Invalid METS Manifest: ?mdSec element with neither mdRef nor mdWrap child.");
605         }
606         catch (JDOMException je)
607         {
608             throw new MetadataValidationException("Error parsing or validating metadata section in mdRef or binData within "+mdSec.toString(), je);
609         }
610 
611     }

613     /**
614      * Return contents of *md element as stream.
615      * Gets content, dereferecing mdRef if necessary, or decoding
616      * a binData element if necessary.
617      * @return Stream containing contents of metadata section.  Never returns null.
618      * @throws MetadataValidationException if METS format does not contain any metadata.
619      */
620     public InputStream getMdContentAsStream(Element mdSec, Mdref callback)
621         throws MetadataValidationException, IOException, SQLException, AuthorizeException
622     {
623         Element mdRef = null;
624         Element mdWrap = mdSec.getChild("mdWrap", metsNS);
625         if (mdWrap != null)
626         {
627             Element xmlData = mdWrap.getChild("xmlData", metsNS);
628             if (xmlData == null)
629             {
630                 Element bin = mdWrap.getChild("binData", metsNS);
631                 if (bin == null)
632                     throw new MetadataValidationException("Invalid METS Manifest: mdWrap element with neither xmlData nor binData child.");
633 
634                 else
635                 {
636                     byte value[] = Base64.decodeBase64(bin.getText().getBytes());
637                     return new ByteArrayInputStream(value);
638                 }
639             }
640             else
641             {
642                 XMLOutputter outputPretty = new XMLOutputter(Format.getPrettyFormat());
643                 return new ByteArrayInputStream(
644                         outputPretty.outputString(xmlData.getChildren()).getBytes());
645             }
646         }
647         else if ((mdRef = mdSec.getChild("mdRef", metsNS)) != null)
648         {
649             return callback.getInputStream(mdRef);
650         }
651         else
652             throw new MetadataValidationException("Invalid METS Manifest: ?mdSec element with neither mdRef nor mdWrap child.");
653     }

735      * Gets all dmdSec elements containing metadata for the DSpace Item.
736      *
737      * @return array of Elements, each a dmdSec.  May be empty but NOT null.

739      *          DMDs in the correct place.

741     public Element[] getItemDmds()
742         throws MetadataValidationException
743     {
744         // div@DMDID is actually IDREFS, a space-separated list of IDs:
745         Element firstDiv = getFirstDiv();
746         String dmds = firstDiv.getAttributeValue("DMDID");
747         if (dmds == null)
748             throw new MetadataValidationException("Invalid METS: Missing reference to Item descriptive metadata, first div on first structmap must have a DMDID attribute.");
749         String dmdID[] = dmds.split("\\s+");
750         Element result[] = new Element[dmdID.length];
751 
752         for (int i = 0; i < dmdID.length; ++i)
753             result[i] = getElementByXPath("mets:dmdSec[@ID=\""+dmdID[i]+"\"]", false);
754         return result;

758      * Return rights metadata section(s) relevant to item as a whole.
759      * @return array of rightsMd elements, possibly empty but never null.
760      * @throws MetadataValidationException if METS is invalid, e.g. referenced amdSec is missing.
761      */
762     public Element[] getItemRightsMD()
763         throws MetadataValidationException
764     {
765         // div@ADMID is actually IDREFS, a space-separated list of IDs:
766         Element firstDiv = getFirstDiv();
767         String amds = firstDiv.getAttributeValue("ADMID");
768         if (amds == null)
769         {
770             log.debug("getItemRightsMD: No ADMID references found.");
771             return new Element[0];
772         }
773         String amdID[] = amds.split("\\s+");
774         List resultList = new ArrayList();
775         for (int i = 0; i < amdID.length; ++i)

802      * Crosswalk the metadata associated with a particular <code>file</code>
803      * element into the bitstream it corresponds to.
804      * @param context a dspace context.
805      * @param bs bitstream target of the crosswalk
806      * @param fileId value of ID attribute in the file element responsible
807      *  for the contents of that bitstream.
808      */
809     public void crosswalkBitstream(Context context, Bitstream bitstream,
810                                    String fileId, Mdref callback)
811         throws MetadataValidationException,
812                CrosswalkException, IOException, SQLException, AuthorizeException
813     {
814         Element file = getElementByXPath("descendant::mets:file[@ID=\""+fileId+"\"]", false);
815         if (file == null)
816             throw new MetadataValidationException("Failed in Bitstream crosswalk, Could not find file element with ID="+fileId);
817 
818         // In DSpace METS SIP spec, admin metadata is only "highly
819         // recommended", not "required", so it is OK if there is no ADMID.
820         String amds = file.getAttributeValue("ADMID");
821         if (amds == null)
822         {
823             log.warn("Got no bitstream ADMID, file@ID="+fileId);
824             return;
825         }
826         String amdID[] = amds.split("\\s+");
827         for (int i = 0; i < amdID.length; ++i)
828         {
829             List techMDs = getElementByXPath("mets:amdSec[@ID=\""+amdID[i]+"\"]", false).
830                                  getChildren("techMD", metsNS);
831             Iterator ti = techMDs.iterator();
832             while (ti.hasNext())
833             {
834                 Element techMD = (Element)ti.next();
835                 if (techMD != null)
836                 {
837                     String type = getMdType(techMD);
838                     IngestionCrosswalk xwalk = getCrosswalk(type);
839                     log.debug("Got bitstream techMD of type="+type+", for file ID="+fileId);
840                      
841                     if (xwalk == null)
842                         throw new MetadataValidationException("Cannot process METS Manifest: "+
843                             "No crosswalk found for techMD MDTYPE="+type);
844                     crosswalkMdContent(techMD, callback, xwalk, context, bitstream);
845                 }
846             }
847         }
848     }
