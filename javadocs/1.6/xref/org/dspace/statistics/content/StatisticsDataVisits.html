310     private void processAxis(DatasetGenerator datasetGenerator, List<DatasetQuery> queries) throws SQLException {
311         if(datasetGenerator instanceof DatasetDSpaceObjectGenerator){
312             DatasetDSpaceObjectGenerator dspaceObjAxis = (DatasetDSpaceObjectGenerator) datasetGenerator;
313             //Get the types involved
314             List<DSORepresentation> dsoRepresentations = dspaceObjAxis.getDsoRepresentations();
315             for (int i = 0; i < dsoRepresentations.size(); i++){
316                 DatasetQuery datasetQuery = new DatasetQuery();
317                 Integer dsoType = dsoRepresentations.get(i).getType();
318                 boolean seperate = dsoRepresentations.get(i).getSeparate();
319                 Integer dsoLength = dsoRepresentations.get(i).getNameLength();
320                 //Check if our type is our current object
321                 if(currentDso != null && dsoType == currentDso.getType()){
322                     Query query = new Query();
323                     query.setDso(currentDso.getID(), currentDso.getType(), dsoLength);
324                     datasetQuery.addQuery(query);
325                 }else{
326                     //TODO: only do this for bitstreams from an item
327                     Query query = new Query();
328                     if(currentDso != null && seperate && dsoType == Constants.BITSTREAM){
329                         //CURRENTLY THIS IS ONLY POSSIBLE FOR AN ITEM ! ! ! ! ! ! !
330                         //We need to get the seperate bitstreams from our item and make a query for each of them
331                         Item item = (Item) currentDso;
332                         for (int j = 0; j < item.getBundles().length; j++) {
333                             Bundle bundle = item.getBundles()[j];
334                             for (int k = 0; k < bundle.getBitstreams().length; k++) {
335                                 Bitstream bitstream = bundle.getBitstreams()[k];
336                                 if(!bitstream.getFormat().isInternal()){
337                                     //Add a seperate query for each bitstream
338                                     query.setDso(bitstream.getID(), bitstream.getType(), dsoLength);
339                                 }
340                             }
341                         }
342                     } else {
343                         //We have something else then our current object
344                         //So we need some kind of children from it, so put this in our query
345                         query.setOwningDso(currentDso);
346                         query.setDsoLength(dsoLength);
347 
348                         String title = "";
349                         switch(dsoType){
350                             case Constants.BITSTREAM:
351                                 title = "Files";
352                                 break;
353                             case Constants.ITEM:
354                                 title = "Items";
355                                 break;
356                             case Constants.COLLECTION:
357                                 title = "Collections";
358                                 break;
359                             case Constants.COMMUNITY:
360                                 title = "Communities";
361                                 break;
362                         }
363                         datasetQuery.setName(title);
364                         //Put the type in so we only get the children of the type specified
365                         query.setDsoType(dsoType);
366                     }
367                     datasetQuery.addQuery(query);
368                 }
369                 datasetQuery.setFacetField("id");
370                 datasetQuery.setMax(dsoRepresentations.get(i).getMax());
371 
372                 queries.add(datasetQuery);
373             }
374         }else
375         if(datasetGenerator instanceof DatasetTypeGenerator){
376             DatasetTypeGenerator typeAxis = (DatasetTypeGenerator) datasetGenerator;
377             DatasetQuery datasetQuery = new DatasetQuery();
378 
379             //First make sure our query is in order
380             Query query = new Query();
381             if(currentDso != null)
382                 query.setDso(currentDso.getID(), currentDso.getType());
383             datasetQuery.addQuery(query);
384 
385             //Then add the rest
386             datasetQuery.setMax(typeAxis.getMax());
387             datasetQuery.setFacetField(typeAxis.getType());
388             datasetQuery.setName(typeAxis.getType());
389 
390             queries.add(datasetQuery);
391         }
392     }

398     private String getResultName(String value, DatasetQuery datasetQuery, Context context) throws SQLException {
399         if("continent".equals(datasetQuery.getName())){
400             value = LocationUtils.getContinentName(value);
401         }else
402         if("countryCode".equals(datasetQuery.getName())){
403             value = LocationUtils.getCountryName(value);
404         }else{
405             Query query = datasetQuery.getQueries().get(0);
406             //TODO: CHANGE & THROW AWAY THIS ENTIRE METHOD
407             //Check if int
408             int dsoId;
409             int dsoLength = query.getDsoLength();
410             try {
411                 dsoId = Integer.parseInt(value);
412             }catch(Exception e){
413                 dsoId = -1;
414             }
415             if(dsoId == -1 && query.getDsoId() != -1 && value == null)
416                 dsoId = query.getDsoId();
417 
418             if(dsoId != -1 && query.dsoType != -1){
419                 DSpaceObject dso = DSpaceObject.find(context, query.getDsoType(), dsoId);
420                 if(dso != null){
421                     switch(dso.getType()){
422                         case Constants.BITSTREAM:
423                             Bitstream bit = (Bitstream) dso;
424                             return bit.getName();
425                         case Constants.ITEM:
426                             Item item = (Item) dso;
427                             String name = "untitled";
428                             DCValue[] vals = item.getMetadata("dc", "title", null, Item.ANY);
429                             if(vals != null && 0 < vals.length)
430                                 name = vals[0].value;
431                             if(dsoLength != -1 && name.length() > dsoLength){
432                                 //Cut it off at the first space
433                                 int firstSpace = name.indexOf(" ", dsoLength);
434                                 if(firstSpace != -1){
435                                     name = name.substring(0, firstSpace) + " ...";
436                                 }
437                             }
438 
439                             return name;
440 
441                         case Constants.COLLECTION:
442                             Collection coll = (Collection) dso;
443                             name = coll.getName();
444 
445                             if(dsoLength != -1 && name.length() > dsoLength){
446                                 //Cut it off at the first space
447                                 int firstSpace = name.indexOf(" ", dsoLength);
448                                 if(firstSpace != -1){
449                                     name = name.substring(0, firstSpace) + " ...";
450                                 }
451                             }
452                             return name;
453 
454                         case Constants.COMMUNITY:
455                             Community comm = (Community) dso;
456                             name = comm.getName();
457 
458                             if(dsoLength != -1 && name.length() > dsoLength){
459                                 //Cut it off at the first space
460                                 int firstSpace = name.indexOf(" ", dsoLength);
461                                 if(firstSpace != -1){
462                                     name = name.substring(0, firstSpace) + " ...";
463                                 }
464                             }
465                             return name;
466                     }
467                 }
468             }
469         }
470         return value;
471     }

473     private Map<String, String> getAttributes(String value, DatasetQuery datasetQuery, Context context) throws SQLException {
474         HashMap<String, String> attrs = new HashMap<String, String>();
475         Query query = datasetQuery.getQueries().get(0);
476         //TODO: CHANGE & THROW AWAY THIS ENTIRE METHOD
477         //Check if int
478         int dsoId;
479         try {
480             dsoId = Integer.parseInt(value);
481         }catch(Exception e){
482             dsoId = -1;
483         }
484         if(dsoId == -1 && query.getDsoId() != -1 && value == null)
485             dsoId = query.getDsoId();
486 
487         if(dsoId != -1 && query.dsoType != -1){
488             DSpaceObject dso = DSpaceObject.find(context, query.getDsoType(), dsoId);
489             if(dso != null){
490                 switch(dso.getType()){
491                     case Constants.BITSTREAM:
492                         Bitstream bit = (Bitstream) dso;
493 
494                         //Get our owning item
495                         Item owningItem = null;
496                         Bundle[] bunds = bit.getBundles();
497                         if(0 < bunds.length)
498                                 if(0 < bunds[0].getItems().length)
499                                         owningItem = bunds[0].getItems()[0];
500 
501                         // If possible refrence this bitstream via a handle, however this may
502                         // be null if a handle has not yet been assigned. In this case refrence the
503                         // item its internal id. In the last case where the bitstream is not associated
504                         // with an item (such as a community logo) then refrence the bitstreamID directly.
505                         String identifier = null;
506                         if (owningItem != null && owningItem.getHandle() != null)
507                             identifier = "handle/"+owningItem.getHandle();
508                         else if (owningItem != null)
509                             identifier = "item/"+owningItem.getID();
510                         else
511                             identifier = "id/"+bit.getID();
512 
513 
514                         String url = ConfigurationManager.getProperty("dspace.url") + "/bitstream/"+identifier+"/";
515 
516                         // If we can put the pretty name of the bitstream on the end of the URL
517                         try
518                         {
519                             if (bit.getName() != null)
520                                 url += Util.encodeBitstreamName(bit.getName(), "UTF-8");
521                         }
522                         catch (UnsupportedEncodingException uee)
523                         {
524                             // just ignore it, we don't have to have a pretty
525                             // name on the end of the url because the sequence id will
526                             // locate it. However it means that links in this file might
527                             // not work....
528                         }
529 
530                         url += "?sequence="+bit.getSequenceID();
531 
532                         attrs.put("url", url);
533                         break;
534 
535                     case Constants.ITEM:
536                         Item item = (Item) dso;
537 
538                         attrs.put("url", HandleManager.resolveToURL(context, item.getHandle()));
539                         break;
540 
541                     case Constants.COLLECTION:
542                         Collection coll = (Collection) dso;
543 
544                         attrs.put("url", HandleManager.resolveToURL(context, coll.getHandle()));
545                         break;
546 
547                     case Constants.COMMUNITY:
548                         Community comm = (Community) dso;
549 
550                         attrs.put("url", HandleManager.resolveToURL(context, comm.getHandle()));
551                         break;
552                 }
553             }
554         }
555         return attrs;
556     }
