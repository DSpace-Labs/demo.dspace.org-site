793             {"NA", "North America"},

801     public static String getCountryName(String countryCode){
802         int index = countryCodeList.indexOf(countryCode);
803         return countryNameList.get(index).toString();           
804     }

806     public static String getContinentCode(String countryCode){
807         int index = countryCodeList.indexOf(countryCode);
808         return continentCodeList.get(index).toString();
809     }

811     public static String getContinentName(String continentCode){
812 //        String continentCode = getContinentCode(countryCode);
813         for (String[] contCodeName : continentCodeToName) {
814             if (contCodeName[0].equals(continentCode))
815                 return contCodeName[1];
816         }
817         return continentCode;
818     }
