18  /**
19   * XBill DNS resolver to retrieve hostnames for client IP addresses.

21   * @author kevinvandevelde at atmire.com
22   * @author ben at atmire.com
23   */
24  public class DnsLookup {
25  
26      public static String reverseDns(String hostIp) throws IOException {
27           Record opt = null;
28           Resolver res = new ExtendedResolver();
29           res.setTimeout(0, 20);
30  
31           Name name = ReverseMap.fromAddress(hostIp);
32           int type = Type.PTR;
33           int dclass = DClass.IN;
34           Record rec = Record.newRecord(name, type, dclass);
35           Message query = Message.newQuery(rec);
36           Message response = res.send(query);
37  
38           Record[] answers = response.getSectionArray(Section.ANSWER);
39           if (answers.length == 0)
40              return hostIp;
41           else
42              return answers[0].rdataToString();
43     }
