157     public static void post(DSpaceObject dspaceObject, String ip,
158             EPerson currentUser)
159     {
160         if (solr == null || locationService == null)
161             return;
162         try
163         {
164             SolrInputDocument doc1 = new SolrInputDocument();
165             // Save our basic info that we already have
166             doc1.addField("ip", ip);
167             doc1.addField("id", dspaceObject.getID());
168             doc1.addField("type", dspaceObject.getType());
169             // Save the current time
170             doc1.addField("time", DateFormatUtils.format(new Date(),
171                     DATE_FORMAT_8601));
172             if (currentUser != null)
173                 doc1.addField("epersonid", currentUser.getID());
174 
175             try
176             {
177             	String dns = DnsLookup.reverseDns(ip);
178                 doc1.addField("dns", dns.toLowerCase());
179             }
180             catch (Exception e)
181             {
182     			log.error(e.getMessage());
183     		}
184             
185             // Save the location information if valid, save the event without
186             // location information if not valid
187             Location location = locationService.getLocation(ip);
188             if (location != null
189                     && !("--".equals(location.countryCode)
190                             && location.latitude == -180 && location.longitude == -180))
191             {
192                 try
193                 {
194                     doc1.addField("continent", LocationUtils
195                             .getContinentCode(location.countryCode));
196                 }
197                 catch (Exception e)
198                 {
199                     System.out
200                             .println("COUNTRY ERROR: " + location.countryCode);
201                 }
202                 doc1.addField("countryCode", location.countryCode);
203                 doc1.addField("city", location.city);
204                 doc1.addField("latitude", location.latitude);
205                 doc1.addField("longitude", location.longitude);
206             }
207             if (dspaceObject instanceof Item)
208             {
209                 Item item = (Item) dspaceObject;
210                 // Store the metadata
211                 for (Object storedField : metadataStorageInfo.keySet())
212                 {
213                     String dcField = (String) metadataStorageInfo
214                             .get(storedField);
215 
216                     DCValue[] vals = item.getMetadata(dcField.split("\\.")[0],
217                             dcField.split("\\.")[1], dcField.split("\\.")[2],
218                             Item.ANY);
219                     for (DCValue val1 : vals)
220                     {
221                         String val = val1.value;
222                         doc1.addField(String.valueOf(storedField), val);
223                         doc1.addField(storedField + "_search", val
224                                 .toLowerCase());
225                     }
226                 }
227             }
228 
229             storeParents(doc1, dspaceObject);
230 
231             solr.add(doc1);
232             // TODO: requires further load testing, very fast commits might cause issues
233             solr.commit(false, false);
234    
235         } catch (Exception e) {
236         	log.error(e.getMessage(), e);
237         }
238     }

245     /**
246      * Method just used to log the parents Community log: owning comms
247      * Collection log: owning comms & their comms Item log: owning colls/comms
248      * Bitstream log: owning item/colls/comms
249      * 
250      * @param doc1
251      *            the current solrinputdoc
252      * @param dso
253      *            the current dspace object we want to log
254      * @throws java.sql.SQLException
255      *             ignore it
256      */
257     public static void storeParents(SolrInputDocument doc1, DSpaceObject dso)
258             throws SQLException
259     {
260         if (dso instanceof Community)
261         {
262             Community comm = (Community) dso;
263             while (comm != null && comm.getParentCommunity() != null)
264             {
265                 doc1.addField("owningComm", comm.getID());
266                 comm = comm.getParentCommunity();
267             }
268         }
269         else if (dso instanceof Collection)
270         {
271             Collection coll = (Collection) dso;
272             for (int i = 0; i < coll.getCommunities().length; i++)
273             {
274                 Community community = coll.getCommunities()[i];
275                 doc1.addField("owningComm", community.getID());
276                 storeParents(doc1, community);
277             }
278         }
279         else if (dso instanceof Item)
280         {
281             Item item = (Item) dso;
282             for (int i = 0; i < item.getCollections().length; i++)
283             {
284                 Collection collection = item.getCollections()[i];
285                 doc1.addField("owningColl", collection.getID());
286                 storeParents(doc1, collection);
287             }
288         }
289         else if (dso instanceof Bitstream)
290         {
291             Bitstream bitstream = (Bitstream) dso;
292             for (int i = 0; i < bitstream.getBundles().length; i++)
293             {
294                 Bundle bundle = bitstream.getBundles()[i];
295                 for (int j = 0; j < bundle.getItems().length; j++)
296                 {
297                     Item item = bundle.getItems()[j];
298                     doc1.addField("owningItem", item.getID());
299                     storeParents(doc1, item);
300                 }
301             }
302         }
303     }

317     public static Map<String, List<String>> queryField(String query,
318             List oldFieldVals, String field)
319     {
320         Map<String, List<String>> currentValsStored = new HashMap<String, List<String>>();
321         try
322         {
323             // Get one document (since all the metadata for all the values
324             // should be the same just get the first one we find
325             Map<String, String> params = new HashMap<String, String>();
326             params.put("q", query);
327             params.put("rows", "1");
328             MapSolrParams solrParams = new MapSolrParams(params);
329             QueryResponse response = solr.query(solrParams);
330             // Make sure we at least got a document
331             if (response.getResults().getNumFound() == 0)
332                 return currentValsStored;
333 
334             // We have at least one document good
335             SolrDocument document = response.getResults().get(0);
336             for (Object storedField : metadataStorageInfo.keySet())
337             {
338                 // For each of these fields that are stored we are to create a
339                 // list of the values it holds now
340                 java.util.Collection collection = document
341                         .getFieldValues((String) storedField);
342                 List<String> storedVals = new ArrayList<String>();
343                 storedVals.addAll(collection);
344                 // Now add it to our hashmap
345                 currentValsStored.put((String) storedField, storedVals);
346             }
347 
348             // System.out.println("HERE");
349             // Get the info we need
350         }
351         catch (SolrServerException e)
352         {
353             e.printStackTrace();
354         }
355         return currentValsStored;
356     }

359      * //TODO: below are not used public static void
360      * update(String query, boolean addField, String fieldName, Object
361      * fieldValue, Object oldFieldValue) throws SolrServerException, IOException
362      * { List<Object> vals = new ArrayList<Object>(); vals.add(fieldValue);
363      * List<Object> oldvals = new ArrayList<Object>(); oldvals.add(fieldValue);
364      * update(query, addField, fieldName, vals, oldvals); }
365      */
366     public static void update(String query, String action,
367             List<String> fieldNames, List<List<Object>> fieldValuesList)
368             throws SolrServerException, IOException
369     {
370         long start = new Date().getTime();
371         // Since there is NO update
372         // We need to get our documents
373         // QueryResponse queryResponse = solr.query()//query(query, null, -1,
374         // null, null, null);
375         Map<String, String> params = new HashMap<String, String>();
376         params.put("q", query);
377         params.put("rows", "10");
378         MapSolrParams solrParams = new MapSolrParams(params);
379         QueryResponse response = solr.query(solrParams);
380 
381         long numbFound = response.getResults().getNumFound();
382         List<SolrDocument> docsToUpdate = new ArrayList<SolrDocument>();
383         docsToUpdate.addAll(response.getResults());
384 
385         // Run over the rest
386         for (int i = 10; i < numbFound; i += 10)
387         {
388             params.put("start", String.valueOf(i));
389             solrParams = new MapSolrParams(params);
390             response = solr.query(solrParams);
391             docsToUpdate.addAll(response.getResults());
392         }
393         // We have all the docs delete the once we don't need
394         solr.deleteByQuery(query);
395 
396         // Add the new (updated onces
397         for (int i = 0; i < docsToUpdate.size(); i++)
398         {
399             SolrDocument solrDocument = docsToUpdate.get(i);
400             // Now loop over our fieldname actions
401             for (int j = 0; j < fieldNames.size(); j++)
402             {
403                 String fieldName = fieldNames.get(j);
404                 List<Object> fieldValues = fieldValuesList.get(j);
405 
406                 if (action.equals("addOne") || action.equals("replace"))
407                 {
408                     if (action.equals("replace"))
409                         solrDocument.removeFields(fieldName);
410 
411                     for (Object fieldValue : fieldValues)
412                     {
413                         solrDocument.addField(fieldName, fieldValue);
414                     }
415                 }
416                 else if (action.equals("remOne"))
417                 {
418                     // Remove the field
419                     java.util.Collection<Object> values = solrDocument
420                             .getFieldValues(fieldName);
421                     solrDocument.removeFields(fieldName);
422                     for (Object value : values)
423                     {
424                         // Keep all the values besides the one we need to remove
425                         if (!fieldValues.contains((value)))
426                         {
427                             solrDocument.addField(fieldName, value);
428                         }
429                     }
430                 }
431             }
432             SolrInputDocument newInput = ClientUtils
433                     .toSolrInputDocument(solrDocument);
434             solr.add(newInput);
435         }
436         solr.commit();
437         // System.out.println("SolrLogger.update(\""+query+"\"):"+(new
438         // Date().getTime() - start)+"ms,"+numbFound+"records");
439     }

446     /**
447      * Query used to get values grouped by the given facetfield
448      * 
449      * @param query
450      *            the query to be used
451      * @param facetField
452      *            the facet field on which to group our values
453      * @param max
454      *            the max number of values given back (in case of 10 the top 10
455      *            will be given)
456      * @param showTotal
457      *            a boolean determening whether the total amount should be given
458      *            back as the last element of the array
459      * @return an array containing our results
460      * @throws SolrServerException
461      *             ...
462      */
463     public static ObjectCount[] queryFacetField(String query,
464             String filterQuery, String facetField, int max, boolean showTotal,
465             List<String> facetQueries) throws SolrServerException
466     {
467         QueryResponse queryResponse = query(query, filterQuery, facetField,
468                 max, null, null, null, facetQueries);
469         if (queryResponse == null)
470             return new ObjectCount[0];
471 
472         FacetField field = queryResponse.getFacetField(facetField);
473         // At least make sure we have one value
474         if (0 < field.getValueCount())
475         {
476             // Create an array for our result
477             ObjectCount[] result = new ObjectCount[field.getValueCount()
478                     + (showTotal ? 1 : 0)];
479             // Run over our results & store them
480             for (int i = 0; i < field.getValues().size(); i++)
481             {
482                 FacetField.Count fieldCount = field.getValues().get(i);
483                 result[i] = new ObjectCount();
484                 result[i].setCount(fieldCount.getCount());
485                 result[i].setValue(fieldCount.getName());
486             }
487             if (showTotal)
488             {
489                 result[result.length - 1] = new ObjectCount();
490                 result[result.length - 1].setCount(queryResponse.getResults()
491                         .getNumFound());
492                 result[result.length - 1].setValue("total");
493             }
494             return result;
495         }
496         else
497         {
498             // Return an empty array cause we got no data
499             return new ObjectCount[0];
500         }
501     }

579     private static String getDateView(String name, String type)
580     {
581         if (name != null && name.matches("^[0-9]{4}\\-[0-9]{2}.*"))
582         {
583             /*
584              * if("YEAR".equalsIgnoreCase(type)) return name.substring(0, 4);
585              * else if("MONTH".equalsIgnoreCase(type)) return name.substring(0,
586              * 7); else if("DAY".equalsIgnoreCase(type)) return
587              * name.substring(0, 10); else if("HOUR".equalsIgnoreCase(type))
588              * return name.substring(11, 13);
589              */
590             // Get our date
591             Date date = null;
592             try
593             {
594                 SimpleDateFormat format = new SimpleDateFormat(DATE_FORMAT_8601);
595                 date = format.parse(name);
596             }
597             catch (ParseException e)
598             {
599                 try
600                 {
601                     // We should use the dcdate (the dcdate is used when
602                     // generating random data)
603                     SimpleDateFormat format = new SimpleDateFormat(
604                             DATE_FORMAT_DCDATE);
605                     date = format.parse(name);
606                 }
607                 catch (ParseException e1)
608                 {
609                     e1.printStackTrace();
610                 }
611                 // e.printStackTrace();
612             }
613             String dateformatString = "dd-MM-yyyy";
614             if ("DAY".equals(type))
615             {
616                 dateformatString = "dd-MM-yyyy";
617             }
618             else if ("MONTH".equals(type))
619             {
620                 dateformatString = "MMMM yyyy";
621 
622             }
623             else if ("YEAR".equals(type))
624             {
625                 dateformatString = "yyyy";
626             }
627             SimpleDateFormat simpleFormat = new SimpleDateFormat(
628                     dateformatString);
629             if (date != null)
630                 name = simpleFormat.format(date);
631 
632         }
633         return name;
634     }

636     private static QueryResponse query(String query, String filterQuery,
637             String facetField, int max, String dateType, String dateStart,
638             String dateEnd, List<String> facetQueries)
639             throws SolrServerException
640     {
641         if (solr == null)
642             return null;
643 
644         // System.out.println("QUERY");
645         SolrQuery solrQuery = new SolrQuery().setRows(0).setQuery(query)
646                 .setFacetMinCount(1);
647 
648         // Set the date facet if present
649         if (dateType != null)
650         {
651             solrQuery.setParam("facet.date", "time")
652                     .
653                     // EXAMPLE: NOW/MONTH+1MONTH
654                     setParam("facet.date.end",
655                             "NOW/" + dateType + dateEnd + dateType).setParam(
656                             "facet.date.gap", "+1" + dateType)
657                     .
658                     // EXAMPLE: NOW/MONTH-" + nbMonths + "MONTHS
659                     setParam("facet.date.start",
660                             "NOW/" + dateType + dateStart + dateType + "S")
661                     .setFacet(true);
662         }
663         if (facetQueries != null)
664         {
665             for (int i = 0; i < facetQueries.size(); i++)
666             {
667                 String facetQuery = facetQueries.get(i);
668                 solrQuery.addFacetQuery(facetQuery);
669             }
670             if (0 < facetQueries.size())
671                 solrQuery.setFacet(true);
672         }
673 
674         if (facetField != null)
675             solrQuery.addFacetField(facetField);
676 
677         // Set the top x of if present
678         if (max != -1)
679             solrQuery.setFacetLimit(max);
680 
681         // We gebruiken filter aangezien dit performanter is dan dit gewoon aan
682         // de query toe te voegen
683         solrQuery.addFilterQuery(getIgnoreSpiders());
684         if (filterQuery != null)
685             solrQuery.addFilterQuery(filterQuery);
686 
687         QueryResponse response = null;
688         try
689         {
690             // solr.set
691             response = solr.query(solrQuery);
692         }
693         catch (SolrServerException e)
694         {
695             System.err.println("Error using query " + query);
696             throw e;
697         }
