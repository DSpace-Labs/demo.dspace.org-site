104      * This regular expression is used to perform sanity checks 
105      * on database names (i.e. tables and columns). 
106      * 
107      * FIXME: Regular expressions can be slow to solve this in the future we should
108      * probably create a system where we don't pass in column and table names to these low
109      * level database methods. This approach is highly exploitable for injection 
110      * type attacks because we are unable to determine where the input came from. Instead
111      * we could pass in static integer constants which are then mapped to their sql name. 
112      */

198     /**
199      * Return an iterator with the results of the query. The table parameter
200      * indicates the type of result. If table is null, the column names are read
201      * from the ResultSetMetaData.
202      * 
203      * @param context
204      *            The context object
205      * @param table
206      *            The name of the table which results
207      * @param query
208      *            The SQL query
209      * @param parameters
210      * 			  A set of SQL parameters to be included in query. The order of 
211      * 			  the parameters must correspond to the order of their reference  
212      * 			  within the query.
213      * @return A TableRowIterator with the results of the query
214      * @exception SQLException
215      *                If a database error occurs
216      */
217     public static TableRowIterator queryTable(Context context, String table,
218             String query, Object... parameters ) throws SQLException
219     {
220         if (log.isDebugEnabled())
221         {
222             StringBuffer sb = new StringBuffer();
223             for (int i = 0; i < parameters.length; i++)
224             {
225                 if (i > 0)
226                {
227                        sb.append(",");
228                }
229                 sb.append(parameters[i].toString());
230             }
231             log.debug("Running query \"" + query + "\"  with parameters: " + sb.toString());
232         }
233         
234         PreparedStatement statement = context.getDBConnection().prepareStatement(query);
235         try
236         {
237             loadParameters(statement,parameters);
238 
239             TableRowIterator retTRI = new TableRowIterator(statement.executeQuery(),
240                     canonicalize(table));
241 
242             retTRI.setStatement(statement);
243             return retTRI;
244         }
245         catch (SQLException sqle)
246         {
247             if (statement != null)
248                 try { statement.close(); } catch (SQLException s) { }
249 
250             throw sqle;
251         }
252     }

348     /**
349      * Return the single row result to this query, or null if no result. If more
350      * than one row results, only the first is returned.
351      * 
352      * @param context
353      *            Current DSpace context
354      * @param query
355      *            The SQL query
356      * @param parameters
357      * 			  A set of SQL parameters to be included in query. The order of 
358      * 			  the parameters must correspond to the order of their reference 
359      * 			  within the query.
360 
361      * @return A TableRow object, or null if no result
362      * @exception SQLException
363      *                If a database error occurs
364      */
365     public static TableRow querySingle(Context context, String query,
366             Object... parameters) throws SQLException
367     {
368         TableRow retRow = null;
369         TableRowIterator iterator = null;
370         try
371         {
372             iterator = query(context, query, parameters);
373             retRow = (!iterator.hasNext()) ? null : iterator.next();
374         }
375         finally
376         {
377             if (iterator != null)
378                 iterator.close();
379         }
380 
381         return (retRow);
382     }

537     public static TableRow findByUnique(Context context, String table,
538             String column, Object value) throws SQLException
539     {
540         String ctable = canonicalize(table);
541 
542         if ( ! DB_SAFE_NAME.matcher(ctable).matches())
543         	throw new SQLException("Unable to execute select query because table name ("+ctable+") contains non alphanumeric characters.");
544 
545         if ( ! DB_SAFE_NAME.matcher(column).matches())
546         	throw new SQLException("Unable to execute select query because column name ("+column+") contains non alphanumeric characters.");
547         
548         String sql = "select * from " + ctable + " where "+ column +" = ? ";
549 
550         return querySingleTable(context, ctable, sql, value);
551     }

660     /**
661      * Insert a table row into the RDBMS.
662      * 
663      * @param context
664      *            Current DSpace context
665      * @param row
666      *            The row to insert
667      * @exception SQLException
668      *                If a database error occurs
669      */
670     public static void insert(Context context, TableRow row)
671             throws SQLException
672     {
673         int newID = -1;
674         String table = canonicalize(row.getTable());
675         Statement statement = null;
676         ResultSet rs = null;
677 
678         try
679         {
680             // Get an ID (primary key) for this row by using the "getnextid"
681             // SQL function in Postgres, or directly with sequences in Oracle
682             String myQuery = "SELECT getnextid('" + table + "') AS result";
683 
684             if ("oracle".equals(ConfigurationManager.getProperty("db.name")))
685             {
686                 myQuery = "SELECT " + table + "_seq" + ".nextval FROM dual";
687             }
688 
689             statement = context.getDBConnection().createStatement();
690             rs = statement.executeQuery(myQuery);
691 
692             rs.next();
693 
694             newID = rs.getInt(1);
695         }
696         finally
697         {
698             if (rs != null)
699             {
700                 try { rs.close(); } catch (SQLException sqle) { }
701             }
702 
703             if (statement != null)
704             {
705                 try { statement.close(); } catch (SQLException sqle) { }
706             }
707         }
708 
709         if (newID < 0)
710             throw new SQLException("Unable to retrieve sequence ID");
711 
712         // Set the ID in the table row object
713         row.setColumn(getPrimaryKeyColumn(table), newID);
714 
715         StringBuffer sql = new StringBuffer().append("INSERT INTO ").append(
716                 table).append(" ( ");
717 
718         ColumnInfo[] info = getColumnInfo(table);
719 
720         for (int i = 0; i < info.length; i++)
721         {
722             sql.append((i == 0) ? "" : ",").append(info[i].getName());
723         }
724 
725         sql.append(") VALUES ( ");
726 
727         // Values to insert
728         for (int i = 0; i < info.length; i++)
729         {
730             sql.append((i == 0) ? "" : ",").append("?");
731         }
732 
733         // Watch the syntax
734         sql.append(")");
735 
736         execute(context.getDBConnection(), sql.toString(), Arrays.asList(info),
737                 row);
738     }

740     /**
741      * Update changes to the RDBMS. Note that if the update fails, the values in
742      * the row will NOT be reverted.
743      * 
744      * @param context
745      *            Current DSpace context
746      * @param row
747      *            The row to update
748      * @return The number of rows affected (1 or 0)
749      * @exception SQLException
750      *                If a database error occurs
751      */
752     public static int update(Context context, TableRow row) throws SQLException
753     {
754         String table = canonicalize(row.getTable());
755 
756         StringBuffer sql = new StringBuffer().append("update ").append(table)
757                 .append(" set ");
758 
759         List columns = new ArrayList();
760         ColumnInfo pk = getPrimaryKeyColumnInfo(table);
761         ColumnInfo[] info = getNonPrimaryKeyColumns(table);
762 
763         String seperator = "";
764         for (int i = 0; i < info.length; i++)
765         {
766             // Only update this column if it has changed
767             if (row.hasColumnChanged(info[i].getName()))
768             {
769                 sql.append(seperator).append(info[i].getName()).append(" = ?");
770                 columns.add(info[i]);
771                 seperator = ", ";
772             }
773         }
774 
775         // Only execute the update if there is anything to update
776         if (columns.size() > 0)
777         {
778             sql.append(" where ").append(pk.getName()).append(" = ?");
779             columns.add(pk);
780 
781             return execute(context.getDBConnection(), sql.toString(), columns, row);
782         }
783 
784         return 1;
785     }

787     /**
788      * Delete row from the RDBMS.
789      * 
790      * @param context
791      *            Current DSpace context
792      * @param row
793      *            The row to delete
794      * @return The number of rows affected (1 or 0)
795      * @exception SQLException
796      *                If a database error occurs
797      */
798     public static int delete(Context context, TableRow row) throws SQLException
799     {
800         String pk = getPrimaryKeyColumn(row);
801 
802         if (row.isColumnNull(pk))

970     /**
971      * Load SQL from a reader into the RDBMS.
972      * 
973      * @param r
974      *            The Reader from which to read the SQL.
975      * @throws SQLException
976      *            If a database error occurs
977      * @throws IOException
978      *            If an error occurs obtaining data from the reader
979      */
980     public static void loadSql(Reader r) throws SQLException, IOException
981     {
982         BufferedReader reader = new BufferedReader(r);
983         StringBuffer sql = new StringBuffer();
984         String SQL = null;
985 
986         String line = null;
987 
988         Connection connection = null;
989         Statement statement = null;
990 
991         try
992         {
993             connection = getConnection();
994             connection.setAutoCommit(true);
995             statement = connection.createStatement();
996 
997             boolean inquote = false;
998 
999             while ((line = reader.readLine()) != null)
1000             {
1001                 // Look for comments
1002                 int commentStart = line.indexOf("--");
1003 
1004                 String input = (commentStart != -1) ? line.substring(0,
1005                         commentStart) : line;
1006 
1007                 // Empty line, skip
1008                 if (input.trim().equals(""))
1009                 {
1010                     continue;
1011                 }
1012 
1013                 // Put it on the SQL buffer
1014                 sql.append(input.replace(';', ' ')); // remove all semicolons
1015                                                      // from sql file!
1016 
1017                 // Add a space
1018                 sql.append(" ");
1019 
1020                 // More to come?
1021                 // Look for quotes
1022                 int index = 0;
1023                 int count = 0;
1024                 int inputlen = input.length();
1025 
1026                 while ((index = input.indexOf("'", count)) != -1)
1027                 {
1028                     // Flip the value of inquote
1029                     inquote = !inquote;
1030 
1031                     // Move the index
1032                     count = index + 1;
1033 
1034                     // Make sure we do not exceed the string length
1035                     if (count >= inputlen)
1036                     {
1037                         break;
1038                     }
1039                 }
1040 
1041                 // If we are in a quote, keep going
1042                 // Note that this is STILL a simple heuristic that is not
1043                 // guaranteed to be correct
1044                 if (inquote)
1045                 {
1046                     continue;
1047                 }
1048 
1049                 int endMarker = input.indexOf(";", index);
1050 
1051                 if (endMarker == -1)
1052                 {
1053                     continue;
1054                 }
1055 
1056                 if (log.isDebugEnabled())
1057                 {
1058                     log.debug("Running database query \"" + sql + "\"");
1059                 }
1060 
1061                 SQL = sql.toString();
1062 
1063                 try
1064                 {
1065                     // Use execute, not executeQuery (which expects results) or
1066                     // executeUpdate
1067                     boolean succeeded = statement.execute(SQL);
1068                 }
1069                 catch (SQLWarning sqlw)
1070                 {
1071                     if (log.isDebugEnabled())
1072                     {
1073                         log.debug("Got SQL Warning: " + sqlw, sqlw);
1074                     }
1075                 }
1076                 catch (SQLException sqle)
1077                 {
1078                     String msg = "Got SQL Exception: " + sqle;
1079                     String sqlmessage = sqle.getMessage();
1080 
1081                     // These are Postgres-isms:
1082                     // There's no easy way to check if a table exists before
1083                     // creating it, so we always drop tables, then create them
1084                     boolean isDrop = ((SQL != null) && (sqlmessage != null)
1085                             && (SQL.toUpperCase().startsWith("DROP")) && (sqlmessage
1086                             .indexOf("does not exist") != -1));
1087 
1088                     // Creating a view causes a bogus warning
1089                     boolean isNoResults = ((SQL != null)
1090                             && (sqlmessage != null)
1091                             && ((SQL.toUpperCase().startsWith("CREATE VIEW")) || (SQL
1092                                     .toUpperCase()
1093                                     .startsWith("CREATE FUNCTION"))) && (sqlmessage
1094                             .indexOf("No results were returned") != -1));
1095 
1096                     // If the messages are bogus, give them a low priority
1097                     if (isDrop || isNoResults)
1098                     {
1099                         if (log.isDebugEnabled())
1100                         {
1101                             log.debug(msg, sqle);
1102                         }
1103                     }
1104                     // Otherwise, we need to know!
1105                     else
1106                     {
1107                         if (log.isEnabledFor(Level.WARN))
1108                         {
1109                             log.warn(msg, sqle);
1110                         }
1111                     }
1112                 }
1113 
1114                 // Reset SQL buffer
1115                 sql = new StringBuffer();
1116                 SQL = null;
1117             }
1118         }
1119         finally
1120         {
1121             if (connection != null)
1122             {
1123                 connection.close();
1124             }
1125 
1126             if (statement != null)
1127             {
1128                 statement.close();
1129             }
1130         }
1131     }

1137     /**
1138      * Convert the current row in a ResultSet into a TableRow object.
1139      * 
1140      * @param results
1141      *            A ResultSet to process
1142      * @param table
1143      *            The name of the table
1144      * @return A TableRow object with the data from the ResultSet
1145      * @exception SQLException
1146      *                If a database error occurs
1147      */
1148     static TableRow process(ResultSet results, String table)
1149             throws SQLException
1150     {
1151         String dbName =ConfigurationManager.getProperty("db.name");
1152         ResultSetMetaData meta = results.getMetaData();
1153         int columns = meta.getColumnCount() + 1;
1154 
1155         List columnNames = (table == null) ? getColumnNames(meta)
1156                 : getColumnNames(table);
1157 
1158         TableRow row = new TableRow(canonicalize(table), columnNames);
1159 
1160         // Process the columns in order
1161         // (This ensures maximum backwards compatibility with
1162         // old JDBC drivers)
1163         for (int i = 1; i < columns; i++)
1164         {
1165             String name = meta.getColumnName(i);
1166             int jdbctype = meta.getColumnType(i);
1167 
1168             if (jdbctype == Types.BIT)
1169             {
1170                 row.setColumn(name, results.getBoolean(i));
1171             }
1172             else if ((jdbctype == Types.INTEGER) || (jdbctype == Types.NUMERIC)
1173                     || (jdbctype == Types.DECIMAL))
1174             {
1175                 // If we are using oracle
1176                 if ("oracle".equals(dbName))
1177                 {
1178                     // Test the value from the record set. If it can be represented using an int, do so.
1179                     // Otherwise, store it as long
1180                     long longValue = results.getLong(i);
1181                     if (longValue <= (long)Integer.MAX_VALUE)
1182                         row.setColumn(name, (int)longValue);
1183                     else
1184                         row.setColumn(name, longValue);
1185                 }
1186                 else
1187                     row.setColumn(name, results.getInt(i));
1188             }
1189             else if (jdbctype == Types.BIGINT)
1190             {
1191                 row.setColumn(name, results.getLong(i));
1192             }
1193             else if (jdbctype == Types.DOUBLE)
1194             {
1195                 row.setColumn(name, results.getDouble(i));
1196             }
1197             else if (jdbctype == Types.CLOB && "oracle".equals(dbName))
1198             {
1199                 // Support CLOBs in place of TEXT columns in Oracle
1200                 row.setColumn(name, results.getString(i));
1201             }
1202             else if (jdbctype == Types.VARCHAR)
1203             {
1204                 try
1205                 {
1206                     byte[] bytes = results.getBytes(i);
1207 
1208                     if (bytes != null)
1209                     {
1210                         String mystring = new String(results.getBytes(i),
1211                                 "UTF-8");
1212                         row.setColumn(name, mystring);
1213                     }
1214                     else
1215                     {
1216                         row.setColumn(name, results.getString(i));
1217                     }
1218                 }
1219                 catch (UnsupportedEncodingException e)
1220                 {
1221                     // do nothing, UTF-8 is built in!
1222                 }
1223             }
1224             else if (jdbctype == Types.DATE)
1225             {
1226                 row.setColumn(name, results.getDate(i));
1227             }
1228             else if (jdbctype == Types.TIME)
1229             {
1230                 row.setColumn(name, results.getTime(i));
1231             }
1232             else if (jdbctype == Types.TIMESTAMP)
1233             {
1234                 row.setColumn(name, results.getTimestamp(i));
1235             }
1236             else
1237             {
1238                 throw new IllegalArgumentException("Unsupported JDBC type: "
1239                         + jdbctype);
1240             }
1241 
1242             if (results.wasNull())
1243             {
1244                 row.setColumnNull(name);
1245             }
1246         }
1247 
1248         // Now that we've prepped the TableRow, reset the flags so that we can detect which columns have changed
1249         row.resetChanged();
1250         return row;
1251     }

1318     /**
1319      * Execute SQL as a PreparedStatement on Connection. Bind parameters in
1320      * columns to the values in the table row before executing.
1321      * 
1322      * @param connection
1323      *            The SQL connection
1324      * @param sql
1325      *            The query to execute
1326      * @param columns
1327      *            The columns to bind
1328      * @param row
1329      *            The row
1330      * @return The number of rows affected by the query.
1331      * @exception SQLException
1332      *                If a database error occurs
1333      */
1334     private static int execute(Connection connection, String sql, List columns,
1335             TableRow row) throws SQLException
1336     {
1337         String dbName =ConfigurationManager.getProperty("db.name");
1338         PreparedStatement statement = null;
1339 
1340         if (log.isDebugEnabled())
1341         {
1342             log.debug("Running query \"" + sql + "\"");
1343         }
1344 
1345         try
1346         {
1347             statement = connection.prepareStatement(sql);
1348         	
1349             int count = 0;
1350 
1351             for (Iterator iterator = columns.iterator(); iterator.hasNext();)
1352             {
1353                 count++;
1354 
1355                 ColumnInfo info = (ColumnInfo) iterator.next();
1356                 String column = info.getName();
1357                 int jdbctype = info.getType();
1358 
1359                 if (row.isColumnNull(column))
1360                 {
1361                     statement.setNull(count, jdbctype);
1362 
1363                     continue;
1364                 }
1365                 else if (jdbctype == Types.BIT)
1366                 {
1367                     statement.setBoolean(count, row.getBooleanColumn(column));
1368 
1369                     continue;
1370                 }
1371                 else if ((jdbctype == Types.INTEGER) || (jdbctype == Types.NUMERIC)
1372                         || (jdbctype == Types.DECIMAL))
1373                 {
1374                     // If we are using Oracle, we can pass in long values, so always do so.
1375                     if ("oracle".equals(dbName))
1376                         statement.setLong(count, row.getLongColumn(column));
1377                     else
1378                         statement.setInt(count, row.getIntColumn(column));
1379 
1380                     continue;
1381                 }
1382                 else if (jdbctype == Types.BIGINT)
1383                 {
1384                     statement.setLong(count, row.getLongColumn(column));
1385                 }
1386                 else if (jdbctype == Types.CLOB && "oracle".equals(dbName))
1387                 {
1388                     // Support CLOBs in place of TEXT columns in Oracle
1389                     statement.setString(count, row.getStringColumn(column));
1390 
1391                     continue;
1392                 }
1393                 else if (jdbctype == Types.VARCHAR)
1394                 {
1395                     statement.setString(count, row.getStringColumn(column));
1396 
1397                     continue;
1398                 }
1399                 else if (jdbctype == Types.DATE)
1400                 {
1401                     java.sql.Date d = new java.sql.Date(row.getDateColumn(
1402                             column).getTime());
1403                     statement.setDate(count, d);
1404 
1405                     continue;
1406                 }
1407                 else if (jdbctype == Types.TIME)
1408                 {
1409                     Time t = new Time(row.getDateColumn(column).getTime());
1410                     statement.setTime(count, t);
1411 
1412                     continue;
1413                 }
1414                 else if (jdbctype == Types.TIMESTAMP)
1415                 {
1416                     Timestamp t = new Timestamp(row.getDateColumn(column)
1417                             .getTime());
1418                     statement.setTimestamp(count, t);
1419 
1420                     continue;
1421                 }
1422                 else
1423                 {
1424                     throw new IllegalArgumentException(
1425                             "Unsupported JDBC type: " + jdbctype);
1426                 }
1427             }
1428 
1429             return statement.executeUpdate();
1430         }
1431         finally
1432         {
1433             if (statement != null)
1434             {
1435                 try
1436                 {
1437                     statement.close();
1438                 }
1439                 catch (SQLException sqle)
1440                 {
1441                 }
1442             }
1443         }
1444     }

1471     /**
1472      * Read metadata about a table from the database.
1473      * 
1474      * @param table
1475      *            The RDBMS table.
1476      * @return A map of information about the columns. The key is the name of
1477      *         the column, a String; the value is a ColumnInfo object.
1478      * @exception SQLException
1479      *                If there is a problem retrieving information from the
1480      *                RDBMS.
1481      */
1482     private static Map retrieveColumnInfo(String table) throws SQLException
1483     {
1484         Connection connection = null;
1485         ResultSet pkcolumns = null;
1486         ResultSet columns = null;
1487         
1488         try
1489         {
1490             String schema = ConfigurationManager.getProperty("db.schema");
1491             String catalog = null;
1492             
1493             int dotIndex = table.indexOf("."); 
1494             if (dotIndex > 0)
1495             {
1496                 catalog = table.substring(0, dotIndex);
1497                 table = table.substring(dotIndex + 1, table.length());
1498                 log.warn("catalog: " + catalog);
1499                 log.warn("table: " + table);
1500             }
1501             
1502             connection = getConnection();
1503 
1504             DatabaseMetaData metadata = connection.getMetaData();
1505             HashMap results = new HashMap();
1506 
1507             int max = metadata.getMaxTableNameLength();
1508             String tname = (table.length() >= max) ? table
1509                     .substring(0, max - 1) : table;
1510             
1511             pkcolumns = metadata.getPrimaryKeys(catalog, schema, tname);
1512             
1513             Set pks = new HashSet();
1514 
1515             while (pkcolumns.next())
1516                 pks.add(pkcolumns.getString(4));
1517 
1518             columns = metadata.getColumns(catalog, schema, tname, null);
1519 
1520             while (columns.next())
1521             {
1522                 String column = columns.getString(4);
1523                 ColumnInfo cinfo = new ColumnInfo();
1524                 cinfo.setName(column);
1525                 cinfo.setType((int) columns.getShort(5));
1526 
1527                 if (pks.contains(column))
1528                 {
1529                     cinfo.setIsPrimaryKey(true);
1530                 }
1531 
1532                 results.put(column, cinfo);
1533             }
1534 
1535             return results;
1536         }
1537         finally
1538         {
1539             if (pkcolumns != null)
1540             {
1541                 try { pkcolumns.close(); } catch (SQLException sqle) { }
1542             }
1543 
1544             if (columns != null)
1545             {
1546                 try { columns.close(); } catch (SQLException sqle) { }
1547             }
1548 
1549             if (connection != null)
1550             {
1551                 try { connection.close(); } catch (SQLException sqle) { }
1552             }
1553         }
1554     }
