69  /**
70   * <P>
71   * Stores, retrieves and deletes bitstreams.
72   * </P>
73   * 
74   * <P>
75   * Presently, asset stores are specified in <code>dspace.cfg</code>. Since
76   * Java does not offer a way of detecting free disk space, the asset store to
77   * use for new bitstreams is also specified in a configuration property. The
78   * drawbacks to this are that the administrators are responsible for monitoring
79   * available space in the asset stores, and DSpace (Tomcat) has to be restarted
80   * when the asset store for new ('incoming') bitstreams is changed.
81   * </P>
82   * 
83   * <P>
84   * Mods by David Little, UCSD Libraries 12/21/04 to allow the registration of
85   * files (bitstreams) into DSpace.
86   * </P>
87   * 
88   * <p>Cleanup integration with checker package by Nate Sarr 2006-01. N.B. The 
89   * dependency on the checker package isn't ideal - a Listener pattern would be 
90   * better but was considered overkill for the purposes of integrating the checker.
91   * It would be worth re-considering a Listener pattern if another package needs to 
92   * be notified of BitstreamStorageManager actions.</p> 
93   *
94   * @author Peter Breton, Robert Tansley, David Little, Nathan Sarr
95   * @version $Revision: 4519 $
96   */

102 	/**
103 	 * The asset store locations. The information for each GeneralFile in the
104 	 * array comes from dspace.cfg, so see the comments in that file.
105 	 *
106 	 * If an array element refers to a conventional (non_SRB) asset store, the
107 	 * element will be a LocalFile object (similar to a java.io.File object)
108 	 * referencing a local directory under which the bitstreams are stored.
109 	 *
110 	 * If an array element refers to an SRB asset store, the element will be an
111 	 * SRBFile object referencing an SRB 'collection' (directory) under which
112 	 * the bitstreams are stored.
113 	 *
114 	 * An SRBFile object is obtained by (1) using dspace.cfg properties to
115 	 * create an SRBAccount object (2) using the account to create an
116 	 * SRBFileSystem object (similar to a connection) (3) using the
117 	 * SRBFileSystem object to create an SRBFile object
118 	 */
119 	private static GeneralFile[] assetStores;

124     // These settings control the way an identifier is hashed into
125     // directory and file names
126     //
127     // With digitsPerLevel 2 and directoryLevels 3, an identifier
128     // like 12345678901234567890 turns into the relative name
129     // /12/34/56/12345678901234567890.
130     //
131     // You should not change these settings if you have data in the
132     // asset store, as the BitstreamStorageManager will be unable
133     // to find your existing data.
134     private static final int digitsPerLevel = 2;
135 
136     private static final int directoryLevels = 3;

143     /* Read in the asset stores from the config. */
144     static
145     {
146         ArrayList stores = new ArrayList();
147 
148 		// 'assetstore.dir' is always store number 0
149 		String sAssetstoreDir = ConfigurationManager
150 				.getProperty("assetstore.dir");
151  
152 		// see if conventional assetstore or srb
153 		if (sAssetstoreDir != null) {
154 			stores.add(sAssetstoreDir); // conventional (non-srb)
155 		} else if (ConfigurationManager.getProperty("srb.host") != null) {
156 			stores.add(new SRBAccount( // srb
157 					ConfigurationManager.getProperty("srb.host"),
158 					ConfigurationManager.getIntProperty("srb.port"),
159 					ConfigurationManager.getProperty("srb.username"),
160 					ConfigurationManager.getProperty("srb.password"),
161 					ConfigurationManager.getProperty("srb.homedirectory"),
162 					ConfigurationManager.getProperty("srb.mdasdomainname"),
163 					ConfigurationManager
164 							.getProperty("srb.defaultstorageresource"),
165 					ConfigurationManager.getProperty("srb.mcatzone")));
166 		} else {
167 			log.error("No default assetstore");
168 		}
169 
170 		// read in assetstores .1, .2, ....
171 		for (int i = 1;; i++) { // i == 0 is default above
172 			sAssetstoreDir = ConfigurationManager.getProperty("assetstore.dir."
173 					+ i);
174 
175 			// see if 'i' conventional assetstore or srb
176 			if (sAssetstoreDir != null) { 		// conventional (non-srb)
177 				stores.add(sAssetstoreDir);
178 			} else if (ConfigurationManager.getProperty("srb.host." + i)
179 					!= null) { // srb
180 				stores.add(new SRBAccount(
181 						ConfigurationManager.getProperty("srb.host." + i),
182 						ConfigurationManager.getIntProperty("srb.port." + i),
183 						ConfigurationManager.getProperty("srb.username." + i),
184 						ConfigurationManager.getProperty("srb.password." + i),
185 						ConfigurationManager
186 								.getProperty("srb.homedirectory." + i),
187 						ConfigurationManager
188 								.getProperty("srb.mdasdomainname." + i),
189 						ConfigurationManager
190 								.getProperty("srb.defaultstorageresource." + i),
191 						ConfigurationManager.getProperty("srb.mcatzone." + i)));
192 			} else {
193 				break; // must be at the end of the assetstores
194 			}
195 		}
196 
197 		// convert list to array
198 		// the elements (objects) in the list are class
199 		//   (1) String - conventional non-srb assetstore
200 		//   (2) SRBAccount - srb assetstore
201 		assetStores = new GeneralFile[stores.size()];
202 		for (int i = 0; i < stores.size(); i++) {
203 			Object o = stores.get(i);
204 			if (o == null) { // I don't know if this can occur
205 				log.error("Problem with assetstore " + i);
206 			}
207 			if (o instanceof String) {
208 				assetStores[i] = new LocalFile((String) o);
209 			} else if (o instanceof SRBAccount) {
210 				SRBFileSystem srbFileSystem = null;
211 				try {
212 					srbFileSystem = new SRBFileSystem((SRBAccount) o);
213 				} catch (NullPointerException e) {
214 					log.error("No SRBAccount for assetstore " + i);
215 				} catch (IOException e) {
216 					log.error("Problem getting SRBFileSystem for assetstore"
217 							+ i);
218 				}
219 				if (srbFileSystem == null) {
220 					log.error("SRB FileSystem is null for assetstore " + i);
221 				}
222 				String sSRBAssetstore = null;
223 				if (i == 0) { // the zero (default) assetstore has no suffix
224 					sSRBAssetstore = ConfigurationManager
225 							.getProperty("srb.parentdir");
226 				} else {
227 					sSRBAssetstore = ConfigurationManager
228 							.getProperty("srb.parentdir." + i);
229 				}
230 				if (sSRBAssetstore == null) {
231 					log.error("srb.parentdir is undefined for assetstore " + i);
232 				}
233 				assetStores[i] = new SRBFile(srbFileSystem, sSRBAssetstore);
234 			} else {
235 				log.error("Unexpected " + o.getClass().toString()
236 						+ " with assetstore " + i);
237 			}
238 		}
239 
240         // Read asset store to put new files in. Default is 0.
241         incoming = ConfigurationManager.getIntProperty("assetstore.incoming");
242     }

244     /**
245      * Store a stream of bits.
246      * 
247      * <p>
248      * If this method returns successfully, the bits have been stored, and RDBMS
249      * metadata entries are in place (the context still needs to be completed to
250      * finalize the transaction).
251      * </p>
252      * 
253      * <p>
254      * If this method returns successfully and the context is aborted, then the
255      * bits will be stored in the asset store and the RDBMS metadata entries
256      * will exist, but with the deleted flag set.
257      * </p>
258      * 
259      * If this method throws an exception, then any of the following may be
260      * true:
261      * 
262      * <ul>
263      * <li>Neither bits nor RDBMS metadata entries have been stored.
264      * <li>RDBMS metadata entries with the deleted flag set have been stored,
265      * but no bits.
266      * <li>RDBMS metadata entries with the deleted flag set have been stored,
267      * and some or all of the bits have also been stored.
268      * </ul>
269      * 
270      * @param context
271      *            The current context
272      * @param is
273      *            The stream of bits to store
274      * @exception IOException
275      *                If a problem occurs while storing the bits
276      * @exception SQLException
277      *                If a problem occurs accessing the RDBMS
278      * 
279      * @return The ID of the stored bitstream
280      */
281     public static int store(Context context, InputStream is)
282             throws SQLException, IOException
283     {
284         // Create internal ID
285         String id = Utils.generateKey();
286 
287         // Create a deleted bitstream row, using a separate DB connection
288         TableRow bitstream;
289         Context tempContext = null;
290 
291         try
292         {
293             tempContext = new Context();
294 
295             bitstream = DatabaseManager.create(tempContext, "Bitstream");
296             bitstream.setColumn("deleted", true);
297             bitstream.setColumn("internal_id", id);
298 
299             /*
300              * Set the store number of the new bitstream If you want to use some
301              * other method of working out where to put a new bitstream, here's
302              * where it should go
303              */
304             bitstream.setColumn("store_number", incoming);
305 
306             DatabaseManager.update(tempContext, bitstream);
307 
308             tempContext.complete();
309         }
310         catch (SQLException sqle)
311         {
312             if (tempContext != null)
313             {
314                 tempContext.abort();
315             }
316 
317             throw sqle;
318         }
319 
320         // Where on the file system will this new bitstream go?
321 		GeneralFile file = getFile(bitstream);
322 
323         // Make the parent dirs if necessary
324 		GeneralFile parent = file.getParentFile();
325 
326         if (!parent.exists())
327         {
328             parent.mkdirs();
329         }
330 
331         //Create the corresponding file and open it
332         file.createNewFile();
333 
334 		GeneralFileOutputStream fos = FileFactory.newFileOutputStream(file);
335 
336 		// Read through a digest input stream that will work out the MD5
337         DigestInputStream dis = null;
338 
339         try
340         {
341             dis = new DigestInputStream(is, MessageDigest.getInstance("MD5"));
342         }
343         // Should never happen
344         catch (NoSuchAlgorithmException nsae)
345         {
346             log.warn("Caught NoSuchAlgorithmException", nsae);
347         }
348 
349         Utils.bufferedCopy(dis, fos);
350         fos.close();
351         is.close();
352 
353         bitstream.setColumn("size_bytes", file.length());
354 
355         if (dis != null)
356         {
357             bitstream.setColumn("checksum", Utils.toHex(dis.getMessageDigest()
358                     .digest()));
359             bitstream.setColumn("checksum_algorithm", "MD5");
360         }
361         
362         bitstream.setColumn("deleted", false);
363         DatabaseManager.update(context, bitstream);
364 
365         int bitstream_id = bitstream.getIntColumn("bitstream_id");
366 
367         if (log.isDebugEnabled())
368         {
369             log.debug("Stored bitstream " + bitstream_id + " in file "
370                     + file.getAbsolutePath());
371         }
372 
373         return bitstream_id;
374     }

376 	/**
377 	 * Register a bitstream already in storage.
378 	 *
379 	 * @param context
380 	 *            The current context
381 	 * @param assetstore The assetstore number for the bitstream to be
382 	 * 			registered
383 	 * @param bitstreamPath The relative path of the bitstream to be registered.
384 	 * 		The path is relative to the path of ths assetstore.
385 	 * @return The ID of the registered bitstream
386 	 * @exception SQLException
387 	 *                If a problem occurs accessing the RDBMS
388 	 * @throws IOException
389 	 */
390 	public static int register(Context context, int assetstore,
391 				String bitstreamPath) throws SQLException, IOException {
392 
393 		// mark this bitstream as a registered bitstream
394 		String sInternalId = REGISTERED_FLAG + bitstreamPath;
395 
396 		// Create a deleted bitstream row, using a separate DB connection
397 		TableRow bitstream;
398 		Context tempContext = null;
399 
400 		try {
401 			tempContext = new Context();
402 
403 			bitstream = DatabaseManager.create(tempContext, "Bitstream");
404 			bitstream.setColumn("deleted", true);
405 			bitstream.setColumn("internal_id", sInternalId);
406 			bitstream.setColumn("store_number", assetstore);
407 			DatabaseManager.update(tempContext, bitstream);
408 
409 			tempContext.complete();
410 		} catch (SQLException sqle) {
411 			if (tempContext != null) {
412 				tempContext.abort();
413 			}
414 			throw sqle;
415 		}
416 
417 		// get a reference to the file
418 		GeneralFile file = getFile(bitstream);
419 
420 		// read through a DigestInputStream that will work out the MD5
421 		//
422 		// DSpace refers to checksum, writes it in METS, and uses it as an
423 		// AIP filename (!), but never seems to validate with it. Furthermore,
424 		// DSpace appears to hardcode the algorithm to MD5 in some places--see 
425 		// METSExport.java.
426 		//
427 		// To remain compatible with DSpace we calculate an MD5 checksum on 
428 		// LOCAL registered files. But for REMOTE (e.g. SRB) files we 
429 		// calculate an MD5 on just the fileNAME. The reasoning is that in the 
430 		// case of a remote file, calculating an MD5 on the file itself will
431 		// generate network traffic to read the file's bytes. In this case it 
432 		// would be better have a proxy process calculate MD5 and store it as 
433 		// an SRB metadata attribute so it can be retrieved simply from SRB.
434 		//
435 		// TODO set this up as a proxy server process so no net activity
436 		
437 		// FIXME this is a first class HACK! for the reasons described above
438 		if (file instanceof LocalFile) 
439 		{
440 
441 			// get MD5 on the file for local file
442 			DigestInputStream dis = null;
443 			try 
444 			{
445 				dis = new DigestInputStream(FileFactory.newFileInputStream(file), 
446 						MessageDigest.getInstance("MD5"));
447 			} 
448 			catch (NoSuchAlgorithmException e) 
449 			{
450 				log.warn("Caught NoSuchAlgorithmException", e);
451 				throw new IOException("Invalid checksum algorithm");
452 			}
453 			catch (IOException e) 
454 			{
455 				log.error("File: " + file.getAbsolutePath() 
456 						+ " to be registered cannot be opened - is it "
457 						+ "really there?");
458 				throw e;
459 			}
460 			final int BUFFER_SIZE = 1024 * 4;
461 			final byte[] buffer = new byte[BUFFER_SIZE];
462 			while (true) 
463 			{
464 				final int count = dis.read(buffer, 0, BUFFER_SIZE);
465 				if (count == -1) 
466 				{
467 					break;
468 				}
469 			}
470 			bitstream.setColumn("checksum", Utils.toHex(dis.getMessageDigest()
471 					.digest()));
472 			dis.close();
473 		} 
474 		else if (file instanceof SRBFile)
475 		{
476 			if (!file.exists())
477 			{
478 				log.error("File: " + file.getAbsolutePath() 
479 						+ " is not in SRB MCAT");
480 				throw new IOException("File is not in SRB MCAT");
481 			}
482 
483 			// get MD5 on just the filename (!) for SRB file
484 			int iLastSlash = bitstreamPath.lastIndexOf('/');
485 			String sFilename = bitstreamPath.substring(iLastSlash + 1);
486 			MessageDigest md = null;
487 			try 
488 			{
489 				md = MessageDigest.getInstance("MD5");
490 			} 
491 			catch (NoSuchAlgorithmException e) 
492 			{
493 				log.error("Caught NoSuchAlgorithmException", e);
494 				throw new IOException("Invalid checksum algorithm");
495 			}
496 			bitstream.setColumn("checksum", 
497 					Utils.toHex(md.digest(sFilename.getBytes())));
498 		}
499 		else
500 		{
501 			throw new IOException("Unrecognized file type - "
502 					+ "not local, not SRB");
503 		}
504 
505 		bitstream.setColumn("checksum_algorithm", "MD5");
506 		bitstream.setColumn("size_bytes", file.length());
507 		bitstream.setColumn("deleted", false);
508 		DatabaseManager.update(context, bitstream);
509 
510 		int bitstream_id = bitstream.getIntColumn("bitstream_id");
511 		if (log.isDebugEnabled()) 
512 		{
513 			log.debug("Stored bitstream " + bitstream_id + " in file "
514 					+ file.getAbsolutePath());
515 		}
516 		return bitstream_id;
517 	}

519 	/**
520 	 * Does the internal_id column in the bitstream row indicate the bitstream
521 	 * is a registered file
522 	 *
523 	 * @param internalId the value of the internal_id column
524 	 * @return true if the bitstream is a registered file
525 	 */
526 	public static boolean isRegisteredBitstream(String internalId) {
527 	    if (internalId.substring(0, REGISTERED_FLAG.length())
528 	            .equals(REGISTERED_FLAG)) 
529 	    {
530 	        return true;
531 	    }
532 	    return false;
533 	}

561      * <p>
562      * Remove a bitstream from the asset store. This method does not delete any
563      * bits, but simply marks the bitstreams as deleted (the context still needs
564      * to be completed to finalize the transaction).
565      * </p>
566      * 
567      * <p>

570      * </p>
571      * 
572      * @param context
573      *            The current context
574      * @param id
575      *            The ID of the bitstream to delete
576      * @exception SQLException
577      *                If a problem occurs accessing the RDBMS
578      */
579     public static void delete(Context context, int id) throws SQLException
580     {
581         DatabaseManager.updateQuery(context,
582                         "update Bitstream set deleted = '1' where bitstream_id = ? ",
583                         id);
584     }

586     /**
587      * Clean up the bitstream storage area. This method deletes any bitstreams
588      * which are more than 1 hour old and marked deleted. The deletions cannot
589      * be undone.
590      * 
591      * @param deleteDbRecords if true deletes the database records otherwise it
592      * 	           only deletes the files and directories in the assetstore  
593      * @exception IOException
594      *                If a problem occurs while cleaning up
595      * @exception SQLException
596      *                If a problem occurs accessing the RDBMS
597      */
598     public static void cleanup(boolean deleteDbRecords, boolean verbose) throws SQLException, IOException
599     {
600         Context context = null;
601         BitstreamInfoDAO bitstreamInfoDAO = new BitstreamInfoDAO();
602         int commit_counter = 0;
603 
604         try
605         {
606             context = new Context();
607 
608             String myQuery = "select * from Bitstream where deleted = '1'";
609 
610             List storage = DatabaseManager.queryTable(context, "Bitstream", myQuery)
611                     .toList();
612 
613             for (Iterator iterator = storage.iterator(); iterator.hasNext();)
614             {
615                 TableRow row = (TableRow) iterator.next();
616                 int bid = row.getIntColumn("bitstream_id");
617 
618 				GeneralFile file = getFile(row);
619 
620                 // Make sure entries which do not exist are removed
621                 if (file == null || !file.exists())
622                 {
623                     log.debug("file is null");
624                     if (deleteDbRecords)
625                     {
626                         log.debug("deleting record");
627                         if (verbose) System.out.println(" - Deleting bitstream information (ID: " + bid + ")");
628                         bitstreamInfoDAO.deleteBitstreamInfoWithHistory(bid);
629                         if (verbose) System.out.println(" - Deleting bitstream record from database (ID: " + bid + ")");
630                         DatabaseManager.delete(context, "Bitstream", bid);
631                     }
632                     continue;
633                 }
634 
635                 // This is a small chance that this is a file which is
636                 // being stored -- get it next time.
637                 if (isRecent(file))
638                 {
639                 	log.debug("file is recent");
640                     continue;
641                 }
642 
643                 if (deleteDbRecords)
644                 {
645                     log.debug("deleting db record");
646                     if (verbose) System.out.println(" - Deleting bitstream information (ID: " + bid + ")");
647                     bitstreamInfoDAO.deleteBitstreamInfoWithHistory(bid);
648                     if (verbose) System.out.println(" - Deleting bitstream record from database (ID: " + bid + ")");
649                     DatabaseManager.delete(context, "Bitstream", bid);
650                 }
651 
652 				if (isRegisteredBitstream(row.getStringColumn("internal_id"))) {
653 				    continue;			// do not delete registered bitstreams
654 				}
655 
656                 boolean success = file.delete();
657 
658                 String message = ("Deleted bitstream " + bid + " (file "
659                             + file.getAbsolutePath() + ") with result "
660                             + success);
661                 if (log.isDebugEnabled())
662                 {
663                     log.debug(message);
664                 }
665                 if (verbose) System.out.println(message);
666 
667                 // if the file was deleted then
668                 // try deleting the parents
669                 // Otherwise the cleanup script is set to 
670                 // leave the db records then the file
671                 // and directories have already been deleted
672                 // if this is turned off then it still looks like the
673                 // file exists
674                 if( success )
675                 {
676                     deleteParents(file);
677                 }
678                 
679                 // Make sure to commit our outstanding work every 100
680                 // iterations. Otherwise you risk losing the entire transaction
681                 // if we hit an exception, which isn't useful at all for large
682                 // amounts of bitstreams.
683                 commit_counter++;
684                 if (commit_counter % 100 == 0)
685                 {
686                 	System.out.print("Committing changes to the database...");
687                     context.commit();
688                     System.out.println(" Done!");
689                 }
690             }
691 
692             context.complete();
693         }
694         // Aborting will leave the DB objects around, even if the
695         // bitstreams are deleted. This is OK; deleting them next
696         // time around will be a no-op.
697         catch (SQLException sqle)
698         {
699             if (verbose) System.err.println("Error: " + sqle.getMessage());
700             context.abort();
701             throw sqle;
702         }
703         catch (IOException ioe)
704         {
705             if (verbose) System.err.println("Error: " + ioe.getMessage());
706             context.abort();
707             throw ioe;
708         }
709     }

716      * Return true if this file is too recent to be deleted, false otherwise.
717      * 
718      * @param file
719      *            The file to check
720      * @return True if this file is too recent to be deleted
721      */
722     private static boolean isRecent(GeneralFile file)
723     {
724         long lastmod = file.lastModified();
725         long now = new java.util.Date().getTime();
726 
727         if (lastmod >= now)
728         {
729             return true;
730         }
731 
732         // Less than one hour old
733         return (now - lastmod) < (1 * 60 * 1000);

737      * Delete empty parent directories.
738      * 
739      * @param file
740      *            The file with parent directories to delete
741      */
742     private synchronized static void deleteParents(GeneralFile file)
743     {
744         if (file == null )
745         {
746             return;
747         }
748  
749 		GeneralFile tmp = file;
750 
751         for (int i = 0; i < directoryLevels; i++)
752         {
753 
754 			GeneralFile directory = tmp.getParentFile();
755 			GeneralFile[] files = directory.listFiles();
756 
757             // Only delete empty directories
758             if (files.length != 0)
759             {
760                 break;
761             }
762 
763             directory.delete();
764             tmp = directory;
765         }

768     /**
769      * Return the file corresponding to a bitstream. It's safe to pass in
770      * <code>null</code>.
771      * 
772      * @param bitstream
773      *            the database table row for the bitstream. Can be
774      *            <code>null</code>
775      * 
776      * @return The corresponding file in the file system, or <code>null</code>
777      * 
778      * @exception IOException
779      *                If a problem occurs while determining the file
780      */
781     private static GeneralFile getFile(TableRow bitstream) throws IOException
782     {
783         // Check that bitstream is not null
784         if (bitstream == null)
785         {
786             return null;
787         }
788 
789         // Get the store to use
790         int storeNumber = bitstream.getIntColumn("store_number");
791 
792         // Default to zero ('assetstore.dir') for backwards compatibility
793         if (storeNumber == -1)
794         {
795             storeNumber = 0;
796         }
797 
798 		GeneralFile assetstore = assetStores[storeNumber];
799 
800 		// turn the internal_id into a file path relative to the assetstore
801 		// directory
802 		String sInternalId = bitstream.getStringColumn("internal_id");
803 
804 		// there are 4 cases:
805 		// -conventional bitstream, conventional storage
806 		// -conventional bitstream, srb storage
807 		// -registered bitstream, conventional storage
808 		// -registered bitstream, srb storage
809 		// conventional bitstream - dspace ingested, dspace random name/path
810 		// registered bitstream - registered to dspace, any name/path
811 		String sIntermediatePath = null;
812 		if (isRegisteredBitstream(sInternalId)) {
813 			sInternalId = sInternalId.substring(REGISTERED_FLAG.length());
814 			sIntermediatePath = "";
815 		} else {
816 			
817 			// Sanity Check: If the internal ID contains a
818 			// pathname separator, it's probably an attempt to
819 			// make a path traversal attack, so ignore the path
820 			// prefix.  The internal-ID is supposed to be just a
821 			// filename, so this will not affect normal operation.
822 			if (sInternalId.indexOf(File.separator) != -1)
823 				sInternalId = sInternalId.substring(sInternalId.lastIndexOf(File.separator)+1);
824 			
825 			sIntermediatePath = getIntermediatePath(sInternalId);
826 		}
827 
828 		StringBuffer bufFilename = new StringBuffer();
829 		if (assetstore instanceof LocalFile) {
830 			bufFilename.append(assetstore.getCanonicalPath());
831 			bufFilename.append(File.separator);
832 			bufFilename.append(sIntermediatePath);
833 			bufFilename.append(sInternalId);
834 			if (log.isDebugEnabled()) {
835 				log.debug("Local filename for " + sInternalId + " is "
836 						+ bufFilename.toString());
837 			}
838 			return new LocalFile(bufFilename.toString());
839 		}
840 		if (assetstore instanceof SRBFile) {
841 			bufFilename.append(sIntermediatePath);
842 			bufFilename.append(sInternalId);
843 			if (log.isDebugEnabled()) {
844 				log.debug("SRB filename for " + sInternalId + " is "
845 						+ ((SRBFile) assetstore).toString()
846 						+ bufFilename.toString());
847 			}
848 			return new SRBFile((SRBFile) assetstore, bufFilename.toString());
849 		}
850 		return null;
851     }

854 	 * Return the intermediate path derived from the internal_id. This method
855 	 * splits the id into groups which become subdirectories.
856 	 *
857 	 * @param iInternalId
858 	 *            The internal_id
859 	 * @return The path based on the id without leading or trailing separators
860 	 */
861 	private static String getIntermediatePath(String iInternalId) {
862 		StringBuffer buf = new StringBuffer();
863 		for (int i = 0; i < directoryLevels; i++) {
864 			int digits = i * digitsPerLevel;
865 			if (i > 0) {
866 				buf.append(File.separator);
867 			}
868 			buf.append(iInternalId.substring(digits, digits
869 							+ digitsPerLevel));
870 		}
871 		buf.append(File.separator);
872 		return buf.toString();
873 	}
