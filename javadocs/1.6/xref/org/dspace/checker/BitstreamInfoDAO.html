68      private static final String FIND_BY_BITSTREAM_ID = "select bitstream.deleted, bitstream.store_number, bitstream.size_bytes, "
69              + "bitstreamformatregistry.short_description, bitstream.bitstream_id,  "
70              + "bitstream.user_format_description, bitstream.internal_id, "

72              + "bitstream.name, most_recent_checksum.last_process_end_date,"
73              + "most_recent_checksum.to_be_processed "

75              + "bitstream.bitstream_format_id = bitstreamformatregistry.bitstream_format_id, "

87              + "last_process_start_date, last_process_end_date, "
88              + "checksum_algorithm, matched_prev_checksum, result ) "
89              + "select bitstream.bitstream_id, "
90              + "CASE WHEN bitstream.deleted = false THEN true ELSE false END, "
91              + "CASE WHEN bitstream.checksum IS NULL THEN '' ELSE bitstream.checksum END, "
92              + "CASE WHEN bitstream.checksum IS NULL THEN '' ELSE bitstream.checksum END, "
93              + "?, ?, CASE WHEN bitstream.checksum_algorithm IS NULL "
94              + "THEN 'MD5' ELSE bitstream.checksum_algorithm END, true, "
95              + "CASE WHEN bitstream.deleted = true THEN 'BITSTREAM_MARKED_DELETED' else 'CHECKSUM_MATCH' END "
96              + "from bitstream where not exists( "
97              + "select 'x' from most_recent_checksum "
98              + "where most_recent_checksum.bitstream_id = bitstream.bitstream_id )";

131      * timestamp is truncated to milliseconds this is because the Date for java

138             + "bitstream_id " + "ASC LIMIT 1";

193     /**
194      * Updates most_recent_checksum with latest checksum and result of
195      * comparison with previous checksum.
196      * 
197      * @param info
198      *            The BitstreamInfo to update.
199      * 
200      * @throws IllegalArgumentException
201      *             if the BitstreamInfo given is null.
202      */
203     public void update(BitstreamInfo info)
204     {
205         if (info == null)
206         {
207             throw new IllegalArgumentException(
208                     "BitstreamInfo parameter may not be null");
209         }
210 
211         Connection conn = null;
212         PreparedStatement stmt = null;
213 
214         try
215         {
216             conn = DatabaseManager.getConnection();
217             stmt = conn.prepareStatement(UPDATE_CHECKSUM);
218             stmt.setString(1, (info.getCalculatedChecksum() != null) ? info
219                     .getCalculatedChecksum() : "");
220             stmt.setString(2, info.getStoredChecksum());
221             stmt.setBoolean(3, ChecksumCheckResults.CHECKSUM_MATCH.equals(info
222                     .getChecksumCheckResult()));
223             stmt.setBoolean(4, info.getToBeProcessed());
224             stmt.setTimestamp(5, new Timestamp(info.getProcessStartDate()
225                     .getTime()));
226             stmt.setTimestamp(6, new Timestamp(info.getProcessEndDate()
227                     .getTime()));
228             stmt.setString(7, info.getChecksumCheckResult());
229             stmt.setInt(8, info.getBitstreamId());
230             stmt.executeUpdate();
231             conn.commit();
232         }
233         catch (SQLException e)
234         {
235             LOG.error("Problem updating checksum row. " + e.getMessage(), e);
236             throw new RuntimeException("Problem updating checksum row. "
237                     + e.getMessage(), e);
238         }
239         finally
240         {
241             cleanup(stmt, conn);
242         }
243     }

245     /**
246      * Find a bitstream by its id.
247      * 
248      * @param id
249      *            the bitstream id
250      * 
251      * @return the bitstream information needed for checksum validation. Returns
252      *         null if bitstream info isn't found.
253      */
254     public BitstreamInfo findByBitstreamId(int id)
255     {
256         Connection conn = null;
257         BitstreamInfo info = null;
258         PreparedStatement prepStmt = null;
259 
260         try
261         {
262             // create the connection and execute the statement
263             conn = DatabaseManager.getConnection();
264 
265             prepStmt = conn.prepareStatement(FIND_BY_BITSTREAM_ID);
266 
267             prepStmt.setInt(1, id);
268 
269             ResultSet rs = prepStmt.executeQuery();
270 
271             // if the bitstream is found return it
272             if (rs.next())
273             {
274                 info = new BitstreamInfo(rs.getBoolean("deleted"), rs
275                         .getInt("store_number"), rs.getLong("size_bytes"), rs
276                         .getString("short_description"), rs
277                         .getInt("bitstream_id"), rs
278                         .getString("user_format_description"), rs
279                         .getString("internal_id"), rs.getString("source"), rs
280                         .getString("checksum_algorithm"), rs
281                         .getString("checksum"), rs.getString("name"), rs
282                         .getTimestamp("last_process_end_date"), rs
283                         .getBoolean("to_be_processed"), new Date());
284             }
285         }
286         catch (SQLException e)
287         {
288             LOG.warn("Bitstream metadata could not be retrieved. "
289                     + e.getMessage(), e);
290         }
291         finally
292         {
293             cleanup(prepStmt, conn);
294         }
295 
296         return info;
297     }

300      * Queries the bitstream table for bitstream IDs that are not yet in the
301      * most_recent_checksum table, and inserts them into the
302      * most_recent_checksum and checksum_history tables.
303      */
304     public void updateMissingBitstreams()
305     {
306         Connection conn = null;
307         PreparedStatement stmt = null;
308 
309         try
310         {
311             LOG.debug("updating missing bitstreams");
312             conn = DatabaseManager.getConnection();
313             if ("oracle".equals(ConfigurationManager.getProperty("db.name")))
314                 stmt = conn.prepareStatement(INSERT_MISSING_CHECKSUM_BITSTREAMS_ORACLE);
315             else
316             	stmt = conn.prepareStatement(INSERT_MISSING_CHECKSUM_BITSTREAMS);
317             stmt.setTimestamp(1, new java.sql.Timestamp(new Date().getTime()));
318             stmt.setTimestamp(2, new java.sql.Timestamp(new Date().getTime()));
319             stmt.executeUpdate();
320 
321             checksumHistoryDAO.updateMissingBitstreams(conn);
322             conn.commit();
323         }
324         catch (SQLException e)

327                     "Problem inserting missing bitstreams. " + e.getMessage(),

338     /**
339      * Deletes the bitstream from the most_recent_checksum table if it exist.
340      * 
341      * @param id
342      *            the bitstream id.
343      * 
344      * @return number of records deleted
345      */
346     protected int deleteBitstreamInfo(int id, Connection conn)
347     {
348         PreparedStatement stmt = null;
349 
350         int numDeleted = 0;
351 
352         try
353         {
354             stmt = conn.prepareStatement(DELETE_BITSTREAM_INFO);
355             stmt.setInt(1, id);
356 
357             numDeleted = stmt.executeUpdate();
358 
359             if (numDeleted > 1)
360             {
361                 conn.rollback();
362                 throw new IllegalStateException(
363                         "Too many rows deleted! Number of rows deleted: "
364                                 + numDeleted
365                                 + " only one row should be deleted for bitstream id "
366                                 + id);
367             }
368         }
369         catch (SQLException e)
370         {
371             LOG.error("Problem deleting bitstream. " + e.getMessage(), e);
372             throw new RuntimeException("Problem deleting bitstream. "
373                     + e.getMessage(), e);
374         }
375         finally
376         {
377             cleanup(stmt);
378         }
379 
380         return numDeleted;
381     }
