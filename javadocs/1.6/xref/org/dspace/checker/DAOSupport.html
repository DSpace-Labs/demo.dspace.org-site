33  package org.dspace.checker;
34  
35  import java.sql.Connection;
36  import java.sql.ResultSet;
37  import java.sql.SQLException;
38  import java.sql.Statement;
39  
40  import org.apache.log4j.Logger;
41  import org.dspace.storage.rdbms.DatabaseManager;
42  
43  /**
44   * Database Helper Class to cleanup database resources
45   * 
46   * @author Jim Downing
47   * @author Grace Carpenter
48   * @author Nathan Sarr
49   * 
50   */
51  public class DAOSupport
52  {
53  
54      private static final Logger LOG = Logger.getLogger(DAOSupport.class);
55  
56      /**
57       * Utility method that cleans up the statement and connection.
58       * 
59       * @param stmt
60       *            A prepared statement to close.
61       * @param conn
62       *            Corresponding connection to close.
63       */
64      protected void cleanup(Statement stmt, Connection conn)
65      {
66          cleanup(stmt);
67          if (conn != null)
68          {
69              DatabaseManager.freeConnection(conn);
70          }
71      }
72  
73      /**
74       * Utility method that cleans up the statement and connection.
75       * 
76       * @param stmt
77       *            A prepared statement to close.
78       * @param conn
79       *            Corresponding connection to close.
80       * @param rs
81       *            Result set to close
82       */
83      protected void cleanup(Statement stmt, Connection conn, ResultSet rs)
84      {
85          if (rs != null)
86          {
87              try
88              {
89                  rs.close();
90              }
91              catch (SQLException e)
92              {
93                  LOG.warn("Problem closing result set. " + e.getMessage(), e);
94              }
95          }
96          cleanup(stmt);
97  
98          if (conn != null)
99          {
100             DatabaseManager.freeConnection(conn);
101         }
102     }
103 
104     protected void cleanup(Statement stmt)
105     {
106         if (stmt != null)
107         {
108             try
109             {
110                 stmt.close();
111             }
112             catch (SQLException e)
113             {
114                 LOG.warn("Problem closing prepared statement. "
115                         + e.getMessage(), e);
116             }
117         }
118     }
119 
120     protected void cleanup(Connection conn)
121     {
122         if (conn != null)
123         {
124             try
125             {
126                 conn.close();
127             }
128             catch (SQLException e)
129             {
130                 LOG.warn(e);
131             }
132         }
133     }
134 
135 }
