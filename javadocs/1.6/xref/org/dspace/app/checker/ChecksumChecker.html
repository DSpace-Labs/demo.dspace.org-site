61  /**
62   * Command line access to the checksum checker. Options are listed in the 
63   * documentation for the main method.</p>
64   * 
65   * @author Jim Downing
66   * @author Grace Carpenter
67   * @author Nathan Sarr
68   */
69  public class ChecksumChecker
70  {
71      private static final Logger LOG = Logger.getLogger(ChecksumChecker.class);
72  
73      /**
74       * Blanked off constructor, this class should be used as a command line
75       * tool.
76       * 
77       */
78      private ChecksumChecker()
79      {
80          ;
81      }
82  
83      /**
84       * Command line access to the checksum package.
85       * 
86       * @param args
87       *            <dl>
88       *            <dt>-h</dt>
89       *            <dd>Print help on command line options</dd>
90       *            <dt>-l</dt>
91       *            <dd>loop through bitstreams once</dd>
92       *            <dt>-L</dt>
93       *            <dd>loop continuously through bitstreams</dd>
94       *            <dt>-d</dt>
95       *            <dd>specify duration of process run</dd>
96       *            <dt>-b</dt>
97       *            <dd>specify bitstream IDs</dd>
98       *            <dt>-a [handle_id]</dt>
99       *            <dd>check anything by handle</dd>
100      *            <dt>-e</dt>
101      *            <dd>Report only errors in the logs</dd>
102      *            <dt>-p</dt>
103      *            <dd>Don't prune results before running checker</dd>
104      *            </dl>
105      */
106     public static void main(String[] args)
107     {
108         // set up command line parser
109         CommandLineParser parser = new PosixParser();
110         CommandLine line = null;
111 
112         // create an options object and populate it
113         Options options = new Options();
114 
115         options.addOption("l", "looping", false, "Loop once through bitstreams");
116         options.addOption("L", "continuous", false,
117                 "Loop continuously through bitstreams");
118         options.addOption("h", "help", false, "Help");
119         options.addOption("d", "duration", true, "Checking duration");
120         options.addOption("c", "count", true, "Check count");
121         options.addOption("a", "handle", true, "Specify a handle to check");
122         options.addOption("v", "verbose", false, "Report all processing");
123 
124         OptionBuilder.withArgName("bitstream-ids").hasArgs().withDescription(
125                 "Space separated list of bitstream ids");
126         Option useBitstreamIds = OptionBuilder.create('b');
127 
128         options.addOption(useBitstreamIds);
129 
130         options.addOption("p", "prune", false, "Prune configuration file");
131         options
132                 .addOption(OptionBuilder
133                         .withArgName("prune")
134                         .hasOptionalArgs(1)
135                         .withDescription(
136                                 "Prune old results (optionally using specified properties file for configuration)")
137                         .create('p'));
138 
139         try
140         {
141             line = parser.parse(options, args);
142         }
143         catch (ParseException e)
144         {
145             LOG.fatal(e);
146             System.exit(1);
147         }
148 
149         // user asks for help
150         if (line.hasOption('h'))
151         {
152             printHelp(options);
153         }
154 
155         // Prune stage
156         if (line.hasOption('p'))
157         {
158             ResultsPruner rp = null;
159             try
160             {
161                 rp = (line.getOptionValue('p') != null) ? ResultsPruner
162                         .getPruner(line.getOptionValue('p')) : ResultsPruner
163                         .getDefaultPruner();
164             }
165             catch (FileNotFoundException e)
166             {
167                 e.printStackTrace();
168                 System.exit(1);
169             }
170             int count = rp.prune();
171             System.out.println("Pruned " + count
172                     + " old results from the database.");
173         }
174 
175         Date processStart = Calendar.getInstance().getTime();
176 
177         BitstreamDispatcher dispatcher = null;
178         
179         // process should loop infinitely through
180         // most_recent_checksum table
181         if (line.hasOption('l'))
182         {
183             dispatcher = new SimpleDispatcher(new BitstreamInfoDAO(), processStart, false); 
184         }
185         else if (line.hasOption('L'))
186         {
187             dispatcher = new SimpleDispatcher(new BitstreamInfoDAO(), processStart, true);
188         }
189         else if (line.hasOption('b'))
190         {
191             // check only specified bitstream(s)
192             String[] ids = line.getOptionValues('b');
193             List idList = new ArrayList(ids.length);
194 
195             for (int i = 0; i < ids.length; i++)
196             {
197                 try
198                 {
199                     idList.add(new Integer(ids[i]));
200                 }
201                 catch (NumberFormatException nfe)
202                 {
203                     System.err.println("The following argument: " + ids[i]
204                             + " is not an integer");
205                     System.exit(0);
206                 }
207             }
208             dispatcher = new ListDispatcher(idList);
209         }
210 
211         else if (line.hasOption('a'))
212         {
213             dispatcher = new HandleDispatcher(new BitstreamInfoDAO(), line.getOptionValue('a'));
214         }
215         else if (line.hasOption('d'))
216         {
217             // run checker process for specified duration
218             try
219             {
220                 dispatcher = new LimitedDurationDispatcher(
221                         new SimpleDispatcher(new BitstreamInfoDAO(), processStart, true), new Date(
222                                 System.currentTimeMillis()
223                                         + Utils.parseDuration(line
224                                                 .getOptionValue('d'))));
225             }
226             catch (Exception e)
227             {
228                 LOG.fatal("Couldn't parse " + line.getOptionValue('d')
229                         + " as a duration: ", e);
230                 System.exit(0);
231             }
232         }
233         else if (line.hasOption('c'))
234         {
235         	int count = new Integer(line.getOptionValue('c')).intValue();
236             
237         	// run checker process for specified number of bitstreams
238             dispatcher = new LimitedCountDispatcher(new SimpleDispatcher(
239                     new BitstreamInfoDAO(), processStart, false), count);
240         }
241         else
242         {
243             dispatcher = new LimitedCountDispatcher(new SimpleDispatcher(
244                     new BitstreamInfoDAO(), processStart, false), 1);
245         }
246         
247         ResultsLogger logger = new ResultsLogger(processStart);
248         CheckerCommand checker = new CheckerCommand();
249         // verbose reporting
250         if (line.hasOption('v'))
251         {
252             checker.setReportVerbose(true);
253         }
254 
255         checker.setProcessStartDate(processStart);
256         checker.setDispatcher(dispatcher);
257         checker.setCollector(logger);
258         checker.process();
259         System.exit(0);
260     }
261 
262     /**
263      * Print the help options for the user
264      * 
265      * @param options that are available for the user
266      */
267     private static void printHelp(Options options)
268     {
269         HelpFormatter myhelp = new HelpFormatter();
270 
271         myhelp.printHelp("Checksum Checker\n", options);
272         System.out
273                 .println("\nSpecify a duration for checker process, using s(seconds),"
274                         + "m(minutes), or h(hours): ChecksumChecker -d 30s"
275                         + " OR ChecksumChecker -d 30m"
276                         + " OR ChecksumChecker -d 2h");
277         System.out
278                 .println("\nSpecify bitstream IDs: ChecksumChecker -b 13 15 17 20");
279         System.out.println("\nLoop once through all bitstreams: "
280                 + "ChecksumChecker -l");
281         System.out
282                 .println("\nLoop continuously through all bitstreams: ChecksumChecker -L");
283         System.out
284                 .println("\nCheck a defined number of bitstreams: ChecksumChecker -c 10");
285         System.out.println("\nReport all processing (verbose)(default reports only errors): ChecksumChecker -v");
286         System.out.println("\nDefault (no arguments) is equivalent to '-c 1'");
287         System.exit(0);
288     }
