102     /**
103      * Authenticate the current DSpace content based upon given authentication
104      * credentials. The AuthenticationManager will consult the configured
105      * authentication stack to determine the best method.
106      * 
107      * @param objectModel
108      *            Cocoon's object model.
109      * @param email
110      *            The email credentials provided by the user.
111      * @param password
112      *            The password credentials provided by the user.
113      * @param realm
114      *            The realm credentials proveded by the user.
115      * @return Return a current context with either the eperson attached if the
116      *         authentication was successfull or or no eperson attached if the
117      *         attempt failed.
118      */
119     public static Context Authenticate(Map objectModel, String email, String password, String realm) 
120     throws SQLException
121     {
122         // Get the real HttpRequest
123         HttpServletRequest request = (HttpServletRequest) objectModel
124                 .get(HttpEnvironment.HTTP_REQUEST_OBJECT);
125         Context context = ContextUtil.obtainContext(objectModel);
126 
127         int implicitStatus = AuthenticationManager.authenticateImplicit(
128                 context, null, null, null, request);
129 
130         if (implicitStatus == AuthenticationMethod.SUCCESS)
131         {
132             log.info(LogManager.getHeader(context, "login", "type=implicit"));
133             AuthenticationUtil.logIn(context, request, context.getCurrentUser());
134         }
135         else
136         {
137             // If implicit authentication failed, fall over to explicit.
138 
139             int explicitStatus = AuthenticationManager.authenticate(context,
140                     email, password, realm, request);
141 
142             if (explicitStatus == AuthenticationMethod.SUCCESS)
143             {
144                 // Logged in OK.
145                 log.info(LogManager
146                         .getHeader(context, "login", "type=explicit"));
147                 AuthenticationUtil.logIn(context, request, context
148                         .getCurrentUser());
149             }
150             else
151             {
152                 log.info(LogManager.getHeader(context, "failed_login", "email="
153                         + email + ", realm=" + realm + ", result="
154                         + explicitStatus));
155             }
156         }
157 
158         return context;
159     }

192     /**
193      * Log the given user in as a real authenticated user. This should only be used after 
194      * a user has presented their credintals and they have been validated. 
195      * 
196      * @param context
197      *            DSpace context
198      * @param request
199      *            HTTP request
200      * @param eperson
201      *            the eperson logged in
202      */
203     private static void logIn(Context context, HttpServletRequest request,
204             EPerson eperson) throws SQLException
205     {
206         if (eperson == null)
207             return;
208         
209         HttpSession session = request.getSession();
210 
211         context.setCurrentUser(eperson);
212 
213         // Check to see if systemwide alerts is restricting sessions
214         if (!AuthorizeManager.isAdmin(context) && !SystemwideAlerts.canUserStartSession())
215         {
216         	// Do not allow this user to login because sessions are being restricted by a systemwide alert.
217         	context.setCurrentUser(null);
218         	return;
219         }
220         
221         // Set any special groups - invoke the authentication mgr.
222         int[] groupIDs = AuthenticationManager.getSpecialGroups(context,
223                 request);
224         for (int groupID : groupIDs)
225             context.setSpecialGroup(groupID);
226 
227         // and the remote IP address to compare against later requests
228         // so we can detect session hijacking.
229         session.setAttribute(CURRENT_IP_ADDRESS, request.getRemoteAddr());
230         
231         // Set both the effective and authenticated user to the same.
232         session.setAttribute(EFFECTIVE_USER_ID, eperson.getID());
233         session.setAttribute(AUTHENTICATED_USER_ID,eperson.getID());
234         
235     }

257     /**
258      * Check to see if there are any session attributes indicating a currently authenticated 
259      * user. If there is then log this user in.
260      * 
261      * @param context
262      *            DSpace context
263      * @param request
264      *            HTTP Request
265      */
266     public static void resumeLogin(Context context, HttpServletRequest request)
267             throws SQLException
268     {
269         HttpSession session = request.getSession(false);
270 
271         if (session != null)
272         {
273             Integer id = (Integer) session.getAttribute(EFFECTIVE_USER_ID);
274             Integer realid = (Integer) session.getAttribute(AUTHENTICATED_USER_ID);
275             
276             if (id != null)
277             {
278                 String address = (String)session.getAttribute(CURRENT_IP_ADDRESS);
279                 if (address != null && address.equals(request.getRemoteAddr()))
280                 {
281                     EPerson eperson = EPerson.find(context, id);
282                     context.setCurrentUser(eperson);
283                     
284                     // Check to see if systemwide alerts is restricting sessions
285                     if (!AuthorizeManager.isAdmin(context) && !SystemwideAlerts.canUserMaintainSession())
286                     {
287                     	// Normal users can not maintain their sessions, check to see if this is really an
288                     	// administrator loging in as someone else.
289                     	
290                     	EPerson realEPerson = EPerson.find(context, realid);
291                     	Group administrators = Group.find(context,1);
292                  	    if (!administrators.isMember(realEPerson))
293                  	    {
294                  	    	// Log this user out because sessions are being restricted by a systemwide alert.
295                  	    	context.setCurrentUser(null);
296                  	    	return;
297                  	    }
298                     }
299                     
300 
301                     // Set any special groups - invoke the authentication mgr.
302                     int[] groupIDs = AuthenticationManager.getSpecialGroups(context, request);
303                     for (int groupID : groupIDs)
304                         context.setSpecialGroup(groupID);
305                 }
306                 else
307                 {
308                     // Possible hack attempt.
309                 }
310             } // if id
311         } // if session
312     }

314     /**
315      * Assume the login as another user. Only site administrators may preform the action.
316      * 
317      * @param context
318      * 		The current DSpace context loged in as a site administrator
319      * @param request
320      * 		The reall HTTP request.
321      * @param loginAs
322      * 		Whom to login as.
323      */
324     public static void loginAs(Context context, HttpServletRequest request, EPerson loginAs ) 
325     throws SQLException, AuthorizeException
326     {
327     	// Only allow loginAs if the administrator has allowed it.
328     	if (!ConfigurationManager.getBooleanProperty("xmlui.user.assumelogin", false))
329     		return;
330     	
331     	// Only super administrators can login as someone else.
332     	if (!AuthorizeManager.isAdmin(context))
333     		throw new AuthorizeException("Only site administrators may assume login as another user.");
334     		
335     	// Just to be double be sure, make sure the administrator
336     	// is the one who actualy authenticated themself.
337 	    HttpSession session = request.getSession(false);
338 	    Integer authenticatedID = (Integer) session.getAttribute(AUTHENTICATED_USER_ID); 
339 	    if (context.getCurrentUser().getID() != authenticatedID)
340 	    	throw new AuthorizeException("Only authenticated users whom are administrators may assume the login as another user.");
341 	    
342 	    // You may not assume the login of another super administrator
343 	    if (loginAs == null)
344 	    	return;
345 	    Group administrators = Group.find(context,1);
346 	    if (administrators.isMember(loginAs))
347 	    	throw new AuthorizeException("You may not assume the login as another super administrator.");
348 	    
349 	    // Success, allow the user to login as another user.
350 	    context.setCurrentUser(loginAs);
351 	
352         // Set any special groups - invoke the authentication mgr.
353         int[] groupIDs = AuthenticationManager.getSpecialGroups(context,request);
354         for (int groupID : groupIDs)
355             context.setSpecialGroup(groupID);
356 	    	        
357         // Set both the effective and authenticated user to the same.
358         session.setAttribute(EFFECTIVE_USER_ID, loginAs.getID());
359     }

362     /**
363      * Log the user out.
364      * 
365      * @param context
366      *            DSpace context
367      * @param request
368      *            HTTP request
369      */
370     public static void logOut(Context context, HttpServletRequest request) throws SQLException
371     {
372         HttpSession session = request.getSession();
373 
374         if (session.getAttribute(EFFECTIVE_USER_ID) != null &&
375         	session.getAttribute(AUTHENTICATED_USER_ID) != null)
376         {
377     	    Integer effectiveID = (Integer) session.getAttribute(EFFECTIVE_USER_ID); 
378     	    Integer authenticatedID = (Integer) session.getAttribute(AUTHENTICATED_USER_ID); 
379     	    
380     	    if (effectiveID.intValue() != authenticatedID.intValue())
381     	    {
382     	    	// The user has login in as another user, instead of logging them out, 
383     	    	// revert back to their previous login name.
384     	    	
385     	    	EPerson authenticatedUser = EPerson.find(context, authenticatedID);
386     	    	context.setCurrentUser(authenticatedUser);
387     	    	session.setAttribute(EFFECTIVE_USER_ID, authenticatedID);
388     	    	return;
389     	    }
390         }
391         
392         // Otherwise, just log the person out as normal.
393         context.setCurrentUser(null);
394         session.removeAttribute(EFFECTIVE_USER_ID);
395         session.removeAttribute(AUTHENTICATED_USER_ID);
396         session.removeAttribute(CURRENT_IP_ADDRESS);
397     }

503      * Interrupt the current request and store if for later resumption. This request will
504      * send an http redirect telling the client to authenticate first. Once that has been finished
505      * then the request can be resumed.
506      * 
507      * @param objectModel The Cocoon object Model
508      * @param header A message header (i18n tag)
509      * @param message A message for why the request was interrupted (i18n tag)
510      * @param characters An untranslated messsage, perhaps an error message?
511      */
512     public static void interruptRequest(Map objectModel, String header, String message, String characters)
513     {
514     	final HttpServletRequest request = 
515     		(HttpServletRequest) objectModel.get(HttpEnvironment.HTTP_REQUEST_OBJECT);        
516 
517     	HttpSession session = request.getSession();
518         
519         // Store this interrupted request untill after the user successfully authenticates.
520         RequestInfo interruptedRequest = new RequestInfo(request);
521         
522         // Set the request as interrupted
523         session.setAttribute(REQUEST_INTERRUPTED,interruptedRequest);
524         session.setAttribute(REQUEST_RESUME, null); // just to be clear.
525         
526         // Set the interrupt message
527         session.setAttribute(REQUEST_INTERRUPTED_HEADER, header);
528         session.setAttribute(REQUEST_INTERRUPTED_MESSAGE, message);
529         session.setAttribute(REQUEST_INTERRUPTED_CHARACTERS, characters);
530         
531         
532     }

535     /**
536      * Set the interrupted request to a resumable state. The
537      * next request that the server recieves (for this session) that
538      * has the same servletPath will be replaced with the previously
539      * inturrupted request.
540      * 
541      * @param objectModel The Cocoon object Model
542      * @return
543      */
544     public static String resumeInterruptedRequest(Map objectModel)
545     {
546     	final HttpServletRequest request = 
547     		(HttpServletRequest) objectModel.get(HttpEnvironment.HTTP_REQUEST_OBJECT);        
548 
549     	HttpSession session = request.getSession();
550         
551     	// Clear the interrupt message
552 	    session.setAttribute(AuthenticationUtil.REQUEST_INTERRUPTED_HEADER, null);
553 	    session.setAttribute(AuthenticationUtil.REQUEST_INTERRUPTED_MESSAGE, null);
554 	    session.setAttribute(AuthenticationUtil.REQUEST_INTERRUPTED_CHARACTERS, null);
555     	
556         // Set the request as interrupted
557         Object interruptedObject = session.getAttribute(REQUEST_INTERRUPTED);
558         if (interruptedObject instanceof RequestInfo)
559         {
560         	RequestInfo interruptedRequest = (RequestInfo) interruptedObject;
561         	
562         	session.setAttribute(REQUEST_INTERRUPTED, null);
563         	session.setAttribute(REQUEST_RESUME, interruptedRequest); 
564         	
565         	// Return the path for which this request belongs too. Only urls
566         	// for this path may be resumed.
567         	return interruptedRequest.getServletPath();
568         }
569         
570         // No request was interrupted.
571         return null;
572     }

576     /**
577      * Check to see if this request should be resumed.
578      * 
579      * @param realHttpRequest The current real request
580      * @return Either the current real request or a stored request that was previously interrupted.
581      */
582     public static HttpServletRequest resumeRequest(HttpServletRequest realHttpRequest) 
583     {
584     	// First check to see if there is a resumed request.
585     	HttpSession session = realHttpRequest.getSession();
586     	//session.setMaxInactiveInterval(60);
587         Object object = session.getAttribute(REQUEST_RESUME);
588     	
589         // Next check to make sure it's the right type of object, 
590         // there should be no condition where it is not - but always 
591         // safe to check.
592         if (object instanceof RequestInfo)
593         {
594         	RequestInfo interruptedRequest = (RequestInfo) object;
595         
596         	// Next, check to make sure this real request if for the same url
597         	// path, if so then resume the previous request.
598         	String interruptedServletPath = interruptedRequest.getServletPath();
599         	String realServletPath = realHttpRequest.getServletPath();
600         	
601         	if (realServletPath != null && 
602         		realServletPath.equals(interruptedServletPath))
603         	{
604         		// Clear the resumed request and send the request back to be resumed.
605         		session.setAttribute(REQUEST_INTERRUPTED, null);
606         		session.setAttribute(REQUEST_RESUME, null);
607  
608         		return interruptedRequest.wrapRequest(realHttpRequest);
609         	}
610         }
611     	// Otherwise return the real request.
612     	return realHttpRequest;
613     }
