41  /**
42   * A class representing an an abstract input control (which is just a fancy name
43   * for a field :) )
44   *
45   * The field element is a container for all information necessary to create a
46   * form field. The required "type" attribute determines the type of the field,
47   * while the children tags carry the information on how to build it. Fields can
48   * only occur in divisions of type "interactive".
49   *
50   * There are several types of possible fields and each of these field types
51   * determine the appropriate parameters on the parameter object. This is the
52   * only place in the schema where this design pattern is used. It limits the
53   * proliferation of elements, such as a special element for textarea, select
54   * lists, text fields etc... as HTML does. It also forces us to treat all fields
55   * the same.
56   *
57   * text: A single-line text input control.
58   *
59   * textarea: A multi-line text input control.
60   *
61   * password: A single-line text input control where the input text is rendered
62   * in such a way as to hide the characters from the user.
63   *
64   * hidden: An input control that is not rendered on the screen and hidden from
65   * the user.
66   *
67   * button: A button input control that when activated by the user will submit
68   * the form, including all the fields, back to the server for processing.
69   *
70   * checkbox: A boolean input control which may be toggled by the user. A
71   * checkbox may have several fields which share the same name and each of those
72   * fields may be toggled independently. This is distinct from a radio button
73   * where only one field may be toggled.
74   *
75   * file: An input control that allows the user to select files to be submitted
76   * with the form. Note that a form which uses a file field must use the
77   * multipart method.
78   *
79   * radio: A boolean input control which may be toggled by the user. Multiple
80   * radio button fields may share the same name. When this occurs only one field
81   * may be selected to be true. This is distinct from a checkbox where multiple
82   * fields may be toggled.
83   *
84   * select: A menu input control which allows the user to select from a list of
85   * available options.
86   *
87   * composite: A combination of multile fields into one input control.
88   *
89   * @author Scott Phillips
90   */

487     /**
488      * Private function to remove all values of a particular type.
489      *
490      * @param removeType
491      *            The type to be removed.
492      */
493     protected void removeValueOfType(String removeType)
494     {
495         List<Value> found = new ArrayList<Value>();
496         for (Value value : values)
497             if (value.getType().equals(removeType))
498                 found.add(value);
499 
500         for (Value remove : found)
501         {
502             values.remove(remove);
503             remove.dispose();
504         }
505     }

507     /**
508      * Translate this element and all contained elements into SAX events. The
509      * events should be routed to the contentHandler found in the WingContext.
510      *
511      * @param contentHandler
512      *            (Required) The registered contentHandler where SAX events
513      *            should be routed too.
514      * @param lexicalHandler
515      *            (Required) The registered lexicalHandler where lexical events
516      *            (such as CDATA, DTD, etc) should be routed too.
517      * @param namespaces
518      *            (Required) SAX Helper class to keep track of namespaces able
519      *            to determine the correct prefix for a given namespace URI.
520      */
521     public void toSAX(ContentHandler contentHandler,
522             LexicalHandler lexicalHandler, NamespaceSupport namespaces)
523             throws SAXException
524     {
525         AttributeMap attributes = new AttributeMap();
526 
527         attributes.put(A_NAME, this.name);
528         attributes.put(A_ID, this.context.generateID(E_FIELD, this.name));
529         attributes.put(A_FIELD_TYPE, this.type);
530         if (this.disabled)
531             attributes.put(A_DISABLED, this.disabled);
532         if (this.required)
533             attributes.put(A_REQUIRED, this.required);
534         if (this.rend != null)
535             attributes.put(A_RENDER, this.rend);
536 
537         startElement(contentHandler, namespaces, E_FIELD, attributes);
538 
539         if (params != null)
540             params.toSAX(contentHandler, lexicalHandler, namespaces);
541 
542         if (label != null)
543                 label.toSAX(contentHandler, lexicalHandler, namespaces);
544         
545         if (help != null)
546             help.toSAX(contentHandler, lexicalHandler, namespaces);
547 
548         for (Error error : errors)
549             error.toSAX(contentHandler, lexicalHandler, namespaces);
550         
551         for (Field field : fields)
552             field.toSAX(contentHandler, lexicalHandler, namespaces);
553 
554         for (Option option : options)
555             option.toSAX(contentHandler, lexicalHandler, namespaces);
556         
557         for (Value value : values)
558             value.toSAX(contentHandler, lexicalHandler, namespaces);
559         
560         for (Instance instance : instances)
561             instance.toSAX(contentHandler, lexicalHandler, namespaces);
562 
563         endElement(contentHandler, namespaces, E_FIELD);
564     }

574         if (label != null)

587         if (fields != null)
588             fields.clear();
589         
590         for (Option option : options)

593             options.clear();
594         
595         for (Value value : values)
596             value.dispose();
597         if (values != null)
598             values.clear();
599 
600         for (Instance instance : instances)
601             instance.dispose();
602         if (instances != null)

608         help = null;
609         errors = null;
610         fields = null;
611         options = null;
612         values = null;
613         instances = null;
614         super.dispose();
615     }
