41  package org.dspace.app.xmlui.wing.element;
42  
43  /**
44   * A class representing a table cell.
45   * 
46   * The cell element contained in a row of a table carries content for that
47   * table. It is a character container, just like p, item, and hi, and its
48   * primary purpose is to display textual data, possibly enhanced with
49   * hyperlinks, emphasized blocks of text, images and form fields.
50   * 
51   * @author Scott Phillips
52   */
53  
54  import org.dspace.app.xmlui.wing.AttributeMap;
55  import org.dspace.app.xmlui.wing.WingContext;
56  import org.dspace.app.xmlui.wing.WingException;
57  import org.xml.sax.ContentHandler;
58  import org.xml.sax.SAXException;
59  import org.xml.sax.ext.LexicalHandler;
60  import org.xml.sax.helpers.NamespaceSupport;
61  
62  public class Cell extends RichTextContainer implements StructuralElement
63  {
64      /** The name of the cell element */
65      public static final String E_CELL = "cell";
66  
67      /** The name of the role attribute */
68      public static final String A_ROLE = "role";
69  
70      /** The name of the rows attribute */
71      public static final String A_ROWS = "rows";
72  
73      /** The name of the cols attribute */
74      public static final String A_COLS = "cols";
75  
76      /** The name of this cell */
77      private String name;
78  
79      /** The role of this cell, see ROLES below */
80      private String role;
81  
82      /** How many rows does this table span */
83      private int rows;
84  
85      /** How many cols does this table span */
86      private int cols;
87  
88      /** Special rendering instructions */
89      private String rend;
90  
91      /** The possible cell role types */
92      public static final String ROLE_DATA = "data";
93  
94      public static final String ROLE_HEADER = "header";
95  
96      /** All the possible cell role types collected into one array */
97      public static final String[] ROLES = { ROLE_DATA, ROLE_HEADER };
98  
99      /**
100      * Construct a new cell.
101      * 
102      * @param context
103      *            (Required) The context this element is contained in, such as
104      *            where to route SAX events and what i18n catalogue to use.
105      * @param name
106      *            (May be null) a local identifier used to differentiate the
107      *            element from its siblings.
108      * @param role
109      *            (May be null) determines what kind of information the cell
110      *            carries, either header or data. See cell.ROLES
111      * @param rows
112      *            (May be zero for no defined value) determines how many rows
113      *            does this cell span.
114      * @param cols
115      *            (May be zero for no defined value) determines how many columns
116      *            does this cell span.
117      * @param rend
118      *            (May be null) a rendering hint used to override the default
119      *            display of the element.
120      */
121     protected Cell(WingContext context, String name, String role, int rows,
122             int cols, String rend) throws WingException
123     {
124         super(context);
125         restrict(role, ROLES,
126                 "The 'role' parameter must be one of these values: 'data' or 'header'.");
127         greater(rows, -1,
128                 "The 'rows' parameter must be greater than or equal to zero.");
129         greater(cols, -1,
130                 "The 'cols' parameter must be greater than or equal to zero.");
131 
132         this.name = name;
133         this.role = role;
134         this.rows = rows;
135         this.cols = cols;
136         this.rend = rend;
137     }
138 
139     /**
140      * Translate this element and all contained elements into SAX events. The
141      * events should be routed to the contentHandler found in the WingContext.
142      * 
143      * @param contentHandler
144      *            (Required) The registered contentHandler where SAX events
145      *            should be routed too.
146      * @param lexicalHandler
147      *            (Required) The registered lexicalHandler where lexical 
148      *            events (such as CDATA, DTD, etc) should be routed too.
149      * @param namespaces
150      *            (Required) SAX Helper class to keep track of namespaces able
151      *            to determine the correct prefix for a given namespace URI.
152      */
153     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler,
154             NamespaceSupport namespaces) throws SAXException
155     {
156         AttributeMap attributes = new AttributeMap();
157         if (name != null)
158         {
159             attributes.put(A_NAME, name);
160             attributes.put(A_ID, context.generateID(E_CELL, name));
161         }
162 
163         if (role != null)
164             attributes.put(A_ROLE, role);
165         // else
166         // attributes.put(A_ROLE, ROLE_DATA);
167         if (rows > 0)
168             attributes.put(A_ROWS, rows);
169         if (cols > 0)
170             attributes.put(A_COLS, cols);
171         if (rend != null)
172             attributes.put(A_RENDER, rend);
173 
174         startElement(contentHandler, namespaces, E_CELL, attributes);
175         super.toSAX(contentHandler, lexicalHandler, namespaces);
176         endElement(contentHandler, namespaces, E_CELL);
177     }
178 
179     /**
180      * dispose
181      */
182     public void dispose()
183     {
184         for (AbstractWingElement content : contents)
185             content.dispose();
186         if (contents != null)
187             contents.clear();
188         contents = null;
189         super.dispose();
190     }
