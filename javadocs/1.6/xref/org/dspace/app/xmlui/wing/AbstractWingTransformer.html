56  /**
57   * This class handles receiving SAX events and translating them into DRI events.
58   * These DRI events are then routed to the individual implementing components
59   * where they fill in and construct the DRI document. The document they
60   * construct is known as the feeder document, this is merged into the main
61   * document that was generated from the previous component in the Cocoon
62   * pipeline. The merge takes place in accordance with the DRI schema's rules for
63   * merging two DRI documents.
64   * 
65   * 
66   * @author Scott Phillips
67   */
68  public abstract class AbstractWingTransformer extends AbstractTransformer
69          implements WingTransformer
70  {
71      /**
72       * Simple variable to indicate weather a new namespace context is needed. If
73       * several namespaces are declared on the same attribute then they are
74       * considered in the same 'context'. Each time an element is opened this
75       * flag is reset to true, and each time a new namespace is declared it is
76       * set to false. Using this information new contexts are opened
77       * conservatively.
78       */
79      private boolean needNewNamespaceContext = true;
80  
81      /**
82       * The namespace support object keeps track of registered URI prefixes. This
83       * is used by the WingElements so that they may attach the correctp prefix
84       * when assigning elements to namespaces.

86      private NamespaceSupport namespaces;

88      /**
89       * The feeder document is the document being merged into the main,
90       * pre-existing document, that is the result of the previous Cocoon
91       * component in the pipeline.
92       */
93      private WingDocument feederDocument;

95      /**
96       * The wing context is where the namespace support is stored along with the
97       * content and lexical handlers so that the wing elements can have access to
98       * them when they perform their toSAX() method.

100     private WingContext wingContext;

109      * Set up the transformer so that it can build a feeder Wing document and
110      * merge it into the main document
111      * 
112      * FIXME: Update document: - this method must be called to initialize the
113      * framework. It must be called after the component's setup has been called

117     public void setupWing() throws WingException
118     {
119         this.wingContext = new WingContext();
120         this.wingContext.setLogger(this.getLogger());
121         this.wingContext.setComponentName(this.getComponentName());
122         this.wingContext.setObjectManager(this.getObjectManager());

124         feederDocument = this.createWingDocument(wingContext);
125         this.stack = new Stack<WingMergeableElement>();

148     /**
149      * Begin the scope of a prefix-URI Namespace mapping.
150      * 
151      * @param prefix
152      *            The Namespace prefix being declared.
153      * @param uri
154      *            The Namespace URI the prefix is mapped to.
155      */
156     public void startPrefixMapping(String prefix, String uri)
157             throws SAXException
158     {
159         if (needNewNamespaceContext)
160         {
161             namespaces.pushContext();
162             needNewNamespaceContext = false;
163         }
164         namespaces.declarePrefix(prefix, uri);
165 
166         super.startPrefixMapping(prefix, uri);
167     }
168 
169     /**
170      * End the scope of a prefix-URI mapping.
171      * 
172      * @param prefix
173      *            The prefix that was being mapping.
174      */
175     public void endPrefixMapping(String prefix) throws SAXException
176     {
177         if (!needNewNamespaceContext)
178         {
179             namespaces.popContext();
180             needNewNamespaceContext = true;
181         }
182         super.endPrefixMapping(prefix);
183     }

185     /**
186      * Receive notification of the beginning of an element.
187      * 
188      * @param namespaceURI
189      *            The Namespace URI, or the empty string if the element has no
190      *            Namespace URI or if Namespace processing is not being
191      *            performed.
192      * @param localName
193      *            The local name (without prefix), or the empty string if
194      *            Namespace processing is not being performed.
195      * @param qName
196      *            The raw XML 1.0 name (with prefix), or the empty string if raw
197      *            names are not available.
198      * @param attributes
199      *            The attributes attached to the element. If there are no
200      *            attributes, it shall be an empty Attributes object.
201      */
202     public void startElement(String namespaceURI, String localName,
203             String qName, Attributes attributes) throws SAXException
204     {
205         // Reset the namespace context flag.
206         needNewNamespaceContext = true;
207 
208         try
209         {
210             if (stack == null)
211             {
212                 throw new WingException("Stack not initialized.");
213             }
214             
215             // Deal with the stack jump start issue of having a document all
216             // ready on the stack.
217             if (stack.size() == 0)
218             {
219                 if (feederDocument.mergeEqual(namespaceURI, localName, qName,
220                         attributes))
221                 {
222                     attributes = feederDocument.merge(attributes);
223                     stack.push(feederDocument);
224                 }
225                 else
226                 {
227                     throw new WingException(
228                             "Attempting to merge DRI documents but the source document is not compatable with the feeder document.");
229                 }
230 
231             }
232             else if (stack.size() > 0)
233             {
234                 WingMergeableElement peek = stack.peek();
235                 WingMergeableElement child = null;
236                 if (peek != null)
237                 {
238                     child = peek.mergeChild(namespaceURI, localName, qName,
239                             attributes);
240                 }
241 
242                 // Check if we should construct a new portion of the document.
243                 if (child instanceof UserMeta)
244                 {
245                     // Create the UserMeta
246                     this.addUserMeta((UserMeta) child);
247                 }
248                 else if (child instanceof PageMeta)
249                 {
250                     // Create the PageMeta
251                     this.addPageMeta((PageMeta) child);
252                 }
253                 else if (child instanceof Body)
254                 {
255                     // Create the Body
256                     this.addBody((Body) child);
257                 }
258                 else if (child instanceof Options)
259                 {
260                     // Create the Options
261                     this.addOptions((Options) child);
262                 }
263 
264                 // Update any attributes of this merged element.
265                 if (child != null)
266                     attributes = child.merge(attributes);
267                 stack.push(child);
268             }
269             // Send off the event with nothing modified except for the
270             // attributes (possibly)
271             super.startElement(namespaceURI, localName, qName, attributes);
272         }
273         catch (SAXException saxe)
274         {
275             throw saxe;
276         }
277         catch (Exception e)
278         {
279             handleException(e);
280         }
281 
282     }

284     /**
285      * Receive notification of the end of an element.
286      * 
287      * @param namespaceURI
288      *            The Namespace URI, or the empty string if the element has no
289      *            Namespace URI or if Namespace processing is not being
290      *            performed.
291      * @param localName
292      *            The local name (without prefix), or the empty string if
293      *            Namespace processing is not being performed.
294      * @param qName
295      *            The raw XML 1.0 name (with prefix), or the empty string if raw
296      *            names are not available.
297      */
298     public void endElement(String namespaceURI, String localName, String qName)
299             throws SAXException
300     {
301         try
302         {
303             if (stack.size() > 0)
304             {
305                 WingMergeableElement poped = stack.pop();
306                 if (poped != null)
307                 {
308 	                poped.toSAX(contentHandler, lexicalHandler, namespaces);
309 	                poped.dispose();
310                 }
311             }
312 
313             // Send the event on unmodified
314             super.endElement(namespaceURI, localName, qName);
315         }
316         catch (SAXException saxe)
317         {
318             throw saxe;
319         }
320         catch (Exception e)
321         {
322             handleException(e);
323         }
324     }

326     /**
327      * Handle exceptions that occurred during the document's creation. When
328      * errors occur a SAX event is being processed it will be sent through this
329      * method. This allows implementing classes to override this method for
330      * specific error handling hooks.
331      * 
332      * @param e
333      *            The thrown exception
334      */
335 
336     protected void handleException(Exception e) throws SAXException
337     {
338         throw new SAXException(
339                 "An error was incountered while processing the Wing based component: "
340                         + this.getClass().getName(), e);
341     }
