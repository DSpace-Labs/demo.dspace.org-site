41  package org.dspace.app.xmlui.wing.element;
42  
43  /**
44   * A class representing a hidden input control. The hidden input control is
45   * never displayed to the user but is passed back to the server when the user
46   * submits a form.
47   * 
48   * @author Scott Phillips
49   */
50  
51  import org.dspace.app.xmlui.wing.Message;
52  import org.dspace.app.xmlui.wing.WingContext;
53  import org.dspace.app.xmlui.wing.WingException;
54  
55  public class Hidden extends Field
56  {
57  
58      /**
59       * Construct a new field.
60       * 
61       * @param context
62       *            (Required) The context this element is contained in, such as
63       *            where to route SAX events and what i18n catalogue to use.
64       * 
65       * @param name
66       *            (Required) a non-unique local identifier used to differentiate
67       *            the element from its siblings within an interactive division.
68       *            This is the name of the field use when data is submitted back
69       *            to the server.
70       * @param rend
71       *            (May be null) a rendering hint used to override the default
72       *            display of the element.
73       */
74      protected Hidden(WingContext context, String name, String rend)
75              throws WingException
76      {
77          super(context, name, Field.TYPE_HIDDEN, rend);
78          this.params = new Params(context);
79      }
80  
81      /** ******************************************************************** */
82      /** Values * */
83      /** ******************************************************************** */
84  
85      /**
86       * Set the raw value of the field removing any previous raw values.
87       * 
88       * @param characters
89       *            (May be null) Field value as a string
90       */
91      public Value setValue() throws WingException
92      {
93          removeValueOfType(Value.TYPE_RAW);
94          Value value = new Value(context, Value.TYPE_RAW);
95          values.add(value);
96          return value;
97      }
98  
99      /**
100      * Set the raw value of the field removing any previous raw values.
101      * 
102      * @param characters
103      *            (May be null) Field value as a string
104      */
105     public void setValue(String characters) throws WingException
106     {
107         Value value = this.setValue();
108         value.addContent(characters);
109     }
110 
111     /**
112      * Set the raw value of the field removing any previous raw values.
113      * 
114      * @param integere
115      *            (May be null) Field value as a string
116      */
117     public void setValue(int integer) throws WingException
118     {
119     	setValue(String.valueOf(integer));
120     }
121     
122     /**
123      * Set the raw value of the field removing any previous raw values.
124      * 
125      * @param message
126      *            (Required) A key into the i18n catalogue for translation into
127      *            the user's preferred language.
128      */
129     public void setValue(Message message) throws WingException
130     {
131         Value value = this.setValue();
132         value.addContent(message);
133     }
134 }
