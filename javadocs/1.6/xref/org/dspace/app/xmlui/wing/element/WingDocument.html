53  /**
54   * A class representing a WingDocument.
55   * 
56   * Documents contain three elements and they are all mandatory: meta, body, and
57   * options. Because they are all mandatory they are created at construction
58   * time.
59   * 
60   * Note: We called the class "WingDocument" instead of just plain old "Document"
61   * so that it won't conflict with all the other documents out there like DOM's
62   * document.
63   * 
64   * @author Scott Phillips
65   */
66  public class WingDocument extends AbstractWingElement implements
67          WingMergeableElement
68  {
69      /** The name of the document element */
70      public static final String E_DOCUMENT = "document";
71  
72      /** The name of the version attribute */
73      public static final String A_VERSION = "version";
74  
75      /** The document version Wing prefer */
76      public static final String DOCUMENT_VERSION = "1.1";
77  
78      /** The divisions contained within this body */
79      private boolean merged = false;
80  
81      /** The meta element */
82      private Meta meta;
83  
84      /** The body element */
85      private Body body;
86  
87      /** The options element */
88      private Options options;
89  
90      /**
91       * Generate a new wing document element.
92       * 
93       * @param context
94       *            (Required) The context this element is contained in.
95       */
96      public WingDocument(WingContext context) throws WingException
97      {
98          super(context);
99  
100         // These are all required so we just create them now.
101         this.meta = new Meta(context);
102         this.body = new Body(context);
103         this.options = new Options(context);
104     }

139     /**
140      * Is this document the same as the given SAX event.
141      * 
142      * Note: this method will throw an error if the given event's document
143      * version number is out of bounds for this implementation of Wing.
144      * 
145      * @param namespace
146      *            The element's name space
147      * @param localName
148      *            The local, unqualified, name for this element
149      * @param qName
150      *            The qualified name for this element
151      * @param attributes
152      *            The element's attributes
153      * @return True if this WingElement is equivalent to the given SAX Event.
154      */
155     public boolean mergeEqual(String namespace, String localName, String qName,
156             Attributes attributes) throws SAXException, WingException
157     {
158         if (!WingConstants.DRI.URI.equals(namespace))
159             return false;
160         if (!E_DOCUMENT.equals(localName))
161             return false;
162         
163         
164         String version = attributes.getValue(A_VERSION);
165         if (!(DOCUMENT_VERSION.equals(version)))
166         	throw new WingException("Incompatable DRI versions, "+DOCUMENT_VERSION +" != "+version);
167         
168         return true;
169     }

171     /**
172      * Merge the given event into this document.
173      * 
174      * @param namespace
175      *            The element's name space
176      * @param localName
177      *            The local, unqualified, name for this element *
178      * @param qName
179      *            The qualified name for this element
180      * @param attributes
181      *            The element's attributes
182      * @return The child element
183      */
184     public WingMergeableElement mergeChild(String namespace, String localName,
185             String qName, Attributes attributes) throws SAXException,
186             WingException
187     {
188         if (this.meta != null)
189             if (this.meta.mergeEqual(namespace, localName, qName, attributes))
190             {
191                 Meta child = this.meta;
192                 this.meta = null;
193                 return child;
194             }
195         if (this.body != null)
196             if (this.body.mergeEqual(namespace, localName, qName, attributes))
197             {
198                 Body child = this.body;
199                 this.body = null;
200                 return child;
201             }
202         if (this.options != null)
203             if (this.options
204                     .mergeEqual(namespace, localName, qName, attributes))
205             {
206                 Options options = this.options;
207                 this.options = null;
208                 return options;
209             }
210         return null;
211     }

225     /**
226      * Translate this document to SAX events.
227      * 
228      * @param contentHandler
229      *            (Required) The registered contentHandler where SAX events
230      *            should be routed too.
231      * @param lexicalHandler
232      *            (Required) The registered lexicalHandler where lexical 
233      *            events (such as CDATA, DTD, etc) should be routed too.
234      * @param namespaces
235      *            (Required) SAX Helper class to keep track of namespaces able
236      *            to determine the correct prefix for a given namespace URI.
237      * 
238      */
239     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler,
240             NamespaceSupport namespaces)
241             throws SAXException
242     {
243         if (!this.merged)
244         {
245             AttributeMap attributes = new AttributeMap();
246             attributes.put(A_VERSION, DOCUMENT_VERSION);
247             startElement(contentHandler, namespaces, E_DOCUMENT, attributes);
248         }
249 
250         if (this.meta != null)
251             meta.toSAX(contentHandler, lexicalHandler, namespaces);
252         if (this.body != null)
253             body.toSAX(contentHandler, lexicalHandler, namespaces);
254         if (this.options != null)
255             options.toSAX(contentHandler, lexicalHandler, namespaces);
256 
257         if (!this.merged)
258         {
259             endElement(contentHandler, namespaces, E_DOCUMENT);
260         }
261     }
262 
263     /**
264      * dispose
265      */
266     public void dispose()
267     {
268         if (this.meta != null)
269             meta.dispose();
270         if (this.body != null)
271             body.dispose();
272         if (this.options != null)
273             options.dispose();
274         this.meta = null;
275         this.body = null;
276         this.options = null;
277         super.dispose();
278     }
