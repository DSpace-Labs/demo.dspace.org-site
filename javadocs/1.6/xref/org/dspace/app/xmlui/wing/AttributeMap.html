41  package org.dspace.app.xmlui.wing;
42  
43  import java.util.HashMap;
44  
45  /**
46   * A simplified Wing version of an AttributeMap. If a namespace is set it is
47   * applied to all attributes in the Map. This is typically not what one would
48   * expect but it makes the syntax easier for Wing Elements. Either all the
49   * attributes come from one namespace or the other, but never two at the same
50   * time.
51   * 
52   * @author Scott Phillips
53   */
54  
55  public class AttributeMap extends HashMap<String, String>
56  {
57      /** Just so there are no compile warnings */
58      private static final long serialVersionUID = 1;
59  
60      /** The namespace of ALL the attributes contained within this map */
61      private Namespace namespace;
62  
63      /**
64       * Set the namespace for all attributes contained within this map.
65       * 
66       * @param namespace
67       *            The new namespace.
68       */
69      public void setNamespace(Namespace namespace)
70      {
71          this.namespace = namespace;
72      }
73  
74      /**
75       * 
76       * @return The namespace for all the attributes contained within this map.
77       */
78      public Namespace getNamespace()
79      {
80          return this.namespace;
81      }
82  
83      /**
84       * Another variation of the put method to convert integer values into
85       * strings.
86       * 
87       * @param key
88       *            The attribute's name.
89       * @param value
90       *            The value of the attribute.
91       * @return
92       */
93      public String put(String key, int value)
94      {
95          return this.put(key, String.valueOf(value));
96      }
97  
98      /**
99       * Another variation of the put method to convert boolean values into
100      * strings. The values "yes" or "no" will be used in replacement of the
101      * boolean value.
102      * 
103      * @param key
104      * @param value
105      * @return
106      */
107     public String put(String key, boolean value)
108     {
109         return this.put(key, (value ? "yes" : "no"));
110     }
111 }
