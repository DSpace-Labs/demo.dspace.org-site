56  /**
57   * This class represents data, by data we mean the translated and untranslated
58   * characters in between XML elements.
59   * 
60   * When data needs to be translated it is enclosed inside the cocoon i18n schema
61   * while untranslated data is enclosed inside nothing.
62   * 
63   * @author Scott Phillips
64   */
65  
66  public class Data extends AbstractWingElement
67  {
68      /** The name of the text element */
69      public static final String E_TEXT = "text";
70  
71      /** The name of the translate element */
72      public static final String E_TRANSLATE = "translate";
73  
74      /** The name of the param element */
75      public static final String E_PARAM = "param";
76  
77      /** The name of the catalogue attribute (used inside text or i18n message) */
78      public static final String A_CATALOGUE = "catalogue";
79  
80      /** The name of the type attribute (used inside params) */
81      public static final String A_TYPE = "type";
82  
83      /** The name of the value attribute (used inside params) */
84      public static final String A_VALUE = "value";
85  
86      /** The date parameter type */
87      private static final String TYPE_DATE = "date";
88  
89      /** The number parameter type */
90      private static final String TYPE_NUMBER = "number";

134     /**
135      * Translate this element into SAX
136      * 
137      * @param contentHandler
138      *            (Required) The registered contentHandler where SAX events
139      *            should be routed too.
140      * @param lexicalHandler
141      *            (Required) The registered lexicalHandler where lexical 
142      *            events (such as CDATA, DTD, etc) should be routed too.
143      * @param namespaces
144      *            (Required) SAX Helper class to keep track of namespaces able
145      *            to determine the correct prefix for a given namespace URI.
146      */
147     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler,
148             NamespaceSupport namespaces)throws SAXException
149     {
150         if (this.characters != null)
151         {
152             sendCharacters(contentHandler, this.characters);
153         }
154         else if (this.message != null)
155         {
156             String catalogue = message.getCatalogue();
157             Object[] dictionaryParameters = message.getDictionaryParameters();
158 
159             if (dictionaryParameters == null
160                     || dictionaryParameters.length == 0)
161             {
162                 // No parameters, we can use the simple method
163                 // <i18n:text> Text to be translated </i18n:text>
164 
165                 AttributeMap attributes = new AttributeMap();
166                 attributes.setNamespace(WingConstants.I18N);
167                 attributes.put(A_CATALOGUE, catalogue);
168 
169                 startElement(contentHandler, namespaces, WingConstants.I18N,
170                         E_TEXT, attributes);
171                 sendCharacters(contentHandler, message.getKey());
172                 endElement(contentHandler, namespaces, WingConstants.I18N,
173                         E_TEXT);
174 
175             }
176             else
177             {
178                 // There are parameters, we need to us the complex method.
179                 // <i18n:translate>
180                 // <i18n:text> some {0} was inserted {1}. </i18n:text>
181                 // <i18n:param> text </i18n:param>
182                 // <i18n:param> here </i18n:param>
183                 // </i18n:translate>
184 
185                 startElement(contentHandler, namespaces, WingConstants.I18N,
186                         E_TRANSLATE, null);
187 
188                 AttributeMap attributes = new AttributeMap();
189                 attributes.setNamespace(WingConstants.I18N);
190                 attributes.put(A_CATALOGUE, catalogue);
191 
192                 startElement(contentHandler, namespaces, WingConstants.I18N,
193                         E_TEXT, attributes);
194                 sendCharacters(contentHandler, message.getKey());
195                 endElement(contentHandler, namespaces, WingConstants.I18N,
196                         E_TEXT);
197 
198                 // i18n:param tags
199                 for (Object dictionaryParameter : dictionaryParameters)
200                 {
201                     toSAX(contentHandler, namespaces, dictionaryParameter);
202                 }
203 
204                 endElement(contentHandler, namespaces, WingConstants.I18N,
205                         E_TRANSLATE);
206             }
207         }
208     }

210     /**
211      * Build the the correct param element for the given dictionaryParameter
212      * based upon the class type. This method can deal with Dates, numbers, and
213      * strings.
214      * 
215      * @param dictionaryParameter
216      *            A dictionary parameter.
217      */
218     private void toSAX(ContentHandler contentHandler,
219             NamespaceSupport namespaces, Object dictionaryParameter)
220             throws SAXException
221     {
222         AttributeMap attributes = new AttributeMap();
223         attributes.setNamespace(WingConstants.I18N);
224 
225         boolean unknownType = false;
226 
227         if (dictionaryParameter.getClass().equals(Date.class.getName()))
228         {
229             Date date = (Date) dictionaryParameter;
230             DateFormat dateFormater = DateFormat
231                     .getDateInstance(DateFormat.DEFAULT);
232 
233             attributes.put(A_TYPE, TYPE_DATE);
234             attributes.put(A_VALUE, dateFormater.format(date));
235             // If no pattern is given then the default format is assumed.
236         }
237         else if (dictionaryParameter.getClass().equals(Integer.class.getName()))
238         {
239             Integer value = (Integer) dictionaryParameter;
240             attributes.put(A_TYPE, TYPE_NUMBER);
241             attributes.put(A_VALUE, String.valueOf(value));
242         }
243         else if (dictionaryParameter.getClass().equals(Double.class.getName()))
244         {
245             Double value = (Double) dictionaryParameter;
246             attributes.put(A_TYPE, TYPE_NUMBER);
247             attributes.put(A_VALUE, String.valueOf(value));
248         }
249         else if (dictionaryParameter.getClass().equals(Long.class.getName()))
250         {
251             Long value = (Long) dictionaryParameter;
252             attributes.put(A_TYPE, TYPE_NUMBER);
253             attributes.put(A_VALUE, String.valueOf(value));
254         }
255         else if (dictionaryParameter.getClass().equals(Short.class.getName()))
256         {
257             Short value = (Short) dictionaryParameter;
258             attributes.put(A_TYPE, TYPE_NUMBER);
259             attributes.put(A_VALUE, String.valueOf(value));
260         }
261         else if (dictionaryParameter.getClass().equals(Float.class.getName()))
262         {
263             Float value = (Float) dictionaryParameter;
264             attributes.put(A_TYPE, TYPE_NUMBER);
265             attributes.put(A_VALUE, String.valueOf(value));
266         }
267         else
268         {
269             // Unknown types or String
270             unknownType = true;
271         }
272 
273         startElement(contentHandler, namespaces, WingConstants.I18N, E_PARAM,
274                 attributes);
275 
276         // If the type is unknown then the value is not included as an attribute
277         // and instead is sent as the contents of the elements.
278         if (unknownType)
279             sendCharacters(contentHandler, dictionaryParameter.toString());
280 
281         endElement(contentHandler, namespaces, WingConstants.I18N, E_PARAM);
282     }
