216     /**
217      * Determine if the given SAX event is a PageMeta element.
218      * 
219      * @param namespace
220      *            The element's name space
221      * @param localName
222      *            The local, unqualified, name for this element
223      * @param qName
224      *            The qualified name for this element
225      * @param attributes
226      *            The element's attributes
227      * @return True if this WingElement is equivalent to the given SAX Event.
228      */
229     public boolean mergeEqual(String namespace, String localName, String qName,
230             Attributes attributes) throws SAXException, WingException
231     {
232 
233         if (!WingConstants.DRI.URI.equals(namespace))
234             return false;
235 
236         if (!E_PAGE_META.equals(localName))
237             return false;
238         return true;
239     }

241     /**
242      * Since metadata can not be merged there are no mergeable children. This
243      * just return's null.
244      * 
245      * @param namespace
246      *            The element's name space
247      * @param localName
248      *            The local, unqualified, name for this element *
249      * @param qName
250      *            The qualified name for this element
251      * @param attributes
252      *            The element's attributes
253      * @return The child element
254      */
255     public WingMergeableElement mergeChild(String namespace, String localName,
256             String qName, Attributes attributes) throws SAXException,
257             WingException
258     {
259     	// We don't merge our children but we do have one special optimization, 
260     	// if a metadata or trail is allready in the document then we do not add 
261     	// our own trail or metadata for that particular item.
262     	if (WingConstants.DRI.URI.equals(namespace) && Trail.E_TRAIL.equals(localName))
263     	{
264             for (Trail trail : trails)
265                 trail.dispose();
266     		trails.clear();
267     	}
268     	
269     	if (WingConstants.DRI.URI.equals(namespace) && Metadata.E_METADATA.equals(localName))
270     	{
271     		String element = attributes.getValue(Metadata.A_ELEMENT);
272     		String qualifier = attributes.getValue(Metadata.A_QUALIFIER);
273     		String language = attributes.getValue(Metadata.A_LANGUAGE);
274     		
275     		List<Metadata> remove = new ArrayList<Metadata>();
276     		for (Metadata metadata : metadatum)
277     		{
278     			if (metadata.equals(element,qualifier,language) && !metadata.allowMultiple())
279     			{
280     				remove.add(metadata);
281     			}
282     		}
283     		
284     		// Remove all the metadata elements we found.
285     		for (Metadata metadata : remove)
286     		{
287     			metadata.dispose();
288     			metadatum.remove(metadata);
289     		}
290     	}
291     	
292         return null;
293     }

305     /**
306      * Translate this element into SAX events.
307      * 
308      * @param contentHandler
309      *            (Required) The registered contentHandler where SAX events
310      *            should be routed too.
311      * @param lexicalHandler
312      *            (Required) The registered lexicalHandler where lexical 
313      *            events (such as CDATA, DTD, etc) should be routed too.
314      * @param namespaces
315      *            (Required) SAX Helper class to keep track of namespaces able
316      *            to determine the correct prefix for a given namespace URI.
317      */
318     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler, 
319             NamespaceSupport namespaces) throws SAXException
320     {
321         if (!merged)
322         {
323             startElement(contentHandler, namespaces, E_PAGE_META, null);
324         }
325 
326         for (Metadata metadata : metadatum)
327             metadata.toSAX(contentHandler, lexicalHandler, namespaces);
328         
329         for (Trail trail : trails)
330             trail.toSAX(contentHandler, lexicalHandler, namespaces);
331 
332         if (!merged)
333             endElement(contentHandler, namespaces, E_PAGE_META);
334     }

336     /**
337      * dispose
338      */
339     public void dispose()
340     {
341     	for (Metadata metadata : metadatum)
342     		metadata.dispose();
343     	
344     	for (Trail trail : trails)
345     		trail.dispose();
346     	
347     	trails.clear();
348     	trails = null;
349     	metadatum.clear();
350     	metadatum = null;
351     	
352         super.dispose();
353     }
