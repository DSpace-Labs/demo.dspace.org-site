70  /**
71   * This class represents data that is translated from simple HTML or plain text.
72   * 
73   * This class represents a simple HTML fragment. It allows for user supplied
74   * HTML to be translated on the fly into DRI.
75   * 
76   * At the present time it only supports the following tags: h1, h2, h3, h4, h5,
77   * p, a, b, i, u, ol, li and img. Each are translated into their DRI equivelents, note
78   * the "h" tags are translated into a paragraph of rend=heading.
79   * 
80   * If the linkbreaks flag is set then line breaks are treated as paragraphs. This 
81   * allows plain text files to also be included and they will be mapped into DRI as 
82   * well.
83   * 
84   * @author Scott Phillips
85   * @author Jay Paz
86   */

117 	 * Translate this element into SAX
118 	 * 
119 	 * @param contentHandler
120 	 *            (Required) The registered contentHandler where SAX events
121 	 *            should be routed too.
122 	 * @param lexicalHandler
123 	 *            (Required) The registered lexicalHandler where lexical events
124 	 *            (such as CDATA, DTD, etc) should be routed too.
125 	 * @param namespaces
126 	 *            (Required) SAX Helper class to keep track of namespaces able
127 	 *            to determine the correct prefix for a given namespace URI.
128 	 */
129 	public void toSAX(ContentHandler contentHandler,
130 			LexicalHandler lexicalHandler, NamespaceSupport namespaces)
131 			throws SAXException {
132 		try {
133 			String xml = "<fragment>" + fragment + "</fragment>";
134 
135 			ByteArrayInputStream inputStream = new ByteArrayInputStream(xml
136 					.getBytes());
137 
138 			SAXBuilder builder = new SAXBuilder();
139 			Document document = builder.build(inputStream);
140 
141 			try {
142 				translate(document.getRootElement());
143 			} catch (Throwable t) {
144 				throw new JDOMException(
145 						"Error translating HTML fragment into DRI", t);
146 			}
147 
148 			SAXFilter filter = new SAXFilter(contentHandler, lexicalHandler,
149 					namespaces);
150 			SAXOutputter outputter = new SAXOutputter();
151 			outputter.setContentHandler(filter);
152 			outputter.setLexicalHandler(filter);
153 
154 			Element root = document.getRootElement();
155 
156 			@SuppressWarnings("unchecked")
157 			// This cast is correct
158 			List<Element> children = root.getChildren();
159 			for (Element child : children) {
160 				outputter.output(child);
161 			}
162 
163 		} catch (JDOMException e) {
164 			startElement(contentHandler, namespaces, Para.E_PARA, null);
165 			sendCharacters(contentHandler, "Error unable to parse: "
166 					+ e.getMessage());
167 			sendCharacters(contentHandler, fragment);
168 			endElement(contentHandler, namespaces, Para.E_PARA);
169 		} catch (IOException ioe) {
170 			throw new SAXException(ioe);
171 		}
172 	}

181 	/**
182 	 * Remove the given content from the Element.
183 	 * 
184 	 * If the content is an element then render it as text and include it's
185 	 * children in the parent.
186 	 * 
187 	 * @param content
188 	 *            The DOM Content to be removed.
189 	 */
190 	private void removeContent(Content content) {
191 		if (content instanceof Element) {
192 			// If it's an element replace the content with a text node.
193 			Element element = (Element) content;
194 
195 			if (element.getContent().size() == 0) {
196 				// The element contains nothing, we can use shorthand notation
197 				// for it.
198 				String replacement = "<" + element.getName();
199 
200 				@SuppressWarnings("unchecked")
201 				// This cast is correct
202 				List<Attribute> attributes = element.getAttributes();
203 				for (Attribute attribute : attributes) {
204 					replacement += " " + attribute.getName() + "=\""
205 							+ attribute.getValue() + "\"";
206 				}
207 				replacement += "/>";
208 
209 				Element parent = element.getParentElement();
210 				int index = parent.indexOf(element);
211 				parent.setContent(index, new Text(replacement));
212 			} else {
213 				// The element contains data
214 				String prepend = "<" + element.getName();
215 
216 				@SuppressWarnings("unchecked")
217 				// This cast is correct
218 				List<Attribute> attributes = element.getAttributes();
219 				for (Attribute attribute : attributes) {
220 					prepend += " " + attribute.getName() + "=\""
221 							+ attribute.getValue() + "\"";
222 				}
223 				prepend += ">";
224 
225 				String postpend = "</" + element.getName() + ">";
226 
227 				Element parent = element.getParentElement();
228 				int index = parent.indexOf(element);
229 
230 				parent.addContent(index, new Text(postpend));
231 				parent.addContent(index, element.removeContent());
232 				parent.addContent(index, new Text(prepend));
233 				parent.removeContent(element);
234 			}
235 		} else {
236 			// If it's not an element just remove the content from the document.
237 			Element parent = content.getParentElement();
238 			parent.removeContent(content);
239 		}
240 	}

242 	/**
243 	 * Wrap the given set of contents into a paragraph and place it at the
244 	 * supplied index.
245 	 * 
246 	 * This method will also check for trivial paragraphs, i.e. those that
247 	 * contain nothing but white space. If they are found then they are removed.
248 	 * 
249 	 * @param parent
250 	 *            The parent element to attach the wrapped paragraph too.
251 	 * @param index
252 	 *            The index within the parent for where the content should be
253 	 *            attached.
254 	 * @param contents
255 	 *            The contents that should be wrapped in a paragraph.
256 	 * @return wheather a paragraph was actualy added.
257 	 */
258 	private boolean paragraphWrap(Element parent, int index,
259 			List<Content> contents) {
260 		if (contents == null || contents.size() <= 0)
261 			return false;
262 
263 		boolean empty = true;
264 		for (Content content : contents) {
265 			if (empty == false)
266 				continue;
267 
268 			if (content instanceof Text) {
269 				Text text = (Text) content;
270 				if (!"".equals(text.getTextNormalize()))
271 					empty = false;
272 			} else {
273 				empty = false;
274 			}
275 		}
276 
277 		if (empty == true)
278 			return false;
279 
280 		// May be usefull for debugging:
281 		// contents.add(0, new Text("("+index+") "));
282 
283 		Element para = new Element(Para.E_PARA);
284 		para.addContent(contents);
285 		if (index >= 0)
286 			parent.addContent(index, para);
287 		else
288 			parent.addContent(para);
289 
290 		return true;
291 	}

294 	 * Ensure that the given element only has the supplied attributes. Also
295 	 * remove any possible namespaces on the attributes.
296 	 * 
297 	 * @param element
298 	 *            The element to be checked.
299 	 * @param names
300 	 *            A list of all allowed attribute names, all others will be
301 	 *            removed.
302 	 */
303 	private void limitAttributes(Element element, String... names) {
304 		Map<String, String> attributes = new HashMap<String, String>();
305 		for (String name : names) {
306 			String value = element.getAttributeValue(name);
307 			if (value != null)
308 				attributes.put(name, value);
309 		}
310 
311 		element.setAttributes(new ArrayList<Attributes>());
312 
313 		for (String name : attributes.keySet()) {
314 			String value = attributes.get(name);
315 			element.setAttribute(name, value);
316 		}
317 	}

335 	/**
336 	 * Translate the given HTML fragment into a DRI document.
337 	 * 
338 	 * The translation is broken up into two steps, 1) recurse through all
339 	 * elements and either translate them into their DRI equivelents or remove
340 	 * them from the document.
341 	 * 
342 	 * The second step, 2) is to iterate over all top level elements and ensure
343 	 * that they only consist of paragraphs. Also at this stage if linkBreaks is
344 	 * true then \n are treated as paragraph breaks.
345 	 * 
346 	 * @param parent
347 	 *            The Element to translate into DRI.
348 	 */
349 	private void translate(Element parent) {
350 		// Step 1:
351 		// Recurse through all elements and either
352 		// translate them or remove them.
353 		for (int i = 0; i < parent.getContentSize(); i++) {
354 			Content decedent = parent.getContent(i);
355 
356 			if (decedent instanceof org.jdom.Text) {
357 
358 			} else if (decedent instanceof Element) {
359 				Element element = (Element) decedent;
360 				String name = element.getName();
361 
362 				// First all the DRI elements, allow them to pass.
363 				if ("p".equals(name)) {
364 					// Paragraphs are tricky, it may be either an HTML
365 					// or DRI <p> element. However, while HTML will allow
366 					// <p> to nest DRI does not, thus first we need to
367 					// check if this is at the block level, if it is then
368 					// we need remove it.
369 
370 					if (parent.isRootElement()) {
371 						// The paragraph is not nested, so translate it to
372 						// a DRI <p>
373 						moveAttribute(element, "class", "rend");
374 						limitAttributes(element, "id", "n", "rend");
375 
376 						translate(element);
377 					} else {
378 						// The paragraph is nested which is not allowed in
379 						// DRI, so remove it.
380 						removeContent(element);
381 					}
382 				} else if ("h1".equals(name) || "h2".equals(name)
383 						|| "h3".equals(name) || "h4".equals(name)
384 						|| "h5".equals(name)) {
385 					// The HTML <H1> tag is translated into the DRI
386 					// <p rend="heading"> tag.
387 					if (parent.isRootElement()) {
388 						limitAttributes(element);
389 						element.setName("p");
390 						element.setAttribute("rend", "heading");
391 
392 						translate(element);
393 					} else {
394 						// DRI paragraphs can not be nested.
395 						removeContent(element);
396 					}
397 				} else if ("a".equals(name)) {
398 					// The HTML <a> tag is translated into the DRI
399 					// <xref> tag.
400 					moveAttribute(element, "href", "target");
401 					limitAttributes(element, "target");
402 					element.setName("xref");
403 
404 					translate(element);
405 				} else if ("ol".equals(name)) {
406 					// the HTML tag <ol> its translated into the DRI
407 					// <list> tag
408 					// <list type="ordered" n="list_part_one"
409 					// id="css.submit.LicenseAgreement.list.list_part_one">
410 					moveAttribute(element, "class", "rend");
411 					limitAttributes(element, "id", "n", "rend");
412 					element.setName("list");
413 					element.setAttribute("type", "ordered");
414 					translate(element);
415 				} else if ("li".equals(name)) {
416 					// the HTML tag <li> its translated into the DRI
417 					// <item> tag
418 					moveAttribute(element, "class", "rend");
419 					limitAttributes(element, "id", "n", "rend");
420 					element.setName("item");
421 					translate(element);
422 				} else if ("b".equals(name)) {
423 					// The HTML <b> tag is translated to a highlight
424 					// element with a rend of bold.
425 					limitAttributes(element);
426 					element.setName("hi");
427 					element.setAttribute("rend", "bold");
428 
429 					translate(element);
430 				} else if ("i".equals(name)) {
431 					// The HTML <i> tag is translated to a highlight
432 					// element with a rend of italic.
433 					limitAttributes(element);
434 					element.setName("hi");
435 					element.setAttribute("rend", "italic");
436 
437 					translate(element);
438 				} else if ("u".equals(name)) {
439 					// The HTML <u> tag is translated to a highlight
440 					// element with a rend of underline.
441 					limitAttributes(element);
442 					element.setName("hi");
443 					element.setAttribute("rend", "underline");
444 
445 					translate(element);
446 				} else if ("img".equals(name)) {
447 					// The HTML <img> element is translated into a DRI figure
448 					moveAttribute(element, "src", "source");
449 					limitAttributes(element, "source");
450 					element.setName("figure");
451 
452 					translate(element);
453 				}
454 				// Next all the DRI elements that we allow to pass through.
455 				else if ("hi".equals(name)) {
456 					limitAttributes(element, "rend");
457 
458 					translate(element);
459 				} else if ("xref".equals(name)) {
460 					limitAttributes(element, "target");
461 
462 					translate(element);
463 				} else if ("figure".equals(name)) {
464 					limitAttributes(element, "rend", "source", "target");
465 
466 					translate(element);
467 				} else {
468 					removeContent(decedent);
469 				}
470 			} else {
471 				removeContent(decedent);
472 			}
473 		}
474 
475 		// Step 2:
476 		// Ensure that all top level elements are encapusalted inside
477 		// a block level element (i.e. a paragraph)
478 		if (parent.isRootElement()) {
479 			List<Content> removed = new ArrayList<Content>();
480 			for (int i = 0; i < parent.getContentSize(); i++) {
481 				Content current = parent.getContent(i);
482 				
483 				if ((current instanceof Element)
484 						&& ("p".equals(((Element) current).getName()))) {
485 					// A paragraph is being open, combine anything up to this
486 					// point into a paragraph.
487 					if (paragraphWrap(parent, i, removed)) {
488 						removed.clear();
489 						i++; // account for the field added
490 					}
491 				} else if ((current instanceof Element)
492 						&& ("list".equals(((Element) current).getName()))) {
493 					if (paragraphWrap(parent, i, removed)) {
494 						removed.clear();
495 						i++; // account for the field added
496 					}
497 				} else {
498 					// If we break paragraphs based upon blank lines then we
499 					// need to check if
500 					// there are any in this text element.
501 					if (this.blankLines && current instanceof Text) {
502 						String rawText = ((Text) current).getText();
503 						parent.removeContent(current);
504 						i--;// account text field removed.
505 
506 						// Regular expressiot to split based upon blank lines.
507 						// FIXME: This may not work for windows people who
508 						// insist on using \r\n for line breaks.
509 						@SuppressWarnings("unchecked")
510 						// This cast is correct
511 						List<String> parts = new ArrayList(Arrays
512 								.asList(rawText.split("\n\\s*\n")));
513 
514 						if (parts.size() > 0) {
515 							String lastPart = parts.remove(parts.size()-1);
516 
517 							for (String part : parts) {
518 								removed.add(new Text(part));
519 
520 								if (paragraphWrap(parent, i+1, removed)) {
521 									removed.clear();
522 									i++;// account for the field added
523 								}
524 							}
525 
526 							removed.add(new Text(lastPart));
527 						}
528 					} else {
529 						removed.add(current);
530 						parent.removeContent(current);
531 						i--; // move back to account for the removed content.
532 					}
533 				}
534 			}
535 
536 			// if anything is left, wrap it up in a para also.
537 			if (removed.size() > 0) {
538 				paragraphWrap(parent, -1, removed);
539 				removed.clear();
540 			}
541 		}
542 	}

546 	 * This class is needed for two reasons, 1) namespaces need to be corrected
