41  package org.dspace.app.xmlui.wing.element;
42  
43  /**
44   * This class represents a generic Wing Container. The Container class adds a
45   * simlpe contents list which may be modified by the extending concret classes.
46   * When it comes time to process the element a toSAX method is provided that
47   * will iterate over the contents. 
48   * 
49   * @author Scott Phillips
50   */
51  
52  import java.util.ArrayList;
53  import java.util.List;
54  
55  import org.dspace.app.xmlui.wing.WingContext;
56  import org.dspace.app.xmlui.wing.WingException;
57  import org.xml.sax.ContentHandler;
58  import org.xml.sax.SAXException;
59  import org.xml.sax.ext.LexicalHandler;
60  import org.xml.sax.helpers.NamespaceSupport;
61  
62  public abstract class Container extends AbstractWingElement
63  {
64  
65      /** The internal contents of this container */
66      protected List<AbstractWingElement> contents = new ArrayList<AbstractWingElement>();
67  
68      /**
69       * @param context
70       *            (Required) The context this element is contained in.
71       */
72      protected Container(WingContext context) throws WingException
73      {
74          super(context);
75      }
76  
77      /**
78       * Translate this element and all contained elements into SAX events. The
79       * events should be routed to the contentHandler found in the WingContext.
80       * 
81       * This method does not create an inclosing block, the implementors of
82       * container class need to implement a method of toSAX() that provides the
83       * surrounding element block for that specific application.
84       * 
85       * @param contentHandler
86       *            (Required) The registered contentHandler where SAX events
87       *            should be routed too.
88       * @param lexicalHandler
89       *            (Required) The registered lexicalHandler where lexical events
90       *            (such as CDATA, DTD, etc) should be routed too.
91       * @param namespaces
92       *            (Required) SAX Helper class to keep track of namespaces able
93       *            to determine the correct prefix for a given namespace URI.
94       */
95      public void toSAX(ContentHandler contentHandler,
96              LexicalHandler lexicalHandler, NamespaceSupport namespaces)
97              throws SAXException
98      {
99          for (AbstractWingElement content : contents)
100         {
101             content.toSAX(contentHandler, lexicalHandler, namespaces);
102         }
103     }
104 
105     /**
106      * Dispose
107      */
108     public void dispose()
109     {
110 
111         if (this.contents != null)
112         {
113             for (AbstractWingElement element : contents)
114                 element.dispose();
115             this.contents.clear();
116         }
117         this.contents = null;
118         super.dispose();
119     }
