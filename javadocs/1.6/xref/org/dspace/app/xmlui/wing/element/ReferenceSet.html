54  /**
55   * Class representing a set of referenced metadata.
56   * 
57   * @author Scott Phillips

59  
60  public class ReferenceSet extends AbstractWingElement implements
61          StructuralElement
62  {
63      /** The name of the referenceSet element */
64      public static final String E_REFERENCE_SET = "referenceSet";
65  
66      /** The name of the orderBy attribute */
67      public static final String A_ORDER_BY = "orderBy";
68  
69      /** The name of the type attribute */
70      public static final String A_TYPE = "type";
71  
72      /** The possible interactive division methods: get,post, or multipart. */
73      public static final String TYPE_SUMMARY_LIST = "summaryList";
74  
75      public static final String TYPE_SUMMARY_VIEW = "summaryView";
76  
77      public static final String TYPE_DETAIL_LIST = "detailList";
78  
79      public static final String TYPE_DETAIL_VIEW = "detailView";
80  
81      /** The possible interactive division methods names collected into one array */
82      public static final String[] TYPES = { TYPE_SUMMARY_LIST, TYPE_SUMMARY_VIEW, TYPE_DETAIL_LIST, TYPE_DETAIL_VIEW };
83  
84      /** The name assigned to this metadata set */
85      private String name;
86  
87      /** The ordering mechanism to use. */
88      private String orderBy;
89  
90      /** The reference type, see TYPES defined above */
91      private String type;
92  
93      /** Special rendering instructions */
94      private String rend;
95  
96      /** The head label for this referenceset */
97      private Head head;
98  
99      /** All content of this container, items & lists */
100     private java.util.List<AbstractWingElement> contents = new ArrayList<AbstractWingElement>();
101 
102     /**
103      * Construct a new referenceSet
104      * 
105      * @param context
106      *            (Required) The context this element is contained in, such as
107      *            where to route SAX events and what i18n catalogue to use.
108      * @param name
109      *            (May be null) a local identifier used to differentiate the
110      *            element from its siblings.
111      * @param informationModel
112      *            (May be null) The information model the enclosed objects
113      *            follow. If no model is given then the default information
114      *            model is used. (INFORMATION_MODEL_DEFAULT)
115      * @param type
116      *            (Required) The type of reference set which determines the level
117      *            of detail for the metadata rendered. See TYPES for a list of
118      *            available types.
119      * @param orderBy
120      *            (May be null) Determines the ordering of referenced metadata.
121      * @param rend
122      *            (May be null) a rendering hint used to override the default
123      *            display of the element.
124      */
125     protected ReferenceSet(WingContext context, boolean childreference, String name, String type, String orderBy, String rend)
126             throws WingException
127     {
128         super(context);
129         // Names are only required for parent reference sets.
130         if (!childreference)
131             require(name, "The 'name' parameter is required for reference sets.");
132         restrict(
133                 type,
134                 TYPES,
135                 "The 'method' parameter must be one of these values: 'summaryList', 'summaryView', 'detailList', or 'detailView'.");
136 
137         this.name = name;
138         this.type = type;
139         this.orderBy = orderBy;
140         this.rend = rend;
141     }

196     /**
197      * Translate this metadata inclusion set to SAX
198      * 
199      * @param contentHandler
200      *            (Required) The registered contentHandler where SAX events
201      *            should be routed too.
202      * @param lexicalHandler
203      *            (Required) The registered lexicalHandler where lexical 
204      *            events (such as CDATA, DTD, etc) should be routed too.
205      * @param namespaces
206      *            (Required) SAX Helper class to keep track of namespaces able
207      *            to determine the correct prefix for a given namespace URI.
208      */
209     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler, 
210             NamespaceSupport namespaces) throws SAXException
211     {
212         AttributeMap attributes = new AttributeMap();
213         if (name != null)
214             attributes.put(A_NAME, name);
215         if (name != null)
216             attributes.put(A_ID, context.generateID(E_REFERENCE_SET, name));
217 
218         attributes.put(A_TYPE, type);
219         if (orderBy != null)
220             attributes.put(A_ORDER_BY, orderBy);
221         if (rend != null)
222             attributes.put(A_RENDER, rend);
223 
224         startElement(contentHandler, namespaces, E_REFERENCE_SET, attributes);
225 
226         if (head != null)
227             head.toSAX(contentHandler, lexicalHandler, namespaces);
228 
229         for (AbstractWingElement content : contents)
230         {
231             content.toSAX(contentHandler, lexicalHandler, namespaces);
232         }
233 
234         endElement(contentHandler, namespaces, E_REFERENCE_SET);
235     }

237     /**
238      * dispose
239      */
240     public void dispose()
241     {
242         for (AbstractWingElement content : contents)
243             content.dispose();
244         if (contents != null)
245             contents.clear();
246         contents = null;
247         super.dispose();
248     }
