116     /**
117      * Generate a unique id based upon the locally unique name and the
118      * application.
119      * 
120      * The format of the unique id typically is:
121      * 
122      * <componentName> dot <application> dot <unique name>
123      * 
124      * typically the componentName is the Java class path of the Wing component.
125      * 
126      * @param application
127      *            The application of this element, typically this is the element
128      *            type that is being identified. Such as p, div, table, field,
129      *            etc...
130      * @param name
131      *            A locally unique name that distinguished this element from
132      *            among its siblings.
133      * @return A globally unique identifier.
134      */
135     public String generateID(String application, String name)
136     {
137         return combine(componentName, application, name);
138     }
139 
140     /**
141      * Generate a unique id with a sub name. Like the other form of generateID
142      * this will append another sub-name. This is typically used for when
143      * duplicate names are used but further identification is needed for
144      * globally unique names.
145      * 
146      * @param application
147      *            The application of this element, typically this is the element
148      *            type that is being identified. Such as p, div, table, field,
149      *            etc...
150      * @param name
151      *            A locally unique name that distinguished this element from
152      *            among its siblings.
153      * @param subName
154      *            An additional name to the original name to further identify it
155      *            in cases when just the name alone does not accomplish this.
156      * @return
157      */
158     public String generateID(String application, String name, String subName)
159     {
160         return combine(componentName, application, name, subName);
161     }
162 
163     /**
164      * Generate a specialized name based on the callers name. This is typically
165      * used in situations where two namespaces are being merged together so that
166      * each namespace is prepended with the current scope's name.
167      * 
168      * @param name
169      *            A locally unique name that distinguished this element from

172     public String generateName(String name)
173     {
174         return combine(componentName, name);
175     }

177     /**
178      * Simple utility function to join the parts into one string separated by a
179      * DOT.
180      * 
181      * @param parts
182      *            The input parts.
183      * @return The combined string.
184      */
185     private String combine(String... parts)
186     {
187 
188         String combine = "";
189         boolean skipDivider = true;
190         for (String part : parts)
191         {
192             if (part == null)
193             {
194                 skipDivider = true;
195                 continue;
196             }
197 
198             if (skipDivider)
199                 skipDivider = false;
200             else
201                 combine += DIVIDER;
202 
203             combine += part;
204 
205         }
206 
207         return combine;
208     }
