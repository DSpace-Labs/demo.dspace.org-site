55  /**
56   * A class representing the page division.
57   * 
58   * The body contains any number of divisions (div elements) which group content
59   * into interactive and non interactive display blocks.
60   * 
61   * @author Scott Phillips
62   */
63  public class Body extends AbstractWingElement implements WingMergeableElement
64  {
65      /** The name of the body element */
66      public static final String E_BODY = "body";
67  
68      /** Has this element been merged */
69      private boolean merged = false;
70  
71      /** The divisions contained within this body */
72      private List<Division> divisions = new ArrayList<Division>();
73  
74      /**
75       * Generate a new Body framework element. This method will NOT open or close
76       * a body element instead it expects that those events are being handled by
77       * the caller. It is important to note that other divisions (div elements)
78       * may precede or follow the divisions created through this object.
79       * 
80       * @param context
81       *            (Required) The context this element is contained in.
82       */
83      protected Body(WingContext context) throws WingException
84      {
85          super(context);
86      }
87  
88      /**
89       * Append a new division (div element) to the document's body. The division
90       * created is not interactive meaning that it may not contain any form
91       * elements, to create an interactive division use addInteractiveDivision().
92       * 
93       * @param name
94       *            (Required) a local identifier used to differentiate the
95       *            element from its siblings.
96       * @param rend
97       *            (May be null) a rendering hint used to override the default
98       *            display of the element.
99       * @return a new division.
100      */
101     public Division addDivision(String name, String rend) throws WingException
102     {
103         Division div = new Division(context, name, rend);
104         divisions.add(div);
105         return div;
106     }
107 
108     /**
109      * Append a new division (div element) to the document's body. This is a
110      * short cut method for divisions with out special rendering instructions.
111      * The division created is not interactive meaning that it may not contain
112      * any form elements, to create an interactive division use
113      * addInteractiveDivision().
114      * 
115      * @param name
116      *            a local identifier used to differentiate the element from its
117      *            siblings
118      * @return A new division.
119      */
120     public Division addDivision(String name) throws WingException
121     {
122         return this.addDivision(name, null);
123     }
124 
125     /**
126      * Append a new interactive division (div element) to the document's body.
127      * An interactive division is able to contain form elements as. The extra
128      * parameters required such as action and method dictate where and how the
129      * form data should be processed.
130      * 
131      * @param name
132      *            (Required) a local identifier used to differentiate the
133      *            element from its siblings.
134      * @param action
135      *            (Required) determines where the form information should be
136      *            sent for processing.
137      * @param method
138      *            (Required) determines the method used to pass gathered field
139      *            values to the handler specified by the action attribute. The
140      *            multipart method should be used if there are any file fields
141      *            used within the division.
142      * @param rend
143      *            (May be null) a rendering hint used to override the default
144      *            display of the element.
145      * @return A new division.
146      */
147     public Division addInteractiveDivision(String name, String action,
148             String method, String rend) throws WingException
149     {
150         Division div = new Division(context, name, action, method, rend);
151         divisions.add(div);
152         return div;
153 
154     }

156     /**
157      * Is this SAX event equivalent to this body?
158      * 
159      * @param namespace
160      *            The element's name space
161      * @param localName
162      *            The local, unqualified, name for this element
163      * @param qName
164      *            The qualified name for this element
165      * @param attributes
166      *            The element's attributes
167      * @return True if it is equivalent.
168      */
169     public boolean mergeEqual(String namespace, String localName, String qName,
170             Attributes attributes) throws SAXException, WingException
171     {
172         if (!WingConstants.DRI.URI.equals(namespace))
173             return false;
174         if (!E_BODY.equals(localName))
175             return false;
176         return true;
177     }
178 
179     /**
180      * Merge this SAX event into the body.
181      * 
182      * @param namespace
183      *            The element's name space
184      * @param localName
185      *            The local, unqualified, name for this element
186      * @param qName
187      *            The qualified name for this element
188      * @param attributes
189      *            The element's attributes
190      * @return
191      */
192     public WingMergeableElement mergeChild(String namespace, String localName,
193             String qName, Attributes attributes) throws SAXException,
194             WingException
195     {
196         Division found = null;
197         for (Division candidate : divisions)
198         {
199             if (candidate.mergeEqual(namespace, localName, qName, attributes))
200                 found = candidate;
201 
202         }
203         divisions.remove(found);
204         return found;
205     }
206 
207     /**
208      * Inform this element that it is being merged with an existing body.
209      * 
210      * @param attributes
211      *            The to-be-merged attributes
212      */
213     public Attributes merge(Attributes attributes) throws SAXException,
214             WingException
215     {
216         this.merged = true;
217         return attributes;
218     }
219 
220     /**
221      * Translate into SAX
222      * 
223      * @param contentHandler
224      *            (Required) The registered contentHandler where SAX events
225      *            should be routed too.
226      * @param lexicalHandler
227      *            (Required) The registered lexicalHandler where lexical events
228      *            (such as CDATA, DTD, etc) should be routed too.
229      * @param namespaces
230      *            (Required) SAX Helper class to keep track of namespaces able
231      *            to determine the correct prefix for a given namespace URI.
232      */
233     public void toSAX(ContentHandler contentHandler,
234             LexicalHandler lexicalHandler, NamespaceSupport namespaces)
235             throws SAXException
236     {
237         if (!merged)
238             startElement(contentHandler, namespaces, E_BODY, null);
239 
240         for (Division division : divisions)
241             division.toSAX(contentHandler, lexicalHandler, namespaces);
242 
243         if (!merged)
244             endElement(contentHandler, namespaces, E_BODY);
245     }
246 
247     /**
248      * dispose
249      */
250     public void dispose()
251     {
252         for (Division division : divisions)
253             division.dispose();
254         super.dispose();
255     }
