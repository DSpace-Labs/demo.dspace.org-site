43  /**
44   * A class representing a table.
45   * 
46   * The table element is a container for information presented in tabular format.
47   * It consists of a set of row elements and an optional header.

49   * @author Scott Phillips
50   */

52  import java.util.ArrayList;
53  import java.util.List;

93      /**
94       * Construct a new row.
95       * 
96       * @param context
97       *            (Required) The context this element is contained in, such as
98       *            where to route SAX events and what i18n catalogue to use.
99       * 
100      * @param name
101      *            (Required) a local identifier used to differentiate the
102      *            element from its siblings.
103      * 
104      * @param rows
105      *            (Required) the number of rows in the table.
106      * @param cols
107      *            (Required) the number of columns in the table.
108      * @param rend
109      *            (May be null) a rendering hint used to override the default
110      *            display of the element.
111      */
112     protected Table(WingContext context, String name, int rows, int cols,
113             String rend) throws WingException
114     {
115         super(context);
116         require(name, "The 'name' parameter is required for all tables.");
117         greater(rows, 0, "The 'rows' parameter must be grater than zero.");
118         greater(cols, 0, "The 'cols' parameter must be grater than zero.");
119 
120         this.name = name;
121         this.rows = rows;
122         this.cols = cols;
123         this.rend = rend;
124     }

218     /**
219      * Translate this element and all contained elements into SAX events. The
220      * events should be routed to the contentHandler found in the WingContext.
221      * 
222      * @param contentHandler
223      *            (Required) The registered contentHandler where SAX events
224      *            should be routed too.
225      * @param lexicalHandler
226      *            (Required) The registered lexicalHandler where lexical 
227      *            events (such as CDATA, DTD, etc) should be routed too.
228      * @param namespaces
229      *            (Required) SAX Helper class to keep track of namespaces able
230      *            to determine the correct prefix for a given namespace URI.
231      */
232 
233     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler, 
234             NamespaceSupport namespaces) throws SAXException
235     {
236         AttributeMap attributes = new AttributeMap();
237         attributes.put(A_NAME, name);
238         attributes.put(A_ID, context.generateID(E_TABLE, name));
239         attributes.put(A_ROWS, rows);
240         attributes.put(A_COLS, cols);
241         if (rend != null)
242             attributes.put(A_RENDER, rend);
243 
244         startElement(contentHandler, namespaces, E_TABLE, attributes);
245         if (head != null)
246             head.toSAX(contentHandler, lexicalHandler, namespaces);
247         for (AbstractWingElement content : contents)
248             content.toSAX(contentHandler, lexicalHandler, namespaces);
249         endElement(contentHandler, namespaces, E_TABLE);
250     }

252     /**
253      * dispose
254      */
255     public void dispose()
256     {
257         if (head != null)
258             head.dispose();
259         for (AbstractWingElement content : contents)
260             content.dispose();
261 
262         head = null;
263         contents.clear();
264         contents = null;
265         super.dispose();
266     }
