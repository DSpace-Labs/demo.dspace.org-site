43  /**
44   * A class that represents a List element.
45   * 
46   * The list element is used to display sets of sequential data. It contains an
47   * optional head element, as well as any number of item elements. Items contain
48   * textual information or other list elements. An item can also be associated
49   * with a label element that annotates an item with a number, a textual
50   * description of some sort, or a simple bullet. The list type (ordered,
51   * bulletted, gloss, etc.) is then determined either by the content of labels on
52   * items or by an explicit value of the "type" attribute. Note that if labels
53   * are used in conjunction with any items in a list, all of the items in that
54   * list must have a label. It is also recommended to avoid mixing label styles
55   * unless an explicit type is specified.
56   * 
57   * typically rendering types are not predefined, but for lists there is a set of
58   * standard rendering options available for the rend attribute to be set too.
59   * This is not an exhaustive list.
60   * 
61   * horizontal: The list should be rendered horizontally.
62   * 
63   * vertical: The list should be rendered vertically.
64   * 
65   * columns: The list should be rendered in equal length columns as determined by
66   * the theme.
67   * 
68   * columns2: The list should be rendered in two equal columns.
69   * 
70   * columns3: The list should be rendered in three equal columns.
71   * 
72   * alphabet: The list should be rendered as an alphabetical index.
73   * 
74   * numeric: The list should be rendered as a numeric index.
75   * 
76   * @author Scott Phillips
77   */

439     /**
440      * Determine if the given SAX startElement event is equivalent to this list.
441      * 
442      * @param namespace
443      *            The element's name space
444      * @param localName
445      *            The local, unqualified, name for this element
446      * @param qName
447      *            The qualified name for this element
448      * @param attributes
449      *            The element's attributes
450      * @return True if this list is equivalent to the given SAX Event.
451      */
452     public boolean mergeEqual(String namespace, String localName, String qName,
453             Attributes attributes)
454     {
455         // Check if it's in our name space and an options element.
456         if (!WingConstants.DRI.URI.equals(namespace))
457             return false;
458         if (!E_LIST.equals(localName))
459             return false;
460         String name = attributes.getValue(A_NAME);
461         if (name == null)
462             return false;
463         if (!name.equals(this.name))
464             return false;
465         return true;
466     }

468     /**
469      * Merge the given SAX startElement event into this list's child. If this
470      * SAX event matches a child element of this list then it should be removed
471      * from the internal book keep of this element and returned. Typically this
472      * is accomplished by looping through all children elements and returned the
473      * first one that returns true for the mergeEqual method.
474      * 
475      * @param namespace
476      *            The element's name space
477      * @param localName
478      *            The local, unqualified, name for this element *
479      * @param qName
480      *            The qualified name for this element
481      * @param attributes
482      *            The element's attributes
483      * @return The child element
484      */
485     public WingMergeableElement mergeChild(String namespace, String localName,
486             String qName, Attributes attributes) throws SAXException,
487             WingException
488     {
489     	this.childMerged = true;
490     	
491         WingMergeableElement found = null;
492         for (AbstractWingElement content : contents)
493         {
494             if (content instanceof WingMergeableElement)
495             {
496                 WingMergeableElement candidate = (WingMergeableElement) content;
497                 if (candidate.mergeEqual(namespace, localName, qName,
498                         attributes))
499                     found = candidate;
500             }
501         }
502         contents.remove(found);
503         return found;
504     }

526     /**
527      * Translate this element and all contained elements into SAX events. The
528      * events should be routed to the contentHandler found in the WingContext.
529      * 
530      * @param contentHandler
531      *            (Required) The registered contentHandler where SAX events
532      *            should be routed too.
533      * @param lexicalHandler
534      *            (Required) The registered lexicalHandler where lexical 
535      *            events (such as CDATA, DTD, etc) should be routed too.
536      * @param namespaces
537      *            (Required) SAX Helper class to keep track of namespaces able
538      *            to determine the correct prefix for a given namespace URI.
539      */
540     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler, 
541             NamespaceSupport namespaces) throws SAXException
542     {
543 
544         if (this.merged == false)
545         {
546             AttributeMap attributes = new AttributeMap();
547             attributes.put(A_NAME, this.name);
548             attributes.put(A_ID, this.context.generateID(E_LIST, this.name));
549             if (this.type != null)
550                 attributes.put(A_TYPE, this.type);
551             if (this.rend != null)
552                 attributes.put(A_RENDER, this.rend);
553 
554             startElement(contentHandler, namespaces, E_LIST, attributes);
555             
556         }
557             
558         if (childMerged == false && head != null)
559         	head.toSAX(contentHandler, lexicalHandler, namespaces);
560 
561         for (AbstractWingElement content : contents)
562             content.toSAX(contentHandler, lexicalHandler,  namespaces);
563 
564         if (this.merged == false)
565             endElement(contentHandler, namespaces, E_LIST);
566     }
