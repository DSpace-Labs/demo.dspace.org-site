43  /**
44   * Class representing a Division, or the div element, in the XML UI schema.
45   * 
46   * The div element represents a major section of content and can contain a wide
47   * variety of other elements to present that content to the user. It can contain
48   * TEI style paragraphs, tables, and lists, as well as references to artifact
49   * information stored in artifactMeta. The div element is also recursive,
50   * allowing it to be further divided into other divs.
51   * 
52   * @author Scott Phillips
53   */

237     /**
238      * Construct an interactive division. Interactive divisions must be
239      * accompanied by both an action and method to determine how to process form
240      * data.
241      * 
242      * The valid values for method may be found in the static variable METHODS.
243      * 
244      * @param context
245      *            (Required) The context this element is contained in, such as
246      *            where to route SAX events and what i18n catalogue to use.
247      * @param name
248      *            (Required) a local identifier used to differentiate the
249      *            element from its siblings.
250      * @param action
251      *            (Required) The form action attribute determines where the form
252      *            information should be sent for processing.
253      * @param method
254      *            (Required) Accepted values are "get", "post", and "multipart".
255      *            Determines the method used to pass gathered field values to
256      *            the handler specified by the action attribute. The multipart
257      *            method should be used if there are any file fields used within
258      *            the division.
259      * @param rend
260      *            (May be null) a rendering hint used to override the default
261      *            display of the element.
262      */
263     protected Division(WingContext context, String name, String action,
264             String method, String rend) throws WingException
265     {
266         super(context);
267         require(name, "The 'name' parameter is required for divisions.");
268         // Blank actions are okay:
269         // require(action,
270         //         "The 'action' parameter is required for interactive divisions.");
271         require(method, "The 'method' parameter is required for divisions.");
272         restrict(
273                 method,
274                 METHODS,
275                 "The 'method' parameter must be one of these values: 'get', 'post', or 'multipart'.");
276 
277         this.name = name;
278         this.interactive = true;
279         this.action = action;
280         this.method = method;
281         this.rend = rend;
282     }

295      * updated each time a request partial page is submitted.
296      * 
297      * @param fieldName
298      *            (Required) The name of a single field (with no spaces).
299      */

301     {
302         require(fieldName, "The fieldName parameter is required for the behaviorSensitiveFields attribute.");
303         if (this.behaviorSensitiveFields == null)

309             this.behaviorSensitiveFields += " " + fieldName;

313     /**
314      * Make this div paginated ( a div that spans multiple pages ) using the
315      * simple page paradigm.
316      * 
317      * @param itemsTotal
318      *            (Required) How many items exist accross all paginated divs.
319      * @param firstItemIndex
320      *            (Required) The index of the first item included in this div.
321      * @param lastItemIndex
322      *            (Required) The index of the last item included in this div.
323      * @param previousPage
324      *            (May be null) The URL of the previous page of the div, if it
325      *            exists.
326      * @param nextPage
327      *            (May be null) The URL of the previous page of the div, if it
328      *            exists.
329      */
330     public void setSimplePagination(int itemsTotal, int firstItemIndex,
331             int lastItemIndex, String previousPage, String nextPage)
332     {
333         this.paginationType = PAGINATION_SIMPLE;
334         this.previousPage = previousPage;
335         this.nextPage = nextPage;
336         this.itemsTotal = itemsTotal;
337         this.firstItemIndex = firstItemIndex;
338         this.lastItemIndex = lastItemIndex;
339     }

343      * masked page paradigm.
344      * 
345      * @param itemsTotal

347      * @param firstItemIndex
348      *            (Required) The index of the first item included in this div.
349      * @param lastItemIndex
350      *            (Required) The index of the last item included in this div.
351      * @param currentPage
352      *            (Required) The index of the page currently displayed for this
353      *            div.
354      * @param pagesTotal
355      *            (Required) How many pages the paginated div spans.
356      * @param pageURLMask
357      *            (Required) The mask of a url to a particular within the
358      *            paginated set. The destination page's number should replace
359      *            the {pageNum} string in the URL mask to generate a full URL to

366         this.paginationType = PAGINATION_MASKED;
367         this.itemsTotal = itemsTotal;
368         this.firstItemIndex = firstItemIndex;
369         this.lastItemIndex = lastItemIndex;
370         this.currentPage = currentPage;
371         this.pagesTotal = pagesTotal;
372         this.pageURLMask = pageURLMask;
373     }

730      * Add a section of translated HTML to the DRI document. This will only handle 
731      * simple transformations such as <p>,<b>,<i>,and <a> tags.
732      * 
733      * Depending on the given HTML this may result in multiple paragraphs being 
734      * opened and several bold tags being included.

738      * @param HTML 
739      * 				(Required) The HTML content

746     /**
747      * Determine if the given SAX event is the same division element. This method 
748      * will compare interactiveness, and rendering.
749      * 
750      * @param namespace
751      *            The element's name space
752      * @param localName
753      *            The local, unqualified, name for this element
754      * @param qName
755      *            The qualified name for this element
756      * @param attributes
757      *            The element's attributes
758      * @return True if this WingElement is equivalent to the given SAX Event.
759      */
760     public boolean mergeEqual(String namespace, String localName, String qName,
761             Attributes attributes) throws SAXException, WingException
762     {
763 
764         if (!WingConstants.DRI.URI.equals(namespace))
765             return false;
766 
767         if (!E_DIVISION.equals(localName))
768             return false;
769 
770         context.getLogger().debug("Merding a division");
771         
772         String name = attributes.getValue(A_NAME);
773         String interactive = attributes.getValue(A_INTERACTIVE);
774         String action = attributes.getValue(A_ACTION);
775         String method = attributes.getValue(A_METHOD);
776         String render = attributes.getValue(A_RENDER);
777         String pagination = attributes.getValue(A_PAGINATION);
778         String behavior = attributes.getValue(A_BEHAVIOR);
779 
780         context.getLogger().debug("Merging got the parameters name="+name+", interactive="+interactive+", action="+action+", method="+method+", render="+render+", pagination="+pagination);
781         
782         // The name must be identical (but id's can differ)
783         if (!this.name.equals(name))
784             return false;
785 
786         // Insure the render attributes are identical.
787         if (this.rend == null)
788         {
789             if (render != null)
790                 return false;
791         }
792         else if (!this.rend.equals(render))
793         {
794             return false;
795         }
796 
797         if (this.interactive)
798         {
799             // Insure all the interactive fields are identical.
800             if (!"yes".equals(interactive))
801                 return false;
802             if (!this.action.equals(action))
803                 return false;
804             if (!this.method.equals(method))
805                 return false;
806             
807             // For now lets just not merge divs that have behavior.
808             if (!(behavior == null || behavior.equals("")))
809                 return false;
810             
811         } else {
812             // Else, insure that it is also not interactive.
813             if (!(interactive == null || "no".equals(interactive)))
814                 return false;
815         }
816 
817         return true;
818     }
