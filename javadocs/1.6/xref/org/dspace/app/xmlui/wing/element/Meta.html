52  /**
53   * A class representing the page division.
54   * 
55   * The body contains any number of divisions (div elements) which group content
56   * into interactive and non interactive display blocks.
57   * 
58   * @author Scott Phillips
59   */
60  public class Meta extends AbstractWingElement implements WingMergeableElement
61  {
62      /** The name of the meta element */
63      public static final String E_META = "meta";
64  
65      /** The divisions contained within this body */
66      private boolean merged = false;
67  
68      /** User oriented metadata associated with this document */
69      private UserMeta userMeta;
70  
71      /** Page oriented metadata associated with this document */
72      private PageMeta pageMeta;
73      
74      /** Repository oriented metadata associated with this document */
75      private RepositoryMeta repositoryMeta;
76      
77      protected Meta(WingContext context) throws WingException
78      {
79          // FIXME: don't statically assign authenticated status or
80          // repositoryIdentifier.
81          super(context);
82          userMeta = new UserMeta(context);
83          pageMeta = new PageMeta(context);
84          repositoryMeta = new RepositoryMeta(context);
85      }
86  
87      /**
88       * Set a new user oriented metadata set.
89       * 
90       * @return The user oriented metadata set.
91       */
92      public UserMeta setUserMeta() throws WingException
93      {
94          return this.userMeta;
95      }
96  
97      /**
98       * Set a new page oriented metadata set.
99       * 
100      * @return The page oriented metadata set.
101      */
102     public PageMeta setPageMeta() throws WingException
103     {
104         return this.pageMeta;
105     }
106     
107     /**
108      * Set a new repository oriented metadata set.
109      * 
110      * @return The repository oriented metadata set.
111      */
112     public RepositoryMeta setRepositoryMeta() throws WingException
113     {
114         return this.repositoryMeta;
115     }
116     
117     /**
118      * Determine if the given SAX event is a Meta element.
119      * 
120      * @param namespace
121      *            The element's name space
122      * @param localName
123      *            The local, unqualified, name for this element
124      * @param qName
125      *            The qualified name for this element
126      * @param attributes
127      *            The element's attributes
128      * @return True if this WingElement is equivalent to the given SAX Event.
129      */
130     public boolean mergeEqual(String namespace, String localName, String qName,
131             Attributes attributes) throws SAXException, WingException
132     {
133         if (!WingConstants.DRI.URI.equals(namespace))
134             return false;
135 
136         if (!E_META.equals(localName))
137             return false;
138         
139         return true;
140     }
141 
142     /**
143      * Merge the given sub-domain of metadata elements.
144      * 
145      * @param namespace
146      *            The element's name space
147      * @param localName
148      *            The local, unqualified, name for this element *
149      * @param qName
150      *            The qualified name for this element
151      * @param attributes
152      *            The element's attributes
153      * @return The child element
154      */
155     public WingMergeableElement mergeChild(String namespace, String localName,
156             String qName, Attributes attributes) throws SAXException,
157             WingException
158     {
159         // User
160         if (this.userMeta != null
161                 && this.userMeta.mergeEqual(namespace, localName, qName,
162                         attributes))
163         {
164             UserMeta userMeta = this.userMeta;
165             this.userMeta = null;
166             return userMeta;
167         }
168 
169         // page
170         if (this.pageMeta != null
171                 && this.pageMeta.mergeEqual(namespace, localName, qName,
172                         attributes))
173         {
174             PageMeta pageMeta = this.pageMeta;
175             this.pageMeta = null;
176             return pageMeta;
177         }
178         
179         // repository
180         if (this.repositoryMeta != null
181                 && this.repositoryMeta.mergeEqual(namespace, localName, qName,
182                         attributes))
183         {
184             RepositoryMeta repositoryMeta = this.repositoryMeta;
185             this.repositoryMeta = null;
186             return repositoryMeta;
187         }
188         
189         return null;
190     }
191 
192     /**
193      * Notify this element that it is being merged.
194      * 
195      * @return The attributes for this merged element
196      */
197     public Attributes merge(Attributes attributes) throws SAXException,
198             WingException
199     {
200         this.merged = true;
201         return attributes;
202     }

204     /**
205      * Translate to SAX events
206      * 
207      * @param contentHandler
208      *            (Required) The registered contentHandler where SAX events
209      *            should be routed too.
210      * @param lexicalHandler
211      *            (Required) The registered lexicalHandler where lexical 
212      *            events (such as CDATA, DTD, etc) should be routed too.
213      * @param namespaces
214      *            (Required) SAX Helper class to keep track of namespaces able
215      *            to determine the correct prefix for a given namespace URI.
216      */
217     public void toSAX(ContentHandler contentHandler, LexicalHandler lexicalHandler, 
218             NamespaceSupport namespaces) throws SAXException
219     {
220         if (!merged)
221             startElement(contentHandler, namespaces, E_META, null);
222 
223         if (this.userMeta != null)
224             this.userMeta.toSAX(contentHandler, lexicalHandler, namespaces);
225         if (this.pageMeta != null)
226             this.pageMeta.toSAX(contentHandler, lexicalHandler, namespaces);
227         if (this.repositoryMeta != null)
228             this.repositoryMeta.toSAX(contentHandler, lexicalHandler, namespaces);
229            
230         if (!merged)
231             endElement(contentHandler, namespaces, E_META);
232     }

234     /**
235      * dispose
236      */
237     public void dispose()
238     {
239         if (this.userMeta != null)
240             this.userMeta.dispose();
241         if (this.pageMeta != null)
242             this.pageMeta.dispose();
243         if (this.repositoryMeta != null)
244             this.repositoryMeta.dispose();
245         
246         this.userMeta = null;
247         this.pageMeta = null;
248         this.repositoryMeta = null;
249         
250         super.dispose();
251     }
