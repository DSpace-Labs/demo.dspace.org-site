68  /**
69   * This is an adapter which translates DSpace containers 
70   * (communities & collections) into METS documents. This adapter follows
71   * the DSpace METS profile however that profile does not define how a
72   * community or collection should be described, but we make the obvious 
73   * decisions to deviate when nessasary from the profile.
74   * 
75   * The METS document consists of three parts: descriptive metadata section,
76   * file section, and a structural map. The descriptive metadata sections holds
77   * metadata about the item being adapted using DSpace crosswalks. This is the 
78   * same way the item adapter works.
79   * 
80   * However the file section and structural map are a bit different. In these
81   * casses the the only files listed is the one logo that may be attached to 
82   * a community or collection.
83   * 
84   * @author Scott Phillips
85   */
86  public class ContainerAdapter extends AbstractAdapter

215     /**
216      * Render the METS descriptive section. This will create a new metadata
217      * section for each crosswalk configured.
218      * 
219      * Example:
220      * <dmdSec>
221      *  <mdWrap MDTYPE="MODS">
222      *    <xmlData>
223      *      ... content from the crosswalk ...
224      *    </xmlDate>
225      *  </mdWrap>
226      * </dmdSec
227      */
228     protected void renderDescriptiveSection() throws WingException, SAXException, CrosswalkException, IOException, SQLException 
229     {
230     	AttributeMap attributes;
231     	
232         String groupID = getGenericID("group_dmd_");
233         dmdSecIDS = new StringBuffer();
234 
235         // Add DIM descriptive metadata if it was requested or if no metadata types 
236     	// were specified. Further more since this is the default type we also use a 
237     	// faster rendering method that the crosswalk API.
238     	if(dmdTypes.size() == 0 || dmdTypes.contains("DIM"))
239     	{
240         	// Metadata element's ID
241         	String dmdID = getGenericID("dmd_");
242         	
243         	// Keep track of all descriptive sections
244             dmdSecIDS.append(dmdID);
245             
246         	
247             // ////////////////////////////////
248             // Start a new dmdSec for each crosswalk.
249             attributes = new AttributeMap();
250             attributes.put("ID", dmdID);
251             attributes.put("GROUPID", groupID);
252             startElement(METS,"dmdSec",attributes);
253             
254             // ////////////////////////////////
255             // Start metadata wrapper
256             attributes = new AttributeMap();
257         	attributes.put("MDTYPE", "OTHER");
258         	attributes.put("OTHERMDTYPE", "DIM");
259             startElement(METS,"mdWrap",attributes);
260 
261             // ////////////////////////////////
262             // Start the xml data
263             startElement(METS,"xmlData");
264             
265             
266             // ///////////////////////////////
267 			// Start the DIM element
268 			attributes = new AttributeMap();
269 			attributes.put("dspaceType", Constants.typeText[dso.getType()]);
270 			startElement(DIM,"dim",attributes);
271 
272             // Add each field for this collection
273             if (dso.getType() == Constants.COLLECTION) 
274             {
275                 Collection collection = (Collection) dso;
276                 
277                 String description = collection.getMetadata("introductory_text");
278                 String description_abstract = collection.getMetadata("short_description");
279                 String description_table = collection.getMetadata("side_bar_text");
280                 String identifier_uri = "http://hdl.handle.net/" + collection.getHandle();
281                 String provenance = collection.getMetadata("provenance_description");
282                 String rights = collection.getMetadata("copyright_text");
283                 String rights_license = collection.getMetadata("license");
284                 String title = collection.getMetadata("name");
285                 
286                 createField("dc","description",null,null,description);
287                 createField("dc","description","abstract",null,description_abstract);
288                 createField("dc","description","tableofcontents",null,description_table);
289                 createField("dc","identifier","uri",null,identifier_uri);
290                 createField("dc","provenance",null,null,provenance);
291                 createField("dc","rights",null,null,rights);
292                 createField("dc","rights","license",null,rights_license);
293                 createField("dc","title",null,null,title);
294                 
295                 boolean useCache = ConfigurationManager.getBooleanProperty("webui.strengths.cache");
296         		
297                 //To improve scalability, XMLUI only adds item counts if they are cached
298                 if (useCache)
299         		{
300 	                try
301 	                {	//try to determine Collection size (i.e. # of items)
302 	                	int size = new ItemCounter(this.dspaceContext).getCount(collection);
303 	                	createField("dc","format","extent",null, String.valueOf(size)); 
304 	                }
305 	                catch(ItemCountException e)
306 	                {
307 	                    IOException ioe = new IOException("Could not obtain Collection item-count");
308 	                    ioe.initCause(e);
309 	                	throw ioe;
310 	                }
311         		}
312             } 
313             else if (dso.getType() == Constants.COMMUNITY) 
314             {
315                 Community community = (Community) dso;
316                 
317                 String description = community.getMetadata("introductory_text");
318                 String description_abstract = community.getMetadata("short_description");
319                 String description_table = community.getMetadata("side_bar_text");
320                 String identifier_uri = "http://hdl.handle.net/" + community.getHandle();
321                 String rights = community.getMetadata("copyright_text");
322                 String title = community.getMetadata("name");
323                 
324                 createField("dc","description",null,null,description);
325                 createField("dc","description","abstract",null,description_abstract);
326                 createField("dc","description","tableofcontents",null,description_table);
327                 createField("dc","identifier","uri",null,identifier_uri);
328                 createField("dc","rights",null,null,rights);
329                 createField("dc","title",null,null,title);
330                 
331                 boolean useCache = ConfigurationManager.getBooleanProperty("webui.strengths.cache");
332         		
333                 //To improve scalability, XMLUI only adds item counts if they are cached
334         		if (useCache)
335         		{
336         			try
337 	                {	//try to determine Community size (i.e. # of items)
338 	                	int size = new ItemCounter(this.dspaceContext).getCount(community);
339 	                	createField("dc","format","extent",null, String.valueOf(size)); 
340 	                }
341 	                catch(ItemCountException e)
342 	                {
343 	                	IOException ioe = new IOException("Could not obtain Collection item-count");
344 		                ioe.initCause(e);
345 		                throw ioe;
346 	                }
347         		}
348             }
349             
350             // ///////////////////////////////
351 			// End the DIM element
352 			endElement(DIM,"dim");
353             
354             // ////////////////////////////////
355             // End elements
356             endElement(METS,"xmlData");
357             endElement(METS,"mdWrap");
358             endElement(METS, "dmdSec");
359           
360         }
361         
362         
363         
364         
365         
366         
367         
368     	for (String dmdType : dmdTypes)
369     	{
370     		// If DIM was requested then it was generated above without using
371     		// the crosswalk API. So we can skip this one.
372     		if ("DIM".equals(dmdType))
373     			continue;
374     		
375     		DisseminationCrosswalk crosswalk = getDisseminationCrosswalk(dmdType);
376     		
377     		if (crosswalk == null)
378     			continue;
379     		
380         	String dmdID = getGenericID("dmd_");
381 	   		// Add our id to the list.
382 	   		dmdSecIDS.append(" " + dmdID);
383 
384             // ////////////////////////////////
385             // Start a new dmdSec for each crosswalk.
386             attributes = new AttributeMap();
387             attributes.put("ID", dmdID);
388             attributes.put("GROUPID", groupID);
389             startElement(METS,"dmdSec",attributes);
390             
391             // ////////////////////////////////
392             // Start metadata wrapper
393             attributes = new AttributeMap();
394             if (isDefinedMETStype(dmdType))
395             {
396             	attributes.put("MDTYPE", dmdType);
397             }
398             else
399             {
400             	attributes.put("MDTYPE", "OTHER");
401             	attributes.put("OTHERMDTYPE", dmdType);
402             }
403             startElement(METS,"mdWrap",attributes);
404 
405             // ////////////////////////////////
406             // Start the xml data
407             startElement(METS,"xmlData");
408             
409             // ///////////////////////////////
410     		// Send the actual XML content
411             try {
412 	    		Element dissemination = crosswalk.disseminateElement(dso);
413 	
414 	    		SAXFilter filter = new SAXFilter(contentHandler, lexicalHandler, namespaces);
415 	    		// Allow the basics for XML
416 	    		filter.allowElements().allowIgnorableWhitespace().allowCharacters().allowCDATA().allowPrefixMappings();
417 	    		
418 	            SAXOutputter outputter = new SAXOutputter();
419 	            outputter.setContentHandler(filter);
420 	            outputter.setLexicalHandler(filter);
421 				outputter.output(dissemination);
422 			} 
423             catch (JDOMException jdome) 
424 			{
425 				throw new WingException(jdome);
426 			}
427 			catch (AuthorizeException ae)
428 			{
429 				// just ignore the authorize exception and continue on with
430 				//out parsing the xml document.
431 			}
432     		
433             
434             // ////////////////////////////////
435             // End elements
436             endElement(METS,"xmlData");
437             endElement(METS,"mdWrap");
438             endElement(METS, "dmdSec");
439             
440             
441             // Record keeping
442             if (dmdSecIDS == null)
443             {
444                 dmdSecIDS = new StringBuffer(dmdID);
445             }
446             else
447             {
448                 dmdSecIDS.append(" " + dmdID);
449 
450             }
451         }
452     }

454     /**
455      * Render the METS file section. If a logo is present for this
456      * container then that single bitstream is listed in the 
457      * file section.
458      * 
459      * Example:
460      * <fileSec>
461      *   <fileGrp USE="LOGO">
462      *     <file ... >
463      *       <fLocate ... >
464      *     </file>
465      *   </fileGrp>
466      * </fileSec>
467      */
468     protected void renderFileSection() throws SAXException
469     {
470     	AttributeMap attributes;
471     	
472         // Get the Community or Collection logo.
473         Bitstream logo = getLogo();
474 
475         if (logo != null)
476         {
477             // ////////////////////////////////
478             // Start the file section
479         	startElement(METS,"fileSec");
480 
481             // ////////////////////////////////
482             // Start a new fileGrp for the logo.
483             attributes = new AttributeMap();
484             attributes.put("USE", "LOGO");
485             startElement(METS,"fileGrp",attributes);
486             
487             // ////////////////////////////////
488             // Add the actual file element
489             String fileID = getFileID(logo);
490             String groupID = getGroupFileID(logo);
491             renderFile(null, logo, fileID, groupID);
492             
493             // ////////////////////////////////
494             // End th file group and file section
495             endElement(METS,"fileGrp");
496             endElement(METS,"fileSec");
497         }
498     }

500     /**
501      * Render the container's structural map. This includes a refrence
502      * to the container's logo, if available, otherwise it is an empty 
503      * division that just states it is a DSpace community or Collection.
504      * 
505      * Examlpe:
506      * <structMap TYPE="LOGICAL" LABEL="DSpace">
507      *   <div TYPE="DSpace Collection" DMDID="space seperated list of ids">
508      *     <fptr FILEID="logo id"/>
509      *   </div>
510      * </structMap>
511      */
512     protected void renderStructureMap() throws SQLException, SAXException
513     {
514     	AttributeMap attributes;
515     	
516     	
517     	// ///////////////////////
518     	// Start a new structure map
519     	attributes = new AttributeMap();
520     	attributes.put("TYPE", "LOGICAL");
521     	attributes.put("LABEL", "DSpace");
522     	startElement(METS,"structMap",attributes);
523 
524     	// ////////////////////////////////
525     	// Start the special first division
526     	attributes = new AttributeMap();
527     	attributes.put("TYPE", getMETSLabel());
528     	// add references to the Descriptive metadata
529     	if (dmdSecIDS != null)
530     		attributes.put("DMDID", dmdSecIDS.toString());
531     	startElement(METS,"div",attributes);
532     	
533     	
534     	// add a fptr pointer to the logo.
535         Bitstream logo = getLogo();
536         if (logo != null)
537         {
538             // ////////////////////////////////
539             // Add a refrence to the logo as the primary bitstream.
540         	attributes = new AttributeMap();
541         	attributes.put("FILEID",getFileID(logo));
542         	startElement(METS,"fptr",attributes);
543         	endElement(METS,"fptr");
544         
545             
546             // ///////////////////////////////////////////////
547             // Add a div for the publicaly viewable bitstreams (i.e. the logo)
548         	attributes = new AttributeMap();
549             attributes.put("ID", getGenericID("div_"));
550             attributes.put("TYPE", "DSpace Content Bitstream");
551         	startElement(METS,"div",attributes);
552         	
553             // ////////////////////////////////
554             // Add a refrence to the logo as the primary bitstream.
555         	attributes = new AttributeMap();
556         	attributes.put("FILEID",getFileID(logo));
557         	startElement(METS,"fptr",attributes);
558         	endElement(METS,"fptr");
559         	
560         	// //////////////////////////
561         	// End the logo division
562         	endElement(METS,"div");
563         }
564 
565     	// ////////////////////////////////
566     	// End the special first division
567     	endElement(METS,"div");
568     	
569     	// ///////////////////////
570     	// End the structure map
571     	endElement(METS,"structMap");
572     }

605     /**
606      * Count how many occurance there is of the given
607      * character in the given string.
608      * 
609      * @param string The string value to be counted.
610      * @param character the character to count in the string.
611      */
612     private int countOccurances(String string, char character)
613     {
614     	if (string == null || string.length() == 0)
615     		return 0;
616     	
617     	int fromIndex = -1;
618         int count = 0;
619         
620         while (true)
621         {
622         	fromIndex = string.indexOf('>', fromIndex+1);
623         	
624         	if (fromIndex == -1)
625         		break;
626         	
627         	count++;
628         }
629         
630         return count;
631     }

633     /**
634      * Check if the given character sequence is located in the given
635      * string at the specified index. If it is then return true, otherwise false.
636      * 
637      * @param string The string to test against
638      * @param index The location within the string
639      * @param characters The character sequence to look for.
640      * @return true if the character sequence was found, otherwise false.
641      */
642     private boolean substringCompare(String string, int index, char ... characters)
643     {
644     	// Is the string long enough?
645     	if (string.length() <= index + characters.length)
646     		return false;
647     	
648     	// Do all the characters match?
649     	for (char character : characters)
650     	{
651     		if (string.charAt(index) != character)
652     			return false;
653     		index++;
654     	}
655     	
656     	return false;
657     }

659     /**
660      * Create a new DIM field element with the given attributes.
661      * 
662      * @param schema The schema the DIM field belongs too.
663      * @param element The element the DIM field belongs too.
664      * @param qualifier The qualifier the DIM field belongs too.
665      * @param language The language the DIM field belongs too.
666      * @param value The value of the DIM field.
667      * @return A new DIM field element
668      * @throws SAXException 
669      */
670     private void createField(String schema, String element, String qualifier, String language, String value) throws SAXException
671     {
672     	// ///////////////////////////////
673     	// Field element for each metadata field.
674     	AttributeMap attributes = new AttributeMap();
675 		attributes.put("mdschema",schema);
676 		attributes.put("element", element);
677 		if (qualifier != null)
678 			attributes.put("qualifier", qualifier);
679 		if (language != null)
680 			attributes.put("language", language);
681 		startElement(DIM,"field",attributes);
682 		
683 		// Only try and add the metadata's value, but only if it is non null.
684     	if (value != null)
685     	{
686     		// First, preform a queck check to see if the value may be XML.
687 	        int countOpen = countOccurances(value,'<');
688 	        int countClose = countOccurances(value, '>');
689 	        
690 	        // If it passed the quick test, then try and parse the value.
691 	        Element xmlDocument = null;
692 	        if (countOpen > 0 && countOpen == countClose)
693 	        {
694 	        	// This may be XML, First try and remove any bad entity refrences.
695 	        	int amp = -1;
696 	        	while ((amp = value.indexOf('&', amp+1)) > -1)
697 	        	{
698 	        		// Is it an xml entity named by number?
699 	        		if (substringCompare(value,amp+1,'#'))
700 	        			continue;
701 	        		
702 	        		// &amp;
703 	        		if (substringCompare(value,amp+1,'a','m','p',';'))
704 	        			continue;
705 	        		
706 	        		// &apos;
707 	        		if (substringCompare(value,amp+1,'a','p','o','s',';'))
708 	        			continue;
709 	        		
710 	        		// &quot;
711 	        		if (substringCompare(value,amp+1,'q','u','o','t',';'))
712 	        			continue;
713 	        			
714 	        		// &lt;
715 	        		if (substringCompare(value,amp+1,'l','t',';'))
716 	        			continue;
717 	        		
718 	        		// &gt;
719 	        		if (substringCompare(value,amp+1,'g','t',';'))
720 	        			continue;
721 	        		
722 	        		// Replace the ampersand with an XML entity.
723 	        		value = value.substring(0,amp) + "&amp;" + value.substring(amp+1);
724 	        	}
725 	        	
726 	        	
727 	        	// Second try and parse the XML into a mini-dom
728 	        	try {
729 	        		// Wrap the value inside a root element (which will be trimed out 
730 	        		// by the SAX filter and set the default namespace to XHTML. 
731 		        	String xml = "<?xml version='1.0' encoding='UTF-8'?><fragment xmlns=\"http://www.w3.org/1999/xhtml\">"+value+"</fragment>";
732 
733 		            ByteArrayInputStream inputStream = new ByteArrayInputStream(xml.getBytes("UTF-8"));
734 		     	   
735 			 	    SAXBuilder builder = new SAXBuilder();
736 					Document document = builder.build(inputStream);
737
