74  /**
75   * This is an adapter which translate a DSpace item into a METS document
76   * following the DSpace METS profile, err well mostly. At least if you use
77   * the proper configuration it will be fully complaint with the profile,
78   * however this adapter will allow you to configure it to be incorrect.
79   *
80   * When we are configured to be non-complaint with the profile the MET's
81   * profile is changed to reflect the diviation. The DSpace profile states
82   * that metadata should be given in MODS format. However you can configure
83   * this adapter to use any metadata crosswalk. When that case is detected we
84   * change the profile to say that we are divating from the standard profile
85   * and it lists what metadata has been added.
86   *
87   * There are four parts to an item's METS document: descriptive metadata,
88   * file section, structural map, and extra sections.
89   *
90   * @author Scott Phillips
91   */

222     /**
223      * Render the METS descriptive section. This will create a new metadata
224      * section for each crosswalk configured. Futher more, a special check
225      * has been aded that will add mods descriptive metadata if it is
226      * available in DSpace.
227      *
228      * Example:
229      * <dmdSec>
230      *  <mdWrap MDTYPE="MODS">
231      *    <xmlData>
232      *      ... content from the crosswalk ...
233      *    </xmlDate>
234      *  </mdWrap>
235      * </dmdSec
236      */
237     protected void renderDescriptiveSection() throws WingException, SAXException, CrosswalkException, IOException, SQLException
238     {
239                 AttributeMap attributes;
240         String groupID = getGenericID("group_dmd_");
241         dmdSecIDS = new StringBuffer();
242 
243         // Add DIM descriptive metadata if it was requested or if no metadata types
244         // were specified. Further more since this is the default type we also use a
245         // faster rendering method that the crosswalk API.
246         if(dmdTypes.size() == 0 || dmdTypes.contains("DIM"))
247         {
248                 // Metadata element's ID
249                 String dmdID = getGenericID("dmd_");
250                 // Keep track of all descriptive sections
251             dmdSecIDS.append("" + dmdID);
252                 
253                         ////////////////////////////////
254                         // Start a metadata wrapper
255                         attributes = new AttributeMap();
256                         attributes.put("ID", dmdID);
257                         attributes.put("GROUPID", groupID);
258                         startElement(METS, "dmdSec", attributes);
259         
260                          ////////////////////////////////
261                         // Start a metadata wrapper
262                         attributes = new AttributeMap();
263                         attributes.put("MDTYPE","OTHER");
264                         attributes.put("OTHERMDTYPE", "DIM");
265                         startElement(METS,"mdWrap",attributes);
266                         
267                         // ////////////////////////////////
268                         // Start the xml data
269                         startElement(METS,"xmlData");
270         
271                         
272                         // ///////////////////////////////
273                         // Start the DIM element
274                         attributes = new AttributeMap();
275                         attributes.put("dspaceType", Constants.typeText[item.getType()]);
276             if (item.isWithdrawn())
277                 attributes.put("withdrawn", "y");
278             startElement(DIM,"dim",attributes);
279                         
280                 DCValue[] dcvs = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
281                 for (DCValue dcv : dcvs)
282                 {
283                         if (!MetadataExposure.isHidden(context, dcv.schema, dcv.element, dcv.qualifier))
284                         {
285                         // ///////////////////////////////
286                         // Field element for each metadata field.
287                         attributes = new AttributeMap();
288                         attributes.put("mdschema",dcv.schema);
289                         attributes.put("element", dcv.element);
290                         if (dcv.qualifier != null)
291                                 attributes.put("qualifier", dcv.qualifier);
292                         if (dcv.language != null)
293                                 attributes.put("language", dcv.language);
294                         if (dcv.authority != null || dcv.confidence != Choices.CF_UNSET)
295                         {
296                                 attributes.put("authority", dcv.authority);
297                                 attributes.put("confidence", Choices.getConfidenceText(dcv.confidence));
298                         }
299                         startElement(DIM,"field",attributes);
300                         sendCharacters(dcv.value);
301                         endElement(DIM,"field");
302                 }
303                 }
304                         
305                 // ///////////////////////////////
306                         // End the DIM element
307                         endElement(DIM,"dim");
308                         
309                 // ////////////////////////////////
310                 // End elements
311                 endElement(METS,"xmlData");
312                 endElement(METS,"mdWrap");
313                 endElement(METS, "dmdSec");
314 
315         }
316                 
317         
318         // Add any extra crosswalks that may configured.
319         for (String dmdType : dmdTypes)
320         {
321                 // If DIM was requested then it was generated above without using
322                 // the crosswalk API. So we can skip this one.
323                 if ("DIM".equals(dmdType))
324                         continue;
325                 
326                 DisseminationCrosswalk crosswalk = getDisseminationCrosswalk(dmdType);
327                 
328                 if (crosswalk == null)
329                         continue;
330                 
331                 String dmdID = getGenericID("dmd_");
332                  // Add our id to the list.
333                 dmdSecIDS.append(" " + dmdID);
334                 
335                 ////////////////////////////////
336                 // Start a metadata wrapper
337                 attributes = new AttributeMap();
338                 attributes.put("ID", dmdID);
339                 attributes.put("GROUPID", groupID);
340                 startElement(METS, "dmdSec", attributes);
341 
342                  ////////////////////////////////
343                 // Start a metadata wrapper
344                 attributes = new AttributeMap();
345                 if (isDefinedMETStype(dmdType))
346                 {
347                         attributes.put("MDTYPE", dmdType);
348                 }
349                 else
350                 {
351                         attributes.put("MDTYPE","OTHER");
352                         attributes.put("OTHERMDTYPE", dmdType);
353                 }
354                 startElement(METS,"mdWrap",attributes);
355                 
356                 // ////////////////////////////////
357                 // Start the xml data
358                 startElement(METS,"xmlData");
359 
360                 
361                 // ///////////////////////////////
362                 // Send the actual XML content
363                 try {
364                         Element dissemination = crosswalk.disseminateElement(item);
365         
366                         SAXFilter filter = new SAXFilter(contentHandler, lexicalHandler, namespaces);
367                         // Allow the basics for XML
368                         filter.allowElements().allowIgnorableWhitespace().allowCharacters().allowCDATA().allowPrefixMappings();
369                         
370                     SAXOutputter outputter = new SAXOutputter();
371                     outputter.setContentHandler(filter);
372                     outputter.setLexicalHandler(filter);
373                                 outputter.output(dissemination);
374                         }
375             catch (JDOMException jdome)
376                         {
377                                 throw new WingException(jdome);
378                         }
379                         catch (AuthorizeException ae)
380                         {
381                                 // just ignore the authorize exception and continue on with
382                                 //out parsing the xml document.
383                         }
384                 
385             
386             // ////////////////////////////////
387             // End elements
388             endElement(METS,"xmlData");
389             endElement(METS,"mdWrap");
390             endElement(METS, "dmdSec");
391         }
392 
393 
394         // Check to see if there is an in-line MODS document
395         // stored as a bitstream. If there is then we should also
396         // include these metadata in our METS document. However
397         // we don't really know what the document describes, so we
398         // but it in it's own dmd group.
399 
400         Boolean include = ConfigurationManager.getBooleanProperty("xmlui.bitstream.mods");
401         if (include && dmdTypes.contains("MODS"))
402         {
403                 // Generate a second group id for any extra metadata added.
404                 String groupID2 = getGenericID("group_dmd_");
405                 
406                 Bundle[] bundles = item.getBundles("METADATA");
407                 for (Bundle bundle : bundles)
408                 {
409                         Bitstream bitstream = bundle.getBitstreamByName("MODS.xml");
410         
411                         if (bitstream == null)
412                                 continue;
413                         
414                         
415                         String dmdID = getGenericID("dmd_");
416                         
417                         
418                         ////////////////////////////////
419                         // Start a metadata wrapper
420                         attributes = new AttributeMap();
421                         attributes.put("ID", dmdID);
422                         attributes.put("GROUPID", groupID2);
423                         startElement(METS, "dmdSec", attributes);
424         
425                          ////////////////////////////////
426                         // Start a metadata wrapper
427                         attributes = new AttributeMap();
428                         attributes.put("MDTYPE", "MODS");
429                         startElement(METS,"mdWrap",attributes);
430                         
431                         // ////////////////////////////////
432                         // Start the xml data
433                         startElement(METS,"xmlData");
434                         
435                         
436                         // ///////////////////////////////
437                         // Send the actual XML content
438                         
439                         SAXFilter filter = new SAXFilter(contentHandler, lexicalHandler, namespaces);
440                         // Allow the basics for XML
441                         filter.allowElements().allowIgnorableWhitespace().allowCharacters().allowCDATA().allowPrefixMappings();
442                         
443                         XMLReader reader = XMLReaderFactory.createXMLReader();
444                         reader.setContentHandler(filter);
445                         reader.setProperty("http://xml.org/sax/properties/lexical-handler", filter);
446                         try {
447                                 InputStream is = bitstream.retrieve();
448                                 reader.parse(new InputSource(is));
449                         }
450                         catch (AuthorizeException ae)
451                         {
452                                 // just ignore the authorize exception and continue on with
453                                 //out parsing the xml document.
454                         }
455                         
456                         // ////////////////////////////////
457                     // End elements
458                     endElement(METS,"xmlData");
459                     endElement(METS,"mdWrap");
460                     endElement(METS, "dmdSec");
461                 }
462         }
463     
464     }

466     /**
467      * Render the METS administrative section.
468      *
469      * Example:
470      * <amdSec>
471      *  <mdWrap MDTYPE="OTHER" OTHERMDTYPE="METSRights">
472      *    <xmlData>
473      *      ... content from the crosswalk ...
474      *    </xmlDate>
475      *  </mdWrap>
476      * </amdSec>
477      */
478     protected void renderAdministrativeSection() throws WingException, SAXException, CrosswalkException, IOException, SQLException
479     {
480         AttributeMap attributes;
481         String groupID;
482 
483         //Only create an <amdSec>, if we have amdTypes (or sub-sections) specified...
484         // (this keeps our METS file small, by default, and hides all our admin metadata)
485         if(amdTypes.size() > 0)
486         {
487           ////////////////////////////////
488           // Start an administrative wrapper
489 
490           // Administrative element's ID
491           String amdID = getGenericID("amd_");
492           attributes = new AttributeMap();
493           attributes.put("ID", amdID);
494           startElement(METS, "amdSec", attributes);
495 
496           groupID = getGenericID("group_amd_");
497           attributes.put("GROUPID", groupID);
498         }
499 
500         // For each administrative metadata section specified
501         for (String amdSecName : amdTypes.keySet())
502         {
503           //get a list of metadata crosswalks which will be used to build
504           // this administrative metadata section
505           List<String> mdTypes = amdTypes.get(amdSecName);
506 
507           // For each crosswalk
508           for (String mdType : mdTypes)
509           {
510             //get our dissemination crosswalk
511             DisseminationCrosswalk crosswalk = getDisseminationCrosswalk(mdType);
512 
513             //skip, if we cannot find this crosswalk in config file
514             if (crosswalk == null)
515               continue;
516 
517             //First, check if this crosswalk can handle disseminating Item-level Administrative metadata
518             if(crosswalk.canDisseminate(item))
519             {
520               //Since this crosswalk works with items, first render a section for entire item
521               renderAmdSubSection(amdSecName, mdType, crosswalk, item);
522             }
523 
524             //Next, we'll try and render Bitstream-level administrative metadata
525             // (Although, we're only rendering this metadata for the bundles specified)
526             List<Bundle> bundles = findEnabledBundles();
527             for (Bundle bundle : bundles)
528             {
529               Bitstream[] bitstreams = bundle.getBitstreams();
530 
531               //Create a sub-section of <amdSec> for each bitstream in bundle
532               for(Bitstream bitstream : bitstreams)
533               {
534                  //Only render the section if crosswalk works with bitstreams
535                  if(crosswalk.canDisseminate(bitstream))
536                  {
537                     renderAmdSubSection(amdSecName, mdType, crosswalk, bitstream);
538                  }
539               }//end for each bitstream
540             }//end for each bundle
541           }//end for each crosswalk
542         }//end for each amdSec
543         
544         if(amdTypes.size() > 0)
545         {
546           //////////////////////////////////
547           // End administrative section
548           endElement(METS,"amdSec");
549         }
550     }

552     /**
553      * Render a sub-section of the administrative metadata section.
554      * Valid sub-sections include: techMD, rightsMD, sourceMD, digiprovMD
555      *
556      * Example:
557      * <techMD>
558      *   <mdWrap MDTYPE="PREMIS">
559      *     <xmlData>
560      *       [PREMIS content ... ]
561      *     </xmlData>
562      *   </mdWrap>
563      * </techMD>
564      *
565      * @param amdSecName Name of administrative metadata section
566      * @param mdType Type of metadata section (e.g. PREMIS)
567      * @param crosswalk The DisseminationCrosswalk to use to generate this section
568      * @param dso The current DSpace object to use the crosswalk on
569      */
570     protected void renderAmdSubSection(String amdSecName, String mdType, DisseminationCrosswalk crosswalk, DSpaceObject dso)
571             throws WingException, SAXException, CrosswalkException, IOException, SQLException
572     {
573         /////////////////////////////////
574         // Start administrative metadata section wrapper
575         String amdSecID = getAmdSecID(amdSecName, mdType, dso);
576         AttributeMap attributes = new AttributeMap();
577         attributes.put("ID", amdSecID);
578         startElement(METS, amdSecName, attributes);
579 
580         //If this is a bitstream
581         if (dso.getType() == Constants.BITSTREAM)
582         {
583           // Add this to our list of each file's administrative section IDs
584           String fileID = getFileID((Bitstream) dso);
585           if(fileAmdSecIDs.containsKey(fileID))
586             fileAmdSecIDs.get(fileID).append(" " + amdSecID);
587           else
588             fileAmdSecIDs.put(fileID, new StringBuffer(amdSecID));
589         }//else if an Item
590         else if (dso.getType() == Constants.ITEM)
591         {
592            //Add this to our list of item's administrative section IDs
593            if(amdSecIDS==null)
594              amdSecIDS = new StringBuffer(amdSecID);
595            else
596              amdSecIDS.append(" " + amdSecID);
597         }
598 
599         ////////////////////////////////
600         // Start a metadata wrapper
601         attributes = new AttributeMap();
602         if (isDefinedMETStype(mdType))
603         {
604             attributes.put("MDTYPE", mdType);
605         }
606         else
607         {
608             attributes.put("MDTYPE","OTHER");
609             attributes.put("OTHERMDTYPE", mdType);
610         }
611         startElement(METS,"mdWrap",attributes);
612 
613         //////////////////////////////////
614         // Start the xml data
615         startElement(METS,"xmlData");
616 
617         /////////////////////////////////
618         // Send the actual XML content,
619         // using the PREMIS crosswalk for each bitstream
620         try {
621             Element dissemination = crosswalk.disseminateElement(dso);
622 
623             SAXFilter filter = new SAXFilter(contentHandler, lexicalHandler, namespaces);
624             // Allow the basics for XML
625             filter.allowElements().allowIgnorableWhitespace().allowCharacters().allowCDATA().allowPrefixMappings();
626 
627             SAXOutputter outputter = new SAXOutputter();
628             outputter.setContentHandler(filter);
629             outputter.setLexicalHandler(filter);
630             outputter.output(dissemination);
631         }
632         catch (JDOMException jdome)
633         {
634             throw new WingException(jdome);
635         }
636         catch (AuthorizeException ae)
637         {
638             // just ignore the authorize exception and continue on with
639             //out parsing the xml document.
640         }
641 
642         // ////////////////////////////////
643         // End elements
644         endElement(METS,"xmlData");
645         endElement(METS,"mdWrap");
646         endElement(METS,amdSecName);
647     }

649     /**
650      * Render the METS file section. This will contain a list of all bitstreams in the
651      * item. Each bundle, even those that are not typically displayed will be listed.
652      *
653      * Example:
654      * <fileSec>
655      *   <fileGrp USE="CONTENT">
656      *     <file ... >
657      *       <fLocate ... >
658      *     </file>
659      *   </fileGrp>
660      *   <fileGrp USE="TEXT">
661      *     <file ... >
662      *       <fLocate ... >
663      *     </file>
664      *   </fileGrp>
665      * </fileSec>
666      */
667     protected void renderFileSection() throws SQLException, SAXException
668     {
669         AttributeMap attributes;
670         
671         // //////////////////////
672         // Start a new file section
673         startElement(METS,"fileSec");
674         
675         // Check if the user is requested a specific bundle or
676         // the all bundles.
677         List<Bundle> bundles = findEnabledBundles();
678         
679         // Loop over all requested bundles
680         for (Bundle bundle : bundles)
681         {
682 
683             // Use the bitstream's name as the use parameter unless we
684             // are the original bundle. In this case rename it to
685             // content.
686             String use = bundle.getName();
687             boolean isContentBundle = false; // remember the content bundle.
688             boolean isDerivedBundle = false;
689             if ("ORIGINAL".equals(use))
690             {
691                 use = "CONTENT";
692                 isContentBundle = true;
693             }
694             if ("TEXT".equals(bundle.getName()) || "THUMBNAIL".equals(bundle.getName()))
695             {
696                 isDerivedBundle = true;
697             }
698 
699             // ///////////////////
700             // Start bundle's file group
701             attributes = new AttributeMap();
702             attributes.put("USE", use);
703             startElement(METS,"fileGrp",attributes);
704             
705             for (Bitstream bitstream : bundle.getBitstreams())
706             {
707                 // //////////////////////////////
708                 // Determine the file's IDs
709                 String fileID = getFileID(bitstream);
710                 
711                 Bitstream originalBitstream = null;
712                 if (isDerivedBundle)
713                         originalBitstream = findOriginalBitstream(item, bitstream);
714                 String groupID = getGroupFileID((originalBitstream == null) ? bitstream : originalBitstream );
715 
716                 //Check if there were administrative metadata sections corresponding to this file
717                 String admIDs = null;
718                 if(fileAmdSecIDs.containsKey(fileID))
719                   admIDs = fileAmdSecIDs.get(fileID).toString();
720   
721                 // Render the actual file & flocate elements.
722                 renderFile(item, bitstream, fileID, groupID, admIDs);
723 
724                 // Remember all the viewable content bitstreams for later in the
725                 // structMap.
726                 if (isContentBundle)
727                 {
728                     contentBitstreams.add(bitstream);
729                     if (bundle.getPrimaryBitstreamID() == bitstream.getID())
730                         primaryBitstream = bitstream;
731                 }
732             }
733             
734             // ///////////////////
735             // End the bundle's file group
736             endElement(METS,"fileGrp");
737         }
738         
739         // //////////////////////
740         // End the file section
741         endElement(METS,"fileSec");
742     }

746      * Render the item's structural map. This includes a list of
747      * content bitstreams, those are bistreams that are typicaly
748      * viewable by the end user.
749      *
750      * Examlpe:
751      * <structMap TYPE="LOGICAL" LABEL="DSpace">
752      *   <div TYPE="DSpace Item" DMDID="space seperated list of ids">
753      *     <fptr FILEID="primary bitstream"/>
754      *     ... a div for each content bitstream.
755      *   </div>
756      * </structMap>
757      */
758     protected void renderStructureMap() throws SQLException, SAXException
759     {
760         AttributeMap attributes;
761         
762         // ///////////////////////
763         // Start a new structure map
764         attributes = new AttributeMap();
765         attributes.put("TYPE", "LOGICAL");
766         attributes.put("LABEL", "DSpace");
767         startElement(METS,"structMap",attributes);
768 
769         // ////////////////////////////////
770         // Start the special first division
771         attributes = new AttributeMap();
772         attributes.put("TYPE", "DSpace Item");
773         // add references to the Descriptive metadata
774         if (dmdSecIDS != null)
775                 attributes.put("DMDID", dmdSecIDS.toString());
776         // add references to the Administrative metadata
777         if (amdSecIDS != null)
778                 attributes.put("AMDID", amdSecIDS.toString());
779         startElement(METS,"div",attributes);
780         
781         // add a fptr pointer to the primary bitstream.
782         if (primaryBitstream != null)
783         {
784                 // ////////////////////////////////
785                 // Start & end a refrence to the primary bistream.
786                 attributes = new AttributeMap();
787                 String fileID = getFileID(primaryBitstream);
788                 attributes.put("FILEID", fileID);
789                 
790                 startElement(METS,"fptr",attributes);
791                 endElement(METS,"fptr");
792         }
793 
794         for (Bitstream bitstream : contentBitstreams)
795         {
796                 // ////////////////////////////////////
797                 // Start a div for each publicaly viewable bitstream
798                 attributes = new AttributeMap();
799                 attributes.put("ID", getGenericID("div_"));
800                 attributes.put("TYPE", "DSpace Content Bitstream");
801                 startElement(METS,"div",attributes);
802 
803                 // ////////////////////////////////
804                 // Start a the actualy pointer to the bitstream
805                 attributes = new AttributeMap();
806                 String fileID = getFileID(bitstream);
807                 attributes.put("FILEID", fileID);
808                 
809                 startElement(METS,"fptr",attributes);
810                 endElement(METS,"fptr");
811                 
812                 // ///////////////////////////////
813                 // End the div
814                 endElement(METS,"div");
815         }
816 
817         // ////////////////////////////////
818         // End the special first division
819         endElement(METS,"div");
820         
821         // ///////////////////////
822         // End the structure map
823         endElement(METS,"structMap");
824     }

828     /**
829      * Render any extra METS section. If the item contains a METS.xml document
830      * then all of that document's sections are included in this document's
831      * METS document.
832      */
833     protected void renderExtraSections() throws SAXException, SQLException, IOException
834     {
835         Boolean include = ConfigurationManager.getBooleanProperty("xmlui.bitstream.mets");
836         if (!include)
837                 return;
838                 
839                 
840         Bundle[] bundles = item.getBundles("METADATA");
841 
842         for (Bundle bundle : bundles)
843         {
844                 Bitstream bitstream = bundle.getBitstreamByName("METS.xml");
845 
846                 if (bitstream == null)
847                         continue;
848 
849                 // ///////////////////////////////
850                 // Send the actual XML content
851                 try {
852                         SAXFilter filter = new SAXFilter(contentHandler, lexicalHandler, namespaces);
853                         // Allow the basics for XML
854                         filter.allowIgnorableWhitespace().allowCharacters().allowCDATA().allowPrefixMappings();
855                         // Special option, only allow elements below the second level to pass through. This
856                         // will trim out the METS declaration and only leave the actual METS parts to be
857                         // included.
858                         filter.allowElements(1);
859                         
860                         
861                         XMLReader reader = XMLReaderFactory.createXMLReader();
862                         reader.setContentHandler(filter);
863                         reader.setProperty("http://xml.org/sax/properties/lexical-handler", filter);
864                         reader.parse(new InputSource(bitstream.retrieve()));
865                 }
866                         catch (AuthorizeException ae)
867                         {
868                                 // just ignore the authorize exception and continue on with
869                                 //out parsing the xml document.
870                         }
871         }
872     }
