67  /**
68   * This is the abstract adapter containing all the common elements between
69   * the three types of adapters: item, container, and repository. Each adapter
70   * translate a given type of DSpace object into a METS document for rendering
71   * into the DRI document.
72   * 
73   * This class provides the chasses for those unique parts of the document to be
74   * build upon, there are seven rendering methods that may be overriden for each
75   * section of the METS document.
76   * 
77   * Header
78   * Descriptive Section
79   * Administrative Section
80   * File Section
81   * Structure Map
82   * Structural Link
83   * Behavioral Section
84   * 
85   * @author Scott Phillips
86   */
87

315 	/**
316 	 * Render the complete METS document.
317 	 */
318     public void renderMETS(ContentHandler contentHandler, LexicalHandler lexicalHandler) throws WingException, SAXException, CrosswalkException, IOException, SQLException 
319     {
320     		this.contentHandler = contentHandler;
321     		this.lexicalHandler = lexicalHandler;
322     		this.namespaces = new NamespaceSupport();
323     	
324     	
325     		// Declare our namespaces
326     		namespaces.pushContext();
327     		namespaces.declarePrefix("mets", METS.URI);
328     		namespaces.declarePrefix("xlink", XLINK.URI);
329     		namespaces.declarePrefix("xsi", XSI.URI);
330     		namespaces.declarePrefix("dim", DIM.URI);
331     		contentHandler.startPrefixMapping("mets", METS.URI);
332     		contentHandler.startPrefixMapping("xlink", XLINK.URI);
333     		contentHandler.startPrefixMapping("xsi", XSI.URI);
334     		contentHandler.startPrefixMapping("dim", DIM.URI);
335     		
336     		// Send the METS element
337     		AttributeMap attributes = new AttributeMap();
338     		attributes.put("ID", getMETSID());
339     		attributes.put("PROFILE", getMETSProfile());
340     		attributes.put("LABEL", getMETSLabel());
341     		String objid = getMETSOBJID();
342     		if (objid != null)
343     			attributes.put("OBJID", objid);
344 
345             // Include the link for editing the item
346             objid = getMETSOBJEDIT();
347             if (objid != null)
348                 attributes.put("OBJEDIT", objid);
349 
350     		startElement(METS,"METS",attributes);
351 
352     		// If the user requested no specefic sections then render them all.
353     		boolean all = (sections.size() == 0);
354     		
355     		if (all || sections.contains("metsHdr"))
356     			renderHeader();
357     		if (all || sections.contains("dmdSec"))
358     			renderDescriptiveSection();
359     		if (all || sections.contains("amdSec"))
360     			renderAdministrativeSection();
361     		if (all || sections.contains("fileSec"))
362     			renderFileSection();
363     		if (all || sections.contains("structMap"))
364     			renderStructureMap();
365     		if (all || sections.contains("structLink"))
366     			renderStructuralLink();
367     		if (all || sections.contains("behaviorSec"))
368     			renderBehavioralSection();
369     		
370     		// FIXME: this is not a met's section, it should be removed
371     		if (all || sections.contains("extraSec"))
372     			renderExtraSections();
373     		
374     		endElement(METS,"METS");
375     		contentHandler.endPrefixMapping("mets");
376     		contentHandler.endPrefixMapping("xlink");
377     		contentHandler.endPrefixMapping("dim");
378     		namespaces.popContext();
379 
380     }

416 	/**
417      * Generate a METS file element for a given bitstream.
418      * 
419      * @param item
420      *            If the bitstream is associated with an item provid the item
421      *            otherwise leave null.
422      * @param bitstream
423      *            The bitstream to build a file element for.
424      * @param fileID
425      *            The unique file id for this file.
426      * @param groupID
427      *            The group id for this file, if it is derived from another file
428      *            then they should share the same groupID.
429      * @param admID
430      *            The IDs of the administrative metadata sections which pertain
431      *            to this file
432      * @return The METS file element.
433      */
434 	protected void renderFile(Item item, Bitstream bitstream, String fileID, String groupID, String admID) throws SAXException
435 	{
436 		AttributeMap attributes;
437 		
438 		// //////////////////////////////
439     	// Determine the file attributes
440         BitstreamFormat format = bitstream.getFormat();
441         String mimeType = null;
442         if (format != null)
443             mimeType = format.getMIMEType();
444         String checksumType = bitstream.getChecksumAlgorithm();
445         String checksum = bitstream.getChecksum();
446         long size = bitstream.getSize();
447     	
448         // ////////////////////////////////
449         // Start the actual file
450         attributes = new AttributeMap();
451         attributes.put("ID", fileID);
452         attributes.put("GROUPID",groupID);
453         if (admID != null && admID.length()>0)
454             attributes.put("ADMID", admID);
455         if (mimeType != null && mimeType.length()>0)
456         	attributes.put("MIMETYPE", mimeType);
457         if (checksumType != null && checksum != null)
458         {
459         	attributes.put("CHECKSUM", checksum);
460         	attributes.put("CHECKSUMTYPE", checksumType);
461         }
462         attributes.put("SIZE", String.valueOf(size));
463         startElement(METS,"file",attributes);
464         
465         
466         // ////////////////////////////////////
467         // Determine the file location attributes
468         String name = bitstream.getName();
469         String description = bitstream.getDescription();
470 
471         
472         // If possible refrence this bitstream via a handle, however this may
473         // be null if a handle has not yet been assigned. In this case refrence the
474         // item its internal id. In the last case where the bitstream is not associated
475         // with an item (such as a community logo) then refrence the bitstreamID directly.
476         String identifier = null;
477         if (item != null && item.getHandle() != null)
478         	identifier = "handle/"+item.getHandle();
479         else if (item != null)
480         	identifier = "item/"+item.getID();
481         else
482         	identifier = "id/"+bitstream.getID();
483         
484         
485         String url = contextPath + "/bitstream/"+identifier+"/";
486         
487         // If we can put the pretty name of the bitstream on the end of the URL
488         try
489         {
490         	if (bitstream.getName() != null)
491         		url += Util.encodeBitstreamName(bitstream.getName(), "UTF-8");
492         }
493         catch (UnsupportedEncodingException uee)
494         {
495             // just ignore it, we don't have to have a pretty
496             // name on the end of the url because the sequence id will 
497         	// locate it. However it means that links in this file might
498         	// not work....
499         }
500         
501         url += "?sequence="+bitstream.getSequenceID();
502         
503         
504         // //////////////////////
505         // Start the file location
506         attributes = new AttributeMap();
507         AttributeMap attributesXLINK = new AttributeMap();
508         attributesXLINK.setNamespace(XLINK);
509         attributes.put("LOCTYPE", "URL");
510         attributesXLINK.put("type","locator");
511         attributesXLINK.put("title", name);
512         if (description != null)
513         	attributesXLINK.put("label",description);
514         attributesXLINK.put("href", url);
515         startElement(METS,"FLocat",attributes,attributesXLINK);
516         
517 
518         // ///////////////////////
519         // End file location
520         endElement(METS,"FLocate");
521         
522         // ////////////////////////////////
523         // End the file
524         endElement(METS,"file");
525 	}

561      * The METS defined types of Metadata, if a format is not listed here 
562      * then it should use the string "OTHER" and provide additional 
563      * attributes describing the metadata type 
564      */
565     public static String[] METS_DEFINED_TYPES = 
566     	{"MARC","MODS","EAD","DC","NISOIMG","LC-AV","VRA","TEIHDR","DDI","FGDC","PREMIS"/*,"OTHER"*/};

568     /**
569      * Determine if the provided metadata type is a stardard METS
570      * defined type. If it is not, use the other string.
571      * 
572      * @param metadataType type name
573      * @return True if METS defined
574      */
575     public boolean isDefinedMETStype(String metadataType)
576     {
577        for (String definedType : METS_DEFINED_TYPES)
578        {
579            if (definedType.equals(metadataType))
580                return true;
581        }
582        return false;
583     }

659     /**
660      * Build the SAX attributes object based upon Java's String map. This
661      * convenience method will build, or add to an existing attributes object,
662      * the attributes detailed in the AttributeMap.
663      * 
664      * @param namespaces
665      *            SAX Helper class to keep track of namespaces able to determine
666      *            the correct prefix for a given namespace URI.
667      * @param attributes
668      *            An existing SAX AttributesImpl object to add attributes too.
669      *            If the value is null then a new attributes object will be
670      *            created to house the attributes.
671      * @param attributeMap
672      *            A map of attributes and values.
673      * @return
674      */
675     private AttributesImpl map2sax(Namespace elementNamespace, AttributeMap ... attributeMaps)
676     {
677 
678         AttributesImpl attributes = new AttributesImpl();
679         for (AttributeMap attributeMap : attributeMaps)
680         {
681             boolean diffrentNamespaces = false;
682             Namespace attributeNamespace = attributeMap.getNamespace();
683             if (attributeNamespace != null)
684             {
685             	if (!(attributeNamespace.URI.equals(elementNamespace.URI)))
686             	{
687             		diffrentNamespaces = true;
688             	}
689             }
690             
691             // copy each one over.
692             for (String name : attributeMap.keySet())
693             {
694                 String value = attributeMap.get(name);
695                 if (value == null)
696                     continue;
697 
698                 if (diffrentNamespaces)
699                 	attributes.addAttribute(attributeNamespace.URI, name, 
700                 			qName(attributeNamespace, name), "CDATA", value);
701                 else
702                     attributes.addAttribute("", name, name, "CDATA", value);
703                 
704             }
705         }
706         return attributes;
707     }
