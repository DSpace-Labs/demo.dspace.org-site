76  /**
77   * Display a form that allows the user to edit their profile.
78   * There are two cases in which this can be used: 1) when an
79   * existing user is attempting to edit their own profile, and
80   * 2) when a new user is registering for the first time.
81   *
82   * There are several parameters this transformer accepts:
83   *
84   * email - The email address of the user registering for the first time.
85   *
86   * registering - A boolean value to indicate whether the user is registering for the first time.
87   *
88   * retryInformation - A boolean value to indicate whether there was an error with the user's profile.
89   *
90   * retryPassword - A boolean value to indicate whether there was an error with the user's password.
91   *
92   * allowSetPassword - A boolean value to indicate whether the user is allowed to set their own password.
93   *
94   * @author Scott Phillips
95   */
96  public class EditProfile extends AbstractDSpaceTransformer

208     public void setup(SourceResolver resolver, Map objectModel, String src,
209             Parameters parameters) throws ProcessingException, SAXException,
210             IOException
211     {
212         super.setup(resolver,objectModel,src,parameters);
213         
214         this.email = parameters.getParameter("email","unknown");
215         this.registering = parameters.getParameterAsBoolean("registering",false);
216         this.allowSetPassword = parameters.getParameterAsBoolean("allowSetPassword",false);
217         
218         String errors = parameters.getParameter("errors","");
219         if (errors.length() > 0)
220             this.errors = Arrays.asList(errors.split(","));
221         else
222             this.errors = new ArrayList<String>();
223         
224         // Ensure that the email variable is set.
225         if (eperson != null)
226             this.email = eperson.getEmail();
227     }

230     {

245    public void addBody(Body body) throws WingException, SQLException
246    {
247        // Log that we are viewing a profile
248        log.info(LogManager.getHeader(context, "view_profile", ""));
249 
250        Request request = ObjectModelHelper.getRequest(objectModel);
251        
252        String defaultFirstName="",defaultLastName="",defaultPhone="";
253        String defaultLanguage=null;
254        if (request.getParameter("submit") != null)
255        {
256            defaultFirstName = request.getParameter("first_name");
257            defaultLastName = request.getParameter("last_name");
258            defaultPhone = request.getParameter("phone");
259            defaultLanguage = request.getParameter("language");
260        }
261        else if (eperson != null)
262        {
263             defaultFirstName = eperson.getFirstName();
264             defaultLastName = eperson.getLastName();
265             defaultPhone = eperson.getMetadata("phone");
266             defaultLanguage = eperson.getLanguage();
267        }
268        
269        String action = contextPath;
270        if (registering)
271            action += "/register";
272        else
273            action += "/profile";
274        
275        
276        
277        
278        Division profile = body.addInteractiveDivision("information",
279                action,Division.METHOD_POST,"primary");
280        
281        if (registering)
282            profile.setHead(T_head_create);
283        else
284            profile.setHead(T_head_update);
285        
286        // Add the progress list if we are registering a new user
287        if (registering)
288            EPersonUtils.registrationProgressList(profile,2);
289        
290        
291        
292        
293        
294        List form = profile.addList("form",List.TYPE_FORM);
295        
296        List identity = form.addList("identity",List.TYPE_FORM);
297        identity.setHead(T_head_identify);
298        
299        // Email
300        identity.addLabel(T_email_address);
301        identity.addItem(email);
302        
303        // First name
304        Text firstName = identity.addItem().addText("first_name");
305        firstName.setRequired();
306        firstName.setLabel(T_first_name);
307        firstName.setValue(defaultFirstName);
308        if (errors.contains("first_name"))
309        {
310            firstName.addError(T_error_required);
311        }
312        if (!registering && !ConfigurationManager.getBooleanProperty("xmlui.user.editmetadata", true))
313            firstName.setDisabled();
314        
315        // Last name
316        Text lastName = identity.addItem().addText("last_name");
317        lastName.setRequired();
318        lastName.setLabel(T_last_name);
319        lastName.setValue(defaultLastName);
320        if (errors.contains("last_name"))
321        {
322            lastName.addError(T_error_required);
323        }
324        if (!registering &&!ConfigurationManager.getBooleanProperty("xmlui.user.editmetadata", true))
325            lastName.setDisabled();
326        
327        // Phone
328        Text phone = identity.addItem().addText("phone");
329        phone.setLabel(T_telephone);
330        phone.setValue(defaultPhone);
331        if (errors.contains("phone"))
332        {
333            phone.addError(T_error_required);
334        }
335        if (!registering && !ConfigurationManager.getBooleanProperty("xmlui.user.editmetadata", true))
336            phone.setDisabled();
337         
338        // Language
339        Select lang = identity.addItem().addSelect("language");
340        lang.setLabel(T_language);
341        if (supportedLocales.length > 0)
342        {
343            for (Locale lc : supportedLocales)
344            {
345                lang.addOption(lc.toString(), lc.getDisplayName());
346            }
347        }
348        else
349        {
350            lang.addOption(I18nUtil.DEFAULTLOCALE.toString(), I18nUtil.DEFAULTLOCALE.getDisplayName());
351        }
352        lang.setOptionSelected((defaultLanguage == null || defaultLanguage.equals("")) ?
353                               I18nUtil.DEFAULTLOCALE.toString() : defaultLanguage);
354        if (!registering && !ConfigurationManager.getBooleanProperty("xmlui.user.editmetadata", true))
355            lang.setDisabled();
356 
357        // Subscriptions
358        if (!registering)
359        {
360            List subscribe = form.addList("subscriptions",List.TYPE_FORM);
361            subscribe.setHead(T_subscriptions);
362            
363            subscribe.addItem(T_subscriptions_help);
364            
365            Collection[] currentList = Subscribe.getSubscriptions(context, context.getCurrentUser());
366            Collection[] possibleList = Collection.findAll(context);
367            
368            Select subscriptions = subscribe.addItem().addSelect("subscriptions");
369            subscriptions.setLabel(T_email_subscriptions);
370            subscriptions.setHelp("");
371            subscriptions.enableAddOperation();
372            subscriptions.enableDeleteOperation();
373            
374            subscriptions.addOption(-1,T_select_collection);
375            for (Collection possible : possibleList)
376            {
377                    String name = possible.getMetadata("name");
378                    if (name.length() > 50)
379                            name = name.substring(0, 47) + "...";
380                    subscriptions.addOption(possible.getID(), name);
381            }
382                    
383            for (Collection collection: currentList)
384                    subscriptions.addInstance().setOptionSelected(collection.getID());
385        }
386        
387        
388        if (allowSetPassword)
389        {
390            List security = form.addList("security",List.TYPE_FORM);
391            security.setHead(T_head_security);
392            
393            if (registering)
394            {
395                    security.addItem().addContent(T_create_password_instructions);
396            }
397            else
398            {
399                    security.addItem().addContent(T_update_password_instructions);
400            }
401            
402            
403            Field password = security.addItem().addPassword("password");
404            password.setLabel(T_password);
405            if (registering)
406                    password.setRequired();
407            if (errors.contains("password"))
408            {
409                password.addError(T_error_invalid_password);
410            }
411            
412            Field passwordConfirm = security.addItem().addPassword("password_confirm");
413            passwordConfirm.setLabel(T_confirm_password);
414            if (registering)
415                    passwordConfirm.setRequired();
416            if (errors.contains("password_confirm"))
417            {
418                passwordConfirm.addError(T_error_unconfirmed_password);
419            }
420        }
421        
422        Button submit = form.addItem().addButton("submit");
423        if (registering)
424            submit.setValue(T_submit_update);
425        else
426            submit.setValue(T_submit_create);
427        
428        profile.addHidden("eperson-continue").setValue(knot.getId());
429        
430        
431        
432        if (!registering)
433        {
434                 // Add a list of groups that this user is apart of.
435                         Group[] memberships = Group.allMemberGroups(context, context.getCurrentUser());
436                 
437                 
438                         // Not a member of any groups then don't do anything.
439                         if (!(memberships.length > 0))
440                                 return;
441                         
442                         List list = profile.addList("memberships");
443                         list.setHead(T_head_auth);
444                         for (Group group: memberships)
445                         {
446                                 list.addItem(group.getName());
447                         }
448        }
449    }

461     /**
462      * get the available Locales for the User Interface as defined in dspace.cfg
463      * property xmlui.supported.locales
464      * returns an array of Locales or null
465      *
466      * @return an array of supported Locales or null
467      */
468     private static Locale[] getSupportedLocales()
469     {
470         String ll = ConfigurationManager.getProperty("xmlui.supported.locales");
471         if (ll != null)
472         {
473             return I18nUtil.parseLocales(ll);
474         }
475         else
476         {
477             Locale result[] = new Locale[1];
478             result[0] =  I18nUtil.DEFAULTLOCALE;
479             return result;
480         }
