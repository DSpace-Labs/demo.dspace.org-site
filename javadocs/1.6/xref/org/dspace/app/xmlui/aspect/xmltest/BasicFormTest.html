445     private String makeURL(boolean help, boolean error)
446 	{
447 		if (help && error)
448 			return "?help&error";
449 		
450 		if (help)
451 			return "?help";
452 		
453 		if (error)
454 			return "?error";
455 		
456 		return "?neither";
457 	}
