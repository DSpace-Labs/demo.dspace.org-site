79   * whishes to expand this they can by setting the num_search_fields to the

81   * results per the page by setting results_per_page
82   * 
83   * FIXME: The list of what fields are search should come from a configurable
84   * place. Possibily the sitemap configuration.

156     /**
157      * Add the body
158      */
159     public void addBody(Body body) throws SAXException, WingException,
160             UIException, SQLException, IOException, AuthorizeException
161     {
162         Request request = ObjectModelHelper.getRequest(objectModel);
163         String numSearchField = request.getParameter("num_search_field");
164         if (numSearchField == null || numSearchField.length() == 0)
165         	numSearchField = "3";
166     	
167         // Build the DRI Body
168         Division search = body.addDivision("advanced-search","primary");
169         search.setHead(T_head);
170         Division query = search.addInteractiveDivision("search-query",
171                 "advanced-search",Division.METHOD_POST,"secondary search");
172         
173         // Use these fields to change the number of search fields, or change the results per page.
174         query.addHidden("num_search_field").setValue(numSearchField);
175         query.addHidden("results_per_page").setValue(getParameterRpp());
176         
177         List queryList = query.addList("search-query",List.TYPE_FORM);
178         
179         if (variableScope())
180         {
181             Select scope = queryList.addItem().addSelect("scope");
182             scope.setLabel(T_search_scope);
183             scope.setHelp(T_search_scope_help);
184             buildScopeList(scope);
185         }
186         
187         Table queryTable = query.addTable("search-query", 4, 3);
188         Row header = queryTable.addRow(Row.ROLE_HEADER);
189         header.addCellContent(T_conjunction);
190         header.addCellContent(T_search_type);
191         header.addCellContent(T_search_for);
192         
193         for (int i = 1; i <= FIELD_DISPLAY_COUNT; i++)
194         {
195             Row row = queryTable.addRow(Row.ROLE_DATA);
196             buildConjunctionField(i, row.addCell());
197             buildTypeField(i, row.addCell());
198             buildQueryField(i, row.addCell());
199         }
200 
201         for (SearchField field : fields)
202         {
203         	// Skip over all the fields we've displayed.
204         	int i = field.getIndex();
205         	if (i <= FIELD_DISPLAY_COUNT)
206         		continue;
207         	
208         	query.addHidden("conjunction"+i).setValue(field.getConjunction());
209         	query.addHidden("field"+i).setValue(field.getField());
210         	query.addHidden("query"+i).setValue(field.getQuery());
211         }
212 
213         buildSearchControls(query);
214         query.addPara(null, "button-list").addButton("submit").setValue(T_go);
215         
216         // Add the result division
217         buildSearchResultsDivision(search);
218 
219     }

221     /**
222      * Build a conjunction field in the given for the given cell. A 
223      * conjunction consists of logical the operators AND, OR, NOT.
224      *
225      * @param row The current row
226      * @param cell The current cell
227      */
228     private void buildConjunctionField(int row, Cell cell) throws WingException
229     {
230         // No conjunction for the first row.
231         if (row == 1)
232             return;
233 
234         Request request = ObjectModelHelper.getRequest(objectModel);
235         String current = request.getParameter("conjunction" + row);
236 
237         // default to AND if nothing specified.
238         if (current == null || current.length() == 0)
239             current = "AND";
240         
241         Select select = cell.addSelect("conjunction" + row);
242 
243         select.addOption("AND".equals(current), "AND").addContent(T_and);
244         select.addOption("OR".equals(current), "OR").addContent(T_or);
245         select.addOption("NOT".equals(current), "NOT").addContent(T_not);
246     }

248     /**
249      * Build a list of all the indexable fields in the given cell.
250      * 
251      * FIXME: This needs to use the dspace.cfg data
252      * 
253      * @param row The current row
254      * @param cell The current cell
255      */
256     private void buildTypeField(int row, Cell cell) throws WingException
257     {
258         Request request = ObjectModelHelper.getRequest(objectModel);
259         String current = request.getParameter("field" + row);
260         
261         Select select = cell.addSelect("field" + row);
262 
263         Map<String, Message> searchTypes = new HashMap<String, Message>();
264         
265         int i = 1;
266         String sindex = ConfigurationManager.getProperty("search.index." + i);
267         while(sindex != null)
268         {
269             String field = sindex.split(":")[0];               
270             searchTypes.put(field, message("xmlui.ArtifactBrowser.AdvancedSearch.type_" + field));
271             
272             sindex = ConfigurationManager.getProperty("search.index." + ++i);
273         }
274             
275         
276         
277         // Special case ANY
278         select.addOption((current == null), "ANY").addContent(
279                 message("xmlui.ArtifactBrowser.AdvancedSearch.type_ANY"));
280 
281         for (String key : searchTypes.keySet())
282         {
283             select.addOption(key.equals(current), key).addContent(
284                     searchTypes.get(key));
285         }
286     }

316     /**
317      * Generate a URL for this search page which includes all the 
318      * search parameters along with the added parameters.
319      * 
320      * @param parameters URL parameters to be included in the generated url.
321      */
322     protected String generateURL(Map<String, String> parameters)
323             throws UIException
324     {
325         Request request = ObjectModelHelper.getRequest(objectModel);
326         
327         String numSearchField = request.getParameter("num_search_field");
328         if (numSearchField != null)
329         	parameters.put("num_search_field", numSearchField);
330 
331         String resultsPerPage = request.getParameter("results_per_page");
332         if (resultsPerPage != null)
333         	parameters.put("results_per_page", resultsPerPage);
334         
335         String scope = request.getParameter("scope");
336         if (scope != null)
337         	parameters.put("scope", scope);
338         
339         for (SearchField searchField : getSearchFields(request))
340         {
341         	int index = searchField.getIndex();
342         	String field = searchField.getField();
343         	String query = searchField.getQuery();
344         	String conjunction = searchField.getConjunction();
345         	
346             parameters.put("conjunction" + index, conjunction);
347             parameters.put("field" + index, field);
348             parameters.put("query" + index, query);
349         }
350         
351         if (parameters.get("page") == null)
352         	parameters.put("page", String.valueOf(getParameterPage()));
353         
354         if (parameters.get("rpp") == null)
355         	parameters.put("rpp", String.valueOf(getParameterRpp()));
356         
357         if (parameters.get("sort_by") == null)
358         	parameters.put("sort_by", String.valueOf(getParameterSortBy()));
359         
360         if (parameters.get("order") == null)
361         	parameters.put("order",getParameterOrder());
362         
363         if (parameters.get("etal") == null)
364         	parameters.put("etal",String.valueOf(getParameterEtAl()));
365         
366         return super.generateURL("advanced-search", parameters);
367     }

381     /**
382      * Given a list of search fields buld a lucene search query string.
383      * 
384      * @param fields The search fields
385      * @return A string
386      */
387     private String buildQuery(ArrayList<SearchField> fields)
388     {
389     	Perl5Util util = new Perl5Util();
390     	
391     	String query = "";
392     	
393     	// Loop through the fields building the search query as we go.
394     	for (SearchField field : fields)
395     	{	
396     		// if the field is empty, then skip it and try a later one.
397     		if (field.getQuery() == null)
398     			continue;
399     		
400     		// Add the conjunction for everything but the first field.
401     		if (fields.indexOf(field) > 0)
402     			query += " " + field.getConjunction() + " ";
403             
404     		// Two cases, one if a specific search field is specified or if 
405     		// ANY is given then just a general search is performed.
406             if ("ANY".equals(field.getField()))
407             {
408             	// No field specified, 
409             	query += "(" + field.getQuery() + ")";
410             }
411             else
412             {   
413             	// Specific search field specified, add the field specific field.
414             	
415             	// Replace singe quote's with double quotes (only if they match)
416             	String subquery = util.substitute("s/\'(.*)\'/\"$1\"/g", field.getQuery());
417             	
418             	// If the field is not quoted ...
419             	if (!util.match("/\".*\"/", subquery))
420                 {
421             		// ... then seperate each word and re-specify the search field.
422                     subquery = util.substitute("s/ / " + field.getField() + ":/g", subquery);
423                 }
424             	
425             	// Put the subquery into the general query
426             	query += "("+field.getField()+":"+subquery+")";
427             }
428     	}
429     	
430     	if (query.length() == 0)
431     		return "";
432     	else
433     		return "("+query+")";
434     }

437     /**
438      * Get a list of search fields from the request object
439      * and parse them into a linear array of fileds. The field's
440      * index is preserved, so if it comes in as index 17 it will 
441      * be outputted as field 17.
442      * 
443      * @param request The http request object
444      * @return Array of search fields
445      * @throws UIException 
446      */
447     public ArrayList<SearchField> getSearchFields(Request request) throws UIException
448 	{
449     	if (this.fields != null)
450     		return this.fields;
451     	
452     	// Get how many fields to search
453 	    int numSearchField;
454 	    try {
455 	    	String numSearchFieldStr = request.getParameter("num_search_field");
456 	    	numSearchField = Integer.valueOf(numSearchFieldStr);
457 	    } 
458 	    catch (NumberFormatException nfe)
459 	    {
460 	    	numSearchField = FIELD_MAX_COUNT;
461 	    }
462 	    	
463     	// Iterate over all the possible fields and add each one to the list of fields.
464 		ArrayList<SearchField> fields = new ArrayList<SearchField>();
465 		for (int i = 1; i <= numSearchField; i++)
466 		{
467 			String field = request.getParameter("field"+i);
468 			String query = URLDecode(request.getParameter("query"+i));
469 			String conjunction = request.getParameter("conjunction"+i);
470 			
471 			if (field != null)
472 			{
473 				field = field.trim();
474 				if (field.length() == 0)
475 					field = null;
476 			}
477 			
478 			
479 			if (query != null)
480 			{
481 				query = query.trim();
482 				if (query.length() == 0)
483 					query = null;
484 			}
485 			
486 			if (conjunction != null)
487 			{
488 				conjunction = conjunction.trim();
489 				if (conjunction.length() == 0)
490 					conjunction = null;
491 			}
492 			
493 			if (field == null)
494 				field = "ANY";
495 			if (conjunction == null)
496 				conjunction = "AND";
497 			
498 			if (query != null)
499 				fields.add(new SearchField(i,field,query,conjunction));
500 		}
501 		
502 		this.fields = fields;
503 		
504 		return this.fields;
505 	}

507     /**
508      * A private record keeping class to relate the various 
509      * components of a search field together.
510      */
511     private static class SearchField {
512     	
513     	/** What index the search field is, typicaly there are just three - but the theme may exand this number */
514     	private int index;
515     	
516     	/** The field to search, ANY if none specified */
517     	private String field;
518     	
519     	/** The query string to search for */
520     	private String query;
521     	
522     	/** the conjunction: either "AND" or "OR" */
523     	private String conjuction;
524     	
525     	public SearchField(int index, String field, String query, String conjunction)
526     	{
527     		this.index = index;
528     		this.field = field;
529     		this.query = query;
530     		this.conjuction = conjunction;
531     	}
532     	
533     	public int    getIndex() { return this.index;}
534     	public String getField() { return this.field;}
535     	public String getQuery() { return this.query;}
536     	public String getConjunction() { return this.conjuction;} 
537     }
