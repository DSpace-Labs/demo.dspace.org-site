61  /**
62   * This is a class to test the capabilities of including HTML inside
63   * a DRI document. This can be usefull for resources out side of a
64   * developers control such as text files on disk or user supplied data.
65   * 
66   * This class is not internationalized because it is never intended
67   * to be used in production. It is merely a tool to aid developers of
68   * aspects and themes.
69   * 
70   * @author Scott Phillips
71   */
72  public class HTMLTest extends AbstractDSpaceTransformer
73  {
74  
75  	// The default string to include in test, may be overridden by the user.
76  	private static final String DEFAULT_HTML_STRING = "<p>This is a test of manakin's ability to render HTML fragments.</p>\n\n<p>Only a few tags are allowed such as: <b>bold</b>, <i>italic</i>, <u>underline</u>, and <a href=\"http://di.tamu.edu/\">link</a>.</p>\n\n<h2>This is a heading</h2>\n\nInvalid tags are treated as plain text: <invalid attribute=\"a\">this is invalid</invalid>\n\nAlso line breaks may be treated as a paragraphs when that action is specified.";
77      
78  	public void addPageMeta(PageMeta pageMeta) throws SAXException,
79              WingException, UIException, SQLException, IOException,
80              AuthorizeException
81      {
82          pageMeta.addMetadata("title").addContent("HTML Test");
83         
84          pageMeta.addTrailLink(contextPath + "/","DSpace Home");
85          pageMeta.addTrail().addContent("HTML Test");
86      }
87  
88    
89      public void addBody(Body body) throws SAXException, WingException,
90              UIException, SQLException, IOException, AuthorizeException
91      {
92      	Request request = ObjectModelHelper.getRequest(objectModel);
93      	String fragment = request.getParameter("fragment");
94      	String[] options = request.getParameterValues("options");
95      	
96      	// If none present set the default HTML string
97      	if (fragment == null || fragment.length() <= 0)
98      		fragment = DEFAULT_HTML_STRING;
99      	
100     	boolean blankLines = false;
101     	if (options != null && options[0].equals("blankLines"))
102     		blankLines = true;
103     	
104         Division div = body.addInteractiveDivision("html-test", "", Division.METHOD_GET, "primary");
105      
106         div.setHead("HTML Test");
107         
108         div.addPara("This page tests Manakin's ability to handle HTML fragments, this ability is normally used to handle user-inputed text. There are two reasons for using this ability of including user supplied HTML fragments 1) it doesn't break the abstraction between themes and aspects, 2) it provides a safety mechanism preventing security vulnerabilities such as cross site scripting.");
109         
110         List form = div.addList("html-test",List.TYPE_FORM);
111         TextArea fragmentField = form.addItem().addTextArea("fragment");
112         fragmentField.setLabel("Fragment");
113         fragmentField.setHelp("Enter free formed text, you may use <p>,<a>,<b>,<i>, or <img> tags.");
114         fragmentField.setSize(15, 50);
115         fragmentField.setValue(fragment);
116         
117         CheckBox optionsField = form.addItem().addCheckBox("options");
118         optionsField.setLabel("Options");
119         optionsField.addOption("blankLines", "Treat blank lines as paragraph breaks.");
120         if (blankLines)
121         	optionsField.setOptionSelected("blankLines");
122         
123         
124         Button submit = form.addItem().addButton("submit");
125         submit.setValue("Test HTML Rendering");
126         
127         Division test = div.addDivision("html-test-sample");
128         test.setHead("Rendered Sample");
129         
130         test.addSimpleHTMLFragment(blankLines, fragment);
131         
132     }
