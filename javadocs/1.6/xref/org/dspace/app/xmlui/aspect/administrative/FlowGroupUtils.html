102 	{
103 		// New group, just return an empty list
104 		if (groupID < 0)
105 			return new String[0];
106 		
107 		Group group = Group.find(context,groupID);
108 		
109 		if (group == null)
110 			return new String[0];
111 		
112 		EPerson[] epeople = group.getMembers();
113 		
114 		String[] epeopleIDs = new String[epeople.length];
115 		for (int i=0; i < epeople.length; i++)

118 		return epeopleIDs;

178 	/**
179 	 * Save the group. If the name has been changed then it will be updated, if any 
180 	 * members have been added or removed then they are updated. 
181 	 * 
182 	 * If the groupID is -1 then a new group is created.
183 	 * 
184 	 * @param context The current dspace context
185 	 * @param groupID The group id, or -1 for a new group.
186 	 * @param newName The group's new name.
187 	 * @param newEPeopleIDsArray All epeople members
188 	 * @param newGroupIDsArray All group members.
189 	 * @return A result
190 	 */
191 	public static FlowResult processSaveGroup(Context context, int groupID, String newName, String[] newEPeopleIDsArray, String[] newGroupIDsArray) throws SQLException, AuthorizeException, UIException
192 	{
193 		FlowResult result = new FlowResult();
194 		
195 		// Decode the name incase it uses non-ascii characters.
196 		try
197         {
198             newName = URLDecoder.decode(newName, Constants.DEFAULT_ENCODING);
199         }
200         catch (UnsupportedEncodingException uee)
201         {
202             throw new UIException(uee);
203         }
204 		
205 		Group group = null;
206 		if (groupID == -1)
207 		{
208 			// First check if the name is blank.
209 			if (newName == null || newName.length() == 0)
210 			{
211 				// Group's can not have blank names.
212 				result.setContinue(false);
213 				result.addError("group_name");
214 				result.setOutcome(false);
215 				result.setMessage(new Message("default","The group name may not be blank."));
216 				
217 				return result;
218 			}
219 			
220 			// Create a new group, check if the newName is allready in use.
221 			Group potentialDuplicate = Group.findByName(context,newName);
222 			
223 			if (potentialDuplicate == null)
224 	    	{
225 				// All good, create the new group.
226 				group = Group.create(context);
227 				group.setName(newName);
228 	    	}
229 			else
230 			{
231 				// The name is allready in use, return in error.
232     			result.setContinue(false);
233     			result.addError("group_name");
234     			result.addError("group_name_duplicate");
235     			result.setOutcome(false);
236     			result.setMessage(new Message("default","The group name is allready in use"));
237     			
238     			return result;
239 			}
240 		}
241 		else
242 		{
243 			group = Group.find(context,groupID);
244 			String name = group.getName();
245 			
246 			// Only update the name if there has been a change.
247 			if (newName != null && newName.length() > 0 && !name.equals(newName))
248 			{
249 				// The group name is to be updated, check if the newName is allready in use.
250 				Group potentialDuplicate = Group.findByName(context,newName);
251 				
252 				if (potentialDuplicate == null)
253 		    	{
254 					// All good, update the name
255 					group.setName(newName);
256 		    	}
257 				else
258 				{
259 					// The name is allready in use, return in error.
260 	    			result.setContinue(false);
261 	    			result.addError("group_name");
262 	    			result.addError("group_name_duplicate");
263 	    			result.setOutcome(false);
264 	    			result.setMessage(new Message("default","The group name is allready in use"));
265 	    			
266 	    			return result;
267 				}
268 			}
269 		}
270 		
271 		// Second, Prepare to check members by turning arrays into lists
272 		List<Integer> newEPeopleIDs = new ArrayList<Integer>();
273 		for (String epeopleID : newEPeopleIDsArray)
274 			newEPeopleIDs.add(Integer.valueOf(epeopleID));
275 		List<Integer> newGroupIDs = new ArrayList<Integer>();
276 		for (String _groupID : newGroupIDsArray)
277 			newGroupIDs.add(Integer.valueOf(_groupID));
278 		
279 		
280 		// Third, check if there are any members to remove
281 		// i.e. scan the list on the group against the ids.
282 		for (EPerson epersonMember : group.getMembers())
283 		{
284 			if (!newEPeopleIDs.contains(epersonMember.getID()))
285 			{
286 				// The current eperson is not contained in the new list.
287 				group.removeMember(epersonMember);
288 			}
289 			else
290 			{
291 				// If they are still in the list then remove them
292 				// from the list of people to add.
293 				newEPeopleIDs.remove((Object)epersonMember.getID());
294 			}
295 		}
296 		for (Group groupMember : group.getMemberGroups())
297 		{
298 			if (!newGroupIDs.contains(groupMember.getID()))
299 			{
300 				// The current group is not contained in the new list.
301 				group.removeMember(groupMember);
302 			}
303 			else
304 			{
305 				// If they are still in the list then remove them
306 				// from the list of groups to add.
307 				newGroupIDs.remove((Object)group.getID());
308 			}
309 		}
310 		
311 		// Third, check if there are any members to add
312 		// i.e. scan the list of ids against the group.
313 		for (Integer epersonID : newEPeopleIDs)
314 		{
315 			EPerson eperson = EPerson.find(context, epersonID);
316 			
317 			group.addMember(eperson);
318 		}
319 		
320 		for (Integer _groupID : newGroupIDs)
321 		{
322 			Group _group = Group.find(context, _groupID);
323 			
324 			group.addMember(_group);
325 		}
326 		
327 		// Last, create the result flow
328 		group.update();
329 		context.commit();
330 		
331 		// Let's record our group id incase we created a new one.
332 		result.setParameter("groupID", group.getID());
333 		result.setContinue(true);
334 		result.setOutcome(true);
335 		result.setMessage(T_edit_group_success_notice);
336 		
337 		return result;
338 	}

340 	/**
341 	 * Remove the specified groups. It is assumed that the user has allready confirm this selection.
342 	 * 
343 	 * @param context The current DSpace context
344 	 * @param groupIDs A list of groups to be removed.
345 	 * @return A results object.
346 	 */
347 	public static FlowResult processDeleteGroups(Context context, String[] groupIDs) throws SQLException, AuthorizeException, IOException
348 	{
349 		FlowResult result = new FlowResult();
350 		result.setContinue(true);
351 		
352     	for (String id : groupIDs) 
353     	{
354     		Group groupDeleted = Group.find(context, Integer.valueOf(id));
355     		
356     		// If this group is related to a collection, then un-link it.
357     		int collectionId = getCollectionId(groupDeleted.getName());
358     		Role role = getCollectionRole(groupDeleted.getName());
359     		if (collectionId != -1 && role != Role.none)
360     		{
361 	    		Collection collection = Collection.find(context, collectionId);
362 	    		
363 	    		if (collection != null)
364 	    		{
365 		    		if (role == Role.Administrators)
366 		    		{
367 		    			collection.removeAdministrators();
368 		    			collection.update();
369 		    		} 
370 		    		else if (role == Role.Submitters)
371 		    		{
372 		    			collection.removeSubmitters();
373 		    			collection.update();
374 		    		}
375 		    		else if (role == Role.WorkflowStep1)
376 		    		{
377 		    			collection.setWorkflowGroup(1, null);
378 		    			collection.update();
379 		    		}
380 		    		else if (role == Role.WorkflowStep2)
381 		    		{
382 		    			collection.setWorkflowGroup(2, null);
383 		    			collection.update();
384 		    		}
385 		    		else if (role == Role.WorkflowStep3)
386 		    		{
387 		    			collection.setWorkflowGroup(3, null);
388 		    			collection.update();
389 		    		}
390 		    		else if (role == Role.DefaultRead)
391 		    		{
392 		    			// Nothing special needs to happen.
393 		    		}
394 	    		}
395     		}
396 
397 			groupDeleted.delete();
398 	    }
399     	
400     	result.setOutcome(true);
401 		result.setMessage(T_delete_group_success_notice);
402     	
403     	return result;
404 	}

423 	 * Extracts the collection id that may be immbedded in the given group name.
424 	 * 
425 	 * @param groupName - the name of a group (ie group.getName())
426 	 * @return the integer collection id or -1 if the group is not that of a collection
427 	 */
428 	public static int getCollectionId(String groupName)
429 	{
430 		if (groupName != null && groupName.startsWith(COLLECTION_PREFIX))
431 		{
432 			for (String suffix : COLLECTION_SUFFIXES)
433 			{
434 				if (groupName.endsWith(suffix))
435 				{
436 					String idString = groupName.substring(COLLECTION_PREFIX.length());
437 					idString = idString.substring(0, idString.length() - suffix.length());
438 
439 					int collectionID = -1;
440 					try {
441 						collectionID = Integer.valueOf(idString); 
442 						
443 						return collectionID;
444 						// All good, we were able to ah 
445 					}
446 					catch (NumberFormatException nfe)
447 					{
448 						// Somethnig went wrong, just ignore the exception and
449 						// continue searching for a collection id
450 					} // try & catch
451 				} // if it ends with a proper suffix.
452 			} // for each possible suffix
453 		} // if it starts with COLLECTION_
454 		
455 		return -1;
456     }

460 	public static Role getCollectionRole(String groupName)
461 	{
462 		if (groupName != null && groupName.startsWith(COLLECTION_PREFIX))
463 		{
464 			for (String suffix : COLLECTION_SUFFIXES)
465 			{
466 				if (groupName.endsWith(suffix))
467 				{
468 					if (COLLECTION_SUFFIXES[0].equals(suffix))
469 						return Role.Submitters;
470 					else if (COLLECTION_SUFFIXES[1].equals(suffix))
471 						return Role.Administrators;
472 					else if (COLLECTION_SUFFIXES[2].equals(suffix))
473 						return Role.WorkflowStep1;
474 					else if (COLLECTION_SUFFIXES[3].equals(suffix))
475 						return Role.WorkflowStep1;
476 					else if (COLLECTION_SUFFIXES[4].equals(suffix))
477 						return Role.WorkflowStep2;
478 					else if (COLLECTION_SUFFIXES[5].equals(suffix))
479 						return Role.WorkflowStep2;
480 					else if (COLLECTION_SUFFIXES[6].equals(suffix))
481 						return Role.WorkflowStep3;
482 					else if (COLLECTION_SUFFIXES[7].equals(suffix))
483 						return Role.WorkflowStep3;
484 					else if (COLLECTION_SUFFIXES[8].equals(suffix))
485 						return Role.DefaultRead;
486 					
487 				} // if it ends with a proper suffix.
488 			} // for each possible suffix
489 		} // if it starts with COLLECTION_
490 		
491 		return Role.none;
