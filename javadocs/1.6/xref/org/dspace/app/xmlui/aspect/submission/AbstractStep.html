188 	/**
189 	 * Grab all the page's parameters from the sitemap. This includes 
190 	 * workspaceID, step, and a list of errored fields.
191 	 * 
192 	 * If the implementer set any required parameters then insure that 
193 	 * they are all present.
194 	 */
195 	public void setup(SourceResolver resolver, Map objectModel, String src, Parameters parameters) 
196 	throws ProcessingException, SAXException, IOException
197 	{ 
198 		super.setup(resolver,objectModel,src,parameters);
199 
200 		try {
201 			this.id = parameters.getParameter("id",null);
202 			this.stepAndPage = Double.valueOf(parameters.getParameter("step","-1"));
203 			this.handle = parameters.getParameter("handle",null);
204 			this.errorFlag = Integer.valueOf(parameters.getParameter("error", String.valueOf(AbstractProcessingStep.STATUS_COMPLETE)));
205 			this.errorFields = getErrorFields(parameters);
206 			
207 			//load in-progress submission
208 			if (this.id != null)
209             {    
210                 this.submissionInfo = FlowUtils.obtainSubmissionInfo(objectModel, this.id);
211 				this.submission = submissionInfo.getSubmissionItem();
212             }
213 			
214 			// Check required error conditions
215 			if (this.requireSubmission && this.submission == null)
216 				throw new ProcessingException("Unable to find submission for id: "+this.id);
217 			
218 			if (this.requireWorkflow && !(submission instanceof WorkflowItem))
219 				throw new ProcessingException("The submission is not a workflow, "+this.id);
220 			
221 			if (this.requireWorkspace && !(submission instanceof WorkspaceItem))
222 				throw new ProcessingException("The submission is not a workspace, "+this.id);
223 			
224 			if (this.requireStep && stepAndPage < 0)
225 				throw new ProcessingException("Step is a required parameter.");
226 			
227 			if (this.requireHandle && handle == null)
228 				throw new ProcessingException("Handle is a required parameter.");
229 			
230 		} 
231 		catch (SQLException sqle) 
232 		{
233 			throw new ProcessingException("Unable to find submission.",sqle);
234 		}
235 	}

239 	 * Base pageMeta that is added to ALL submission stages 
240 	 */
241 	public void addPageMeta(PageMeta pageMeta) throws SAXException,
242 	WingException, UIException, SQLException, IOException,
243 	AuthorizeException
244 	{
245 		if (submission instanceof WorkspaceItem)
246 		{
247 			pageMeta.addMetadata("title").addContent(T_submission_title);
248 	
249 			Collection collection = submission.getCollection();
250 			
251 	        pageMeta.addTrailLink(contextPath + "/",T_dspace_home);
252 	        HandleUtil.buildHandleTrail(collection,pageMeta,contextPath);
253 	        pageMeta.addTrail().addContent(T_submission_trail);
254 		}
255 		else if (submission instanceof WorkflowItem)
256 		{
257 			pageMeta.addMetadata("title").addContent(T_workflow_title);
258 			
259 			Collection collection = submission.getCollection();
260 			
261 	        pageMeta.addTrailLink(contextPath + "/",T_dspace_home);
262 	        HandleUtil.buildHandleTrail(collection,pageMeta,contextPath);
263 	        pageMeta.addTrail().addContent(T_workflow_trail);
264 		}
265 		else
266 		{
267 			// defaults for pages that don't have a workspace item or workflow 
268 			// item such as the submission complete page where the object is in transition.
269 			pageMeta.addMetadata("title").addContent(T_default_title);
270 			
271 			pageMeta.addTrailLink(contextPath + "/",T_dspace_home);
272 	        pageMeta.addTrail().addContent(T_default_trail);
273 		}

277 	/**
278 	 * Add a submission progress list to the current div for this step. 
279 	 * 
280 	 * @param div The division to add the list to.
281 	 */
282 	public void addSubmissionProgressList(Division div) throws WingException
283 	{
284 		//each entry in progress bar is placed under this "submit-progress" div
285 		List progress = div.addList("submit-progress",List.TYPE_PROGRESS);
286 		
287 		//get Map of progress bar information
288 		//key: entry # (i.e. step & page), 
289 		//value: entry name key (i.e. display name)
290 		Map progBarInfo = this.submissionInfo.getProgressBarInfo();
291 		
292 		Iterator i = progBarInfo.keySet().iterator();
293 		
294 		//add each entry to progress bar
295 		while(i.hasNext())
296 		{
297 			String entryNum = (String) i.next();
298 			
299 			//Since we are using XML-UI, we need to prepend the heading key with "xmlui.Submission."
300 			String entryNameKey = "xmlui.Submission." + (String) progBarInfo.get(entryNum);
301 			
302 			//the value of entryNum is current step & page 
303 			//(e.g. 1.2 is page 2 of step 1) 
304 			double currentStepAndPage = Double.valueOf(entryNum).doubleValue();
305 			
306             //add a button to progress bar for this step & page
307             addJumpButton(progress, message(entryNameKey), currentStepAndPage);
308 		}
309 		
310 	}

312     /**
313      * Adds a single "jump-to" button, which when clicked will
314      * jump the user directly to a particular step within the
315      * submission process.
316      * <P>
317      * This method is used by the addSubmissionProgressList() method
318      * to create the progress bar at the top of the submission process.
319      * <P>
320      * This method is also used by the ReviewStep to add buttons to
321      * jump back to a particular step in the submission process
322      *
323      * @param list
324      *          The List which the button will be added to
325      * @param buttonText
326      *          The text to be displayed on the button
327      * @param stepAndPage
328      *          The step and page (a double of the form 'step.page', e.g. 1.2)
329      *          which this button will jump back to
330      */
331     public void addJumpButton(List list, Message buttonText, double stepAndPage)
332         throws WingException
333     {
334         //Only add the button if we have button text and a valid step & page!
335         if(buttonText!=null && stepAndPage>0.0)
336         {    
337             //add a Jump To button for this section
338             Button jumpButton = list.addItem("step_" + stepAndPage, renderJumpButton(stepAndPage))
339                                     .addButton(AbstractProcessingStep.PROGRESS_BAR_PREFIX + stepAndPage);
340             jumpButton.setValue(buttonText);
341         }
342     }

344     /**
345      * Adds the "<-Previous", "Save/Cancel" and "Next->" buttons 
346      * to a given form.  This method ensures that the same
347      * default control/paging buttons appear on each submission page.
348      * <P>
349      * Note: A given step may define its own buttons as necessary,
350      * and not call this method (since it must be explicitly envoked by
351      * the step's addBody() method)
352      *
353      * @param controls
354      *          The List which will contain all control buttons
355      */
356     public void addControlButtons(List controls)
357         throws WingException
358     {
359         Item actions = controls.addItem();
360         
361         //only have "<-Previous" button if not first step
362         if(!isFirstStep())
363             actions.addButton(AbstractProcessingStep.PREVIOUS_BUTTON).setValue(T_previous);
364         
365         //always show "Save/Cancel"
366         actions.addButton(AbstractProcessingStep.CANCEL_BUTTON).setValue(T_save);
367         
368         //If last step, show "Complete Submission"
369         if(isLastStep())
370             actions.addButton(AbstractProcessingStep.NEXT_BUTTON).setValue(T_complete);
371         else //otherwise, show "Next->"
372             actions.addButton(AbstractProcessingStep.NEXT_BUTTON).setValue(T_next);
373     }

397     /**
398      * Return whether this is the first step in
399      * the submission process (the first step is
400      * currently considered the first that appears
401      * in the progress bar)
402      *
403      * @return true if first step
404      */
405     public boolean isFirstStep()
406     {
407         Set submissionPagesSet = submissionInfo.getProgressBarInfo().keySet();
408         String[] submissionPages = (String[]) submissionPagesSet.toArray(new String[submissionPagesSet.size()]);
409         
410         double firstStepAndPage = Double.valueOf(submissionPages[0]).doubleValue();
411         
412         if((FlowUtils.getStep(firstStepAndPage)==getStep())
413                 && (FlowUtils.getPage(firstStepAndPage)==getPage()))
414         {
415             return true;
416         }
417         else
418             return false;
419     }

422     /**
423      * Return whether this is the last step in
424      * the submission process (the last step is
425      * currently considered the one directly *before*
426      * the Complete step in the progress bar)
427      *
428      * @return true if last step
429      */
430     public boolean isLastStep()
431     {
432         boolean inWorkflow = this.submissionInfo.isInWorkflow();
433         
434         Set submissionPagesSet = submissionInfo.getProgressBarInfo().keySet();
435         String[] submissionPages = (String[]) submissionPagesSet.toArray(new String[submissionPagesSet.size()]);
436         
437         double lastStepAndPage;
438         
439         if(!inWorkflow)
440         {
441             // If not in Workflow,
442             // Last step is considered the one *before* the Complete Step
443             lastStepAndPage = Double.valueOf(submissionPages[submissionPages.length-2]).doubleValue();
444         }
445         else
446         {   
447             lastStepAndPage = Double.valueOf(submissionPages[submissionPages.length-1]).doubleValue();
448         }
449         
450         if((FlowUtils.getStep(lastStepAndPage)==getStep())
451                 && (FlowUtils.getPage(lastStepAndPage)==getPage()))
452         {
453             return true;
454         }
455         else
456             return false;
457     }

459     /**
460      * Find the maximum step and page that the user has 
461      * reached in the submission processes. 
462      * If this submission is a workflow then return max-int.
463      * 
464      * @param context The current DSpace content
465      * @param id The unique ID of the current workflow/workspace
466      */
467     public double getMaxStepAndPageReached() throws SQLException {
468         
469         if (this.submission instanceof WorkspaceItem)
470         {
471             WorkspaceItem workspaceItem = (WorkspaceItem) submission;
472             
473             int step = workspaceItem.getStageReached();
474             if(step<0) 
475                 step=0;
476             
477             int page = workspaceItem.getPageReached();
478             if (page < 0)
479                 page = 0;
480             
481             return Double.valueOf(step + "." + page).doubleValue();
482         }
483         
484         // This is a workflow, return infinity.
485         return Double.MAX_VALUE;
486     }

489 	 * Retreive error fields from the list of parameters
490 	 * and return a List of all fields which had errors
491 	 *
492 	 * @return java.util.List of field names with errors
493 	 */
494 	public java.util.List<String> getErrorFields(Parameters parameters)
495 	{
496 		java.util.List<String> fields = new ArrayList<String>();
497 		
498 		String errors = parameters.getParameter("error_fields","");
499 		
500 		if (errors!=null && errors.length() > 0)
501 		{	
502 			if(errors.indexOf(",") > 0)
503 				fields = Arrays.asList(errors.split(","));
504 			else//only one error field
505 				fields.add(errors);
506 		}
507 		
508 		return fields;
509 	}

511 	/**
512 	 * A simple method to determine how the Jump to button
513      * for a given step and page should be rendered.
514      * <P>
515      * If the given step and page corresponds to the current
516      * step and page, render it with "current" style.
517      * <P>
518      * If the given step and page is greater than the max,
519      * render it with "disabled" style.
520 	 * 
521 	 * @param givenStepAndPage 
522      *        This given step & page (e.g. "1.2")
523 	 * @return
524      *        render style for this button
525 	 */
526 	private String renderJumpButton(double givenStepAndPage)
527 	{
528         try
529         {
530             if (Double.compare(givenStepAndPage, this.stepAndPage)==0)
531     			return "current";
532             else if (Double.compare(givenStepAndPage, getMaxStepAndPageReached())>0)
533                 return "disabled";
534             else
535     			return null;
536         }
537         catch(Exception e)
538         {
539             return null;
540         }  
541 	}
