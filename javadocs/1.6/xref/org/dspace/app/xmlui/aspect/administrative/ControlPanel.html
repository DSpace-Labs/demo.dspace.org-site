208 	public void addBody(Body body) throws SAXException, WingException,
209 			UIException, SQLException, IOException, AuthorizeException {
210 		
211 		if (!AuthorizeManager.isAdmin(context))
212 			throw new AuthorizeException("You are not authorized to view this page.");
213 		
214 		Request request = ObjectModelHelper.getRequest(objectModel);
215 		OPTIONS option = null;
216 		if (request.getParameter("java") != null)
217 			option = OPTIONS.java;
218 		if (request.getParameter("dspace") != null)
219 			option = OPTIONS.dspace;
220 		if (request.getParameter("alerts") != null)
221 			option = OPTIONS.alerts;
222 		if (request.getParameter("activity") != null)
223 			option = OPTIONS.activity;
224 		if (request.getParameter("harvest") != null)
225 			option = OPTIONS.harvest;
226 		
227 		Division div = body.addInteractiveDivision("control-panel", contextPath+"/admin/panel", Division.METHOD_POST, "primary administrative");
228 		div.setHead(T_head);
229 		
230 		// LIST: options
231 		List options = div.addList("options",List.TYPE_SIMPLE,"horizontal");
232 		
233 		// our options, selected or not....
234 		if (option == OPTIONS.java)
235 			options.addItem().addHighlight("bold").addXref("?java",T_option_java);
236 		else
237 			options.addItemXref("?java",T_option_java);
238 		
239 		if (option == OPTIONS.dspace)
240 			options.addItem().addHighlight("bold").addXref("?dspace",T_option_dspace);
241 		else
242 			options.addItemXref("?dspace",T_option_dspace);
243 		
244 		if (option == OPTIONS.alerts)
245 			options.addItem().addHighlight("bold").addXref("?alerts",T_option_alerts);
246 		else
247 			options.addItemXref("?alerts",T_option_alerts);
248 		
249 		if (option == OPTIONS.harvest)
250 			options.addItem().addHighlight("bold").addXref("?harvest",T_option_harvest);
251 		else
252 			options.addItemXref("?harvest",T_option_harvest);
253 		
254 		String userSortTarget = "?activity";
255 		if (request.getParameter("sortBy") != null)
256 			userSortTarget += "&sortBy="+request.getParameter("sortBy");
257 		if (option == OPTIONS.activity)
258 			options.addItem().addHighlight("bold").addXref(userSortTarget,"Current Activity");
259 		else
260 			options.addItemXref(userSortTarget,"Current Activity");
261 		
262 		
263 		// The main content:
264 		if (option == OPTIONS.java)
265 			addJavaInformation(div);
266 		else if (option == OPTIONS.dspace)
267 			addDSpaceConfiguration(div);
268 		else if (option == OPTIONS.alerts)
269 			addAlerts(div);
270 		else if (option == OPTIONS.activity)
271 			addActivity(div);
272 		else if (option == OPTIONS.harvest)
273 			addHarvest(div);
274 		else
275 		{
276 			div.addPara(T_select_panel);
277 		}
278 		
279 	}

281 	/**
282 	 * Add specific java information including JRE, OS, and runtime memory statistics.
283 	 */
284 	private void addJavaInformation(Division div) throws WingException
285 	{
286 		// Get memory statistics
287 		int processors = Runtime.getRuntime().availableProcessors();
288 		long maxMemory = Runtime.getRuntime().maxMemory();
289 		long totalMemory = Runtime.getRuntime().totalMemory();
290         long freeMemory = Runtime.getRuntime().freeMemory();
291 		long usedMemory = totalMemory-freeMemory;
292         
293 		// Convert bytes into MiB
294 		maxMemory   = maxMemory   / 1024 / 1024;
295 		totalMemory = totalMemory / 1024 / 1024;
296 		usedMemory  = usedMemory  / 1024 / 1024;
297 		freeMemory  = freeMemory  / 1024 / 1024;
298 		
299 		// LIST: Java
300 		List list = div.addList("javaOs");
301 		list.setHead(T_JAVA_HEAD);
302 		list.addLabel(T_JAVA_VERSION);
303         list.addItem(System.getProperty("java.version"));
304         list.addLabel(T_JAVA_VENDOR);
305         list.addItem(System.getProperty("java.vendor"));
306         list.addLabel(T_OS_NAME);
307         list.addItem(System.getProperty("os.name"));
308         list.addLabel(T_OS_ARCH);
309         list.addItem(System.getProperty("os.arch"));
310         list.addLabel(T_OS_VERSION);
311         list.addItem(System.getProperty("os.version"));
312 		
313 		// LIST: memory
314 		List runtime = div.addList("runtime");
315 		runtime.setHead(T_RUNTIME_HEAD);
316 		runtime.addLabel(T_RUNTIME_PROCESSORS);
317 		runtime.addItem(String.valueOf(processors));
318 		runtime.addLabel(T_RUNTIME_MAX);
319 		runtime.addItem(String.valueOf(maxMemory) + " MiB");
320 		runtime.addLabel(T_RUNTIME_TOTAL);
321 		runtime.addItem(String.valueOf(totalMemory) + " MiB");
322 		runtime.addLabel(T_RUNTIME_USED);
323 		runtime.addItem(String.valueOf(usedMemory) + " MiB");
324 		runtime.addLabel(T_RUNTIME_FREE);
325 		runtime.addItem(String.valueOf(freeMemory) + " MiB");	
326 	}

381 	/**
382 	 * Add a section that allows administrators to activate or deactivate system-wide alerts.
383 	 */
384 	private void addAlerts(Division div) throws WingException 
385 	{
386 		// Remember we're in the alerts section
387 		div.addHidden("alerts").setValue("true");
388 		
389 		List form = div.addList("system-wide-alerts",List.TYPE_FORM);
390 		form.setHead(T_alerts_head);
391 		
392 		form.addItem(T_alerts_warning);
393 		
394 		TextArea message = form.addItem().addTextArea("message");
395 		message.setLabel(T_alerts_message_label);
396 		message.setSize(5, 45);
397 		if (SystemwideAlerts.getMessage() == null)
398 			message.setValue(T_alerts_message_default);
399 		else
400 			message.setValue(SystemwideAlerts.getMessage());
401 		
402 		Select countdown = form.addItem().addSelect("countdown");
403 		countdown.setLabel(T_alerts_countdown_label);
404 		
405 		countdown.addOption(0,T_alerts_countdown_none);
406 		countdown.addOption(5,T_alerts_countdown_5);
407 		countdown.addOption(15,T_alerts_countdown_15);
408 		countdown.addOption(30,T_alerts_countdown_30);
409 		countdown.addOption(60,T_alerts_countdown_60);
410 		
411 		// Is there a current count down active?
412 		if (SystemwideAlerts.isAlertActive() && SystemwideAlerts.getCountDownToo() - System.currentTimeMillis() > 0)
413 			countdown.addOption(true,-1,T_alerts_countdown_keep);
414 		else
415 			countdown.setOptionSelected(0);
416 		
417 		Select restrictsessions = form.addItem().addSelect("restrictsessions");
418 		restrictsessions.setLabel(T_alerts_session_label);
419 		restrictsessions.addOption(SystemwideAlerts.STATE_ALL_SESSIONS,T_alerts_session_all_sessions);
420 		restrictsessions.addOption(SystemwideAlerts.STATE_CURRENT_SESSIONS,T_alerts_session_current_sessions);
421 		restrictsessions.addOption(SystemwideAlerts.STATE_ONLY_ADMINISTRATIVE_SESSIONS,T_alerts_session_only_administrative);
422 		restrictsessions.setOptionSelected(SystemwideAlerts.getRestrictSessions());
423 		
424 		form.addItem(T_alerts_session_note);
425 		
426 		
427 		Item actions = form.addItem();
428 		actions.addButton("submit_activate").setValue(T_alerts_submit_activate);
429 		actions.addButton("submit_deactivate").setValue(T_alerts_submit_deactivate);
430 		
431 	}

579 	/**
580 	 * Comparator to sort activity events by their access times.
581 	 */
582 	public static class ActivitySort<E extends CurrentActivityAction.Event> implements Comparator<E>
583 	{
584 		// Sort parameter
585 		private EventSort sortBy;
586 		
587 		public ActivitySort(EventSort sortBy)
588 		{
589 			this.sortBy = sortBy;
590 		}
591 		
592 		/**
593 		 * Compare these two activity events based upon the given sort parameter. In the case of a tie,
594 		 * allways fallback to sorting based upon the timestamp.
595 		 */
596 		public int compare(E a, E b) 
597 		{
598 			// Protect against null events while sorting
599 			if (a != null && b == null)
600 				return 1; // A > B
601 			else if (a == null && b != null)
602 				return -1; // B > A
603 			else if (a == null && b == null)
604 				return 0; // A == B
605 			
606 			// Sort by the given ordering matrix
607 			if (EventSort.URL == sortBy)
608 			{
609 				String aURL = a.getURL();
610 				String bURL = b.getURL();
611 				int cmp = aURL.compareTo(bURL);
612 				if (cmp != 0)
613 					return cmp;
614 			}
615 			else if (EventSort.AGENT == sortBy)
616 			{
617 				String aAgent = a.getDectectedBrowser();
618 				String bAgent = b.getDectectedBrowser();
619 				int cmp = aAgent.compareTo(bAgent);
620 				if (cmp != 0)
621 					return cmp;
622 			}
623 			else if (EventSort.IP == sortBy)
624 			{
625 				String aIP = a.getIP();
626 				String bIP = b.getIP();
627 				int cmp = aIP.compareTo(bIP);
628 				if (cmp != 0)
629 					return cmp;
630 				
631 			}
632 			else if (EventSort.SESSION == sortBy)
633 			{
634 				// Ensure that all sessions with an EPersonID associated are
635 				// ordered to the top. Otherwise fall back to comparing session
636 				// IDs. Unfortunitaly we can not compare eperson names because 
637 				// we do not have access to a context object.
638 				if (a.getEPersonID() > 0  && b.getEPersonID() < 0)
639 					return 1; // A > B
640 				else if (a.getEPersonID() < 0  && b.getEPersonID() > 0)
641 					return -1; // B > A
642 				
643 				String aSession = a.getSessionID();
644 				String bSession = b.getSessionID();
645 				int cmp = aSession.compareTo(bSession);
646 				if (cmp != 0)
647 					return cmp;
648 			}
649 			
650 			// All ways fall back to sorting by time, when events are equal.
651 			if (a.getTimeStamp() > b.getTimeStamp())
652 				return 1;  // A > B
653 			else if (a.getTimeStamp() > b.getTimeStamp())
654 				return -1; // B > A
655 			return 0; // A == B
656 		}
657 	}
