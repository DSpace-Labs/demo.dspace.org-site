89  	/**
90  	 * Resolve the given identifier to an item. The identifier may be either an
91  	 * internal ID or a handle. If an item is found then the result the internal
92  	 * ID of the item will be placed in the result "itemID" parameter.
93  	 * 
94  	 * If the identifier was unable to be resolved to an item then the "identifier"
95  	 * field is placed in error.
96  	 * 
97  	 * @param context The current DSpace context.
98  	 * @param identifier An Internal ID or a handle
99  	 * @return A flow result
100 	 */
101 	public static FlowResult resolveItemIdentifier(Context context, String identifier) throws SQLException
102 	{
103 		FlowResult result = new FlowResult();
104 		result.setContinue(false);
105 		
106 		//		Check whether it's a handle or internal id (by check ing if it has a slash inthe string)
107 		if (identifier.contains("/")) 
108 		{
109 			DSpaceObject dso = HandleManager.resolveToObject(context, identifier);
110 	
111 			if (dso != null && dso.getType() == Constants.ITEM) 
112 			{ 
113 				result.setParameter("itemID", dso.getID());
114 				result.setParameter("type", Constants.ITEM);
115 				result.setContinue(true);
116 				return result;
117 			}
118 		}
119 		else
120 		{
121 		
122 			Item item = null;
123 			try {
124 				item = Item.find(context, Integer.valueOf(identifier));
125 			} catch (NumberFormatException e) {
126 				// ignoring the exception
127 			}
128 
129 			if (item != null) 
130 			{
131 				result.setParameter("itemID", item.getID());
132 				result.setParameter("type", Constants.ITEM);
133 				result.setContinue(true);
134 				return result;
135 			}
136 		}
137 
138 		result.addError("identifier");
139 		return result;
140 	}

143 	 * Process the request parameters to update the item's metadata and remove any selected bitstreams.
144 	 * 
145 	 * Each metadata entry will have three fields "name_X", "value_X", and "language_X" where X is an
146 	 * integer that relates all three of the fields together. The name parameter stores the metadata name 
147 	 * that is used by the entry (i.e schema_element_qualifier). The value and language paramaters are user
148 	 * inputed fields. If the optional parameter "remove_X" is given then the metadata value is removed.
149 	 * 
150 	 * To support AJAX operations on this page an aditional parameter is considered, the "scope". The scope
151 	 * is the set of metadata entries that are being updated during this request. It the metadata name, 
152 	 * schema_element_qualifier, only fields that have this name are considered! If all fields are to be
153 	 * considered then scope should be set to "*". 
154 	 * 
155 	 * When creating an AJAX query include all the name_X, value_X, language_X, and remove_X for the fields
156 	 * in the set, and then set the scope parameter to be the metadata field.
157 	 * 
158 	 * @param context The current DSpace context
159 	 * @param itemID  internal item id
160 	 * @param request the Cocoon request
161 	 * @return A flow result
162 	 */
163 	public static FlowResult processEditItem(Context context, int itemID, Request request) throws SQLException, AuthorizeException, UIException, IOException
164 	{
165 		FlowResult result = new FlowResult();
166 		result.setContinue(false);
167 
168 		Item item = Item.find(context, itemID);
169 		
170 		
171 		// STEP 1:
172 		// Clear all metadata within the scope
173 		// Only metadata values within this scope will be considered. This
174 		// is so ajax request can operate on only a subset of the values.
175 		String scope = request.getParameter("scope");
176 		if ("*".equals(scope))
177 		{
178 			item.clearMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
179 		}
180 		else
181 		{
182 			String[] parts = parseName(scope);
183 			item.clearMetadata(parts[0],parts[1],parts[2],Item.ANY);
184 		}
185 		
186 		// STEP 2:
187 		// First determine all the metadata fields that are within
188 		// the scope parameter
189 		ArrayList<Integer> indexes = new ArrayList<Integer>();
190 		Enumeration parameters = request.getParameterNames();
191 		while(parameters.hasMoreElements())
192 		{
193 
194 			// Only consider the name_ fields
195 			String parameterName = (String) parameters.nextElement();
196 			if (parameterName.startsWith("name_"))
197 			{
198 				// Check if the name is within the scope
199 				String parameterValue = request.getParameter(parameterName);
200 				if ("*".equals(scope) || scope.equals(parameterValue))
201 				{
202 					// Extract the index from the name.
203 					String indexString = parameterName.substring("name_".length());
204 					Integer index = Integer.valueOf(indexString);
205 					indexes.add(index);
206 				}
207 			}
208 		}
209 		
210 		// STEP 3:
211 		// Iterate over all the indexes within the scope and add them back in.
212 		for (Integer index=1; index <= indexes.size(); ++index)
213 		{
214 			String name = request.getParameter("name_"+index);
215 			String value = request.getParameter("value_"+index);
216                         String authority = request.getParameter("value_"+index+"_authority");
217                         String confidence = request.getParameter("value_"+index+"_confidence");
218 			String lang = request.getParameter("language_"+index);
219 			String remove = request.getParameter("remove_"+index);
220 			
221 			// the user selected the remove checkbox.
222 			if (remove != null)
223 				continue;
224 			
225 			// get the field's name broken up
226 			String[] parts = parseName(name);
227 			
228                         // probe for a confidence value
229                         int iconf = Choices.CF_UNSET;
230                         if (confidence != null && confidence.length() > 0)
231                             iconf = Choices.getConfidenceValue(confidence);
232                         // upgrade to a minimum of NOVALUE if there IS an authority key
233                         if (authority != null && authority.length() > 0 && iconf == Choices.CF_UNSET)
234                             iconf = Choices.CF_NOVALUE;
235                         item.addMetadata(parts[0], parts[1], parts[2], lang,
236                                              value, authority, iconf);
237 		}
238 		
239 		item.update();
240 		context.commit();
241 		
242 		result.setContinue(true);
243 		
244 		result.setOutcome(true);
245 		result.setMessage(T_metadata_updated);
246 		
247 		return result;
248 	}

335     /**
336      * Move the specified item to another collection.
337      *
338      * @param context The DSpace context
339      * @param itemID The id of the to-be-moved item.
340      * @param collectionID The id of the destination collection.
341      * @return A result object
342      */
343     public static FlowResult processMoveItem(Context context, int itemID, int collectionID) throws SQLException, AuthorizeException, IOException
344     {
345         FlowResult result = new FlowResult();
346         result.setContinue(false);
347 
348         Item item = Item.find(context, itemID);
349 
350         if(AuthorizeManager.isAdmin(context, item))
351         {
352           //Add a policy giving this user *explicit* admin permissions on the item itself.
353           //This ensures that the user will be able to call item.update() even if he/she
354           // moves it to a Collection that he/she doesn't administer.
355           AuthorizeManager.addPolicy(context, item, Constants.ADMIN, context.getCurrentUser());
356 
357           Collection destination = Collection.find(context, collectionID);
358           if (destination == null)
359           {
360               result.setOutcome(false);
361               result.setContinue(false);
362               result.setMessage(T_item_move_destination_not_found);
363               return result;
364           }
365 
366           Collection owningCollection = item.getOwningCollection();
367           if (destination.equals(owningCollection))
368           {
369               // nothing to do
370               result.setOutcome(false);
371               result.setContinue(false);
372               return result;
373           }
374 
375           // note: an item.move() method exists, but does not handle several cases:
376           // - no preexisting owning collection (first arg is null)
377           // - item already in collection, but not an owning collection
378           //   (works, but puts item in collection twice)
379 
380           // Don't re-add the item to a collection it's already in.
381           boolean alreadyInCollection = false;
382           for (Collection collection : item.getCollections())
383           {
384               if (collection.equals(destination))
385               {
386                   alreadyInCollection = true;
387                   break;
388               }
389           }
390 
391           // Remove item from its owning collection and add to the destination
392           if (!alreadyInCollection)
393               destination.addItem(item);
394 
395           if (owningCollection != null)
396               owningCollection.removeItem(item);
397 
398           item.setOwningCollection(destination);
399           item.update();
400           context.commit();
401 
402           result.setOutcome(true);
403           result.setContinue(true);
404           result.setMessage(T_item_moved);
405         }
406 
407         return result;
408     }

413 	 * the action has been confirmed.
414 	 * 
415 	 * @param context The DSpace context
416 	 * @param itemID The id of the to-be-deleted item.
417 	 * @return A result object

419 	public static FlowResult processDeleteItem(Context context, int itemID) throws SQLException, AuthorizeException, IOException
420 	{
421 		FlowResult result = new FlowResult();
422 		result.setContinue(false);
423 		
424 		Item item = Item.find(context, itemID);
425 			
426         Collection[] collections = item.getCollections();
427 
428         // Remove item from all the collections it's in
429         for (Collection collection : collections)
430         {
431             collection.removeItem(item);
432         }	
433         
434         // Note: when removing an item from the last collection it will
435         // be removed from the system. So there is no need to also call
436         // an item.delete() method.        
437         
438         context.commit();
439 		
440         result.setContinue(true);
441         
442 		return result;
443 	}

446 	/**
447 	 * Add a new bitstream to the item. The bundle, bitstream (aka file), and description 
448 	 * will be used to create a new bitstream. If the format needs to be adjusted then they 
449 	 * will need to access the edit bitstream form after it has been uploaded.
450 	 * 
451 	 * @param context The DSpace content
452 	 * @param itemID The item to add a new bitstream too
453 	 * @param request The request.
454 	 * @return A flow result
455 	 */
456 	public static FlowResult processAddBitstream(Context context, int itemID, Request request) throws SQLException, AuthorizeException, IOException 
457 	{
458 		FlowResult result = new FlowResult();
459 		result.setContinue(false);
460 		
461 		// Upload a new file
462 		Item item = Item.find(context, itemID);
463 		
464 		
465 		Object object = request.get("file");
466 		Part filePart = null;
467 		if (object instanceof Part)
468 			filePart = (Part) object;
469 
470 		if (filePart != null && filePart.getSize() > 0)
471 		{
472 			InputStream is = filePart.getInputStream();
473 
474 			String bundleName = request.getParameter("bundle");
475 			
476 			Bitstream bitstream;
477 			Bundle[] bundles = item.getBundles(bundleName);
478 			if (bundles.length < 1)
479 			{
480 				// set bundle's name to ORIGINAL
481 				bitstream = item.createSingleBitstream(is, bundleName);
482 				
483 				// set the permission as defined in the owning collection
484 				Collection owningCollection = item.getOwningCollection();
485 				if (owningCollection != null)
486 				{
487 				    Bundle bnd = bitstream.getBundles()[0]; 
488 				    bnd.inheritCollectionDefaultPolicies(owningCollection);
489 				}
490 			}
491 			else
492 			{
493 				// we have a bundle already, just add bitstream
494 				bitstream = bundles[0].createBitstream(is);
495 			}
496 
497 			// Strip all but the last filename. It would be nice
498 			// to know which OS the file came from.
499 			String name = filePart.getUploadName();
500 
501 			while (name.indexOf('/') > -1)
502 			{
503 				name = name.substring(name.indexOf('/') + 1);
504 			}
505 
506 			while (name.indexOf('\\') > -1)
507 			{
508 				name = name.substring(name.indexOf('\\') + 1);
509 			}
510 
511 			bitstream.setName(name);
512 			bitstream.setSource(filePart.getUploadName());
513 			bitstream.setDescription(request.getParameter("description"));
514 
515 			// Identify the format
516 			BitstreamFormat format = FormatIdentifier.guessFormat(context, bitstream);
517 			bitstream.setFormat(format);
518 
519 			// Update to DB
520 			bitstream.update();
521 			item.update();
522 			
523 			context.commit();
524 			
525 			result.setContinue(true);
526 	        result.setOutcome(true);
527 	        result.setMessage(T_bitstream_added); 
528 		}
529 		else
530 		{
531 			result.setContinue(false);
532 	        result.setOutcome(false);
533 	        result.setMessage(T_bitstream_failed); 
534 		}
535 		return result;
536 	}

539 	/**
540 	 * Update a bitstream's metadata.
541 	 * 
542 	 * @param context The DSpace content
543 	 * @param itemID The item to which the bitstream belongs
544 	 * @param bitstreamID The bitstream being updated.
545 	 * @param description The new description of the bitstream
546 	 * @param formatID The new format ID of the bitstream
547 	 * @param userFormat Any user supplied formats.
548 	 * @return A flow result object.
549 	 */
550 	public static FlowResult processEditBitstream(Context context, int itemID, int bitstreamID, String primary, String description, int formatID, String userFormat) throws SQLException, AuthorizeException 
551 	{
552 		FlowResult result = new FlowResult();
553 		result.setContinue(false);
554 		
555 		Bitstream bitstream = Bitstream.find(context, bitstreamID);
556 		BitstreamFormat currentFormat = bitstream.getFormat();
557 
558 		//Step 1:
559 		// Update the bitstream's description
560 		if (description != null)
561 		{
562 			bitstream.setDescription(description);
563 		}
564 		
565 		//Step 2:
566 		// Check if the primary bitstream status has changed
567 		Bundle[] bundles = bitstream.getBundles();
568 		if (bundles != null && bundles.length > 0)
569 		{
570 			if (bitstreamID == bundles[0].getPrimaryBitstreamID())
571 			{
572 				// currently the bitstream is primary
573 				if ("no".equals(primary))
574 				{
575 					// However the user has removed this bitstream as a primary bitstream.
576 					bundles[0].unsetPrimaryBitstreamID();
577 					bundles[0].update();
578 				}
579 			}
580 			else
581 			{
582 				// currently the bitstream is non-primary
583 				if ("yes".equals(primary))
584 				{
585 					// However the user has set this bitstream as primary.
586 					bundles[0].setPrimaryBitstreamID(bitstreamID);
587 					bundles[0].update();
588 				}
589 			}
590 		}
591 		
592 		
593 		//Step 2:
594 		// Update the bitstream's format
595 		if (formatID > 0)
596 		{
597 			if (currentFormat == null || currentFormat.getID() != formatID)
598 			{
599 				BitstreamFormat newFormat = BitstreamFormat.find(context, formatID);
600 				if (newFormat != null)
601 				{
602 					bitstream.setFormat(newFormat);
603 				}
604 			}
605 		}
606 		else
607 		{
608 			if (userFormat != null && userFormat.length() > 0)
609 			{
610 				bitstream.setUserFormatDescription(userFormat);
611 			}
612 		}
613 		
614 		//Step 3:
615 		// Save our changes
616 		bitstream.update();
617 		context.commit();
618 		
619 		 result.setContinue(true);
620 	     result.setOutcome(true);
621 	     result.setMessage(T_bitstream_updated);
622 	        
623 		
624 		return result;
625 	}

627 	/**
628 	 * Delete the given bitstreams from the bundle and item. If there are no more bitstreams 
629 	 * left in a bundle then also remove it.
630 	 * 
631 	 * @param context Current dspace content
632 	 * @param itemID The item id from which to remove bitstreams
633 	 * @param bitstreamIDs A bundle slash bitstream id pair of bitstreams to be removed.
634 	 * @return A flow result
635 	 */
636 	public static FlowResult processDeleteBitstreams(Context context, int itemID, String[] bitstreamIDs) throws SQLException, AuthorizeException, IOException, UIException
637 	{
638 		FlowResult result = new FlowResult();
639 		result.setContinue(false);
640 		
641 		Item item = Item.find(context, itemID);
642 		
643 		for (String id : bitstreamIDs)
644 		{
645 			String[] parts = id.split("/");
646 			
647 			if (parts.length != 2)
648 				throw new UIException("Unable to parse id into bundle and bitstream id: "+id);
649 			
650 			int bundleID = Integer.valueOf(parts[0]);
651 			int bitstreamID = Integer.valueOf(parts[1]);
652 			
653 			Bundle bundle = Bundle.find(context, bundleID);
654 			Bitstream bitstream = Bitstream.find(context,bitstreamID);
655 			
656 			bundle.removeBitstream(bitstream);
657 			
658 			if (bundle.getBitstreams().length == 0)
659 			{
660 				item.removeBundle(bundle);
661 			}
662 		}
663 		
664 		item.update();
665 		
666 		context.commit();
667 		
668 		result.setContinue(true);
669 		result.setOutcome(true);
670 		result.setMessage(T_bitstream_delete);
671 		
672 		return result;
673 	}

676 	/**
677 	 * Parse the given name into three parts, divided by an _. Each part should represent the 
678 	 * schema, element, and qualifier. You are guaranteed that if no qualifier was supplied the 
679 	 * third entry is null.
680 	 * 
681 	 * @param name The name to be parsed.
682 	 * @return An array of name parts.
683 	 */
684 	private static String[] parseName(String name) throws UIException
