63  /**
64   * This generic transformer is used to generate the DRI
65   * for any Submission Step which extends the class
66   * org.dspace.app.xmlui.submission.AbstractStep
67   * <p>
68   * This transformer just initializes the current step class
69   * and calls the appropriate method(s) for the step.
70   * 
71   * @author Tim Donohue
72   * @see AbstractStep
73   */
74  public class StepTransformer extends AbstractDSpaceTransformer
75  {
76  	private static Logger log = Logger.getLogger(StepTransformer.class);
77  
78  	
79  	/** 
80       * The id of the currently active workspace or workflow, this contains 
81       * the incomplete DSpace item 
82       */
83  	protected String id;
84  	
85  	/**
86  	 * The current step and page's numeric value that it is at currently. This 
87  	 * number is dynamic between submissions and is a double where the integer
88  	 * value is the step #, and the decimal value is the page # 
89  	 * (e.g. 1.2 is page 2 of step 1)
90  	 */
91  	private double stepAndPage;
92  	
93  	/**
94  	 * Full name of the step's transformer class (which must extend
95  	 * org.dspace.app.xmlui.submission.AbstractStep).
96  	 */
97  	private String transformerClassName;
98  	
99  	/** 
100      * The handle of the collection into which this DSpace
101      * item is being submitted
102      */
103 	protected String collectionHandle;
104 	
105 	/**
106 	 * An instance of the current step's transformer class (which must extend
107 	 * org.dspace.app.xmlui.submission.AbstractStep).  This class is
108 	 * used to generate the actual DRI for this step.
109 	 */
110 	private AbstractSubmissionStep step;
111 	
112 	
113 	/**
114 	 * Grab all the step's parameters from the sitemap. This includes 
115 	 * workspaceID, step, and a list of errored fields.
116 	 * 
117 	 * If the implementer set any required parameters then insure that 
118 	 * they are all present.
119 	 */
120 	public void setup(SourceResolver resolver, Map objectModel, String src, Parameters parameters) 
121 	throws ProcessingException, SAXException, IOException
122 	{ 
123 		super.setup(resolver,objectModel,src,parameters);
124 
125 		//retrieve id and transformer information
126 		//(This ID should always exist by this point, since the
127 		// selection of the collection should have already happened!)
128 		this.id = parameters.getParameter("id",null);
129 		this.transformerClassName = parameters.getParameter("transformer",null);
130 		
131 		//even though its not used in this class, this "step" parameter
132 		//is heavily used by the Transformers which extend the 
133 		//org.dspace.app.xmlui.submission.AbstractStep
134 		this.stepAndPage = Double.valueOf(parameters.getParameter("step","-1"));
135 		
136 		//retrieve collection handle if it's there
137 		this.collectionHandle = parameters.getParameter("handle",null);
138 		
139 		try
140 		{
141 			//retrieve an instance of the transformer class
142 	        ClassLoader loader = this.getClass().getClassLoader();
143 	        Class stepClass = loader
144 	                .loadClass(this.transformerClassName);
145 	
146 	        // this XML-UI class *must* be a valid AbstractSubmissionStep, 
147 	        // or else we'll have problems here
148 	        step = (AbstractSubmissionStep) stepClass
149 	                    .newInstance();
150 		}
151 		catch(ClassNotFoundException cnfe)
152 		{
153 			//means that we couldn't find a class by the given name
154 			throw new ProcessingException("Class Not Found: " + this.transformerClassName, cnfe);
155 		}
156 	    catch(Exception e)
157 	    {
158 	    	//means we couldn't instantiate the class as an AbstractStep
159 	    	throw new ProcessingException("Unable to instantiate class " + this.transformerClassName + ". " +
160 	    								  "Please make sure it extends org.dspace.app.xmlui.submission.AbstractSubmissionStep!", e);
161 	    }
162 	
163 	    
164 	    //call the setup for this step
165 	    if(step!=null)
166 	    	step.setup(resolver, objectModel, src, parameters);
167 	    else
168 	    	throw new ProcessingException("Step class is null!  We do not have a valid AbstractStep in " + this.transformerClassName + ". ");
169 	}
170

172 	/** What to add at the end of the body */
173     public void addBody(Body body) throws SAXException, WingException,
174             UIException, SQLException, IOException, AuthorizeException
175     {
176         //call addBody for this step
177     	step.addBody(body);
178     }
179 
180     /** What to add to the options list */
181     public void addOptions(Options options) throws SAXException, WingException,
182             UIException, SQLException, IOException, AuthorizeException
183     {
184         //call addOptions for this step
185     	step.addOptions(options);
186     }
187 
188     /** What user metadata to add to the document */
189     public void addUserMeta(UserMeta userMeta) throws SAXException,
190             WingException, UIException, SQLException, IOException,
191             AuthorizeException
192     {
193     	//call addUserMeta for this step
194     	step.addUserMeta(userMeta);
195     }
196 
197     /** What page metadata to add to the document */
198     public void addPageMeta(PageMeta pageMeta) throws SAXException,
199             WingException, UIException, SQLException, IOException,
200             AuthorizeException
201     {
202     	//call addPageMeta for this step
203     	step.addPageMeta(pageMeta);
204     }

206 
207 	/**
208 	 * Recycle
209 	 */
210 	public void recycle() 
211 	{
212 		this.id = null;
213 		this.transformerClassName = null;
214 		this.collectionHandle = null;
215 		this.stepAndPage = -1;
216 		if(step!=null)
217         {    
218 		    this.step.recycle();
219 		    this.step = null;
220         }
221 		super.recycle();
222 	}
