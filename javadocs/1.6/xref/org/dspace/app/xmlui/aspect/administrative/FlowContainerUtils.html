98  	/**
99  	 * Process the collection metadata edit form.
100 	 * 
101 	 * @param context The current DSpace context.
102 	 * @param collectionID The collection id.
103 	 * @param deleteLogo Determines if the logo should be deleted along with the metadata editing action.
104 	 * @param request the Cocoon request object
105 	 * @return A process result's object.
106 	 */
107 	public static FlowResult processEditCollection(Context context, int collectionID, boolean deleteLogo, Request request) throws SQLException, IOException, AuthorizeException
108 	{
109 		FlowResult result = new FlowResult();
110 		
111 		Collection collection = Collection.find(context, collectionID);
112 		
113 		// Get the metadata
114 		String name = request.getParameter("name");
115 		String shortDescription = request.getParameter("short_description");
116 		String introductoryText = request.getParameter("introductory_text");
117 		String copyrightText = request.getParameter("copyright_text");
118 		String sideBarText = request.getParameter("side_bar_text");
119 		String license = request.getParameter("license");
120 		String provenanceDescription = request.getParameter("provenance_description");
121 		
122 		// If they don't have a name then make it untitled.
123 		if (name == null || name.length() == 0)
124 			name = "Untitled";
125 
126 		// If empty, make it null.
127 		if (shortDescription != null && shortDescription.length() == 0)
128 			shortDescription = null;
129 		if (introductoryText != null && introductoryText.length() == 0)
130 			introductoryText = null;
131 		if (copyrightText != null && copyrightText.length() == 0)
132 			copyrightText = null;
133 		if (sideBarText != null && sideBarText.length() == 0)
134 			sideBarText = null;
135 		if (license != null && license.length() == 0)
136 			license = null;
137 		if (provenanceDescription != null && provenanceDescription.length() == 0)
138 			provenanceDescription = null;
139 		
140 		// Save the metadata
141 		collection.setMetadata("name", name);
142 		collection.setMetadata("short_description", shortDescription);
143 		collection.setMetadata("introductory_text", introductoryText);
144 		collection.setMetadata("copyright_text", copyrightText);
145 		collection.setMetadata("side_bar_text", sideBarText);
146 		collection.setMetadata("license", license);
147 		collection.setMetadata("provenance_description", provenanceDescription);
148 		
149         
150 		// Change or delete the logo
151         if (deleteLogo)
152         {
153         	// Remove the logo
154         	collection.setLogo(null);
155         }
156         else
157         {
158         	// Update the logo
159     		Object object = request.get("logo");
160     		Part filePart = null;
161     		if (object instanceof Part)
162     			filePart = (Part) object;
163 
164     		if (filePart != null && filePart.getSize() > 0)
165     		{
166     			InputStream is = filePart.getInputStream();
167     			
168     			collection.setLogo(is);
169     		}
170         }
171         
172         // Save everything
173         collection.update();
174         context.commit();
175         
176         
177         // No notice...
178         result.setContinue(true);
179 		
180 		return result;
181 	}

183 	/**
184 	 * Process the collection harvesting options form.
185 	 * 
186 	 * @param context The current DSpace context.
187 	 * @param collectionID The collection id.
188 	 * @param request the Cocoon request object
189 	 * @return A process result's object.
190 	 */
191 	public static FlowResult processSetupCollectionHarvesting(Context context, int collectionID, Request request) throws SQLException, IOException, AuthorizeException
192 	{
193 		FlowResult result = new FlowResult();
194 		HarvestedCollection hc = HarvestedCollection.find(context, collectionID);
195 
196 		String contentSource = request.getParameter("source");
197 
198 		// First, if this is not a harvested collection (anymore), set the harvest type to 0; possibly also wipe harvest settings  
199 		if (contentSource.equals("source_normal")) 
200 		{
201 			if (hc != null) 
202 				hc.delete();
203 			
204 			result.setContinue(true);
205 		}
206 		else 
207 		{
208 			FlowResult subResult = testOAISettings(context, request);
209 			
210 			// create a new harvest instance if all the settings check out
211 			if (hc == null) {
212 				hc = HarvestedCollection.create(context, collectionID);
213 			}
214 			
215 			// if the supplied options all check out, set the harvesting parameters on the collection
216 			if (subResult.getErrors().isEmpty()) {
217 				String oaiProvider = request.getParameter("oai_provider");
218 				String oaiSetId = request.getParameter("oai_setid");
219 				String metadataKey = request.getParameter("metadata_format");
220 				String harvestType = request.getParameter("harvest_level");
221 				
222 				hc.setHarvestParams(Integer.parseInt(harvestType), oaiProvider, oaiSetId, metadataKey);
223 				hc.setHarvestStatus(HarvestedCollection.STATUS_READY);
224 			}
225 			else {
226 				result.setErrors(subResult.getErrors());
227 				result.setContinue(false);
228 				return result;
229 			}
230 			
231 			hc.update();
232 		}
233 		        
234         // Save everything
235         context.commit();
236         
237         // No notice...
238         //result.setMessage(new Message("default","Harvesting options successfully modified."));
239         result.setOutcome(true);
240         result.setContinue(true);
241 		
242 		return result;
243 	}

247 	 * Use the collection's harvest settings to immediately perform a harvest cycle.
248 	 * 
249 	 * @param context The current DSpace context.
250 	 * @param collectionID The collection id.
251 	 * @param request the Cocoon request object
252 	 * @return A process result's object.
253 	 * @throws TransformerException 
254 	 * @throws SAXException 
255 	 * @throws ParserConfigurationException 
256 	 * @throws CrosswalkException 
257 	 */
258 	public static FlowResult processRunCollectionHarvest(Context context, int collectionID, Request request) throws SQLException, IOException, AuthorizeException, CrosswalkException, ParserConfigurationException, SAXException, TransformerException
259 	{
260 		FlowResult result = new FlowResult();
261 		OAIHarvester harvester;
262 		List<String> testErrors = new ArrayList<String>();
263 		Collection collection = Collection.find(context, collectionID);
264 		HarvestedCollection hc = HarvestedCollection.find(context, collectionID);
265 
266 		//TODO: is there a cleaner way to do this?
267 		try {
268 			if (HarvestScheduler.status != HarvestScheduler.HARVESTER_STATUS_STOPPED) {
269 				synchronized(HarvestScheduler.lock) {
270 					HarvestScheduler.interrupt = HarvestScheduler.HARVESTER_INTERRUPT_INSERT_THREAD;
271 					HarvestScheduler.interruptValue = collection.getID();
272 					HarvestScheduler.lock.notify();
273 				}
274 			}
275 			else {
276 				harvester = new OAIHarvester(context, collection, hc);
277 				harvester.runHarvest();
278 			}
279 		}
280 		catch (Exception e) {
281 			testErrors.add(e.getMessage());
282 			result.setErrors(testErrors);
283 			result.setContinue(false);
284 			return result;
285 		}
286 		
287         result.setContinue(true);
288 		
289 		return result;
290 	}

294 	 * Purge the collection of all items, then run a fresh harvest cycle.
295 	 * 
296 	 * @param context The current DSpace context.
297 	 * @param collectionID The collection id.
298 	 * @param request the Cocoon request object
299 	 * @return A process result's object.
300 	 * @throws TransformerException 
301 	 * @throws SAXException 
302 	 * @throws ParserConfigurationException 
303 	 * @throws CrosswalkException 
304 	 * @throws BrowseException 
305 	 */
306 	public static FlowResult processReimportCollection(Context context, int collectionID, Request request) throws SQLException, IOException, AuthorizeException, CrosswalkException, ParserConfigurationException, SAXException, TransformerException, BrowseException 
307 	{
308 		FlowResult result = new FlowResult();
309 		Collection collection = Collection.find(context, collectionID);
310 		HarvestedCollection hc = HarvestedCollection.find(context, collectionID);
311 		
312 		ItemIterator it = collection.getAllItems();
313 		//IndexBrowse ib = new IndexBrowse(context);
314 		while (it.hasNext()) {
315 			Item item = it.next();
316 			//System.out.println("Deleting: " + item.getHandle());
317 			//ib.itemRemoved(item);
318 			collection.removeItem(item);
319 		}
320 		hc.setHarvestResult(null,"");
321 		hc.update();
322 		collection.update();
323 		context.commit();
324 		
325 		result = processRunCollectionHarvest(context, collectionID, request);		
326 		
327 		return result;		
328 	}

333 	 * 
334 	 * @param context
335 	 * @param request
336 	 * @return
337 	 */
338 	public static FlowResult testOAISettings(Context context, Request request)  
339 	{
340 		FlowResult result  = new FlowResult();
341 		
342 		String oaiProvider = request.getParameter("oai_provider");
343 		String oaiSetId = request.getParameter("oai_setid");
344 		String metadataKey = request.getParameter("metadata_format");
345 		String harvestType = request.getParameter("harvest_level");
346 		int harvestTypeInt = 0;
347 		
348 		if (oaiProvider == null || oaiProvider.length() == 0)
349 			result.addError("oai_provider");
350 		if (oaiSetId == null || oaiSetId.length() == 0)
351 			result.addError("oai_setid");
352 		if (metadataKey == null || metadataKey.length() == 0)
353 			result.addError("metadata_format");
354 		if (harvestType == null || harvestType.length() == 0)
355 			result.addError("harvest_level");
356 		else
357 			harvestTypeInt = Integer.parseInt(harvestType);
358 			
359 		
360 		if (result.getErrors() == null) {
361 			List<String> testErrors = OAIHarvester.verifyOAIharvester(oaiProvider, oaiSetId, metadataKey, (harvestTypeInt>1));
362 			result.setErrors(testErrors);
363 		}
364 		
365 		if (result.getErrors() == null || result.getErrors().isEmpty()) {
366 			result.setOutcome(true);
367 			// On a successful test we still want to stay in the loop, not continue out of it
368 			//result.setContinue(true);
369 			result.setMessage(new Message("default","Harvesting settings are valid."));
370 		}
371 		else {
372 			result.setOutcome(false);
373 			result.setContinue(false);
374 			// don't really need a message when the errors are highlighted already
375 			//result.setMessage(new Message("default","Harvesting is not properly configured."));
376 		}
377 
378 		return result;
379 	}

411 	 * Look up the id of a group authorized for one of the given roles. If no group is currently 
412 	 * authorized to preform this role then a new group will be created and assigned the role.
413 	 * 
414 	 * @param context The current DSpace context.
415 	 * @param collectionID The collection id.
416 	 * @param roleName ADMIN, WF_STEP1,	WF_STEP2, WF_STEP3,	SUBMIT, DEFAULT_READ.
417 	 * @return The id of the group associated with that particular role, or -1 if the role was not found.
418 	 */
419 	public static int getCollectionRole(Context context, int collectionID, String roleName) throws SQLException, AuthorizeException, IOException
420 	{
421 		Collection collection = Collection.find(context, collectionID);
422 		
423 		// Determine the group based upon wich role we are looking for.
424 		Group role = null;
425 		if (ROLE_ADMIN.equals(roleName))
426 		{
427 			role = collection.getAdministrators();
428 			if (role == null)
429 				role = collection.createAdministrators();
430 		} 
431 		else if (ROLE_SUBMIT.equals(roleName))
432 		{
433 			role = collection.getSubmitters();
434 			if (role == null)
435 				role = collection.createSubmitters();
436 		}
437 		else if (ROLE_WF_STEP1.equals(roleName))
438 		{	
439 			role = collection.getWorkflowGroup(1);
440 			if (role == null)
441 				role = collection.createWorkflowGroup(1);
442 			
443 		}
444 		else if (ROLE_WF_STEP2.equals(roleName))
445 		{
446 			role = collection.getWorkflowGroup(2);
447 			if (role == null)
448 				role = collection.createWorkflowGroup(2);
449 		}
450 		else if (ROLE_WF_STEP3.equals(roleName))
451 		{
452 			role = collection.getWorkflowGroup(3);
453 			if (role == null)
454 				role = collection.createWorkflowGroup(3);
455 			
456 		}
457 		
458 		// In case we needed to create a group, save our changes
459 		collection.update();
460 		context.commit();
461 		
462 		// If the role name was valid then role should be non null,
463 		if (role != null)
464 			return role.getID();
465 		
466 		return -1;
467 	}

470 	/**
471 	 * Delete one of collection's roles
472 	 * 
473 	 * @param context The current DSpace context.
474 	 * @param collectionID The collection id.
475 	 * @param roleName ADMIN, WF_STEP1,	WF_STEP2, WF_STEP3,	SUBMIT, DEFAULT_READ.
476 	 * @param groupID The id of the group associated with this role.
477 	 * @return A process result's object.
478 	 */
479 	public static FlowResult processDeleteCollectionRole(Context context, int collectionID, String roleName, int groupID) throws SQLException, UIException, IOException, AuthorizeException
480 	{
481 		FlowResult result = new FlowResult();
482 		
483 		Collection collection = Collection.find(context,collectionID);
484 		Group role = Group.find(context, groupID);
485 		
486 		// First, Unregister the role
487 		if (ROLE_ADMIN.equals(roleName))
488 		{
489 			collection.removeAdministrators();
490 		} 
491 		else if (ROLE_SUBMIT.equals(roleName))
492 		{
493 			collection.removeSubmitters();
494 		}
495 		else if (ROLE_WF_STEP1.equals(roleName))
496 		{	
497 			collection.setWorkflowGroup(1, null);
498 		}
499 		else if (ROLE_WF_STEP2.equals(roleName))
500 		{
501 			collection.setWorkflowGroup(2, null);
502 		}
503 		else if (ROLE_WF_STEP3.equals(roleName))
504 		{
505 			collection.setWorkflowGroup(3, null);
506 			
507 		}
508 		
509 		// Second, remove all outhorizations for this role by searching for all policies that this 
510 		// group has on the collection and remove them otherwise the delete will fail because 
511 		// there are dependencies.
512 		@SuppressWarnings("unchecked") // the cast is correct
513 		List<ResourcePolicy> policies = AuthorizeManager.getPolicies(context,collection);
514 		for (ResourcePolicy policy : policies)
515 		{
516 			if (policy.getGroupID() == groupID)
517 				policy.delete();
518 		}
519 		
520 		// Finally, Delete the role's actual group.
521 		collection.update();
522 		role.delete();
523 		context.commit();
524 	
525 		result.setContinue(true);
526 		result.setOutcome(true);
527 		result.setMessage(new Message("default","The role was successfully deleted."));
528 		return result;
529 	}

533 	 * Look up the id of a group authorized for one of the given roles. If no group is currently 
534 	 * authorized to preform this role then a new group will be created and assigned the role.
535 	 * 
536 	 * @param context The current DSpace context.
537 	 * @param collectionID The collection id.
538 	 * @param roleName ADMIN, WF_STEP1,	WF_STEP2, WF_STEP3,	SUBMIT, DEFAULT_READ.
539 	 * @return The id of the group associated with that particular role.
540 	 */
541 	public static int getCollectionDefaultRead(Context context, int collectionID) throws SQLException, AuthorizeException
542 	{
543 		Collection collection = Collection.find(context,collectionID);
544 		
545 		Group[] itemGroups = AuthorizeManager.getAuthorizedGroups(context, collection, Constants.DEFAULT_ITEM_READ);
546 		Group[] bitstreamGroups = AuthorizeManager.getAuthorizedGroups(context, collection, Constants.DEFAULT_BITSTREAM_READ);
547 		
548 		if (itemGroups.length != 1 && bitstreamGroups.length != 1)
549 			// If there are more than one groups assigned either of these privleges then this role based method will not work.
550 			// The user will need to go to the authorization section to manualy straight this out.
551 			return -1;
552 		
553 		Group itemGroup = itemGroups[0];
554 		Group bitstreamGroup = bitstreamGroups[0];
555 		
556 		if (itemGroup.getID() != bitstreamGroup.getID())
557 			// If the same group is not assigned both of these priveleges then this role based method will not work. The user 
558 			// will need to go to the authorization section to manualy straighten this out.
559 			return -1;
560 		
561 		
562 		
563 		return itemGroup.getID();
564 	}

566 	/**
567 	 * Change default privleges from the anonymous group to a new group that will be created and
568 	 * approrpate privleges assigned. The id of this new group will be returned.
569 	 * 
570 	 * @param context The current DSpace context.
571 	 * @param collectionID The collection id.
572 	 * @return The group ID of the new group.
573 	 */
574 	public static int createCollectionDefaultReadGroup(Context context, int collectionID) throws SQLException, AuthorizeException, UIException
575 	{
576 		int roleID = getCollectionDefaultRead(context, collectionID);
577 		
578 		if (roleID != 0)
579 			throw new UIException("Unable to create a new default read group because either the group allready exists or multiple groups are assigned the default privleges.");
580 		
581 		Collection collection = Collection.find(context,collectionID);
582 		Group role = Group.create(context);
583 		role.setName("COLLECTION_"+collection.getID() +"_DEFAULT_READ");
584 		
585 		// Remove existing privleges from the anynomous group.
586 		AuthorizeManager.removePoliciesActionFilter(context, collection, Constants.DEFAULT_ITEM_READ);
587 		AuthorizeManager.removePoliciesActionFilter(context, collection, Constants.DEFAULT_BITSTREAM_READ);
588 		
589 		// Grant our new role the default privleges.
590 		AuthorizeManager.addPolicy(context, collection, Constants.DEFAULT_ITEM_READ,      role);
591 		AuthorizeManager.addPolicy(context, collection, Constants.DEFAULT_BITSTREAM_READ, role);
592 		
593 		// Committ the changes
594 		role.update();
595 		context.commit();
596 		
597 		return role.getID();
598 	}

600 	/**
601 	 * Change the default read priveleges to the anonymous group.
602 	 * 
603 	 * If getCollectionDefaultRead() returns -1 or the anonymous group then nothing 
604 	 * is done. 
605 	 * 
606 	 * @param context The current DSpace context.
607 	 * @param collectionID The collection id.
608 	 * @return A process result's object.
609 	 */
610 	public static FlowResult changeCollectionDefaultReadToAnonymous(Context context, int collectionID) throws SQLException, AuthorizeException, UIException
611 	{
612 		FlowResult result = new FlowResult();
613 		
614 		int roleID = getCollectionDefaultRead(context, collectionID);
615 		
616 		if (roleID < 1)
617 		{
618 			throw new UIException("Unable to delete the default read role because the role is either allready assigned to the anonymous group or multiple groups are assigned the default priveleges.");
619 		}
620 		
621 		Collection collection = Collection.find(context,collectionID);
622 		Group role = Group.find(context, roleID);
623 		Group anonymous = Group.find(context,0);
624 		
625 		// Delete the old role, this will remove the default privleges.
626 		role.delete();
627 		
628 		// Set anonymous as the default read group.
629 		AuthorizeManager.addPolicy(context, collection, Constants.DEFAULT_ITEM_READ,      anonymous);
630 		AuthorizeManager.addPolicy(context, collection, Constants.DEFAULT_BITSTREAM_READ, anonymous);
631 		
632 		// Commit the changes
633 		context.commit();
634

1069     /** 
1070      * Sanatize any XML that was inputed by the user, this will clean up
1071      * any unescaped characters so that they can be stored as proper XML.
1072      * These are errors that in general we want to take care of on behalf
1073      * of the user.
1074      * 
1075      * @param value The unsantized value
1076      * @return A sanatized value
1077      */
1078 	public static String escapeXMLEntities(String value)
1079 	{
1080 		if (value == null)
1081 			return null;
1082 		
1083 		// Escape any XML entities
1084     	int amp = -1;
1085     	while ((amp = value.indexOf('&', amp+1)) > -1)
1086     	{
1087     		// Is it an xml entity named by number?
1088     		if (substringCompare(value,amp+1,'#'))
1089     			continue;
1090     		
1091     		// &amp;
1092     		if (substringCompare(value,amp+1,'a','m','p',';'))
1093     			continue;
1094     		
1095     		// &apos;
1096     		if (substringCompare(value,amp+1,'a','p','o','s',';'))
1097     			continue;
1098     		
1099     		// &quot;
1100     		if (substringCompare(value,amp+1,'q','u','o','t',';'))
1101     			continue;
1102     			
1103     		// &lt;
1104     		if (substringCompare(value,amp+1,'l','t',';'))
1105     			continue;
1106     		
1107     		// &gt;
1108     		if (substringCompare(value,amp+1,'g','t',';'))
1109     			continue;
1110     		
1111     		// Replace the ampersand with an XML entity.
1112     		value = value.substring(0,amp) + "&amp;" + value.substring(amp+1);
1113     	}
1114     	
1115     	return value;
1116 	}
