166     throws ProcessingException, SAXException, IOException
167     {

169

171         //was requested, then we need to load EditFileStep instead!
172         if(this.errorFlag==org.dspace.submit.step.UploadStep.STATUS_EDIT_BITSTREAM)

181     public void addBody(Body body) throws SAXException, WingException,
182             UIException, SQLException, IOException, AuthorizeException
183     {
184         //If we are actually editing an uploaded file's information,
185         //then display that body instead!
186     	if(this.editFile!=null)
187         {
188             editFile.addBody(body);
189             return;
190         }
191         
192         // Get a list of all files in the original bundle
193 		Item item = submission.getItem();
194 		Collection collection = submission.getCollection();
195 		String actionURL = contextPath + "/handle/"+collection.getHandle() + "/submit/" + knot.getId() + ".continue";
196 		boolean workflow = submission instanceof WorkflowItem;
197 		Bundle[] bundles = item.getBundles("ORIGINAL");
198 		Bitstream[] bitstreams = new Bitstream[0];
199 		if (bundles.length > 0)
200 		{
201 			bitstreams = bundles[0].getBitstreams();
202 		}
203 		
204 		// Part A: 
205 		//  First ask the user if they would like to upload a new file (may be the first one)
206     	Division div = body.addInteractiveDivision("submit-upload", actionURL, Division.METHOD_MULTIPART, "primary submission");
207     	div.setHead(T_submission_head);
208     	addSubmissionProgressList(div);
209     	
210     	
211     	List upload = null;
212     	if (!workflow)
213     	{
214     		// Only add the upload capabilities for new item submissions
215 	    	upload = div.addList("submit-upload-new", List.TYPE_FORM);
216 	        upload.setHead(T_head);    
217 	        
218 	        File file = upload.addItem().addFile("file");
219 	        file.setLabel(T_file);
220 	        file.setHelp(T_file_help);
221 	        file.setRequired();
222 	        
223 	        //if no files found error was thrown by processing class, display it!
224 	        if (this.errorFlag==org.dspace.submit.step.UploadStep.STATUS_NO_FILES_ERROR)
225 	        {
226                 file.addError(T_file_error);
227             }
228 
229             // if an upload error was thrown by processing class, display it!
230             if (this.errorFlag == org.dspace.submit.step.UploadStep.STATUS_UPLOAD_ERROR)
231             {
232                 file.addError(T_upload_error);
233             }
234 	        	
235 	        Text description = upload.addItem().addText("description");
236 	        description.setLabel(T_description);
237 	        description.setHelp(T_description_help);
238 	        
239 	        Button uploadSubmit = upload.addItem().addButton("submit_upload");
240 	        uploadSubmit.setValue(T_submit_upload);
241     	}
242         
243         // Part B:
244         //  If the user has allready uploaded files provide a list for the user.
245         if (bitstreams.length > 0 || workflow)
246 		{
247 	        Table summary = div.addTable("submit-upload-summary",(bitstreams.length * 2) + 2,7);
248 	        summary.setHead(T_head2);
249 	        
250 	        Row header = summary.addRow(Row.ROLE_HEADER);
251 	        header.addCellContent(T_column0); // primary bitstream
252 	        header.addCellContent(T_column1); // select checkbox
253 	        header.addCellContent(T_column2); // file name
254 	        header.addCellContent(T_column3); // size
255 	        header.addCellContent(T_column4); // description
256 	        header.addCellContent(T_column5); // format
257 	        header.addCellContent(T_column6); // edit button
258 	        
259 	        for (Bitstream bitstream : bitstreams)
260 	        {
261 	        	int id = bitstream.getID();
262 	        	String name = bitstream.getName();
263 	        	String url = makeBitstreamLink(item, bitstream);
264 	        	long bytes = bitstream.getSize();
265 	        	String desc = bitstream.getDescription();
266 	        	String algorithm = bitstream.getChecksumAlgorithm();
267 	        	String checksum = bitstream.getChecksum();
268 	        	BitstreamFormat format = bitstream.getFormat();
269 	        	int support = format.getSupportLevel();
270 	        	
271 	        	
272 	        	Row row = summary.addRow();
273 
274 	            // Add radio-button to select this as the primary bitstream
275                 Radio primary = row.addCell().addRadio("primary_bitstream_id");
276                 primary.addOption(String.valueOf(id));
277                 
278                 // If this bitstream is already marked as the primary bitstream
279                 // mark it as such.
280                 if(bundles[0].getPrimaryBitstreamID() == id) {
281                     primary.setOptionSelected(String.valueOf(id));
282                 }
283 	        	
284 	        	if (!workflow)
285 	        	{
286 	        		// Workflow users can not remove files.
287 		            CheckBox remove = row.addCell().addCheckBox("remove");
288 		            remove.setLabel("remove");
289 		            remove.addOption(id);
290 	        	}
291 	        	else
292 	        	{
293 	        		row.addCell();
294 	        	}
295 	        	
296 	            row.addCell().addXref(url,name);
297 	            row.addCellContent(bytes + " bytes");
298 	            if (desc == null || desc.length() == 0)
299 	            	row.addCellContent(T_unknown_name);
300 	            else
301 	            	row.addCellContent(desc);
302 	            
303 	            if (format == null)
304 	            {
305 	            	row.addCellContent(T_unknown_format);
306 	            }
307 	            else
308 	            {
309 	            	Cell cell = row.addCell();
310 	            	cell.addContent(format.getMIMEType());
311 	            	cell.addContent(" ");
312 	            	switch (support)
313 	            	{
314 	            	case 1:
315 	            		cell.addContent(T_supported);
316 	            		break;
317 	            	case 2:
318 	            		cell.addContent(T_known);
319 	            		break;
320 	            	case 3:
321 	            		cell.addContent(T_unsupported);
322 	            		break;
323 	            	}
324 	            }
325 	            
326 	            Button edit = row.addCell().addButton("submit_edit_"+id);
327 	            edit.setValue(T_submit_edit);  
328 	            
329 	            Row checksumRow = summary.addRow();
330 	            checksumRow.addCell();
331 	            Cell checksumCell = checksumRow.addCell(null, null, 0, 6, null);
332 	            checksumCell.addHighlight("bold").addContent(T_checksum);
333 	            checksumCell.addContent(" ");
334 	            checksumCell.addContent(algorithm + ":" + checksum);
335 	        }
336 	        
337 	        if (!workflow)
338 	        {
339 	        	// Workflow user's can not remove files.
340 		        Row actionRow = summary.addRow();
341 		        actionRow.addCell();
342 		        Button removeSeleceted = actionRow.addCell(null, null, 0, 6, null).addButton("submit_remove_selected");
343 		        removeSeleceted.setValue(T_submit_remove);
344 	        }
345 	        
346 	        upload = div.addList("submit-upload-new-part2", List.TYPE_FORM);
347 		}
348         
349         // Part C:
350         // add standard control/paging buttons
351         addControlButtons(upload);
352     }

354     /** 
355      * Each submission step must define its own information to be reviewed
356      * during the final Review/Verify Step in the submission process.
357      * <P>
358      * The information to review should be tacked onto the passed in 
359      * List object.
360      * <P>
361      * NOTE: To remain consistent across all Steps, you should first
362      * add a sub-List object (with this step's name as the heading),
363      * by using a call to reviewList.addList().   This sublist is
364      * the list you return from this method!
365      * 
366      * @param reviewList
367      *      The List to which all reviewable information should be added
368      * @return 
369      *      The new sub-List object created by this step, which contains
370      *      all the reviewable information.  If this step has nothing to
371      *      review, then return null!   
372      */
373     public List addReviewSection(List reviewList) throws SAXException,
374         WingException, UIException, SQLException, IOException,
375         AuthorizeException
376     {
377         // Create a new list section for this step (and set its heading)
378         List uploadSection = reviewList.addList("submit-review-" + this.stepAndPage, List.TYPE_FORM);
379         uploadSection.setHead(T_head);
380         
381         //Review all uploaded files
382         Item item = submission.getItem();
383         Bundle[] bundles = item.getBundles("ORIGINAL");
384         Bitstream[] bitstreams = new Bitstream[0];
385         if (bundles.length > 0)
386         {
387             bitstreams = bundles[0].getBitstreams();
388         }
389         
390         for (Bitstream bitstream : bitstreams)
391         {
392             BitstreamFormat bitstreamFormat = bitstream.getFormat();
393             
394             int id = item.getID();
395             String name = bitstream.getName();
396             String url = makeBitstreamLink(item, bitstream);
397             String format = bitstreamFormat.getShortDescription();
398             Message support = ReviewStep.T_unknown;
399             if (bitstreamFormat.getSupportLevel() == BitstreamFormat.KNOWN)
400                 support = T_known;
401             else if (bitstreamFormat.getSupportLevel() == BitstreamFormat.SUPPORTED)
402                 support = T_supported;
403             
404             org.dspace.app.xmlui.wing.element.Item file = uploadSection.addItem();
405             file.addXref(url,name);
406             file.addContent(" - "+ format + " ");
407             file.addContent(support);
408             
409         }
410         
411         //return this new "upload" section
412         return uploadSection;
413     }

415     /**
416      * Returns canonical link to a bitstream in the item.
417      *
418      * @param item The DSpace Item that the bitstream is part of
419      * @param bitstream The bitstream to link to
420      * @returns a String link to the bistream
421      */
422     private String makeBitstreamLink(Item item, Bitstream bitstream)
423     {
424         String name = bitstream.getName();
425         StringBuilder result = new StringBuilder(contextPath);
426         result.append("/bitstream/item/").append(String.valueOf(item.getID()));
427         // append name although it isn't strictly necessary
428         try
429         {
430             if (name != null)
431             {
432                 result.append("/").append(Util.encodeBitstreamName(name, "UTF-8"));
433             }
434         }
435         catch (UnsupportedEncodingException uee)
436         {
437             // just ignore it, we don't have to have a pretty
438             // name on the end of the url because the sequence id will
439             // locate it. However it means that links in this file might
440             // not work....
441         }
442         result.append("?sequence=").append(String.valueOf(bitstream.getSequenceID()));
443         return result.toString();
444     }
