147      * Generate the unique caching key.
148      * This key must be unique inside the space of this component.

150     public Serializable getKey()
151     {
152         try 
153         {
154             String key = "";
155             
156             // Page Parameter
157             Request request = ObjectModelHelper.getRequest(objectModel);
158             key += "-" + getParameterPage();
159             key += "-" + getParameterRpp();
160             key += "-" + getParameterSortBy();
161             key += "-" + getParameterOrder();
162             key += "-" + getParameterEtAl();
163 
164             // What scope the search is at
165             DSpaceObject scope = getScope();
166             if (scope != null)
167                 key += "-" + scope.getHandle();
168             
169             // The actualy search query.
170             key += "-" + getQuery();
171 
172             return HashUtil.hash(key);
173         }
174         catch (Exception e)
175         {
176             // Ignore all errors and just don't cache.
177             return "0";
178         }
179     }

181     /**
182      * Generate the cache validity object.
183      * 
184      * This validity object should never "over cache" because it will
185      * perform the search, and serialize the results using the
186      * DSpaceValidity object.
187      */
188     public SourceValidity getValidity()
189     {
190     	if (this.validity == null)
191     	{
192 	        try
193 	        {
194 	            DSpaceValidity validity = new DSpaceValidity();
195 	            
196 	            DSpaceObject scope = getScope();
197 	            validity.add(scope);
198 	            
199 	            performSearch();
200 	            
201 	            @SuppressWarnings("unchecked") // This cast is correct
202 	            java.util.List<String> handles = queryResults.getHitHandles();
203 	            for (String handle : handles)
204 	            {
205 	                DSpaceObject resultDSO = HandleManager.resolveToObject(context, handle);
206 	                validity.add(resultDSO);
207 	            }
208 	            
209 	            this.validity = validity.complete();
210             }
211 	        catch (Exception e)
212 	        {
213 	            // Just ignore all errors and return an invalid cache.
214 	        }
215 
216             // add log message that we are viewing the item
217             // done here, as the serialization may not occur if the cache is valid
218             logSearch();
219     	}
220     	return this.validity;
221     }

230     /**
231      * 
232      * Attach a division to the given search division named "search-results"
233      * which contains results for this search query.
234      * 
235      * @param search
236      *            The search division to contain the search-results division.
237      */
238     protected void buildSearchResultsDivision(Division search)
239             throws IOException, SQLException, WingException
240     {
241         if (getQuery().length() > 0)
242         {
243 
244             // Preform the actual search
245             performSearch();
246             DSpaceObject searchScope = getScope();
247             
248             Para para = search.addPara("result-query","result-query");
249 
250             String query = getQuery();
251             int hitCount = queryResults.getHitCount();
252             para.addContent(T_result_query.parameterize(query,hitCount));
253             
254             Division results = search.addDivision("search-results","primary");
255             
256             if (searchScope instanceof Community)
257             {
258                 Community community = (Community) searchScope;
259                 String communityName = community.getMetadata("name");
260                 results.setHead(T_head1_community.parameterize(communityName));
261             }
262             else if (searchScope instanceof Collection)
263             {
264                 Collection collection = (Collection) searchScope;
265                 String collectionName = collection.getMetadata("name");
266                 results.setHead(T_head1_collection.parameterize(collectionName));
267             }
268             else
269             {
270                 results.setHead(T_head1_none);
271             }
272 
273             if (queryResults.getHitCount() > 0)
274             {
275                 // Pagination variables.
276                 int itemsTotal = queryResults.getHitCount();
277                 int firstItemIndex = queryResults.getStart() + 1;
278                 int lastItemIndex = queryResults.getStart()
279                         + queryResults.getPageSize();
280                 if (itemsTotal < lastItemIndex)
281                     lastItemIndex = itemsTotal;
282                 int currentPage = (queryResults.getStart() / queryResults
283                         .getPageSize()) + 1;
284                 int pagesTotal = ((queryResults.getHitCount() - 1) / queryResults
285                         .getPageSize()) + 1;
286                 Map<String, String> parameters = new HashMap<String, String>();
287                 parameters.put("page", "{pageNum}");
288                 String pageURLMask = generateURL(parameters);
289 
290                 results.setMaskedPagination(itemsTotal, firstItemIndex,
291                         lastItemIndex, currentPage, pagesTotal, pageURLMask);
292 
293                 // Look for any communities or collections in the mix
294                 ReferenceSet referenceSet = null;
295                 boolean resultsContainsBothContainersAndItems = false;
296                 
297                 @SuppressWarnings("unchecked") // This cast is correct
298                 java.util.List<String> containerHandles = queryResults.getHitHandles();
299                 for (String handle : containerHandles)
300                 {
301                     DSpaceObject resultDSO = HandleManager.resolveToObject(
302                             context, handle);
303 
304                     if (resultDSO instanceof Community
305                             || resultDSO instanceof Collection)
306                     {
307                         if (referenceSet == null) {
308                             referenceSet = results.addReferenceSet("search-results-repository",
309                                     ReferenceSet.TYPE_SUMMARY_LIST,null,"repository-search-results");
310                             // Set a heading showing that we will be listing containers that matched:
311                             referenceSet.setHead(T_head2);
312                             resultsContainsBothContainersAndItems = true;
313                         }
314                         referenceSet.addReference(resultDSO);
315                     }
316                 }
317                 
318                 
319                 // Look for any items in the result set.
320                 referenceSet = null;
321                 
322                 @SuppressWarnings("unchecked") // This cast is correct
323                 java.util.List<String> itemHandles = queryResults.getHitHandles();
324                 for (String handle : itemHandles)
325                 {
326                     DSpaceObject resultDSO = HandleManager.resolveToObject(
327                             context, handle);
328 
329                     if (resultDSO instanceof Item)
330                     {
331                         if (referenceSet == null) {
332                             referenceSet = results.addReferenceSet("search-results-repository",
333                                     ReferenceSet.TYPE_SUMMARY_LIST,null,"repository-search-results");
334                             // Only set a heading if there are both containers and items.
335                             if (resultsContainsBothContainersAndItems)
336                             	referenceSet.setHead(T_head3);  
337                         }
338                         referenceSet.addReference(resultDSO);
339                     }
340                 }
341                 
342             }
343             else
344             {
345                 results.addPara(T_no_results);
346             }
347         }// Empty query
348     }

350     /**
351      * Add options to the search scope field. This field determines in what
352      * communities or collections to search for the query.
353      * 
354      * The scope list will depend upon the current search scope. There are three
355      * cases:
356      * 
357      * No current scope: All top level communities are listed.
358      * 
359      * The current scope is a community: All collections contained within the
360      * community are listed.
361      * 
362      * The current scope is a collection: All parent communities are listed.
363      * 
364      * @param scope
365      *            The current scope field.
366      */
367     protected void buildScopeList(Select scope) throws SQLException,
368             WingException
369     {
370 
371         DSpaceObject scopeDSO = getScope();
372         if (scopeDSO == null)
373         {
374             // No scope, display all root level communities
375             scope.addOption("/",T_all_of_dspace);
376             scope.setOptionSelected("/");
377             for (Community community : Community.findAllTop(context))
378             {
379                 scope.addOption(community.getHandle(),community.getMetadata("name"));
380             }
381         }
382         else if (scopeDSO instanceof Community)
383         {
384             // The scope is a community, display all collections contained
385             // within
386             Community community = (Community) scopeDSO;
387             scope.addOption("/",T_all_of_dspace);
388             scope.addOption(community.getHandle(),community.getMetadata("name"));
389             scope.setOptionSelected(community.getHandle());
390 
391             for (Collection collection : community.getCollections())
392             {
393                 scope.addOption(collection.getHandle(),collection.getMetadata("name"));
394             }
395         }
396         else if (scopeDSO instanceof Collection)
397         {
398             // The scope is a collection, display all parent collections.
399             Collection collection = (Collection) scopeDSO;
400             scope.addOption("/",T_all_of_dspace);
401             scope.addOption(collection.getHandle(),collection.getMetadata("name"));
402             scope.setOptionSelected(collection.getHandle());
403             
404             Community[] communities = collection.getCommunities()[0]
405                     .getAllParents();
406             for (Community community : communities)
407             {
408                 scope.addOption(community.getHandle(),community.getMetadata("name"));
409             }
410         }
411     }

413     /**
414      * Query DSpace for a list of all items / collections / or communities that
415      * match the given search query.
416      * 
417      * @return The associated query results.
418      */
419     protected void performSearch() throws SQLException, IOException, UIException
420     {
421         if (queryResults != null)
422             return;
423         
424         Context context = ContextUtil.obtainContext(objectModel);
425         String query = getQuery();
426         DSpaceObject scope = getScope();
427         int page = getParameterPage();
428 
429         queryArgs = new QueryArgs();
430         queryArgs.setPageSize(getParameterRpp());
431         try
432         {
433             queryArgs.setSortOption(SortOption.getSortOption(getParameterSortBy()));
434         }
435         catch (SortException se)
436         {
437         }
438         
439         queryArgs.setSortOrder(getParameterOrder());
440 
441         queryArgs.setQuery(query);
442         if (page > 1)
443             queryArgs.setStart((Integer.valueOf(page) - 1) * queryArgs.getPageSize());
444         else
445             queryArgs.setStart(0);
446 
447         QueryResults qResults = null;
448         if (scope instanceof Community)
449         {
450             qResults = DSQuery.doQuery(context, queryArgs, (Community) scope);
451         }
452         else if (scope instanceof Collection)
453         {
454             qResults = DSQuery.doQuery(context, queryArgs, (Collection) scope);
455         }
456         else
457         {
458             qResults = DSQuery.doQuery(context, queryArgs);
459         }
460 
461         this.queryResults = qResults;
462     }

464     /**
465      * Determine the current scope. This may be derived from the current url
466      * handle if present or the scope parameter is given. If no scope is
467      * specified then null is returned.
468      * 
469      * @return The current scope.
470      */
471     protected DSpaceObject getScope() throws SQLException
472     {
473         Request request = ObjectModelHelper.getRequest(objectModel);
474         String scopeString = request.getParameter("scope");
475 
476         // Are we in a community or collection?
477         DSpaceObject dso;
478         if (scopeString == null || "".equals(scopeString))
479             // get the search scope from the url handle
480             dso = HandleUtil.obtainHandle(objectModel);
481         else
482             // Get the search scope from the location parameter
483             dso = HandleManager.resolveToObject(context, scopeString);
484 
485         return dso;
486     }

653     protected void logSearch()
654     {
655         int countCommunities = 0;
656         int countCollections = 0;
657         int countItems       = 0;
658 
659         for (Object type : queryResults.getHitTypes())
660         {
661             if (type instanceof Integer)
662             {
663                 switch (((Integer)type).intValue())
664                 {
665                 case Constants.ITEM:       countItems++;        break;
666                 case Constants.COLLECTION: countCollections++;  break;
667                 case Constants.COMMUNITY:  countCommunities++;  break;
668                 }
669             }
670         }
671 
672         String logInfo = "";
673 
674         try
675         {
676             DSpaceObject dsoScope = getScope();
677 
678             if (dsoScope instanceof Collection)
679             {
680                 logInfo = "collection_id=" + dsoScope.getID() + ",";
681             }
682             else if (dsoScope instanceof Community)
683             {
684                 logInfo = "community_id=" + dsoScope.getID() + ",";
685             }
686         }
687         catch (SQLException sqle)
688         {
689             // Ignore, as we are only trying to get the scope to add detail to the log message
690         }
691         
692         log.info(LogManager.getHeader(context, "search", logInfo + "query=\""
693                 + queryArgs.getQuery() + "\",results=(" + countCommunities + ","
694                 + countCollections + "," + countItems + ")"));
695     }
