90  	 * Add a new metadata schema. The ID of the new schema will be added
91  	 * as the "schemaID" parameter on the results object.
92  	 * 
93  	 * @param context The DSpace context
94  	 * @param namespace The new schema's namespace
95  	 * @param name The new schema's name.
96  	 * @return A flow result
97  	 */
98  	public static FlowResult processAddMetadataSchema(Context context, String namespace, String name) throws SQLException, AuthorizeException, NonUniqueMetadataException, UIException
99  	{
100 		FlowResult result = new FlowResult();
101 		result.setContinue(false);
102 		
103 		// Decode the namespace and name
104 		try
105         {
106             namespace = URLDecoder.decode(namespace, Constants.DEFAULT_ENCODING);
107             name = URLDecoder.decode(name,Constants.DEFAULT_ENCODING);
108         }
109         catch (UnsupportedEncodingException uee)
110         {
111             throw new UIException(uee);
112         }
113 		
114 		if (namespace == null || 
115 			namespace.length() <= 0)
116 			result.addError("namespace");
117 		if (name == null || 
118 			name.length() <= 0 ||
119 			name.indexOf('.') != -1 ||
120 			name.indexOf('_') != -1 ||
121 			name.indexOf(' ') != -1)
122 			// The name must not be empty nor contain dot, underscore or spaces.
123 			result.addError("name");
124 		
125 		
126 		if (result.getErrors() == null)
127 		{
128 			MetadataSchema schema = new MetadataSchema();
129 		    schema.setNamespace(namespace);
130 		    schema.setName(name);
131 		    schema.create(context);
132 
133 		    context.commit();
134 		    
135 		    result.setContinue(true);
136 		    result.setOutcome(true);
137 		    result.setMessage(T_add_metadata_schema_success_notice);   
138 		    result.setParameter("schemaID", schema.getSchemaID());
139 		}
140 		
141 		return result;
142 	}

146 	 * 
147 	 * @param context The DSpace context
148 	 * @param schemaIDs A list of schema IDs to be deleted.
149 	 * @return A flow result
150 	 */
151 	public static FlowResult processDeleteMetadataSchemas(Context context, String[] schemaIDs) throws SQLException, AuthorizeException, NonUniqueMetadataException
152 	{
153 		FlowResult result = new FlowResult();
154 		
155 		int count = 0;
156 		for (String id : schemaIDs) 
157     	{
158 			MetadataSchema schema = MetadataSchema.find(context, Integer.valueOf(id));
159 			
160 			// First remove and fields in the schema
161 			MetadataField[] fields = MetadataField.findAllInSchema(context, schema.getSchemaID());
162 			for (MetadataField field : fields)
163 				field.delete(context);
164 			
165 			// Once all the fields are gone, then delete the schema.
166 	        schema.delete(context);
167 	        count++;
168     	}
169 		
170 		if (count > 0)
171 		{
172 			context.commit();
173 			
174 			result.setContinue(true);
175 			result.setOutcome(true);
176 			result.setMessage(T_delete_metadata_schema_success_notice);
177 		}
178         
179 		return result;
180 	}

194 	{

197

247 	 * Edit a metadata field.
248 	 * 
249 	 * @param context The DSpace context.
250 	 * @param schemaID The ID of the schema for this field.
251 	 * @param fieldID The id of this field.
252 	 * @param element A new element value
253 	 * @param qualifier A new qualifier value
254 	 * @param note A new note value.
255 	 * @return A results object.
256 	 */
257 	public static FlowResult processEditMetadataField(Context context, int schemaID, int fieldID, String element, String qualifier, String note) throws IOException, AuthorizeException, SQLException, UIException
258 	{
259 		FlowResult result = new FlowResult();
260 		result.setContinue(false);
261 
262 		// Decode the element, qualifier, and note.
263 		try
264         {
265             element = URLDecoder.decode(element, Constants.DEFAULT_ENCODING);
266             qualifier = URLDecoder.decode(qualifier,Constants.DEFAULT_ENCODING);
267             note = URLDecoder.decode(note,Constants.DEFAULT_ENCODING);
268         }
269         catch (UnsupportedEncodingException uee)
270         {
271             throw new UIException(uee);
272         }
273 		
274 		// Check if the field name is good.
275 		result.setErrors(checkMetadataFieldName(element, qualifier));
276 		
277 		// Make sure qualifier is null if blank.
278 		if ("".equals(qualifier))
279 			qualifier = null;
280 		
281 		// Check to make sure the field is unique, sometimes the NonUniqueMetadataException is not thrown.
282 		MetadataField possibleDuplicate = MetadataField.findByElement(context, schemaID, element, qualifier);
283 		if (possibleDuplicate != null && possibleDuplicate.getFieldID() != fieldID)
284 			result.addError("duplicate_field");
285 		
286 		if (result.getErrors() == null)
287 		{	
288 			try
289 			{
290 				// Update the metadata for a DC type
291 				MetadataField field = MetadataField.find(context, fieldID);
292 				field.setElement(element);
293 				field.setQualifier(qualifier);
294 				field.setScopeNote(note);
295 				field.update(context);
296 				
297 				context.commit();
298 				
299 				result.setContinue(true);
300 				result.setOutcome(true);
301 				result.setMessage(T_edit_metadata_field_success_notice);
302 			} 
303 			catch (NonUniqueMetadataException nume)
304 			{
305 				// This shouldn't ever occure.
306 				result.addError("duplicate_field");
307 			}	
308 		}
309 		
310 		return result;
311 	}

314 	 * Simple method to check the a metadata field's name: element and qualifier.
315 	 * 
316 	 * @param element The field's element.
317 	 * @param qualifier The field's qualifier
318 	 * @return A list of errors found, null if none are found.
319 	 */
320 	private static List<String> checkMetadataFieldName(String element, String qualifier)
321 	{
322 		List<String> errors = new ArrayList<String>();
323 		
324 		
325 		// Is the element empty?
326 		if (element == null || element.length() <= 0)
327 		{
328 			element = ""; // so that the rest of the checks don't fail.
329 			errors.add("element_empty");
330 		}
331 		
332 		// Is there a bad character in the element?
333 		if (element.indexOf('.') != -1 ||
334 			element.indexOf('_') != -1 ||
335 			element.indexOf(' ') != -1)
336 			errors.add("element_badchar");
337 		
338 		// Is the element too long?
339 		if (element.length() > 64)
340 			errors.add("element_tolong");
341 		
342 
343 		// The qualifier can be empty.
344 		if (qualifier != null && qualifier.length() > 0)
345 		{
346 			if (qualifier.length() > 64)
347 				errors.add("qualifier_tolong");
348 			
349 			if (qualifier.indexOf('.') != -1 ||
350 				qualifier.indexOf('_') != -1 ||
351 				qualifier.indexOf(' ') != -1)
352 				errors.add("qualifier_badchar");
353 		}
354 		
355 		// If there were no errors then just return null.
356 		if (errors.size() == 0)
357 			return null;
358 		
359 		return errors;
360 	}

429 	 * Edit a bitstream format. If the formatID is -1 then a new format is created.
430 	 * The formatID of the new format is added as a parameter to the results object.
431 	 * 
432 	 * FIXME: the reason we accept a request object is so that we can use the 
433 	 * RequestUtils.getFieldvalues() to get the multivalue field values.
434 	 * 
435 	 * @param context The dspace context
436 	 * @param formatID The id of the format being updated.
437 	 * @param request The request object, for all the field entries.
438 	 * @return A results object
439 	 */
440 	public static FlowResult processEditBitstreamFormat(Context context, int formatID, Request request) throws SQLException, AuthorizeException
441 	{
442 		FlowResult result = new FlowResult();
443 		result.setContinue(false);
444 		
445 		// Get the values
446         String mimeType = request.getParameter("mimetype");
447         String shortDescription = request.getParameter("short_description");
448         String description = request.getParameter("description");
449         String supportLevel = request.getParameter("support_level");
450         String internal = request.getParameter("internal");
451         List<String> extensionsList = RequestUtils.getFieldValues(request, "extensions");
452         String[] extensions = extensionsList.toArray(new String[extensionsList.size()]);
453 		
454         // The format must at least have a name.
455         if (formatID != 1 && (shortDescription == null || shortDescription.length() == 0))
456         {
457         	result.addError("short_description");
458         	return result;
459         }
460         
461         // Remove leading periods from file extensions.
462         for (int i = 0; i < extensions.length; i++)
463         	if (extensions[i].startsWith("."))
464         		extensions[i] = extensions[i].substring(1);
465         
466         
467         // Get or create the format
468         BitstreamFormat format;
469 		if (formatID >= 0)
470 			format = BitstreamFormat.find(context, formatID);
471 		else
472 			format = BitstreamFormat.create(context);
473         
474 		// Update values
475 		format.setMIMEType(mimeType);
476 		if (formatID != 1) // don't change the unknow format.
477 			format.setShortDescription(shortDescription);
478 		format.setDescription(description);
479 		format.setSupportLevel(Integer.valueOf(supportLevel));
480 		if (internal == null)
481 			format.setInternal(false);
482 		else
483 			format.setInternal(true);
484 		format.setExtensions(extensions);
485 
486 		
487 		// Commit the change
488         format.update();
489         context.commit();
490 		
491 		// Return status
492         result.setContinue(true);
493 		result.setOutcome(true);
494 		result.setMessage(T_edit_bitstream_format_success_notice);
495 		result.setParameter("formatID",format.getID());
496         
497 		return result;
498 	}
