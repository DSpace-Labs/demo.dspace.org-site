150 	public void addBody(Body body) throws WingException, SQLException 
151 	{
152 		// Get all our parameters
153 		String baseURL  = contextPath +"/admin/groups?administrative-continue="+knot.getId();
154 		String query    = URLDecode(parameters.getParameter("query",""));
155 		int page        = parameters.getParameterAsInteger("page",0);
156 		int highlightID = parameters.getParameterAsInteger("highlightID",-1);
157         int resultCount = Group.searchResultCount(context, query);
158         Group[] groups  = Group.search(context, query, page*PAGE_SIZE, PAGE_SIZE);
159 		
160 		
161 		
162 		// DIVISION: groups-main
163 		Division main = body.addInteractiveDivision("groups-main",contextPath +"/admin/groups",Division.METHOD_POST,"primary administrative groups");
164 		main.setHead(T_main_head);
165 		
166 		
167 		
168 		
169 		// DIVISION: group-actions
170         Division actions = main.addDivision("group-actions");
171         actions.setHead(T_actions_head);
172         
173         // Browse Epeople
174         List actionsList = actions.addList("actions");
175         actionsList.addLabel(T_actions_create);
176         actionsList.addItemXref(baseURL+"&submit_add", T_actions_create_link);
177         actionsList.addLabel(T_actions_browse);
178         actionsList.addItemXref(baseURL+"&query&submit_search",T_actions_browse_link);
179 	    
180         actionsList.addLabel(T_actions_search);
181         org.dspace.app.xmlui.wing.element.Item actionItem = actionsList.addItem();
182         Text queryField = actionItem.addText("query");
183         if (query != null)
184         	queryField.setValue(query);
185         queryField.setHelp(T_search_help);
186         actionItem.addButton("submit_search").setValue(T_go);
187         
188         
189         
190         
191         
192         
193         
194         // DIVISION: group-search
195         Division search = main.addDivision("group-search");
196 		search.setHead(T_search_head);
197         
198 
199         if (resultCount > PAGE_SIZE) 
200 		{
201         	// If there are enough results then paginate the results
202         	int firstIndex = page*PAGE_SIZE+1; 
203         	int lastIndex = page*PAGE_SIZE + groups.length;
204        
205         	String nextURL = null, prevURL = null;
206         	if (page < (resultCount / PAGE_SIZE))
207         		nextURL = baseURL+"&page="+(page+1);
208         	if (page > 0)
209         		prevURL = baseURL+"&page="+(page-1);
210         	
211 			search.setSimplePagination(resultCount,firstIndex,lastIndex,prevURL, nextURL);
212 		}
213         
214         
215         Table table = search.addTable("groups-search-table",groups.length + 1, 1);
216         Row header = table.addRow(Row.ROLE_HEADER);
217         header.addCell().addContent(T_search_column1);
218         header.addCell().addContent(T_search_column2);
219         header.addCell().addContent(T_search_column3);
220         header.addCell().addContent(T_search_column4);
221         header.addCell().addContent(T_search_column5);
222         
223         for (Group group : groups)
224         {
225         	Row row;
226         	if (group.getID() == highlightID)
227         		row = table.addRow(null,null,"highlight");
228         	else
229         		row = table.addRow();
230         	
231         	if (group.getID() > 1)
232         	{
233 	        	CheckBox select = row.addCell().addCheckBox("select_group");
234 	        	select.setLabel(new Integer(group.getID()).toString());
235 	        	select.addOption(new Integer(group.getID()).toString());
236         	}
237         	else
238         	{
239         		// Don't allow the user to remove the administrative (id:1) or 
240         		// anonymous group (id:0) 
241         		row.addCell();
242         	}
243         	
244         	row.addCell().addContent(group.getID());
245         	row.addCell().addXref(baseURL+"&submit_edit&groupID="+group.getID(), group.getName());
246         	
247         	int memberCount = group.getMembers().length + group.getMemberGroups().length;
248         	row.addCell().addContent(memberCount == 0 ? "-" : String.valueOf(memberCount));
249         	
250         	Cell cell = row.addCell();
251         	String groupName = group.getName();
252             DSpaceObject collectionOrCommunity = null;
253             String collectionOrCommunityName = null;
254             int id;
255             id = FlowGroupUtils.getCollectionId(groupName);
256             if (id > -1)
257             {
258                 Collection collection = Collection.find(context, id);
259                 if (collection != null)
260                 {
261                     collectionOrCommunityName = collection.getMetadata("name");
262                     collectionOrCommunity = collection;
263                 }
264             } 
265             else
266             {
267                 id = FlowGroupUtils.getCommunityId(groupName);
268                 if (id > -1)
269                 {
270                     Community community = Community.find(context, id);
271                     if (community != null)
272                     {
273                         collectionOrCommunityName = community.getMetadata("name");
274                         collectionOrCommunity = community;
275                     }
276                 }
277             }
278             if (collectionOrCommunity != null)
279             {
280                 if (collectionOrCommunityName == null)
281                     collectionOrCommunityName = "";
282                 else if (collectionOrCommunityName.length() > MAX_COLLECTION_OR_COMMUNITY_NAME)
283                     collectionOrCommunityName = collectionOrCommunityName.substring(0,MAX_COLLECTION_OR_COMMUNITY_NAME-3) + "...";
284 	        		
285                 cell.addContent(collectionOrCommunityName + " ");
286 	        		
287 	        	Highlight highlight = cell.addHighlight("fade");
288 
289                 highlight.addContent("[");
290                 highlight.addXref(contextPath+"/handle/"+collectionOrCommunity.getHandle(), T_collection_link);
291 	        	highlight.addContent("]");
292         	}
293         	
294         }
295         
296         if (groups.length <= 0)
297         {
298         	Cell cell = table.addRow().addCell(1,5);
299         	cell.addHighlight("italic").addContent(T_no_results);
300         }
301         else
302         {
303         	search.addPara().addButton("submit_delete").setValue(T_submit_delete);	
304         }
305         
306         search.addHidden("administrative-continue").setValue(knot.getId());
307    }
