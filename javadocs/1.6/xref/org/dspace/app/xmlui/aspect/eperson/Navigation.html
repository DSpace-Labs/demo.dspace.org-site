68  /**
69   * Add the eperson navigation items to the document. This includes:

71   * 1) Login and Logout links

105     /**
106      * Generate the unique key.
107      * This key must be unique inside the space of this component.
108      *
109      * @return The generated key hashes the src
110      */
111     public Serializable getKey() 
112     {
113         Request request = ObjectModelHelper.getRequest(objectModel);
114         
115         // Special case, don't cache anything if the user is logging 
116         // in. The problem occures because of timming, this cache key
117         // is generated before we know whether the operation has 
118         // succeded or failed. So we don't know whether to cache this 
119         // under the user's specific cache or under the anonymous user.
120         if (request.getParameter("login_email")    != null ||
121             request.getParameter("login_password") != null ||
122             request.getParameter("login_realm")    != null )
123         {
124             return null;
125         }
126                 
127         // FIXME:
128         // Do not cache the home page. There is a bug that is causing the
129         // homepage to be cached with user's data after a logout. This
130         // polutes the cache. As a work-around this problem we just won't
131         // cache this page.
132         if (request.getSitemapURI().length() == 0)
133         {
134         	return null;
135         }
136         
137     	String key;
138         if (context.getCurrentUser() != null)
139             key = context.getCurrentUser().getEmail();
140         else
141         	key = "anonymous";
142         
143         // Add the user's language
144         Enumeration locales = request.getLocales();
145         while (locales.hasMoreElements())
146         {
147             Locale locale = (Locale) locales.nextElement();
148             key += "-" + locale.toString();    
149         }
150         
151         return HashUtil.hash(key);
152     }

154     /**
155      * Generate the validity object.
156      *
157      * @return The generated validity object or <code>null</code> if the
158      *         component is currently not cacheable.
159      */
160     public SourceValidity getValidity() 
161     {
162     	if (this.validity == null)
163     	{
164     		// Only use the DSpaceValidity object is someone is logged in.
165     		if (context.getCurrentUser() != null)
166     		{
167 		        try {
168 		            DSpaceValidity validity = new DSpaceValidity();
169 		            
170 		            validity.add(eperson);
171 		            
172 		            Group[] groups = Group.allMemberGroups(context, eperson);
173 		            for (Group group : groups)
174 		            {
175 		            	validity.add(group);
176 		            }
177 		            
178 		            this.validity = validity.complete();
179 		        } 
180 		        catch (SQLException sqle)
181 		        {
182 		            // Just ignore it and return invalid.
183 		        }
184     		}
185     		else
186     		{
187     			this.validity = NOPValidity.SHARED_INSTANCE;
188     		}
189     	}
190     	return this.validity;
191     }

193     /**
194      * Add the eperson aspect navigational options.
195      */
196     public void addOptions(Options options) throws SAXException, WingException,
197             UIException, SQLException, IOException, AuthorizeException
198     {
199     	/* Create skeleton menu structure to ensure consistent order between aspects,
200     	 * even if they are never used 
201     	 */
202         options.addList("browse");
203         List account = options.addList("account");
204         options.addList("context");
205         options.addList("administrative");
206         
207         account.setHead(T_my_account);
208         EPerson eperson = this.context.getCurrentUser();
209         if (eperson != null)
210         {
211             String fullName = eperson.getFullName();
212             account.addItemXref(contextPath+"/logout",T_logout);
213             account.addItemXref(contextPath+"/profile",T_profile.parameterize(fullName));
214         } 
215         else 
216         {
217             account.addItemXref(contextPath+"/login",T_login);
218             if (ConfigurationManager.getBooleanProperty("xmlui.user.registration", true))
219             	account.addItemXref(contextPath+"/register",T_register);
220         }
221     }

223     /**
224      * Add the user metadata
225      */
226     public void addUserMeta(UserMeta userMeta) throws SAXException,
227             WingException, UIException, SQLException, IOException,
228             AuthorizeException
229     {
230         EPerson eperson = context.getCurrentUser();
231         if (eperson != null)
232         {
233             userMeta.setAuthenticated(true);
234             userMeta.addMetadata("identifier").addContent(eperson.getID());
235             userMeta.addMetadata("identifier","email").addContent(eperson.getEmail());
236             userMeta.addMetadata("identifier","firstName").addContent(eperson.getFirstName());
237             userMeta.addMetadata("identifier","lastName").addContent(eperson.getLastName());
238             userMeta.addMetadata("identifier","logoutURL").addContent(contextPath+"/logout");
239             userMeta.addMetadata("identifier","url").addContent(contextPath+"/profile");
240         }
241         else
242         {
243             userMeta.setAuthenticated(false);
244         }
245 
246         // Allways have a login URL.
247         userMeta.addMetadata("identifier","loginURL").addContent(contextPath+"/login");
248         
249         // Allways add language information
250         Request request = ObjectModelHelper.getRequest(objectModel);
251         Enumeration locales = request.getLocales();
252         while (locales.hasMoreElements())
253         {
254             Locale locale = (Locale) locales.nextElement();
255             userMeta.addMetadata("language","RFC3066").addContent(locale.toString());    
256         }
257     }
