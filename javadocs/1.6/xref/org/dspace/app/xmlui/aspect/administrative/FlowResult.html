50  /**
51   * This class represents the results that may be generated during a flow script
52   * in this administrative section. Typicaly, some method preforms an operation
53   * and returns an object of type FlowResult, then the flow script can inspect
54   * the results object to determine what the next course of action is.
55   * 
56   * Basicaly, this results object stores all the errors and contiunation states
57   * that need to be represented. There are four types of information stored:
58   * 
59   * 1) Continuation, this is a simple boolean variable that indicates wheather
60   * the required operation is complete and the user may continue on to the next step.
61   * 
62   * 2) Notice information, this is a simple encoding of a notice message to be displayed
63   * to the user on their next step. There are four parts: outcome, header, message, and
64   * characters. See each field for more description on each part. Note: either a message
65   * or characters are required.
66   * 
67   * 3) Errors, this is a list of errors that were encountered during processing. 
68   * Typical, it just consists of a list of errored fields. However occasionaly there 
69   * may be other specialized errors listed.
70   * 
71   * 4) Parameters, this is a map of attached parameters that may be relevnt to the 
72   * result. This should be used for things such as generated id's when objects are newly
73   * created.
74   * 
75   * @author Scott Phillips
76   */
77  public class FlowResult {
78  
79  	/**
80  	 * Determine wheather the operation has been completed enough that the user
81  	 * may successufully continue on to the next step.
82  	 */
83  	private boolean continuep;
84  	
85  	/**
86  	 * Notice parameters:
87  	 * 
88  	 * Outcome: The outcome of the notice, may be either success, failure, or netural.
89  	 * 
90  	 * Header: The notice's label, an i18n dictionary key.
91  	 * 
92  	 * message: The main body of the notice, an i18n dictionary key.
93  	 * 
94  	 * characters: Supplementaiol information for the notice, plain text. This is 
95  	 * typicaly used for exepctions.
96  	 *
97  	 */
98  	private enum Outcome{ SUCCESS, FAILURE, NETURAL};
99  	private Outcome outcome = Outcome.NETURAL;
100 	private Message header;
101 	private Message message;
102 	private String characters;
103 	
104 	/**
105 	 * A list of errors encountered while processing this operation.
106 	 */
107 	private List<String> errors;
108 	
109 	
110 	/**
111 	 * Any parameters that may be attached to this result.
112 	 */
113 	private Map<String,Object> parameters;

117 	 * user should progress to the next step in the flow.
118 	 */

122 	}

124 	/**
125 	 * Determine if the user should progress to the
126 	 * next step in the flow.
127 	 * 
128 	 * @return the continuation parameter
129 	 */
130 	public boolean getContinue()
131 	{
132 		return this.continuep;
133 	}

135 	
136 	
137 	/**
138 	 * Set the notice outcome to either success or failure. Note, 
139 	 * the default outcome is netural, once an outcome is set the
140 	 * netural outcome can never be atained again.
141 	 * 
142 	 * @param success True for success, false for failure.
143 	 */
144 	public void setOutcome(boolean success)
145 	{
146 		if (success)
147 			outcome = Outcome.SUCCESS;
148 		else
149 			outcome = Outcome.FAILURE;
150 	}
151 	
152 	/**
153 	 * Get the notice outcome in string form, either success 
154 	 * or failure. If the outcome is netural then null is returned.
155 	 */
156 	public String getOutcome()
157 	{
158 		if (outcome == Outcome.SUCCESS)
159 		{
160 			return "success";
161 		}
162 		else if (outcome == Outcome.FAILURE)
163 		{
164 			return "failure";
165 		}
166 		return null;
167 	}

243 			this.errors = new ArrayList<String>();
244 		
245 		this.errors.add(newError);
246 	}

254 	}

256 	/**
257 	 * Return the list of errors in string form, i.e. a comma seperated list
258 	 * of errors. If there are no errors then null is returned.
259 	 */
260 	public String getErrorString()
261 	{
262 		if (errors == null || errors.size() == 0)
263 			return null;
264 		
265 		String errorString = null;
266 		for (String error : errors)
267 		{
268 			if (errorString == null)
269 				errorString = error;
270 			else
271 				errorString += ","+error;
272 		}
273 		return errorString;
274 	}

288

295 	 * 
296 	 * @param name The parameter's name.
297 	 * @return The parameter's value.
298 	 */
299 	public Object getParameter(String name)
300 	{
301 		if (this.parameters == null)
302 			return null;
303 		
304 		return this.parameters.get(name);
305 	}
