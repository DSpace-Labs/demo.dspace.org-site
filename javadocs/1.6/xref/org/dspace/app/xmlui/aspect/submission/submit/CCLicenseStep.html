62  /**
63   * This is an optional page of the item submission processes. The Creative 
64   * Commons license may be added to an item in addition to the standard distribution 
65   * license. This step will allow the user to go off to the creative commons website 
66   * select a license and then when returned view what license was added to the item.
67   * <P>
68   * This class is called by org.dspace.app.xmlui.submission.step.LicenseStep
69   * when the Creative Commons license is enabled
70   * <P>
71   * The form is divided into three major divisions: 1) A global div surrounds the 
72   * whole page, 2) a specific interactive div displays the button that goes off to the 
73   * creative commons website to select a license, and 3) a local division that displays 
74   * the selected license and standard action bar.
75   * 
76   * @author Scott Phillips
77   * @author Tim Donohue (updated for Configurable Submission)
78   */

110 	public void addBody(Body body) throws SAXException, WingException,
111 	UIException, SQLException, IOException, AuthorizeException
112 	{
113 		// Build the url to and from creative commons
114 		Item item = submission.getItem();
115 		Collection collection = submission.getCollection();
116 		String actionURL = contextPath + "/handle/"+collection.getHandle() + "/submit/" + knot.getId() + ".continue";
117 
118 		Request request = ObjectModelHelper.getRequest(objectModel);
119 		boolean https = request.isSecure();
120 		String server = request.getServerName();
121 		int port = request.getServerPort();
122 	
123 	    String exitURL = (https) ? "https://" : "http://";
124 	    exitURL += server;
125 	    if (! (port == 80 || port == 443))
126 	    	exitURL += ":"+port;
127 			
128 	    exitURL += actionURL + "?submission-continue="+knot.getId()+"&cc_license_url=[license_url]";
129 	
130 	    
131 	    
132 	    // Division 1:
133 	    //  Global division
134 		Division div = body.addDivision("submit-cclicense", "primary submission");
135 		div.setHead(T_submission_head);
136         
137         // Division 2:
138 		//Progress bar division
139         Division progressDiv = div.addInteractiveDivision("submit-cclicense-progress", actionURL, Division.METHOD_POST);
140         addSubmissionProgressList(progressDiv);
141         //need 'submission-continue' in order to keep current state
142         progressDiv.addHidden("submission-continue").setValue(knot.getId()); 
143         
144 		// Division 3:
145 		//  Creative commons offsite division
146 	    Division offsiteDiv = div.addInteractiveDivision("submit-cclicense-offsite", CREATIVE_COMMONS_URL, Division.METHOD_POST);
147         offsiteDiv.setHead(T_head);
148 	    offsiteDiv.addPara(T_info1);
149 		
150 	    offsiteDiv.addHidden("submission-continue").setValue(knot.getId()); 
151 	    offsiteDiv.addHidden("partner").setValue("dspace");
152 	    offsiteDiv.addHidden("exit_url").setValue(exitURL);
153 
154         String jurisdiction = ConfigurationManager.getProperty("webui.submit.cc-jurisdiction");
155         if ((jurisdiction != null) && (!"".equals(jurisdiction)))
156         {
157             offsiteDiv.addHidden("jurisdiction").setValue(jurisdiction.trim());
158         }
159 
160 	    Para ccPara = offsiteDiv.addPara("creative-commons-button","creative-commons-button");
161 	    ccPara.addButton("submit_to_creative_commons").setValue(T_submit_to_creative_commons);
162 	
163 	    // Division 4:
164 		//  Local onsite division
165 		Division onsiteDiv = div.addInteractiveDivision("submit-cclicense-offsite", actionURL, Division.METHOD_POST);
166 		List form = onsiteDiv.addList("submit-review", List.TYPE_FORM);
167 	
168 		form.addLabel(T_license);
169 		if (CreativeCommons.hasLicense(context, item))
170 		{
171 			String url = CreativeCommons.getLicenseURL(item);
172 			form.addItem().addXref(url,url);
173 			
174 			form.addItem().addButton("submit_no_cc").setValue(T_submit_remove);
175 			form.addItem().addHidden("cc_license_url").setValue(url);
176         }
177 		else
178 		{
179 			form.addItem().addHighlight("italic").addContent(T_no_license);
180 		}
181 		
182         
183 		// add standard control/paging buttons
184         addControlButtons(form);
185         
186 	}
