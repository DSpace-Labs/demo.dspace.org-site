207     /**
208      * If the user has any workflow tasks, either assigned to them or in an 
209      * available pool of tasks, then build two tables listing each of these queues.
210      * 
211      * If the user dosn't have any workflows then don't do anything.
212      * 
213      * @param division The division to add the two queues too.
214      */
215     private void addWorkflowTasks(Division division) throws SQLException, WingException
216     {
217     	@SuppressWarnings("unchecked") // This cast is correct
218     	java.util.List<WorkflowItem> ownedItems = WorkflowManager.getOwnedTasks(context, context
219                 .getCurrentUser());
220     	@SuppressWarnings("unchecked") // This cast is correct.
221     	java.util.List<WorkflowItem> pooledItems = WorkflowManager.getPooledTasks(context, context
222                 .getCurrentUser());
223     	
224     	if (!(ownedItems.size() > 0 || pooledItems.size() > 0))
225     		// No tasks, so don't show the table.
226     		return;
227     	
228     	
229     	Division workflow = division.addDivision("workflow-tasks");
230     	workflow.setHead(T_w_head1);
231     	workflow.addPara(T_w_info1);
232     	
233     	// Tasks you own
234     	Table table = workflow.addTable("workflow-tasks",ownedItems.size() + 2,5);
235         table.setHead(T_w_head2);
236         Row header = table.addRow(Row.ROLE_HEADER);
237         header.addCellContent(T_w_column1);
238         header.addCellContent(T_w_column2);
239         header.addCellContent(T_w_column3);
240         header.addCellContent(T_w_column4);
241         header.addCellContent(T_w_column5);
242     	
243         if (ownedItems.size() > 0)
244         {
245         	for (WorkflowItem owned : ownedItems)
246         	{
247         		int workflowItemID = owned.getID();
248         		String url = contextPath+"/handle/"+owned.getCollection().getHandle()+"/workflow?workflowID="+workflowItemID;
249         		DCValue[] titles = owned.getItem().getDC("title", null, Item.ANY);
250         		String collectionName = owned.getCollection().getMetadata("name");
251         		EPerson submitter = owned.getSubmitter();
252         		String submitterName = submitter.getFullName();
253         		String submitterEmail = submitter.getEmail();
254         		
255         		Message state = getWorkflowStateMessage(owned);
256 
257         		Row row = table.addRow();
258         		
259         		CheckBox remove = row.addCell().addCheckBox("workflowID");
260 	        	remove.setLabel("selected");
261 	        	remove.addOption(workflowItemID);
262         		
263         		// The task description
264 	        	row.addCell().addXref(url,state);
265 
266         		// The item description
267         		if (titles != null && titles.length > 0)
268         		{
269         			String displayTitle = titles[0].value;
270         			if (displayTitle.length() > 50)
271         				displayTitle = displayTitle.substring(0,50)+ " ...";
272         			row.addCell().addXref(url,displayTitle);
273         		}
274         		else
275         			row.addCell().addXref(url,T_untitled);
276 
277         		// Submitted too
278         		row.addCell().addXref(url,collectionName);
279 
280         		// Submitted by
281 	        	Cell cell = row.addCell();
282 	        	cell.addContent(T_email);
283 	        	cell.addXref("mailto:"+submitterEmail,submitterName);
284         	}
285         	
286         	Row row = table.addRow();
287  	    	row.addCell(0,5).addButton("submit_return_tasks").setValue(T_w_submit_return);
288         	
289         }
290         else
291         {
292         	Row row = table.addRow();
293         	row.addCell(0,5).addHighlight("italic").addContent(T_w_info2);
294         }
295        
296         
297         
298         
299         // Tasks in the pool
300         table = workflow.addTable("workflow-tasks",pooledItems.size()+2,5);
301         table.setHead(T_w_head3);
302         
303         header = table.addRow(Row.ROLE_HEADER);
304         header.addCellContent(T_w_column1);
305         header.addCellContent(T_w_column2);
306         header.addCellContent(T_w_column3);
307         header.addCellContent(T_w_column4);
308         header.addCellContent(T_w_column5);
309         
310         if (pooledItems.size() > 0)
311         {
312 
313         	for (WorkflowItem pooled : pooledItems)
314         	{
315         		int workflowItemID = pooled.getID();
316         		String url = contextPath+"/handle/"+pooled.getCollection().getHandle()+"/workflow?workflowID="+workflowItemID;
317         		DCValue[] titles = pooled.getItem().getDC("title", null, Item.ANY);
318         		String collectionName = pooled.getCollection().getMetadata("name");
319         		EPerson submitter = pooled.getSubmitter();
320         		String submitterName = submitter.getFullName();
321         		String submitterEmail = submitter.getEmail();
322 
323         		Message state = getWorkflowStateMessage(pooled);
324         		
325         		
326         		Row row = table.addRow();
327         		
328         		CheckBox remove = row.addCell().addCheckBox("workflowID");
329 	        	remove.setLabel("selected");
330 	        	remove.addOption(workflowItemID);
331         		
332         		// The task description
333 	        	row.addCell().addXref(url,state);
334 
335         		// The item description
336         		if (titles != null && titles.length > 0)
337         		{
338         			String displayTitle = titles[0].value;
339         			if (displayTitle.length() > 50)
340         				displayTitle = displayTitle.substring(0,50)+ " ...";
341         			
342         			row.addCell().addXref(url,displayTitle);
343         		}
344         		else
345         			row.addCell().addXref(url,T_untitled);
346 
347         		// Submitted too
348         		row.addCell().addXref(url,collectionName);
349 
350         		// Submitted by
351         		Cell cell = row.addCell();
352 	        	cell.addContent(T_email);
353 	        	cell.addXref("mailto:"+submitterEmail,submitterName);
354         		
355         	}
356         	Row row = table.addRow();
357  	    	row.addCell(0,5).addButton("submit_take_tasks").setValue(T_w_submit_take);
358         }
359         else
360         {
361         	Row row = table.addRow();
362         	row.addCell(0,5).addHighlight("italic").addContent(T_w_info3);
363         }
364     }

368     /**
369      * There are two options, the user has some unfinished submissions 
370      * or the user does not.
371      * 
372      * If the user does not, then we just display a simple paragraph 
373      * explaining that the user may submit new items to dspace.
374      * 
375      * If the user does have unfinisshed submissions then a table is 
376      * presented listing all the unfinished submissions that this user has.
377      * 
378      */
379     private void addUnfinishedSubmissions(Division division) throws SQLException, WingException
380     {
381     	
382         // User's WorkflowItems
383     	WorkspaceItem[] unfinishedItems = WorkspaceItem.findByEPerson(context,context.getCurrentUser());
384     	SupervisedItem[] supervisedItems = SupervisedItem.findbyEPerson(context, context.getCurrentUser());
385 
386     	if (unfinishedItems.length <= 0 && supervisedItems.length <= 0)
387     	{
388     		Collection[] collections = Collection.findAuthorized(context, null, Constants.ADD);
389     		
390     		if (collections.length > 0)
391     		{
392     			Division start = division.addDivision("start-submision");
393 	        	start.setHead(T_s_head1);
394 	        	Para p = start.addPara();
395 	        	p.addContent(T_s_info1a);
396 	        	p.addXref(contextPath+"/submit",T_s_info1b);
397 	        	p.addContent(T_s_info1c);
398 	        	return;
399     		}
400     	}
401     	
402     	Division unfinished = division.addInteractiveDivision("unfinished-submisions", contextPath+"/submit", Division.METHOD_POST);
403     	unfinished.setHead(T_s_head2);
404     	Para p = unfinished.addPara();
405     	p.addContent(T_s_info2a);
406     	p.addHighlight("bold").addXref(contextPath+"/submit",T_s_info2b);
407     	p.addContent(T_s_info2c);
408     	
409     	// Calculate the number of rows.
410     	// Each list pluss the top header and bottom row for the button.
411     	int rows = unfinishedItems.length + supervisedItems.length + 2;
412     	if (supervisedItems.length > 0 && unfinishedItems.length > 0)
413     		rows++; // Authoring heading row
414     	if (supervisedItems.length > 0)
415     		rows++; // Supervising heading row
416     	
417     	
418     	Table table = unfinished.addTable("unfinished-submissions",rows,5);
419         Row header = table.addRow(Row.ROLE_HEADER);
420         header.addCellContent(T_s_column1);
421         header.addCellContent(T_s_column2);
422         header.addCellContent(T_s_column3);
423         header.addCellContent(T_s_column4);
424         
425         if (supervisedItems.length > 0 && unfinishedItems.length > 0)
426         {
427             header = table.addRow();
428             header.addCell(null,Cell.ROLE_HEADER,0,5,null).addContent(T_s_head3);
429         }
430         
431         if (unfinishedItems.length > 0)
432         {
433 	        for (WorkspaceItem workspaceItem : unfinishedItems) 
434 	        {
435 	        	DCValue[] titles = workspaceItem.getItem().getDC("title", null, Item.ANY);
436 	        	EPerson submitterEPerson = workspaceItem.getItem().getSubmitter();
437 	        	
438 	        	int workspaceItemID = workspaceItem.getID();
439 	        	String url = contextPath+"/submit?workspaceID="+workspaceItemID;
440 	        	String submitterName = submitterEPerson.getFullName();
441 	        	String submitterEmail = submitterEPerson.getEmail();
442 	        	String collectionName = workspaceItem.getCollection().getMetadata("name");
443 	
444 	        	Row row = table.addRow(Row.ROLE_DATA);
445 	        	CheckBox remove = row.addCell().addCheckBox("workspaceID");
446 	        	remove.setLabel("remove");
447 	        	remove.addOption(workspaceItemID);
448 	        	
449 	        	if (titles.length > 0)
450 	        	{
451 	        		String displayTitle = titles[0].value;
452         			if (displayTitle.length() > 50)
453         				displayTitle = displayTitle.substring(0,50)+ " ...";
454 	        		row.addCell().addXref(url,displayTitle);
455 	        	}
456 	        	else
457 	        		row.addCell().addXref(url,T_untitled);
458 	        	row.addCell().addXref(url,collectionName);
459 	        	Cell cell = row.addCell();
460 	        	cell.addContent(T_email);
461 	        	cell.addXref("mailto:"+submitterEmail,submitterName);
462 	        }
463         } 
464         else
465         {
466         	header = table.addRow();
467         	header.addCell(0,5).addHighlight("italic").addContent(T_s_info3);
468         }
469         
470         if (supervisedItems.length > 0)
471         {
472             header = table.addRow();
473             header.addCell(null,Cell.ROLE_HEADER,0,5,null).addContent(T_s_head4);
474         }
475         
476         for (WorkspaceItem workspaceItem : supervisedItems) 
477         {
478         	
479         	DCValue[] titles = workspaceItem.getItem().getDC("title", null, Item.ANY);
480         	EPerson submitterEPerson = workspaceItem.getItem().getSubmitter();
481         	
482         	int workspaceItemID = workspaceItem.getID();
483         	String url = contextPath+"/submit?workspaceID="+workspaceItemID;
484         	String submitterName = submitterEPerson.getFullName();
485         	String submitterEmail = submitterEPerson.getEmail();
486         	String collectionName = workspaceItem.getCollection().getMetadata("name");
487         	
488         	
489         	Row row = table.addRow(Row.ROLE_DATA);
490         	CheckBox selected = row.addCell().addCheckBox("workspaceID");
491         	selected.setLabel("select");
492         	selected.addOption(workspaceItemID);
493         	
494         	if (titles.length > 0)
495         	{
496         		String displayTitle = titles[0].value;
497     			if (displayTitle.length() > 50)
498     				displayTitle = displayTitle.substring(0,50)+ " ...";
499         		row.addCell().addXref(url,displayTitle);
500         	}
501         	else
502         		row.addCell().addXref(url,T_untitled);
503         	row.addCell().addXref(url,collectionName);
504         	Cell cell = row.addCell();
505         	cell.addContent(T_email);
506         	cell.addXref("mailto:"+submitterEmail,submitterName);
507         }
508         
509         
510         header = table.addRow();
511         Cell lastCell = header.addCell(0,5);
512         if (unfinishedItems.length > 0 || supervisedItems.length > 0)
513         	lastCell.addButton("submit_submissions_remove").setValue(T_s_submit_remove);
514     }

517     /**
518      * This section lists all the submissions that this user has submitted which are currently under review.
519      * 
520      * If the user has none, this nothing is displayed.
521      */
522     private void addSubmissionsInWorkflow(Division division) throws SQLException, WingException
523     {
524     	WorkflowItem[] inprogressItems = WorkflowItem.findByEPerson(context,context.getCurrentUser());
525 
526     	// If there is nothing in progress then don't add anything.
527     	if (!(inprogressItems.length > 0))
528     			return;
529     	
530     	Division inprogress = division.addDivision("submissions-inprogress");
531     	inprogress.setHead(T_p_head1);
532     	inprogress.addPara(T_p_info1);
533     	
534     	
535     	Table table = inprogress.addTable("submissions-inprogress",inprogressItems.length+1,3);
536         Row header = table.addRow(Row.ROLE_HEADER);
537         header.addCellContent(T_p_column1);
538         header.addCellContent(T_p_column2);
539         header.addCellContent(T_p_column3);
540         
541         
542         for (WorkflowItem workflowItem : inprogressItems)
543         {
544         	DCValue[] titles = workflowItem.getItem().getDC("title", null, Item.ANY);
545         	String collectionName = workflowItem.getCollection().getMetadata("name");
546         	Message state = getWorkflowStateMessage(workflowItem);
547         	
548         	
549         	Row row = table.addRow();
550         	
551         	// Add the title column
552         	if (titles.length > 0)
553         	{
554         		String displayTitle = titles[0].value;
555     			if (displayTitle.length() > 50)
556     				displayTitle = displayTitle.substring(0,50)+ " ...";
557         		row.addCellContent(displayTitle);
558         	}
559         	else
560         		row.addCellContent(T_untitled);
561         	
562         	// Collection name column
563         	row.addCellContent(collectionName);
564         	
565         	// Status column
566         	row.addCellContent(state);
567         }
568     }

574     /**
575      * Determine the correct message that describes this workflow item's state.
576      * 
577      * FIXME: change to return type of message;
578      */
579     private Message getWorkflowStateMessage(WorkflowItem workflowItem)
580     {
581 		switch (workflowItem.getState())
582 		{
583 			case WorkflowManager.WFSTATE_SUBMIT:
584 				return T_status_0;
