130 	public void addBody(Body body) throws WingException, SQLException 
131 	{
132 		// Get our parameters & state
133 		int formatID = parameters.getParameterAsInteger("formatID",-1);
134 		BitstreamFormat format = null;
135 		
136 		if (formatID >= 0)
137 			format = BitstreamFormat.find(context,formatID);
138 	
139 		String errorString = parameters.getParameter("errors",null);
140 		ArrayList<String> errors = new ArrayList<String>();
141 		if (errorString != null)
142 			for (String error : errorString.split(","))
143 				errors.add(error);
144 		
145 		Request request = ObjectModelHelper.getRequest(objectModel);
146         String mimetypeValue = request.getParameter("mimetype");
147         String nameValue = request.getParameter("short_description");
148         String descriptionValue = request.getParameter("description");
149         String supportLevelValue = request.getParameter("support_level");
150         String internalValue = request.getParameter("internal");
151         java.util.List<String> extensionsList = RequestUtils.getFieldValues(request, "extensions");
152         String[] extensionValues = extensionsList.toArray(new String[extensionsList.size()]);
153         
154         // Remove leading periods from file extensions.
155         for (int i = 0; i < extensionValues.length; i++)
156         	if (extensionValues[i].startsWith("."))
157         		extensionValues[i] = extensionValues[i].substring(1);
158         
159         if (format != null)
160         {
161         	if (mimetypeValue == null)
162         		mimetypeValue = format.getMIMEType();
163         	if (nameValue == null)
164         		nameValue = format.getShortDescription();
165         	if (descriptionValue == null)
166         		descriptionValue = format.getDescription();
167         	if (supportLevelValue == null)
168         		supportLevelValue = String.valueOf(format.getSupportLevel());
169         	if (request.getParameter("mimetype") == null)
170         		internalValue = format.isInternal() ? "true" : null;
171         	if (request.getParameter("extensions") == null)
172         		extensionValues = format.getExtensions();
173         }
174 		
175         
176 	
177         
178         // DIVISION: edit-bitstream-format
179 		Division main = body.addInteractiveDivision("edit-bitstream-format",contextPath+"/admin/format-registry",Division.METHOD_POST,"primary administrative format-registry");
180 		if (formatID == -1)
181 			main.setHead(T_head1);
182 		else
183 			main.setHead(T_head2.parameterize(nameValue));
184 		main.addPara(T_para1);
185 	
186 		List form = main.addList("edit-bitstream-format",List.TYPE_FORM);
187 		
188 		Text name = form.addItem().addText("short_description");
189 		name.setRequired();
190 		name.setLabel(T_name);
191 		name.setHelp(T_name_help);
192 		name.setValue(nameValue);
193 		name.setSize(35);
194 		if (errors.contains("short_description"))
195 			name.addError(T_name_error);
196 		
197 		Text mimeType = form.addItem().addText("mimetype");
198 		mimeType.setLabel(T_mimetype);
199 		mimeType.setHelp(T_mimetype_help);
200 		mimeType.setValue(mimetypeValue);
201 		mimeType.setSize(35);
202 		
203 		// Do not allow anyone to change the name of the unknown format.
204 		if (format != null && format.getID() == 1)
205 			name.setDisabled(); 
206 
207 		TextArea description = form.addItem().addTextArea("description");
208 		description.setLabel(T_description);
209 		description.setValue(descriptionValue);
210 		description.setSize(3, 35);
211 		
212 		Select supportLevel = form.addItem().addSelect("support_level");
213 		supportLevel.setLabel(T_support);
214 		supportLevel.setHelp(T_support_help);
215 		supportLevel.addOption(0,T_support_0);
216 		supportLevel.addOption(1,T_support_1);
217 		supportLevel.addOption(2,T_support_2);
218 		supportLevel.setOptionSelected(supportLevelValue);
219 		
220 		CheckBox internal = form.addItem().addCheckBox("internal");
221 		internal.setLabel(T_internal);
222 		internal.setHelp(T_internal_help);
223 		internal.addOption((internalValue != null),"true");
224 		
225 		Text extensions = form.addItem().addText("extensions");
226 		extensions.setLabel(T_extensions);
227 		extensions.setHelp(T_extensions_help);
228 		extensions.enableAddOperation();
229 		extensions.enableDeleteOperation();
230 		for (String extensionValue : extensionValues)
231 		{
232 			extensions.addInstance().setValue(extensionValue);
233 		}
234 		
235 		Item actions = form.addItem();
236 		actions.addButton("submit_save").setValue(T_submit_save);
237 		actions.addButton("submit_cancel").setValue(T_submit_cancel);
238 		
239 		
240 		main.addHidden("administrative-continue").setValue(knot.getId());
241         
242    }
