58  /**
59   * Unclaim all the selected workflows. This action returns these
60   * tasks to the general pool for other users to select from.
61   * 
62   * 
63   * @author Scott Phillips
64   */
65  public class UnclaimTasksAction extends AbstractAction
66  {
67      private static final Logger log = Logger.getLogger(UnclaimTasksAction.class);
68  
69      /**
70       * @param redirector
71       *            un-used.
72       * @param resolver
73       * @param objectModel
74       *            Cocoon's object model
75       */
76      public Map act(Redirector redirector, SourceResolver resolver, Map objectModel,
77              String source, Parameters parameters) throws Exception
78      {
79          Request request = ObjectModelHelper.getRequest(objectModel);
80          Context context = ContextUtil.obtainContext(objectModel);
81          
82      	// Or the user selected a checkbox full of workflow IDs
83      	String[] workflowIDs = request.getParameterValues("workflowID");
84      	if (workflowIDs != null)
85      	{
86      		for (String workflowID : workflowIDs)
87      		{
88      			WorkflowItem workflowItem = WorkflowItem.find(context, Integer.valueOf(workflowID));
89      			//workflowItem.get
90      			
91      			int state = workflowItem.getState();
92      			// only claim tasks that are in the pool.
93      			if ( state == WorkflowManager.WFSTATE_STEP1 || 
94      				 state == WorkflowManager.WFSTATE_STEP2 || 
95      				 state == WorkflowManager.WFSTATE_STEP3 )
96      			{
97                      log.info(LogManager.getHeader(context, "unclaim_workflow", "workflow_id=" + workflowItem.getID()));
98      				WorkflowManager.unclaim(context, workflowItem, context.getCurrentUser());
99      			}
100     		}
101     		context.commit();
102     	}
103     	
104     	return null;
105     }
