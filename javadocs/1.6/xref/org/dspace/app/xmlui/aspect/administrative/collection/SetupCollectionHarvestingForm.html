131 	public void addBody(Body body) throws WingException, SQLException, AuthorizeException
132 	{
133 		int collectionID = parameters.getParameterAsInteger("collectionID", -1);
134 		Collection thisCollection = Collection.find(context, collectionID);
135 		Request request = ObjectModelHelper.getRequest(objectModel);
136 		
137 		HarvestedCollection hc = HarvestedCollection.find(context, collectionID);
138 		String baseURL = contextPath + "/admin/collection?administrative-continue=" + knot.getId();
139 		
140 		String errorString = parameters.getParameter("errors",null);
141 		String[] errors = errorString.split(",");
142 		HashMap<String,String> errorMap = new HashMap<String,String>();
143 		for (String error : errors) {
144 			//System.out.println(errorString);
145 			String[] errorPieces = error.split(":",2);
146 			
147 			if (errorPieces.length > 1)
148 				errorMap.put(errorPieces[0], errorPieces[1]);
149 			else
150 				errorMap.put(errorPieces[0], errorPieces[0]);
151 		}
152 		
153 		
154 		String oaiProviderValue;
155 		String oaiSetIdValue;
156 		String metadataFormatValue;
157 		int harvestLevelValue;
158 				
159 		if (hc != null && request.getParameter("submit_test") == null) {
160 			oaiProviderValue = hc.getOaiSource();
161 			oaiSetIdValue = hc.getOaiSetId();
162 			metadataFormatValue = hc.getHarvestMetadataConfig();
163 			harvestLevelValue = hc.getHarvestType();			
164 		}
165 		else {
166 			oaiProviderValue = parameters.getParameter("oaiProviderValue", "");
167 			oaiSetIdValue = parameters.getParameter("oaiSetIdValue", "");
168 			metadataFormatValue = parameters.getParameter("metadataFormatValue", "");
169 			String harvestLevelString = parameters.getParameter("harvestLevelValue","0");
170 			if (harvestLevelString.length() == 0)
171 				harvestLevelValue = 0;
172 			else
173 				harvestLevelValue = Integer.parseInt(harvestLevelString);
174 		}
175 		
176 		// DIVISION: main
177 	    Division main = body.addInteractiveDivision("collection-harvesting-setup",contextPath+"/admin/collection",Division.METHOD_MULTIPART,"primary administrative collection");
178 	    main.setHead(T_main_head.parameterize(thisCollection.getMetadata("name")));   
179 	    
180 	    List options = main.addList("options",List.TYPE_SIMPLE,"horizontal");
181 	    options.addItem().addXref(baseURL+"&submit_metadata",T_options_metadata);
182 	    options.addItem().addXref(baseURL+"&submit_roles",T_options_roles);
183 	    options.addItem().addHighlight("bold").addXref(baseURL+"&submit_harvesting",T_options_harvest);
184 	    
185 	    
186 	    // The top-level, all-setting, countent source radio button
187 	    List harvestSource = main.addList("harvestSource", "form");
188 	    
189 	    harvestSource.addLabel(T_label_source);
190 	    Radio source = harvestSource.addItem().addRadio("source");
191     	source.addOption(hc == null || harvestLevelValue == -1, "source_normal", T_source_normal);
192     	source.addOption(hc != null, "source_harvested", T_source_harvested);
193 	    
194 	    List settings = main.addList("harvestSettings", "form");
195 	    settings.setHead(T_main_settings_head);
196 	    
197 	    settings.addLabel(T_label_oai_provider);
198 	    Text oaiProvider = settings.addItem().addText("oai_provider");
199 	    oaiProvider.setSize(40);
200 	    oaiProvider.setValue(oaiProviderValue);
201 	    oaiProvider.setHelp(T_help_oaiurl);
202 	    
203 	    if (errorMap.containsKey(OAIHarvester.OAI_ADDRESS_ERROR)) {
204 	    	oaiProvider.addError(errorMap.get(OAIHarvester.OAI_ADDRESS_ERROR));
205 	    }
206 	    if (errorMap.containsKey("oai_provider")) {
207 	    	oaiProvider.addError(T_error_oaiurl);
208 	    	//oaiProvider.addError("You must provide a set id of the target collection.");
209 	    }
210 	    
211 	    settings.addLabel(T_label_setid);
212 	    Text oaiSetId = settings.addItem().addText("oai_setid");
213 	    oaiSetId.setSize(40);
214 	    oaiSetId.setValue(oaiSetIdValue);
215 	    oaiSetId.setHelp(T_help_oaisetid);
216 	    if (errorMap.containsKey(OAIHarvester.OAI_SET_ERROR)) {
217 	    	oaiSetId.addError(errorMap.get(OAIHarvester.OAI_SET_ERROR));
218 	    }
219 	    if (errorMap.containsKey("oai_setid")) {
220 	    	oaiSetId.addError(T_error_oaisetid);
221 	    }
222 	    
223 	    settings.addLabel(T_label_metadata_format);
224 	    Select metadataFormat = settings.addItem().addSelect("metadata_format");
225 	    if (errorMap.containsKey(OAIHarvester.OAI_ORE_ERROR)) {	    	
226 	    	metadataFormat.addError(errorMap.get(OAIHarvester.OAI_ORE_ERROR));
227 	    }
228 		if (errorMap.containsKey(OAIHarvester.OAI_DMD_ERROR)) {
229 	    	metadataFormat.addError(errorMap.get(OAIHarvester.OAI_DMD_ERROR));
230 	    }
231 	
232 	
233 	    // Add an entry for each instance of ingestion crosswalks configured for harvesting 
234         String metaString = "harvester.oai.metadataformats.";
235         Enumeration pe = ConfigurationManager.propertyNames();
236         while (pe.hasMoreElements())
237         {
238             String key = (String)pe.nextElement();
239             if (key.startsWith(metaString)) {
240             	String metadataString = ConfigurationManager.getProperty(key);
241             	String metadataKey = key.substring(metaString.length());
242             	String displayName;
243 
244             	if (metadataString.indexOf(',') != -1)
245             		displayName = metadataString.substring(metadataString.indexOf(',') + 1);
246             	else
247             		displayName = metadataKey + "(" + metadataString + ")";
248             	
249             	metadataFormat.addOption(metadataKey.equalsIgnoreCase(metadataFormatValue), metadataKey, displayName);
250             }
251         }
252         
253         
254         settings.addLabel();
255 	    Item harvestButtons = settings.addItem();
256 	    harvestButtons.addButton("submit_test").setValue(T_submit_test);
257 	    
258 	    // Various non-critical harvesting options 
259 	    //Division optionsDiv = main.addDivision("collection-harvesting-options","secondary");
260 	    //optionsDiv.setHead(T_options_head);
261 	    
262 	    List harvestOptions = main.addList("harvestOptions", "form");
263 	    harvestOptions.setHead(T_options_head);
264 	    
265 	    harvestOptions.addLabel(T_label_harvest_level);
266 	    Radio harvestLevel = harvestOptions.addItem().addRadio("harvest_level");
267 	    harvestLevel.addOption(harvestLevelValue == 1, 1, T_option_md_only);
268 	    harvestLevel.addOption(harvestLevelValue == 2, 2, T_option_md_and_ref);
269 	    harvestLevel.addOption(harvestLevelValue != 1 && harvestLevelValue != 2, 3, T_option_md_and_bs);
270 	    
271 		Para buttonList = main.addPara();
272 	    buttonList.addButton("submit_save").setValue(T_submit_save);
273 	    buttonList.addButton("submit_return").setValue(T_submit_return);
274 	    
275     	main.addHidden("administrative-continue").setValue(knot.getId());
276     }
