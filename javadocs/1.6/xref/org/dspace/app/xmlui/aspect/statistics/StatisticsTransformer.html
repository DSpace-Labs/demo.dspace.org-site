269 	 * Adds a table layout to the page
270 	 * 
271 	 * @param mainDiv
272 	 *            the div to add the table to
273 	 * @param display
274 	 * @throws SAXException
275 	 * @throws WingException
276 	 * @throws ParseException
277 	 * @throws IOException
278 	 * @throws SolrServerException
279 	 * @throws SQLException
280 	 */
281 	private void addDisplayTable(Division mainDiv, StatisticsTable display)
282 			throws SAXException, WingException, SQLException,
283 			SolrServerException, IOException, ParseException {
284 
285 		String title = display.getTitle();
286 
287 		Dataset dataset = display.getDataset();
288 
289 		if (dataset == null) {
290 			/** activate dataset query */
291 			dataset = display.getDataset(context);
292 		}
293 
294 		if (dataset != null) {
295 
296 			String[][] matrix = dataset.getMatrixFormatted();
297 
298 			/** Generate Table */
299 			Division wrapper = mainDiv.addDivision("tablewrapper");
300 			Table table = wrapper.addTable("list-table", 1, 1,
301 					title == null ? "" : "tableWithTitle");
302 			if (title != null)
303 				table.setHead(message(title));
304 
305 			/** Generate Header Row */
306 			Row headerRow = table.addRow();
307 			headerRow.addCell("spacer", Cell.ROLE_DATA, "labelcell");
308 
309 			String[] cLabels = dataset.getColLabels().toArray(new String[0]);
310 			for (int row = 0; row < cLabels.length; row++) {
311 				Cell cell = headerRow.addCell(0 + "-" + row + "-h",
312 						Cell.ROLE_DATA, "labelcell");
313 				cell.addContent(cLabels[row]);
314 			}
315 
316 			/** Generate Table Body */
317 			for (int row = 0; row < matrix.length; row++) {
318 				Row valListRow = table.addRow();
319 
320 				/** Add Row Title */
321 				valListRow.addCell("" + row, Cell.ROLE_DATA, "labelcell")
322 						.addContent(dataset.getRowLabels().get(row));
323 
324 				/** Add Rest of Row */
325 				for (int col = 0; col < matrix[row].length; col++) {
326 					Cell cell = valListRow.addCell(row + "-" + col,
327 							Cell.ROLE_DATA, "datacell");
328 					cell.addContent(matrix[row][col]);
329 				}
330 			}
331 		}

355 					title == null ? "" : "tableWithTitle");
356 			if (title != null)
357 				table.setHead(message(title));
358 
359 			Row headerRow = table.addRow();
360 
361 			headerRow.addCell("", Cell.ROLE_DATA, "labelcell");
362 			
363 			headerRow.addCell("", Cell.ROLE_DATA, "labelcell").addContent(message(T_head_visits_views));
364 
365 			/** Generate Table Body */
366 			for (int col = 0; col < matrix[0].length; col++) {
367 				Row valListRow = table.addRow();
368 
369 				Cell catCell = valListRow.addCell(col + "1", Cell.ROLE_DATA,
370 						"labelcell");
371 				catCell.addContent(dataset.getColLabels().get(col));
372 
373 				Cell valCell = valListRow.addCell(col + "2", Cell.ROLE_DATA,
374 						"datacell");
375 				valCell.addContent(matrix[0][col]);
376 
377 			}
378 
379 			if (!"".equals(display.getCss())) {
380 				List attrlist = mainDiv.addList("divattrs");
381 				attrlist.addItem("style", display.getCss());
382 			}
383 
384 		}
