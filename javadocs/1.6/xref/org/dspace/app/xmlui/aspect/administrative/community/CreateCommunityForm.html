85  	public void addPageMeta(PageMeta pageMeta) throws WingException
86      {
87          pageMeta.addMetadata("title").addContent(T_title);
88          pageMeta.addTrailLink(contextPath + "/", T_dspace_home);
89          pageMeta.addTrail().addContent(T_trail);
90      }
91  	
92  	public void addBody(Body body) throws WingException, SQLException, AuthorizeException
93  	{
94  		int communityID = parameters.getParameterAsInteger("communityID", -1);
95  		Community parentCommunity = Community.find(context, communityID);
96  		
97  		// DIVISION: main
98  	    Division main = body.addInteractiveDivision("create-community",contextPath+"/admin/community",Division.METHOD_MULTIPART,"primary administrative community");
99  	    /* Whether the parent community is null is what determines if 
100 		  we are creating a top-level community or a sub-community */
101 	    if (parentCommunity != null)
102 	    	main.setHead(T_main_head_sub.parameterize(parentCommunity.getMetadata("name")));
103 	    else
104 	    	main.setHead(T_main_head_top);
105 	        
106 	    
107 	    // The grand list of metadata options
108 	    List metadataList = main.addList("metadataList", "form");
109 	    
110 	    // community name
111 	    metadataList.addLabel(T_label_name);
112 	    Text name = metadataList.addItem().addText("name");
113 	    name.setSize(40);
114 	    
115 	    // short description
116 	    metadataList.addLabel(T_label_short_description);
117 	    Text short_description = metadataList.addItem().addText("short_description");
118 	    short_description.setSize(40);
119 	    
120 	    // introductory text
121 	    metadataList.addLabel(T_label_introductory_text);
122 	    TextArea introductory_text = metadataList.addItem().addTextArea("introductory_text");
123 	    introductory_text.setSize(6, 40);
124 	    
125 	    // copyright text
126 	    metadataList.addLabel(T_label_copyright_text);
127 	    TextArea copyright_text = metadataList.addItem().addTextArea("copyright_text");
128 	    copyright_text.setSize(6, 40);
129 	    
130 	    // legacy sidebar text; may or may not be used for news 
131 	    metadataList.addLabel(T_label_side_bar_text);
132 	    TextArea side_bar_text = metadataList.addItem().addTextArea("side_bar_text");
133 	    side_bar_text.setSize(6, 40);
134 	    
135 	    // the row to upload a new logo 
136 	    metadataList.addLabel(T_label_logo);
137 	    metadataList.addItem().addFile("logo");
138 
139 	    Para buttonList = main.addPara();
140 	    buttonList.addButton("submit_save").setValue(T_submit_save);
141 	    buttonList.addButton("submit_cancel").setValue(T_submit_cancel);
142 	    
143     	main.addHidden("administrative-continue").setValue(knot.getId());
144     }
