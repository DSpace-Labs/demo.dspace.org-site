95  	public void addPageMeta(PageMeta pageMeta) throws WingException
96      {
97          pageMeta.addMetadata("title").addContent(T_title);
98          pageMeta.addTrailLink(contextPath + "/", T_dspace_home);
99          pageMeta.addTrailLink(contextPath + "/admin/metadata-registry",T_metadata_registry_trail);
100         pageMeta.addTrail().addContent(T_trail);
101     }

103 	
104 	public void addBody(Body body) throws WingException, SQLException, AuthorizeException
105 	{
106 		// Get all our parameters
107 		MetadataSchema[] schemas = MetadataSchema.findAll(context); 
108 		String idsString = parameters.getParameter("fieldIDs", null);
109 		
110 		ArrayList<MetadataField> fields = new ArrayList<MetadataField>();
111 		for (String id : idsString.split(","))
112 		{
113 			MetadataField field = MetadataField.find(context,Integer.valueOf(id));
114 			fields.add(field);
115 		}
116  
117 		
118 		// DIVISION: metadata-field-move
119     	Division moved = body.addInteractiveDivision("metadata-field-move",contextPath+"/admin/metadata-registry",Division.METHOD_POST,"primary administrative metadata-registry");
120     	moved.setHead(T_head1);
121     	moved.addPara(T_para1);
122     	
123     	Table table = moved.addTable("metadata-field-move",fields.size() + 1, 3);
124         Row header = table.addRow(Row.ROLE_HEADER);
125         header.addCell().addContent(T_column1);
126         header.addCell().addContent(T_column2);
127         header.addCell().addContent(T_column3);
128     	
129     	for (MetadataField field : fields) 
130     	{
131     		String fieldID = String.valueOf(field.getFieldID());
132 			String fieldEelement = field.getElement();
133 			String fieldQualifier = field.getQualifier();
134 			
135 			MetadataSchema schema = MetadataSchema.find(context, field.getSchemaID());
136 			String schemaName = schema.getName();
137 			
138 			String fieldName = schemaName +"."+ fieldEelement;
139 			if (fieldQualifier != null && fieldQualifier.length() > 0)
140 				fieldName += "."+fieldQualifier;
141 				
142 			String fieldScopeNote = field.getScopeNote();
143     		
144     		Row row = table.addRow();
145     		row.addCell().addContent(fieldID);
146         	row.addCell().addContent(fieldName);
147         	row.addCell().addContent(fieldScopeNote);
148 	    }
149 
150     	Row row = table.addRow();
151     	Cell cell = row.addCell(1,3);
152     	cell.addContent(T_para2);
153     	Select toSchema = cell.addSelect("to_schema");
154     	for (MetadataSchema schema : schemas)
155     	{
156     		toSchema.addOption(schema.getSchemaID(), schema.getNamespace());
157     	}
158     	
159     	Para buttons = moved.addPara();
160     	buttons.addButton("submit_move").setValue(T_submit_move);
161     	buttons.addButton("submit_cancel").setValue(T_submit_cancel);
162     	
163     	
164     	
165     	moved.addHidden("administrative-continue").setValue(knot.getId());
166     }
