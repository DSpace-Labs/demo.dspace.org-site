222 	public void addBody(Body body) throws WingException, SQLException 
223 	{		
224 		// Find the group in question
225 		int groupID = parameters.getParameterAsInteger("groupID",-1);
226 		String currentName = URLDecode(parameters.getParameter("groupName",null));
227 		if (currentName == null || currentName.length() == 0)
228 			currentName = FlowGroupUtils.getName(context, groupID);
229 		
230 		Group group = null;
231 		if (groupID >= 0)
232 			group = Group.find(context,groupID);
233 			
234 		
235 		// Find the collection or community if applicable
236 		Collection collection = null;
237 		Community community = null;
238 		if (group != null)
239 		{
240 			int collectionID = FlowGroupUtils.getCollectionId(group.getName());
241 			if (collectionID > -1)
242 				collection = Collection.find(context, collectionID);
243 			else
244 			{
245 			    int communityID = FlowGroupUtils.getCommunityId(group.getName());
246 			    if (communityID > -1)
247 			        community = Community.find(context, communityID);
248 		    }
249 		}
250 		
251 		// Get list of member groups
252 		String memberGroupIDsString = parameters.getParameter("memberGroupIDs",null);
253 		ArrayList<Integer> memberGroupIDs = new ArrayList<Integer>();
254 		if (memberGroupIDsString != null)
255 		{
256 			for (String id : memberGroupIDsString.split(","))
257 			{
258 				if (id.length() > 0)
259 					memberGroupIDs.add(Integer.valueOf(id));
260 			}
261 		}
262 	    
263 		// Get list of member epeople
264 		String memberEPeopleIDsString = parameters.getParameter("memberEPeopleIDs",null);
265 		ArrayList<Integer> memberEPeopleIDs = new ArrayList<Integer>();
266 		if (memberEPeopleIDsString != null)
267 		{
268 			for (String id : memberEPeopleIDsString.split(","))
269 				if (id.length() > 0)
270 					memberEPeopleIDs.add(Integer.valueOf(id));
271 		}
272 		
273 		// Get highlight parameters
274 		int highlightEPersonID = parameters.getParameterAsInteger("highlightEPersonID",-1);
275 		int highlightGroupID = parameters.getParameterAsInteger("highlightGroupID",-1);
276 		
277 		// Get search parameters
278 	    String query = URLDecode(parameters.getParameter("query",null));
279 	    int page     = parameters.getParameterAsInteger("page",0);
280 	    String type  = parameters.getParameter("type",null);
281 	    
282 	    // Get any errors
283 	    String errorString = parameters.getParameter("errors",null);
284 		ArrayList<String> errors = new ArrayList<String>();
285 		if (errorString != null)
286 			for (String error : errorString.split(","))
287 				errors.add(error);
288 	    
289     
290 	    
291 	    //DIVISION: group-edit
292 	    Division main = body.addInteractiveDivision("group-edit",contextPath+"/admin/groups",Division.METHOD_POST,"primary administrative groups");
293 	    if (group == null)
294 		    main.setHead(T_main_head_new);
295 	    else
296 	    	main.setHead(T_main_head.parameterize(group.getName(),groupID));
297 	 
298 	    
299 	    if(collection != null)
300 	    {
301 	    	Para para = main.addPara();
302 	    	para.addContent(T_collection_para);
303 	    	para.addXref(contextPath+"/handle/"+collection.getHandle(), collection.getMetadata("name"));
304 	    }
305 	    else if(community != null)
306         {
307             Para para = main.addPara();
308             para.addContent(T_community_para);
309             para.addXref(contextPath+"/handle/"+community.getHandle(), community.getMetadata("name"));
310         }
311 	   
312 	    // DIVISION: group-actions
313 	    Division actions = main.addDivision("group-edit-actions");
314 	    Para groupName = actions.addPara();
315         groupName.addContent(T_label_name);
316         Text groupText = groupName.addText("group_name");
317         groupText.setValue(currentName);
318         if(collection != null || community != null)
319         {
320         	// If this group is associated with a collection or community then it is special,
321         	// thus they shouldn't be able to update it.
322         	groupText.setDisabled();
323         	groupText.setHelp(T_label_instructions);
324         }
325         else if (errors.contains("group_name") || errors.contains("group_name_duplicate"))
326         	groupText.addError("");
327         
328 
329         Para searchBoxes = actions.addPara();
330         searchBoxes.addContent(T_label_search);
331         Text queryField = searchBoxes.addText("query");
332         queryField.setValue(query);
333         queryField.setSize(15);
334         searchBoxes.addButton("submit_search_epeople").setValue(T_submit_search_people);
335         searchBoxes.addButton("submit_search_groups").setValue(T_submit_search_groups);
336 		
337         
338         if (query != null)
339         {
340 	        if ("eperson".equals(type))
341 	        {
342 	        	searchBoxes.addButton("submit_clear").setValue(T_submit_clear);
343 	        	addEPeopleSearch(main,query,page,group,memberEPeopleIDs);
344 	        }
345 	        else if ("group".equals(type))
346 	        {
347 	        	searchBoxes.addButton("submit_clear").setValue(T_submit_clear);
348         		addGroupSearch(main,group,query,page,group,memberGroupIDs);
349 	        }
350         }
351                      
352         boolean changes = false;
353         if (group != null)
354         	changes = addMemberList(main,group,memberGroupIDs,memberEPeopleIDs,highlightEPersonID,highlightGroupID);
355         
356                
357     	Para buttons = main.addPara();
358     	buttons.addButton("submit_save").setValue(T_submit_save);
359     	buttons.addButton("submit_cancel").setValue(T_submit_cancel);
360        
361     	if (changes)
362     		main.addPara().addHighlight("warn").addContent(T_pending_warn);
363     	
364         
365         main.addHidden("administrative-continue").setValue(knot.getId());
366 	}

370 	/**
371 	 * Search for epeople to add to this group.
372 	 */
373 	private void addEPeopleSearch(Division div, String query, int page, Group group, ArrayList<Integer> memberEPeopleIDs) throws SQLException, WingException
374 	{
375 		int resultCount = EPerson.searchResultCount(context, query);
376         EPerson[] epeople = EPerson.search(context, query, page*RESULTS_PER_PAGE, RESULTS_PER_PAGE);
377 		
378 		Division results = div.addDivision("results");
379 		
380 		if (resultCount > RESULTS_PER_PAGE) 
381 		{
382         	// If there are enough results then paginate the results
383         	String baseURL = contextPath +"/admin/groups?administrative-continue="+knot.getId();
384         	int firstIndex = page*RESULTS_PER_PAGE+1; 
385         	int lastIndex = page*RESULTS_PER_PAGE + epeople.length;
386        
387         	String nextURL = null, prevURL = null;
388         	if (page < (resultCount / RESULTS_PER_PAGE))
389         		nextURL = baseURL+"&page="+(page+1);
390         	if (page > 0)
391         		prevURL = baseURL+"&page="+(page-1);
392         	
393 			results.setSimplePagination(resultCount,firstIndex,lastIndex,prevURL, nextURL);
394 		}
395 		
396 		/* Set up a table with search results (if there are any). */
397 		Table table = results.addTable("group-edit-search-eperson",epeople.length + 1, 1);
398 		Row header = table.addRow(Row.ROLE_HEADER);
399 		header.addCell().addContent(T_epeople_column1);
400 		header.addCell().addContent(T_epeople_column2);
401 		header.addCell().addContent(T_epeople_column3);
402 		header.addCell().addContent(T_epeople_column4);
403 
404 		for (EPerson person : epeople)
405 		{
406 			String epersonID = String.valueOf(person.getID());
407 			String fullName = person.getFullName();
408 			String email = person.getEmail();
409 			String url = contextPath+"/admin/epeople?administrative-continue="+knot.getId()+"&submit_edit_eperson&epersonID="+epersonID;
410 			
411 			
412 			
413 			Row personData = table.addRow();
414 
415 			personData.addCell().addContent(person.getID());
416 			personData.addCell().addXref(url, fullName);
417 			personData.addCell().addXref(url, email);
418 			
419 			// check if they are allready a member of the group
420 			if (memberEPeopleIDs.contains(person.getID()))
421 			{
422 				// Check if they really members or just pending members
423 				if (group != null && group.isMember(person))
424 					personData.addCellContent(T_member);
425 				else
426 					personData.addCell().addHighlight("warn").addContent(T_pending);
427 			}
428 			else
429 			{
430 				personData.addCell().addButton("submit_add_eperson_"+epersonID).setValue(T_submit_add);
431 			}
432 		}
433 
434 		if (epeople.length <= 0) {
435 			table.addRow().addCell(1, 4).addContent(T_no_results);
436 		}
437 	}

443 	/**
444 	 * Search for groups to add to this group.
445 	 */
446 	private void addGroupSearch(Division div, Group sourceGroup, String query, int page, Group parent, ArrayList<Integer> memberGroupIDs) throws WingException, SQLException
447 	{
448 		int resultCount = Group.searchResultCount(context, query);
449         Group[] groups = Group.search(context, query, page*RESULTS_PER_PAGE, RESULTS_PER_PAGE);
450 		
451 		Division results = div.addDivision("results");
452 		
453 		if (resultCount > RESULTS_PER_PAGE) 
454 		{
455         	// If there are enough results then paginate the results
456         	String baseURL = contextPath +"/admin/groups?administrative-continue="+knot.getId();
457         	int firstIndex = page*RESULTS_PER_PAGE+1; 
458         	int lastIndex = page*RESULTS_PER_PAGE + groups.length;
459        
460         	String nextURL = null, prevURL = null;
461         	if (page < (resultCount / RESULTS_PER_PAGE))
462         		nextURL = baseURL+"&page="+(page+1);
463         	if (page > 0)
464         		prevURL = baseURL+"&page="+(page-1);
465         	
466 			results.setSimplePagination(resultCount,firstIndex,lastIndex,prevURL, nextURL);
467 		}
468 		
469         Table table = results.addTable("roup-edit-search-group",groups.length + 1, 1);
470         Row header = table.addRow(Row.ROLE_HEADER);
471         header.addCell().addContent(T_groups_column1);
472         header.addCell().addContent(T_groups_column2);
473         header.addCell().addContent(T_groups_column3);
474         header.addCell().addContent(T_groups_column4);
475         header.addCell().addContent(T_groups_column5);  
476         
477         for (Group group : groups)
478         {
479         	String groupID = String.valueOf(group.getID());
480         	String name = group.getName();
481         	String url = contextPath+"/admin/groups?administrative-continue="+knot.getId()+"&submit_edit_group&groupID="+groupID;
482         	int memberCount = group.getMembers().length + group.getMemberGroups().length;
483         	
484         	Row row = table.addRow();
485 	        	
486         	row.addCell().addContent(groupID);
487         	if (AuthorizeManager.isAdmin(context))
488         		// Only administrators can edit other groups.
489         		row.addCell().addXref(url,name);
490         	else
491         		row.addCell().addContent(name);
492         	
493         	
494         	
495         	row.addCell().addContent(memberCount == 0 ? "-" : String.valueOf(memberCount));
496         	
497         	Cell cell = row.addCell();
498         	if (FlowGroupUtils.getCollectionId(group.getName()) > -1)
499         	{
500         		Collection collection = Collection.find(context, FlowGroupUtils.getCollectionId(group.getName()) );
501         		if (collection != null)
502         		{
503 	        		String collectionName = collection.getMetadata("name");
504 	        		
505 	        		if (collectionName == null)
506 	        			collectionName = "";
507 	        		else if (collectionName.length() > MAX_COLLECTION_NAME)
508 	        			collectionName = collectionName.substring(0,MAX_COLLECTION_NAME-3) + "...";
509 	        		
510 	        		cell.addContent(collectionName+" ");
511 	        		
512 	        		Highlight highlight = cell.addHighlight("fade");
513 	        		highlight.addContent("[");
514 	        		highlight.addXref(contextPath+"/handle/"+collection.getHandle(), T_groups_collection_link);
515 	        		highlight.addContent("]");
516         		}
517         	}
518         	
519         	
520         	// Check if the group is allready a member or would create a cycle.
521         	if (memberGroupIDs.contains(group.getID()))
522         	{
523         		// Check if they really members or just pending members
524 				if (parent != null && parent.isMember(group))
525 					row.addCellContent(T_member);
526 				else
527 					row.addCell().addHighlight("warn").addContent(T_pending);
528         	}
529         	else if (isDescendant(sourceGroup, group, memberGroupIDs)) 
530         	{
531         		row.addCellContent(T_cycle);
532 	        }
533 	        else 
534 	        {
535 	        	row.addCell().addButton("submit_add_group_"+groupID).setValue(T_submit_add);
536 	        }
537             
538         }
539         if (groups.length <= 0) {
540 			table.addRow().addCell(1, 4).addContent(T_no_results);
541 		}
542 	}

549 	/**
550 	 * Method to extensively check whether the first group has the second group as a distant
551 	 * parent. This is used to avoid creating cycles like A->B, B->C, C->D, D->A which leads 
552 	 * all the groups involved to essentially include themselves.  
553 	 */
554 	private boolean isDescendant(Group descendant, Group ancestor, ArrayList<Integer> memberGroupIDs) throws SQLException 
555 	{
556 		Queue<Group> toVisit = new LinkedList<Group>();
557 		Group currentGroup;
558 		
559 		toVisit.offer(ancestor);
560 		
561 		// Initialize by adding a list of our current list of group members.
562 		for (Integer groupid : memberGroupIDs)
563 		{
564 			Group member = Group.find(context,groupid);
565 			toVisit.offer(member);
566 		}
567 		
568 		while (!toVisit.isEmpty()) {
569 			// 1. Grab a group from the queue
570 			currentGroup = toVisit.poll();
571 			
572 			// 2. See if it's the descendant we're looking for
573 			if (currentGroup.equals(descendant)) return true;
574 			
575 			// 3. If not, add that group's children to the queue
576 			for (Group nextBatch : currentGroup.getMemberGroups()) {
577 				toVisit.offer(nextBatch);
578 			}
579 		}
580 		return false;
581 	}
