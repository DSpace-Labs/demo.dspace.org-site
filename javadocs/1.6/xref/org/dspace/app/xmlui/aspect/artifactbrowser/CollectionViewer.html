137     public Serializable getKey()
138     {
139         try
140         {
141             DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
142 
143             if (dso == null)
144                 return "0";
145                 
146             return HashUtil.hash(dso.getHandle());
147         }
148         catch (SQLException sqle)
149         {
150             // Ignore all errors and just return that the component is not
151             // cachable.
152             return "0";
153         }
154     }

156     /**
157      * Generate the cache validity object.
158      * 
159      * The validity object will include the collection being viewed and 
160      * all recently submitted items. This does not include the community / collection
161      * hierarch, when this changes they will not be reflected in the cache.
162      */
163     public SourceValidity getValidity()
164     {
165     	if (this.validity == null)
166     	{
167             Collection collection = null;
168 	        try
169 	        {
170 	            DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
171 	
172 	            if (dso == null)
173 	                return null;
174 	
175 	            if (!(dso instanceof Collection))
176 	                return null;
177 	
178 	            collection = (Collection) dso;
179 	
180 	            DSpaceValidity validity = new DSpaceValidity();
181 	            
182 	            // Add the actual collection;
183 	            validity.add(collection);
184 	
185 	            // add reciently submitted items
186 	            for(BrowseItem item : getRecientlySubmittedIems(collection))
187 	            {
188 	                validity.add(item);
189 	            }
190 	            
191 	            this.validity = validity.complete();
192 	        }
193 	        catch (Exception e)
194 	        {
195 	            // Just ignore all errors and return an invalid cache.
196 	        }
197 
198     	}
199     	return this.validity;
200     }

204      * Add a page title and trail links.
205      */
206     public void addPageMeta(PageMeta pageMeta) throws SAXException,
207             WingException, UIException, SQLException, IOException,
208             AuthorizeException
209     {
210         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
211         if (!(dso instanceof Collection))
212             return;
213 
214         Collection collection = (Collection) dso;
215 
216         // Set the page title
217         String name = collection.getMetadata("name");
218         if (name == null || name.length() == 0)
219         	pageMeta.addMetadata("title").addContent(T_untitled);
220         else
221         	pageMeta.addMetadata("title").addContent(name);
222 
223         pageMeta.addTrailLink(contextPath + "/",T_dspace_home);
224         HandleUtil.buildHandleTrail(collection,pageMeta,contextPath);
225         
226         // Add RSS links if available
227         String formats = ConfigurationManager.getProperty("webui.feed.formats");
228 		if ( formats != null )
229 		{
230 			for (String format : formats.split(","))
231 			{
232 				// Remove the protocol number, i.e. just list 'rss' or' atom'
233 				String[] parts = format.split("_");
234 				if (parts.length < 1) 
235 					continue;
236 				
237 				String feedFormat = parts[0].trim()+"+xml";
238 					
239 				String feedURL = contextPath+"/feed/"+format.trim()+"/"+collection.getHandle();
240 				pageMeta.addMetadata("feed", feedFormat).addContent(feedURL);
241 			}
242 		}

245     /**
246      * Display a single collection
247      */
248     public void addBody(Body body) throws SAXException, WingException,
249             UIException, SQLException, IOException, AuthorizeException
250     {
251         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
252         if (!(dso instanceof Collection))
253             return;
254 
255         // Set up the major variables
256         Collection collection = (Collection) dso;
257 
258         // Build the collection viewer division.
259         Division home = body.addDivision("collection-home", "primary repository collection");
260         String name = collection.getMetadata("name");
261         if (name == null || name.length() == 0)
262         	home.setHead(T_untitled);
263         else
264         	home.setHead(name);
265 
266         // The search / browse box.
267         {
268             Division search = home.addDivision("collection-search-browse",
269                     "secondary search-browse");
270 
271             // Search query
272             Division query = search.addInteractiveDivision("collection-search",
273                     contextPath + "/handle/" + collection.getHandle() + "/search", 
274                     Division.METHOD_POST, "secondary search");
275             
276             Para para = query.addPara("search-query", null);
277             para.addContent(T_full_text_search);
278             para.addContent(" ");
279             para.addText("query");
280             para.addContent(" ");
281             para.addButton("submit").setValue(T_go);
282             query.addPara().addXref(contextPath + "/handle/" + collection.getHandle()+ "/advanced-search", T_advanced_search_link);
283             
284             // Browse by list
285             Division browseDiv = search.addDivision("collection-browse","secondary browse");
286             List browse = browseDiv.addList("collection-browse", List.TYPE_SIMPLE,
287                     "collection-browse");
288             browse.setHead(T_head_browse);
289             String url = contextPath + "/handle/" + collection.getHandle();
290             browse.addItemXref(url + "/browse?type=title",T_browse_titles);
291             browse.addItemXref(url + "/browse?type=author",T_browse_authors);
292             browse.addItemXref(url + "/browse?type=dateissued",T_browse_dates);
293         }
294 
295         // Add the reference
296         {
297         	Division viewer = home.addDivision("collection-view","secondary");
298             ReferenceSet mainInclude = viewer.addReferenceSet("collection-view",
299                     ReferenceSet.TYPE_DETAIL_VIEW);
300             mainInclude.addReference(collection);
301         }
302 
303         // Recently submitted items
304         {
305             java.util.List<BrowseItem> items = getRecientlySubmittedIems(collection);
306 
307             Division lastSubmittedDiv = home
308                     .addDivision("collection-recent-submission","secondary recent-submission");
309             lastSubmittedDiv.setHead(T_head_recent_submissions);
310             ReferenceSet lastSubmitted = lastSubmittedDiv.addReferenceSet(
311                     "collection-last-submitted", ReferenceSet.TYPE_SUMMARY_LIST,
312                     null, "recent-submissions");
313             for (BrowseItem item : items)
314             {
315                 lastSubmitted.addReference(item);
316             }
317         }
318         
319         
320         
321     }

323     /**
324      * Get the recently submitted items for the given collection.
325      * 
326      * @param collection The collection.
327      */
328     @SuppressWarnings("unchecked") // The cast from getLastSubmitted is correct, it dose infact return a list of Items.
329     private java.util.List<BrowseItem> getRecientlySubmittedIems(Collection collection) 
330         throws SQLException
331     {
332         if (recentSubmissionItems != null)
333             return recentSubmissionItems;
334         
335         String source = ConfigurationManager.getProperty("recent.submissions.sort-option");
336         BrowserScope scope = new BrowserScope(context);
337         scope.setCollection(collection);
338         scope.setResultsPerPage(RECENT_SUBMISISONS);
339         
340         // FIXME Exception Handling
341         try
342         {
343         	scope.setBrowseIndex(BrowseIndex.getItemBrowseIndex());
344             for (SortOption so : SortOption.getSortOptions())
345             {
346                 if (so.getName().equals(source))
347                 {
348                     scope.setSortBy(so.getNumber());
349                     scope.setOrder(SortOption.DESCENDING);
350                 }
351             }
352 
353         	BrowseEngine be = new BrowseEngine(context);
354         	this.recentSubmissionItems = be.browse(scope).getResults();
355         }
356         catch (SortException se)
357         {
358             log.error("Caught SortException", se);
359         }
360         catch (BrowseException bex)
361         {
362         	log.error("Caught BrowseException", bex);
363         }
364         
365         return this.recentSubmissionItems;
366     }
