98  	public void addBody(Body body) throws SAXException, WingException, SQLException
99  	{
100 		// Get our parameters and state;
101 		int collectionID = parameters.getParameterAsInteger("collectionID",-1);
102 		Collection collection = Collection.find(context,collectionID);
103 		
104 		ArrayList<Item> items = getMappedItems(collection);
105 		
106 		// DIVISION: browse-items
107 		Division div = body.addInteractiveDivision("browse-items",contextPath + "/admin/mapper", Division.METHOD_GET,"primary administrative mapper");
108 		div.setHead(T_head1);
109 		
110 		if (AuthorizeManager.authorizeActionBoolean(context, collection, Constants.REMOVE))
111 		{
112 			Para actions = div.addPara();
113 			actions.addButton("submit_unmap").setValue(T_submit_unmap);
114 			actions.addButton("submit_return").setValue(T_submit_return);
115 		}
116 		else
117 		{
118 			Para actions = div.addPara();
119 			Button button = actions.addButton("submit_unmap");
120 			button.setValue(T_submit_unmap);
121 			button.setDisabled();
122 			actions.addButton("submit_return").setValue(T_submit_return);
123 			
124 			div.addPara().addHighlight("fade").addContent(T_no_remove);
125 		}
126 		
127 		Table table = div.addTable("browse-items-table",1,1);
128 		
129 		Row header = table.addRow(Row.ROLE_HEADER);
130 		header.addCellContent(T_column1);
131 		header.addCellContent(T_column2);
132 		header.addCellContent(T_column3);
133 		header.addCellContent(T_column4);
134 		
135 		for (Item item : items)
136 		{
137 			String itemID = String.valueOf(item.getID());
138 			Collection owningCollection = item.getOwningCollection();
139 			String owning = owningCollection.getMetadata("name");
140 			String author = "unkown";
141 			DCValue[] dcAuthors = item.getDC("contributor",Item.ANY,Item.ANY);
142 			if (dcAuthors != null && dcAuthors.length >= 1)
143 				author = dcAuthors[0].value;
144 			
145 			String title = "untitled";
146 			DCValue[] dcTitles = item.getDC("title",null,Item.ANY);
147 			if (dcTitles != null && dcTitles.length >= 1)
148 				title = dcTitles[0].value;
149 
150 			String url = contextPath+"/handle/"+item.getHandle();
151 			
152 			Row row = table.addRow();
153 			
154 			CheckBox select = row.addCell().addCheckBox("itemID");
155 			select.setLabel("Select");
156 			select.addOption(itemID);
157 			
158 			row.addCellContent(owning);
159 			row.addCell().addXref(url,author);
160 			row.addCell().addXref(url,title);
161 		}
162 		
163 		if (AuthorizeManager.authorizeActionBoolean(context, collection, Constants.REMOVE))
164 		{
165 			Para actions = div.addPara();
166 			actions.addButton("submit_unmap").setValue(T_submit_unmap);
167 			actions.addButton("submit_return").setValue(T_submit_return);
168 		}
169 		else
170 		{
171 			Para actions = div.addPara();
172 			Button button = actions.addButton("submit_unmap");
173 			button.setValue(T_submit_unmap);
174 			button.setDisabled();
175 			actions.addButton("submit_return").setValue(T_submit_return);
176 			
177 			div.addPara().addHighlight("fade").addContent(T_no_remove);
178 		}
179 		
180 		
181 		
182 		div.addHidden("administrative-continue").setValue(knot.getId());
183 	}

186 	/**
187 	 * Get a list of all items that are mapped from other collections.
188 	 * 
189 	 * @param collection The collection to look in.
190 	 */
191 	private ArrayList<Item> getMappedItems(Collection collection) throws SQLException
192 	{
193 
194 		ArrayList<Item> items = new ArrayList<Item>();
195 		
196 		// get all items from that collection
197         ItemIterator iterator = collection.getItems();
198         try
199         {
200             while (iterator.hasNext())
201             {
202                 Item item = iterator.next();
203 
204                 if (! item.isOwningCollection(collection))
205                     items.add(item);
206             }
207         }
208         finally
209         {
210             if (iterator != null)
211                 iterator.close();
212         }
213         
214         return items;
215 	}
