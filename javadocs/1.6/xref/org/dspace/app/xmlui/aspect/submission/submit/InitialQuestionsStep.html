59  /**
60   * This is the first official step of the item submission processes. This
61   * step will ask the user two questions which will direct whether future
62   * questions will be asked of the user. Since they user may move next or
63   * previous or even jump around between the stages some metadata values
64   * may all ready be filled for selected values. i.e. if the user selected
65   * that there may be multiple titles and then later comes back and unchecks
66   * then multiple titles box. In this case these metadata entries are removed
67   * from the item, a note is displayed informing the user of this discrepency.
68   * 
69   * Questions:
70   *  - Multiple titles
71   *  - Published Before
72   * 
73   * @author Scott Phillips
74   * @author Tim Donohue (updated for Configurable Submission)
75   */

120     public void addBody(Body body) throws SAXException, WingException,
121             UIException, SQLException, IOException, AuthorizeException
122     {
123     	// Get any metadata that may be removed by unselecting one of these options.
124     	Item item = submission.getItem();
125 		Collection collection = submission.getCollection();
126 		String actionURL = contextPath + "/handle/"+collection.getHandle() + "/submit/" + knot.getId() + ".continue";
127 
128 		DCValue[] titles = item.getDC("title", "alternative", Item.ANY);
129 		
130 		DCValue[] dateIssued = item.getDC("date", "issued", Item.ANY);
131         DCValue[] citation = item.getDC("identifier", "citation", Item.ANY);
132         DCValue[] publisher = item.getDC("publisher", null, Item.ANY);
133     	
134     	
135         // Generate a from asking the user two questions: multiple 
136         // titles & published before.
137     	Division div = body.addInteractiveDivision("submit-initial-questions", actionURL, Division.METHOD_POST, "primary submission");
138     	div.setHead(T_submission_head);
139     	addSubmissionProgressList(div);
140     	
141     	List form = div.addList("submit-initial-questions", List.TYPE_FORM);
142         form.setHead(T_head);    
143         
144         CheckBox multipleTitles = form.addItem().addCheckBox("multiple_titles");
145         multipleTitles.setLabel(T_multiple_titles);
146         multipleTitles.setHelp(T_multiple_titles_help);
147         multipleTitles.addOption("true");
148         if (submission.hasMultipleTitles())
149         {
150         	multipleTitles.setOptionSelected("true");
151         }
152         
153         // If any titles would be removed if the user unselected this box then 
154         // warn the user!
155         if (titles.length > 0)
156         {
157 	        org.dspace.app.xmlui.wing.element.Item note = form.addItem();
158 	        note.addHighlight("bold").addContent(T_important_note);
159 	        note.addContent(T_multiple_titles_note);
160 	        for (int i=0; i< titles.length; i++)
161 	        {
162 	        	if (i > 0)
163 	        		note.addContent(T_separator);
164 	        	note.addContent("\"");
165 	        	note.addHighlight("bold").addContent(titles[i].value);
166 	        	note.addContent("\"");
167 	        }
168         }
169         
170         CheckBox publishedBefore = form.addItem().addCheckBox("published_before");
171         publishedBefore.setLabel(T_published_before);
172         publishedBefore.setHelp(T_published_before_help);
173         publishedBefore.addOption("true");
174         if (submission.isPublishedBefore())
175         {
176         	publishedBefore.setOptionSelected("true");
177         }
178         
179         // If any metadata would be removed if the user unselected the box then warn 
180         // the user about what will be removed.
181         if (dateIssued.length > 0 || citation.length > 0 || publisher.length > 0)
182         {
183         	org.dspace.app.xmlui.wing.element.Item note = form.addItem();
184 	        note.addHighlight("bold").addContent(T_important_note);
185 	        note.addContent(T_published_before_note);
186 	        
187 	        // Start a convoluted processes to build an english list of values.
188 	        // Format: Date Issued (value, value, value), Citation (value, value, value), Publisher (value, value, value)
189 	        if (dateIssued.length > 0)
190 	        {
191 	        	note.addHighlight("bold").addContent(T_date_issued);
192 	        	note.addContent(T_open);
193 	        	for (int i=0; i< dateIssued.length; i++)
194 	        	{
195 	        		if (i > 0)
196 		        		note.addContent(T_separator);
197 	        		note.addContent(dateIssued[i].value);
198 	        	}
199 	        	note.addContent(T_close);
200 	        }
201 	        
202 	        // Conjunction
203 	        if (dateIssued.length > 0 && (citation.length > 0 || publisher.length > 0))
204 	        	note.addContent(T_separator);
205 	        
206 	        if (dateIssued.length > 0 && citation.length > 0 && publisher.length == 0)
207 	        {
208 	        	note.addContent(T_and);
209 	        }
210 	        
211 	        // Citation
212 	        if (citation.length > 0)
213 	        {
214 	        	note.addHighlight("bold").addContent(T_citation);
215 	        	note.addContent(T_open);
216 	        	for (int i=0; i< citation.length; i++)
217 	        	{
218 	        		if (i > 0)
219 		        		note.addContent(T_separator);
220 	        		note.addContent(citation[i].value);
221 	        	}
222 	        	note.addContent(T_close);
223 	        }
224 	        
225 	        
226 	        // Conjunction
227 	        if (citation.length > 0 && publisher.length > 0)
228 	        {
229 	        	note.addContent(T_separator);
230 	        }
231 	        
232 	        if ((dateIssued.length > 0 || citation.length > 0) && publisher.length > 0)
233 	        {
234 	        	note.addContent(T_and);
235 	        }
236 	        
237 	        
238 	        // Publisher
239 	        if (publisher.length > 0)
240 	        {
241 	        	note.addHighlight("bold").addContent(T_publisher);
242 	        	note.addContent(T_open);
243 	        	for (int i=0; i< publisher.length; i++)
244 	        	{
245 	        		if (i > 0)
246 		        		note.addContent(T_separator);
247 	        		note.addContent(publisher[i].value);
248 	        	}
249 	        	note.addContent(T_close);
250 	        }
251         }
252         
253         //add standard control/paging buttons
254         addControlButtons(form);
255         
256         //Since we already warn users about the metadata pruning to happen
257         //if they uncheck an already checked box, then
258         //we can let the prune process occur immediately!
259         div.addHidden("prune").setValue("true");
260         
261     }

263     /** 
264      * Each submission step must define its own information to be reviewed
265      * during the final Review/Verify Step in the submission process.
266      * <P>
267      * The information to review should be tacked onto the passed in 
268      * List object.
269      * <P>
270      * NOTE: To remain consistent across all Steps, you should first
271      * add a sub-List object (with this step's name as the heading),
272      * by using a call to reviewList.addList().   This sublist is
273      * the list you return from this method!
274      * 
275      * @param reviewList
276      *      The List to which all reviewable information should be added
277      * @return 
278      *      The new sub-List object created by this step, which contains
279      *      all the reviewable information.  If this step has nothing to
280      *      review, then return null!   
281      */
282     public List addReviewSection(List reviewList) throws SAXException,
283         WingException, UIException, SQLException, IOException,
284         AuthorizeException
285     {
286         //Create a new section for this Initial Questions information
287         List initSection = reviewList.addList("submit-review-" + this.stepAndPage, List.TYPE_FORM);
288         initSection.setHead(T_head);
289         
290         //add information to review
291         Message multipleTitles = ReviewStep.T_no;
292         if (submission.hasMultipleTitles())
293             multipleTitles = ReviewStep.T_yes;
294     
295         Message publishedBefore = ReviewStep.T_no;
296         if (submission.isPublishedBefore())
297             publishedBefore = ReviewStep.T_yes;
298         
299         initSection.addLabel(T_multiple_titles);
300         initSection.addItem(multipleTitles);
301         initSection.addLabel(T_published_before);
302         initSection.addItem(publishedBefore);
303         
304         //return this new review section
305         return initSection;
306     }
