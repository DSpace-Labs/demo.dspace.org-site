162     public Serializable getKey()
163     {
164         try
165         {
166             BrowseParams params = getUserParams();
167 
168             String key = params.getKey();
169 
170             if (key != null)
171             {
172                 DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
173                 if (dso != null)
174                     key += "-" + dso.getHandle();
175 
176                 return HashUtil.hash(key);
177             }
178         }
179         catch (Exception e)
180         {
181             // Ignore all errors and just don't cache.
182         }
183 
184         return "0";
185     }

187     public SourceValidity getValidity()
188     {
189         if (validity == null)
190         {
191             try
192             {
193                 DSpaceValidity validity = new DSpaceValidity();
194                 DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
195 
196                 if (dso != null)
197                     validity.add(dso);
198 
199                 BrowseInfo info = getBrowseInfo();
200 
201                 // Are we browsing items, or unique metadata?
202                 if (isItemBrowse(info))
203                 {
204                     // Add the browse items to the validity
205                     for (BrowseItem item : (java.util.List<BrowseItem>) info.getResults())
206                     {
207                         validity.add(item);
208                     }
209                 }
210                 else
211                 {
212                     // Add the metadata to the validity
213                     for (String[] singleEntry : browseInfo.getStringResults())
214                     {
215                         validity.add(singleEntry[0]+"#"+singleEntry[1]);
216                     }
217                 }
218             }
219             catch (Exception e)
220             {
221                 // Just ignore all errors and return an invalid cache.
222             }
223 
224         }
225 
226         return this.validity;
227     }

251     /**
252      * Add the browse-title division.
253      */
254     public void addBody(Body body) throws SAXException, WingException, UIException, SQLException,
255             IOException, AuthorizeException
256     {
257         BrowseParams params = getUserParams();
258         BrowseInfo info = getBrowseInfo();
259 
260         String type = "withdrawn";
261 
262         // Build the DRI Body
263         Division div = body.addDivision("browse-by-" + type, "primary");
264 
265         div.setHead(getTitleMessage(info));
266 
267         // Build the internal navigation (jump lists)
268         addBrowseJumpNavigation(div, info, params);
269 
270         // Build the sort and display controls
271         addBrowseControls(div, info, params);
272 
273         // This div will hold the browsing results
274         Division results = div.addDivision("browse-by-" + type + "-results", "primary");
275 
276         // Add the pagination
277         if (info.getTotal() <= 0)
278         {
279             results.setSimplePagination(0, 0, 0, null, null);
280 
281         }
282         else
283         {
284             results.setSimplePagination(info.getTotal(), browseInfo.getOverallPosition() + 1,
285                     browseInfo.getOverallPosition() + browseInfo.getResultCount(), getPreviousPageURL(
286                             params, info), getNextPageURL(params, info));
287         }
288 
289         // Reference all the browsed items
290         ReferenceSet referenceSet = results.addReferenceSet("browse-by-" + type,
291                 ReferenceSet.TYPE_SUMMARY_LIST, type, null);
292 
293         // Are we browsing items, or unique metadata?
294         if (isItemBrowse(info))
295         {
296             // Add the items to the browse results
297             for (BrowseItem item : (java.util.List<BrowseItem>) info.getResults())
298             {
299                 referenceSet.addReference(item);
300             }
301         }
302         else    // browsing a list of unique metadata entries
303         {
304             // Create a table for the results
305             Table singleTable = results.addTable("browse-by-" + type + "-results",
306                     browseInfo.getResultCount() + 1, 1);
307 
308             // Add the column heading
309             singleTable.addRow(Row.ROLE_HEADER).addCell().addContent(
310                     message("xmlui.ArtifactBrowser.ConfigurableBrowse." + type + ".column_heading"));
311 
312             // Iterate each result
313             for (String[] singleEntry : browseInfo.getStringResults())
314             {
315                 // Create a Map of the query parameters for the link
316                 Map<String, String> queryParams = new HashMap<String, String>();
317                 queryParams.put(BrowseParams.TYPE, URLEncode(type));
318                 if (singleEntry[1] != null)
319                 {
320                     queryParams.put(BrowseParams.FILTER_VALUE[1], URLEncode(
321                         singleEntry[1]));
322                 }
323                 else
324                 {
325                     queryParams.put(BrowseParams.FILTER_VALUE[0], URLEncode(
326                         singleEntry[0]));
327                 }
328                 // Create an entry in the table, and a linked entry
329                 Cell cell = singleTable.addRow().addCell();
330                 cell.addXref(super.generateURL(WITHDRAWN_URL_BASE, queryParams), singleEntry[0]);
331             }
332         }
333     }

348     /**
349      * Makes the jump-list navigation for the results
350      *
351      * @param div
352      * @param info
353      * @param params
354      * @throws WingException
355      */
356     private void addBrowseJumpNavigation(Division div, BrowseInfo info, BrowseParams params)
357             throws WingException
358     {
359         // Get the name of the index
360         String type = info.getBrowseIndex().getName();
361 
362         // Prepare a Map of query parameters required for all links
363         Map<String, String> queryParams = new HashMap<String, String>();
364         queryParams.putAll(params.getCommonParameters());
365         queryParams.putAll(params.getControlParameters());
366 
367         // Navigation aid (really this is a poor version of pagination)
368         Division jump = div.addInteractiveDivision("browse-navigation", WITHDRAWN_URL_BASE,
369                 Division.METHOD_POST, "secondary navigation");
370 
371         // Add all the query parameters as hidden fields on the form
372         for (String key : queryParams.keySet())
373             jump.addHidden(key).setValue(queryParams.get(key));
374 
375         // If this is a date based browse, render the date navigation
376         if (isSortedByDate(info))
377         {
378             Para jumpForm = jump.addPara();
379 
380             // Create a select list to choose a month
381             jumpForm.addContent(T_jump_select);
382             Select month = jumpForm.addSelect(BrowseParams.MONTH);
383             month.addOption(false, "-1", T_choose_month);
384             for (int i = 1; i <= 12; i++)
385             {
386                 month.addOption(false, String.valueOf(i), DCDate.getMonthName(i, Locale
387                         .getDefault()));
388             }
389 
390             // Create a select list to choose a year
391             Select year = jumpForm.addSelect(BrowseParams.YEAR);
392             year.addOption(false, "-1", T_choose_year);
393             int currentYear = DCDate.getCurrent().getYear();
394             int i = currentYear;
395 
396             // Calculate where to move from 1, 5 to 10 year jumps
397             int oneYearBreak = ((currentYear - ONE_YEAR_LIMIT) / 5) * 5;
398             int fiveYearBreak = ((currentYear - FIVE_YEAR_LIMIT) / 10) * 10;
399             int tenYearBreak = (currentYear - TEN_YEAR_LIMIT);
400             do
401             {
402                 year.addOption(false, String.valueOf(i), String.valueOf(i));
403 
404                 if (i <= fiveYearBreak)
405                     i -= 10;
406                 else if (i <= oneYearBreak)
407                     i -= 5;
408                 else
409                     i--;
410             }
411             while (i > tenYearBreak);
412 
413             // Create a free text entry box for the year
414             jumpForm = jump.addPara();
415             jumpForm.addContent(T_jump_year);
416             jumpForm.addText(BrowseParams.STARTS_WITH).setHelp(T_jump_year_help);
417 
418             jumpForm.addButton("submit").setValue(T_go);
419         }
420         else
421         {
422             // Create a clickable list of the alphabet
423             List jumpList = jump.addList("jump-list", List.TYPE_SIMPLE, "alphabet");
424             for (char c = 'A'; c <= 'Z'; c++)
425             {
426                 Map<String, String> cQuery = new HashMap<String, String>(queryParams);
427                 cQuery.put(BrowseParams.STARTS_WITH, Character.toString(c));
428                 jumpList.addItemXref(super.generateURL(WITHDRAWN_URL_BASE, cQuery), Character
429                         .toString(c));
430             }
431 
432             // Create a free text field for the initial characters
433             Para jumpForm = jump.addPara();
434             jumpForm.addContent(T_starts_with);
435             jumpForm.addText(BrowseParams.STARTS_WITH).setHelp(T_starts_with_help);
436 
437             jumpForm.addButton("submit").setValue(T_go);
438         }
439     }

528      * The URL query string of of the previous page.
529      *
530      * Note: the query string does not start with a "?" or "&" those need to be
531      * added as appropriate by the caller.
532      */
533     private String getPreviousPageURL(BrowseParams params, BrowseInfo info) throws SQLException,
534             UIException
535     {
536         // Don't create a previous page link if this is the first page
537         if (info.isFirst())
538             return null;
539 
540         Map<String, String> parameters = new HashMap<String, String>();
541         parameters.putAll(params.getCommonParameters());
542         parameters.putAll(params.getControlParameters());
543 
544         if (info.hasPrevPage())
545         {
546             parameters.put(BrowseParams.OFFSET, URLEncode(String.valueOf(info.getPrevOffset())));
547         }
548 
549         return super.generateURL(WITHDRAWN_URL_BASE, parameters);
550 
551     }

578     /**
579      * Get the query parameters supplied to the browse.
580      *
581      * @return
582      * @throws SQLException
583      * @throws UIException
584      */
585     private BrowseParams getUserParams() throws SQLException, UIException
586     {
587         if (this.userParams != null)
588             return this.userParams;
589 
590         Context context = ContextUtil.obtainContext(objectModel);
591         Request request = ObjectModelHelper.getRequest(objectModel);
592 
593         BrowseParams params = new BrowseParams();
594 
595         params.month = request.getParameter(BrowseParams.MONTH);
596         params.year = request.getParameter(BrowseParams.YEAR);
597         params.etAl = RequestUtils.getIntParameter(request, BrowseParams.ETAL);
598 
599         params.scope = new BrowserScope(context);
600 
601         // Are we in a community or collection?
602         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
603         if (dso instanceof Community)
604             params.scope.setCommunity((Community) dso);
605         if (dso instanceof Collection)
606             params.scope.setCollection((Collection) dso);
607 
608         try
609         {
610             int    sortBy = RequestUtils.getIntParameter(request, BrowseParams.SORT_BY);
611 
612             BrowseIndex bi = BrowseIndex.getWithdrawnBrowseIndex();
613 
614             // If we don't have a sort column
615             if (sortBy == -1)
616             {
617                 // Get the default one
618                 SortOption so = bi.getSortOption();
619                 if (so != null)
620                 {
621                     sortBy = so.getNumber();
622                 }
623             }
624 
625             params.scope.setBrowseIndex(bi);
626             params.scope.setSortBy(sortBy);
627 
628             params.scope.setJumpToItem(RequestUtils.getIntParameter(request, BrowseParams.JUMPTO_ITEM));
629             params.scope.setOrder(request.getParameter(BrowseParams.ORDER));
630             params.scope.setOffset(RequestUtils.getIntParameter(request, BrowseParams.OFFSET));
631             params.scope.setResultsPerPage(RequestUtils.getIntParameter(request,
632                     BrowseParams.RESULTS_PER_PAGE));
633             params.scope.setStartsWith(request.getParameter(BrowseParams.STARTS_WITH));
634             String filterValue = request.getParameter(BrowseParams.FILTER_VALUE[0]);
635             if (filterValue == null)
636             {
637                 filterValue = request.getParameter(BrowseParams.FILTER_VALUE[1]);
638             }
639             else
640             {
641                 params.scope.setAuthorityValue(filterValue);
642             }
643             params.scope.setFilterValue(filterValue);
644             params.scope.setJumpToValue(request.getParameter(BrowseParams.JUMPTO_VALUE));
645             params.scope.setJumpToValueLang(request.getParameter(BrowseParams.JUMPTO_VALUE_LANG));
646             params.scope.setFilterValueLang(request.getParameter(BrowseParams.FILTER_VALUE_LANG));
647 
648             // Filtering to a value implies this is a second level browse
649             if (params.scope.getFilterValue() != null)
650                 params.scope.setBrowseLevel(1);
651 
652             // if year and perhaps month have been selected, we translate these
653             // into "startsWith"
654             // if startsWith has already been defined then it is overwritten
655             if (params.year != null && !"".equals(params.year) && !"-1".equals(params.year))
656             {
657                 String startsWith = params.year;
658                 if ((params.month != null) && !"-1".equals(params.month)
659                         && !"".equals(params.month))
660                 {
661                     // subtract 1 from the month, so the match works
662                     // appropriately
663                     if ("ASC".equals(params.scope.getOrder()))
664                     {
665                         params.month = Integer.toString((Integer.parseInt(params.month) - 1));
666                     }
667 
668                     // They've selected a month as well
669                     if (params.month.length() == 1)
670                     {
671                         // Ensure double-digit month number
672                         params.month = "0" + params.month;
673                     }
674 
675                     startsWith = params.year + "-" + params.month;
676 
677                     if ("ASC".equals(params.scope.getOrder()))
678                     {
679                         startsWith = startsWith + "-32";
680                     }
681                 }
682 
683                 params.scope.setStartsWith(startsWith);
684             }
685         }
686         catch (BrowseException bex)
687         {
688             throw new UIException("Unable to create browse parameters", bex);
689         }
690 
691         this.userParams = params;
692         return params;
693     }

699      * @return
700      * @throws SQLException
701      * @throws UIException
702      */
703     private BrowseInfo getBrowseInfo() throws SQLException, UIException
704     {
705         if (this.browseInfo != null)
706             return this.browseInfo;
707

709

711         // (this includes a browse scope)
712         BrowseParams params = getUserParams();

716             // Create a new browse engine, and perform the browse
717             BrowseEngine be = new BrowseEngine(context);
718             this.browseInfo = be.browse(params.scope);
719 
720             // figure out the setting for author list truncation

736             else

739                 this.browseInfo.setEtAl(params.etAl);
