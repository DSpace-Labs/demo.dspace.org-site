124 	 /**
125      * Generate the unique cache key.
126      *
127      * @return The generated key hashes the src
128      */
129     public Serializable getKey() 
130     {
131         Request request = ObjectModelHelper.getRequest(objectModel);
132         
133         // Special case, don't cache anything if the user is logging 
134         // in. The problem occures because of timming, this cache key
135         // is generated before we know whether the operation has 
136         // succeded or failed. So we don't know whether to cache this 
137         // under the user's specific cache or under the anonymous user.
138         if (request.getParameter("login_email")    != null ||
139             request.getParameter("login_password") != null ||
140             request.getParameter("login_realm")    != null )
141         {
142             return "0";
143         }
144                 
145         String key;
146         if (context.getCurrentUser() != null)
147         {
148         	key = context.getCurrentUser().getEmail();
149         	if(availableExports!=null && availableExports.size()>0){
150         		for(String fileName:availableExports){
151         			key+= ":"+fileName;
152         		}
153         	}
154         }
155         else
156         	key = "anonymous";
157         
158         return HashUtil.hash(key);
159     }

161     /**
162      * Generate the validity object.
163      *
164      * @return The generated validity object or <code>null</code> if the
165      *         component is currently not cacheable.
166      */
167     public SourceValidity getValidity() 
168     {
169     	if (this.validity == null)
170     	{
171     		// Only use the DSpaceValidity object is someone is logged in.
172     		if (context.getCurrentUser() != null)
173     		{
174 		        try {
175 		            DSpaceValidity validity = new DSpaceValidity();
176 		            
177 		            validity.add(eperson);
178 		            
179 		            Group[] groups = Group.allMemberGroups(context, eperson);
180 		            for (Group group : groups)
181 		            {
182 		            	validity.add(group);
183 		            }
184 		            
185 		            this.validity = validity.complete();
186 		        } 
187 		        catch (SQLException sqle)
188 		        {
189 		            // Just ignore it and return invalid.
190 		        }
191     		}
192     		else
193     		{
194     			this.validity = NOPValidity.SHARED_INSTANCE;
195     		}
196     	}
197     	return this.validity;
198     }

214     public void addOptions(Options options) throws SAXException, WingException,
215             UIException, SQLException, IOException, AuthorizeException
216     {
217     	/* Create skeleton menu structure to ensure consistent order between aspects,
218     	 * even if they are never used 
219     	 */
220         options.addList("browse");
221         List account = options.addList("account");
222         List context = options.addList("context");
223         List admin = options.addList("administrative");
224         account.setHead(T_my_account);	        
225         
226         // My Account options
227         if(availableExports!=null && availableExports.size()>0){
228             account.addItem().addXref(contextPath+"/admin/export", T_account_export);
229         }
230 
231         //Check if a system administrator
232         boolean isSystemAdmin = AuthorizeManager.isAdmin(this.context);
233 
234         // Context Administrative options
235         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
236     	if (dso instanceof Item)
237     	{
238     		Item item = (Item) dso;
239     		if (item.canEdit())
240     		{
241                     context.setHead(T_context_head);
242                     context.addItem().addXref(contextPath+"/admin/item?itemID="+item.getID(), T_context_edit_item);
243                     if (AuthorizeManager.isAdmin(this.context, dso))
244                     {
245                         context.addItem().addXref(contextPath+"/admin/export?itemID="+item.getID(), T_context_export_item );
246                         context.addItem().addXref(contextPath+ "/csv/handle/"+dso.getHandle(),T_context_export_metadata );
247                     }
248                 }
249     	}
250     	else if (dso instanceof Collection)
251     	{
252     		Collection collection = (Collection) dso;
253     		
254     		// can they admin this collection?
255             if (collection.canEditBoolean(true))
256             {
257             	context.setHead(T_context_head);
258             	context.addItemXref(contextPath+"/admin/collection?collectionID=" + collection.getID(), T_context_edit_collection);            	
259             	context.addItemXref(contextPath+"/admin/mapper?collectionID="+collection.getID(), T_context_item_mapper); 
260             	if (AuthorizeManager.isAdmin(this.context, dso))
261                 {
262                     context.addItem().addXref(contextPath+"/admin/export?collectionID="+collection.getID(), T_context_export_collection );
263                     context.addItem().addXref(contextPath+ "/csv/handle/"+dso.getHandle(),T_context_export_metadata );
264                 }
265             }
266     	}
267     	else if (dso instanceof Community)
268     	{
269     		Community community = (Community) dso;
270     		
271     		// can they admin this collection?
272             if (community.canEditBoolean())
273             {
274             	context.setHead(T_context_head);
275             	context.addItemXref(contextPath+"/admin/community?communityID=" + community.getID(), T_context_edit_community); 
276             	if (AuthorizeManager.isAdmin(this.context, dso))
277                     context.addItem().addXref(contextPath+"/admin/export?communityID="+community.getID(), T_context_export_community );
278                     context.addItem().addXref(contextPath+ "/csv/handle/"+dso.getHandle(),T_context_export_metadata );
279             }
280             
281             // can they add to this community?
282             if (AuthorizeManager.authorizeActionBoolean(this.context, community,Constants.ADD))
283             {
284             	context.setHead(T_context_head);
285             	context.addItemXref(contextPath+"/admin/collection?createNew&communityID=" + community.getID(), T_context_create_collection);
286                 context.addItemXref(contextPath+"/admin/community?createNew&communityID=" + community.getID(), T_context_create_subcommunity);      
287             }
288     	}
289     	
290     	if ("community-list".equals(this.sitemapURI))
291     	{
292             // Only System administrators can create top-level communities
293     		if (isSystemAdmin)
294             {
295             	context.setHead(T_context_head);
296     			context.addItemXref(contextPath+"/admin/community?createNew", T_context_create_community);    			
297             }
298     	}
299         
300         
301         // System Administrator options!
302         if (isSystemAdmin)
303         {
304 	        admin.setHead(T_administrative_head);
305 	                
306 	        List epeople = admin.addList("epeople");
307 	        List registries = admin.addList("registries");
308 	        
309 	        epeople.setHead(T_administrative_access_control);	        
310 	        epeople.addItemXref(contextPath+"/admin/epeople", T_administrative_people);	        
311 	        epeople.addItemXref(contextPath+"/admin/groups", T_administrative_groups);	        
312 	        epeople.addItemXref(contextPath+"/admin/authorize", T_administrative_authorizations);	        
313 	        
314 	        registries.setHead(T_administrative_registries);	        
315 	        registries.addItemXref(contextPath+"/admin/metadata-registry",T_administrative_metadata);	        
316 	        registries.addItemXref(contextPath+"/admin/format-registry",T_administrative_format);	        
317 	        
318 	        admin.addItemXref(contextPath+"/admin/item", T_administrative_items);
319             admin.addItemXref(contextPath+"/admin/withdrawn", T_administrative_withdrawn);	        
320 	        admin.addItemXref(contextPath+"/admin/panel", T_administrative_control_panel);
321             admin.addItemXref(contextPath+"/statistics", T_statistics);
322             admin.addItemXref(contextPath+ "/admin/metadataimport", T_administrative_import_metadata);
323         }
324     }

327     public int addContextualOptions(List context) throws SQLException, WingException
328     {
329     	// How many options were added.
330     	int options = 0;
331     	
332     	DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
333     	
334     	if (dso instanceof Item)
335     	{
336     		Item item = (Item) dso;
337     		if (item.canEdit())
338     		{   			
339     			context.addItem().addXref(contextPath+"/admin/item?itemID="+item.getID(), T_context_edit_item);
340     			options++;
341     		}
342     	}
343     	else if (dso instanceof Collection)
344     	{
345     		Collection collection = (Collection) dso;
346     		
347     		
348     		// can they admin this collection?
349             if (AuthorizeManager.authorizeActionBoolean(this.context, collection, Constants.ADMIN))
350             {            	
351             	context.addItemXref(contextPath+"/admin/collection?collectionID=" + collection.getID(), T_context_edit_collection);
352             	context.addItemXref(contextPath+"/admin/mapper?collectionID="+collection.getID(), T_context_item_mapper);            	
353             	options++;
354             }
355     	}
356     	else if (dso instanceof Community)
357     	{
358     		Community community = (Community) dso;
359     		
360     		// can they admin this collection?
361             if (community.canEditBoolean())
362             {           	
363             	context.addItemXref(contextPath+"/admin/community?communityID=" + community.getID(), T_context_edit_community);
364             	options++;
365             }
366             
367             // can they add to this community?
368             if (AuthorizeManager.authorizeActionBoolean(this.context, community,Constants.ADD))
369             {        	
370             	context.addItemXref(contextPath+"/admin/collection?createNew&communityID=" + community.getID(), T_context_create_collection);         	
371             	context.addItemXref(contextPath+"/admin/community?createNew&communityID=" + community.getID(), T_context_create_subcommunity);
372             	options++;
373             }
374     	}
375     	
376     	if ("community-list".equals(this.sitemapURI))
377     	{
378     		if (AuthorizeManager.isAdmin(this.context))
379             {
380     			context.addItemXref(contextPath+"/admin/community?createNew", T_context_create_community);
381     			options++;
382             }
383     	}
384     	
385     	return options;
386     }
