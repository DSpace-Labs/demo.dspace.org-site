63  /**
64   * 
65   * This action simply records pipeline events that it sees, keeping track of the users and
66   * pages they are viewing. Later the control panel's activity viewer can access this data to
67   * get a realtime snap shot of current activity of the repository.
68   * 
69   * @author Scott Phillips
70   */

75  	/** The maximum number of events that are recorded */
76  	public static int MAX_EVENTS = 250;
77  	
78  	/** The HTTP header that contains the real IP for this request, this is used when DSpace is placed behind a load balancer */
79  	public static String IP_HEADER = "X-Forwarded-For";
80  	
81  	/** The ordered list of events, by access time */
82  	private static Queue<Event> events = new LinkedList<Event>();
83  	
84  	/** record events that are from anynmous users */
85  	private static boolean recordAnonymousEvents = true;
86  	
87  	/** record events from automatic bots */
88  	private static boolean recordBotEvents = false;
89  	
90  	/**
91  	 * Allow the DSpace.cfg to override our activity max and ip header.
92  	 */
93  	static {
94  		// If the dspace.cfg has a max event count then use it.
95  		if (ConfigurationManager.getProperty("xmlui.controlpanel.activity.max") != null)
96  			MAX_EVENTS = ConfigurationManager.getIntProperty("xmlui.controlpanel.activity.max");
97  		
98  		if (ConfigurationManager.getProperty("xmlui.controlpanel.activity.ipheader") != null)
99  			IP_HEADER = ConfigurationManager.getProperty("xmlui.controlpanel.activity.ipheader");
100 	}

103     /**
104      * Record this current event.
105      */
106     public Map act(Redirector redirector, SourceResolver resolver, Map objectModel,
107             String source, Parameters parameters) throws Exception
108     {
109         Request request = ObjectModelHelper.getRequest(objectModel);
110         Context context = ContextUtil.obtainContext(objectModel);
111         
112         // Ensure only one thread is manipulating the events queue at a time.
113         synchronized (events) {
114 	        // Create and store our events
115 	        Event event = new Event(context,request);
116 	        
117 	        // Check if we should record the event
118 	        boolean record = true;
119 	        if (!recordAnonymousEvents && event.isAnonymous())
120 	        	record = false;
121 	        if (!recordBotEvents && event.isBot())
122 	        	record = false;
123 	        
124 	        if (record)
125 	        	events.add(event);
126 	        
127 	        // Remove the oldest element from the list if we are over our max
128 	        // number of elements.
129 	        while(events.size() > MAX_EVENTS)
130 	        	events.poll();
131         }
132        
133         return null;
134     }

136     /**
137      * @return a list of all current events.
138      */
139     public static List<Event> getEvents()
140     {
141     	List<Event> list = new ArrayList<Event>();
142     	// Make sure only one thread is manipulating the events queue at a time.
143     	synchronized (events) {
144 	    	list.addAll(events);
145     	}
146     	return list;
147     }

166     /**
167      * An object that represents an activity event.
168      */
169     public static class Event
170     {
171     	/** The Servlet session */
172     	private String sessionID;
173     	
174     	/** The eperson ID */
175     	private int epersonID = -1;
176     	
177     	/** The url being viewed */
178     	private String url;
179     	
180     	/** A timestap of this event */
181     	private long timestamp;
182     	
183     	/** The user-agent for the request */
184     	private String userAgent;
185     	
186     	/** The ip address of the requester */
187     	private String ip;
188     	
189     	/**
190     	 * Construct a new activity event, grabing various bits of data about the request from the context and request.
191     	 */
192     	public Event(Context context, Request request)
193     	{
194     		if (context != null)
195     		{
196     			EPerson eperson = context.getCurrentUser();
197     			if (eperson != null)
198     				epersonID = eperson.getID();
199     		}
200     		
201     		if (request != null)
202     		{
203     			url = request.getSitemapURI();
204     			HttpSession session = request.getSession(true);
205     			if (session != null)
206     				sessionID = session.getId();
207     			
208     			userAgent = request.getHeader("User-Agent");
209     			
210     			ip = request.getHeader(IP_HEADER);
211     			if (ip == null)
212     				ip = request.getRemoteAddr();
213     		}
214     		
215     		// The current time
216     		timestamp = System.currentTimeMillis();
217     	}

251     	 * Is this event anonymous?
252     	 * @return
253     	 */
254     	public boolean isAnonymous()
255     	{
256     		if (epersonID == -1)
257     			return true;
258     		else
259     			return false;
260     	}
261     	
262     	/**
263     	 * Is this event from a bot?
264     	 * @return
265     	 */
266     	public boolean isBot()
267     	{
268     		if (userAgent == null)
269     			return false;
270     		String ua = userAgent.toLowerCase();
271     		
272     		if (ua.contains("google/") || 
273     			ua.contains("msnbot/") ||
274     			ua.contains("googlebot/") || 
275     			ua.contains("webcrawler/") ||
276     			ua.contains("inktomi") ||
277     			ua.contains("teoma") ||
278     			ua.contains("bot"))
279     			return true;
280     		else
281     			return false;
282     	}
