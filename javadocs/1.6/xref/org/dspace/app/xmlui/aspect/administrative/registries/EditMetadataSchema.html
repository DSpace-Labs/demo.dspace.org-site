148 	public void addBody(Body body) throws WingException, SQLException 
149 	{
150 		// Get our parameters & state
151 		int schemaID = parameters.getParameterAsInteger("schemaID",-1);
152 		int updateID = parameters.getParameterAsInteger("updateID",-1);
153 		int highlightID = parameters.getParameterAsInteger("highlightID",-1);
154 		MetadataSchema schema = MetadataSchema.find(context,schemaID);
155 		MetadataField[] fields = MetadataField.findAllInSchema(context, schemaID);
156 		String schemaName = schema.getName();
157 		String schemaNamespace = schema.getNamespace();
158 		
159 		String errorString = parameters.getParameter("errors",null);
160 		ArrayList<String> errors = new ArrayList<String>();
161 		if (errorString != null)
162 		{
163 			for (String error : errorString.split(","))
164 				errors.add(error);
165 		}
166 		
167 	
168         // DIVISION: edit-schema
169 		Division main = body.addInteractiveDivision("metadata-schema-edit",contextPath+"/admin/metadata-registry",Division.METHOD_POST,"primary administrative metadata-registry");
170 		main.setHead(T_head1.parameterize(schemaName));
171 		main.addPara(T_para1.parameterize(schemaNamespace));
172 		
173 		
174 		// DIVISION: add or updating a metadata field
175 		if (updateID >= 0)
176 			// Updating an existing field
177 			addUpdateFieldForm(main, schemaName, updateID,  errors);
178 		else
179 			// Add a new field
180 			addNewFieldForm(main, schemaName, errors);
181 		
182 		
183 		
184 		// DIVISION: existing fields
185 		Division existingFields = main.addDivision("metadata-schema-edit-existing-fields");
186 		existingFields.setHead(T_head2);
187 		
188 		Table table = existingFields.addTable("metadata-schema-edit-existing-fields", fields.length+1, 5);
189 		
190 		Row header = table.addRow(Row.ROLE_HEADER);
191 		header.addCellContent(T_column1);
192 		header.addCellContent(T_column2);
193 		header.addCellContent(T_column3);
194 		header.addCellContent(T_column4);
195 		
196 		for (MetadataField field : fields)
197 		{
198 			String id = String.valueOf(field.getFieldID());
199 			String fieldElement = field.getElement();
200 			String fieldQualifier = field.getQualifier();
201 			
202 			String fieldName = schemaName +"."+ fieldElement;
203 			if (fieldQualifier != null && fieldQualifier.length() > 0)
204 				fieldName += "."+fieldQualifier;
205 				
206 			boolean highlight = false;
207 			if (field.getFieldID() == highlightID)
208 				highlight = true;
209 			
210 			String fieldScopeNote = field.getScopeNote();
211 			
212 			String url = contextPath + "/admin/metadata-registry?administrative-continue="+knot.getId()+"&submit_edit&fieldID="+id;
213 			
214 			Row row;
215 			if (highlight)
216 				row = table.addRow(null,null,"highlight");
217 			else
218 				row = table.addRow();
219 			
220 			CheckBox select = row.addCell().addCheckBox("select_field");
221 			select.setLabel(id);
222 			select.addOption(id);
223 			
224 			row.addCell().addContent(id);
225 			row.addCell().addXref(url,fieldName);
226 			row.addCell().addContent(fieldScopeNote);
227 		}
228 		
229 		if (fields.length == 0)
230 		{
231 			// No fields, let the user know.
232 			table.addRow().addCell(1,4).addHighlight("italic").addContent(T_empty);
233 			main.addPara().addButton("submit_return").setValue(T_submit_return);
234 		}
235 		else
236 		{
237 			// Only show the actions if there are fields available to preform them on.
238 			Para actions = main.addPara();
239 			actions.addButton("submit_delete").setValue(T_submit_delete);
240 			if (MetadataSchema.findAll(context).length > 1)
241 				actions.addButton("submit_move").setValue(T_submit_move);	
242 			actions.addButton("submit_return").setValue(T_submit_return);
243 		}
244 		
245 		main.addHidden("administrative-continue").setValue(knot.getId());
246         
247    }

250 	/**
251 	 * Add a form prompting the user to add a new field to the this schema.
252 	 *  
253 	 * @param div The division to add the form too.
254 	 * @param schemaName The schemaName currently being operated on.
255 	 * @param errors A list of errors from previous attempts at adding new fields.
256 	 */
257 	public void addNewFieldForm(Division div, String schemaName, ArrayList<String> errors) throws WingException
258 	{
259 		Request request = ObjectModelHelper.getRequest(objectModel);
260 		String elementValue = request.getParameter("newElement");
261 		String qualifierValue = request.getParameter("newQualifier");
262 		String noteValue = request.getParameter("newNote");
263 		
264 		Division newField = div.addDivision("edit-schema-new-field");
265 		newField.setHead(T_head3);
266 		
267 		List form = newField.addList("edit-schema-new-field-form",List.TYPE_FORM);
268 		addFieldErrors(form, errors);
269 		
270 		form.addLabel(T_name);
271 		Highlight item =form.addItem().addHighlight("big");
272 		
273 		item.addContent(schemaName+" . ");
274 		Text element = item.addText("newElement");
275 		item.addContent(" . ");
276 		Text qualifier = item.addText("newQualifier");
277 		
278 		
279 		element.setSize(15);
280 		element.setValue(elementValue);
281 		
282 		qualifier.setSize(15);
283 		qualifier.setValue(qualifierValue);
284 		
285 		TextArea scopeNote =form.addItem().addTextArea("newNote");
286 		scopeNote.setLabel(T_note);
287 		scopeNote.setHelp(T_note_help);
288 		scopeNote.setSize(2, 35);
289 		scopeNote.setValue(noteValue);
290 		
291 		form.addItem().addButton("submit_add").setValue(T_submit_add);
292 	}

296 	 * Update an existing field by promting the user for it's values.
297 	 *  
298 	 * @param div The division to add the form too.
299 	 * @param schemaName The schemaName currently being operated on.
300 	 * @param fieldID The id of the field being updated.
301 	 * @param errors A list of errors from previous attempts at updaating the field.
302 	 */
303 	public void addUpdateFieldForm(Division div, String schemaName, int fieldID, ArrayList<String> errors) throws WingException, SQLException
304 	{
305 		
306 		MetadataField field = MetadataField.find(context, fieldID);
307 		
308 		Request request = ObjectModelHelper.getRequest(objectModel);
309 		String elementValue = request.getParameter("updateElement");
310 		String qualifierValue = request.getParameter("updateQualifier");
311 		String noteValue = request.getParameter("updateNote");
312 		
313 		if (elementValue == null)
314 			elementValue = field.getElement();
315 		if (qualifierValue == null)
316 			qualifierValue = field.getQualifier();
317 		if (noteValue == null)
318 			noteValue = field.getScopeNote();
319 		
320 		
321 		Division newField = div.addDivision("edit-schema-update-field");
322 		newField.setHead(T_head4.parameterize(field.getFieldID()));
323 		
324 		List form = newField.addList("edit-schema-update-field-form",List.TYPE_FORM);
325 		
326 
327 		addFieldErrors(form, errors);
328 		
329 		form.addLabel(T_name);
330 		Highlight item =form.addItem().addHighlight("big");
331 		
332 		item.addContent(schemaName+" . ");
333 		Text element = item.addText("updateElement");
334 		item.addContent(" . ");
335 		Text qualifier = item.addText("updateQualifier");
336 		
337 		
338 		element.setSize(13);
339 		element.setValue(elementValue);
340 		
341 		qualifier.setSize(13);
342 		qualifier.setValue(qualifierValue);
343 		
344 		TextArea scopeNote =form.addItem().addTextArea("updateNote");
345 		scopeNote.setLabel(T_note);
346 		scopeNote.setHelp(T_note_help);
347 		scopeNote.setSize(2, 35);
348 		scopeNote.setValue(noteValue);
349 		
350 		Item actions = form.addItem();
351 		actions.addButton("submit_update").setValue(T_submit_update);
352 		actions.addButton("submit_cancel").setValue(T_submit_cancel);
353 		
354 	}

356 	/**
357 	 * Determine if there were any special errors and display approparte 
358 	 * text. Because of the inline nature of the element and qualifier 
359 	 * fields these errors can not be placed on the field. Instead they 
360 	 * have to be added as seperate items above the field.
361 	 * 
362 	 * @param form The form to add errors to.
363 	 * @param errors A list of errors.
364 	 */
365 	
366 	public void addFieldErrors(List form, ArrayList<String> errors) throws WingException 
367 	{
368 		if (errors.contains("duplicate_field"))
369 		{
370 			form.addLabel(T_error);
371 			form.addItem(T_error_duplicate_field);
372 		}
373 		if (errors.contains("element_empty"))
374 		{
375 			form.addLabel(T_error);
376 			form.addItem(T_error_element_empty);
377 		}
378 		if (errors.contains("element_badchar"))
379 		{
380 			form.addLabel(T_error);
381 			form.addItem(T_error_element_badchar);
382 		}
383 		if (errors.contains("element_tolong"))
384 		{
385 			form.addLabel(T_error);
386 			form.addItem(T_error_element_tolong);
387 		}
388 		if (errors.contains("qualifier_tolong"))
389 		{
390 			form.addLabel(T_error);
391 			form.addItem(T_error_qualifier_tolong);
392 		}
393 		if (errors.contains("qualifier_badchar"))
394 		{
395 			form.addLabel(T_error);
396 			form.addItem(T_error_qualifier_badchar);
397 		}
398 	}
