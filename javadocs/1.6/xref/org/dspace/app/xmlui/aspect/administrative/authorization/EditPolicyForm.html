141 	public void addBody(Body body) throws WingException, SQLException 
142 	{
143 		/* Get and setup our parameters. We should always have an objectType and objectID, since every policy 
144 		 * has to have a parent resource. policyID may, however, be -1 for new, not-yet-created policies and
145 		 * the groupID is only set if the group is being changed. */
146         int objectType = parameters.getParameterAsInteger("objectType",-1);
147         int objectID = parameters.getParameterAsInteger("objectID",-1);
148         int policyID = parameters.getParameterAsInteger("policyID",-1);
149         int groupID = parameters.getParameterAsInteger("groupID",-1);
150         int actionID = parameters.getParameterAsInteger("actionID",-1);
151         int page = parameters.getParameterAsInteger("page",0);
152         String query = URLDecode(parameters.getParameter("query","-1"));
153         
154         // The current policy, if it exists (i.e. we are not creating a new one)
155         ResourcePolicy policy = ResourcePolicy.find(context, policyID);
156         
157         // The currently set group; it's value depends on wether previously clicked the "Set" button to change 
158         // the associated group, came here to edit an existing group, or create a new one. 
159         Group currentGroup;
160         if (groupID != -1) {
161         	currentGroup = Group.find(context, groupID);
162         }
163         else if (policy != null) {
164         	currentGroup = policy.getGroup();
165         }
166         else currentGroup = null;
167         
168         // Same for the current action; it can either blank (-1), manually set, or inherited from the current policy
169         if (policy != null && actionID == -1)
170         	actionID = policy.getAction();
171                 
172         String errorString = parameters.getParameter("errors",null);
173 		ArrayList<String> errors = new ArrayList<String>();
174 		if (errorString != null)
175 		{
176 			for (String error : errorString.split(","))
177 				errors.add(error);
178 		}
179 		
180 		
181 		/* Set up our current Dspace object */
182 		DSpaceObject dso;
183 		switch (objectType) {
184 			case Constants.COMMUNITY: dso = Community.find(context, objectID); break;
185 			case Constants.COLLECTION: dso = Collection.find(context, objectID); break;
186 			case Constants.ITEM: dso = org.dspace.content.Item.find(context, objectID); break;
187 			case Constants.BUNDLE: dso = Bundle.find(context, objectID); break;
188 			case Constants.BITSTREAM: dso = Bitstream.find(context, objectID); break;
189 			default: dso = null;
190 		}
191 
192 		
193         // DIVISION: edit-container-policies
194         Division main = body.addInteractiveDivision("edit-policy",contextPath+"/admin/authorize",Division.METHOD_POST,"primary administrative authorization");
195         
196         if (policyID >= 0) {
197         	objectID = policy.getResourceID();
198         	objectType = policy.getResourceType();
199         	main.setHead(T_main_head_edit.parameterize(policyID,Constants.typeText[objectType],objectID));
200         }
201         else
202         	main.setHead(T_main_head_new.parameterize(Constants.typeText[objectType],objectID));
203 		
204 	    int resourceRelevance = 1 << objectType; 
205 		
206 		
207 	    // DIVISION: authorization-actions
208         Division actions = main.addDivision("edit-policy-actions");
209         
210         List actionsList = actions.addList("actions","form");
211         
212         // actions radio buttons
213         actionsList.addLabel(T_label_action);
214         Item actionSelectItem = actionsList.addItem();
215         Radio actionSelect = actionSelectItem.addRadio("action_id");
216         actionSelect.setLabel(T_label_action);
217 		        //Select actionSelect = actionSelectItem.addSelect("action_id");
218 		        //actionsBox.addContent(T_label_action);
219 		        //Select actionSelect = actionsBox.addSelect("action_id");
220         for( int i = 0; i < Constants.actionText.length; i++ )
221         {
222             // only display if action i is relevant
223             //  to resource type resourceRelevance                             
224             if( (Constants.actionTypeRelevance[i] & resourceRelevance) > 0)
225             {
226             	if (actionID == i)
227             		actionSelect.addOption(true, i, Constants.actionText[i]);
228             	else
229             		actionSelect.addOption(i, Constants.actionText[i]);
230             }
231         }
232         if (errors.contains("action_id"))
233         	actionSelect.addError(T_error_no_action);        
234         
235         
236         // currently set group
237         actionsList.addLabel(T_policy_currentGroup);
238     	Select groupSelect = actionsList.addItem().addSelect("group_id");
239     	groupSelect.setSize(5);
240     	for (Group group : Group.findAll(context, Group.NAME))
241     	{
242     		if (group == currentGroup)
243     			groupSelect.addOption(true, group.getID(), group.getName());
244     		else
245     			groupSelect.addOption(group.getID(), group.getName());
246     	}
247     	if (errors.contains("group_id"))
248     		groupSelect.addError(T_error_no_group);
249     	
250         
251         // the search function
252         actionsList.addLabel(T_label_search);
253         Item searchItem = actionsList.addItem();
254         Text searchText = searchItem.addText("query");
255         if (!query.equals(new String("-1")))
256         	searchText.setValue(query);
257         searchItem.addButton("submit_search_groups").setValue(T_submit_search_groups);
258         
259         
260         actionsList.addLabel();
261         Item buttons = actionsList.addItem();
262         buttons.addButton("submit_save").setValue(T_submit_save);
263         buttons.addButton("submit_cancel").setValue(T_submit_cancel);
264         
265         
266 	    // Display the search results table
267         if (!query.equals(new String("-1"))) {
268         	Division groupsList = main.addDivision("edit-policy-groupsList");
269             groupsList.setHead(T_groups_head);  
270         	this.addGroupSearch(groupsList, currentGroup, dso, query, page);
271         }       	
272        	
273     	main.addHidden("administrative-continue").setValue(knot.getId());
274    }

278 	/**
279 	 * Search for groups to add to this group.
280 	 */
281 	private void addGroupSearch(Division div, Group sourceGroup, DSpaceObject dso, String query, int page) throws WingException, SQLException
282 	{
283 		Group[] groups = Group.search(context, query, page*RESULTS_PER_PAGE, (page+1)*RESULTS_PER_PAGE);
284 		int totalResults = Group.searchResultCount(context, query);
285 		ArrayList<ResourcePolicy> otherPolicies = (ArrayList<ResourcePolicy>)AuthorizeManager.getPolicies(context, dso);
286 		
287 		
288 		if (totalResults > RESULTS_PER_PAGE) {
289 			int firstIndex = page*RESULTS_PER_PAGE+1; 
290         	int lastIndex = page*RESULTS_PER_PAGE + groups.length;
291 			String baseURL = contextPath+"/admin/authorize?administrative-continue="+knot.getId();
292 			
293 			String nextURL = null, prevURL = null;
294         	if (page < ((totalResults - 1) / RESULTS_PER_PAGE))
295         		nextURL = baseURL+"&page="+(page+1);
296         	if (page > 0)
297         		prevURL = baseURL+"&page="+(page-1);
298         	
299         	div.setSimplePagination(totalResults,firstIndex,lastIndex,prevURL, nextURL);
300 		}
301 		
302 		
303 		Table table = div.addTable("policy-edit-search-group",groups.length + 1, 1);
304          
305         Row header = table.addRow(Row.ROLE_HEADER);
306         
307         // Add the header row 
308 	    header = table.addRow(Row.ROLE_HEADER);
309         header.addCell().addContent(T_groups_column1);
310         header.addCell().addContent(T_groups_column2);
311         header.addCell().addContent(T_groups_column3);
312         header.addCell().addContent(T_groups_column4);
313         
314         // The rows of search results
315         for (Group group : groups)
316         {
317         	String groupID = String.valueOf(group.getID());
318         	String name = group.getName();
319         	url = contextPath+"/admin/groups?administrative-continue="+knot.getId()+"&submit_edit_group&group_id="+groupID;
320         	
321     		Row row = table.addRow();
322         	row.addCell().addContent(groupID);
323         	row.addCell().addXref(url,name);
324         	
325         	// Iterate other other polices of our parent resource to see if any match the currently selected group
326         	String otherAuthorizations = new String();
327         	int groupsMatched = 0;
328         	for (ResourcePolicy otherPolicy : otherPolicies) {
329         		if (otherPolicy.getGroup() == group) {
330         			otherAuthorizations += otherPolicy.getActionText() + ", ";
331         			groupsMatched++;
332         		}
333         	}
334         	
335         	if (groupsMatched > 0) {
336         		row.addCell().addContent(otherAuthorizations.substring(0,otherAuthorizations.lastIndexOf(", ")));
337         	}
338         	else 
339         		row.addCell().addContent("-");
340         	
341         	if (group != sourceGroup)
342     			row.addCell().addButton("submit_group_id_"+groupID).setValue(T_set_group);
343     		else
344     			row.addCell().addContent(T_current_group);
345         	
346         }
347         if (groups.length <= 0) {
348 			table.addRow().addCell(1, 4).addContent(T_no_results);
349 		}
350 	}
