65  /**
66   * Add a single link to the display item page that allows
67   * the user to submit a new item to this collection.
68   * 
69   * @author Scott Phillips
70   */
71  public class CollectionViewer extends AbstractDSpaceTransformer implements CacheableProcessingComponent
72  {
73  	
74  	/** Language Strings */
75      protected static final Message T_title = 
76          message("xmlui.Submission.SelectCollection.title");
77      protected static final Message T_submit = 
78      	message("xmlui.Submission.CollectionViewer.link1");
79      
80  	
81      /** Cached validity object */
82      private SourceValidity validity;
83      
84      /**
85       * Generate the unique caching key.
86       * This key must be unique inside the space of this component.
87       */
88      public Serializable getKey()
89      {
90          try
91          {
92              DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
93  
94              if (dso == null)
95                  return "0";
96                  
97              return HashUtil.hash(dso.getHandle());
98          }
99          catch (SQLException sqle)
100         {
101             // Ignore all errors and just return that the component is not
102             // cachable.
103             return "0";
104         }
105     }
106 
107     /**
108      * Generate the cache validity object.
109      * 
110      * The validity object will include the collection being viewed and 
111      * all recently submitted items. This does not include the community / collection
112      * hierarch, when this changes they will not be reflected in the cache.
113      */
114     public SourceValidity getValidity()
115     {
116     	if (this.validity == null)
117     	{
118 	        try
119 	        {
120 	            DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
121 	
122 	            if (dso == null)
123 	                return null;
124 	
125 	            if (!(dso instanceof Collection))
126 	                return null;
127 	
128 	            Collection collection = (Collection) dso;
129 	
130 	            DSpaceValidity validity = new DSpaceValidity();
131 	            
132 	            // Add the actual collection;
133 	            validity.add(collection);
134 	            
135 	            // Add the eperson viewing the collection
136 	            validity.add(eperson);
137 	            
138 	            // Include any groups they are a member of
139 	            Group[] groups = Group.allMemberGroups(context, eperson);
140 	            for (Group group : groups)
141 	            {
142 	            	validity.add(group);
143 	            }
144 	            
145 	            this.validity = validity.complete();
146 	        }
147 	        catch (Exception e)
148 	        {
149 	            // Just ignore all errors and return an invalid cache.
150 	        }
151     	}
152     	return this.validity;
153     }
154 
155     /**
156      * Add a single link to the view item page that allows the user 
157      * to submit to the collection.
158      */
159     public void addBody(Body body) throws SAXException, WingException,
160             UIException, SQLException, IOException, AuthorizeException
161     {
162         DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
163         if (!(dso instanceof Collection))
164             return;
165  
166         // Set up the major variables
167         Collection collection = (Collection) dso;
168         
169         // Only add the submit link if the user has the ability to add items.
170         if (AuthorizeManager.authorizeActionBoolean(context, collection, Constants.ADD))
171         {
172 	        Division home = body.addDivision("collection-home","primary repository collection");
173 	        Division viewer = home.addDivision("collection-view","secondary");
174 	        String submitURL = contextPath + "/handle/" + collection.getHandle() + "/submit";
175 	        viewer.addPara().addXref(submitURL,T_submit); 
176         }
177         
178     }
179     
180     /**
181      * Recycle
182      */
183     public void recycle() 
184     {   
185         this.validity = null;
186         super.recycle();
187     }
