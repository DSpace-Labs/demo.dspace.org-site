147     /**
148      * Generate the cache validity object.
149      * 
150      * The validity object will include a list of all communities 
151      * & collection being browsed along with there logo bitstreams.
152      */
153     public SourceValidity getValidity()
154     {
155     	if (validity == null)
156     	{
157 	        try {
158 	            DSpaceValidity validity = new DSpaceValidity();
159 	            
160 	            TreeNode root = buildTree(Community.findAllTop(context));
161 	            
162 	            Stack<TreeNode> stack = new Stack<TreeNode>();
163 	            stack.push(root);
164 	            
165 	            int objectCount = 0;
166 	            while (!stack.empty())
167 	            {
168 	            	objectCount++;
169 	                TreeNode node = stack.pop();
170 	                
171 	                validity.add(node.getDSO());
172 	                
173 	                for (TreeNode child : node.getChildren())
174 	                {
175 	                    stack.push(child);
176 	                }
177 	            }
178 	            
179 	            // Check if we are configured to assume validity.
180 	            String assumeCacheValidity = ConfigurationManager.getProperty("xmlui.community-list.cache");
181 	            if (assumeCacheValidity != null)
182 	            	validity.setAssumedValidityDelay(assumeCacheValidity);
183 	            
184 	            this.validity = validity.complete();
185 	        } 
186 	        catch (SQLException sqle) 
187 	        {
188 	            // ignore all errors and return an invalid cache.
189 	        }
190             log.info(LogManager.getHeader(context, "view_community_list", ""));
191     	}
192     	return this.validity;
193     }

250     /**
251      * Recursively build an includeset of the community / collection hierarcher based upon
252      * the given NodeTree.
253      * 
254      * @param referenceSet The include set
255      * @param node The current node of the hierarch.
256      */
257     public void buildReferenceSet(ReferenceSet referenceSet, TreeNode node) throws WingException
258     {
259         DSpaceObject dso = node.getDSO();
260         
261         Reference objectInclude = referenceSet.addReference(dso);
262         
263         // Add all the sub-collections;
264         ArrayList<TreeNode> collectionNodes = node.getChildrenOfType(Constants.COLLECTION);
265         if (collectionNodes != null && collectionNodes.size() > 0)
266         {
267             ReferenceSet collectionSet = objectInclude.addReferenceSet(ReferenceSet.TYPE_SUMMARY_LIST);
268             
269             for (TreeNode collectionNode : collectionNodes)
270             {
271                 collectionSet.addReference(collectionNode.getDSO());
272             }
273         }
274         
275         // Add all the sub-communities
276         ArrayList<TreeNode> communityNodes = node.getChildrenOfType(Constants.COMMUNITY);
277         if (communityNodes != null && communityNodes.size() > 0)
278         {
279             ReferenceSet communitySet = objectInclude.addReferenceSet(ReferenceSet.TYPE_SUMMARY_LIST);
280             
281             for (TreeNode communityNode : communityNodes)
282             {
283                 buildReferenceSet(communitySet,communityNode);
284             }
285         }

288     /**
289      * Recursively build a list of the community / collection hierarchery based upon
290      * the given NodeTree.
291      * 
292      * @param List The parent list
293      * @param node The current node of the hierarch.
294      */
295     public void buildList(List list, TreeNode node) throws WingException
296     {
297         DSpaceObject dso = node.getDSO();
298         
299         String name = null;
300         if (dso instanceof Community)
301         	name = ((Community) dso).getMetadata("name");
302         else if (dso instanceof Collection)
303         	name = ((Collection) dso).getMetadata("name");
304         
305         String url = contextPath + "/handle/"+dso.getHandle();
306         list.addItem().addHighlight("bold").addXref(url, name);
307         
308         List subList = null;
309         
310         // Add all the sub-collections;
311         ArrayList<TreeNode> collectionNodes = node.getChildrenOfType(Constants.COLLECTION);
312         if (collectionNodes != null && collectionNodes.size() > 0)
313         {
314         	subList = list.addList("sub-list-"+dso.getID());
315         
316             for (TreeNode collectionNode : collectionNodes)
317             {
318                 String collectionName = ((Collection) collectionNode.getDSO()).getMetadata("name");
319                 String collectionUrl = contextPath + "/handle/"+collectionNode.getDSO().getHandle();
320                 subList.addItemXref(collectionUrl, collectionName);
321             }
322         }
323         
324         
325         // Add all the sub-communities
326         ArrayList<TreeNode> communityNodes = node.getChildrenOfType(Constants.COMMUNITY);
327         if (communityNodes != null && communityNodes.size() > 0)
328         {
329         	if (subList == null)
330         		subList = list.addList("sub-list-"+dso.getID());
331             
332             for (TreeNode communityNode : communityNodes)
333             {
334                 buildList(subList,communityNode);
335             }
336         }
337     }

349     /**
350      * construct a tree structure of communities and collections. The results 
351      * of this hirarchy are cached so calling it multipletimes is acceptable.
352      * 
353      * @param communities The root level communities
354      * @return A root level node.
355      */
356     private TreeNode buildTree(Community[] communities) throws SQLException
357     {
358         if (root != null)
359             return root;
360         
361         TreeNode newRoot = new TreeNode();
362 
363         // Setup for breath-first traversal
364         Stack<TreeNode> stack = new Stack<TreeNode>();
365 
366         for (Community community : communities)
367             stack.push(newRoot.addChild(community));
368 
369         while (!stack.empty())
370         {
371             TreeNode node = stack.pop();
372 
373             // Short circuit if we have reached our max depth.
374             if (node.getLevel() >= this.depth)
375                 continue;
376 
377             // Only communities nodes are pushed on the stack.
378             Community community = (Community) node.getDSO();
379 
380             for (Community subcommunity : community.getSubcommunities())
381             {
382                 stack.push(node.addChild(subcommunity));
383             }
384 
385             // Add any collections to the document.
386             if (!excludeCollections)
387             {
388                 for (Collection collection : community.getCollections())
389                 {
390                     node.addChild(collection);
391                 }
392             }
393         }
394         
395         this.root = newRoot;
396         return root;
397     }

399     /**
400      * Private class to represent the tree structure of communities & collections. 
401      */
402     protected static class TreeNode
403     {
404         /** The object this node represents */
405         private DSpaceObject dso;
406 
407         /** The level in the hirarchy that this node is at. */
408         private int level;
409 
410         /** All children of this node */
411         private ArrayList<TreeNode> children = new ArrayList<TreeNode>();
412 
413         /** 
414          * Construct a new root level node 
415          */
416         public TreeNode()
417         {
418             // Root level node is add the zero level.
419             this.level = 0;
420         }
421 
422         /**
423          * @return The DSpaceObject this node represents
424          */
425         public DSpaceObject getDSO()
426         {
427             return this.dso;
428         }
429 
430         /**
431          * Add a child DSpaceObject
432          * 
433          * @param dso The child
434          * @return A new TreeNode object attached to the tree structure.
435          */
436         public TreeNode addChild(DSpaceObject dso)
437         {
438             TreeNode child = new TreeNode();
439             child.dso = dso;
440             child.level = this.level + 1;
441             children.add(child);
442             return child;
443         }
444 
445         /**
446          * @return The current level in the hirarchy of this node.
447          */
448         public int getLevel()
449         {
450             return this.level;
451         }
452 
453         /**
454          * @return All children
455          */
456         public ArrayList<TreeNode> getChildren()
457         {
458             return children;
459         }
460 
461         /**
462          * @return All children of the given @type.
463          */
464         public ArrayList<TreeNode> getChildrenOfType(int type)
465         {
466             ArrayList<TreeNode> results = new ArrayList<TreeNode>();
467             for (TreeNode node : children)
468             {
469                 if (node.dso.getType() == type)
470                     results.add(node);
471             }
472             return results;
473         }
474     }
