282     /**
283      * Helpfull method to generate the return url to this page given the 
284      * error & help parameters.
285      */
286     private String makeURL(boolean help, boolean error)
287 	{
288 		if (help && error)
289 			return "?help&error";
290 		
291 		if (help)
292 			return "?help";
293 		
294 		if (error)
295 			return "?error";
296 		
297 		return "?neither";
298 	}
