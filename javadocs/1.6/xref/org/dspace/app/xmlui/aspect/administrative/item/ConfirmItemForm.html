60  /**
61   * This page is used as a general confirmation page for any
62   * actions on items. It will display the item and ask if the
63   * user is sure they want to preform the action.
64   * 
65   * The "confirm" parameter determines what action is confirmed.
66   * There are three possible values, "delete", "withdraw", or
67   * "reinstate"
68   * 
69   * @author Jay Paz
70   * @author Scott Phillips
71   */

103 	@SuppressWarnings("unchecked") // the cast is correct
104 	public void addBody(Body body) throws WingException, SQLException 
105 	{
106 		// Get our parameters and state
107 		int itemID = parameters.getParameterAsInteger("itemID",-1);
108 		Item item = Item.find(context, itemID);
109 		final DCValue[] values = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
110 		Arrays.sort(values, new DCValueComparator());
111 
112 		String confirm = parameters.getParameter("confirm",null);
113 
114 
115 		// DIVISION: Main
116 		Division main = body.addInteractiveDivision("confirm-item", contextPath+"/admin/item", Division.METHOD_POST,"primary administrative item");
117 		main.setHead(T_head1.parameterize(item.getHandle()));
118 
119 		// PARA: descriptive instructions
120 		if("delete".equals(confirm))
121 		{
122 			main.addPara(T_para_delete);
123 		}
124 		else if ("reinstate".equals(confirm))
125 		{
126 			main.addPara(T_para_reinstate);
127 		}
128 		else if ("withdraw".equals(confirm))
129 		{
130 			main.addPara(T_para_withdraw);
131 		}
132 
133 
134 		// TABLE: metadata table
135 		Table table = main.addTable("withdrawValues", values.length+1, 3);
136 		final Row header = table.addRow(Row.ROLE_HEADER);
137 		header.addCell().addContent(T_column1);
138 		header.addCell().addContent(T_column2);
139 		header.addCell().addContent(T_column3);
140 		for(final DCValue value:values){
141 			final String dcValue = value.schema + ". " + value.element + (value.qualifier==null?"":(". " + value.qualifier));
142 			final Row row = table.addRow();
143 			row.addCell().addContent(dcValue);
144 			row.addCell().addContent(value.value);
145 			row.addCell().addContent(value.language);
146 		}
147 
148 		// LIST: actions, confirm or return
149 		org.dspace.app.xmlui.wing.element.Item actions = main.addList("actions", List.TYPE_FORM).addItem();
150 
151 		Button confirmButton = actions.addButton("submit_confirm");
152 
153 		if("delete".equals(confirm))
154 		{
155 			confirmButton.setValue(T_submit_delete);
156 		}
157 		else if ("reinstate".equals(confirm))
158 		{
159 			confirmButton.setValue(T_submit_reinstate);
160 		}
161 		else if ("withdraw".equals(confirm))
162 		{
163 			confirmButton.setValue(T_submit_withdraw);
164 		}
165 		actions.addButton("submit_cancel").setValue(T_submit_cancel);
166 
167 		main.addHidden("administrative-continue").setValue(knot.getId());
168 	}

171 	/**
172 	 * Compare two metadata element's name so that they may be sorted.
173 	 */
174 	class DCValueComparator implements Comparator{
175 		public int compare(Object arg0, Object arg1) {
176 			final DCValue o1 = (DCValue)arg0;
177 			final DCValue o2 = (DCValue)arg1;
178 			final String s1 = o1.schema + o1.element + (o1.qualifier==null?"":("." + o1.qualifier));
179 			final  String s2 = o2.schema + o2.element + (o2.qualifier==null?"":("." + o2.qualifier));
180 			return s1.compareTo(s2);
181 		}
182 	}
