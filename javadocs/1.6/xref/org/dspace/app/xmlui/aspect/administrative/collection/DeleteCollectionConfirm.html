56  /**
57   * Confirmation step for the deletion of an entire collection
58   * @author Alexey Maslov
59   */
60  public class DeleteCollectionConfirm extends AbstractDSpaceTransformer   
61  {
62  	/** Language Strings */
63  	private static final Message T_dspace_home =
64          message("xmlui.general.dspace_home");
65  	
66  	private static final Message T_title = message("xmlui.administrative.collection.DeleteCollectionConfirm.title");
67  	private static final Message T_trail = message("xmlui.administrative.collection.DeleteCollectionConfirm.trail");
68  
69  	private static final Message T_main_head = message("xmlui.administrative.collection.DeleteCollectionConfirm.main_head");
70  	private static final Message T_main_para = message("xmlui.administrative.collection.DeleteCollectionConfirm.main_para");
71  	private static final Message T_confirm_item1 = message("xmlui.administrative.collection.DeleteCollectionConfirm.confirm_item1");
72  	private static final Message T_confirm_item2 = message("xmlui.administrative.collection.DeleteCollectionConfirm.confirm_item2");
73  	private static final Message T_confirm_item3 = message("xmlui.administrative.collection.DeleteCollectionConfirm.confirm_item3");
74  
75  	private static final Message T_submit_confirm = message("xmlui.general.delete");
76  	private static final Message T_submit_cancel = message("xmlui.general.cancel");
77  	
78  	
79  	public void addPageMeta(PageMeta pageMeta) throws WingException
80      {
81          pageMeta.addMetadata("title").addContent(T_title);
82          pageMeta.addTrailLink(contextPath + "/", T_dspace_home);
83          pageMeta.addTrail().addContent(T_trail);
84      }
85  	
86  	public void addBody(Body body) throws WingException, SQLException, AuthorizeException
87  	{
88  		int collectionID = parameters.getParameterAsInteger("collectionID", -1);
89  		Collection thisCollection = Collection.find(context, collectionID);
90  		
91  		// DIVISION: main
92  	    Division main = body.addInteractiveDivision("collection-confirm-delete",contextPath+"/admin/collection",Division.METHOD_POST,"primary administrative collection");
93  	    main.setHead(T_main_head.parameterize(collectionID));
94  	    main.addPara(T_main_para.parameterize(thisCollection.getMetadata("name")));	    
95  	    List deleteConfirmHelp = main.addList("consequences",List.TYPE_BULLETED);
96  	    deleteConfirmHelp.addItem(T_confirm_item1);
97  	    deleteConfirmHelp.addItem(T_confirm_item2);
98  	    deleteConfirmHelp.addItem(T_confirm_item3);
99  	    
100 	    Para buttonList = main.addPara();
101 	    buttonList.addButton("submit_confirm").setValue(T_submit_confirm);
102 	    buttonList.addButton("submit_cancel").setValue(T_submit_cancel);
103 	    
104     	main.addHidden("administrative-continue").setValue(knot.getId());
105     }
