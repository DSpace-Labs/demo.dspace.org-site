99  	public void addBody(Body body) throws SAXException, WingException, SQLException, IOException
100 	{
101 		// Get our parameters and state;
102 		int collectionID = parameters.getParameterAsInteger("collectionID",-1);
103 		Collection collection = Collection.find(context,collectionID);	
104 		
105 		String query = URLDecode(parameters.getParameter("query",null));
106 		ArrayList<Item> items = preformSearch(collection,query);
107 		
108 		
109 		
110 		// DIVISION: manage-mapper
111 		Division div = body.addInteractiveDivision("search-items",contextPath + "/admin/mapper", Division.METHOD_GET,"primary administrative mapper");
112 		div.setHead(T_head1.parameterize(query));
113 		
114 		Para actions = div.addPara();
115 		actions.addButton("submit_map").setValue(T_submit_map);
116 		actions.addButton("submit_cancel").setValue(T_submit_cancel);
117 		
118 		Table table = div.addTable("search-items-table",1,1);
119 		
120 		Row header = table.addRow(Row.ROLE_HEADER);
121 		header.addCellContent(T_column1);
122 		header.addCellContent(T_column2);
123 		header.addCellContent(T_column3);
124 		header.addCellContent(T_column4);
125 		
126 		for (Item item : items)
127 		{
128 			String itemID = String.valueOf(item.getID());
129 			Collection owningCollection = item.getOwningCollection();
130 			String owning = owningCollection.getMetadata("name");
131 			String author = "unkown";
132 			DCValue[] dcAuthors = item.getDC("contributor",Item.ANY,Item.ANY);
133 			if (dcAuthors != null && dcAuthors.length >= 1)
134 				author = dcAuthors[0].value;
135 			
136 			String title = "untitled";
137 			DCValue[] dcTitles = item.getDC("title",null,Item.ANY);
138 			if (dcTitles != null && dcTitles.length >= 1)
139 				title = dcTitles[0].value;
140 
141 			String url = contextPath+"/handle/"+item.getHandle();
142 			
143 			Row row = table.addRow();
144 
145             boolean canBeMapped = true;
146             Collection[] collections = item.getCollections();
147             for (Collection c : collections)
148             {
149                 if (c.getID() == collectionID)
150                 {
151                     canBeMapped = false;
152                 }
153             }
154 
155             if (canBeMapped)
156             {
157                 CheckBox select = row.addCell().addCheckBox("itemID");
158                 select.setLabel("Select");
159                 select.addOption(itemID);
160             }
161             else
162             {
163                 row.addCell().addContent("");
164             }
165 
166 			row.addCellContent(owning);
167 			row.addCell().addXref(url,author);
168 			row.addCell().addXref(url,title);
169 		}
170 		
171 		actions = div.addPara();
172 		actions.addButton("submit_map").setValue(T_submit_map);
173 		actions.addButton("submit_cancel").setValue(T_submit_cancel);
174 		
175 		
176 		
177 		
178 		div.addHidden("administrative-continue").setValue(knot.getId());
179 	}

182 	/**
183 	 * Search the repository for items in other collections that can be mapped into this one.
184 	 * 
185 	 * @param collection The collection to mapp into
186 	 * @param query The search query.
187 	 */
188 	private ArrayList<Item> preformSearch(Collection collection, String query) throws SQLException, IOException
189 	{
190 		
191 		// Search the repository
192         QueryArgs queryArgs = new QueryArgs();
193         queryArgs.setQuery(query);
194         queryArgs.setPageSize(Integer.MAX_VALUE);
195         QueryResults results = DSQuery.doQuery(context, queryArgs);
196         
197 
198         // Get a list of found items
199         ArrayList<Item> items = new ArrayList<Item>();
200         @SuppressWarnings("unchecked")
201         java.util.List<String> handles = results.getHitHandles();
202         for (String handle : handles)
203         {
204             DSpaceObject resultDSO = HandleManager.resolveToObject(context, handle);
205 
206             if (resultDSO instanceof Item)
207             {
208             	Item item = (Item) resultDSO;
209             	
210             	if (!item.isOwningCollection(collection))
211             		items.add(item);
212             }
213         }
214         
215         return items;
216 	}
