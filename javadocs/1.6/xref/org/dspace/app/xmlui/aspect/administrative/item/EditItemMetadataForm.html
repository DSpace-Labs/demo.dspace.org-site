114         public void addPageMeta(PageMeta pageMeta) throws WingException, SQLException
115         {
116                 Item item = Item.find(context, parameters.getParameterAsInteger("itemID",-1));
117                 Collection owner = item.getOwningCollection();
118                 int collectionID = (owner == null) ? -1 : owner.getID();
119         
120                 pageMeta.addMetadata("choice", "collection").addContent(String.valueOf(collectionID));
121                 pageMeta.addMetadata("title").addContent(T_title);
122 
123                 pageMeta.addTrailLink(contextPath + "/", T_dspace_home);
124                 pageMeta.addTrailLink(contextPath + "/admin/item", T_item_trail);
125                 pageMeta.addTrail().addContent(T_trail);
126         }

128         /**
129          * Add either the simple Ajax response document or the full
130          * document with header and full edit item form based on the
131          * mode parameter.  Mode parameter values are set in the Flowscipt
132          * and can be either 'ajax' or 'normal'.
133          */
134         @SuppressWarnings("unchecked") // the cast is correct
135         public void addBody(Body body) throws SQLException, WingException
136         {
137                 // Get our parameters and state
138                 int itemID = parameters.getParameterAsInteger("itemID",-1);
139                 Item item = Item.find(context, itemID);
140                 DCValue[] values = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
141                 Arrays.sort(values, new DCValueComparator());
142                 String baseURL = contextPath+"/admin/item?administrative-continue="+knot.getId();
143 
144                 Request request = ObjectModelHelper.getRequest(objectModel);
145                 String previousFieldID = request.getParameter("field");
146 
147         // Metadata editing is the only type of editing available for a template item.
148         boolean editingTemplateItem = false;
149         int templateCollectionID = parameters.getParameterAsInteger("templateCollectionID",-1);
150         Collection templateCollection = templateCollectionID == -1 ? null : Collection.find(context, templateCollectionID);
151         if (templateCollection != null)
152         {
153             Item templateItem = templateCollection.getTemplateItem();
154             if (templateItem != null && templateItem.getID() == itemID)
155                 editingTemplateItem = true;
156         }
157 
158                 // DIVISION: main
159                 Division main = body.addInteractiveDivision("edit-item-status", contextPath+"/admin/item", Division.METHOD_POST,"primary administrative item");
160                 if (editingTemplateItem)
161         {
162                     main.setHead(T_template_head.parameterize(templateCollection.getName()));
163                 }
164         else
165         {
166                 main.setHead(T_option_head);
167                 }
168 
169                 Collection owner = item.getOwningCollection();
170                 int collectionID = (owner == null) ? -1 : owner.getID();
171                 
172                 
173                 // LIST: options
174                 if (!editingTemplateItem)
175         {
176                   List options = main.addList("options",List.TYPE_SIMPLE,"horizontal");
177                   options.addItem().addXref(baseURL+"&submit_status",T_option_status);
178                   options.addItem().addXref(baseURL+"&submit_bitstreams",T_option_bitstreams);
179                   options.addItem().addHighlight("bold").addXref(baseURL+"&submit_metadata",T_option_metadata);
180                   options.addItem().addXref(baseURL + "&view_item", T_option_view);
181                 }
182 
183                 // LIST: add new metadata
184                 List addForm = main.addList("addItemMetadata",List.TYPE_FORM);
185                 addForm.setHead(T_head1);
186 
187                 Select addName = addForm.addItem().addSelect("field");
188                 addName.setLabel(T_name_label);
189                 MetadataField[] fields = MetadataField.findAll(context);
190                 for (MetadataField field : fields)
191                 {
192                         int fieldID = field.getFieldID();
193                         MetadataSchema schema = MetadataSchema.find(context, field.getSchemaID());
194                         String name = schema.getName() +"."+field.getElement();
195                         if (field.getQualifier() != null)
196                                 name += "."+field.getQualifier();
197 
198                         addName.addOption(fieldID, name);
199                 }
200                 if (previousFieldID != null)
201                         addName.setOptionSelected(previousFieldID);
202 
203 
204                 Composite addComposite = addForm.addItem().addComposite("value");
205                 addComposite.setLabel(T_value_label);
206                 TextArea addValue = addComposite.addTextArea("value");
207                 Text addLang = addComposite.addText("language");
208 
209                 addValue.setSize(4, 35);
210                 addLang.setLabel(T_lang_label);
211                 addLang.setSize(6);
212 
213                 addForm.addItem().addButton("submit_add").setValue(T_submit_add);
214 
215                 
216                 
217 
218                 // PARA: Disclaimer
219                 main.addPara(T_para1);
220 
221                 
222                 Para actions = main.addPara(null,"edit-metadata-actions top" );
223                 actions.addButton("submit_update").setValue(T_submit_update);
224                 actions.addButton("submit_return").setValue(T_submit_return);
225                 
226                 
227 
228                 // TABLE: Metadata
229                 main.addHidden("scope").setValue("*");
230                 int index = 1;
231                 Table table = main.addTable("editItemMetadata",1,1);
232                 table.setHead(T_head2);
233 
234                 Row header = table.addRow(Row.ROLE_HEADER);
235                 header.addCell().addContent(T_column1);
236                 header.addCell().addContent(T_column2);
237                 header.addCell().addContent(T_column3);
238                 header.addCell().addContent(T_column4);
239 
240                 ChoiceAuthorityManager cmgr = ChoiceAuthorityManager.getManager();
241                 for(DCValue value : values)
242                 {
243                         String name = value.schema + "_" + value.element;
244                         if (value.qualifier != null)
245                                 name += "_" + value.qualifier;
246 
247                         Row row = table.addRow(name,Row.ROLE_DATA,"metadata-value");
248 
249                         CheckBox remove = row.addCell().addCheckBox("remove_"+index);
250                         remove.setLabel("remove");
251                         remove.addOption(index);
252 
253                         Cell cell = row.addCell();
254                         cell.addContent(name.replaceAll("_", ". "));
255                         cell.addHidden("name_"+index).setValue(name);
256 
257                         // value entry cell:
258                         Cell mdCell = row.addCell();
259                         String fieldKey = MetadataAuthorityManager.makeFieldKey(value.schema, value.element, value.qualifier);
260 
261                         // put up just a selector when preferred choice presentation is select:
262                         if (cmgr.isChoicesConfigured(fieldKey) &&
263                             Params.PRESENTATION_SELECT.equals(cmgr.getPresentation(fieldKey)))
264                         {
265                             Select mdSelect = mdCell.addSelect("value_"+index);
266                             mdSelect.setSize(1);
267                             Choices cs = cmgr.getMatches(fieldKey, value.value, collectionID, 0, 0, null);
268                             if (cs.defaultSelected < 0)
269                                 mdSelect.addOption(true, value.value, value.value);
270                             for (int i = 0; i < cs.values.length; ++i)
271                             {
272                                 mdSelect.addOption(i == cs.defaultSelected, cs.values[i].value, cs.values[i].label);
273                             }
274                         }
275                         else
276                         {
277                             TextArea mdValue = mdCell.addTextArea("value_"+index);
278                         mdValue.setSize(4,35);
279                         mdValue.setValue(value.value);
280                             boolean isAuth = MetadataAuthorityManager.getManager().isAuthorityControlled(fieldKey);
281                             if (isAuth)
282                             {
283                                 mdValue.setAuthorityControlled();
284                                 mdValue.setAuthorityRequired(MetadataAuthorityManager.getManager().isAuthorityRequired(fieldKey));
285                                 Value authValue =  mdValue.setAuthorityValue((value.authority == null)?"":value.authority, Choices.getConfidenceText(value.confidence));
286                                 // add the "unlock" button to auth field
287                                 Button unlock = authValue.addButton("authority_unlock_"+index,"ds-authority-lock");
288                                 unlock.setHelp(T_unlock);
289                             }
290                             if (ChoiceAuthorityManager.getManager().isChoicesConfigured(fieldKey))
291                             {
292                                 mdValue.setChoices(fieldKey);
293                                 mdValue.setChoicesPresentation(Params.PRESENTATION_LOOKUP);
294                                 mdValue.setChoicesClosed(ChoiceAuthorityManager.getManager().isClosed(fieldKey));
295                             }
296                         }
297                         Text mdLang = row.addCell().addText("language_"+index);
298                         mdLang.setSize(6);
299                         mdLang.setValue(value.language);
300 
301                         // Tick the index counter;
302                         index++;
303                 }
304 
305                 
306                 
307                 
308                 // PARA: actions
309                 actions = main.addPara(null,"edit-metadata-actions bottom" );
310                 actions.addButton("submit_update").setValue(T_submit_update);
311                 actions.addButton("submit_return").setValue(T_submit_return);
312 
313 
314                 main.addHidden("administrative-continue").setValue(knot.getId());
315         }

319         /**
320          * Compare two metadata element's name so that they may be sorted.
321          */
322         class DCValueComparator implements Comparator{
323                 public int compare(Object arg0, Object arg1) {
324                         final DCValue o1 = (DCValue)arg0;
325                         final DCValue o2 = (DCValue)arg1;
326                         final String s1 = o1.schema + o1.element + (o1.qualifier==null?"":("." + o1.qualifier));
327                         final  String s2 = o2.schema + o2.element + (o2.qualifier==null?"":("." + o2.qualifier));
328                         return s1.compareTo(s2);
329                 }
330         }
