65  /**
66   * Preform a simple search of the repository. The user provides a simple one
67   * field query (the url parameter is named query) and the results are processed.
68   * 
69   * @author Scott Phillips
70   */
71  public class SimpleSearch extends AbstractSearch implements CacheableProcessingComponent
72  {
73      /** Language Strings */
74      private static final Message T_title =
75          message("xmlui.ArtifactBrowser.SimpleSearch.title");
76      
77      private static final Message T_dspace_home =
78          message("xmlui.general.dspace_home");
79      
80      private static final Message T_trail =
81          message("xmlui.ArtifactBrowser.SimpleSearch.trail");
82      
83      private static final Message T_head = 
84          message("xmlui.ArtifactBrowser.SimpleSearch.head");
85      
86      private static final Message T_search_scope =
87          message("xmlui.ArtifactBrowser.SimpleSearch.search_scope");
88      
89      private static final Message T_full_text_search =
90          message("xmlui.ArtifactBrowser.SimpleSearch.full_text_search");
91      
92      private static final Message T_go =
93          message("xmlui.general.go");
94      
95      /**
96       * Add Page metadata.
97       */
98      public void addPageMeta(PageMeta pageMeta) throws WingException, SQLException
99      {
100         pageMeta.addMetadata("title").addContent(T_title);
101         pageMeta.addTrailLink(contextPath + "/", T_dspace_home);
102 		
103 		DSpaceObject dso = HandleUtil.obtainHandle(objectModel);
104         if ((dso instanceof Collection) || (dso instanceof Community))
105         {
106 	        HandleUtil.buildHandleTrail(dso,pageMeta,contextPath);
107 		} 
108 		
109         pageMeta.addTrail().addContent(T_trail);
110     }
111     
112     /**
113      * build the DRI page representing the body of the search query. This
114      * provides a widget to generate a new query and list of search results if
115      * present.
116      */
117     public void addBody(Body body) throws SAXException, WingException,
118             UIException, SQLException, IOException, AuthorizeException
119     {
120         String queryString = getQuery();
121 
122         // Build the DRI Body
123         Division search = body.addDivision("search","primary");
124         search.setHead(T_head);
125         Division query = search.addInteractiveDivision("general-query",
126                 "search",Division.METHOD_GET,"secondary search");
127 
128         List queryList = query.addList("search-query",List.TYPE_FORM);
129         
130         if (variableScope())
131         {
132             Select scope = queryList.addItem().addSelect("scope");
133             scope.setLabel(T_search_scope);
134             buildScopeList(scope);
135         }
136         
137         Text text = queryList.addItem().addText("query");
138         text.setLabel(T_full_text_search);
139         text.setValue(queryString);
140         
141         buildSearchControls(query);
142         query.addPara(null, "button-list").addButton("submit").setValue(T_go);
143 
144         // Add the result division
145         buildSearchResultsDivision(search);
146 
147     }
148 
149     /**
150      * Get the search query from the URL parameter, if none is found the empty
151      * string is returned.
152      */
153     protected String getQuery() throws UIException
154     {
155         Request request = ObjectModelHelper.getRequest(objectModel);
156         String query = URLDecode(request.getParameter("query"));
157         if (query == null)
158             return "";
159         return query;
160     }
161 
162     /**
163      * Generate a url to the simple search url.
164      */
165     protected String generateURL(Map<String, String> parameters)
166             throws UIException
167     {
168         String query = getQuery();
169         if (!"".equals(query))
170             parameters.put("query", URLEncode(query));
171         
172         if (parameters.get("page") == null)
173         	parameters.put("page", String.valueOf(getParameterPage()));
174         
175         if (parameters.get("rpp") == null)
176         	parameters.put("rpp", String.valueOf(getParameterRpp()));
177         
178         if (parameters.get("sort_by") == null)
179         	parameters.put("sort_by", String.valueOf(getParameterSortBy()));
180         
181         if (parameters.get("order") == null)
182         	parameters.put("order",getParameterOrder());
183         
184         if (parameters.get("etal") == null)
185         	parameters.put("etal",String.valueOf(getParameterEtAl()));
186         
187         return super.generateURL("search", parameters);
188     }
