127 	public void addBody(Body body) throws WingException, SQLException, AuthorizeException
128 	{
129 		int collectionID = parameters.getParameterAsInteger("collectionID", -1);
130 		Collection thisCollection = Collection.find(context, collectionID);
131 		
132 		String baseURL = contextPath + "/admin/collection?administrative-continue=" + knot.getId();
133 		
134 		Group admins = thisCollection.getAdministrators();
135 		Group wfStep1 = thisCollection.getWorkflowGroup(1);
136 		Group wfStep2 = thisCollection.getWorkflowGroup(2);
137 		Group wfStep3 = thisCollection.getWorkflowGroup(3);
138 		Group submitters = thisCollection.getSubmitters();
139 
140 		Group defaultRead = null;
141 		int defaultReadID = FlowContainerUtils.getCollectionDefaultRead(context, collectionID);
142 		if (defaultReadID >= 0)
143 			defaultRead = Group.find(context, defaultReadID);
144 		
145 		// DIVISION: main
146 	    Division main = body.addInteractiveDivision("collection-assign-roles",contextPath+"/admin/collection",Division.METHOD_POST,"primary administrative collection");
147 	    main.setHead(T_main_head.parameterize(thisCollection.getMetadata("name")));
148 	    
149 	    List options = main.addList("options", List.TYPE_SIMPLE, "horizontal");
150 	    options.addItem().addXref(baseURL+"&submit_metadata",T_options_metadata);
151 	    options.addItem().addHighlight("bold").addXref(baseURL+"&submit_roles",T_options_roles);
152 	    options.addItem().addXref(baseURL+"&submit_harvesting",T_options_harvest);
153 	    	    
154 	    // The table of admin roles
155 	    Table rolesTable = main.addTable("roles-table", 6, 5);
156 	    Row tableRow;
157 	    
158 	    // The header row
159 	    Row tableHeader = rolesTable.addRow(Row.ROLE_HEADER);
160 	    tableHeader.addCell().addContent(T_role_name);
161 	    tableHeader.addCell().addContent(T_role_group);
162 	    tableHeader.addCell().addContent(T_role_buttons);
163 	    rolesTable.addRow();
164 	    	    
165 	    
166 	    /* 
167 	     * The collection admins 
168 	     */
169 	    // data row
170 	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
171 	    tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_admins);
172 	    if (admins != null) 
173 	    {
174 	        try
175             {
176                 AuthorizeUtil.authorizeManageAdminGroup(context, thisCollection);
177                 tableRow.addCell().addXref(baseURL + "&submit_edit_admin", admins.getName());
178             }
179 	        catch (AuthorizeException authex) {
180                 // add a notice, the user is not authorized to create/edit collection's admin group
181                 tableRow.addCell().addContent(T_not_allowed);
182             }
183             try
184             {
185                 AuthorizeUtil.authorizeRemoveAdminGroup(context, thisCollection);
186                 tableRow.addCell().addButton("submit_delete_admin").setValue(T_delete);
187             }
188             catch (AuthorizeException authex)
189             {
190                 // nothing to add, the user is not allowed to delete the group
191             }
192 	    }
193 	    else 
194 	    {
195 	    	tableRow.addCell().addContent(T_no_role);
196 	    	try
197             {
198                 AuthorizeUtil.authorizeManageAdminGroup(context, thisCollection);
199                 tableRow.addCell().addButton("submit_create_admin").setValue(T_create);
200             }
201             catch (AuthorizeException authex) {
202                 // add a notice, the user is not authorized to create/edit collection's admin group
203                 tableRow.addCell().addContent(T_not_allowed);
204             }
205 	    }
206 	    // help and directions row
207 	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
208 	    tableRow.addCell();
209 	    tableRow.addCell(1,2).addHighlight("fade offset").addContent(T_help_admins);
210 	    
211 	    
212 	    /* 
213 	     * Workflow steps 1-3 
214 	     */
215 	    // data row
216 	    try
217         {
218             AuthorizeUtil.authorizeManageWorkflowsGroup(context, thisCollection);
219     	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
220     	    tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_wf_step1);
221     	    if (wfStep1 != null) 
222     	    {
223     	    	tableRow.addCell().addXref(baseURL + "&submit_edit_wf_step1", wfStep1.getName());
224                 tableRow.addCell().addButton("submit_delete_wf_step1").setValue(T_delete);
225     	    }
226     	    else 
227     	    {
228     	    	tableRow.addCell().addContent(T_no_role);
229                 tableRow.addCell().addButton("submit_create_wf_step1").setValue(T_create);
230     	    }
231     	    // help and directions row
232     	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
233     	    tableRow.addCell();
234     	    tableRow.addCell(1,2).addHighlight("fade offset").addContent(T_help_wf_step1);
235     	    
236     	    
237     	    // data row
238     	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
239     	    tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_wf_step2);
240     	    if (wfStep2 != null) 
241     	    {
242     	    	tableRow.addCell().addXref(baseURL + "&submit_edit_wf_step2", wfStep2.getName());
243                 tableRow.addCell().addButton("submit_delete_wf_step2").setValue(T_delete);
244     	    }
245     	    else 
246     	    {
247     	    	tableRow.addCell().addContent(T_no_role);
248                 tableRow.addCell().addButton("submit_create_wf_step2").setValue(T_create);
249     	    }
250     	    // help and directions row
251     	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
252     	    tableRow.addCell();
253     	    tableRow.addCell(1,2).addHighlight("fade offset").addContent(T_help_wf_step2);
254     	    
255     	    
256     	    // data row
257     	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
258     	    tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_wf_step3);
259     	    if (wfStep3 != null) 
260     	    {
261     	    	tableRow.addCell().addXref(baseURL + "&submit_edit_wf_step3", wfStep3.getName());
262                 tableRow.addCell().addButton("submit_delete_wf_step3").setValue(T_delete);
263     	    }
264     	    else 
265     	    {
266     	    	tableRow.addCell().addContent(T_no_role);
267                 tableRow.addCell().addButton("submit_create_wf_step3").setValue(T_create);
268     	    }
269     	    // help and directions row
270     	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
271     	    tableRow.addCell();
272     	    tableRow.addCell(1,2).addHighlight("fade offset").addContent(T_help_wf_step3);
273         }
274 	    catch (AuthorizeException authex) {
275             // add a notice, the user is not allowed to manage workflow group
276 	        tableRow = rolesTable.addRow(Row.ROLE_DATA);
277             tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_wf);
278             tableRow.addCell().addContent(T_not_allowed);
279         }
280 	    	
281 	    /*
282 	     * The collection submitters 
283 	     */
284 	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
285 	    tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_submitters);
286 	    try
287 	    {
288 	        AuthorizeUtil.authorizeManageSubmittersGroup(context, thisCollection);
289     	    if (submitters != null) 
290     	    {
291     	    	tableRow.addCell().addXref(baseURL + "&submit_edit_submit", submitters.getName());
292                 tableRow.addCell().addButton("submit_delete_submit").setValue(T_delete);
293     	    }
294     	    else 
295     	    {
296     	    	tableRow.addCell().addContent(T_no_role);
297                 tableRow.addCell().addButton("submit_create_submit").setValue(T_create);
298     	    }
299 	    }
300 	    catch (AuthorizeException authex)
301 	    {
302 	        tableRow.addCell().addContent(T_not_allowed);
303 	    }
304 	    // help and directions row
305 	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
306 	    tableRow.addCell();
307 	    tableRow.addCell(1,2).addHighlight("fade offset").addContent(T_help_submitters);
308 	    
309 	    
310 	    /* 
311 	     * The collection's default read authorizations 
312 	     */
313 	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
314 	    tableRow.addCell(Cell.ROLE_HEADER).addContent(T_label_default_read);
315 	    if (defaultRead == null) 
316 	    {
317 	    	// Custome reading permissions, we can't handle it, just provide a link to the 
318 	    	// authorizations manager.
319 	    	tableRow.addCell(1,2).addContent(T_default_read_custom);
320 	    }
321 	    else if (defaultRead.getID() == 0) {
322 	    	// Anonymous reading
323 	    	tableRow.addCell().addContent(T_default_read_anonymous);
324 	    	addAdministratorOnlyButton(tableRow.addCell(),"submit_create_default_read",T_restrict);
325 	    }
326 	    else 
327 	    {
328 	    	// A specific group is dedicated to reading.
329 	    	tableRow.addCell().addXref(baseURL + "&submit_edit_default_read", defaultRead.getName());
330 		    addAdministratorOnlyButton(tableRow.addCell(),"submit_delete_default_read",T_delete);
331 	    }
332 	 
333 	    // help and directions row
334 	    tableRow = rolesTable.addRow(Row.ROLE_DATA);
335 	    tableRow.addCell();
336 	    tableRow.addCell(1,2).addHighlight("fade offset").addContent(T_help_default_read);
337 	    
338 	    try
339 	    {
340 	        AuthorizeUtil.authorizeManageCollectionPolicy(context, thisCollection);
341 		    // add one last link to edit the raw authorizations
342 		    Cell authCell =rolesTable.addRow().addCell(1,3);
343 		    authCell.addXref(baseURL + "&submit_authorizations", T_edit_authorization);
344 	    }
345 	    catch (AuthorizeException authex) {
346             // nothing to add, the user is not authorized to edit collection's policies
347         }
348 
349 	    Para buttonList = main.addPara();
350 	    buttonList.addButton("submit_return").setValue(T_submit_return);
351 	    
352 	    
353 	    
354     	main.addHidden("administrative-continue").setValue(knot.getId());
355     	
356     }

359 	private void addAdministratorOnlyButton(Cell cell, String buttonName, Message buttonLabel) throws WingException, SQLException
360 	{
361     	Button button = cell.addButton(buttonName);
362     	button.setValue(buttonLabel);
363     	if (!AuthorizeManager.isAdmin(context))
364     	{
365     		// Only admins can create or delete
366     		button.setDisabled();
367     		cell.addHighlight("fade").addContent(T_sysadmins_only);
368     	}
