85  /**
86   * The BitstreamReader will query DSpace for a particular bitstream and transmit
87   * it to the user. There are several method of specifing the bitstream to be
88   * develivered. You may refrence a bitstream by either it's id or attempt to
89   * resolve the bitstream's name.
90   *
91   *  /bitstream/{handle}/{sequence}/{name}
92   *
93   *  &lt;map:read type="BitstreamReader">
94   *    &lt;map:parameter name="handle" value="{1}/{2}"/&gt;
95   *    &lt;map:parameter name="sequence" value="{3}"/&gt;
96   *    &lt;map:parameter name="name" value="{4}"/&gt;
97   *  &lt;/map:read&gt;
98   *
99   *  When no handle is assigned yet you can access a bistream
100  *  using it's internal ID.
101  *
102  *  /bitstream/id/{bitstreamID}/{sequence}/{name}
103  *
104  *  &lt;map:read type="BitstreamReader">
105  *    &lt;map:parameter name="bitstreamID" value="{1}"/&gt;
106  *    &lt;map:parameter name="sequence" value="{2}"/&gt;
107  *  &lt;/map:read&gt;
108  *
109  *  Alternativly, you can access the bitstream via a name instead
110  *  of directly through it's sequence.
111  *
112  *  /html/{handle}/{name}
113  *
114  *  &lt;map:read type="BitstreamReader"&gt;
115  *    &lt;map:parameter name="handle" value="{1}/{2}"/&gt;
116  *    &lt;map:parameter name="name" value="{3}"/&gt;
117  *  &lt;/map:read&gt;
118  *
119  *  Again when no handle is available you can also access it
120  *  via an internal itemID & name.
121  *
122  *  /html/id/{itemID}/{name}
123  *
124  *  &lt;map:read type="BitstreamReader"&gt;
125  *    &lt;map:parameter name="itemID" value="{1}"/&gt;
126  *    &lt;map:parameter name="name" value="{2}"/&gt;
127  *  &lt;/map:read&gt;
128  *
129  * @author Scott Phillips
130  */

188     /**
189      * Set up the bitstream reader.
190      *
191      * See the class description for information on configuration options.
192      */
193     public void setup(SourceResolver resolver, Map objectModel, String src,
194             Parameters par) throws ProcessingException, SAXException,
195             IOException
196     {
197         super.setup(resolver, objectModel, src, par);
198 
199         try
200         {
201             this.request = ObjectModelHelper.getRequest(objectModel);
202             this.response = ObjectModelHelper.getResponse(objectModel);
203             Context context = ContextUtil.obtainContext(objectModel);
204             
205             // Get our parameters that identify the bitstream
206             int itemID = par.getParameterAsInteger("itemID", -1);
207             int bitstreamID = par.getParameterAsInteger("bitstreamID", -1);
208             String handle = par.getParameter("handle", null);
209             
210             int sequence = par.getParameterAsInteger("sequence", -1);
211             String name = par.getParameter("name", null);
212         
213             this.isSpider = par.getParameter("userAgent", "").equals("spider");
214 
215             // Reslove the bitstream
216             Bitstream bitstream = null;
217             DSpaceObject dso = null;
218             
219             if (bitstreamID > -1)
220             {
221                 // Direct refrence to the individual bitstream ID.
222                 bitstream = Bitstream.find(context, bitstreamID);
223             }
224             else if (itemID > -1)
225             {
226                 // Referenced by internal itemID
227                 item = Item.find(context, itemID);
228                 
229                 if (sequence > -1)
230                 {
231                         bitstream = findBitstreamBySequence(item, sequence);
232                 }
233                 else if (name != null)
234                 {
235                         bitstream = findBitstreamByName(item, name);
236                 }
237             }
238             else if (handle != null)
239             {
240                 // Reference by an item's handle.
241                 dso = HandleManager.resolveToObject(context,handle);
242 
243                 if (dso instanceof Item)
244                 {
245                     item = (Item)dso;
246 
247                     if (sequence > -1)
248                     {
249                         bitstream = findBitstreamBySequence(item,sequence);
250                     }
251                     else if (name != null)
252                     {
253                         bitstream = findBitstreamByName(item,name);
254                     }
255                 }
256             }
257 
258             //if initial search was by sequence number and found nothing,
259             //then try to find bitstream by name (assuming we have a file name)
260             if((sequence > -1 && bitstream==null) && name!=null)
261             {
262                 bitstream = findBitstreamByName(item,name);
263 
264                 //if we found bitstream by name, send a redirect to its new sequence number location
265                 if(bitstream!=null)
266                 {
267                     String redirectURL = "";
268 
269                     //build redirect URL based on whether item has a handle assigned yet
270                     if(item.getHandle()!=null && item.getHandle().length()>0)
271                       redirectURL = request.getContextPath() + "/bitstream/handle/" + item.getHandle();
272                     else
273                       redirectURL = request.getContextPath() + "/bitstream/item/" + item.getID();
274 
275                         redirectURL += "/" + name + "?sequence=" + bitstream.getSequenceID();
276 
277                         HttpServletResponse httpResponse = (HttpServletResponse)
278                         objectModel.get(HttpEnvironment.HTTP_RESPONSE_OBJECT);
279                         httpResponse.sendRedirect(redirectURL);
280                         return;
281                 }
282             }
283 
284             // Was a bitstream found?
285             if (bitstream == null)
286             {
287                 throw new ResourceNotFoundException("Unable to locate bitstream");
288             }
289 
290             // Is there a User logged in and does the user have access to read it?
291             boolean isAuthorized = AuthorizeManager.authorizeActionBoolean(context, bitstream, Constants.READ);
292             if (item != null && item.isWithdrawn() && !AuthorizeManager.isAdmin(context))
293             {
294                 isAuthorized = false;
295                 log.info(LogManager.getHeader(context, "view_bitstream", "handle=" + item.getHandle() + ",withdrawn=true"));
296             }
297 
298             if (!isAuthorized)
299             {
300                 if(context.getCurrentUser() != null){
301                         // A user is logged in, but they are not authorized to read this bitstream,
302                         // instead of asking them to login again we'll point them to a friendly error
303                         // message that tells them the bitstream is restricted.
304                         String redictURL = request.getContextPath() + "/handle/";
305                         if (item!=null){
306                                 redictURL += item.getHandle();
307                         }
308                         else if(dso!=null){
309                                 redictURL += dso.getHandle();
310                         }
311                         redictURL += "/restricted-resource?bitstreamId=" + bitstream.getID();
312 
313                         HttpServletResponse httpResponse = (HttpServletResponse)
314                         objectModel.get(HttpEnvironment.HTTP_RESPONSE_OBJECT);
315                         httpResponse.sendRedirect(redictURL);
316                         return;
317                 }
318                 else{
319 
320                         // The user does not have read access to this bitstream. Inturrupt this current request
321                         // and then forward them to the login page so that they can be authenticated. Once that is
322                         // successfull they will request will be resumed.
323                         AuthenticationUtil.interruptRequest(objectModel, AUTH_REQUIRED_HEADER, AUTH_REQUIRED_MESSAGE, null);
324 
325                         // Redirect
326                         String redictURL = request.getContextPath() + "/login";
327 
328                         HttpServletResponse httpResponse = (HttpServletResponse)
329                         objectModel.get(HttpEnvironment.HTTP_RESPONSE_OBJECT);
330                         httpResponse.sendRedirect(redictURL);
331                         return;
332                 }
333             }
334                 
335             // Success, bitstream found and the user has access to read it.
336             // Store these for later retreval:
337             this.bitstreamInputStream = bitstream.retrieve();
338             this.bitstreamSize = bitstream.getSize();
339             this.bitstreamMimeType = bitstream.getFormat().getMIMEType();
340             this.bitstreamName = bitstream.getName();
341             if (context.getCurrentUser() == null)
342             {
343                 this.isAnonymouslyReadable = true;
344             }
345             else
346             {
347                 this.isAnonymouslyReadable = false;
348                 for (ResourcePolicy rp : AuthorizeManager.getPoliciesActionFilter(context, bitstream, Constants.READ))
349                 {
350                     if (rp.getGroupID() == 0)
351                         this.isAnonymouslyReadable = true;
352                 }
353             }
354 
355             // Trim any path information from the bitstream
356             if (bitstreamName != null && bitstreamName.length() >0 )
357             {
358                         int finalSlashIndex = bitstreamName.lastIndexOf("/");
359                         if (finalSlashIndex > 0)
360                         {
361                                 bitstreamName = bitstreamName.substring(finalSlashIndex+1);
362                         }
363             }
364             else
365             {
366                 // In-case there is no bitstream name...
367                 bitstreamName = "bitstream";
368             }
369             
370             // Log that the bitstream has been viewed, this is none-cached and the complexity
371             // of adding it to the sitemap for every possible bitstre uri is not very tractable
372             new DSpace().getEventService().fireEvent(
373                                 new UsageEvent(
374                                                 UsageEvent.Action.VIEW,
375                                                 ObjectModelHelper.getRequest(objectModel),
376                                                 ContextUtil.obtainContext(ObjectModelHelper.getRequest(objectModel)),
377                                                 bitstream));
378 
379         }
380         catch (SQLException sqle)
381         {
382             throw new ProcessingException("Unable to read bitstream.",sqle);
383         }
384         catch (AuthorizeException ae)
385         {
386             throw new ProcessingException("Unable to read bitstream.",ae);
387         }
388     }

394     /**
395      * Find the bitstream identified by a sequence number on this item.
396      *
397      * @param item A DSpace item
398      * @param sequence The sequence of the bitstream
399      * @return The bitstream or null if none found.
400      */
401     private Bitstream findBitstreamBySequence(Item item, int sequence) throws SQLException
402     {
403         if (item == null)
404                 return null;
405         
406         Bundle[] bundles = item.getBundles();
407         for (Bundle bundle : bundles)
408         {
409             Bitstream[] bitstreams = bundle.getBitstreams();
410 
411             for (Bitstream bitstream : bitstreams)
412             {
413                 if (bitstream.getSequenceID() == sequence)
414                 {
415                         return bitstream;
416                 }
417             }
418         }
419         return null;
420     }

422     /**
423      * Return the bitstream from the given item that is identified by the
424      * given name. If the name has prepended directories they will be removed
425      * one at a time until a bitstream is found. Note that if two bitstreams
426      * have the same name then the first bitstream will be returned.
427      *
428      * @param item A DSpace item
429      * @param name The name of the bitstream
430      * @return The bitstream or null if none found.
431      */
432     private Bitstream findBitstreamByName(Item item, String name) throws SQLException
433     {
434         if (name == null || item == null)
435                 return null;
436     
437         // Determine our the maximum number of directories that will be removed for a path.
438         int maxDepthPathSearch = 3;
439         if (ConfigurationManager.getProperty("xmlui.html.max-depth-guess") != null)
440                 maxDepthPathSearch = ConfigurationManager.getIntProperty("xmlui.html.max-depth-guess");
441         
442         // Search for the named bitstream on this item. Each time through the loop
443         // a directory is removed from the name until either our maximum depth is
444         // reached or the bitstream is found. Note: an extra pass is added on to the
445         // loop for a last ditch effort where all directory paths will be removed.
446         for (int i = 0; i < maxDepthPathSearch+1; i++)
447         {
448                 // Search through all the bitstreams and see
449                 // if the name can be found
450                 Bundle[] bundles = item.getBundles();
451                 for (Bundle bundle : bundles)
452                 {
453                     Bitstream[] bitstreams = bundle.getBitstreams();
454         
455                     for (Bitstream bitstream : bitstreams)
456                     {
457                         if (name.equals(bitstream.getName()))
458                         {
459                                 return bitstream;
460                         }
461                     }
462                 }
463                 
464                 // The bitstream was not found, so try removing a directory
465                 // off of the name and see if we lost some path information.
466                 int indexOfSlash = name.indexOf('/');
467                 
468                 if (indexOfSlash < 0)
469                         // No more directories to remove from the path, so return null for no
470                         // bitstream found.
471                         return null;
472                
473                 name = name.substring(indexOfSlash+1);
474                 
475                 // If this is our next to last time through the loop then
476                 // trim everything and only use the trailing filename.
477                 if (i == maxDepthPathSearch-1)
478                 {
479                         int indexOfLastSlash = name.lastIndexOf('/');
480                         if (indexOfLastSlash > -1)
481                                 name = name.substring(indexOfLastSlash+1);
482                 }
483                 
484         }
485         
486         // The named bitstream was not found and we exausted our the maximum path depth that
487         // we search.
488         return null;
489     }

492     /**
493          * Write the actual data out to the response.
494          *
495          * Some implementation notes,
496          *
497          * 1) We set a short expires time just in the hopes of preventing someone
498          * from overloading the server by clicking reload a bunch of times. I
499          * realize that this is nowhere near 100% effective but it may help in some
500          * cases and shouldn't hurt anything.
501          *
502          * 2) We accept partial downloads, thus if you lose a connection half way
503          * through most web browser will enable you to resume downloading the
504          * bitstream.
505          */
506     public void generate() throws IOException, SAXException,
507             ProcessingException
508     {
509         if (this.bitstreamInputStream == null)
510                 return;
511         
512         // Only allow If-Modified-Since protocol if request is from a spider
513         // since response headers would encourage a browser to cache results
514         // that might change with different authentication..
515         if (isSpider)
516         {
517             // Check for if-modified-since header -- ONLY if not authenticated
518             long modSince = request.getDateHeader("If-Modified-Since");
519             if (modSince != -1 && item != null && item.getLastModified().getTime() < modSince)
520             {
521                 // Item has not been modified since requested date,
522                 // hence bitstream has not; return 304
523                 response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
524                 return;
525             }
526         }
527 
528         // Only set Last-Modified: header for spiders or anonymous
529         // access, since it might encourage browse to cache the result
530         // which might leave a result only available to authenticated
531         // users in the cache for a response later to anonymous user.
532         try
533         {
534             if (item != null && (isSpider || ContextUtil.obtainContext(request).getCurrentUser() == null))
535             {
536                 // TODO:  Currently just borrow the date of the item, since
537                 // we don't have last-mod dates for Bitstreams
538                 response.setDateHeader("Last-Modified", item.getLastModified()
539                         .getTime());
540             }
541         }
542         catch (SQLException e)
543         {
544             throw new ProcessingException(e);
545         }
546 
547         byte[] buffer = new byte[BUFFER_SIZE];
548         int length = -1;
549 
550         // Only encourage caching if this is not a restricted resource, i.e.
551         // if it is accessed anonymously or is readable by Anonymous:
552         if (isAnonymouslyReadable)
553             response.setDateHeader("Expires", System.currentTimeMillis()
554                     + expires);
555         
556         // If this is a large bitstream then tell the browser it should treat it as a download.
557         int threshold = ConfigurationManager.getIntProperty("xmlui.content_disposition_threshold");
558         if (bitstreamSize > threshold && threshold != 0)
559         {
560                 String name  = bitstreamName;
561                 
562                 // Try and make the download file name formated for each browser.
563                 try {
564                         String agent = request.getHeader("USER-AGENT");
565                         if (agent != null && agent.contains("MSIE"))
566                                 name = URLEncoder.encode(name,"UTF8");
567                         else if (agent != null && agent.contains("Mozilla"))
568                                 name = MimeUtility.encodeText(name, "UTF8", "B");
569                 }
570                 catch (UnsupportedEncodingException see)
571                 {
572                         // do nothing
573                 }
574                 response.setHeader("Content-Disposition", "attachment;filename=" + name);
575         }
576 
577         // Turn off partial downloads, they cause problems
578         // and are only rarely used. Specifically some windows pdf
579         // viewers are incapable of handling this request. By
580         // uncommenting the following two lines you will turn this feature back on.
581         // response.setHeader("Accept-Ranges", "bytes");
582         // String ranges = request.getHeader("Range");
583         String ranges = null;
584         
585 
586         ByteRange byteRange = null;
587         if (ranges != null)
588         {
589             try
590             {
591                 ranges = ranges.substring(ranges.indexOf('=') + 1);
592                 byteRange = new ByteRange(ranges);
593             }
594             catch (NumberFormatException e)
595             {
596                 byteRange = null;
597                 if (response instanceof HttpResponse)
598                 {
599                     // Respond with status 416 (Request range not
600                     // satisfiable)
601                     ((HttpResponse) response).setStatus(416);
602                 }
603             }
604         }
605 
606         if (byteRange != null)
607         {
608             String entityLength;
609             String entityRange;
610             if (this.bitstreamSize != -1)
611             {
612                 entityLength = "" + this.bitstreamSize;
613                 entityRange = byteRange.intersection(
614                         new ByteRange(0, this.bitstreamSize)).toString();
615             }
616             else
617             {
618                 entityLength = "*";
619                 entityRange = byteRange.toString();
620             }
621 
622             response.setHeader("Content-Range", entityRange + "/"
623                     + entityLength);
624             if (response instanceof HttpResponse)
625             {
626                 // Response with status 206 (Partial content)
627                 ((HttpResponse) response).setStatus(206);
628             }
629 
630             int pos = 0;
631             int posEnd;
632             while ((length = this.bitstreamInputStream.read(buffer)) > -1)
633             {
634                 posEnd = pos + length - 1;
635                 ByteRange intersection = byteRange
636                         .intersection(new ByteRange(pos, posEnd));
637                 if (intersection != null)
638                 {
639                     out.write(buffer, (int) intersection.getStart()
640                             - pos, (int) intersection.length());
641                 }
642                 pos += length;
643             }
644         }
645         else
646         {
647             response.setHeader("Content-Length", String
648                     .valueOf(this.bitstreamSize));
649 
650             while ((length = this.bitstreamInputStream.read(buffer)) > -1)
651             {
652                 out.write(buffer, 0, length);
653             }
654             out.flush();
655         }
656     }
