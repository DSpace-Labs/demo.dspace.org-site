108 /**
109  *
110  * Generate a syndication feed for DSpace, either a community or collection
111  * or the whole repository. This code was adapted from the syndication found
112  * in DSpace's JSP implementation, "org.dspace.app.webui.servlet.FeedServlet".
113  *
114  * Once thing that has been modified from DSpace's JSP implementation is what
115  * is placed inside an item's description, we've changed it so that the list
116  * of metadata fields is scanned until a value is found and the first one
117  * found is used as the description. This means that we look at the abstract,
118  * description, alternative title, title, etc... to and the first metadata
119  * value found is used.
120  *
121  * I18N: Feed's are internationalized, meaning that they may contain refrences
122  * to messages contained in the global messages.xml file using cocoon's i18n
123  * schema. However the library used to build the feeds does not understand
124  * this schema to work around this limitation I created a little hack. It
125  * basicaly works like this, when text that needs to be localized is put into
126  * the feed it is allways mangled such that a prefix is added to the messages's
127  * key. Thus if the key were "xmlui.feed.text" then the resulting text placed
128  * into the feed would be "I18N:xmlui.feed.text". After the library is finished
129  * and produced it's final result the output is traversed to find these
130  * occurances ande replace them with proper cocoon i18n elements.
131  *
132  *
133  *
134  * @author Scott Phillips, Ben Bosman, Richard Rodgers
135  */
136

187     /**
188      * Generate the cache validity object.
189      *
190      * The validity object will include the collection being viewed and
191      * all recently submitted items. This does not include the community / collection
192      * hierarch, when this changes they will not be reflected in the cache.
193      */
194     public SourceValidity getValidity()
195     {
196         if (this.validity == null)
197         {
198             try
199             {
200                 DSpaceValidity validity = new FeedValidity();
201                 
202                 Context context = ContextUtil.obtainContext(objectModel);
203 
204                 DSpaceObject dso = null;
205                 
206                 if (handle != null && !handle.contains("site"))
207                     dso = HandleManager.resolveToObject(context, handle);
208                 
209                 validity.add(dso);
210                 
211                 // add reciently submitted items
212                 for(Item item : getRecentlySubmittedItems(context,dso))
213                 {
214                     validity.add(item);
215                 }
216 
217                 this.validity = validity.complete();
218             }
219             catch (Exception e)
220             {
221                 // Just ignore all errors and return an invalid cache.
222             }
223         }
224         return this.validity;
225     }

252     /**
253      * Generate the syndication feed.
254      */
255     public void generate() throws IOException, SAXException, ProcessingException
256     {
257         try
258         {
259             Context context = ContextUtil.obtainContext(objectModel);
260             DSpaceObject dso = null;
261             
262             if (handle != null && !handle.contains("site"))
263             {
264                 dso = HandleManager.resolveToObject(context, handle);
265                 if (dso == null)
266                 {
267                     // If we were unable to find a handle then return page not found.
268                     throw new ResourceNotFoundException("Unable to find DSpace object matching the given handle: "+handle);
269                 }
270                 
271                 if (!(dso.getType() == Constants.COLLECTION || dso.getType() == Constants.COMMUNITY))
272                 {
273                     // The handle is valid but the object is not a container.
274                     throw new ResourceNotFoundException("Unable to syndicate DSpace object: "+handle);
275                 }
276             }
277         
278             SyndicationFeed feed = new SyndicationFeed(SyndicationFeed.UITYPE_XMLUI);
279             feed.populate(ObjectModelHelper.getRequest(objectModel),
280                           dso, getRecentlySubmittedItems(context,dso), FeedUtils.i18nLabels);
281             feed.setType(this.format);
282             Document dom = feed.outputW3CDom();
283             FeedUtils.unmangleI18N(dom);
284             DOMStreamer streamer = new DOMStreamer(contentHandler, lexicalHandler);
285             streamer.stream(dom);
286         }
287         catch (IllegalArgumentException iae)
288         {
289                 throw new ResourceNotFoundException("Syndication feed format, '"+this.format+"', is not supported.");
290         }
291         catch (FeedException fe)
292         {
293                 throw new SAXException(fe);
294         }
295         catch (SQLException sqle)
296         {
297                 throw new SAXException(sqle);
298         }
299     }

301     /**
302      * @return recently submitted Items within the indicated scope
303      */
304     @SuppressWarnings("unchecked")
305     private Item[] getRecentlySubmittedItems(Context context, DSpaceObject dso)
306             throws SQLException
307     {
308         if (recentSubmissionItems != null)
309                 return recentSubmissionItems;
310 
311         String source = ConfigurationManager.getProperty("recent.submissions.sort-option");
312         BrowserScope scope = new BrowserScope(context);
313         if (dso instanceof Collection)
314                 scope.setCollection((Collection) dso);
315         else if (dso instanceof Community)
316                 scope.setCommunity((Community) dso);
317         scope.setResultsPerPage(itemCount);
318 
319         // FIXME Exception handling
320         try
321         {
322             scope.setBrowseIndex(BrowseIndex.getItemBrowseIndex());
323             for (SortOption so : SortOption.getSortOptions())
324             {
325                 if (so.getName().equals(source))
326                 {
327                     scope.setSortBy(so.getNumber());
328                     scope.setOrder(SortOption.DESCENDING);
329                 }
330             }
331 
332             BrowseEngine be = new BrowseEngine(context);
333             this.recentSubmissionItems = be.browseMini(scope).getItemResults(context);
334 
335             // filter out Items taht are not world-readable
336             if (!includeRestrictedItems)
337             {
338                 List<Item> result = new ArrayList<Item>();
339                 for (Item item : this.recentSubmissionItems)
340                 {
341                 checkAccess:
342                     for (Group group : AuthorizeManager.getAuthorizedGroups(context, item, Constants.READ))
343                     {
344                         if ((group.getID() == 0))
345                         {
346                             result.add(item);
347                             break checkAccess;
348                         }
349                     }
350                 }
351                 this.recentSubmissionItems = result.toArray(new Item[result.size()]);
352             }
353         }
354         catch (BrowseException bex)
355         {
356             log.error("Caught browse exception", bex);
357         }
358         catch (SortException e)
359         {
360             log.error("Caught sort exception", e);
361         }
362         return this.recentSubmissionItems;
363     }

378     /**
379      * Extend the standard DSpaceValidity object to support assumed
380      * caching. Since feeds will constantly be requested we want to
381      * assume that a feed is still valid instead of checking it
382      * against the database anew everytime.
383      *
384      * This validity object will assume that a cache is still valid,
385      * without rechecking it, for 24 hours.
386      *
387      */
388     private class FeedValidity extends DSpaceValidity
389     {
390                 private static final long serialVersionUID = 1L;
391                         
392         /** When the cache's validity expires */
393         private long expires = 0;
394         
395         /**
396          * When the validity is completed record a timestamp to check later.
397          */
398         public DSpaceValidity complete()
399         {
400                 this.expires = System.currentTimeMillis() + CACHE_AGE;
401                 
402                 return super.complete();
403         }
404         
405         
406         /**
407          * Determine if the cache is still valid
408          */
409         public int isValid()
410         {
411             // Return true if we have a hash.
412             if (this.completed)
413             {
414                 if (System.currentTimeMillis() < this.expires)
415                 {
416                         // If the cache hasn't expired the just assume that it is still valid.
417                         return SourceValidity.VALID;
418                 }
419                 else
420                 {
421                         // The cache is past its age
422                         return SourceValidity.UNKNOWN;
423                 }
424             }
425             else
426             {
427                 // This is an error, state. We are being asked whether we are valid before
428                 // we have been initialized.
429                 return SourceValidity.INVALID;
430             }
431         }
432 
433         /**
434          * Determine if the cache is still valid based
435          * upon the other validity object.
436          *
437          * @param other
438          *          The other validity object.
439          */
440         public int isValid(SourceValidity otherValidity)
441         {
442             if (this.completed)
443             {
444                 if (otherValidity instanceof FeedValidity)
445                 {
446                     FeedValidity other = (FeedValidity) otherValidity;
447                     if (hash == other.hash)
448                     {
449                         // Update both cache's expiration time.
450                         this.expires = System.currentTimeMillis() + CACHE_AGE;
451                         other.expires = System.currentTimeMillis() + CACHE_AGE;
452                         
453                         return SourceValidity.VALID;
454                     }
455                 }
456             }
457 
458             return SourceValidity.INVALID;
459         }
