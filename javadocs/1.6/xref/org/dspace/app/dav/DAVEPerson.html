160     /**
161      * If authenticated user is an Administrator, show all epersons, otherwise
162      * just show current user (if there is one).
163      * 
164      * @return array of children, all the EPerson records on site.
165      * 
166      * @throws SQLException the SQL exception
167      */
168     @Override
169     protected DAVResource[] children() throws SQLException
170     {
171         if (AuthorizeManager.isAdmin(this.context))
172         {
173             EPerson all[] = EPerson.findAll(this.context, EPerson.ID);
174             DAVResource result[] = new DAVResource[all.length];
175             log.debug("EPerson children(), got " + String.valueOf(all.length)
176                     + " e-people.");
177             for (int i = 0; i < all.length; ++i)
178             {
179                 result[i] = new DAVEPersonEPerson(this.context, this.request, this.response,
180                         makeChildPath(DAVEPersonEPerson.getPathElt(all[i]
181                                 .getID())), all[i]);
182             }
183             return result;
184         }
185         EPerson self = this.context.getCurrentUser();
186         if (self == null)
187         {
188             return new DAVResource[0];
189         }
190         else
191         {
192             DAVResource result[] = new DAVResource[1];
193             result[0] = new DAVEPersonEPerson(this.context, this.request, this.response,
194                     makeChildPath(DAVEPersonEPerson.getPathElt(self.getID())),
195                     self);
196             return result;
197         }
198     }

200     /* (non-Javadoc)
201      * @see org.dspace.app.dav.DAVResource#propfindInternal(org.jdom.Element)
202      */
203     @Override
204     protected Element propfindInternal(Element property) throws SQLException,
205             AuthorizeException, IOException, DAVStatusException
206     {
207         String value = null;
208 
209         // displayname - title or handle.
210         if (elementsEqualIsh(property, displaynameProperty))
211         {
212             value = this.pathElt[0];
213         }
214         else
215         {
216             return commonPropfindInternal(property, true);
217         }
218 
219         // value was set up by "if" clause:
220         if (value == null)
221         {
222             throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
223                     "Not found.");
224         }
225         Element p = new Element(property.getName(), property.getNamespace());
226         p.setText(filterForXML(value));
227         return p;
228     }
