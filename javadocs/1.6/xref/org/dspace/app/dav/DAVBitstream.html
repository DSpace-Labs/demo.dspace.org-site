200     /**
201      * Attempt to locate Bitstream object from URI. pathElt is
202      * "bitstream_{sid}.ext" or "retrieve_{db-id}.ext"
203      * 
204      * @param context the context
205      * @param item the item
206      * @param pathElt the path elt
207      * 
208      * @return the bitstream found (any errors throw an exception)
209      * 
210      * @throws SQLException the SQL exception
211      * @throws DAVStatusException the DAV status exception
212      */
213     protected static Bitstream findBitstream(Context context, Item item,
214             String pathElt) throws SQLException, DAVStatusException
215     {
216         try
217         {
218             // get rid of extension, if any, e.g. ".pdf"
219             int dot = pathElt.indexOf('.');
220             String strId = (dot >= 0) ? pathElt.substring(0, dot) : new String(
221                     pathElt);
222             Bitstream result = null;
223 
224             if (strId.startsWith("bitstream_"))
225             {
226                 strId = strId.substring(10);
227                 result = getBitstreamBySequenceID(item, Integer.parseInt(strId));
228             }
229             else if (strId.startsWith("retrieve_"))
230             {
231                 strId = strId.substring(9);
232                 result = Bitstream.find(context, Integer.parseInt(strId));
233             }
234             else
235             {
236                 throw new DAVStatusException(
237                         HttpServletResponse.SC_BAD_REQUEST,
238                         "Unrecognized bitstream URI format.");
239             }
240             if (result == null)
241             {
242                 throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
243                         "No bitstream at this sequence ID: " + pathElt);
244             }
245             return result;
246         }
247         catch (NumberFormatException nfe)
248         {
249             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
250                     "Invalid Bitstream Sequence ID in URI: " + pathElt);
251         }
252     }

254     /**
255      * Find bitstream with matching sequence id.
256      * 
257      * @param item the item
258      * @param sid the sid
259      * 
260      * @return bitstream, or null if none found.
261      * 
262      * @throws SQLException the SQL exception
263      */
264     protected static Bitstream getBitstreamBySequenceID(Item item, int sid)
265             throws SQLException
266     {
267 
268         Bundle[] bundles = item.getBundles();
269         for (Bundle element : bundles)
270         {
271             Bitstream[] bitstreams = element.getBitstreams();
272 
273             for (Bitstream element0 : bitstreams)
274             {
275                 if (sid == element0.getSequenceID())
276                 {
277                     return element0;
278                 }
279             }
280         }
281         return null;
282     }

303      * @see org.dspace.app.dav.DAVDSpaceObject#propfindInternal(org.jdom.Element)
304      */
305     @Override
306     protected Element propfindInternal(Element property) throws SQLException,
307             AuthorizeException, IOException, DAVStatusException
308     {
309         String value = null;
310 
311         /*
312          * FIXME: This implements permission check that really belongs in
313          * business logic. Although communities and collections don't check for
314          * read auth, Bitstream may contain sensitive data and should always
315          * check for READ permission.
316          */
317         AuthorizeManager.authorizeAction(this.context, this.bitstream, Constants.READ);
318 
319         // displayname - title or handle.
320         if (elementsEqualIsh(property, displaynameProperty))
321         {
322             value = this.bitstream.getName();
323             if (value == null)
324             {
325                 value = makeDisplayname();
326             }
327         }
328         else if (elementsEqualIsh(property, getcontentlengthProperty))
329         {
330             value = String.valueOf(this.bitstream.getSize());
331         }
332         else if (elementsEqualIsh(property, getcontenttypeProperty))
333         {
334             value = this.bitstream.getFormat().getMIMEType();
335         }
336         else if (elementsEqualIsh(property, sourceProperty))
337         {
338             value = this.bitstream.getSource();
339         }
340         else if (elementsEqualIsh(property, descriptionProperty))
341         {
342             value = this.bitstream.getDescription();
343         }
344         else if (elementsEqualIsh(property, formatProperty))
345         {
346             BitstreamFormat bsf = this.bitstream.getFormat();
347             value = bsf == null ? null : bsf.getShortDescription();
348         }
349         else if (elementsEqualIsh(property, format_descriptionProperty))
350         {
351             value = this.bitstream.getFormatDescription();
352         }
353         else if (elementsEqualIsh(property, checksumProperty))
354         {
355             value = this.bitstream.getChecksum();
356         }
357         else if (elementsEqualIsh(property, checksum_algorithmProperty))
358         {
359             value = this.bitstream.getChecksumAlgorithm();
360         }
361         else if (elementsEqualIsh(property, sequence_idProperty))
362         {
363             int sid = this.bitstream.getSequenceID();
364             if (sid >= 0)
365             {
366                 value = String.valueOf(sid);
367             }
368         }
369         else if (elementsEqualIsh(property, bundleProperty))
370         {
371             Bundle bn[] = this.bitstream.getBundles();
372             if (bn != null && bn.length > 0)
373             {
374                 value = bn[0].getName();
375             }
376         }
377         else
378         {
379             return super.propfindInternal(property);
380         }
381 
382         // value was set up by "if" clause:
383         if (value == null)
384         {
385             throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
386                     "Not found.");
387         }
388         Element p = new Element(property.getName(), property.getNamespace());
389         p.setText(filterForXML(value));
390         return p;
391     }

393     /* (non-Javadoc)
394      * @see org.dspace.app.dav.DAVResource#proppatchInternal(int, org.jdom.Element)
395      */
396     @Override
397     protected int proppatchInternal(int action, Element prop)
398             throws SQLException, AuthorizeException, IOException,
399             DAVStatusException
400     {
401         Namespace ns = prop.getNamespace();
402         String propName = prop.getName();
403         boolean nsDspace = ns != null && ns.equals(DAV.NS_DSPACE);
404         String newValue = (action == DAV.PROPPATCH_REMOVE) ? null : prop
405                 .getText();
406 
407         // displayname - arbitrary string
408         if (elementsEqualIsh(prop, displaynameProperty))
409         {
410             this.bitstream.setName(newValue);
411         }
412         else if (nsDspace && propName.equals("description"))
413         {
414             this.bitstream.setDescription(newValue);
415         }
416         else if (nsDspace && propName.equals("source"))
417         {
418             this.bitstream.setSource(newValue);
419         }
420         else if (nsDspace && propName.equals("format_description"))
421         {
422             this.bitstream.setUserFormatDescription(newValue);
423         }
424         else if (nsDspace && propName.equals("format"))
425         {
426             if (action == DAV.PROPPATCH_REMOVE)
427             {
428                 throw new DAVStatusException(DAV.SC_CONFLICT,
429                         "The format property cannot be removed.");
430             }
431             BitstreamFormat bsf = BitstreamFormat.findByShortDescription(
432                     this.context, newValue);
433             if (bsf == null)
434             {
435                 throw new DAVStatusException(DAV.SC_CONFLICT,
436                         "Cannot set format, no such Bitstream Format: "
437                                 + newValue);
438             }
439             this.bitstream.setFormat(bsf);
440         }
441         else
442         {
443             throw new DAVStatusException(DAV.SC_CONFLICT, "The "
444                     + prop.getName() + " property cannot be changed.");
445         }
446 
447         // this assumes we got through an IF clause and changed something:
448         this.bitstream.update();
449         return HttpServletResponse.SC_OK;
450     }

455     @Override
456     protected void get() throws SQLException, AuthorizeException,
457             IOException, DAVStatusException
458     {
459         if (this.bitstream == null)
460         {
461             throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
462                     "Bitstream not found, URI=\"" + hrefURL() + "\"");
463         }
464         else
465         {
466             if (this.item != null)
467             {
468                 log.info(LogManager.getHeader(this.context, "DAV GET Bitstream",
469                         "item handle=" + this.item.getHandle() + ", bitstream_id="
470                                 + this.bitstream.getID()));
471             }
472 
473             // Set the response MIME type
474             this.response.setContentType(this.bitstream.getFormat().getMIMEType());
475 
476             // Response length
477             this.response.setHeader("Content-Length", String.valueOf(this.bitstream
478                     .getSize()));
479 
480             // Pipe the bits
481             InputStream is = this.bitstream.retrieve();
482             Utils.bufferedCopy(is, this.response.getOutputStream());
483             is.close();
484             this.response.getOutputStream().flush();
485         }
486     }

560     /**
561      * Returns an XML representation of a bitstream -- either inline content or
562      * a link reference. The XML looks like:
563      * 
564      * <pre>
565      * &lt;dspace:bitstream&gt;
566      * &lt;dspace:link href=&quot;url-to-bitstream&quot;&gt;
567      * &lt;/dspace:bitstream&gt;
568      * ...or...
569      * &lt;dspace:bitstream&gt;
570      * &lt;dspace:content contenttype=&quot;image/gif&quot; contentlength=&quot;299&quot; contentencoding=&quot;base64&quot;&gt;
571      * ...text of base64..
572      * &lt;/dspace:content&gt;
573      * &lt;/dspace:bitstream&gt;
574      * NOTE: contentlength is the DECODED length of the content.
575      * </pre>
576      * 
577      * Used by the "logo" property on collections and communities.
578      * 
579      * @param bitstream the bitstream
580      * @param resource the resource
581      * 
582      * @return the element
583      * 
584      * @throws AuthorizeException the authorize exception
585      * @throws SQLException the SQL exception
586      * @throws IOException Signals that an I/O exception has occurred.
587      */
588     protected static Element makeXmlBitstream(Bitstream bitstream,
589             DAVResource resource) throws AuthorizeException, SQLException,
590             IOException
591     {
592         Element b = new Element("bitstream", DAV.NS_DSPACE);
593         long length = bitstream.getSize();
594         BitstreamFormat bf = bitstream.getFormat();
595 
596         if (length > BITSTREAM_INLINE_THRESHOLD)
597         {
598             Element e = new Element("link", DAV.NS_DSPACE);
599             e.setAttribute("href", resource.hrefPrefix() + "retrieve_"
600                     + String.valueOf(bitstream.getID()));
601             b.addContent(e);
602         }
603         else
604         {
605             Element e = new Element("content", DAV.NS_DSPACE);
606             if (bf != null)
607             {
608                 e.setAttribute("contenttype", bf.getMIMEType());
609             }
610             e.setAttribute("contentlength", String.valueOf(length));
611             e.setAttribute("contentencoding", "base64");
612             b.addContent(e);
613 
614             // write encoding of bitstream contents
615             ByteArrayOutputStream baos = new ByteArrayOutputStream((int) length);
616             Utils.copy(bitstream.retrieve(), baos);
617             e.setText(new String(Base64.encodeBase64(baos.toByteArray())));
618         }
619         return b;
620     }

622     /**
623      * Extract bitstream from the XML representation, i.e.
624      * 
625      * <pre>
626      * &lt;dspace:bitstream&gt;
627      * &lt;dspace:content contenttype=&quot;image/gif&quot;
628      * contentlength=&quot;299&quot;
629      * contentencoding=&quot;base64&quot;&gt;
630      * ...text of base64..
631      * &lt;/dspace:content&gt;
632      * &lt;/dspace:bitstream&gt;
633      * </pre>
634      * 
635      * In the above format, contenttype and contentencoding attributes of
636      * content are REQUIRED.
637      * 
638      * @param context the context
639      * @param xb the xb
640      * 
641      * @return inputstream of the contents of the data, or null on error.
642      */
643     protected static InputStream getXmlBitstreamContent(Context context,
644             Element xb)
645     {
646         Element c = xb.getChild("content", DAV.NS_DSPACE);
647         if (c != null)
648         {
649             String enc = c.getAttributeValue("contentencoding");
650             if (enc != null && enc.equals("base64"))
651             {
652                 byte value[] = Base64.decodeBase64(c.getText().getBytes());
653                 return new ByteArrayInputStream(value);
654             }
655         }
656         return null;
657     }

659     /**
660      * Get the content-type from an XML-encoded bitstream.
661      * 
662      * @param context required for reading the DB.
663      * @param xb XML bitstream representation in JDOM.
664      * 
665      * @return First BitstreamFormat matching content-type string, or null if
666      * none.
667      * 
668      * @throws SQLException the SQL exception
669      */
670     protected static BitstreamFormat getXmlBitstreamFormat(Context context,
671             Element xb) throws SQLException
672     {
673         Element c = xb.getChild("content", DAV.NS_DSPACE);
674         if (c != null)
675         {
676             String ctype = c.getAttributeValue("contenttype");
677             if (ctype != null)
678             {
679                 BitstreamFormat af[] = BitstreamFormat.findAll(context);
680                 for (BitstreamFormat element : af)
681                 {
682                     if (ctype.equals(element.getMIMEType()))
683                     {
684                         return element;
685                     }
686                 }
687             }
688         }
689         return null;
690     }
