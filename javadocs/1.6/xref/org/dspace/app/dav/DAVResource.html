77  /**
78   * Superclass for all DSpace "resources" in the WebDAV interface. Maps DAV
79   * operations onto DSpace objects. An instance is created for one HTTP request
80   * and discarded thereafter.
81   * <p>
82   * This class has the high-level driver for each DAV (or SOAP) method. It calls
83   * on subclasses for implementation details, e.g. how to get properties in and
84   * out of each type of resource, PUT, COPY, etc.
85   * <p>
86   * All of the interpretation (and generation) of DAV Resource URIs should be in
87   * this class, in methods such as <code>findResource()</code>,
88   * <code>hrefURL()</code> etc, calling on the <code>matchResource()</code>
89   * methods in some subclasses for help. NOTE: <code>matchResource</code>
90   * should be an abstract method but it cannot be since it is necessarily static.
91   * <p>
92   * DAVResource instances are generally only created by the
93   * <code>findResource()</code> method and the <code>children()</code>
94   * methods in resource subclasses.
95   */

349     /**
350      * Get resource named by a DAV URI. The URI can be relative (from the Site)
351      * or absolute if it starts with the WebDAV server's prefix.
352      * 
353      * @param uri the uri
354      * 
355      * @return resource object or null upon error.
356      * 
357      * @throws IOException Signals that an I/O exception has occurred.
358      * @throws SQLException the SQL exception
359      * @throws DAVStatusException the DAV status exception
360      * @throws AuthorizeException the authorize exception
361      */
362     protected DAVResource URIToResource(String uri) throws IOException,
363             SQLException, DAVStatusException, AuthorizeException
364     {
365         String dest = uri;
366         String prefix = hrefPrefix();
367         if (dest.startsWith(prefix))
368         {
369             dest = dest.substring(prefix.length());
370         }
371         if (dest.startsWith("/"))
372         {
373             dest = dest.substring(1);
374         }
375         return findResource(this.context, this.request, this.response, dest.split("/"));
376     }

378     /**
379      * Creates a resource object corresponding to the path.
380      * 
381      * @param context the context
382      * @param request the request
383      * @param response the response
384      * @param pathElt the path elt
385      * 
386      * @return resource, or null if path cannot be interpreted. Sends HTTP
387      * status in the event of failure.
388      * 
389      * @throws IOException Signals that an I/O exception has occurred.
390      * @throws SQLException the SQL exception
391      * @throws DAVStatusException the DAV status exception
392      * @throws AuthorizeException the authorize exception
393      */
394     protected static DAVResource findResource(Context context,
395             HttpServletRequest request, HttpServletResponse response,
396             String pathElt[]) throws IOException, SQLException,
397             DAVStatusException, AuthorizeException
398     {
399         DAVResource result = null;
400 
401         if ((result = DAVSite.matchResourceURI(context, request, response,
402                 pathElt)) == null
403                 && (result = DAVLookup.matchResourceURI(context, request,
404                         response, pathElt)) == null
405                 && (result = DAVWorkspace.matchResourceURI(context, request,
406                         response, pathElt)) == null
407                 && (result = DAVWorkflow.matchResourceURI(context, request,
408                         response, pathElt)) == null
409                 && (result = DAVEPerson.matchResourceURI(context, request,
410                         response, pathElt)) == null
411                 && (result = DAVItem.matchResourceURI(context, request,
412                         response, pathElt)) == null
413                 && (result = DAVBitstream.matchResourceURI(context, request,
414                         response, pathElt)) == null
415                 && (result = DAVDSpaceObject.matchResourceURI(context, request,
416                         response, pathElt)) == null)
417         {
418             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
419                     "Unrecognized DSpace resource URI");
420         }
421         return result;
422     }

426     /**
427      * Construct URL prefix up to the resource path, for making up "href" values
428      * in e.g. PROPFIND responses.
429      * 
430      * @return start of resource URI ending with a '/'.
431      */
432     protected String hrefPrefix()
433     {
434         // Note: contextPath and servletPath include leading slashes but no
435         // trailing..
436         if (this.request == null)
437         {
438             return "/";
439         }
440         else
441         {
442             return this.request.getScheme() + "://" + this.request.getServerName() + ":"
443                     + String.valueOf(this.request.getServerPort())
444                     + this.request.getContextPath() + this.request.getServletPath() + "/";
445         }
446     }
447 
448     /**
449      * Construct the whole absolute URL to resource - prefix plus path elements.
450      * 
451      * @return the string
452      */
453     protected String hrefURL()
454     {
455         StringBuffer result = new StringBuffer(hrefPrefix());
456         for (int i = 0; i < this.pathElt.length; ++i)
457         {
458             if (i + 1 < this.pathElt.length)
459             {
460                 result.append(this.pathElt[i] + "/");
461             }
462             else
463             {
464                 result.append(this.pathElt[i]);
465             }
466         }
467 
468         return result.toString();
469     }

529     /**
530      * PROPFIND method service: Collect parameters and launch the recursive
531      * generic propfind driver which in turn calls resource methods.
532      * 
533      * @throws SQLException the SQL exception
534      * @throws AuthorizeException the authorize exception
535      * @throws IOException Signals that an I/O exception has occurred.
536      * @throws DAVStatusException the DAV status exception
537      */
538     protected void propfind() throws SQLException,
539             AuthorizeException, IOException, DAVStatusException
540     {
541         // set all incoming encoding to UTF-8
542         this.request.setCharacterEncoding("UTF-8");
543 
544         /*
545          * FIXME:(?) this is technically wrong, wrt. WebDAV protocol, but it's
546          * more efficient; default depth should be DAV.DAV_INFINITY; we cheat
547          * and make it 0.
548          */
549         int depth = 0;
550 
551         String sdepth = this.request.getHeader("Depth");
552         if (sdepth != null)
553         {
554             sdepth = sdepth.trim();
555             try
556             {
557                 if (sdepth.equalsIgnoreCase("infinity"))
558                 {
559                     depth = DAV.DAV_INFINITY;
560                 }
561                 else
562                 {
563                     depth = Integer.parseInt(sdepth);
564                 }
565             }
566             catch (NumberFormatException nfe)
567             {
568                 throw new DAVStatusException(
569                         HttpServletResponse.SC_BAD_REQUEST,
570                         "Bad Depth header: " + sdepth);
571             }
572         }
573 
574         // get object-type mask from request query args, e.g.
575         // type=ITEM,type=BITSTREAM ...
576         String types[] = this.request.getParameterValues("type");
577         int typeMask = typesToMask(types);
578 
579         Document outdoc = propfindDriver(depth, this.request.getInputStream(),
580                 typeMask);
581         if (outdoc != null)
582         {
583             this.response.setStatus(DAV.SC_MULTISTATUS);
584             this.response.setContentType("text/xml");
585 
586             outputRaw.output(outdoc, this.response.getOutputStream());
587 
588             if (debugXML)
589             {
590                 log.debug("PROPFIND response = "
591                         + outputPretty.outputString(outdoc));
592             }
593         }
594     }

596     /**
597      * Inner logic for propfind. Shared with SOAP servlet
598      * 
599      * @param depth the depth
600      * @param pfDoc the pf doc
601      * @param typeMask the type mask
602      * 
603      * @return the document
604      * 
605      * @throws SQLException the SQL exception
606      * @throws AuthorizeException the authorize exception
607      * @throws IOException Signals that an I/O exception has occurred.
608      * @throws DAVStatusException the DAV status exception
609      */
610     protected Document propfindDriver(int depth, InputStream pfDoc, int typeMask)
611             throws SQLException, AuthorizeException, IOException,
612             DAVStatusException
613     {
614         // When there is no document, type defaults to <allprop>.
615         int pfType = DAV.PROPFIND_ALLPROP;
616         List pfProps = null;
617 
618         try
619         {
620             SAXBuilder builder = new SAXBuilder();
621             Document reqdoc = builder.build(pfDoc);
622             Element propfind = reqdoc.getRootElement();
623             if (!propfind.getName().equals("propfind")
624                     || !propfind.getNamespace().equals(DAV.NS_DAV))
625             {
626                 throw new DAVStatusException(
627                         HttpServletResponse.SC_BAD_REQUEST,
628                         "Bad Root Element, must be propfind in DAV:");
629             }
630 
631             // Propfind child is be ONE of: allprop | propname | prop
632             // if "prop", also get list of type names.
633             List pfChild = propfind.getChildren();
634             if (pfChild.size() > 0)
635             {
636                 Element child0 = (Element) pfChild.get(0);
637                 String rawType = child0.getName();
638                 if (rawType.equalsIgnoreCase("prop"))
639                 {
640                     pfType = DAV.PROPFIND_PROP;
641                     pfProps = child0.getChildren();
642                 }
643                 else if (rawType.equalsIgnoreCase("allprop"))
644                 {
645                     pfType = DAV.PROPFIND_ALLPROP;
646                 }
647                 else if (rawType.equalsIgnoreCase("propname"))
648                 {
649                     pfType = DAV.PROPFIND_PROPNAME;
650                 }
651                 else
652                 {
653                     log.warn(LogManager.getHeader(this.context, "propfind",
654                             "Unknown TYPE child of <propfind>, named: \""
655                                     + rawType + "\""));
656                 }
657             }
658 
659             if (debugXML)
660             {
661                 log.debug("PROPFIND request = "
662                         + outputPretty.outputString(reqdoc));
663             }
664         }
665         catch (JDOMParseException je)
666         {
667             // if there is no document we get error at line -1, so let it pass.
668             if (je.getLineNumber() >= 0)
669             {
670                 log.error(LogManager.getHeader(this.context, "propfind", je
671                         .toString()));
672                 throw new DAVStatusException(
673                         HttpServletResponse.SC_BAD_REQUEST,
674                         "Could not parse request document: " + je.toString());
675             }
676         }
677         catch (JDOMException je)
678         {
679             log.error(LogManager.getHeader(this.context, "propfind", je.toString()));
680             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
681                     "Could not parse request document: " + je.toString());
682         }
683 
684         // At this point, pfProps, pfType and URI define the whole request.
685 
686         // Construct response XML
687         Element multistatus = new Element("multistatus", DAV.NS_DAV);
688         Document outdoc = new Document(multistatus);
689 
690         propfindCrawler(multistatus, pfType, pfProps, depth, typeMask, 0);
691         return outdoc;
692     }

694     // Recursive propfind driver: each call to
695     // resource.propfindInternal accumulates <response>s in multistatus.
696     /**
697      * Propfind crawler.
698      * 
699      * @param multistatus the multistatus
700      * @param pfType the pf type
701      * @param pfProps the pf props
702      * @param depth the depth
703      * @param typeMask the type mask
704      * @param count the count
705      * 
706      * @return the int
707      * 
708      * @throws DAVStatusException the DAV status exception
709      * @throws SQLException the SQL exception
710      * @throws AuthorizeException the authorize exception
711      * @throws IOException Signals that an I/O exception has occurred.
712      */
713     private int propfindCrawler(Element multistatus, int pfType, List pfProps,
714             int depth, int typeMask, int count) throws DAVStatusException,
715             SQLException, AuthorizeException, IOException
716     {
717         if ((this.type & typeMask) != 0 || this.type == TYPE_OTHER)
718         {
719             // check the count of resources visited
720             ++count;
721             if (propfindResourceLimit > 0 && count > propfindResourceLimit)
722             {
723                 throw new DAVStatusException(HttpServletResponse.SC_CONFLICT,
724                         "PROPFIND request exceeded server's limit on number of resources to query.");
725             }
726 
727             String uri = hrefURL();
728             log.debug("PROPFIND returning (count=" + String.valueOf(count)
729                     + ") href=" + uri);
730             Vector notFound = new Vector();
731             Vector forbidden = new Vector();
732             Element responseElt = new Element("response", DAV.NS_DAV);
733             multistatus.addContent(responseElt);
734 
735             Element href = new Element("href", DAV.NS_DAV);
736             href.setText(uri);
737             responseElt.addContent(href);
738 
739             // just get the names
740             if (pfType == DAV.PROPFIND_PROPNAME)
741             {
742                 responseElt.addContent(makePropstat(
743                         copyElementList(getAllProperties()),
744                         HttpServletResponse.SC_OK, "OK"));
745             }
746             else
747             {
748                 List success = new LinkedList();
749                 List props = (pfType == DAV.PROPFIND_ALLPROP) ? getAllProperties()
750                         : pfProps;
751                 ListIterator pi = props.listIterator();
752                 while (pi.hasNext())
753                 {
754                     Element property = (Element) pi.next();
755                     try
756                     {
757                         Element value = propfindInternal(property);
758                         if (value != null)
759                         {
760                             success.add(value);
761                         }
762                         else
763                         {
764                             notFound.add((Element) property.clone());
765                         }
766                     }
767                     catch (DAVStatusException se)
768                     {
769                         if (se.getStatus() == HttpServletResponse.SC_NOT_FOUND)
770                         {
771                             notFound.add((Element) property.clone());
772                         }
773                         else
774                         {
775                             responseElt.addContent(makePropstat(
776                                     (Element) property.clone(), se.getStatus(),
777                                     se.getMessage()));
778                         }
779                     }
780                     catch (AuthorizeException ae)
781                     {
782                         forbidden.add((Element) property.clone());
783                     }
784                 }
785 
786                 if (success.size() > 0)
787                 {
788                     responseElt.addContent(makePropstat(success,
789                             HttpServletResponse.SC_OK, "OK"));
790                 }
791                 if (notFound.size() > 0)
792                 {
793                     responseElt.addContent(makePropstat(notFound,
794                             HttpServletResponse.SC_NOT_FOUND, "Not found"));
795                 }
796                 if (forbidden.size() > 0)
797                 {
798                     responseElt
799                             .addContent(makePropstat(forbidden,
800                                     HttpServletResponse.SC_FORBIDDEN,
801                                     "Not authorized."));
802                 }
803             }
804         }
805 
806         // recurse on children; filter on types first.
807         // child types are all lower bits than this type, so (type - 1) is
808         // bit-mask of all the possible child types. Skip recursion if
809         // all child types would be masked out anyway.
810         if (depth != 0 && ((this.type - 1) & typeMask) != 0)
811         {
812             DAVResource[] kids = children();
813             for (DAVResource element : kids)
814             {
815                 count = element.propfindCrawler(multistatus, pfType, pfProps,
816                         depth == DAV.DAV_INFINITY ? depth : depth - 1,
817                         typeMask, count);
818             }
819         }
820 
821         return count;
822     }

853     /**
854      * Inner logic for proppatch. Shared with SOAP servlet
855      * 
856      * @param docStream the doc stream
857      * 
858      * @return the document
859      * 
860      * @throws SQLException the SQL exception
861      * @throws AuthorizeException the authorize exception
862      * @throws IOException Signals that an I/O exception has occurred.
863      * @throws DAVStatusException the DAV status exception
864      */
865     protected Document proppatchDriver(InputStream docStream)
866             throws SQLException, AuthorizeException, IOException,
867             DAVStatusException
868     {
869 
870         Document reqdoc = null;
871         try
872         {
873             SAXBuilder builder = new SAXBuilder();
874             reqdoc = builder.build(docStream);
875         }
876         catch (JDOMParseException je)
877         {
878             // if there is no document we get error at line -1, so let it pass.
879             if (je.getLineNumber() >= 0)
880             {
881                 log.error(LogManager.getHeader(this.context, "proppatch", je
882                         .toString()));
883                 throw new DAVStatusException(
884                         HttpServletResponse.SC_BAD_REQUEST,
885                         "Could not parse PROPERTYUPDATE request document: "
886                                 + je.toString());
887             }
888         }
889         catch (JDOMException je)
890         {
891             log
892                     .error(LogManager.getHeader(this.context, "proppatch", je
893                             .toString()));
894             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
895                     "Could not parse PROPERTYUPDATE request document: "
896                             + je.toString());
897         }
898         if (reqdoc == null)
899         {
900             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
901                     "Failed to parse any valid PROPERTYUPDATE document in request.");
902         }
903 
904         Element pupdate = reqdoc.getRootElement();
905         if (!pupdate.getName().equals("propertyupdate")
906                 || !pupdate.getNamespace().equals(DAV.NS_DAV))
907         {
908             log.warn(LogManager.getHeader(this.context, "proppatch",
909                     "Got bad root element, XML=" + pupdate.toString()));
910             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
911                     "Bad Root Element, must be propertyupdate in DAV:");
912         }
913 
914         Element multistatus = new Element("multistatus", DAV.NS_DAV);
915         Element msResponse = new Element("response", DAV.NS_DAV);
916         multistatus.addContent(msResponse);
917         Element href = new Element("href", DAV.NS_DAV);
918         msResponse.addContent(href);
919         href.addContent(hrefURL());
920 
921         // result status and accumulation:
922         boolean failing = false;
923         List failedDep = new LinkedList();
924         List success = new LinkedList();
925 
926         // process the SET and REMOVE elements under PROPERTYUPDATE:
927         ListIterator ci = pupdate.getChildren().listIterator();
928         while (ci.hasNext())
929         {
930             int action = -1;
931             Element e = (Element) ci.next();
932             if (e.getName().equals("set")
933                     && e.getNamespace().equals(DAV.NS_DAV))
934             {
935                 action = DAV.PROPPATCH_SET;
936             }
937             else if (e.getName().equals("remove")
938                     && e.getNamespace().equals(DAV.NS_DAV))
939             {
940                 action = DAV.PROPPATCH_REMOVE;
941             }
942             else
943             {
944                 log.warn(LogManager.getHeader(this.context, "proppatch",
945                         "Got unrecognized PROPERTYUPDATE element:"
946                                 + e.toString()));
947                 continue;
948             }
949             // PROP elements under SET or REMOVE
950             ListIterator pi = e.getChildren().listIterator();
951             while (pi.hasNext())
952             {
953                 Element p = (Element) pi.next();
954                 if (p.getName().equals("prop")
955                         && p.getNamespace().equals(DAV.NS_DAV))
956                 {
957                     ListIterator propi = p.getChildren().listIterator();
958                     while (propi.hasNext())
959                     {
960                         Element thisprop = (Element) propi.next();
961 
962                         // if this PROPPATCH request is failing, just
963                         // accumulate properties for Failed Dependency status
964                         if (failing)
965                         {
966                             failedDep.add(new Element(thisprop.getName(),
967                                     thisprop.getNamespace()));
968                         }
969                         else
970                         {
971                             int status = 0;
972                             String statusMsg = null;
973                             try
974                             {
975                                 status = proppatchCommonInternal(action,
976                                         thisprop);
977                             }
978                             catch (DAVStatusException se)
979                             {
980                                 status = se.getStatus();
981                                 statusMsg = se.getMessage();
982                             }
983                             catch (AuthorizeException ae)
984                             {
985                                 status = HttpServletResponse.SC_FORBIDDEN;
986                                 statusMsg = "Permission denied.";
987                             }
988                             if (status == HttpServletResponse.SC_OK)
989                             {
990                                 success.add(new Element(thisprop.getName(),
991                                         thisprop.getNamespace()));
992                                 log.debug("proppatch SET/REMOVE OK, action="
993                                         + String.valueOf(action) + ", prop="
994                                         + thisprop.toString());
995                             }
996                             else
997                             {
998                                 failing = true;
999                                 msResponse.addContent(makePropstat(thisprop,
1000                                         status, statusMsg));
1001                                 log
1002                                         .debug("proppatch SET/REMOVE FAILED with status="
1003                                                 + String.valueOf(status)
1004                                                 + ", on prop="
1005                                                 + thisprop.toString());
1006                             }
1007                         }
1008                     }
1009                 }
1010                 else
1011                 {
1012                     log.warn(LogManager.getHeader(this.context, "proppatch",
1013                             "No PROP element where expected, found:"
1014                                     + p.toString()));
1015                 }
1016             }
1017         }
1018 
1019         // add success and failure propstat elements to response
1020         if (failing)
1021         {
1022             failedDep.addAll(success);
1023             if (failedDep.size() > 0)
1024             {
1025                 msResponse.addContent(makePropstat(failedDep,
1026                         DAV.SC_FAILED_DEPENDENCY,
1027                         "Failed because another property failed."));
1028             }
1029             this.context.abort();
1030         }
1031         else
1032         {
1033             if (success.size() > 0)
1034             {
1035                 msResponse.addContent(makePropstat(success,
1036                         HttpServletResponse.SC_OK, "OK"));
1037             }
1038             this.context.complete();
1039         }
1040         return new Document(multistatus);
1041 
1042     }

1137     /**
1138      * Service routine for COPY HTTP request.
1139      * 
1140      * @throws IOException Signals that an I/O exception has occurred.
1141      * @throws SQLException the SQL exception
1142      * @throws AuthorizeException the authorize exception
1143      * @throws DAVStatusException the DAV status exception
1144      */
1145     protected void copy() throws IOException, SQLException, AuthorizeException,
1146             DAVStatusException
1147     {
1148         // Destination arg from header
1149         String destination = this.request.getHeader("Destination");
1150         if (destination == null)
1151         {
1152             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
1153                     "Missing the required request header \"Destination\"");
1154         }
1155 
1156         // Fix a common misfeature in clients: they will append
1157         // the final pathname element of the "source" URI to the
1158         // "destination" URI, which is misleading in our URI scheme so
1159         // we have to strip it off:
1160         try
1161         {
1162             String srcPath = (new URI(this.request.getRequestURI())).getPath();
1163             String destPath = (new URI(destination)).getPath();
1164             int slash = srcPath.lastIndexOf("/");
1165             if (slash > -1)
1166             {
1167                 String lastElt = srcPath.substring(slash);
1168                 if (destPath.endsWith(lastElt))
1169                 {
1170                     destination = destination.substring(0, destination.length()
1171                             - lastElt.length() + 1);
1172                 }
1173             }
1174             log.debug("Copy dest. URI repair: srcPath=" + srcPath
1175                     + ", destPath=" + destPath + ", final dest=" + destination);
1176         }
1177         catch (URISyntaxException e)
1178         {
1179             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
1180                     "Illegal URI syntax in value of \"Destination\" header: "
1181                             + destination);
1182         }
1183 
1184         // Depth arg from header
1185         int depth = DAV.DAV_INFINITY;
1186         String sdepth = this.request.getHeader("Depth");
1187         if (sdepth != null)
1188         {
1189             sdepth = sdepth.trim();
1190             try
1191             {
1192                 if (sdepth.equalsIgnoreCase("infinity"))
1193                 {
1194                     depth = DAV.DAV_INFINITY;
1195                 }
1196                 else
1197                 {
1198                     depth = Integer.parseInt(sdepth);
1199                 }
1200             }
1201             catch (NumberFormatException nfe)
1202             {
1203                 throw new DAVStatusException(
1204                         HttpServletResponse.SC_BAD_REQUEST,
1205                         "Illegal value in Depth request header: " + sdepth);
1206             }
1207         }
1208 
1209         // overwrite header logic
1210         boolean overwrite = false;
1211         String soverwrite = this.request.getHeader("Overwrite");
1212         if (soverwrite != null && soverwrite.trim().equalsIgnoreCase("T"))
1213         {
1214             overwrite = true;
1215         }
1216 
1217         // keepProperties - extract from XML doc in request, if any..
1218         boolean keepProperties = false;
1219         Document reqdoc = null;
1220         try
1221         {
1222             SAXBuilder builder = new SAXBuilder();
1223             reqdoc = builder.build(this.request.getInputStream());
1224         }
1225         catch (JDOMParseException je)
1226         {
1227             // if there is no document we get error at line -1, so let it pass.
1228             if (je.getLineNumber() >= 0)
1229             {
1230                 throw new DAVStatusException(
1231                         HttpServletResponse.SC_BAD_REQUEST,
1232                         "Error parsing XML document in COPY request.");
1233             }
1234         }
1235         catch (JDOMException je)
1236         {
1237             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
1238                     "Error parsing XML document in COPY request: "
1239                             + je.toString());
1240         }
1241         if (reqdoc != null)
1242         {
1243             Element propertybehavior = reqdoc.getRootElement();
1244             Namespace ns = propertybehavior.getNamespace();
1245             if (!(ns != null && ns.equals(DAV.NS_DAV) && propertybehavior
1246                     .getName().equals("propertybehavior")))
1247             {
1248                 throw new DAVStatusException(
1249                         HttpServletResponse.SC_BAD_REQUEST,
1250                         "Illegal XML document in COPY request, root= "
1251                                 + propertybehavior.toString());
1252             }
1253 
1254             // FIXME: (?) Punt on parsing exact list of properties to
1255             // "keepalive" since we don't implement it anyway.
1256             if (propertybehavior.getChild("keepalive", DAV.NS_DAV) != null)
1257             {
1258                 keepProperties = true;
1259             }
1260             else if (propertybehavior.getChild("omit", DAV.NS_DAV) == null)
1261             {
1262                 throw new DAVStatusException(
1263                         HttpServletResponse.SC_BAD_REQUEST,
1264                         "Illegal propertybehavior document in COPY request, no omit or keepalive child.");
1265             }
1266         }
1267 
1268         int result = copyDriver(destination, depth, overwrite, keepProperties);
1269         if (result >= 200 && result < 300)
1270         {
1271             this.response.setStatus(result);
1272         }
1273         else
1274         {
1275             throw new DAVStatusException(result, "COPY Failed.");
1276         }
1277     }

1318     /**
1319      * Make bitmask out resource-type keywords. Used by DAV and SOAP interfaces.
1320      * 
1321      * @param types the types
1322      * 
1323      * @return binary mask of types to allow
1324      * 
1325      * @throws DAVStatusException if an unrecognized type keyword is given.
1326      */
1327     protected static int typesToMask(String types[]) throws DAVStatusException
1328     {
1329         int typeMask = 0;
1330         if (types == null || types.length == 0)
1331         {
1332             typeMask = TYPE_ALL;
1333         }
1334         else
1335             for (String element : types)
1336             {
1337                 String key = element.trim();
1338                 if (key.equalsIgnoreCase("SITE"))
1339                 {
1340                     typeMask |= TYPE_SITE;
1341                 }
1342                 else if (key.equalsIgnoreCase("COMMUNITY"))
1343                 {
1344                     typeMask |= TYPE_COMMUNITY;
1345                 }
1346                 else if (key.equalsIgnoreCase("COLLECTION"))
1347                 {
1348                     typeMask |= TYPE_COLLECTION;
1349                 }
1350                 else if (key.equalsIgnoreCase("ITEM"))
1351                 {
1352                     typeMask |= TYPE_ITEM;
1353                 }
1354                 else if (key.equalsIgnoreCase("BITSTREAM"))
1355                 {
1356                     typeMask |= TYPE_BITSTREAM;
1357                 }
1358                 else
1359                 {
1360                     throw new DAVStatusException(
1361                             HttpServletResponse.SC_BAD_REQUEST,
1362                             "Unrecognized type keyword: " + key);
1363                 }
1364             }
1365         return typeMask;
1366     }
