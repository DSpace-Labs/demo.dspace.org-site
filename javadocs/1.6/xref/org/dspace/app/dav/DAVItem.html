159     /**
160      * Match Item URIs that identify the item by a database ID. Handle URIs are
161      * matched by DAVDSpaceObject.
162      * 
163      * @param context the context
164      * @param request the request
165      * @param response the response
166      * @param pathElt the path elt
167      * 
168      * @return the DAV resource
169      * 
170      * @throws DAVStatusException the DAV status exception
171      * @throws SQLException the SQL exception
172      */
173     protected static DAVResource matchResourceURI(Context context,
174             HttpServletRequest request, HttpServletResponse response,
175             String pathElt[]) throws DAVStatusException, SQLException
176     {
177         int id = -1;
178         String bsElt = null;
179 
180         try
181         {
182             // The "/item_db_" element in last or next-to-last element
183             if (pathElt[pathElt.length - 1].startsWith("item_db_"))
184             {
185                 id = Integer.parseInt(pathElt[pathElt.length - 1].substring(8));
186             }
187             else if (pathElt[pathElt.length - 1].startsWith("bitstream_")
188                     && pathElt.length > 1
189                     && pathElt[pathElt.length - 2].startsWith("item_db_"))
190             {
191                 id = Integer.parseInt(pathElt[pathElt.length - 2].substring(8));
192                 bsElt = pathElt[pathElt.length - 1];
193             }
194             if (id >= 0)
195             {
196                 Item item = Item.find(context, id);
197                 if (item == null)
198                 {
199                     throw new DAVStatusException(
200                             HttpServletResponse.SC_NOT_FOUND, "Item with ID="
201                                     + String.valueOf(id) + " not found.");
202                 }
203                 if (bsElt != null)
204                 {
205                     Bitstream bs = DAVBitstream.findBitstream(context, item,
206                             bsElt);
207                     if (bs == null)
208                     {
209                         throw new DAVStatusException(
210                                 HttpServletResponse.SC_NOT_FOUND,
211                                 "Bitstream not found.");
212                     }
213                     return new DAVBitstream(context, request, response,
214                             pathElt, item, bs);
215                 }
216                 else
217                 {
218                     return new DAVItem(context, request, response, pathElt,
219                             item);
220                 }
221             }
222             return null;
223         }
224         catch (NumberFormatException ne)
225         {
226             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
227                     "Error parsing number in request URI.");
228         }
229     }

264     /**
265      * Return this resource's children. Item's children are its bitstreams.
266      * 
267      * @return the DAV resource[]
268      * 
269      * @throws SQLException the SQL exception
270      */
271     @Override
272     protected DAVResource[] children() throws SQLException
273     {
274         // Check for overall read permission on Item
275         if (!AuthorizeManager.authorizeActionBoolean(this.context, this.item,
276                 Constants.READ))
277         {
278             return new DAVResource[0];
279         }
280 
281         Vector result = new Vector();
282         Bundle[] bundles = this.item.getBundles();
283         for (Bundle element : bundles)
284         {
285             // check read permission on this Bundle
286             if (!AuthorizeManager.authorizeActionBoolean(this.context, element,
287                     Constants.READ))
288             {
289                 continue;
290             }
291 
292             Bitstream[] bitstreams = element.getBitstreams();
293             for (Bitstream element0 : bitstreams)
294             {
295                 String ext[] = element0.getFormat().getExtensions();
296                 result.add(new DAVBitstream(this.context, this.request, this.response,
297                         makeChildPath(DAVBitstream.getPathElt(element0
298                                 .getSequenceID(), ext.length < 1 ? null
299                                 : ext[0])), this.item, element0));
300             }
301         }
302         return (DAVResource[]) result.toArray(new DAVResource[result.size()]);
303     }

305     /* (non-Javadoc)
306      * @see org.dspace.app.dav.DAVDSpaceObject#propfindInternal(org.jdom.Element)
307      */
308     @Override
309     protected Element propfindInternal(Element property) throws SQLException,
310             AuthorizeException, IOException, DAVStatusException
311     {
312         String value = null;
313 
314         /*
315          * FIXME: This implements permission check that really belongs in
316          * business logic. Although communities and collections don't check for
317          * read auth, Item may contain sensitive data and should always check
318          * for READ permission. Exception: allow "withdrawn" property to be
319          * checked regardless of authorization since all permissions are removed
320          * when item is withdrawn.
321          */
322         if (!elementsEqualIsh(property, withdrawnProperty))
323         {
324             AuthorizeManager.authorizeAction(this.context, this.item, Constants.READ);
325         }
326 
327         if (elementsEqualIsh(property, withdrawnProperty))
328         {
329             value = String.valueOf(this.item.isWithdrawn());
330         }
331         else if (elementsEqualIsh(property, displaynameProperty))
332         {
333             // displayname - title or handle.
334             DCValue titleDc[] = this.item.getDC("title", Item.ANY, Item.ANY);
335             value = titleDc.length > 0 ? titleDc[0].value : this.item.getHandle();
336         }
337         else if (elementsEqualIsh(property, handleProperty))
338         {
339             value = canonicalizeHandle(this.item.getHandle());
340         }
341         else if (elementsEqualIsh(property, submitterProperty))
342         {
343             EPerson ep = this.item.getSubmitter();
344             if (ep != null)
345             {
346                 value = hrefToEPerson(ep);
347             }
348         }
349         else if (elementsEqualIsh(property, owning_collectionProperty))
350         {
351             Collection owner = this.item.getOwningCollection();
352             if (owner != null)
353             {
354                 value = canonicalizeHandle(owner.getHandle());
355             }
356         }
357         else if (elementsEqualIsh(property, getlastmodifiedProperty))
358         {
359             value = DAV.HttpDateFormat.format(this.item.getLastModified());
360         }
361         else if (elementsEqualIsh(property, licenseProperty))
362         {
363             value = getLicenseAsString();
364         }
365         else if (elementsEqualIsh(property, cc_license_textProperty))
366         {
367             value = CreativeCommons.getLicenseText(this.item);
368         }
369         else if (elementsEqualIsh(property, cc_license_rdfProperty))
370         {
371             value = CreativeCommons.getLicenseRDF(this.item);
372         }
373         else if (elementsEqualIsh(property, cc_license_urlProperty))
374         {
375             value = CreativeCommons.getLicenseURL(this.item);
376         }
377         else
378         {
379             return super.propfindInternal(property);
380         }
381 
382         // value was set up by "if" clause:
383         if (value == null)
384         {
385             throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
386                     "Not found.");
387         }
388         Element p = new Element(property.getName(), property.getNamespace());
389         p.setText(filterForXML(value));
390         return p;
391     }

393     /**
394      * Get the license for this Item as String. License is the first bitstream
395      * named "license.txt" in a LICENSE bundle, apparently?
396      * <p>
397      * FIXME: is this correct? there's no counterexample..
398      * 
399      * @return license string, or null if none found.
400      * 
401      * @throws SQLException the SQL exception
402      * @throws AuthorizeException the authorize exception
403      * @throws IOException Signals that an I/O exception has occurred.
404      */
405     private String getLicenseAsString() throws SQLException,
406             AuthorizeException, IOException
407     {
408         Bundle lb[] = this.item.getBundles(Constants.LICENSE_BUNDLE_NAME);
409         for (Bundle element : lb)
410         {
411             Bitstream lbs = element.getBitstreamByName("license.txt");
412             if (lbs != null)
413             {
414                 ByteArrayOutputStream baos = new ByteArrayOutputStream(
415                         (int) lbs.getSize());
416                 Utils.copy(lbs.retrieve(), baos);
417                 return baos.toString();
418             }
419         }
420         return null;
421     }

423     /* (non-Javadoc)
424      * @see org.dspace.app.dav.DAVResource#proppatchInternal(int, org.jdom.Element)
425      */
426     @Override
427     protected int proppatchInternal(int action, Element prop)
428             throws SQLException, AuthorizeException, IOException,
429             DAVStatusException
430     {
431         // Don't need any authorization checks since the Item layer
432         // checks authorization for write and delete.
433 
434         Namespace ns = prop.getNamespace();
435         String propName = prop.getName();
436 
437         // "submitter" is the only Item-specific mutable property, rest are
438         // live and unchangeable.
439         if (ns != null && ns.equals(DAV.NS_DSPACE)
440                 && propName.equals("submitter"))
441         {
442             if (action == DAV.PROPPATCH_REMOVE)
443             {
444                 throw new DAVStatusException(DAV.SC_CONFLICT,
445                         "The submitter property cannot be removed.");
446             }
447             String newName = prop.getText();
448             EPerson ep = EPerson.findByEmail(this.context, newName);
449             if (ep == null)
450             {
451                 throw new DAVStatusException(DAV.SC_CONFLICT,
452                         "Cannot set submitter, no EPerson found for email address: "
453                                 + newName);
454             }
455             this.item.setSubmitter(ep);
456             this.item.update();
457             return HttpServletResponse.SC_OK;
458         }
459         throw new DAVStatusException(DAV.SC_CONFLICT, "The " + prop.getName()
460                 + " property cannot be changed.");
461     }

463     /**
464      * GET implementation returns the contents of the Item as a package. The
465      * query arg "package" must be specified.
466      * 
467      * @throws SQLException the SQL exception
468      * @throws AuthorizeException the authorize exception
469      * @throws ServletException the servlet exception
470      * @throws IOException Signals that an I/O exception has occurred.
471      * @throws DAVStatusException the DAV status exception
472      */
473     @Override
474     protected void get() throws SQLException, AuthorizeException,
475             IOException, DAVStatusException
476     {
477         // Check for overall read permission on Item, because nothing else will
478         AuthorizeManager.authorizeAction(this.context, this.item, Constants.READ);
479 
480         String packageType = this.request.getParameter("package");
481         Bundle[] original = this.item.getBundles("ORIGINAL");
482         int bsid;
483 
484         if (packageType == null)
485         {
486             packageType = "default";
487         }
488         PackageDisseminator dip = (PackageDisseminator) PluginManager
489                 .getNamedPlugin(PackageDisseminator.class, packageType);
490         if (dip == null)
491         {
492             throw new DAVStatusException(HttpServletResponse.SC_BAD_REQUEST,
493                     "Cannot find a disseminate plugin for package="
494                             + packageType);
495         }
496         else
497         {
498             try
499             {
500                 PackageParameters pparams = PackageParameters.create(this.request);
501                 this.response.setContentType(dip.getMIMEType(pparams));
502                 dip.disseminate(this.context, this.item, pparams, this.response
503                         .getOutputStream());
504             }
505             catch (CrosswalkException pe)
506             {
507                 throw new DAVStatusException(
508                         HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
509                         "Failed in crosswalk of metadata: " + pe.toString());
510             }
511             catch (PackageException pe)
512             {
513                 pe.log(log);
514                 throw new DAVStatusException(
515                         HttpServletResponse.SC_INTERNAL_SERVER_ERROR, pe
516                                 .toString());
517             }
518         }
519     }

557     /**
558      * Do the work of "copy" method; this code is shared with e.g.
559      * DAVInProgressSubmission.
560      * 
561      * @param context the context
562      * @param item the item
563      * @param destination the destination
564      * @param overwrite the overwrite
565      * 
566      * @return HTTP status code.
567      * 
568      * @throws DAVStatusException the DAV status exception
569      * @throws SQLException the SQL exception
570      * @throws AuthorizeException the authorize exception
571      * @throws IOException Signals that an I/O exception has occurred.
572      */
573     protected static int addItemToCollection(Context context, Item item,
574             DAVResource destination, boolean overwrite)
575             throws DAVStatusException, SQLException, AuthorizeException,
576             IOException
577     {
578         // sanity checks
579         if (!(destination instanceof DAVCollection))
580         {
581             throw new DAVStatusException(
582                     HttpServletResponse.SC_METHOD_NOT_ALLOWED,
583                     "COPY of Item is only allowed when destination is a DSpace Collection.");
584         }
585 
586         // access check
587         AuthorizeManager.authorizeAction(context, item, Constants.READ);
588 
589         // make sure item doesn't belong to this collection
590         Collection destColl = ((DAVCollection) destination).getCollection();
591 
592         log.debug("COPY from=" + item.toString() + " (" + item.getHandle()
593                 + "), to=" + destColl.toString() + " (" + destColl.getHandle()
594                 + ")");
595 
596         // check if it's already a member
597         Collection refs[] = item.getCollections();
598         for (Collection element : refs)
599         {
600             if (destColl.equals(element))
601             {
602                 log.debug("COPY - item @ " + item.getHandle()
603                         + " is already a member of collection @ "
604                         + destColl.getHandle());
605                 if (overwrite)
606                 {
607                     return DAV.SC_NO_CONTENT;
608                 }
609                 else
610                 {
611                     throw new DAVStatusException(DAV.SC_CONFLICT,
612                             "This Item is already a member of collection handle="
613                                     + destColl.getHandle());
614                 }
615             }
616         }
617 
618         destColl.addItem(item);
619         return DAV.SC_NO_CONTENT;
620     }
