60  /**
61   * The "workflow_*" resources are collections (in the DAV sense) of all the
62   * current user's <code>WorkflowItem</code>s. Each one is nothing more than a
63   * read-only collection to list the items.
64   * <p>
65   * The <code>workflow_own</code> resource lists items <em>owned</em> by the
66   * current user, while the <code>workflow_pool</code> lists items in the
67   * workflow pool which may be acquired by the current user.
68   * <p>
69   * Its children are all the relevant <code>WorkflowItem</code>s. These
70   * resources cannot be altered.
71   * <p>
72   * 
73   * @author Larry Stone
74   * @see DAVWorkflowItem
75   */

121     /**
122      * Match the URIs this subclass understands and return the corresponding
123      * resource.
124      * 
125      * @param context the context
126      * @param request the request
127      * @param response the response
128      * @param pathElt the path elt
129      * 
130      * @return the DAV resource
131      * 
132      * @throws DAVStatusException the DAV status exception
133      * @throws SQLException the SQL exception
134      */
135     protected static DAVResource matchResourceURI(Context context,
136             HttpServletRequest request, HttpServletResponse response,
137             String pathElt[]) throws DAVStatusException, SQLException
138     {
139         // The "/workflow" URI:
140         if (pathElt.length > 0 && pathElt[0].startsWith("workflow_"))
141         {
142             if (pathElt.length > 1)
143             {
144                 return DAVWorkflowItem.matchResourceURI(context, request,
145                         response, pathElt);
146             }
147             else if (pathElt[0].equals("workflow_own")
148                     || pathElt[0].equals("workflow_pool"))
149             {
150                 return new DAVWorkflow(context, request, response, pathElt);
151             }
152             else
153             {
154                 throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
155                         "Unrecognized URI path element: " + pathElt[0]);
156             }
157         }
158         return null;
159     }

170     /* (non-Javadoc)
171      * @see org.dspace.app.dav.DAVResource#children()
172      */
173     @Override
174     protected DAVResource[] children() throws SQLException
175     {
176 
177         EPerson ep = this.context.getCurrentUser();
178         if (ep != null)
179         {
180             List wi = null;
181             if (this.pathElt[0].equals("workflow_own"))
182             {
183                 wi = WorkflowManager.getOwnedTasks(this.context, ep);
184             }
185             else if (this.pathElt[0].equals("workflow_pool"))
186             {
187                 wi = WorkflowManager.getPooledTasks(this.context, ep);
188             }
189             if (wi != null)
190             {
191                 log.debug("children(): Got " + String.valueOf(wi.size())
192                         + " Workflow Items.");
193                 DAVResource result[] = new DAVResource[wi.size()];
194                 ListIterator wii = wi.listIterator();
195                 int i = 0;
196                 while (wii.hasNext())
197                 {
198                     WorkflowItem wfi = (WorkflowItem) wii.next();
199                     result[i++] = new DAVWorkflowItem(this.context, this.request,
200                             this.response, makeChildPath(DAVWorkflowItem
201                                     .getPathElt(wfi.getID())), wfi);
202                 }
203                 return result;
204             }
205         }
206         return new DAVResource[0];
207     }

209     /* (non-Javadoc)
210      * @see org.dspace.app.dav.DAVResource#propfindInternal(org.jdom.Element)
211      */
212     @Override
213     protected Element propfindInternal(Element property) throws SQLException,
214             AuthorizeException, IOException, DAVStatusException
215     {
216         String value = null;
217 
218         // displayname - title or handle.
219         if (elementsEqualIsh(property, displaynameProperty))
220         {
221             value = this.pathElt[0];
222         }
223         else
224         {
225             return commonPropfindInternal(property, true);
226         }
227 
228         // value was set up by "if" clause:
229         if (value == null)
230         {
231             throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
232                     "Not found.");
233         }
234         Element p = new Element(property.getName(), property.getNamespace());
235         p.setText(filterForXML(value));
236         return p;
237     }
