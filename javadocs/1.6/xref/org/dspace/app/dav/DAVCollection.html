155     /* (non-Javadoc)
156      * @see org.dspace.app.dav.DAVResource#children()
157      */
158     @Override
159     protected DAVResource[] children() throws SQLException
160     {
161         Vector result = new Vector();
162         ItemIterator ii = this.collection.getItems();
163         try
164         {
165             while (ii.hasNext())
166             {
167                 Item item = ii.next();
168                 result.add(new DAVItem(this.context, this.request, this.response,
169                         makeChildPath(item), item));
170             }
171         }
172         finally
173         {
174             if (ii != null)
175                 ii.close();
176         }
177         
178         return (DAVResource[]) result.toArray(new DAVResource[result.size()]);
179     }

191     /* (non-Javadoc)
192      * @see org.dspace.app.dav.DAVDSpaceObject#propfindInternal(org.jdom.Element)
193      */
194     @Override
195     protected Element propfindInternal(Element property) throws SQLException,
196             AuthorizeException, IOException, DAVStatusException
197     {
198         String value = null;
199 
200         // displayname - title or handle.
201         if (elementsEqualIsh(property, displaynameProperty))
202         {
203             value = getObjectMetadata("name");
204             if (value == null)
205             {
206                 value = this.collection.getHandle();
207             }
208         }
209 
210         else if (elementsEqualIsh(property, handleProperty))
211         {
212             value = canonicalizeHandle(this.collection.getHandle());
213         }
214         else if (elementsEqualIsh(property, logoProperty))
215         {
216             Bitstream lbs = this.collection.getLogo();
217             Element le;
218             if (lbs != null
219                     && (le = DAVBitstream.makeXmlBitstream(lbs, this)) != null)
220             {
221                 Element p = new Element("logo", DAV.NS_DSPACE);
222                 p.addContent(le);
223                 return p;
224             }
225         }
226 
227         else if (elementsEqualIsh(property, short_descriptionProperty))
228         {
229             value = getObjectMetadata("short_description");
230         }
231         else if (elementsEqualIsh(property, introductory_textProperty))
232         {
233             value = getObjectMetadata("introductory_text");
234         }
235         else if (elementsEqualIsh(property, side_bar_textProperty))
236         {
237             value = getObjectMetadata("side_bar_text");
238         }
239         else if (elementsEqualIsh(property, copyright_textProperty))
240         {
241             value = getObjectMetadata("copyright_text");
242         }
243         else if (elementsEqualIsh(property, default_licenseProperty))
244         {
245             value = this.collection.hasCustomLicense() ? this.collection.getLicense()
246                     : null;
247         }
248         else if (elementsEqualIsh(property, provenance_descriptionProperty))
249         {
250             value = getObjectMetadata("provenance_description");
251         }
252         else
253         {
254             return super.propfindInternal(property);
255         }
256 
257         // value was set up by "if" clause:
258         if (value == null)
259         {
260             throw new DAVStatusException(HttpServletResponse.SC_NOT_FOUND,
261                     "Not found.");
262         }
263         Element p = new Element(property.getName(), property.getNamespace());
264         p.setText(filterForXML(value));
265         return p;
266     }

288     /* (non-Javadoc)
289      * @see org.dspace.app.dav.DAVResource#proppatchInternal(int, org.jdom.Element)
290      */
291     @Override
292     protected int proppatchInternal(int action, Element prop)
293             throws SQLException, AuthorizeException, IOException,
294             DAVStatusException
295     {
296         String newValue = (action == DAV.PROPPATCH_REMOVE) ? null : prop
297                 .getText();
298 
299         // these are "metadata" values..
300         if (elementsEqualIsh(prop, short_descriptionProperty)
301                 || elementsEqualIsh(prop, introductory_textProperty)
302                 || elementsEqualIsh(prop, side_bar_textProperty)
303                 || elementsEqualIsh(prop, copyright_textProperty)
304                 || elementsEqualIsh(prop, provenance_descriptionProperty))
305         {
306             this.collection.setMetadata(prop.getName(), newValue);
307         }
308         else if (elementsEqualIsh(prop, displaynameProperty))
309         {
310             this.collection.setMetadata("name", newValue);
311         }
312         else if (elementsEqualIsh(prop, default_licenseProperty))
313         {
314             this.collection.setLicense(newValue);
315         }
316         else if (elementsEqualIsh(prop, logoProperty))
317         {
318             if (action == DAV.PROPPATCH_REMOVE)
319             {
320                 this.collection.setLogo(null);
321             }
322             else
323             {
324                 Element bs = prop.getChild("bitstream", DAV.NS_DSPACE);
325                 if (bs != null)
326                 {
327                     InputStream bis = DAVBitstream.getXmlBitstreamContent(
328                             this.context, bs);
329                     BitstreamFormat bsf = DAVBitstream.getXmlBitstreamFormat(
330                             this.context, bs);
331                     if (bis == null || bsf == null)
332                     {
333                         throw new DAVStatusException(DAV.SC_CONFLICT,
334                                 "Unacceptable value for logo property.");
335                     }
336                     Bitstream nbs = this.collection.setLogo(bis);
337                     nbs.setFormat(bsf);
338                     nbs.update();
339                 }
340                 else
341                 {
342                     throw new DAVStatusException(DAV.SC_CONFLICT,
343                             "No <bitstream> element value found for logo property.");
344                 }
345             }
346         }
347         else
348         {
349             throw new DAVStatusException(DAV.SC_CONFLICT, "The "
350                     + prop.getName() + " property cannot be changed.");
351         }
352 
353         this.collection.update();
354         return HttpServletResponse.SC_OK;
355     }

368     /**
369      * Wrapped input stream that hits end-of-file after reading a counted number
370      * of bytes, even if its client stream appears to have more data. This fixes
371      * a problem in the Servlet container's input stream which will try to read
372      * past the end of the request body.
373      */
374     private class CountedInputStream extends FilterInputStream
375     {
376         
377         /** The count. */
378         private long count = 0;
379 
380         /** The length. */
381         private long length = -1;
382 
383         /**
384          * Instantiates a new counted input stream.
385          * 
386          * @param is the is
387          * @param length the length
388          */
389         protected CountedInputStream(InputStream is, long length)
390         {
391             super(is);
392             this.length = length;
393         }
394 
395         /* (non-Javadoc)
396          * @see java.io.FilterInputStream#read()
397          */
398         @Override
399         public int read() throws IOException
400         {
401             if (++this.count > this.length)
402             {
403                 return -1;
404             }
405             return super.read();
406         }
407 
408         /* (non-Javadoc)
409          * @see java.io.FilterInputStream#read(byte[])
410          */
411         @Override
412         public int read(byte[] b) throws IOException
413         {
414             if (this.count >= this.length)
415             {
416                 return -1;
417             }
418             int result = super.read(b);
419             if (this.count > 0)
420             {
421                 this.count += result;
422             }
423             return result;
424         }
425 
426         /* (non-Javadoc)
427          * @see java.io.FilterInputStream#read(byte[], int, int)
428          */
429         @Override
430         public int read(byte[] b, int off, int len) throws IOException
431         {
432             if (this.count >= this.length)
433             {
434                 return -1;
435             }
436             int result = super.read(b, off, len);
437             if (this.count > 0)
438             {
439                 this.count += result;
440             }
441             return result;
442         }
443 
444         /* (non-Javadoc)
445          * @see java.io.FilterInputStream#skip(long)
446          */
447         @Override
448         public long skip(long n) throws IOException
449         {
450             long result = super.skip(n);
451             this.count += result;
452             return result;
453         }
454     }

456     /**
457      * PUT ingests a package as a new Item. Package type (must match pluggable
458      * packager name) is in either (a) "package" query arg in URI (b)
459      * content-type request header
460      * 
461      * @throws SQLException the SQL exception
462      * @throws AuthorizeException the authorize exception
463      * @throws IOException Signals that an I/O exception has occurred.
464      * @throws DAVStatusException the DAV status exception
465      */
466     @Override
467     protected void put() throws SQLException, AuthorizeException,
468             IOException, DAVStatusException
469     {
470         try
471         {
472             String packageType = this.request.getParameter("package");
473             if (packageType == null)
474             {
475                 packageType = this.request.getContentType();
476             }
477             if (packageType == null)
478             {
479                 throw new DAVStatusException(
480                         HttpServletResponse.SC_BAD_REQUEST,
481                         "Cannot determine package type,  need content-type header or package param");
482             }
483 
484             PackageIngester sip = (PackageIngester) PluginManager
485                     .getNamedPlugin(PackageIngester.class, packageType);
486             if (sip == null)
487             {
488                 throw new DAVStatusException(
489                         HttpServletResponse.SC_BAD_REQUEST,
490                         "Cannot find importer for package type: " + packageType);
491             }
492 
493             /*
494              * Ugh. Servlet container doesn't get end-of-file right on input
495              * stream so we have to count it, when possible.
496              */
497             int contentLength = this.request.getIntHeader("Content-Length");
498             InputStream pis = this.request.getInputStream();
499             if (contentLength >= 0)
500             {
501                 pis = new CountedInputStream(pis, contentLength);
502                 log.debug("put: Using CountedInputStream, length="
503                         + String.valueOf(contentLength));
504             }
505             WorkspaceItem wi = sip.ingest(this.context, this.collection, pis,
506                     PackageParameters.create(this.request), null);
507             WorkflowItem wfi = WorkflowManager.startWithoutNotify(this.context, wi);
508 
509             // get new item's location: if workflow completed, then look
510             // for handle (but be ready for disappointment); otherwise,
511             // return the workflow item's resource.
512             int state = wfi.getState();
513             String location = null;
514             if (state == WorkflowManager.WFSTATE_ARCHIVE)
515             {
516                 Item ni = wfi.getItem();
517                 String handle = HandleManager.findHandle(this.context, ni);
518                 String end = (handle != null) ? DAVDSpaceObject
519                         .getPathElt(handle) : DAVItem.getPathElt(ni);
520                 DAVItem newItem = new DAVItem(this.context, this.request, this.response,
521                         makeChildPath(end), ni);
522                 location = newItem.hrefURL();
523             }
524             else if (state == WorkflowManager.WFSTATE_SUBMIT
525                     || state == WorkflowManager.WFSTATE_STEP1POOL)
526             {
527                 location = hrefPrefix() + DAVWorkflow.getPath(wfi);
528             }
529             else
530             {
531                 throw new DAVStatusException(
532                         HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
533                         "Workflow object in unexpected state, state="
534                                 + String.valueOf(state) + ", aborting PUT.");
535             }
536 
537             this.context.commit();
538             log.info("Created new Item, location=" + location);
539             this.response.setHeader("Location", location);
540             this.response.setStatus(HttpServletResponse.SC_CREATED);
541         }
542         catch (PackageException pe)
543         {
544             pe.log(log);
545             throw new DAVStatusException(
546                     HttpServletResponse.SC_INTERNAL_SERVER_ERROR, pe.toString());
547         }
548         catch (CrosswalkException ie)
549         {
550             String reason = "";
551             if (ie.getCause() != null)
552             {
553                 reason = ", Reason: " + ie.getCause().toString();
554             }
555             log.error(ie.toString() + reason);
556             throw new DAVStatusException(
557                     HttpServletResponse.SC_INTERNAL_SERVER_ERROR, ie.toString()
558                             + reason);
559         }
560     }
