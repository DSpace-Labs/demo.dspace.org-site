134     /**
135      * Gimme cookie.
136      * 
137      * @param request the request
138      * 
139      * @return the cookie
140      */
141     private static Cookie gimmeCookie(HttpServletRequest request)
142     {
143         Cookie cookies[] = request.getCookies();
144         if (cookies != null)
145             for (Cookie element : cookies)
146             {
147                 if (element.getName().equals(COOKIE_NAME))
148                 {
149                     return element;
150                 }
151             }
152         return null;
153     }

155     /**
156      * Get Session Cookie.
157      * <p>
158      * DAVServlet rolls its own session cookie because the Servlet container's
159      * session <em>cannot</em> be constrained to use ONLY cookies and NOT
160      * URL-rewriting, and the latter would break the DAV protocol so we cannot
161      * use it. Since we really only need to cache the authenticated EPerson (an
162      * integer ID) anyway, it's easy enough so simply stuff that into a cookie.
163      * <p>
164      * Cookie format is: <br>
165      * {timestamp}!{epersonID}!{client-IP}!{MAC} <br>
166      * where timestamp and eperson are integers; client IP is dotted IP
167      * notation, and MAC is the hex MD5 of the preceding fields plus the
168      * "cookieSecret" string. The MAC ensures that the cookie was issued by this
169      * servlet.
170      * <p>
171      * Look for authentication cookie and try to get a previously-authenticated
172      * EPerson from it if found. Also check the timestamp to be sure the cookie
173      * isn't "stale".
174      * <p>
175      * NOTE This is also used by the SOAP servlet.
176      * <p>
177      * 
178      * @param context -
179      * set user in this context
180      * @param request -
181      * HTTP request.
182      * 
183      * @return true when a fresh cookie yields a valid eperson.
184      * 
185      * @throws SQLException the SQL exception
186      */
187     protected static boolean getAuthFromCookie(Context context,
188             HttpServletRequest request) throws SQLException
189     {
190         Cookie cookie = gimmeCookie(request);
191         if (cookie == null)
192         {
193             return false;
194         }
195         String crumb[] = cookie.getValue().split("\\!");
196         if (crumb.length != 4)
197         {
198             log
199                     .warn("Got invalid cookie value = \"" + cookie.getValue()
200                             + "\"");
201             return false;
202         }
203         long timestamp = 0;
204         int epersonID = 0;
205         try
206         {
207             timestamp = Long.parseLong(crumb[0]);
208             epersonID = Integer.parseInt(crumb[1]);
209         }
210         catch (NumberFormatException e)
211         {
212             log.warn("Error groveling cookie, " + e.toString());
213             return false;
214         }
215 
216         // check freshness
217         long now = new Date().getTime();
218         if (timestamp > now || (now - timestamp) > COOKIE_SELL_BY)
219         {
220             log.warn("Cookie is stale or has weird time, value = \""
221                     + cookie.getValue() + "\"");
222             return false;
223         }
224 
225         // check IP address
226         if (!crumb[2].equals(request.getRemoteAddr()))
227         {
228             log.warn("Cookie fails IP Addr test, value = \""
229                     + cookie.getValue() + "\"");
230             return false;
231         }
232 
233         // check MAC
234         String mac = Utils.getMD5(crumb[0] + "!" + crumb[1] + "!" + crumb[2]
235                 + "!" + cookieSecret);
236         if (!mac.equals(crumb[3]))
237         {
238             log.warn("Cookie fails MAC test, value = \"" + cookie.getValue()
239                     + "\"");
240             return false;
241         }
242 
243         // looks like the browser reguritated a good one:
244         EPerson cuser = EPerson.find(context, epersonID);
245         if (cuser != null)
246         {
247             context.setCurrentUser(cuser);
248             log.debug("Got authenticated user from cookie, id=" + crumb[1]);
249             return true;
250         }
251         return false;
252     }

254     /**
255      * Set a new cookie -- only bother if there is no existing cookie or it's at
256      * least halfway stale, so you're not churning it.. When force is true,
257      * always set a fresh cookie. (e.g. after mac failure upon server restart,
258      * etc)
259      * <p>
260      * 
261      * @param context -
262      * get user from context
263      * @param request the request
264      * @param response the response
265      * @param force the force
266      */
267     protected static void putAuthCookie(Context context,
268             HttpServletRequest request, HttpServletResponse response,
269             boolean force)
270     {
271         Cookie cookie = gimmeCookie(request);
272         long now = new Date().getTime();
273         if (!force && cookie != null)
274         {
275             String crumb[] = cookie.getValue().split("\\!");
276             if (crumb.length == 4)
277             {
278                 long timestamp = -1;
279                 try
280                 {
281                     timestamp = Long.parseLong(crumb[0]);
282                 }
283                 catch (NumberFormatException e)
284                 {
285                 }
286 
287                 // check freshness - skip setting cookie if old one isn't stale
288                 if (timestamp > 0 && (now - timestamp) < (COOKIE_SELL_BY / 2))
289                 {
290                     return;
291                 }
292             }
293         }
294         EPerson user = context.getCurrentUser();
295         if (user == null)
296         {
297             return;
298         }
299         String value = String.valueOf(now) + "!" + String.valueOf(user.getID())
300                 + "!" + request.getRemoteAddr() + "!";
301         String mac = Utils.getMD5(value + cookieSecret);
302         cookie = new Cookie(COOKIE_NAME, value + mac);
303         cookie.setPath(request.getContextPath());
304         response.addCookie(cookie);
305 
306         log.debug("Setting new cookie, value = \"" + value + mac + "\"");
307     }

309     /**
310      * Get authenticated user for this service. Returns null upon failure, with
311      * the implication that an error repsonse has already been "sent", so caller
312      * should not set anything else in servlet response.
313      * 
314      * @param request the request
315      * @param response the response
316      * @param username the username
317      * @param password the password
318      * 
319      * @return the context
320      * 
321      * @throws IOException Signals that an I/O exception has occurred.
322      * @throws SQLException the SQL exception
323      */
324     private static Context authenticate(HttpServletRequest request,
325             HttpServletResponse response, String username, String password)
326             throws IOException, SQLException
327     {
328         Context context = new Context();
329 
330         if (getAuthFromCookie(context, request))
331         {
332             putAuthCookie(context, request, response, false);
333             return context;
334         }
335 
336         // get username/password from Basic auth header if avail:
337         String cred = request.getHeader("Authorization");
338         if (cred != null && username == null && password == null)
339         {
340             log.info(LogManager.getHeader(context, "got creds", "Authorize: "
341                     + cred));
342             StringTokenizer ct = new StringTokenizer(cred);
343 
344             // format: Basic {username:password in base64}
345             if (ct.nextToken().equalsIgnoreCase("Basic"))
346             {
347                 String crud = ct.nextToken();
348                 String dcrud = new String(Base64.decodeBase64(crud.getBytes()));
349                 int colon = dcrud.indexOf(":");
350                 if (colon > 0)
351                 {
352                     username = URLDecode(dcrud.substring(0, colon));
353                     password = URLDecode(dcrud.substring(colon + 1));
354                     log
355                             .info(LogManager.getHeader(context, "auth",
356                                     "Got username=\"" + username
357                                             + "\" out of \"" + crud + "\"."));
358                 }
359             }
360         }
361         if (AuthenticationManager.authenticate(context, username, password,
362                 null, request) == AuthenticationMethod.SUCCESS)
363         {
364             log.info(LogManager.getHeader(context, "auth",
365                     "Authentication returned SUCCESS, eperson="
366                             + context.getCurrentUser().getEmail()));
367         }
368         else
369         {
370             if (username == null)
371             {
372                 log.info(LogManager.getHeader(context, "auth",
373                         "No credentials, so sending WWW-Authenticate header."));
374             }
375             else
376             {
377                 log.warn(LogManager.getHeader(context, "auth",
378                         "Authentication FAILED, cred=" + cred));
379             }
380 
381             // ...EXCEPT if dav.access.anonymous is true in config:
382             if (!allowAnonymousAccess)
383             {
384                 if (response != null)
385                 {
386                     response.setHeader("WWW-Authenticate",
387                             "Basic realm=\"dspace\"");
388                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
389                 }
390                 return null;
391             }
392         }
393 
394         // Set any special groups - invoke the authentication mgr.
395         int[] groupIDs = AuthenticationManager.getSpecialGroups(context,
396                 request);
397         for (int element : groupIDs)
398         {
399             context.setSpecialGroup(element);
400             log.debug("Adding Special Group id=" + String.valueOf(element));
401         }
402         putAuthCookie(context, request, response, true);
403         return context;
404     }

406     /**
407      * Return portion of URI path relevant to the DAV resource. We go through
408      * the extra pain of chopping up getRequestURI() because it is NOT
409      * URL-decoded by the Servlet container, while unfortunately getPathInfo()
410      * IS pre-decoded, leaving a redudndant "/" (and who knows what else) in the
411      * handle. Since the "handle" may not even be a CNRI Handle, we don't want
412      * to assume it even has a "/" (escaped or not).
413      * <p>
414      * Finally, search for doubled-up '/' separators and coalesce them.
415      * 
416      * @param request the request
417      * 
418      * @return String of undecoded path NOT starting with '/'.
419      */
420     private static String getDavResourcePath(HttpServletRequest request)
421     {
422         String path = request.getRequestURI();
423         String ppath = path.substring(request.getContextPath().length());
424         String scriptName = request.getServletPath();
425         if (ppath.startsWith(scriptName))
426         {
427             ppath = ppath.substring(scriptName.length());
428         }
429         // log.debug("Got DAV URI: BEFORE // FIXUP: PATH_INFO=\"" + ppath+"\"");
430         // turn all double '/' ("//") in URI into single '/'
431         StringBuffer sb = new StringBuffer(ppath);
432         int i = ppath.length() - 2;
433         if (i > 0)
434         {
435             while ((i = ppath.lastIndexOf("//", i)) > -1)
436             {
437                 sb.deleteCharAt(i + 1);
438                 --i;
439             }
440         }
441         // remove leading '/'
442         if (sb.length() > 0 && sb.charAt(0) == '/')
443         {
444             sb.deleteCharAt(0);
445         }
446         ppath = sb.toString();
447         log.debug("Got DAV URI: PATH_INFO=\"" + ppath + "\"");
448         return ppath;
449     }

451     /**
452      * override service() to add DAV methods.
453      * 
454      * @param request the request
455      * @param response the response
456      * 
457      * @throws ServletException the servlet exception
458      * @throws IOException Signals that an I/O exception has occurred.
459      */
460     @Override
461     protected void service(HttpServletRequest request,
462             HttpServletResponse response) throws ServletException, IOException
463     {
464         String method = request.getMethod();
465 
466         // no authentication needed for OPTIONS
467         if (method.equals(METHOD_OPTIONS))
468         {
469             doOptions(request, response);
470         }
471         else if (!serviceInternal(method, request, response))
472         {
473             super.service(request, response);
474         }
475     }

479      * 
480      * @param msg the msg

484     private static String truncateForStatus(String msg)

488                 + "... [Message truncated, see logs for details.]" : msg;

491     /**
492      * Pass this request along to the appropriate resource and method. Includes
493      * authentication, where needed. Return true if we handle this request,
494      * false otherwise. True means response has been "sent", false not.
495      * 
496      * @param method the method
497      * @param request the request
498      * @param response the response
499      * 
500      * @return true, if service internal
501      * @throws IOException Signals that an I/O exception has occurred.
502      */
503     protected static boolean serviceInternal(String method,
504             HttpServletRequest request, HttpServletResponse response)
505             throws IOException
506     {
507         // Fake new DAV methods not understood by the Apache Servlet base class
508         // (returns HTTP/500 when it sees unrecognised method)
509         // The way it is faked is by submitting "delete=true" in the PUT URL's
510         // query parameters (for a delete)
511         // The way it is faked is by submitting "mkcol=true" in the PUT URL's
512         // query parameters (for a mk-collection)
513         if (method.equals(METHOD_PUT)
514                 && request.getQueryString().indexOf("delete=true") >= 0)
515         {
516             method = METHOD_DELETE;
517         }
518         if (method.equals(METHOD_PUT)
519                 && request.getQueryString().indexOf("mkcol=true") >= 0)
520         {
521             method = METHOD_MKCOL;
522         }
523 
524         // if not a DAV method (i.e. POST), defer to superclass.
525         if (!(method.equals(METHOD_PROPFIND) || method.equals(METHOD_PROPPATCH)
526                 || method.equals(METHOD_MKCOL) || method.equals(METHOD_COPY)
527                 || method.equals(METHOD_MOVE) || method.equals(METHOD_DELETE)
528                 || method.equals(METHOD_GET) || method.equals(METHOD_PUT)))
529         {
530             return false;
531         }
532 
533         // set all incoming encoding to UTF-8
534         request.setCharacterEncoding("UTF-8");
535         String pathElt[] = getDavResourcePath(request).split("/");
536 
537         Context context = null;
538         try
539         {
540             // this sends a response on failure, unless it throws.
541             context = authenticate(request, response, null, null);
542             if (context == null)
543             {
544                 return true;
545             }
546 
547             // Note: findResource sends error response if it fails.
548             DAVResource resource = DAVResource.findResource(context, request,
549                     response, pathElt);
550             if (resource != null)
551             {
552                 if (method.equals(METHOD_PROPFIND))
553                 {
554                     resource.propfind();
555                 }
556                 else if (method.equals(METHOD_PROPPATCH))
557                 {
558                     resource.proppatch();
559                 }
560                 else if (method.equals(METHOD_COPY))
561                 {
562                     resource.copy();
563                 }
564                 else if (method.equals(METHOD_DELETE))
565                 {
566                     resource.delete();
567                 }
568                 else if (method.equals(METHOD_MKCOL))
569                 {
570                     resource.mkcol();
571                 }
572                 else if (method.equals(METHOD_GET))
573                 {
574                     resource.get();
575                 }
576                 else if (method.equals(METHOD_PUT))
577                 {
578                     resource.put();
579                 }
580                 else
581                 {
582                     response.sendError(HttpServletResponse.SC_NOT_IMPLEMENTED);
583                 }
584                 context.complete();
585                 context = null;
586             }
587         }
588         catch (SQLException e)
589         {
590             log.error(e.toString(),e);
591             response
592                     .sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
593                             truncateForStatus("Database access error: "
594                                     + e.toString()));
595         }
596         catch (AuthorizeException e)
597         {
598             if(log.isDebugEnabled())
599             {
600                 log.debug(e.toString(),e);
601             }
602             else
603             {
604                 log.info(e.toString());
605             }
606             response.sendError(HttpServletResponse.SC_FORBIDDEN,
607                     truncateForStatus("Access denied: " + e.toString()));
608         }
609         catch (DAVStatusException e)
610         {
611             log.error(e.toString(),e);
612             response.sendError(e.getStatus(), truncateForStatus(e
613                     .getMessage()));
614         }
615         catch (IOException e)
616         {
617             log.error(e.toString(),e);
618             response
619                     .sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
620                             truncateForStatus("IO Error: "
621                                     + e.toString()));
622         }
623         catch (Exception e)
624         {
625             log.error(e.toString(),e);
626             response
627                     .sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
628                             truncateForStatus("IO Error: "
629                                     + e.toString()));
630         }
631         finally
632         {
633             // Abort the context if it's still valid
634             if (context != null && context.isValid())
635             {
636                 context.abort();
637             }
638         }
639         return true;
640     }

643      * Handler for HTTP OPTIONS method. Same for all resources under the WeDAV
644      * root. Add DAV methods so client knows we handle DAV.
645      * 
646      * @param request the request
647      * @param response the response
