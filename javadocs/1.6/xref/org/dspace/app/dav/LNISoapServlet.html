155     /**
156      * Authenticate and return the filled-in DSpace context This is the prologue
157      * to all calls.
158      * 
159      * @return the context
160      * 
161      * @throws SQLException the SQL exception
162      * @throws IOException Signals that an I/O exception has occurred.
163      */
164     private Context prologue() throws SQLException, IOException
165     {
166         MessageContext mc = MessageContext.getCurrentContext();
167         String username = null, password = null;
168 
169         if (mc.getUsername() != null)
170         {
171             username = DAVServlet.URLDecode(mc.getUsername());
172         }
173         if (mc.getPassword() != null)
174         {
175             password = DAVServlet.URLDecode(mc.getPassword());
176         }
177 
178         /***********************************************************************
179          * ** XXX TEMPORARY *** Instrumentation to explore the guts of Axis at
180          * runtime, *** leave this commented-out. java.util.Iterator pi =
181          * mc.getPropertyNames(); while (pi.hasNext()) log.debug("SOAP: request
182          * has property named \""+((String)pi.next())+"\""); log.debug("SOAP:
183          * getSOAPActionURI = \""+mc.getSOAPActionURI()+"\""); log.debug("SOAP:
184          * property(servletEndpointContext) = =
185          * \""+mc.getProperty("servletEndpointContext").toString()+"\"");
186          * log.debug("SOAP: property(realpath) = =
187          * \""+mc.getProperty("realpath").toString()+"\""); log.debug("SOAP:
188          * property(transport.http.servletLocation) = =
189          * \""+mc.getProperty("transport.http.servletLocation").toString()+"\"");
190          * *** end TEMPORARY INSTRUMENTATION
191          **********************************************************************/
192 
193         this.request = (HttpServletRequest) mc
194                 .getProperty("transport.http.servletRequest");
195         this.response = (HttpServletResponse) mc
196                 .getProperty("transport.http.servletResponse");
197         
198         Context context = new Context();
199 
200         // try cookie shortcut
201         if (DAVServlet.getAuthFromCookie(context, this.request))
202         {
203             DAVServlet.putAuthCookie(context, this.request, this.response, false);
204             log.debug("SOAP service " + this.getClass().getName()
205                     + " authenticated with cookie.");
206             return context;
207         }
208 
209         int status = AuthenticationManager.authenticate(context, username,
210                 password, null, this.request);
211         if (status == AuthenticationMethod.SUCCESS)
212         {
213             EPerson cu = context.getCurrentUser();
214             log.debug("SOAP service " + this.getClass().getName()
215                     + " authenticated as " + cu.getEmail() + " ("
216                     + cu.getFirstName() + " " + cu.getLastName() + ")");
217             DAVServlet.putAuthCookie(context, this.request, this.response, true);
218             return context;
219         }
220         else if (status == AuthenticationMethod.BAD_CREDENTIALS)
221         {
222             context.abort();
223             throw new LNIRemoteException(
224                     "Authentication failed: Bad Credentials.");
225         }
226         else if (status == AuthenticationMethod.CERT_REQUIRED)
227         {
228             context.abort();
229             throw new LNIRemoteException(
230                     "Authentication failed: This user may only login with X.509 certificate.");
231         }
232         else if (status == AuthenticationMethod.NO_SUCH_USER)
233         {
234             context.abort();
235             throw new LNIRemoteException("Authentication failed: No such user.");
236         }
237         else
238         {
239             context.abort();
240             /** AuthenticationMethod.BAD_ARGS and etc * */
241             throw new LNIRemoteException(
242                     "Authentication failed: Cannot authenticate.");
243         }
244     }

246     /**
247      * Propfind.
248      * 
249      * @param uri the uri
250      * @param doc the doc
251      * @param depth the depth
252      * @param types the types
253      * 
254      * @return the string
255      * 
256      * @throws LNIRemoteException the LNI remote exception
257      */
258     public String propfind(String uri, String doc, int depth, String types)
259             throws LNIRemoteException
260     {
261         // break up path into elements.
262         if (uri.startsWith("/"))
263         {
264             uri = uri.substring(1);
265         }
266         String pathElt[] = uri.split("/");
267         Context context = null;
268         try
269         {
270             context = prologue();
271 
272             // return properties only for resources of these types, comma-sep
273             // list
274             String aTypes[] = (types == null) ? null : types.split(",");
275             int typeMask = DAVResource.typesToMask(aTypes);
276 
277             DAVResource resource = DAVResource.findResource(context, null,
278                     null, pathElt);
279             if (resource == null)
280             {
281                 throw new LNIRemoteException("Resource not found.");
282             }
283             else
284             {
285                 Document outdoc = resource.propfindDriver(depth,
286                         new ByteArrayInputStream(doc.getBytes()), typeMask);
287 
288                 if (outdoc == null)
289                 {
290                     // this should never happen, it should throw an error
291                     // before returning null
292                     throw new LNIRemoteException(
293                             "propfind failed, no document returned.");
294                 }
295                 
296                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
297                 outputPretty.output(outdoc, baos);
298                 context.complete();
299                 return baos.toString();
300             }
301         }
302         catch (IOException ie)
303         {
304             throw new LNIRemoteException("Exception executing PROPFIND", ie);
305         }
306         catch (SQLException e)
307         {
308             throw new LNIRemoteException("Failure accessing database", e);
309         }
310         catch (DAVStatusException e)
311         {
312             throw new LNIRemoteException("PROPFIND request failed: "
313                     + e.getStatusLine());
314         }
315         catch (AuthorizeException e)
316         {
317             throw new LNIRemoteException(
318                     "You are not authorized for the requested operation.", e);
319         }
320         finally
321         {
322             if (context != null && context.isValid())
323             {
324                 context.abort();
325             }
326         }
327     }

408     /**
409      * Return "absolute" DAV URI for the given handle (and optional bitstream
410      * persistent identifier). Always returns a valid URI; if resource is not
411      * found it throws an exception.
412      * 
413      * @param handle the handle
414      * @param bitstreamPid the bitstream pid
415      * 
416      * @return the string
417      * 
418      * @throws LNIRemoteException the LNI remote exception
419      */
420     public String lookup(String handle, String bitstreamPid)
421             throws LNIRemoteException
422     {
423         Context context = null;
424         try
425         {
426             context = prologue();
427 
428             // trim leading scheme if any:
429             if (handle.startsWith("hdl:"))
430             {
431                 handle = handle.substring(4);
432             }
433 
434             DAVLookup resource = new DAVLookup(context, this.request, this.response,
435                     lookupPathElt);
436             String result = resource.makeURI(handle, bitstreamPid);
437             if (result == null)
438             {
439                 throw new LNIRemoteException("Resource not found.");
440             }
441             
442             context.complete();
443 
444             return result;
445         }
446         catch (IOException ie)
447         {
448             throw new LNIRemoteException("Exception executing LOOKUP", ie);
449         }
450         catch (SQLException e)
451         {
452             throw new LNIRemoteException("Failure accessing database", e);
453         }
454         finally
455         {
456             if (context != null && context.isValid())
457             {
458                 context.abort();
459             }
460         }
461     }
