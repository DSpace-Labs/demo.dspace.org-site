99      /**
100      * Create a new instance, reading the lines in from file
101      *
102      * @param f The file to read from
103      *
104      * @throws Exception thrown if there is an error reading or processing the file
105      */
106     public DSpaceCSV(File f) throws Exception
107     {
108         // Initalise the class
109         init();
110 
111         // Open the CSV file
112         BufferedReader input = new BufferedReader(new InputStreamReader(new FileInputStream(f),"UTF8"));
113 
114         // Read the heading line
115         String head = input.readLine();
116         String[] headingElements = head.split(escapedFieldSeparator);
117         for (String element : headingElements)
118         {
119             // Remove surrounding quotes if there are any
120             if ((element.startsWith("\"")) && (element.endsWith("\"")))
121             {
122                 element = element.substring(1, element.length() - 1);
123             }
124 
125             if (!"id".equals(element))
126             {
127                 // Store the heading
128                 headings.add(element);
129             }
130         }
131 
132         // Read each subsequent line
133         String line;
134         while ((line = input.readLine()) != null){
135             // Are there an odd number of quotes?
136             while (((" " + line + " ").split("\"").length)%2 == 0)
137             {
138                 line = line + "\n" + input.readLine();
139             }
140 
141             // Parse the item metadata
142             addItem(line);
143         }
144     }

185     /**
186      * Set the value separator for multiple values stored in one csv value.
187      *
188      * Is set in dspace.cfg as bulkedit.valueseparator
189      *
190      * If not set, defaults to double pipe '||'
191      */
192     private void setValueSeparator()
193     {
194         // Get the value separator
195         valueSeparator = ConfigurationManager.getProperty("bulkedit.valueseparator");
196         if ((valueSeparator != null) && (!"".equals(valueSeparator.trim())))
197         {
198             valueSeparator = valueSeparator.trim();
199         }
200         else
201         {
202             valueSeparator = "||";
203         }
204 
205         // Now store the escaped version
206         Pattern spchars = Pattern.compile("([\\\\*+\\[\\](){}\\$.?\\^|])");
207         Matcher match = spchars.matcher(valueSeparator);
208         escpaedValueSeparator = match.replaceAll("\\\\$1");
209     }

211     /**
212      * Set the field separator use to separate fields in the csv.
213      *
214      * Is set in dspace.cfg as bulkedit.fieldseparator
215      *
216      * If not set, defaults to comma ','.
217      *
218      * Special values are 'tab', 'hash' and 'semicolon' which will
219      * get substituted from the text to the value.
220      */
221     private void setFieldSeparator()
222     {
223         // Get the value separator
224         fieldSeparator = ConfigurationManager.getProperty("bulkedit.fieldseparator");
225         if ((fieldSeparator != null) && (!"".equals(fieldSeparator.trim())))
226         {
227             fieldSeparator = fieldSeparator.trim();
228             if ("tab".equals(fieldSeparator))
229             {
230                 fieldSeparator = "\t";
231             }
232             else if ("semicolon".equals(fieldSeparator))
233             {
234                 fieldSeparator = ";";
235             }
236             else if ("hash".equals(fieldSeparator))
237             {
238                 fieldSeparator = "#";
239             }
240             else
241             {
242                 fieldSeparator = fieldSeparator.trim();
243             }
244         }
245         else
246         {
247             fieldSeparator = ",";
248         }
249 
250         // Now store the escaped version
251         Pattern spchars = Pattern.compile("([\\\\*+\\[\\](){}\\$.?\\^|])");
252         Matcher match = spchars.matcher(fieldSeparator);
253         escapedFieldSeparator = match.replaceAll("\\\\$1");
254     }

256     /**
257      * Add a DSpace item to the CSV file
258      *
259      * @param i The DSpace item
260      *
261      * @throws Exception if something goes wrong with adding the Item
262      */
263     public void addItem(Item i) throws Exception
264     {
265         // Create the CSV line
266         DSpaceCSVLine line = new DSpaceCSVLine(i.getID());
267 
268         // Add in owning collections
269         Collection[] collections = i.getCollections();
270         for (Collection c : collections)
271         {
272            line.add("collection", c.getHandle());
273         }
274 
275         // Populate it
276         DCValue md[] = i.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
277         for (DCValue value : md)
278         {
279             // Get the key (schema.element)
280             String key = value.schema + "." + value.element;
281 
282             // Add the qualifer if there is one (schema.element.qualifier)
283             if (value.qualifier != null)
284             {
285                 key = key + "." + value.qualifier;
286             }
287 
288             // Add the language if there is one (schema.element.qualifier[langauge])
289             //if ((value.language != null) && (!"".equals(value.language)))
290             if (value.language != null)
291             {
292                 key = key + "[" + value.language + "]";
293             }
294 
295             // Store the item
296             if (exportAll || okToExport(value))
297             {
298                 line.add(key, value.value);
299                 if (!headings.contains(key))
300                 {
301                     headings.add(key);
302                 }
303             }
304         }
305         lines.add(line);
306         counter++;
307     }

309     /**
310      * Add an item to the CSV file, from a CSV line of elements
311      *
312      * @param line The line of elements
313      * @throws Exception Thrown if an error occurs when adding the item
314      */
315     public void addItem(String line) throws Exception
316     {
317         // Check to see if the last character is a field separator, which hides the last empy column
318         boolean last = false;
319         if (line.endsWith(fieldSeparator))
320         {
321             // Add a space to the end, then remove it later
322             last = true;
323             line += " ";
324         }
325         
326         // Split up on field separator
327         String[] parts = line.split(escapedFieldSeparator);
328         ArrayList<String> bits = new ArrayList<String>();
329         bits.addAll(Arrays.asList(parts));
330 
331         // Merge parts with embedded separators
332         boolean alldone = false;
333         while (!alldone)
334         {
335             boolean found = false;
336             int i = 0;
337             for (String part : bits)
338             {
339                 int bitcounter = part.length() - part.replaceAll("\"", "").length();
340                 if ((part.startsWith("\"")) && ((!part.endsWith("\"")) || ((bitcounter %2) == 1)))
341                 {
342                     found = true;
343                     String add = bits.get(i) + fieldSeparator + bits.get(i + 1);
344                     bits.remove(i);
345                     bits.add(i, add);
346                     bits.remove(i + 1);
347                     break;
348                 }
349                 i++;
350             }
351             alldone = !found;
352         }
353 
354         // Deal with quotes around the elements
355         int i = 0;
356         for (String part : bits)
357         {
358             if ((part.startsWith("\"")) && (part.endsWith("\"")))
359             {
360                 part = part.substring(1, part.length() - 1);
361                 bits.set(i, part);
362             }
363             i++;
364         }
365 
366         // Remove embedded quotes
367         i = 0;
368         for (String part : bits)
369         {
370             if (part.contains("\"\""))
371             {
372                 part = part.replaceAll("\"\"", "\"");
373                 bits.set(i, part);
374             }
375             i++;
376         }
377 
378         // Add elements to a DSpaceCSVLine
379         String id = parts[0].replaceAll("\"", "");
380         DSpaceCSVLine csvLine;
381                 
382         // Is this an existing item, or a new item (where id = '+')
383         if ("+".equals(id))
384         {
385             csvLine = new DSpaceCSVLine();
386         }
387         else
388         {
389             try
390             {
391                 csvLine = new DSpaceCSVLine(Integer.parseInt(id));
392             }
393             catch (NumberFormatException nfe)
394             {
395                 System.err.println("Invalid item identifier: " + id);
396                 System.err.println("Please check your CSV file for informaton. " +
397                                    "Item id must be numeric, or a '+' to add a new item");
398                 throw(nfe);
399             }
400         }
401 
402         // Add the rest of the parts
403         i = 0;
404         for (String part : bits)
405         {
406             if (i > 0)
407             {
408                 // Is this a last empty item?
409                 if ((last) && (i == headings.size()))
410                 {
411                     part = "";
412                 }
413 
414                 // Make sure we register that this column was there
415                 csvLine.add(headings.get(i - 1), null);
416                 String[] elements = part.split(escpaedValueSeparator);
417                 for (String element : elements)
418                 {
419                     if ((element != null) && (!"".equals(element)))
420                     {
421                         csvLine.add(headings.get(i - 1), element);
422                     }
423                 }
424             }
425             i++;
426         }
427         lines.add(csvLine);
428         counter++;
429     }

442     /**
443      * Get the CSV lines as an array of CSV formatted strings
444      *
445      * @return the array of CSV formatted Strings
446      */
447     public String[] getCSVLinesAsStringArray()
448     {
449         // Create the headings line
450         String[] csvLines = new String[counter + 1];
451         csvLines[0] = "id" + fieldSeparator + "collection";
452         Collections.sort(headings);
453         for (String value : headings)
454         {
455             csvLines[0] = csvLines[0] + fieldSeparator + value;
456         }
457 
458         Iterator<DSpaceCSVLine> i = lines.iterator();
459         int c = 1;
460         while (i.hasNext())
461         {
462             csvLines[c++] = i.next().toCSV(headings);
463         }
464 
465         return csvLines;
466     }

476     {
477         // Save the file
478         BufferedWriter out = new BufferedWriter(

488     /**
489      * Is it Ok to export this value? When exportAll is set to false, we don't export
490      * some of the metadata elements.
491      *
492      * The list can be configured via the key bulkedit.ignore-on-export in dspace.cfg
493      *
494      * @param md The DCValue to examine
495      * @return Whether or not it is OK to export this element
496      */
497     private boolean okToExport(DCValue md)
498     {
499         // First check the metadata format, and K all non DC elements
500         if (!"dc".equals(md.schema))
501         {
502             return true;
503         }
504 
505         // Now compare with the list to ignore
506         String key = md.schema + "." + md.element;
507         if (md.qualifier != null)
508         {
509             key += "." + md.qualifier;
510         }
511         if (ignore.get(key) != null) {
512             return false;
513         }
514 
515         // Must be OK, so don't ignore
516         return true;
517     }
