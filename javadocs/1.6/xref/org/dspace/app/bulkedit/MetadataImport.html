89      /**
90       * Run an import. The import can either be read-only to detect changes, or
91       * can write changes as it goes.
92       *
93       * @param change Whether or not to write the changes to the database
94       * @param useWorkflow Whether the workflows should be used when creating new items
95       * @param workflowNotify If the workflows should be used, whether to send notifications or not
96       * @param useTemplate Use collection template if create new item
97       * @return An array of BulkEditChange elements representing the items that have changed
98       *
99       * @throws MetadataImportException if something goes wrong
100      */
101     public ArrayList<BulkEditChange> runImport(boolean change,
102                                                boolean useWorkflow,
103                                                boolean workflowNotify,
104                                                boolean useTemplate) throws MetadataImportException
105     {
106         // Store the changes
107         ArrayList<BulkEditChange> changes = new ArrayList<BulkEditChange>();
108 
109         // Make the changes
110         try
111         {
112             // Process each change
113             for (DSpaceCSVLine line : toImport)
114             {
115                 // Get the DSpace item to compare with
116                 int id = line.getID();
117 
118                 // Is this a new item?
119                 if (id != -1)
120                 {
121                     // Get the item
122                     Item item = Item.find(c, id);
123                     BulkEditChange whatHasChanged = new BulkEditChange(item);
124 
125                     // Has it moved collection?
126                     ArrayList<String> collections = line.get("collection");
127                     if (collections != null)
128                     {
129                         // Sanity check we're not orphaning it
130                         if (collections.size() == 0)
131                         {
132                             throw new MetadataImportException("Missing collection from item " + item.getHandle());
133                         }
134                         Collection[] actualCollections = item.getCollections();
135                         compare(item, collections, actualCollections, whatHasChanged, change);
136                     }
137 
138                     // Iterate through each metadata element in the csv line
139                     Enumeration<String> e = line.keys();
140                     while (e.hasMoreElements())
141                     {
142                         // Get the values we already have
143                         String md = e.nextElement();
144                         if (!"id".equals(md))
145                         {
146                             // Get the values from the CSV
147                             String[] fromCSV = line.get(md).toArray(new String[line.get(md).size()]);
148 
149                             // Compare
150                             compare(item, fromCSV, change, md, whatHasChanged);
151                         }
152                     }
153 
154                     // Only record if changes have been made
155                     if (whatHasChanged.hasChanges())
156                     {
157                         changes.add(whatHasChanged);
158                     }
159                 }
160                 else
161                 {
162                     // This is marked as a new item, so no need to compare
163 
164                     // First check a user is set, otherwise this can't happen
165                     if (c.getCurrentUser() == null)
166                     {
167                         throw new MetadataImportException("When adding new items, a user must be specified with the -e option");
168                     }
169 
170                     // Iterate through each metadata element in the csv line
171                     Enumeration<String> e = line.keys();
172                     BulkEditChange whatHasChanged = new BulkEditChange();
173                     while (e.hasMoreElements())
174                     {
175                         // Get the values we already have
176                         String md = e.nextElement();
177                         if (!"id".equals(md))
178                         {
179                             // Get the values from the CSV
180                             String[] fromCSV = line.get(md).toArray(new String[line.get(md).size()]);
181 
182                             // Add all the values from the CSV line
183                             add(fromCSV, md, whatHasChanged);
184                         }
185                     }
186 
187                     // Check it has an owning collection
188                     ArrayList<String> collections = line.get("collection");
189                     if (collections == null)
190                     {
191                         throw new MetadataImportException("New items must have a 'collection' assiged in the form of a handle");
192                     }
193                     
194                     // Check collections are really collections
195                     ArrayList<Collection> check = new ArrayList<Collection>();
196                     Collection collection;
197                     for (String handle : collections)
198                     {
199                         try
200                         {
201                             // Resolve the handle to the collection
202                             collection = (Collection)HandleManager.resolveToObject(c, handle);
203 
204                             // Check it resolved OK
205                             if (collection == null)
206                             {
207                                 throw new MetadataImportException("'" + handle + "' is not a Collection! You must specify a valid collection for new items");
208                             }
209 
210                             // Check for duplicate
211                             if (check.contains(collection))
212                             {
213                                 throw new MetadataImportException("Duplicate collection assignement detected in new item! " + handle);
214                             }
215                             else
216                             {
217                                 check.add(collection);
218                             }
219                         }
220                         catch (Exception ex)
221                         {
222                             throw new MetadataImportException("'" + handle + "' is not a Collection! You must specify a valid collection for new items");
223                         }
224                     }
225 
226                     // Record the addition to collections
227                     for (String handle : collections)
228                     {
229                         Collection extra = (Collection)HandleManager.resolveToObject(c, handle);
230                         whatHasChanged.registerNewOwningCollection(extra);
231                     }
232 
233                     // Create the new item?
234                     if (change)
235                     {
236                         // Create the item
237                         String collectionHandle = line.get("collection").get(0);
238                         collection = (Collection)HandleManager.resolveToObject(c, collectionHandle);
239                         WorkspaceItem wsItem = WorkspaceItem.create(c, collection, useTemplate);
240                         Item item = wsItem.getItem();
241 
242                         // Add the metadata to the item
243                         for (DCValue dcv : whatHasChanged.getAdds())
244                         {
245                             item.addMetadata(dcv.schema,
246                                              dcv.element,
247                                              dcv.qualifier, 
248                                              dcv.language,
249                                              dcv.value);
250                         }
251 
252                         // Should the workflow be used?
253                         if ((useWorkflow) && (workflowNotify))
254                         {
255                             WorkflowManager.start(c, wsItem);
256                         }
257                         else if (useWorkflow)
258                         {
259                             WorkflowManager.startWithoutNotify(c, wsItem);
260                         }
261                         else
262                         {
263                             // Install the item
264                             InstallItem.installItem(c, wsItem);
265                         }
266                                               
267                         // Add to extra collections
268                         if (line.get("collection").size() > 0)
269                         {
270                             for (int i = 1; i < collections.size(); i++)
271                             {
272                                 String handle = collections.get(i);
273                                 Collection extra = (Collection)HandleManager.resolveToObject(c, handle);
274                                 extra.addItem(item);
275                             }
276                         }
277 
278                         // Commit changes to the object
279                         c.commit();
280                         whatHasChanged.setItem(item);
281                     }
282 
283                     // Record the changes
284                     changes.add(whatHasChanged);
285                 }
286             }
287         }
288         catch (MetadataImportException mie)
289         {
290             throw mie;
291         }
292         catch (Exception e)
293         {
294             e.printStackTrace();
295         }
296 
297         // Return the changes
298         return changes;
299     }

301     /**
302      * Compare an item metadata with a line from CSV, and optionally update the item
303      *
304      * @param item The current item metadata
305      * @param fromCSV The metadata from the CSV file
306      * @param change Whether or not to make the update
307      * @param md The element to compare
308      * @param changes The changes object to populate
309      *
310      * @throws SQLException if there is a problem accessing a Collection from the database, from its handle
311      * @throws AuthorizeException if there is an authorization problem with permissions
312      */
313     private void compare(Item item, String[] fromCSV, boolean change,
314                          String md, BulkEditChange changes) throws SQLException, AuthorizeException
315     {
316         // Don't compare collections
317         if ("collection".equals(md))
318         {
319             return;
320         }
321 
322         // Make a String array of the current values stored in this element
323         // First, strip of language if it is there
324         String language = null;
325         if (md.contains("["))
326         {
327             String[] bits = md.split("\\[");
328             language = bits[1].substring(0, bits[1].length() - 1);
329         }
330         String[] bits = md.split("\\.");
331         String schema = bits[0];
332         String element = bits[1];
333         // If there is a language on the element, strip if off
334         if (element.contains("["))
335         {
336             element = element.substring(0, element.indexOf("["));
337         }
338         String qualifier = null;
339         if (bits.length > 2)
340         {
341             qualifier = bits[2];
342 
343             // If there is a language, strip if off
344             if (qualifier.contains("["))
345             {
346                 qualifier = qualifier.substring(0, qualifier.indexOf("["));
347             }
348         }
349         DCValue[] current = item.getMetadata(schema, element, qualifier, language);
350         
351         String[] dcvalues = new String[current.length];
352         int i = 0;
353         for (DCValue dcv : current)
354         {
355             dcvalues[i] = dcv.value;
356             i++;
357         }
358 
359         // Compare from csv->current
360         for (String value : dcvalues)
361         {
362             // Look to see if it should be removed
363             DCValue dcv = new DCValue();
364             dcv.schema = schema;
365             dcv.element = element;
366             dcv.qualifier = qualifier;
367             dcv.language = language;
368             dcv.value = value;
369             if ((value != null) && (!"".equals(value)) && (!contains(value, fromCSV)))
370             {
371                 // Remove it
372                 changes.registerRemove(dcv);
373             }
374             else
375             {
376                 // Keep it
377                 changes.registerConstant(dcv);
378             }
379         }
380 
381         // Compare from current->csv
382         for (String value : fromCSV)
383         {
384             // Look to see if it should be added
385             DCValue dcv = new DCValue();
386             dcv.schema = schema;
387             dcv.element = element;
388             dcv.qualifier = qualifier;
389             dcv.language = language;
390             dcv.value = value;
391             if ((value != null) && (!"".equals(value)) && (!contains(value, dcvalues)))
392             {
393                 changes.registerAdd(dcv);
394             }
395         }
396 
397         // Update the item if it has changed
398         if ((change) &&
399             ((changes.getAdds().size() > 0) || (changes.getRemoves().size() > 0)))
400         {
401             // Get the complete list of what values should now be in that element
402             ArrayList<DCValue> list = changes.getComplete();
403             ArrayList<String> values = new ArrayList<String>();
404             for (DCValue value : list)
405             {
406                 if ((qualifier == null) && (language == null))
407                 {
408                     if ((schema.equals(value.schema)) &&
409                         (element.equals(value.element)) &&
410                          (value.qualifier == null) &&
411                          (value.language == null))
412                     {
413                         values.add(value.value);
414                     }
415                 }
416                 else if (qualifier == null)
417                 {
418                     if ((schema.equals(value.schema)) &&
419                         (element.equals(value.element)) &&
420                         (language.equals(value.language)) &&
421                         (value.qualifier == null))
422                     {
423                         values.add(value.value);
424                     }
425                 }
426                 else if (language == null)
427                 {
428                     if ((schema.equals(value.schema)) &&
429                         (element.equals(value.element)) &&
430                         (qualifier.equals(value.qualifier)) &&
431                         (value.language == null))
432                     {
433                         values.add(value.value);
434                     }
435                 }
436                 else
437                 {
438                     if ((schema.equals(value.schema)) &&
439                         (element.equals(value.element)) &&
440                         (qualifier.equals(value.qualifier)) &&
441                         (language.equals(value.language)))
442                     {
443                         values.add(value.value);
444                     }
445                 }
446             }
447 
448             // Set those values
449             item.clearMetadata(schema, element, qualifier, language);
450             String[] theValues = values.toArray(new String[values.size()]);
451             item.addMetadata(schema, element, qualifier, language, theValues);
452             item.update();
453         }
454     }

456     /**
457      * Compare changes between an items owning collections and what is in the CSV file
458      *
459      * @param item The item in question
460      * @param collections The collection handles fro mthe CSV file
461      * @param actualCollections The Collections from the actual item
462      * @param bechange The bulkedit change object for this item
463      * @param change Whether or not to actuate a change
464      *
465      * @throws SQLException if there is a problem accessing a Collection from the database, from its handle
466      * @throws AuthorizeException if there is an authorization problem with permissions
467      * @throws IOException Can be thrown when moving items in communities
468      * @throws MetadataImportException If something goes wrong to be reported back to the user
469      */
470     private void compare(Item item,
471                          ArrayList<String> collections,
472                          Collection[] actualCollections,
473                          BulkEditChange bechange,
474                          boolean change) throws SQLException, AuthorizeException, IOException, MetadataImportException
475     {
476         // First, loop through the strings from the CSV
477         for (String csvcollection : collections)
478         {
479             // Look for it in the actual list of Collections
480             boolean found = false;
481             for (Collection collection : actualCollections)
482             {
483                 // Is it there?
484                 if (csvcollection.equals(collection.getHandle()))
485                 {
486                     found = true;
487                 }
488             }
489 
490             // Was it found?
491             if (!found)
492             {
493                 // Register than change
494                 DSpaceObject dso = HandleManager.resolveToObject(c, csvcollection);
495                 if ((dso == null) || (dso.getType() != Constants.COLLECTION))
496                 {
497                     throw new MetadataImportException("Collection defined for item " + item.getID() +
498                                                       " (" + item.getHandle() + ") is not a collection");
499                 }
500                 else
501                 {
502                     Collection col = (Collection)dso;
503                     bechange.registerNewOwningCollection(col);
504 
505                     // Execute the change
506                     if (change)
507                     {
508                         // Add the item to the community
509                         col.addItem(item);
510                     }
511                 }
512             }
513         }
514 
515         // Second, loop through the strings from the current item
516         for (Collection collection : actualCollections)
517         {
518             // Look for it in the actual list of Collections
519             boolean found = false;
520             for (String csvcollection : collections)
521             {
522                 // Is it there?
523                 if (collection.getHandle().equals(csvcollection))
524                 {
525                     found = true;
526                 }
527             }
528 
529             // Was it found?
530             if (!found)
531             {
532                 // Reocrd that it isn't there any more
533                 bechange.registerOldOwningCollection(collection);
534 
535                 // Execute the change
536                 if (change)
537                 {
538                     // Sanity check it is in another collection
539                     if (item.getCollections().length > 1)
540                     {
541                         collection.removeItem(item);
542                     }
543                     else
544                     {
545                         throw new MetadataImportException("Not removing item " + item.getHandle() +
546                                                           " from collection " + collection.getHandle() +
547                                                           " as it would leave it oprhaned!");
548                     }
549                 }
550             }
551         }
552 
553         // Now check the owning_collection (item table) is the same as
554         // the first collection given in the CSV file entry
555         if ((change) && (!collections.get(0).equals(item.getOwningCollection().getHandle())))
556         {
557             Collection owner = (Collection)HandleManager.resolveToObject(c, collections.get(0));
558             item.setOwningCollection(Collection.find(c, owner.getID()));
559             item.update();
560         }
561     }

563     /**
564      * Add an item metadata with a line from CSV, and optionally update the item
565      *
566      * @param fromCSV The metadata from the CSV file
567      * @param md The element to compare
568      * @param changes The changes object to populate
569      *
570      * @throws SQLException when an SQL error has occured (querying DSpace)
571      * @throws AuthorizeException If the user can't make the changes
572      */
573     private void add(String[] fromCSV, String md, BulkEditChange changes)
574                                             throws SQLException, AuthorizeException
575     {
576         // Don't add owning collection
577         if ("collection".equals(md))
578         {
579             return;
580         }
581 
582         // Make a String array of the values
583         // First, strip of language if it is there
584         String language = null;
585         if (md.contains("["))
586         {
587             String[] bits = md.split("\\[");
588             language = bits[1].substring(0, bits[1].length() - 1);
589         }
590         String[] bits = md.split("\\.");
591         String schema = bits[0];
592         String element = bits[1];
593         // If there is a language on the element, strip if off
594         if (element.contains("["))
595         {
596             element = element.substring(0, element.indexOf("["));
597         }
598         String qualifier = null;
599         if (bits.length > 2)
600         {
601             qualifier = bits[2];
602 
603             // If there is a language, strip if off
604             if (qualifier.contains("["))
605             {
606                 qualifier = qualifier.substring(0, qualifier.indexOf("["));
607             }
608         }
609 
610         // Add all the values
611         for (String value : fromCSV)
612         {
613             // Look to see if it should be removed
614             DCValue dcv = new DCValue();
615             dcv.schema = schema;
616             dcv.element = element;
617             dcv.qualifier = qualifier;
618             dcv.language = language;
619             dcv.value = value;
620 
621             // Add it
622             if ((value != null) && (!"".equals(value)))
623             {
624                 changes.registerAdd(dcv);
625             }
626         }
627     }

629     /**
630      * Method to find if a String occurs in an array of Strings
631      *
632      * @param needle The String to look for
633      * @param haystack The array of Strings to search through
634      * @return Whether or not it is contained
635      */
636     private boolean contains(String needle, String[] haystack)
637     {
638         // Look for the needle in the haystack
639         for (String examine : haystack)
640         {
641             if (clean(examine).equals(clean(needle)))
642             {
643                 return true;
644             }
645         }
646         return false;
647     }
648 
649     /**
650      * Clean elements before comparing
651      *
652      * @param in The element to clean
653      * @return The cleaned up element
654      */
655     private String clean(String in)
656     {
657         // Check for nulls
658         if (in == null)
659         {
660             return null;
661         }
662         
663         // Remove newlines as different operating systems sometimes use different formats
664         return in.replaceAll("\r\n", "").replaceAll("\n", "").trim();
665     }

682     /**
683      * Display the changes that have been detected, or that have been made
684      *
685      * @param changes The changes detected
686      * @param changed Whether or not the changes have been made
687      * @return The number of items that have changed
688      */
689     private static int displayChanges(ArrayList<BulkEditChange> changes, boolean changed)
690     {
691         // Display the changes
692         int changeCounter = 0;
693         for (BulkEditChange change : changes)
694         {
695             // Get the changes
696             ArrayList<DCValue> adds = change.getAdds();
697             ArrayList<DCValue> removes = change.getRemoves();
698             ArrayList<Collection> newCollections = change.getNewOwningCollections();
699             ArrayList<Collection> oldCollections = change.getOldOwningCollections();
700             if ((adds.size() > 0) || (removes.size() > 0) ||
701                 (newCollections.size() > 0) || (oldCollections.size() > 0))
702             {
703                 // Show the item
704                 Item i = change.getItem();
705 
706                 System.out.println("-----------------------------------------------------------");
707                 if (!change.isNewItem())
708                 {
709                     System.out.println("Changes for item: " + i.getID() + " (" + i.getHandle() + ")");
710                 }
711                 else
712                 {
713                     System.out.print("New item: ");
714                     if (i != null)
715                     {
716                         if (i.getHandle() != null)
717                         {
718                             System.out.print(i.getID() + " (" + i.getHandle() + ")");
719                         }
720                         else
721                         {
722                             System.out.print(i.getID() + " (in workflow)");
723                         }
724                     }
725                     System.out.println();
726                 }
727                 changeCounter++;
728             }
729 
730             // Show new collections
731             for (Collection c : newCollections)
732             {
733                 String cHandle = c.getHandle();
734                 String cName = c.getName();
735                 if (!changed)
736                 {
737                     System.out.print(" + Add to collection (" + cHandle + "): ");
738                 }
739                 else
740                 {
741                     System.out.print(" + Added to collection  (" + cHandle + "): ");
742                 }
743                 System.out.println(cName);
744             }
745 
746             // Show old collections
747             for (Collection c : oldCollections)
748             {
749                 String cHandle = c.getHandle();
750                 String cName = c.getName();
751                 if (!changed)
752                 {
753                     System.out.print(" + Remove from collection (" + cHandle + "): ");
754                 }
755                 else
756                 {
757                     System.out.print(" + Removed from collection  (" + cHandle + "): ");
758                 }
759                 System.out.println(cName);
760             }
761 
762             // Show additions
763             for (DCValue dcv : adds)
764             {
765                 String md = dcv.schema + "." + dcv.element;
766                 if (dcv.qualifier != null)
767                 {
768                     md += "." + dcv.qualifier;
769                 }
770                 if (dcv.language != null)
771                 {
772                     md += "[" + dcv.language + "]";
773                 }
774                 if (!changed)
775                 {
776                     System.out.print(" + Add    (" + md + "): ");
777                 }
778                 else
779                 {
780                     System.out.print(" + Added   (" + md + "): ");
781                 }
782                 System.out.println(dcv.value);
783             }
784 
785             // Show removals
786             for (DCValue dcv : removes)
787             {

805             }
