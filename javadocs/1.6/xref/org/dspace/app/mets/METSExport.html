112     public static void main(String[] args) throws Exception
113     {
114         Context context = new Context();
115 
116         init(context);
117 
118         // create an options object and populate it
119         CommandLineParser parser = new PosixParser();
120 
121         Options options = new Options();
122 
123         options.addOption("c", "collection", true,
124                 "Handle of collection to export");
125         options.addOption("i", "item", true, "Handle of item to export");
126         options.addOption("a", "all", false, "Export all items in the archive");
127         options.addOption("d", "destination", true, "Destination directory");
128         options.addOption("h", "help", false, "Help");
129 
130         CommandLine line = parser.parse(options, args);
131 
132         if (line.hasOption('h'))
133         {
134             HelpFormatter myhelp = new HelpFormatter();
135             myhelp.printHelp("metsexport", options);
136             System.out
137                     .println("\nExport a collection:  metsexport -c hdl:123.456/789");
138             System.out
139                     .println("Export an item:       metsexport -i hdl:123.456/890");
140             System.out.println("Export everything:    metsexport -a");
141 
142             System.exit(0);
143         }
144 
145         String dest = "";
146 
147         if (line.hasOption('d'))
148         {
149             dest = line.getOptionValue('d');
150 
151             // Make sure it ends with a file separator
152             if (!dest.endsWith(File.separator))
153             {
154                 dest = dest + File.separator;
155             }
156         }
157 
158         if (line.hasOption('i'))
159         {
160             String handle = getHandleArg(line.getOptionValue('i'));
161 
162             // Exporting a single item
163             DSpaceObject o = HandleManager.resolveToObject(context, handle);
164 
165             if ((o != null) && o instanceof Item)
166             {
167                 writeAIP(context, (Item) o, dest);
168                 System.exit(0);
169             }
170             else
171             {
172                 System.err.println(line.getOptionValue('i')
173                         + " is not a valid item Handle");
174                 System.exit(1);
175             }
176         }
177 
178         ItemIterator items = null;
179         try
180         {
181             if (line.hasOption('c'))
182             {
183                 String handle = getHandleArg(line.getOptionValue('c'));
184 
185                 // Exporting a collection's worth of items
186                 DSpaceObject o = HandleManager.resolveToObject(context, handle);
187 
188                 if ((o != null) && o instanceof Collection)
189                 {
190                     items = ((Collection) o).getItems();
191                 }
192                 else
193                 {
194                     System.err.println(line.getOptionValue('c')
195                             + " is not a valid collection Handle");
196                     System.exit(1);
197                 }
198             }
199 
200             if (line.hasOption('a'))
201             {
202                 items = Item.findAll(context);
203             }
204 
205             if (items == null)
206             {
207                 System.err.println("Nothing to export specified!");
208                 System.exit(1);
209             }
210 
211             while (items.hasNext())
212             {
213                 writeAIP(context, items.next(), dest);
214             }
215         }
216         finally
217         {
218             if (items != null)
219                 items.close();
220         }
221         
222         context.abort();
223         System.exit(0);
224     }

232     private static void init(Context context) throws SQLException, IOException
233     {
234         // Don't init again if initialised already
235         if (licenseFormat != -1)

240         // Find the License format
241         BitstreamFormat bf = BitstreamFormat.findByShortDescription(context,
242                 "License");
243         licenseFormat = bf.getID();
244 
245         // get path to DC->MODS map info file
246         String configFile = ConfigurationManager.getProperty("dspace.dir")
247                 + File.separator + "config" + File.separator + "dc2mods.cfg";
248 
249         // Read it in
250         InputStream is = null;

254             dcToMODS = new Properties();
255             dcToMODS.load(is);
256         }
257         finally
258         {
259             if (is != null)
260                 try { is.close(); } catch (IOException ioe) { }
261         }

264     /**
265      * Write out the AIP for the given item to the given directory. A new
266      * directory will be created with the Handle (URL-encoded) as the directory
267      * name, and inside, a mets.xml file written, together with the bitstreams.
268      * 
269      * @param context
270      *            DSpace context to use
271      * @param item
272      *            Item to write
273      * @param dest
274      *            destination directory
275      */
276     public static void writeAIP(Context context, Item item, String dest)
277             throws SQLException, IOException, AuthorizeException, MetsException
278     {
279         System.out.println("Exporting item hdl:" + item.getHandle());
280 
281         // Create aip directory
282         java.io.File aipDir = new java.io.File(dest
283                 + URLEncoder.encode("hdl:" + item.getHandle(), "UTF-8"));
284 
285         if (!aipDir.mkdir())
286         {
287             // Couldn't make the directory for some reason
288             throw new IOException("Couldn't create " + aipDir.toString());
289         }
290 
291         // Write the METS file
292         FileOutputStream out = new FileOutputStream(aipDir.toString()
293                 + java.io.File.separator + "mets.xml");
294         writeMETS(context, item, out, false);
295         out.close();
296 
297         // Write bitstreams
298         Bundle[] bundles = item.getBundles();
299 
300         for (int i = 0; i < bundles.length; i++)
301         {
302             Bitstream[] bitstreams = bundles[i].getBitstreams();
303 
304             for (int b = 0; b < bitstreams.length; b++)
305             {
306                 // Skip license bitstream and unauthorized resources
307                 if ((bitstreams[b].getFormat().getID() != licenseFormat)
308                         && AuthorizeManager.authorizeActionBoolean(context,
309                                 bitstreams[b], Constants.READ))
310                 {
311                     out = new FileOutputStream(aipDir.toString()
312                             + java.io.File.separator
313                             + bitstreams[b].getName());
314 
315                     InputStream in = bitstreams[b].retrieve();
316                     Utils.bufferedCopy(in, out);
317                     out.close();
318                     in.close();
319                 }
320             }
321         }
322     }

324     /**
325      * Write METS metadata corresponding to the metadata for an item
326      * 
327      * @param context
328      *            DSpace context
329      * @param item
330      *            DSpace item to create METS object for
331      * @param os
332      *            A stream to write METS package to (UTF-8 encoding will be used)
333      * @param fullURL
334      *            if <code>true</code>, the &lt;FLocat&gt; values for each
335      *            bitstream will be the full URL for that bitstream. Otherwise,
336      *            only the filename itself will be used.
337      */
338     public static void writeMETS(Context context, Item item, OutputStream os, boolean fullURL)
339             throws SQLException, IOException, AuthorizeException
340     {
341         try
342         {
343             init(context);
344 
345             // Create the METS file
346             Mets mets = new Mets();
347 
348             // Top-level stuff
349             mets.setOBJID("hdl:" + item.getHandle());
350             mets.setLABEL("DSpace Item");
351             mets.setSchema("mods", "http://www.loc.gov/mods/v3",
352                     "http://www.loc.gov/standards/mods/v3/mods-3-0.xsd");
353 
354             // MetsHdr
355             MetsHdr metsHdr = new MetsHdr();
356             metsHdr.setCREATEDATE(new Date()); // FIXME: CREATEDATE is now:
357                                                // maybe should be item create
358                                                // date?
359 
360             // Agent
361             Agent agent = new Agent();
362             agent.setROLE(Role.CUSTODIAN);
363             agent.setTYPE(Type.ORGANIZATION);
364 
365             Name name = new Name();
366             name.getContent()
367                     .add(
368                             new PCData(ConfigurationManager
369                                     .getProperty("dspace.name")));
370             agent.getContent().add(name);
371 
372             metsHdr.getContent().add(agent);
373 
374             mets.getContent().add(metsHdr);
375 
376             DmdSec dmdSec = new DmdSec();
377             dmdSec.setID("DMD_hdl_" + item.getHandle());
378 
379             MdWrap mdWrap = new MdWrap();
380             mdWrap.setMDTYPE(Mdtype.MODS);
381 
382             XmlData xmlData = new XmlData();
383             createMODS(item, xmlData);
384 
385             mdWrap.getContent().add(xmlData);
386             dmdSec.getContent().add(mdWrap);
387             mets.getContent().add(dmdSec);
388 
389             // amdSec
390             AmdSec amdSec = new AmdSec();
391             amdSec.setID("TMD_hdl_" + item.getHandle());
392 
393             // FIXME: techMD here
394             // License as <rightsMD><mdWrap><binData>base64encoded</binData>...
395             InputStream licenseStream = findLicense(context, item);
396 
397             if (licenseStream != null)
398             {
399                 RightsMD rightsMD = new RightsMD();
400                 MdWrap rightsMDWrap = new MdWrap();
401                 rightsMDWrap.setMIMETYPE("text/plain");
402                 rightsMDWrap.setMDTYPE(Mdtype.OTHER);
403                 rightsMDWrap.setOTHERMDTYPE("TEXT");
404 
405                 BinData binData = new BinData();
406                 Base64 base64 = new Base64(licenseStream);
407 
408                 binData.getContent().add(base64);
409                 rightsMDWrap.getContent().add(binData);
410                 rightsMD.getContent().add(rightsMDWrap);
411                 amdSec.getContent().add(rightsMD);
412             }
413 
414             // FIXME: History data???? Nooooo!!!!
415             mets.getContent().add(amdSec);
416 
417             // fileSec
418             FileSec fileSec = new FileSec();
419             boolean fileSecEmpty = true;
420 
421             Bundle[] bundles = item.getBundles();
422 
423             for (int i = 0; i < bundles.length; i++)
424             {
425                 Bitstream[] bitstreams = bundles[i].getBitstreams();
426 
427                 // Unusual condition, but if no bitstreams, skip this bundle
428                 if (bitstreams.length == 0)
429                 {
430                     continue;
431                 }
432                         
433                 // First: we skip the license bundle, since it's included
434                 // elsewhere
435                 if (bitstreams[0].getFormat().getID() == licenseFormat)
436                 {
437                     continue;
438                 }
439 
440                 // Create a fileGrp
441                 FileGrp fileGrp = new FileGrp();
442 
443                 // Bundle name for USE attribute
444                 if ((bundles[i].getName() != null)
445                         && !bundles[i].getName().equals(""))
446                 {
447                     fileGrp.setUSE(bundles[i].getName());
448                 }
449 
450                 for (int bits = 0; bits < bitstreams.length; bits++)
451                 {
452                     // What's the persistent(-ish) ID?
453                     String bitstreamPID = ConfigurationManager
454                             .getProperty("dspace.url")
455                             + "/bitstream/"
456                             + item.getHandle()
457                             + "/"
458                             + bitstreams[bits].getSequenceID()
459                             + "/"
460                             + Util.encodeBitstreamName(bitstreams[bits].getName(),
461                                     "UTF-8");
462 
463                     edu.harvard.hul.ois.mets.File file = new edu.harvard.hul.ois.mets.File();
464 
465                     /*
466                      * ID: we use the unique part of the persistent ID, i.e. the
467                      * Handle + sequence number, but with _'s instead of /'s so
468                      * it's a legal xsd:ID.
469                      */
470                     String xmlIDstart = item.getHandle().replaceAll("/", "_")
471                             + "_";
472 
473                     file.setID(xmlIDstart + bitstreams[bits].getSequenceID());
474 
475                     String groupID = "GROUP_" + xmlIDstart
476                             + bitstreams[bits].getSequenceID();
477 
478                     /*
479                      * If we're in THUMBNAIL or TEXT bundles, the bitstream is
480                      * extracted text or a thumbnail, so we use the name to work
481                      * out which bitstream to be in the same group as
482                      */
483                     if ((bundles[i].getName() != null)
484                             && (bundles[i].getName().equals("THUMBNAIL") || bundles[i]
485                                     .getName().equals("TEXT")))
486                     {
487                         // Try and find the original bitstream, and chuck the
488                         // derived
489                         // bitstream in the same group
490                         Bitstream original = findOriginalBitstream(item,
491                                 bitstreams[bits]);
492 
493                         if (original != null)
494                         {
495                             groupID = "GROUP_" + xmlIDstart
496                                     + original.getSequenceID();
497                         }
498                     }
499 
500                     file.setGROUPID(groupID);
501                     file.setOWNERID(bitstreamPID);
502 
503                     // FIXME: ADMID should point to appropriate TechMD section
504                     // above
505                     file
506                             .setMIMETYPE(bitstreams[bits].getFormat()
507                                     .getMIMEType());
508 
509                     // FIXME: CREATED: no date
510                     file.setSIZE(bitstreams[bits].getSize());
511                     file.setCHECKSUM(bitstreams[bits].getChecksum());
512                     file.setCHECKSUMTYPE(Checksumtype.MD5);
513 
514                     // FLocat: filename is as in records, or full URL
515                     // FIXME: Duplicate filenames and characters illegal to
516                     // local OS may cause problems
517                     FLocat flocat = new FLocat();
518                     flocat.setLOCTYPE(Loctype.URL);
519                     if (fullURL)
520                     {
521                         flocat.setXlinkHref(bitstreamPID);
522                     }
523                     else
524                     {
525                         flocat.setXlinkHref(bitstreams[bits].getName());
526                     }
527 
528                     // Add FLocat to File, and File to FileGrp
529                     file.getContent().add(flocat);
530                     fileGrp.getContent().add(file);
531                 }
532 
533                 // Add fileGrp to fileSec
534                 fileSec.getContent().add(fileGrp);
535                 fileSecEmpty = false;
536             }
537 
538             // Add fileSec to document
539             if (!fileSecEmpty)
540             {
541                 mets.getContent().add(fileSec);
542             }
543             
544             // FIXME: Add Structmap here, but it is empty and we won't use it now.
545             StructMap structMap = new StructMap();
546             Div div = new Div();
547             structMap.getContent().add(div);
548             mets.getContent().add(structMap);
549 
550             
551             mets.validate(new MetsValidator());
552 
553             mets.write(new MetsWriter(os));
554         }
555         catch (MetsException e)
556         {
557             // We don't pass up a MetsException, so callers don't need to
558             // know the details of the METS toolkit
559             e.printStackTrace();
560             throw new IOException(e.getMessage());
561         }
562     }

564     /**
565      * Utility to find the license bitstream from an item
566      * 
567      * @param context
568      *            DSpace context
569      * @param item
570      *            the item
571      * @return the license as a string
572      * 
573      * @throws IOException
574      *             if the license bitstream can't be read
575      */
576     private static InputStream findLicense(Context context, Item item)
577             throws SQLException, IOException, AuthorizeException
578     {
579         Bundle[] bundles = item.getBundles();
580 
581         for (int i = 0; i < bundles.length; i++)
582         {
583             // Assume license will be in its own bundle
584             Bitstream[] bitstreams = bundles[i].getBitstreams();
585 
586             if (bitstreams.length > 0)
587             {
588                 if (bitstreams[0].getFormat().getID() == licenseFormat)
589                 {
590                     // Read the license into a string
591                     return bitstreams[0].retrieve();
592                 }
593             }
594         }
595 
596         // Oops! No license!
597         return null;
598     }

600     /**
601      * For a bitstream that's a thumbnail or extracted text, find the
602      * corresponding bitstream in the ORIGINAL bundle
603      * 
604      * @param item
605      *            the item we're dealing with
606      * @param derived
607      *            the derived bitstream
608      * 
609      * @return the corresponding original bitstream (or null)
610      */
611     private static Bitstream findOriginalBitstream(Item item, Bitstream derived)
612     				throws SQLException
613     {
614         Bundle[] bundles = item.getBundles();
615 
616         // Filename of original will be filename of the derived bitstream
617         // minus the extension (last 4 chars - .jpg or .txt)
618         String originalFilename = derived.getName().substring(0,
619                 derived.getName().length() - 4);
620 
621         // First find "original" bundle
622         for (int i = 0; i < bundles.length; i++)
623         {
624             if ((bundles[i].getName() != null)
625                     && bundles[i].getName().equals("ORIGINAL"))
626             {
627                 // Now find the corresponding bitstream
628                 Bitstream[] bitstreams = bundles[i].getBitstreams();
629 
630                 for (int bsnum = 0; bsnum < bitstreams.length; bsnum++)
631                 {
632                     if (bitstreams[bsnum].getName().equals(originalFilename))
633                     {
634                         return bitstreams[bsnum];
635                     }
636                 }
637             }
638         }
639 
640         // Didn't find it
641         return null;
642     }

644     /**
645      * Create MODS metadata from the DC in the item, and add to the given
646      * XmlData METS object.
647      * 
648      * @param item
649      *            the item
650      * @param xmlData
651      *            xmlData to add MODS to.
652      */
653     private static void createMODS(Item item, XmlData xmlData)
654     {
655         DCValue[] dc = item.getDC(Item.ANY, Item.ANY, Item.ANY);
656 
657         StringBuffer modsXML = new StringBuffer();
658 
659         for (int i = 0; i < dc.length; i++)
660         {
661             // Get the property name - element[.qualifier]
662             String propName = ((dc[i].qualifier == null) ? dc[i].element
663                     : (dc[i].element + "." + dc[i].qualifier));
664 
665             String modsMapping = dcToMODS.getProperty(propName);
666 
667             if (modsMapping == null)
668             {
669                 System.err.println("WARNING: No MODS mapping for " + propName);
670             }
671             else
672             {
673                 String value = dc[i].value;
674 
675                 // Replace all $'s with \$ so it doesn't trip up the replaceAll!
676                 if (value != null && value.length() > 0)
677                 {
678                     // RegExp note: Yes, there really does need to be this many backslashes!
679                     // To have \$ inserted in the replacement, both the backslash and the dollar
680                     // have to be escaped (backslash) - so the replacemenet string has to be
681                     // passed as \\\$. All of those backslashes then have to escaped in the literal
682                     // for them to be in string used!!!
683                     value = dc[i].value.replaceAll("\\$", "\\\\\\$");
684                 }
685 
686                 if (!(("description.provenance".equals(propName)) &&
687                     ((ConfigurationManager.getBooleanProperty("oai.mets.hide-provenance", false)))))
688                 {
689                     // Replace '%s' with DC value (with entities encoded)
690                     modsXML.append(modsMapping.replaceAll("%s", Utils
691                             .addEntities(value)));
692                     modsXML.append("\n"); // For readability
693                 }
694             }
695         }
696 
697         PreformedXML pXML = new PreformedXML(modsXML.toString());
698         xmlData.getContent().add(pXML);
699     }

701     /**
702      * Get the handle from the command line in the form 123.456/789. Doesn't
703      * matter if incoming handle has 'hdl:' or 'http://hdl....' before it.
704      * 
705      * @param original
706      *            Handle as passed in by user
707      * @return Handle as can be looked up in our table
708      */
709     private static String getHandleArg(String original)
710     {
711         if (original.startsWith("hdl:"))
712         {
713             return original.substring(4);
714         }
715 
716         if (original.startsWith("http://hdl.handle.net/"))
717         {
718             return original.substring(22);
719         }
720 
721         return original;
722     }
