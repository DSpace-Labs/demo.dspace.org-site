89  /**
90   * Item exporter to create simple AIPs for DSpace content. Currently exports
91   * individual items, or entire collections. For instructions on use, see
92   * printUsage() method.
93   * <P>
94   * ItemExport creates the simple AIP package that the importer also uses. It
95   * consists of:
96   * <P>
97   * /exportdir/42/ (one directory per item) / dublin_core.xml - qualified dublin
98   * core in RDF schema / contents - text file, listing one file per line / file1
99   * - files contained in the item / file2 / ...
100  * <P>
101  * issues -doesn't handle special characters in metadata (needs to turn &'s into
102  * &amp;, etc.)
103  * <P>
104  * Modified by David Little, UCSD Libraries 12/21/04 to allow the registration
105  * of files (bitstreams) into DSpace.
106  * 
107  * @author David Little
108  * @author Jay Paz
109  */
110 public class ItemExport

125     public static void main(String[] argv) throws Exception
126     {
127         // create an options object and populate it
128         CommandLineParser parser = new PosixParser();
129 
130         Options options = new Options();
131 
132         options.addOption("t", "type", true, "type: COLLECTION or ITEM");
133         options.addOption("i", "id", true, "ID or handle of thing to export");
134         options.addOption("d", "dest", true,
135                 "destination where you want items to go");
136         options.addOption("m", "migrate", false, "export for migration (remove handle and metadata that will be re-created in new system)");
137         options.addOption("n", "number", true,
138                 "sequence number to begin exporting items with");
139         options.addOption("z", "zip", true, "export as zip file (specify filename e.g. export.zip)");
140         options.addOption("h", "help", false, "help");
141 
142         CommandLine line = parser.parse(options, argv);
143 
144         String typeString = null;
145         String destDirName = null;
146         String myIDString = null;
147         int seqStart = -1;
148         int myType = -1;
149 
150         Item myItem = null;
151         Collection mycollection = null;
152 
153         if (line.hasOption('h'))
154         {
155             HelpFormatter myhelp = new HelpFormatter();
156             myhelp.printHelp("ItemExport\n", options);
157             System.out
158                     .println("\nfull collection: ItemExport -t COLLECTION -i ID -d dest -n number");
159             System.out
160                     .println("singleitem:       ItemExport -t ITEM -i ID -d dest -n number");
161 
162             System.exit(0);
163         }
164 
165         if (line.hasOption('t')) // type
166         {
167             typeString = line.getOptionValue('t');
168 
169             if (typeString.equals("ITEM"))
170             {
171                 myType = Constants.ITEM;
172             }
173             else if (typeString.equals("COLLECTION"))
174             {
175                 myType = Constants.COLLECTION;
176             }
177         }
178 
179         if (line.hasOption('i')) // id
180         {
181             myIDString = line.getOptionValue('i');
182         }
183 
184         if (line.hasOption('d')) // dest
185         {
186             destDirName = line.getOptionValue('d');
187         }
188 
189         if (line.hasOption('n')) // number
190         {
191             seqStart = Integer.parseInt(line.getOptionValue('n'));
192         }
193 
194         boolean migrate = false;
195         if (line.hasOption('m')) // number
196         {
197             migrate = true;
198         }
199 
200         boolean zip = false;
201         String zipFileName = "";
202         if (line.hasOption('z'))
203         {
204             zip = true;
205             zipFileName = line.getOptionValue('z');
206         }
207 
208         // now validate the args
209         if (myType == -1)
210         {
211             System.out
212                     .println("type must be either COLLECTION or ITEM (-h for help)");
213             System.exit(1);
214         }
215 
216         if (destDirName == null)
217         {
218             System.out
219                     .println("destination directory must be set (-h for help)");
220             System.exit(1);
221         }
222 
223         if (seqStart == -1)
224         {
225             System.out
226                     .println("sequence start number must be set (-h for help)");
227             System.exit(1);
228         }
229 
230         if (myIDString == null)
231         {
232             System.out
233                     .println("ID must be set to either a database ID or a handle (-h for help)");
234             System.exit(1);
235         }
236 
237         Context c = new Context();
238         c.setIgnoreAuthorization(true);
239 
240         if (myType == Constants.ITEM)
241         {
242             // first, is myIDString a handle?
243             if (myIDString.indexOf('/') != -1)
244             {
245                 myItem = (Item) HandleManager.resolveToObject(c, myIDString);
246 
247                 if ((myItem == null) || (myItem.getType() != Constants.ITEM))
248                 {
249                     myItem = null;
250                 }
251             }
252             else
253             {
254                 myItem = Item.find(c, Integer.parseInt(myIDString));
255             }
256 
257             if (myItem == null)
258             {
259                 System.out
260                         .println("Error, item cannot be found: " + myIDString);
261             }
262         }
263         else
264         {
265             if (myIDString.indexOf('/') != -1)
266             {
267                 // has a / must be a handle
268                 mycollection = (Collection) HandleManager.resolveToObject(c,
269                         myIDString);
270 
271                 // ensure it's a collection
272                 if ((mycollection == null)
273                         || (mycollection.getType() != Constants.COLLECTION))
274                 {
275                     mycollection = null;
276                 }
277             }
278             else if (myIDString != null)
279             {
280                 mycollection = Collection.find(c, Integer.parseInt(myIDString));
281             }
282 
283             if (mycollection == null)
284             {
285                 System.out.println("Error, collection cannot be found: "
286                         + myIDString);
287                 System.exit(1);
288             }
289         }
290 
291         if (zip)
292         {
293             ItemIterator items;
294             if (myItem != null)
295             {
296                 items = new ItemIterator(c, new ArrayList(myItem.getID()));
297             }
298             else
299             {
300                 System.out.println("Exporting from collection: " + myIDString);
301                 items = mycollection.getItems();
302             }
303             exportAsZip(c, items, destDirName, zipFileName, seqStart, migrate);
304         }
305         else
306         {
307             if (myItem != null)
308             {
309                 // it's only a single item
310                 exportItem(c, myItem, destDirName, seqStart, migrate);
311             }
312             else
313             {
314                 System.out.println("Exporting from collection: " + myIDString);
315 
316                 // it's a collection, so do a bunch of items
317                 ItemIterator i = mycollection.getItems();
318                 try
319                 {
320                     exportItem(c, i, destDirName, seqStart, migrate);
321                 }
322                 finally
323                 {
324                     if (i != null)
325                         i.close();
326                 }
327             }
328         }
329 
330         c.complete();
331     }

333     private static void exportItem(Context c, ItemIterator i,
334             String destDirName, int seqStart, boolean migrate) throws Exception
335     {
336         int mySequenceNumber = seqStart;
337         int counter = SUBDIR_LIMIT - 1;
338         int subDirSuffix = 0;
339         String fullPath = destDirName;
340         String subdir = "";
341         File dir;
342 
343         if (SUBDIR_LIMIT > 0)
344         {
345             dir = new File(destDirName);
346             if (!dir.isDirectory())
347             {
348                 throw new IOException(destDirName + " is not a directory.");
349             }
350         }
351 
352         System.out.println("Beginning export");
353 
354         while (i.hasNext())
355         {
356             if (SUBDIR_LIMIT > 0 && ++counter == SUBDIR_LIMIT)
357             {
358                 subdir = new Integer(subDirSuffix++).toString();
359                 fullPath = destDirName + File.separatorChar + subdir;
360                 counter = 0;
361 
362                 if (!new File(fullPath).mkdirs())
363                 {
364                     throw new IOException("Error, can't make dir " + fullPath);
365                 }
366             }
367 
368             System.out.println("Exporting item to " + mySequenceNumber);
369             exportItem(c, i.next(), fullPath, mySequenceNumber, migrate);
370             mySequenceNumber++;
371         }
372     }

374     private static void exportItem(Context c, Item myItem, String destDirName,
375             int seqStart, boolean migrate) throws Exception
376     {
377         File destDir = new File(destDirName);
378 
379         if (destDir.exists())
380         {
381             // now create a subdirectory
382             File itemDir = new File(destDir + "/" + seqStart);
383 
384             System.out.println("Exporting Item " + myItem.getID() + " to "
385                     + itemDir);
386 
387             if (itemDir.exists())
388             {
389                 throw new Exception("Directory " + destDir + "/" + seqStart
390                         + " already exists!");
391             }
392 
393             if (itemDir.mkdir())
394             {
395                 // make it this far, now start exporting
396                 writeMetadata(c, myItem, itemDir, migrate);
397                 writeBitstreams(c, myItem, itemDir);
398                 if (!migrate)
399                 {
400                     writeHandle(c, myItem, itemDir);
401                 }
402             }
403             else
404             {
405                 throw new Exception("Error, can't make dir " + itemDir);
406             }
407         }
408         else
409         {
410             throw new Exception("Error, directory " + destDirName
411                     + " doesn't exist!");
412         }
413     }

415     /**
416      * Discover the different schemas in use and output a seperate metadata XML
417      * file for each schema.
418      * 
419      * @param c
420      * @param i
421      * @param destDir
422      * @throws Exception
423      */
424     private static void writeMetadata(Context c, Item i, File destDir, boolean migrate)
425             throws Exception
426     {
427         // Build a list of schemas for the item
428         HashMap map = new HashMap();
429         DCValue[] dcorevalues = i.getMetadata(Item.ANY, Item.ANY, Item.ANY,
430                 Item.ANY);
431         for (int ii = 0; ii < dcorevalues.length; ii++)
432         {
433             map.put(dcorevalues[ii].schema, null);
434         }
435 
436         // Save each of the schemas into it's own metadata file
437         Iterator iterator = map.keySet().iterator();
438         while (iterator.hasNext())
439         {
440             String schema = (String) iterator.next();
441             writeMetadata(c, schema, i, destDir, migrate);
442         }
443     }

445     // output the item's dublin core into the item directory
446     private static void writeMetadata(Context c, String schema, Item i,
447             File destDir, boolean migrate) throws Exception
448     {
449         String filename;
450         if (schema.equals(MetadataSchema.DC_SCHEMA))
451         {
452             filename = "dublin_core.xml";
453         }
454         else
455         {
456             filename = "metadata_" + schema + ".xml";
457         }
458 
459         File outFile = new File(destDir, filename);
460 
461         System.out.println("Attempting to create file " + outFile);
462 
463         if (outFile.createNewFile())
464         {
465             BufferedOutputStream out = new BufferedOutputStream(
466                     new FileOutputStream(outFile));
467 
468             DCValue[] dcorevalues = i.getMetadata(schema, Item.ANY, Item.ANY,
469                     Item.ANY);
470 
471             // XML preamble
472             byte[] utf8 = "<?xml version=\"1.0\" encoding=\"utf-8\" standalone=\"no\"?>\n"
473                     .getBytes("UTF-8");
474             out.write(utf8, 0, utf8.length);
475 
476             String dcTag = "<dublin_core schema=\"" + schema + "\">\n";
477             utf8 = dcTag.getBytes("UTF-8");
478             out.write(utf8, 0, utf8.length);
479 
480             String dateIssued = null;
481             String dateAccessioned = null;
482 
483             for (DCValue dcv : dcorevalues)
484             {
485                 String qualifier = dcv.qualifier;
486 
487                 if (qualifier == null)
488                 {
489                     qualifier = "none";
490                 }
491 
492                 String language = dcv.language;
493 
494                 if (language != null)
495                 {
496                     language = " language=\"" + language + "\"";
497                 }
498                 else
499                 {
500                     language = "";
501                 }
502 
503                 utf8 = ("  <dcvalue element=\"" + dcv.element + "\" "
504                         + "qualifier=\"" + qualifier + "\""
505                         + language + ">"
506                         + Utils.addEntities(dcv.value) + "</dcvalue>\n")
507                         .getBytes("UTF-8");
508 
509                 if ((!migrate) ||
510                     (migrate && !(
511                      (dcv.element.equals("date") && qualifier.equals("issued")) ||
512                      (dcv.element.equals("date") && qualifier.equals("accessioned")) ||
513                      (dcv.element.equals("date") && qualifier.equals("available")) ||
514                      (dcv.element.equals("identifier") && qualifier.equals("uri") &&
515                       (dcv.value.startsWith("http://hdl.handle.net/" +
516                        ConfigurationManager.getProperty("handle.prefix") + "/"))) ||
517                      (dcv.element.equals("description") && qualifier.equals("provenance")) ||
518                      (dcv.element.equals("format") && qualifier.equals("extent")) ||
519                      (dcv.element.equals("format") && qualifier.equals("mimetype")))))
520                 {
521                     out.write(utf8, 0, utf8.length);
522                 }
523 
524                 // Store the date issued and assection to see if they are different
525                 // because we need to keep date.issued if they are, when migrating
526                 if ((dcv.element.equals("date") && qualifier.equals("issued")))
527                 {
528                     dateIssued = dcv.value;
529                 }
530                 if ((dcv.element.equals("date") && qualifier.equals("accessioned")))
531                 {
532                     dateAccessioned = dcv.value;
533                 }
534             }
535 
536             // When migrating, only keep date.issued if it is different to date.accessioned
537             if ((migrate) &&
538                 (dateIssued != null) &&
539                 (dateAccessioned != null) &&
540                 (!dateIssued.equals(dateAccessioned)))
541             {
542                 utf8 = ("  <dcvalue element=\"date\" "
543                         + "qualifier=\"issued\">"
544                         + Utils.addEntities(dateIssued) + "</dcvalue>\n")
545                         .getBytes("UTF-8");
546                 out.write(utf8, 0, utf8.length);
547             }
548 
549             utf8 = "</dublin_core>\n".getBytes("UTF-8");
550             out.write(utf8, 0, utf8.length);
551 
552             out.close();
553         }
554         else
555         {
556             throw new Exception("Cannot create dublin_core.xml in " + destDir);
557         }
558     }

588     /**
589      * Create both the bitstreams and the contents file. Any bitstreams that
590      * were originally registered will be marked in the contents file as such.
591      * However, the export directory will contain actual copies of the content
592      * files being exported.
593      * 
594      * @param c
595      *            the DSpace context
596      * @param i
597      *            the item being exported
598      * @param destDir
599      *            the item's export directory
600      * @throws Exception
601      *             if there is any problem writing to the export directory
602      */
603     private static void writeBitstreams(Context c, Item i, File destDir)
604             throws Exception
605     {
606         File outFile = new File(destDir, "contents");
607 
608         if (outFile.createNewFile())
609         {
610             PrintWriter out = new PrintWriter(new FileWriter(outFile));
611 
612             Bundle[] bundles = i.getBundles();
613 
614             for (int j = 0; j < bundles.length; j++)
615             {
616                 // bundles can have multiple bitstreams now...
617                 Bitstream[] bitstreams = bundles[j].getBitstreams();
618 
619                 String bundleName = bundles[j].getName();
620 
621                 for (int k = 0; k < bitstreams.length; k++)
622                 {
623                     Bitstream b = bitstreams[k];
624 
625                     String myName = b.getName();
626                     String oldName = myName;
627 
628                     String description = b.getDescription();
629                     if ((description != null) && (!description.equals("")))
630                     {
631                         description = "\tdescription:" + description;
632                     } else
633                     {
634                         description = "";
635                     }
636 
637                     String primary = "";
638                     if (bundles[j].getPrimaryBitstreamID() == b.getID()) {
639                         primary = "\tprimary:true ";
640                     }
641 
642                     int myPrefix = 1; // only used with name conflict
643 
644                     InputStream is = b.retrieve();
645 
646                     boolean isDone = false; // done when bitstream is finally
647                     // written
648 
649                     while (!isDone)
650                     {
651                         if (myName.contains(File.separator))
652                         {
653                             String dirs = myName.substring(0, myName
654                                     .lastIndexOf(File.separator));
655                             File fdirs = new File(destDir + File.separator
656                                     + dirs);
657                             fdirs.mkdirs();
658                         }
659 
660                         File fout = new File(destDir, myName);
661 
662                         if (fout.createNewFile())
663                         {
664                             FileOutputStream fos = new FileOutputStream(fout);
665                             Utils.bufferedCopy(is, fos);
666                             // close streams
667                             is.close();
668                             fos.close();
669 
670                             // write the manifest file entry
671                             if (b.isRegisteredBitstream())
672                             {
673                                 out.println("-r -s " + b.getStoreNumber()
674                                         + " -f " + myName +
675                                         "\tbundle:" + bundleName +
676                                         primary + description);
677                             }
678                             else
679                             {
680                                 out.println(myName + "\tbundle:" + bundleName +
681                                             primary + description);
682                             }
683 
684                             isDone = true;
685                         }
686                         else
687                         {
688                             myName = myPrefix + "_" + oldName; // keep
689                             // appending
690                             // numbers to the
691                             // filename until
692                             // unique
693                             myPrefix++;
694                         }
695                     }
696                 }
697             }
698 
699             // close the contents file
700             out.close();
701         }
702         else
703         {
704             throw new Exception("Cannot create contents in " + destDir);
705         }
706     }

708     /**
709      * Method to perform an export and save it as a zip file.
710      *
711      * @param context The DSpace Context
712      * @param items The items to export
713      * @param destDirName The directory to save the export in
714      * @param zipFileName The name to save the zip file as
715      * @param seqStart The first number in the sequence
716      * @param migrate Whether to use the migrate option or not
717      * @throws Exception
718      */
719     public static void exportAsZip(Context context, ItemIterator items,
720                                    String destDirName, String zipFileName,
721                                    int seqStart, boolean migrate) throws Exception
722     {
723         String workDir = getExportWorkDirectory() +
724                          System.getProperty("file.separator") +
725                          zipFileName;
726 
727         File wkDir = new File(workDir);
728         if (!wkDir.exists())
729         {
730             wkDir.mkdirs();
731         }
732 
733         File dnDir = new File(destDirName);
734         if (!dnDir.exists())
735         {
736             dnDir.mkdirs();
737         }
738 
739         // export the items using normal export method
740         exportItem(context, items, workDir, seqStart, migrate);
741 
742         // now zip up the export directory created above
743         zip(workDir, destDirName + System.getProperty("file.separator") + zipFileName);
744     }

823      * Does the work creating a List with all the Items in the Community or
824      * Collection It then kicks off a new Thread to export the items, zip the
825      * export directory and send confirmation email
826      * 
827      * @param dsObjects
828      *            - List of dspace objects to process
829      * @param context
830      *            - the dspace context
831      * @param additionalEmail
832      *            - email address to cc in addition the the current user email
833      * @throws Exception
834      */
835     private static void processDownloadableExport(List<DSpaceObject> dsObjects,
836             Context context, final String additionalEmail, boolean toMigrate) throws Exception
837     {
838         final EPerson eperson = context.getCurrentUser();
839         final boolean migrate = toMigrate;
840 
841         // before we create a new export archive lets delete the 'expired'
842         // archives
843         deleteOldExportArchives(eperson.getID());
844 
845         // keep track of the commulative size of all bitstreams in each of the
846         // items
847         // it will be checked against the config file entry
848         float size = 0;
849         final ArrayList<Integer> items = new ArrayList<Integer>();
850         for (DSpaceObject dso : dsObjects)
851         {
852             if (dso.getType() == Constants.COMMUNITY)
853             {
854                 Community community = (Community) dso;
855                 // get all the collections in the community
856                 Collection[] collections = community.getCollections();
857                 for (Collection collection : collections)
858                 {
859                     // get all the items in each collection
860                     ItemIterator iitems = collection.getItems();
861                     try
862                     {
863                         while (iitems.hasNext())
864                         {
865                             Item item = iitems.next();
866                             // get all the bundles in the item
867                             Bundle[] bundles = item.getBundles();
868                             for (Bundle bundle : bundles)
869                             {
870                                 // get all the bitstreams in each bundle
871                                 Bitstream[] bitstreams = bundle.getBitstreams();
872                                 for (Bitstream bit : bitstreams)
873                                 {
874                                     // add up the size
875                                     size += bit.getSize();
876                                 }
877                             }
878                             items.add(item.getID());
879                         }
880                     }
881                     finally
882                     {
883                         if (iitems != null)
884                             iitems.close();
885                     }
886                 }
887             }
888             else if (dso.getType() == Constants.COLLECTION)
889             {
890                 Collection collection = (Collection) dso;
891                 // get all the items in the collection
892                 ItemIterator iitems = collection.getItems();
893                 try
894                 {
895                     while (iitems.hasNext())
896                     {
897                         Item item = iitems.next();
898                         // get all thebundles in the item
899                         Bundle[] bundles = item.getBundles();
900                         for (Bundle bundle : bundles)
901                         {
902                             // get all the bitstreams in the bundle
903                             Bitstream[] bitstreams = bundle.getBitstreams();
904                             for (Bitstream bit : bitstreams)
905                             {
906                                 // add up the size
907                                 size += bit.getSize();
908                             }
909                         }
910                         items.add(item.getID());
911                     }
912                 }
913                 finally
914                 {
915                     if (iitems != null)
916                         iitems.close();
917                 }
918             }
919             else if (dso.getType() == Constants.ITEM)
920             {
921                 Item item = (Item) dso;
922                 // get all the bundles in the item
923                 Bundle[] bundles = item.getBundles();
924                 for (Bundle bundle : bundles)
925                 {
926                     // get all the bitstreams in the bundle
927                     Bitstream[] bitstreams = bundle.getBitstreams();
928                     for (Bitstream bit : bitstreams)
929                     {
930                         // add up the size
931                         size += bit.getSize();
932                     }
933                 }
934                 items.add(item.getID());
935             }
936             else
937             {
938                 // nothing to do just ignore this type of DSPaceObject
939             }
940         }
941 
942         // check the size of all the bitstreams against the configuration file
943         // entry if it exists
944         String megaBytes = ConfigurationManager
945                 .getProperty("org.dspace.app.itemexport.max.size");
946         if (megaBytes != null)
947         {
948             float maxSize = 0;
949             try
950             {
951                 maxSize = Float.parseFloat(megaBytes);
952             }
953             catch (Exception e)
954             {
955                 // ignore...configuration entry may not be present
956             }
957 
958             if (maxSize > 0)
959             {
960                 if (maxSize < (size / 1048576.00))
961                 { // a megabyte
962                     throw new ItemExportException(ItemExportException.EXPORT_TOO_LARGE,
963                                                   "The overall size of this export is too large.  Please contact your administrator for more information.");
964                 }
965             }
966         }
967 
968         // if we have any items to process then kick off annonymous thread
969         if (items.size() > 0)
970         {
971             Thread go = new Thread()
972             {
973                 public void run()
974                 {
975                     Context context = null;
976                     ItemIterator iitems = null;
977                     try
978                     {
979                         // create a new dspace context
980                         context = new Context();
981                         // ignore auths
982                         context.setIgnoreAuthorization(true);
983                         iitems = new ItemIterator(context, items);
984 
985                         String fileName = assembleFileName("item", eperson,
986                                 new Date());
987                         String workDir = getExportWorkDirectory()
988                                 + System.getProperty("file.separator")
989                                 + fileName;
990                         String downloadDir = getExportDownloadDirectory(eperson
991                                 .getID());
992 
993                         File wkDir = new File(workDir);
994                         if (!wkDir.exists())
995                         {
996                             wkDir.mkdirs();
997                         }
998 
999                         File dnDir = new File(downloadDir);
1000                         if (!dnDir.exists())
1001                         {
1002                             dnDir.mkdirs();
1003                         }
1004 
1005                         // export the items using normal export method
1006                         exportItem(context, iitems, workDir, 1, migrate);
1007                         // now zip up the export directory created above
1008                         zip(workDir, downloadDir
1009                                 + System.getProperty("file.separator")
1010                                 + fileName + ".zip");
1011                         // email message letting user know the file is ready for
1012                         // download
1013                         emailSuccessMessage(context, eperson, fileName + ".zip");
1014                         // return to enforcing auths
1015                         context.setIgnoreAuthorization(false);
1016                     }
1017                     catch (Exception e1)
1018                     {
1019                         try
1020                         {
1021                             emailErrorMessage(eperson, e1.getMessage());
1022                         }
1023                         catch (Exception e)
1024                         {
1025                             // wont throw here
1026                         }
1027                         throw new RuntimeException(e1);
1028                     }
1029                     finally
1030                     {
1031                         if (iitems != null)
1032                             iitems.close();
1033                         
1034                         // Make sure the database connection gets closed in all conditions.
1035                     	try {
1036 							context.complete();
1037 						} catch (SQLException sqle) {
1038 							context.abort();
1039 						}
1040                     }
1041                 }
1042 
1043             };
1044 
1045             go.isDaemon();
1046             go.start();
1047         }

1397     private static void zipFiles(File cpFile, String strSource,
1398             String strTarget, ZipOutputStream cpZipOutputStream)
1399             throws Exception
1400     {
1401         int byteCount;
1402         final int DATA_BLOCK_SIZE = 2048;
1403         FileInputStream cpFileInputStream;
1404         if (cpFile.isDirectory())
1405         {
1406             File[] fList = cpFile.listFiles();
1407             for (int i = 0; i < fList.length; i++)
1408             {
1409                 zipFiles(fList[i], strSource, strTarget, cpZipOutputStream);
1410             }
1411         }
1412         else
1413         {
1414             try
1415             {
1416                 if (cpFile.getAbsolutePath().equalsIgnoreCase(strTarget))
1417                 {
1418                     return;
1419                 }
1420                 String strAbsPath = cpFile.getPath();
1421                 String strZipEntryName = strAbsPath.substring(strSource
1422                         .length() + 1, strAbsPath.length());
1423 
1424                 // byte[] b = new byte[ (int)(cpFile.length()) ];
1425 
1426                 cpFileInputStream = new FileInputStream(cpFile);
1427                 ZipEntry cpZipEntry = new ZipEntry(strZipEntryName);
1428                 cpZipOutputStream.putNextEntry(cpZipEntry);
1429 
1430                 byte[] b = new byte[DATA_BLOCK_SIZE];
1431                 while ((byteCount = cpFileInputStream.read(b, 0,
1432                         DATA_BLOCK_SIZE)) != -1)
1433                 {
1434                     cpZipOutputStream.write(b, 0, byteCount);
1435                 }
1436 
1437                 // cpZipOutputStream.write(b, 0, (int)cpFile.length());
1438                 cpZipOutputStream.closeEntry();
1439             }
1440             catch (Exception e)
1441             {
1442                 throw e;
1443             }
