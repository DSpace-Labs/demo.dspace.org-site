89      /**      
90       * 
91       *  Working around Item API to delete a value-specific DCValue
92       *  For a given element/qualifier/lang:
93       *      get all DCValues
94       *      clear (i.e. delete) all of these DCValues
95       *      add them back, minus the one to actually delete
96       *  
97       * 
98       * @param item
99       * @param dtom
100      * @param isLanguageStrict - 
101      * 
102      * @return true if metadata field is found with matching value and was deleted
103      */
104     public static boolean deleteMetadataByValue(Item item, DtoMetadata dtom, boolean isLanguageStrict)
105     {   	
106     	DCValue[] ar = null;
107     	
108     	if (isLanguageStrict)
109     	{   // get all for given type
110     		ar = item.getMetadata(dtom.schema, dtom.element, dtom.qualifier, dtom.language);  
111     	}
112     	else
113     	{
114     		ar = item.getMetadata(dtom.schema, dtom.element, dtom.qualifier, Item.ANY);  
115     	}
116     	
117     	boolean found = false;
118     	
119     	//build new set minus the one to delete
120     	List<String> vals = new ArrayList<String>();
121     	for (DCValue dcv : ar)
122     	{
123     		if (dcv.value.equals(dtom.value))
124     		{
125     			found = true;
126     		}
127     		else
128     		{
129     			vals.add(dcv.value);
130     		}
131     	}
132     	
133     	if (found)  //remove all for given type  ??synchronize this block??
134     	{   
135         	if (isLanguageStrict)
136         	{           		
137         		item.clearMetadata(dtom.schema, dtom.element, dtom.qualifier, dtom.language);
138         	}
139         	else
140         	{
141         		item.clearMetadata(dtom.schema, dtom.element, dtom.qualifier, Item.ANY);
142         	}
143     	
144     		item.addMetadata(dtom.schema, dtom.element, dtom.qualifier, dtom.language, vals.toArray(new String[0]));   	
145     	}
146 		return found;
147     }

211     /**
212      *  Modification of method from ItemImporter.loadDublinCore 
213      *  as a Factory method
214      * 
215      * @param docBuilder  - 
216      * @param is - InputStream of dublin_core.xml
217      * @return list of DtoMetadata representing the metadata fields relating to an Item
218      * @throws SQLException
219      * @throws IOException
220      * @throws ParserConfigurationException
221      * @throws SAXException
222      * @throws TransformerException
223      * @throws AuthorizeException
224      */
225     public static List<DtoMetadata> loadDublinCore(DocumentBuilder docBuilder, InputStream is)
226     throws SQLException, IOException, ParserConfigurationException,
227            SAXException, TransformerException, AuthorizeException
228 	{    	
229 		Document document = docBuilder.parse(is);
230 		
231 		List<DtoMetadata> dtomList = new ArrayList<DtoMetadata>();
232 		
233 		// Get the schema, for backward compatibility we will default to the
234 		// dublin core schema if the schema name is not available in the import file		
235 		String schema = null;
236 		NodeList metadata = XPathAPI.selectNodeList(document, "/dublin_core");		
237 		Node schemaAttr = metadata.item(0).getAttributes().getNamedItem("schema");
238 		if (schemaAttr == null)
239 		{
240 		    schema = MetadataSchema.DC_SCHEMA;
241 		}
242 		else
243 		{
244 		    schema = schemaAttr.getNodeValue();
245 		}
246 
247 		// Get the nodes corresponding to formats
248 		NodeList dcNodes = XPathAPI.selectNodeList(document, "/dublin_core/dcvalue");
249 		
250 		for (int i = 0; i < dcNodes.getLength(); i++)
251 		{
252 		    Node n = dcNodes.item(i);		    
253 	        String value = getStringValue(n).trim(); 
254 	        // compensate for empty value getting read as "null", which won't display
255 	        if (value == null)
256 	        {
257 	            value = "";
258 	        }
259 	        String element = getAttributeValue(n, "element");
260 	        if (element != null)
261 	        {
262 	        	element = element.trim();
263 	        }
264 	        String qualifier = getAttributeValue(n, "qualifier"); 
265 	        if (qualifier != null)
266 	        {
267 	        	qualifier = qualifier.trim();
268 	        }
269 	        String language = getAttributeValue(n, "language");
270 	        if (language != null)
271 	        {
272 	        	language = language.trim();
273 	        }
274 
275 	        if (qualifier.equals("none") || "".equals(qualifier))
276 	        {
277 	            qualifier = null;
278 	        }
279 	        
280 	        // a goofy default, but consistent with DSpace treatment elsewhere  
281 	        if (language == null)
282 	        {
283 	            language = "en";
284 	        }
285 	        else if (language.equals(""))
286 	        {
287 	            language = ConfigurationManager.getProperty("default.language");
288 	        }
289 	        
290 		    DtoMetadata dtom = DtoMetadata.create(schema, element, qualifier, language, value);
291 		    ItemUpdate.pr(dtom.toString());
292 		    dtomList.add(dtom);
293 		}
294 		return dtomList;
295 	}

297     /**
298      *    Write dublin_core.xml 
299      * 
300      * @param docBuilder
301      * @param dtomList
302      * @return xml document
303      * @throws ParserConfigurationException
304      * @throws TransformerConfigurationException
305      * @throws TransformerException
306      */
307 	public static Document writeDublinCore(DocumentBuilder docBuilder, List<DtoMetadata> dtomList)
308 	throws ParserConfigurationException, TransformerConfigurationException, TransformerException
309 	{		
310         Document doc = docBuilder.newDocument();
311         Element root = doc.createElement("dublin_core");
312         doc.appendChild(root);
313     
314         for (DtoMetadata dtom : dtomList)
315         {
316         	Element mel = doc.createElement("dcvalue");
317         	mel.setAttribute("element", dtom.element);
318         	if (dtom.qualifier == null)
319         	{
320         		mel.setAttribute("qualifier", "none");
321         	}
322         	else
323         	{
324         		mel.setAttribute("qualifier", dtom.qualifier);
325         	}
326  
327         	if ((dtom.language == null) || (dtom.language.equals("")))
328         	{
329         		mel.setAttribute("language", "en");
330         	}
331         	else
332         	{
333         		mel.setAttribute("language", dtom.language);
334         	}
335         	mel.setTextContent(dtom.value);
336         	root.appendChild(mel);
337         }
338         
339         return doc;       
340 	}

361     /**
362      * Lookup an attribute from a DOM node.
363      * @param n
364      * @param name
365      * @return
366      */
367     private static String getAttributeValue(Node n, String name)
368     {
369         NamedNodeMap nm = n.getAttributes();
370 
371         for (int i = 0; i < nm.getLength(); i++)
372         {
373             Node node = nm.item(i);
374 
375             if (name.equals(node.getNodeName()))
376             {
377                 return node.getNodeValue();
378             }
379         }
380 
381         return "";
382     }

384     /**
385      * Return the String value of a Node.
386      * @param node
387      * @return
388      */
389     private static String getStringValue(Node node)
390     {
391         String value = node.getNodeValue();
392 
393         if (node.hasChildNodes())
394         {
395             Node first = node.getFirstChild();
396 
397             if (first.getNodeType() == Node.TEXT_NODE)
398             {
399                 return first.getNodeValue();
400             }
401         }
402 
403         return value;
404     }

535 	/**
536 	 *    Parses metadata field given in the form <schema>.<element>[.<qualifier>|.*]
537 	 *    checks for correct number of elements (2 or 3) and for empty strings
538 	 *    
539 	 *    @return String Array
540 	 *    @throws ParseException if validity checks fail
541 	 *    
542 	 */
543 	public static String[] parseCompoundForm(String compoundForm)
544 	throws ParseException
545 	{
546 		String[] ar = compoundForm.split("\\s*\\.\\s*");  //trim ends
547 				
548 		if (ar[0].equals(""))
549 		{
550 			throw new ParseException("schema is empty string: " + compoundForm, 0);
551 		}
552 		
553 		if ((ar.length < 2) || (ar.length > 3) || ar[1].equals(""))
554 		{
555 			throw new ParseException("element is malformed or empty string: " + compoundForm, 0);
556 		}
557 		
558 		return ar;
559 	}
