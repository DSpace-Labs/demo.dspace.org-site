77      /**
78       * Obtain a new context object. If a context object has already been created
79       * for this HTTP request, it is re-used, otherwise it is created. If a user
80       * has authenticated with the system, the current user of the context is set
81       * appropriately.
82       * 
83       * @param request
84       *            the HTTP request
85       * 
86       * @return a context object
87       */
88      public static Context obtainContext(HttpServletRequest request)
89              throws SQLException
90      {
91          
92          //Set encoding to UTF-8, if not set yet
93          //This avoids problems of using the HttpServletRequest
94          //in the getSpecialGroups() for an AuthenticationMethod,  
95          //which causes the HttpServletRequest to default to 
96          //non-UTF-8 encoding.
97          try
98          {
99              if(request.getCharacterEncoding()==null)
100                 request.setCharacterEncoding(Constants.DEFAULT_ENCODING);
101         }
102         catch(Exception e)
103         {
104             log.error("Unable to set encoding to UTF-8.", e);
105         }
106         
107         Context c = (Context) request.getAttribute("dspace.context");
108         
109 
110         if (c == null)
111         {
112             // No context for this request yet
113             c = new Context();
114             HttpSession session = request.getSession();
115 
116             // See if a user has authentication
117             Integer userID = (Integer) session.getAttribute(
118                     "dspace.current.user.id");
119 
120             if (userID != null)
121             {
122                 String remAddr = (String)session.getAttribute("dspace.current.remote.addr");
123                 if (remAddr != null && remAddr.equals(request.getRemoteAddr()))
124                 {
125                 	EPerson e = EPerson.find(c, userID.intValue());
126 
127                 	Authenticate.loggedIn(c, request, e);
128                 }
129                 else
130                 {
131                     log.warn("POSSIBLE HIJACKED SESSION: request from "+
132                              request.getRemoteAddr()+" does not match original "+
133                              "session address: "+remAddr+". Authentication rejected.");
134                 }
135             }
136 
137             // Set any special groups - invoke the authentication mgr.
138             int[] groupIDs = AuthenticationManager.getSpecialGroups(c, request);
139 
140             for (int i = 0; i < groupIDs.length; i++)
141             {
142                 c.setSpecialGroup(groupIDs[i]);
143                 log.debug("Adding Special Group id="+String.valueOf(groupIDs[i]));
144             }
145 
146             // Set the session ID and IP address
147             c.setExtraLogInfo("session_id=" + request.getSession().getId() + ":ip_addr=" + request.getRemoteAddr());
148 
149             // Store the context in the request
150             request.setAttribute("dspace.context", c);
151         }
152         
153         // Set the locale to be used
154         Locale sessionLocale = getSessionLocale(request);
155         Config.set(request.getSession(), Config.FMT_LOCALE, sessionLocale);
156         c.setCurrentLocale(sessionLocale);
157 
158         return c;
159     }

191     /**
192      * Put the original request URL into the request object as an attribute for
193      * later use. This is necessary because forwarding a request removes this
194      * information. The attribute is only written if it hasn't been before; thus
195      * it can be called after a forward safely.
196      * 
197      * @param request
198      *            the HTTP request
199      */
200     public static void storeOriginalURL(HttpServletRequest request)
201     {
202         String orig = (String) request.getAttribute("dspace.original.url");
203 
204         if (orig == null)
205         {
206             String fullURL = request.getRequestURL().toString();
207 
208             if (request.getQueryString() != null)
209             {
210                 fullURL = fullURL + "?" + request.getQueryString();
211             }
212 
213             request.setAttribute("dspace.original.url", fullURL);
214         }
215     }

233     /**
234      * Write a human-readable version of a DCDate.
235      * 
236      * @param d
237      *            the date
238      * @param time
239      *            if true, display the time with the date
240      * @param localTime
241      *            if true, adjust for local timezone, otherwise GMT
242      * @param request
243      *            the servlet request           
244      * 
245      * @return the date in a human-readable form.
246      */
247     public static String displayDate(DCDate d, boolean time, boolean localTime, HttpServletRequest request)
248     {
249         StringBuffer sb = new StringBuffer();
250         Locale locale = ((Context)request.getAttribute("dspace.context")).getCurrentLocale();
251         if (locale == null) locale = I18nUtil.DEFAULTLOCALE;
252 
253         if (d != null)
254         {
255             int year;
256             int month;
257             int day;
258             int hour;
259             int minute;
260             int second;
261 
262             if (localTime)
263             {
264                 year = d.getYear();
265                 month = d.getMonth();
266                 day = d.getDay();
267                 hour = d.getHour();
268                 minute = d.getMinute();
269                 second = d.getSecond();
270             }
271             else
272             {
273                 year = d.getYearGMT();
274                 month = d.getMonthGMT();
275                 day = d.getDayGMT();
276                 hour = d.getHourGMT();
277                 minute = d.getMinuteGMT();
278                 second = d.getSecondGMT();
279             }
280 
281             if (year > -1)
282             {
283                 if (month > -1)
284                 {
285                     if (day > -1)
286                     {
287                         sb.append(day + "-");
288                     }
289                     String monthName = DCDate.getMonthName(month, getSessionLocale(request));
290                     int monthLength = monthName.length();
291                     monthLength = monthLength > 2 ? 3 : monthLength;
292                     sb.append(monthName.substring(0, monthLength) + "-");
293                 }
294 
295                 sb.append(year + " ");
296             }
297 
298             if (time && (hour > -1))
299             {
300                 String hr = String.valueOf(hour);
301 
302                 while (hr.length() < 2)
303                 {
304                     hr = "0" + hr;
305                 }
306 
307                 String mn = String.valueOf(minute);
308 
309                 while (mn.length() < 2)
310                 {
311                     mn = "0" + mn;
312                 }
313 
314                 String sc = String.valueOf(second);
315 
316                 while (sc.length() < 2)
317                 {
318                     sc = "0" + sc;
319                 }
320 
321                 sb.append(hr + ":" + mn + ":" + sc + " ");
322             }
323         }
324         else
325         {
326             sb.append("Unset");
327         }
328 
329         return (sb.toString());
330     }

332     /**
333      * Return a string for logging, containing useful information about the
334      * current request - the URL, the method and parameters.
335      * 
336      * @param request
337      *            the request object.
338      * @return a multi-line string containing information about the request.
339      */
340     public static String getRequestLogInfo(HttpServletRequest request)
341     {
342         String report;
343 
344         report = "-- URL Was: " + getOriginalURL(request) + "\n";
345         report = report + "-- Method: " + request.getMethod() + "\n";
346 
347         // First write the parameters we had
348         report = report + "-- Parameters were:\n";
349 
350         Enumeration e = request.getParameterNames();
351 
352         while (e.hasMoreElements())
353         {
354             String name = (String) e.nextElement();
355 
356             if (name.equals("login_password"))
357             {
358                 // We don't want to write a clear text password
359                 // to the log, even if it's wrong!
360                 report = report + "-- " + name + ": *not logged*\n";
361             }
362             else
363             {
364                 report = report + "-- " + name + ": \""
365                         + request.getParameter(name) + "\"\n";
366             }
367         }
368 
369         return report;
370     }

373     /**
374      * Get the Locale for a session according to the user's language selection or language preferences.
375      * Order of selection
376      * - language selected via UI
377      * - language as set by application
378      * - language browser default
379      * 
380      * @param request
381      *        the request Object
382      * @return supportedLocale
383      *         Locale supported by this DSpace Instance for this request
384      */
385     public static Locale getSessionLocale(HttpServletRequest request)
386 
387     {
388         String paramLocale = request.getParameter("locale");
389         Locale sessionLocale = null;
390         Locale supportedLocale = null;
391 
392         if (paramLocale != null && paramLocale != "")
393         {
394             /* get session locale according to user selection */
395             sessionLocale = new Locale(paramLocale);
396         }
397         
398      
399         if (sessionLocale == null)
400         {
401             /* get session locale set by application */
402             HttpSession session = request.getSession();
403             sessionLocale = (Locale) Config.get(session, Config.FMT_LOCALE);
404         }
405 
406         /*
407          * if session not set by selection or application then default browser
408          * locale
409          */
410         if (sessionLocale == null)
411         {
412             sessionLocale = request.getLocale();
413         }
414         
415         if (sessionLocale == null)
416         {
417             sessionLocale = I18nUtil.DEFAULTLOCALE;
418         }
419         supportedLocale =  I18nUtil.getSupportedLocale(sessionLocale);
420         
421         return supportedLocale;
422     }

425     /**
426      * Send an alert to the designated "alert recipient" - that is, when a
427      * database error or internal error occurs, this person is sent an e-mail
428      * with details.
429      * <P>
430      * The recipient is configured via the "alert.recipient" property in
431      * <code>dspace.cfg</code>. If this property is omitted, no alerts are
432      * sent.
433      * <P>
434      * This method "swallows" any exception that might occur - it will just be
435      * logged. This is because this method will usually be invoked as part of an
436      * error handling routine anyway.
437      * 
438      * @param request
439      *            the HTTP request leading to the error
440      * @param exception
441      *            the exception causing the error, or null
442      */
443     public static void sendAlert(HttpServletRequest request, Exception exception)
444     {
445         String logInfo = UIUtil.getRequestLogInfo(request);
446         Context c = (Context) request.getAttribute("dspace.context");
447         Locale locale = getSessionLocale(request);
448         EPerson user = null;
449 
450         try
451         {
452             String recipient = ConfigurationManager
453                     .getProperty("alert.recipient");
454 
455             if (recipient != null)
456             {
457                 Email email = ConfigurationManager.getEmail(I18nUtil.getEmailFilename(locale, "internal_error"));
458                 email.addRecipient(recipient);
459                 email.addArgument(ConfigurationManager
460                         .getProperty("dspace.url"));
461                 email.addArgument(new Date());
462                 email.addArgument(request.getSession().getId());
463                 email.addArgument(logInfo);
464 
465                 String stackTrace;
466 
467                 if (exception != null)
468                 {
469                     StringWriter sw = new StringWriter();
470                     PrintWriter pw = new PrintWriter(sw);
471                     exception.printStackTrace(pw);
472                     pw.flush();
473                     stackTrace = sw.toString();
474                 }
475                 else
476                 {
477                     stackTrace = "No exception";
478                 }
479 
480                 email.addArgument(stackTrace);
481                 try
482                 {
483                     user = c.getCurrentUser();
484                 }
485                 catch (Exception e)
486                 {
487                     log.warn("No context, the database might be down or the connection pool exhausted.");
488                 }
489                 
490                 if (user != null)
491                 {
492                     email.addArgument(user.getFullName() + " (" + user.getEmail() + ")");
493                 }
494                 else
495                 {
496                     email.addArgument("Anonymous");
497                 }
498                 email.addArgument(request.getRemoteAddr());
499                 email.send();
500             }
501         }
502         catch (Exception e)
503         {
504             // Not much we can do here!
505             log.warn("Unable to send email alert", e);
506         }
507     }
