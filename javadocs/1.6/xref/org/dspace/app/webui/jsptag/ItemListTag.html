157     public int doStartTag() throws JspException
158     {
159         JspWriter out = pageContext.getOut();
160         HttpServletRequest hrq = (HttpServletRequest) pageContext.getRequest();
161 
162         boolean emphasiseDate = false;
163         boolean emphasiseTitle = false;
164 
165         if (emphColumn != null)
166         {
167             emphasiseDate = emphColumn.equalsIgnoreCase("date");
168             emphasiseTitle = emphColumn.equalsIgnoreCase("title");
169         }
170 
171         // get the elements to display
172         String configLine = null;
173         String widthLine  = null;
174 
175         if (sortOption != null)
176         {
177             if (configLine == null)
178             {
179                 configLine = ConfigurationManager.getProperty("webui.itemlist.sort." + sortOption.getName() + ".columns");
180                 widthLine  = ConfigurationManager.getProperty("webui.itemlist.sort." + sortOption.getName() + ".widths");
181             }
182 
183             if (configLine == null)
184             {
185                 configLine = ConfigurationManager.getProperty("webui.itemlist." + sortOption.getName() + ".columns");
186                 widthLine  = ConfigurationManager.getProperty("webui.itemlist." + sortOption.getName() + ".widths");
187             }
188         }
189 
190         if (configLine == null)
191         {
192             configLine = ConfigurationManager.getProperty("webui.itemlist.columns");
193             widthLine  = ConfigurationManager.getProperty("webui.itemlist.widths");
194         }
195 
196         // Have we read a field configration from dspace.cfg?
197         if (configLine != null)
198         {
199             // If thumbnails are disabled, strip out any thumbnail column from the configuration
200             if (!showThumbs && configLine.contains("thumbnail"))
201             {
202                 // Ensure we haven't got any nulls
203                 configLine = configLine  == null ? "" : configLine;
204                 widthLine  = widthLine   == null ? "" : widthLine;
205 
206                 // Tokenize the field and width lines
207                 StringTokenizer llt = new StringTokenizer(configLine,  ",");
208                 StringTokenizer wlt = new StringTokenizer(widthLine, ",");
209 
210                 StringBuilder newLLine = new StringBuilder();
211                 StringBuilder newWLine = new StringBuilder();
212                 while (llt.hasMoreTokens() || wlt.hasMoreTokens())
213                 {
214                     String listTok  = llt.hasMoreTokens() ? llt.nextToken() : null;
215                     String widthTok = wlt.hasMoreTokens() ? wlt.nextToken() : null;
216 
217                     // Only use the Field and Width tokens, if the field isn't 'thumbnail'
218                     if (listTok == null || !listTok.trim().equals("thumbnail"))
219                     {
220                         if (listTok != null)
221                         {
222                             if (newLLine.length() > 0)
223                                 newLLine.append(",");
224 
225                             newLLine.append(listTok);
226                         }
227 
228                         if (widthTok != null)
229                         {
230                             if (newWLine.length() > 0)
231                                 newWLine.append(",");
232 
233                             newWLine.append(widthTok);
234                         }
235                     }
236                 }
237 
238                 // Use the newly built configuration file
239                 configLine  = newLLine.toString();
240                 widthLine = newWLine.toString();
241             }
242 
243             listFields = configLine;
244             listWidths = widthLine;
245         }
246 
247         // get the date and title fields
248         String dateLine = ConfigurationManager.getProperty("webui.browse.index.date");
249         if (dateLine != null)
250         {
251             dateField = dateLine;
252         }
253 
254         String titleLine = ConfigurationManager.getProperty("webui.browse.index.title");
255         if (titleLine != null)
256         {
257             titleField = titleLine;
258         }
259 
260         String authorLine = ConfigurationManager.getProperty("webui.browse.author-field");
261         if (authorLine != null)
262         {
263             authorField = authorLine;
264         }
265 
266         // Arrays used to hold the information we will require when outputting each row
267         String[] fieldArr  = listFields.split("\\s*,\\s*");
268         String[] widthArr  = listWidths == null ? new String[0] : listWidths.split("\\s*,\\s*");
269         boolean isDate[]   = new boolean[fieldArr.length];
270         boolean emph[]     = new boolean[fieldArr.length];
271         boolean isAuthor[] = new boolean[fieldArr.length];
272         boolean viewFull[] = new boolean[fieldArr.length];
273         String[] browseType = new String[fieldArr.length];
274         String[] cOddOrEven = new String[fieldArr.length];
275 
276         try
277         {
278             // Get the interlinking configuration too
279             CrossLinks cl = new CrossLinks();
280 
281             // Get a width for the table
282             String tablewidth = ConfigurationManager.getProperty("webui.itemlist.tablewidth");
283 
284             // If we have column widths, output a fixed layout table - faster for browsers to render
285             // but not if we have to add an 'edit item' button - we can't know how big it will be
286             if (widthArr.length > 0 && widthArr.length == fieldArr.length && !linkToEdit)
287             {
288                 // If the table width has been specified, we can make this a fixed layout
289                 if (!StringUtils.isEmpty(tablewidth))
290                 {
291                     out.println("<table style=\"width: " + tablewidth + "; table-layout: fixed;\" align=\"center\" class=\"miscTable\" summary=\"This table browses all dspace content\">");
292                 }
293                 else
294                 {
295                     // Otherwise, don't constrain the width
296                     out.println("<table align=\"center\" class=\"miscTable\" summary=\"This table browses all dspace content\">");
297                 }
298 
299                 // Output the known column widths
300                 out.print("<colgroup>");
301 
302                 for (int w = 0; w < widthArr.length; w++)
303                 {
304                     out.print("<col width=\"");
305 
306                     // For a thumbnail column of width '*', use the configured max width for thumbnails
307                     if (fieldArr[w].equals("thumbnail") && widthArr[w].equals("*"))
308                     {
309                         out.print(thumbItemListMaxWidth);
310                     }
311                     else
312                     {
313                         out.print(StringUtils.isEmpty(widthArr[w]) ? "*" : widthArr[w]);
314                     }
315 
316                     out.print("\" />");
317                 }
318 
319                 out.println("</colgroup>");
320             }
321             else if (!StringUtils.isEmpty(tablewidth))
322             {
323                 out.println("<table width=\"" + tablewidth + "\" align=\"center\" class=\"miscTable\" summary=\"This table browses all dspace content\">");
324             }
325             else
326             {
327                 out.println("<table align=\"center\" class=\"miscTable\" summary=\"This table browses all dspace content\">");
328             }
329 
330             // Output the table headers
331             out.println("<tr>");
332 
333             for (int colIdx = 0; colIdx < fieldArr.length; colIdx++)
334             {
335                 String field = fieldArr[colIdx].toLowerCase().trim();
336                 cOddOrEven[colIdx] = (((colIdx + 1) % 2) == 0 ? "Odd" : "Even");
337 
338                 // find out if the field is a date
339                 if (field.indexOf("(date)") > 0)
340                 {
341                     field = field.replaceAll("\\(date\\)", "");
342                     isDate[colIdx] = true;
343                 }
344 
345                 // Cache any modifications to field
346                 fieldArr[colIdx] = field;
347 
348                 // find out if this is the author column
349                 if (field.equals(authorField))
350                 {
351                     isAuthor[colIdx] = true;
352                 }
353 
354                 // find out if this field needs to link out to other browse views
355                 if (cl.hasLink(field))
356                 {
357                     browseType[colIdx] = cl.getLinkType(field);
358                     viewFull[colIdx] = BrowseIndex.getBrowseIndex(browseType[colIdx]).isItemIndex();
359                 }
360 
361                 // find out if we are emphasising this field
362                 if (field.equals(emphColumn))
363                 {
364                     emph[colIdx] = true;
365                 }
366                 else if ((field.equals(dateField) && emphasiseDate) ||
367                         (field.equals(titleField) && emphasiseTitle))
368                 {
369                     emph[colIdx] = true;
370                 }
371 
372                 // prepare the strings for the header
373                 String id = "t" + Integer.toString(colIdx + 1);
374                 String css = "oddRow" + cOddOrEven[colIdx] + "Col";
375                 String message = "itemlist." + field;
376 
377                 // output the header
378                 out.print("<th id=\"" + id +  "\" class=\"" + css + "\">"
379                         + (emph[colIdx] ? "<strong>" : "")
380                         + LocaleSupport.getLocalizedMessage(pageContext, message)
381                         + (emph[colIdx] ? "</strong>" : "") + "</th>");
382             }
383 
384             if (linkToEdit)
385             {
386                 String id = "t" + Integer.toString(cOddOrEven.length + 1);
387                 String css = "oddRow" + cOddOrEven[cOddOrEven.length - 2] + "Col";
388 
389                 // output the header
390                 out.print("<th id=\"" + id +  "\" class=\"" + css + "\">"
391                         + (emph[emph.length - 2] ? "<strong>" : "")
392                         + "&nbsp;" //LocaleSupport.getLocalizedMessage(pageContext, message)
393                         + (emph[emph.length - 2] ? "</strong>" : "") + "</th>");
394             }
395 
396             out.print("</tr>");
397 
398             // now output each item row
399             for (int i = 0; i < items.length; i++)
400             {
401                 // now prepare the XHTML frag for this division
402             	out.print("<tr>"); 
403                 String rOddOrEven;
404                 if (i == highlightRow)
405                 {
406                     rOddOrEven = "highlight";
407                 }
408                 else
409                 {
410                     rOddOrEven = ((i % 2) == 1 ? "odd" : "even");
411                 }
412 
413                 for (int colIdx = 0; colIdx < fieldArr.length; colIdx++)
414                 {
415                     String field = fieldArr[colIdx];
416 
417                     // get the schema and the element qualifier pair
418                     // (Note, the schema is not used for anything yet)
419                     // (second note, I hate this bit of code.  There must be
420                     // a much more elegant way of doing this.  Tomcat has
421                     // some weird problems with variations on this code that
422                     // I tried, which is why it has ended up the way it is)
423                     StringTokenizer eq = new StringTokenizer(field, ".");
424 
425                     String[] tokens = { "", "", "" };
426                     int k = 0;
427                     while(eq.hasMoreTokens())
428                     {
429                         tokens[k] = eq.nextToken().toLowerCase().trim();
430                         k++;
431                     }
432                     String schema = tokens[0];
433                     String element = tokens[1];
434                     String qualifier = tokens[2];
435 
436                     // first get hold of the relevant metadata for this column
437                     DCValue[] metadataArray;
438                     if (qualifier.equals("*"))
439                     {
440                         metadataArray = items[i].getMetadata(schema, element, Item.ANY, Item.ANY);
441                     }
442                     else if (qualifier.equals(""))
443                     {
444                         metadataArray = items[i].getMetadata(schema, element, null, Item.ANY);
445                     }
446                     else
447                     {
448                         metadataArray = items[i].getMetadata(schema, element, qualifier, Item.ANY);
449                     }
450 
451                     // save on a null check which would make the code untidy
452                     if (metadataArray == null)
453                     {
454                         metadataArray = new DCValue[0];
455                     }
456 
457                     // now prepare the content of the table division
458                     String metadata = "-";
459                     if (field.equals("thumbnail"))
460                     {
461                         metadata = getThumbMarkup(hrq, items[i]);
462                     }
463                     if (metadataArray.length > 0)
464                     {
465                         // format the date field correctly
466                         if (isDate[colIdx])
467                         {
468                             DCDate dd = new DCDate(metadataArray[0].value);
469                             metadata = UIUtil.displayDate(dd, false, false, hrq);
470                         }
471                         // format the title field correctly for withdrawn items (ie. don't link)
472                         else if (field.equals(titleField) && items[i].isWithdrawn())
473                         {
474                             metadata = Utils.addEntities(metadataArray[0].value);
475                         }
476                         // format the title field correctly
477                         else if (field.equals(titleField))
478                         {
479                             metadata = "<a href=\"" + hrq.getContextPath() + "/handle/"
480                             + items[i].getHandle() + "\">"
481                             + Utils.addEntities(metadataArray[0].value)
482                             + "</a>";
483                         }
484                         // format all other fields
485                         else
486                         {
487                             // limit the number of records if this is the author field (if
488                             // -1, then the limit is the full list)
489                             boolean truncated = false;
490                             int loopLimit = metadataArray.length;
491                             if (isAuthor[colIdx])
492                             {
493                                 int fieldMax = (authorLimit > 0 ? authorLimit : metadataArray.length);
494                                 loopLimit = (fieldMax > metadataArray.length ? metadataArray.length : fieldMax);
495                                 truncated = (fieldMax < metadataArray.length);
496                                 log.debug("Limiting output of field " + field + " to " + Integer.toString(loopLimit) + " from an original " + Integer.toString(metadataArray.length));
497                             }
498 
499                             StringBuffer sb = new StringBuffer();
500                             for (int j = 0; j < loopLimit; j++)
501                             {
502                                 String startLink = "";
503                                 String endLink = "";
504                                 if (!StringUtils.isEmpty(browseType[colIdx]) && !disableCrossLinks)
505                                 {
506                                     String argument;
507                                     String value;
508                                     if (metadataArray[j].authority != null &&
509                                             metadataArray[j].confidence >= MetadataAuthorityManager.getManager()
510                                                 .getMinConfidence(metadataArray[j].schema, metadataArray[j].element, metadataArray[j].qualifier))
511                                     {
512                                         argument = "authority";
513                                         value = metadataArray[j].authority;
514                                     }
515                                     else
516                                     {
517                                         argument = "value";
518                                         value = metadataArray[j].value;
519                                     }
520                                     if (viewFull[colIdx])
521                                     {
522                                         argument = "vfocus";
523                                     }
524                                     startLink = "<a href=\"" + hrq.getContextPath() + "/browse?type=" + browseType[colIdx] + "&amp;" +
525                                         argument + "=" + URLEncoder.encode(value,"UTF-8");
526 
527                                     if (metadataArray[j].language != null)
528                                     {
529                                         startLink = startLink + "&amp;" +
530                                             argument + "_lang=" + URLEncoder.encode(metadataArray[j].language, "UTF-8");
531                                     }
532 
533                                     if ("authority".equals(argument))
534                                     {
535                                         startLink += "\" class=\"authority " +browseType[colIdx] + "\">";
536                                     }
537                                     else
538                                     {
539                                         startLink = startLink + "\">";
540                                     }
541                                     endLink = "</a>";
542                                 }
543                                 sb.append(startLink);
544                                 sb.append(Utils.addEntities(metadataArray[j].value));
545                                 sb.append(endLink);
546                                 if (j < (loopLimit - 1))
547                                 {
548                                     sb.append("; ");
549                                 }
550                             }
551                             if (truncated)
552                             {
553                                 String etal = LocaleSupport.getLocalizedMessage(pageContext, "itemlist.et-al");
554                                 sb.append(", " + etal);
555                             }
556                             metadata = "<em>" + sb.toString() + "</em>";
557                         }
558                     }
559 
560                     // prepare extra special layout requirements for dates
561                     String extras = "";
562                     if (isDate[colIdx])
563                     {
564                         extras = "nowrap=\"nowrap\" align=\"right\"";
565                     }
566 
567                     String id = "t" + Integer.toString(colIdx + 1);
568                     out.print("<td headers=\"" + id + "\" class=\""
569                         + rOddOrEven + "Row" + cOddOrEven[colIdx] + "Col\" " + extras + ">"
570                         + (emph[colIdx] ? "<strong>" : "") + metadata + (emph[colIdx] ? "</strong>" : "")
571                         + "</td>");
572                 }
573 
574                 // Add column for 'edit item' links
575                 if (linkToEdit)
576                 {
577                     String id = "t" + Integer.toString(cOddOrEven.length + 1);
578 
579                         out.print("<td headers=\"" + id + "\" class=\""
580                             + rOddOrEven + "Row" + cOddOrEven[cOddOrEven.length - 2] + "Col\" nowrap>"
581                             + "<form method=\"get\" action=\"" + hrq.getContextPath() + "/tools/edit-item\">"
582                             + "<input type=\"hidden\" name=\"handle\" value=\"" + items[i].getHandle() + "\" />"
583                             + "<input type=\"submit\" value=\"Edit Item\" /></form>"
584                             + "</td>");
585                     }
586 
587                 out.println("</tr>");
588             }
589 
590             // close the table
591             out.println("</table>");
592         }
593         catch (IOException ie)
594         {
595             throw new JspException(ie);
596         }
597         catch (BrowseException e)
598         {
599             throw new JspException(e);
600         }
601 
602         return SKIP_BODY;
603     }

729     /* get the required thumbnail config items */
730     private void getThumbSettings()
731     {
732         showThumbs = ConfigurationManager
733                 .getBooleanProperty("webui.browse.thumbnail.show");
734 
735         if (showThumbs)
736         {
737             thumbItemListMaxHeight = ConfigurationManager
738                     .getIntProperty("webui.browse.thumbnail.maxheight");
739 
740             if (thumbItemListMaxHeight == 0)

768     /*
769      * Get the (X)HTML width and height attributes. As the browser is being used
770      * for scaling, we only scale down otherwise we'll get hideously chunky
771      * images. This means the media filter should be run with the maxheight and
772      * maxwidth set greater than or equal to the size of the images required in
773      * the search/browse
774      */
775     private String getScalingAttr(HttpServletRequest hrq, Bitstream bitstream)
776             throws JspException
777     {
778         BufferedImage buf;
779 
780         try
781         {
782             Context c = UIUtil.obtainContext(hrq);
783 
784             InputStream is = BitstreamStorageManager.retrieve(c, bitstream
785                     .getID());
786 
787             //AuthorizeManager.authorizeAction(bContext, this, Constants.READ);
788             // 	read in bitstream's image
789             buf = ImageIO.read(is);
790             is.close();
791         }
792         catch (SQLException sqle)
793         {
794             throw new JspException(sqle.getMessage());
795         }
796         catch (IOException ioe)
797         {
798             throw new JspException(ioe.getMessage());
799         }
800 
801         // now get the image dimensions
802         float xsize = (float) buf.getWidth(null);
803         float ysize = (float) buf.getHeight(null);
804 
805         // scale by x first if needed
806         if (xsize > (float) thumbItemListMaxWidth)
807         {
808             // calculate scaling factor so that xsize * scale = new size (max)
809             float scale_factor = (float) thumbItemListMaxWidth / xsize;
810 
811             // now reduce x size and y size
812             xsize = xsize * scale_factor;
813             ysize = ysize * scale_factor;
814         }
815 
816         // scale by y if needed
817         if (ysize > (float) thumbItemListMaxHeight)
818         {
819             float scale_factor = (float) thumbItemListMaxHeight / ysize;
820 
821             // now reduce x size
822             // and y size
823             xsize = xsize * scale_factor;
824             ysize = ysize * scale_factor;
825         }
826 
827         StringBuffer sb = new StringBuffer("width=\"").append(xsize).append(
828                 "\" height=\"").append(ysize).append("\"");
829 
830         return sb.toString();
831     }
