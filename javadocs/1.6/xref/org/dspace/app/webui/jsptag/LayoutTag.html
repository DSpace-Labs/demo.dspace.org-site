62  /**
63   * Tag for HTML page layout ("skin").
64   * <P>
65   * This tag <em>sets</em> request attributes that should be used by the header
66   * and footer to render the page appropriately:
67   * <P>
68   * <ul>
69   * <li><code>dspace.layout.title</code> - title of page</li>
70   * <li><code>dspace.layout.locbar</code> - value will Boolean true or false
71   * </li>
72   * <li><code>dspace.layout.parenttitles</code> - a <code>List</code> of
73   * <code>String</code>s corresponding with titles to put in the location bar.
74   * Only set if <code>dspace.layout.locbar</code> is true</li>
75   * <li><code>dspace.layout.parentlinks</code> - a <code>List</code> of
76   * <code>String</code>s corresponding with links to put in the location bar.
77   * Empty strings mean no link. Will only be set if
78   * <code>dspace.layout.locbar</code> is true.</li>
79   * <li><code>dspace.layout.navbar</code> - value will be "off", or the
80   * navigation bar to include, e.g. "/layout/navbar_default.jsp"</li>
81   * <li><code>dspace.layout.sidebar</code> - contents of the sidebar</li>
82   * <li><code>dspace.current.user</code> - the EPerson currently logged in, or
83   * <code>null</code> if anonymous access</li>
84   * <li><code>dspace.layout.feeddata</code> - <code>String</code>. "NONE"
85   * means no feed from this page; otherwise the Handle of object (community or
86   * collection) the feed is from</li>
87   * <li><code>dspace.layout.linkparts</code> - <code>String[]</code>. A
88   * cycling sequence of: 2nd part of MIME type (e.g. <code>rdf+rss</code>);
89   * title of feed; path component to go after /feed/ for the actual feed URL
90   * (e.g. <code>rss_1.0</code>). Hence, this array will have 3<i>n</i>
91   * elements where <i>n</i> is the number of feeds.</li>
92   * </ul>
93   * <p>
94   *
95   * * Additionally the following parameter may be set elsewhere in a Servlet.
96   * <p>
97   * <ul>
98   * <li><code>dspace.layout.head</code> - extra data to include verbatim in
99   * the &lt;head&gt; element of the page</li>
100  * </ul>
101  * 
102  * Furthermore it sets the content type of the response to text/html using UTF-8
103  * to ensure this will be returned in the HTTP header.
104  * </p>
105  * 
106  * @author Robert Tansley
107  * @version $Revision: 3705 $
108  */

152     public int doStartTag() throws JspException
153     {
154         ServletRequest request = pageContext.getRequest();
155 
156         // header file
157         String header = "/layout/header-default.jsp";
158 
159         // Choose default style unless one is specified
160         if (style != null)
161         {
162             header = "/layout/header-" + style.toLowerCase() + ".jsp";
163         }
164 
165         // Sort out location bar
166         if (locbar == null)
167         {
168             locbar = "auto";
169         }
170 
171         // These lists will contain titles and links to put in the location
172         // bar
173         List parents = new ArrayList();
174         List parentLinks = new ArrayList();
175 
176         if (locbar.equalsIgnoreCase("off"))
177         {
178             // No location bar
179             request.setAttribute("dspace.layout.locbar", new Boolean(false));
180         }
181         else
182         {
183             // We'll always add "DSpace Home" to the a location bar
184             parents.add(ConfigurationManager.getProperty("dspace.name"));
185 
186             if (locbar.equalsIgnoreCase("nolink"))
187             {
188                 parentLinks.add("");
189             }
190             else
191             {
192                 parentLinks.add("/");
193             }
194 
195             // Add other relevant components to the location bar
196             if (locbar.equalsIgnoreCase("link"))
197             {
198                 // "link" mode - next thing in location bar is taken from
199                 // parameters of tag, with a link
200                 if (parentTitle != null)
201                 {
202                     parents.add(parentTitle);
203                     parentLinks.add(parentLink);
204                 }
205                 else if (parentTitleKey != null)
206                 {
207                     parents.add(LocaleSupport.getLocalizedMessage(pageContext,
208                             parentTitleKey));
209                     parentLinks.add(parentLink);
210                 }
211 
212             }
213             else if (locbar.equalsIgnoreCase("commLink"))
214             {
215                 // "commLink" mode - show all parent communities
216                 Community[] comms = (Community[]) request
217                         .getAttribute("dspace.communities");
218 
219                 if (comms != null)
220                 {
221                     for (int i = 0; i < comms.length; i++)
222                     {
223                         parents.add(comms[i].getMetadata("name"));
224                         parentLinks.add("/handle/" + comms[i].getHandle());
225                     }
226                 }
227             }
228             else if (locbar.equalsIgnoreCase("nolink"))
229             {
230                 // "nolink" mode - next thing in location bar is taken from
231                 // parameters of tag, with no link
232                 if (parentTitle != null)
233                 {
234                     parents.add(parentTitle);
235                     parentLinks.add("");
236                 }
237             }
238             else
239             {
240                 // Grab parents from the URL - these should have been picked up
241                 // by the HandleServlet
242                 Collection col = (Collection) request
243                         .getAttribute("dspace.collection");
244                 Community[] comms = (Community[]) request
245                         .getAttribute("dspace.communities");
246 
247                 if (comms != null)
248                 {
249                     for (int i = 0; i < comms.length; i++)
250                     {
251                         parents.add(comms[i].getMetadata("name"));
252                         parentLinks.add("/handle/" + comms[i].getHandle());
253                     }
254 
255                     if (col != null)
256                     {
257                         parents.add(col.getMetadata("name"));
258                         parentLinks.add("/handle/" + col.getHandle());
259                     }
260                 }
261             }
262 
263             request.setAttribute("dspace.layout.locbar", new Boolean(true));
264         }
265 
266         request.setAttribute("dspace.layout.parenttitles", parents);
267         request.setAttribute("dspace.layout.parentlinks", parentLinks);
268 
269         // Navigation bar: "default" is default :)
270         if (navbar == null)
271         {
272             navbar = "default";
273         }
274 
275         if (navbar.equals("off"))
276         {
277             request.setAttribute("dspace.layout.navbar", "off");
278         }
279         else
280         {
281             request.setAttribute("dspace.layout.navbar", "/layout/navbar-"
282                     + navbar + ".jsp");
283         }
284 
285         // Set title
286         if (title != null)
287         {
288             request.setAttribute("dspace.layout.title", title);
289         }
290         else if (titleKey != null)
291         {
292             request.setAttribute("dspace.layout.title", LocaleSupport
293                     .getLocalizedMessage(pageContext, titleKey));
294         }
295         else
296         {
297             request.setAttribute("dspace.layout.title", "NO TITLE");
298         }
299         
300         // Set feedData if present
301         if (feedData != null && ! "NONE".equals(feedData))
302         {
303             // set the links' reference - community or collection
304         	boolean commLinks = feedData.startsWith("comm:");
305         	boolean collLinks = feedData.startsWith("coll:");
306         	if ( commLinks )
307         	{
308                 Community com = (Community)request.getAttribute("dspace.community");
309         		request.setAttribute("dspace.layout.feedref", com.getHandle());
310         	}
311         	else if( collLinks )
312         	{
313         		Collection col = (Collection)request.getAttribute("dspace.collection");
314         		request.setAttribute("dspace.layout.feedref", col.getHandle());
315         	}
316         	else //feed is across all of DSpace and not Community/Collection specific
317         	{
318         		request.setAttribute("dspace.layout.feedref", FeedServlet.SITE_FEED_KEY);
319         	}
320         	
321         	// build a list of link attributes for each link format
322         	String[] formats = feedData.substring(feedData.indexOf(":")+1).split(",");
323         	List linkParts = new ArrayList();
324         	// each link has a mime-type, title, and format (used in href URL)
325         	for (int i = 0; i < formats.length; i++)
326         	{
327         		if("rss_1.0".equals(formats[i]))
328         		{
329         			linkParts.add("rdf+xml");
330         		}
331         		else
332         		{
333         			linkParts.add("rss+xml");
334         		}
335         		
336         		if (commLinks)
337         		{
338         			linkParts.add("Items in Community");
339         		}
340         		else if(collLinks)
341         		{
342         			linkParts.add("Items in Collection");
343         		}
344         		else
345         		{
346         			linkParts.add("Items in " + ConfigurationManager.getProperty("dspace.name"));
347         		}
348         		
349         		linkParts.add(formats[i]);
350         	}
351         	request.setAttribute("dspace.layout.linkparts", linkParts);
352         }
353         else
354         {
355         	request.setAttribute("dspace.layout.feedref", "NONE" );
356         }
357 
358         // Now include the header
359         try
360         {
361             HttpServletResponse response = (HttpServletResponse) pageContext
362                     .getResponse();
363 
364             // Set headers to prevent browser caching, if appropriate
365             if ((noCache != null) && noCache.equalsIgnoreCase("true"))
366             {
367                 response.addDateHeader("expires", 1);
368                 response.addHeader("Pragma", "no-cache");
369                 response.addHeader("Cache-control", "no-store");
370             }
371 
372             // Ensure the HTTP header will declare that UTF-8 is used
373             // in the response.
374             response.setContentType("text/html; charset=UTF-8");
375 
376             ServletConfig config = pageContext.getServletConfig();
377 
378             RequestDispatcher rd = config.getServletContext()
379                     .getRequestDispatcher(header);
380 
381             rd.include(request, response);
382         }
383         catch (IOException ioe)
384         {
385             throw new JspException("Got IOException: " + ioe);
386         }
387         catch (ServletException se)
388         {
389             log.warn("Exception", se.getRootCause());
390             throw new JspException("Got ServletException: " + se);
391         }
392 
393         return EVAL_BODY_INCLUDE;
394     }
395 
396     public int doEndTag() throws JspException
397     {
398         // Footer file to use
399         String footer = "/layout/footer-default.jsp";
400 
401         // Choose default flavour unless one is specified
402         if (style != null)
403         {
404             footer = "/layout/footer-" + style.toLowerCase() + ".jsp";
405         }
406 
407         try
408         {
409             // Ensure body is included before footer
410             pageContext.getOut().flush();
411 
412             // Context objects
413             ServletRequest request = pageContext.getRequest();
414             ServletResponse response = pageContext.getResponse();
415             ServletConfig config = pageContext.getServletConfig();
416 
417             if (sidebar != null)
418             {
419                 request.setAttribute("dspace.layout.sidebar", sidebar);
420             }
421 
422             RequestDispatcher rd = config.getServletContext()
423                     .getRequestDispatcher(footer);
424 
425             rd.include(request, response);
426         }
427         catch (ServletException se)
428         {
429             throw new JspException("Got ServletException: " + se);
430         }
431         catch (IOException ioe)
432         {
433             throw new JspException("Got IOException: " + ioe);
434         }
435 
436         return EVAL_PAGE;
437     }
