78  /**
79   * <P>
80   * JSP tag for displaying an item.
81   * </P>
82   * <P>
83   * The fields that are displayed can be configured in <code>dspace.cfg</code>
84   * using the <code>webui.itemdisplay.(style)</code> property. The form is
85   * </P>
86   * 
87   * <PRE>
88   * 
89   * &lt;schema prefix&gt;.&lt;element&gt;[.&lt;qualifier&gt;|.*][(date)|(link)], ...
90   * 
91   * </PRE>
92   * 
93   * <P>
94   * For example:
95   * </P>
96   * 
97   * <PRE>
98   * 
99   * dc.title = Dublin Core element 'title' (unqualified)
100  * dc.title.alternative = DC element 'title', qualifier 'alternative'
101  * dc.title.* = All fields with Dublin Core element 'title' (any or no qualifier)
102  * dc.identifier.uri(link) = DC identifier.uri, render as a link
103  * dc.date.issued(date) = DC date.issued, render as a date
104  * dc.identifier.doi(doi) = DC identifier.doi, render as link to http://dx.doi.org
105  * dc.identifier.hdl(handle) = DC identifier.hanlde, render as link to http://hdl.handle.net
106  * dc.relation.isPartOf(resolver) = DC relation.isPartOf, render as link to the base url of the resolver 
107  *                                  according to the specified urn in the metadata value (doi:xxxx, hdl:xxxxx, 
108  *                                  urn:issn:xxxx, etc.)
109  * 
110  * </PRE>
111  * 
112  * <P>
113  * When using "resolver" in webui.itemdisplay to render identifiers as resolvable
114  * links, the base URL is taken from <code>webui.resolver.<n>.baseurl</code> 
115  * where <code>webui.resolver.<n>.urn</code> matches the urn specified in the metadata value.
116  * The value is appended to the "baseurl" as is, so the baseurl need to end with slash almost in any case.
117  * If no urn is specified in the value it will be displayed as simple text.
118  * 
119  * <PRE>
120  * 
121  * webui.resolver.1.urn = doi
122  * webui.resolver.1.baseurl = http://dx.doi.org/
123  * webui.resolver.2.urn = hdl
124  * webui.resolver.2.baseurl = http://hdl.handle.net/
125  * 
126  * </PRE>
127  * 
128  * For the doi and hdl urn defaults values are provided, respectively http://dx.doi.org/ and 
129  * http://hdl.handle.net/ are used.<br> 
130  * 
131  * If a metadata value with style: "doi", "handle" or "resolver" matches a URL
132  * already, it is simply rendered as a link with no other manipulation.
133  * </P>
134  * 
135  * <PRE>
136  * 
137  * <P>
138  * If an item has no value for a particular field, it won't be displayed. The
139  * name of the field for display will be drawn from the current UI dictionary,
140  * using the key:
141  * </P>
142  * 
143  * <PRE>
144  * 
145  * &quot;metadata.&lt;style.&gt;.&lt;field&gt;&quot;
146  * 
147  * e.g. &quot;metadata.thesis.dc.title&quot; &quot;metadata.thesis.dc.contributor.*&quot;
148  * &quot;metadata.thesis.dc.date.issued&quot;
149  * 
150  * 
151  * if this key is not found will be used the more general one
152  * 
153  * &quot;metadata.&lt;field&gt;&quot;
154  * 
155  * e.g. &quot;metadata.dc.title&quot; &quot;metadata.dc.contributor.*&quot;
156  * &quot;metadata.dc.date.issued&quot;
157  * 
158  * </PRE>
159  * 
160  * <P>
161  * You need to specify which strategy use for select the style for an item.
162  * </P>
163  * 
164  * <PRE>
165  * 
166  * plugin.single.org.dspace.app.webui.util.StyleSelection = \
167  *                      org.dspace.app.webui.util.CollectionStyleSelection
168  *                      #org.dspace.app.webui.util.MetadataStyleSelection
169  * 
170  * </PRE>
171  * 
172  * <P>
173  * With the Collection strategy you can also specify which collections use which
174  * views.
175  * </P>
176  * 
177  * <PRE>
178  * 
179  * webui.itemdisplay.&lt;style&gt;.collections = &lt;collection handle&gt;, ...
180  * 
181  * </PRE>
182  * 
183  * <P>
184  * FIXME: This should be more database-driven
185  * </P>
186  * 
187  * <PRE>
188  * 
189  * webui.itemdisplay.thesis.collections = 123456789/24, 123456789/35
190  * 
191  * </PRE>
192  * 
193  * <P>
194  * With the Metadata strategy you MUST specify which metadata use as name of the
195  * style.
196  * </P>
197  * 
198  * <PRE>
199  * 
200  * webui.itemdisplay.metadata-style = schema.element[.qualifier|.*]
201  * 
202  * e.g. &quot;dc.type&quot;
203  * 
204  * </PRE>
205  * 
206  * @author Robert Tansley
207  * @version $Revision: 4386 $
208  */

244     public ItemTag()
245     {
246         super();
247         getThumbSettings();
248         linkedMetadata = new HashMap<String, String>();
249         String linkMetadata;
250         for (int i = 1; null != (linkMetadata = ConfigurationManager.getProperty("webui.browse.link."+i)); i++)
251         {            
252             String[] linkedMetadataSplit = linkMetadata.split(":");
253             String indexName = linkedMetadataSplit[0].trim();
254             String metadataName = linkedMetadataSplit[1].trim();
255             linkedMetadata.put(indexName, metadataName);
256         }
257         
258         urn2baseurl = new HashMap<String, String>();
259 
260         String urn;
261         for (int i = 1; null != (urn = ConfigurationManager.getProperty("webui.resolver."+i+".urn")); i++){
262             String baseurl = ConfigurationManager.getProperty("webui.resolver."+i+".baseurl"); 
263             if (baseurl != null){
264             urn2baseurl.put(ConfigurationManager
265                     .getProperty("webui.resolver."+i+".urn"),
266                     baseurl);
267             } else {
268                 log.warn("Wrong webui.resolver configuration, you need to specify both webui.resolver.<n>.urn and webui.resolver.<n>.baseurl: missing baseurl for n = "+i);
269             }
270         }
271         
272         // Set sensible default if no config is found for doi & handle
273         if (!urn2baseurl.containsKey("doi")){
274             urn2baseurl.put("doi",DOI_DEFAULT_BASEURL);
275         }
276         
277         if (!urn2baseurl.containsKey("hdl")){
278             urn2baseurl.put("hdl",HANDLE_DEFAULT_BASEURL);
279         }
280     }

282     public int doStartTag() throws JspException
283     {
284         try
285         {
286             if (style == null || style.equals(""))
287             {
288                 style = styleSelection.getStyleForItem(item);
289             }
290 
291             if (style.equals("full"))
292             {
293                 renderFull();
294             }
295             else
296             {
297                 render();
298             }
299         }
300         catch (SQLException sqle)
301         {
302             throw new JspException(sqle);
303         }
304         catch (IOException ie)
305         {
306             throw new JspException(ie);
307         }
308 
309         return SKIP_BODY;
310     }

382     /**
383      * Render an item in the given style
384      */
385     private void render() throws IOException, SQLException
386     {
387         JspWriter out = pageContext.getOut();
388         HttpServletRequest request = (HttpServletRequest)pageContext.getRequest();
389         Context context = UIUtil.obtainContext(request);
390         
391         String configLine = styleSelection.getConfigurationForStyle(style);
392 
393         if (configLine == null)
394             configLine = defaultFields;
395 
396         out.println("<center><table class=\"itemDisplayTable\">");
397 
398         /*
399          * Break down the configuration into fields and display them
400          * 
401          * FIXME?: it may be more efficient to do some processing once, perhaps
402          * to a more efficient intermediate class, but then it would become more
403          * difficult to reload the configuration "on the fly".
404          */
405         StringTokenizer st = new StringTokenizer(configLine, ",");
406 
407         while (st.hasMoreTokens())
408         {
409         	String field = st.nextToken().trim();
410             boolean isDate = false;
411             boolean isLink = false;
412             boolean isResolver = false;
413             
414             String style = null;
415             Matcher fieldStyleMatcher = fieldStylePatter.matcher(field);
416             if (fieldStyleMatcher.matches()){
417                 style = fieldStyleMatcher.group(1);
418             }
419             
420             String browseIndex;
421             try
422             {
423                 browseIndex = getBrowseField(field);
424             }
425             catch (BrowseException e)
426             {
427                 log.error(e);
428                 browseIndex = null;
429             }
430 
431             // Find out if the field should rendered with a particular style
432 
433             if (style != null)
434             {
435                 isDate = style.equals("date");
436                 isLink = style.equals("link");
437                 isResolver = style.equals("resolver") || urn2baseurl.keySet().contains(style);
438                 field = field.replaceAll("\\("+style+"\\)", "");
439             } 
440 
441             // Get the separate schema + element + qualifier
442 
443             String[] eq = field.split("\\.");
444             String schema = eq[0];
445             String element = eq[1];
446             String qualifier = null;
447             if (eq.length > 2 && eq[2].equals("*"))
448             {
449                 qualifier = Item.ANY;
450             }
451             else if (eq.length > 2)
452             {
453                 qualifier = eq[2];
454             }
455 
456             // check for hidden field, even if it's configured..
457             if (MetadataExposure.isHidden(context, schema, element, qualifier))
458                 continue;
459 
460             // FIXME: Still need to fix for metadata language?
461             DCValue[] values = item.getMetadata(schema, element, qualifier, Item.ANY);
462             
463             if (values.length > 0)
464             {
465                 out.print("<tr><td class=\"metadataFieldLabel\">");
466 
467                 String label = null;
468                 try
469                 {
470                     label = I18nUtil.getMessage("metadata."
471                             + (style != null ? style + "." : "") + field,
472                             pageContext.getRequest().getLocale());
473                 }
474                 catch (MissingResourceException e)
475                 {
476                     // if there is not a specific translation for the style we
477                     // use the default one
478                     label = LocaleSupport.getLocalizedMessage(pageContext,
479                             "metadata." + field);
480                 }
481                 
482                 out.print(label);
483                 out.print(":&nbsp;</td><td class=\"metadataFieldValue\">");
484 
485                 for (int j = 0; j < values.length; j++)
486                 {
487                     if (values[j] != null && values[j].value != null)
488                     {
489                         if (j > 0)
490                         {
491                             out.print("<br />");
492                         }
493 
494                         if (isLink)
495                         {
496                             out.print("<a href=\"" + values[j].value + "\">"
497                                     + Utils.addEntities(values[j].value) + "</a>");
498                         }
499                         else if (isDate)
500                         {
501                             DCDate dd = new DCDate(values[j].value);
502 
503                             // Parse the date
504                             out.print(UIUtil.displayDate(dd, false, false, (HttpServletRequest)pageContext.getRequest()));
505                         }
506                         else if (isResolver)
507                         {
508                             String value = values[j].value;
509                             if (value.startsWith("http://")
510                                     || value.startsWith("https://")
511                                     || value.startsWith("ftp://")
512                                     || value.startsWith("ftps://"))
513                             {
514                                 // Already a URL, print as if it was a regular link
515                                 out.print("<a href=\"" + value + "\">"
516                                         + Utils.addEntities(value) + "</a>");
517                             }
518                             else
519                             {
520                                 String foundUrn = null;
521                                 if (!style.equals("resolver"))
522                                 {
523                                     foundUrn = style;
524                                 }
525                                 else
526                                 {
527                                     for (String checkUrn : urn2baseurl.keySet())
528                                     {
529                                         if (value.startsWith(checkUrn))
530                                         {
531                                             foundUrn = checkUrn;
532                                         }
533                                     }
534                                 }
535 
536                                 if (foundUrn != null)
537                                 {
538 
539                                     if (value.startsWith(foundUrn + ":"))
540                                     {
541                                         value = value.substring(foundUrn.length()+1);
542                                     }
543 
544                                     String url = urn2baseurl.get(foundUrn);
545                                     out.print("<a href=\"" + url
546                                             + value + "\">"
547                                             + Utils.addEntities(values[j].value)
548                                             + "</a>");
549                                 }
550                                 else
551                                 {
552                                     out.print(value);
553                                 }
554                             }
555 
556                         }
557                         else if (browseIndex != null)
558                         {
559 	                        String argument, value;
560 	                        if ( values[j].authority != null &&
561 	                                            values[j].confidence >= MetadataAuthorityManager.getManager()
562 	                                                .getMinConfidence( values[j].schema,  values[j].element,  values[j].qualifier))
563 	                        {
564 	                            argument = "authority";
565 	                            value = values[j].authority;
566 	                        }
567 	                        else
568 	                        {
569 	                            argument = "value";
570 	                            value = values[j].value;
571 	                        }
572 	                    	out.print("<a class=\"" + ("authority".equals(argument)?"authority ":"") + browseIndex + "\""
573 	                                                + "href=\"" + request.getContextPath() + "/browse?type=" + browseIndex + "&amp;" + argument + "="
574 	                    				+ URLEncoder.encode(value, "UTF-8") + "\">" + Utils.addEntities(values[j].value)
575 	                    				+ "</a>");
576 	                    }
577                         else
578                         {
579                             out.print(Utils.addEntities(values[j].value));
580                         }
581                     }
582                 }
583 
584                 out.println("</td></tr>");
585             }
586         }
587 
588         listCollections();
589 
590         out.println("</table></center><br/>");
591 
592         listBitstreams();
593 
594         if (ConfigurationManager
595                 .getBooleanProperty("webui.licence_bundle.show"))
596 
597         {
598             out.println("<br/><br/>");
599             showLicence();
600         }
601     }

603     /**
604      * Render full item record
605      */
606     private void renderFull() throws IOException, SQLException
607     {
608         JspWriter out = pageContext.getOut();
609         HttpServletRequest request = (HttpServletRequest)pageContext.getRequest();
610         Context context = UIUtil.obtainContext(request);
611 
612         // Get all the metadata
613         DCValue[] values = item.getMetadata(Item.ANY, Item.ANY, Item.ANY, Item.ANY);
614 
615         out.println("<p align=\"center\">"
616                 + LocaleSupport.getLocalizedMessage(pageContext,
617                         "org.dspace.app.webui.jsptag.ItemTag.full") + "</p>");
618 
619         // Three column table - DC field, value, language
620         out.println("<center><table class=\"itemDisplayTable\">");
621         out.println("<tr><th id=\"s1\" class=\"standard\">"
622                 + LocaleSupport.getLocalizedMessage(pageContext,
623                         "org.dspace.app.webui.jsptag.ItemTag.dcfield")
624                 + "</th><th id=\"s2\" class=\"standard\">"
625                 + LocaleSupport.getLocalizedMessage(pageContext,
626                         "org.dspace.app.webui.jsptag.ItemTag.value")
627                 + "</th><th id=\"s3\" class=\"standard\">"
628                 + LocaleSupport.getLocalizedMessage(pageContext,
629                         "org.dspace.app.webui.jsptag.ItemTag.lang")
630                 + "</th></tr>");
631 
632         for (int i = 0; i < values.length; i++)
633         {
634             if (!MetadataExposure.isHidden(context, values[i].schema, values[i].element, values[i].qualifier))
635             {
636                 out.print("<tr><td headers=\"s1\" class=\"metadataFieldLabel\">");
637                 out.print(values[i].schema);
638                 out.print("." + values[i].element);
639 
640                 if (values[i].qualifier != null)
641                 {
642                     out.print("." + values[i].qualifier);
643                 }
644 
645                 out.print("</td><td headers=\"s2\" class=\"metadataFieldValue\">");
646                 out.print(Utils.addEntities(values[i].value));
647                 out.print("</td><td headers=\"s3\" class=\"metadataFieldValue\">");
648 
649                 if (values[i].language == null)
650                 {
651                     out.print("-");
652                 }
653                 else
654                 {
655                     out.print(values[i].language);
656                 }
657 
658                 out.println("</td></tr>");
659             }
660         }
661 
662         listCollections();
663 
664         out.println("</table></center><br/>");
665 
666         listBitstreams();
667 
668         if (ConfigurationManager
669                 .getBooleanProperty("webui.licence_bundle.show"))
670         {
671             out.println("<br/><br/>");
672             showLicence();
673         }
674     }

676     /**
677      * List links to collections if information is available
678      */
679     private void listCollections() throws IOException
680     {
681         JspWriter out = pageContext.getOut();
682         HttpServletRequest request = (HttpServletRequest) pageContext
683                 .getRequest();
684 
685         if (collections != null)
686         {
687             out.print("<tr><td class=\"metadataFieldLabel\">");
688             if (item.getHandle()==null)  // assume workspace item
689             {
690                 out.print(LocaleSupport.getLocalizedMessage(pageContext,
691                         "org.dspace.app.webui.jsptag.ItemTag.submitted"));
692             }
693             else
694             {
695                 out.print(LocaleSupport.getLocalizedMessage(pageContext,
696                           "org.dspace.app.webui.jsptag.ItemTag.appears"));
697             }
698             out.print("</td><td class=\"metadataFieldValue\">");
699 
700             for (int i = 0; i < collections.length; i++)
701             {
702                 out.print("<a href=\"");
703                 out.print(request.getContextPath());
704                 out.print("/handle/");
705                 out.print(collections[i].getHandle());
706                 out.print("\">");
707                 out.print(collections[i].getMetadata("name"));
708                 out.print("</a><br/>");
709             }
710 
711             out.println("</td></tr>");
712         }
713     }

715     /**
716      * List bitstreams in the item
717      */
718     private void listBitstreams() throws IOException
719     {
720         JspWriter out = pageContext.getOut();
721         HttpServletRequest request = (HttpServletRequest) pageContext
722                 .getRequest();
723

1019     /**
1020      * Return the browse index related to the field. <code>null</code> if the field is not a browse field
1021      * (look for <cod>webui.browse.link.<n></code> in dspace.cfg) 
1022      * 
1023      * @param field
1024      * @return the browse index related to the field. Null otherwise 
1025      * @throws BrowseException 
1026      */
1027     private String getBrowseField(String field) throws BrowseException
1028     {
1029         for (String indexName : linkedMetadata.keySet())
1030         {            
1031             StringTokenizer bw_dcf = new StringTokenizer(linkedMetadata.get(indexName), ".");
1032             
1033             String[] bw_tokens = { "", "", "" };
1034             int i = 0;
1035             while(bw_dcf.hasMoreTokens())
1036             {
1037                 bw_tokens[i] = bw_dcf.nextToken().toLowerCase().trim();
1038                 i++;
1039             }
1040             String bw_schema = bw_tokens[0];
1041             String bw_element = bw_tokens[1];
1042             String bw_qualifier = bw_tokens[2];
1043             
1044             StringTokenizer dcf = new StringTokenizer(field, ".");
1045             
1046             String[] tokens = { "", "", "" };
1047             int j = 0;
1048             while(dcf.hasMoreTokens())
1049             {
1050                 tokens[j] = dcf.nextToken().toLowerCase().trim();
1051                 j++;
1052             }
1053             String schema = tokens[0];
1054             String element = tokens[1];
1055             String qualifier = tokens[2];
1056             if (schema.equals(bw_schema) // schema match
1057                     && element.equals(bw_element) // element match
1058                     && (
1059                               (bw_qualifier != null) && ((qualifier != null && qualifier.equals(bw_qualifier)) // both not null and equals 
1060                                       || bw_qualifier.equals("*")) // browse link with jolly
1061                            || (bw_qualifier == null && qualifier == null)) // both null
1062                         )
1063             {
1064                 return indexName;
1065             }
1066         }
1067         return null;
