140     {
141     	// read rest of config info if enabled

145     		if ( fmtsStr != null )
146     		{
147     			formats = new ArrayList();

164     protected void doDSGet(Context context, HttpServletRequest request,
165             HttpServletResponse response) throws ServletException, IOException,
166             SQLException, AuthorizeException
167 	{
168         includeAll = ConfigurationManager.getBooleanProperty("harvest.includerestricted.rss", true);
169         String path = request.getPathInfo();
170         String feedType = null;
171         String handle = null;
172 
173         // build label map from localized Messages resource bundle
174             Locale locale = request.getLocale();
175         ResourceBundle msgs = ResourceBundle.getBundle("Messages", locale);
176         Map<String, String> labelMap = new HashMap<String, String>();
177         labelMap.put(SyndicationFeed.MSG_UNTITLED, msgs.getString(clazz + ".notitle"));
178         labelMap.put(SyndicationFeed.MSG_LOGO_TITLE, msgs.getString(clazz + ".logo.title"));
179         labelMap.put(SyndicationFeed.MSG_FEED_DESCRIPTION, msgs.getString(clazz + ".general-feed.description"));
180         labelMap.put(SyndicationFeed.MSG_UITYPE, SyndicationFeed.UITYPE_JSPUI);
181         for (String selector : SyndicationFeed.getDescriptionSelectors())
182         {
183             labelMap.put("metadata." + selector, msgs.getString(SyndicationFeed.MSG_METADATA + selector));
184         }
185         
186         if (path != null)
187         {
188             // substring(1) is to remove initial '/'
189             path = path.substring(1);
190             int split = path.indexOf("/");
191             if (split != -1)
192             {
193             	feedType = path.substring(0,split);
194             	handle = path.substring(split+1);
195             }
196         }
197 
198         DSpaceObject dso = null;
199         
200         //as long as this is not a site wide feed, 
201         //attempt to retrieve the Collection or Community object
202         if(handle != null && !handle.equals(SITE_FEED_KEY))
203         { 	
204         	// Determine if handle is a valid reference
205         	dso = HandleManager.resolveToObject(context, handle);
206                 if (dso == null)
207                 {
208                     log.info(LogManager.getHeader(context, "invalid_handle", "path=" + path));
209                     JSPManager.showInvalidIDError(request, response, handle, -1);
210                     return;
211         }
212         }
213         
214         if (! enabled || (dso != null && 
215         	(dso.getType() != Constants.COLLECTION && dso.getType() != Constants.COMMUNITY)) )
216         {
217             log.info(LogManager.getHeader(context, "invalid_id", "path=" + path));
218             JSPManager.showInvalidIDError(request, response, path, -1);
219             return;
220         }
221         
222         // Determine if requested format is supported
223         if( feedType == null || ! formats.contains( feedType ) )
224         {
225             log.info(LogManager.getHeader(context, "invalid_syndformat", "path=" + path));
226             JSPManager.showInvalidIDError(request, response, path, -1);
227             return;
228         }
229         
230         if (dso != null &&  dso.getType() == Constants.COLLECTION)
231         {
232             labelMap.put(SyndicationFeed.MSG_FEED_TITLE,
233                 MessageFormat.format(msgs.getString(clazz + ".feed.title"),
234                                      new Object[]{ msgs.getString(clazz + ".feed-type.collection"),
235                                                    ((Collection)dso).getMetadata("short_description")}));
236         }
237         else if (dso != null &&  dso.getType() == Constants.COMMUNITY)
238         {
239             labelMap.put(SyndicationFeed.MSG_FEED_TITLE,
240                 MessageFormat.format(msgs.getString(clazz + ".feed.title"),
241                                      new Object[]{ msgs.getString(clazz + ".feed-type.community"),
242                                                    ((Community)dso).getMetadata("short_description")}));
243         }
244 
245         // Lookup or generate the feed
246         // Cache key is handle + locale
247         String cacheKey = (handle==null?"site":handle)+"."+locale.toString();
248         SyndicationFeed feed = null;
249         if (feedCache != null)
250         {
251                 CacheFeed cFeed = (CacheFeed)feedCache.get(cacheKey);
252         	if (cFeed != null)  // cache hit, but...
253         	{
254         		// Is the feed current?
255         		boolean cacheFeedCurrent = false;
256         		if (cFeed.timeStamp + (cacheAge * HOUR_MSECS) < System.currentTimeMillis())
257         		{
258         			cacheFeedCurrent = true;
259         		}
260         		// Not current, but have any items changed since feed was created/last checked?
261         		else if ( ! itemsChanged(context, dso, cFeed.timeStamp))
262         		{
263         			// no items have changed, re-stamp feed and use it
264           			cFeed.timeStamp = System.currentTimeMillis();
265           			cacheFeedCurrent = true;
266         		}
267         		if (cacheFeedCurrent)
268         		{
269                                 feed = cFeed.access();
270         		}
271         	}
272         }
273         
274         // either not caching, not found in cache, or feed in cache not current
275         if (feed == null)
276         {
277                 feed = new SyndicationFeed(SyndicationFeed.UITYPE_JSPUI);
278                 feed.populate(request, dso, getItems(context, dso), labelMap);
279         	if (feedCache != null)
280         	{
281                         cache(cacheKey, new CacheFeed(feed));
282         	}
283         }
284         
285         // set the feed to the requested type & return it
286         try
287         {
288                 feed.setType(feedType);
289         	response.setContentType("text/xml; charset=UTF-8");
290                 feed.output(response.getWriter());
291         }
292         catch( FeedException fex )
293         {
294         	throw new IOException(fex.getMessage());
295         }
296     }

298     private boolean itemsChanged(Context context, DSpaceObject dso, long timeStamp)
299             throws SQLException
300     {
301         // construct start and end dates
302         DCDate dcStartDate = new DCDate( new Date(timeStamp) );
303         DCDate dcEndDate = new DCDate( new Date(System.currentTimeMillis()) );
304 
305         // convert dates to ISO 8601, stripping the time
306         String startDate = dcStartDate.toString().substring(0, 10);
307         String endDate = dcEndDate.toString().substring(0, 10);
308         
309         // this invocation should return a non-empty list if even 1 item has changed
310         try {
311             return (Harvest.harvest(context, dso, startDate, endDate,
312         		                0, 1, !includeAll, false, false, includeAll).size() > 0);
313         }
314         catch (ParseException pe)
315         {
316         	// This should never get thrown as we have generated the dates ourselves
317         	return false;
318         }
319     }

321     // returns recently changed items, checking for accessibility
322     private Item[] getItems(Context context, DSpaceObject dso)
323     		throws IOException, SQLException
324     {
325     	try
326     	{
327     		// new method of doing the browse:
328     		String idx = ConfigurationManager.getProperty("recent.submissions.sort-option");
329     		if (idx == null)
330     		{
331     			throw new IOException("There is no configuration supplied for: recent.submissions.sort-option");
332     		}
333     		BrowseIndex bix = BrowseIndex.getItemBrowseIndex();
334     		if (bix == null)
335     		{
336     			throw new IOException("There is no browse index with the name: " + idx);
337     		}
338     		
339     		BrowserScope scope = new BrowserScope(context);
340     		scope.setBrowseIndex(bix);
341                 if (dso != null)
342                     scope.setBrowseContainer(dso);
343 
344             for (SortOption so : SortOption.getSortOptions())
345             {
346                 if (so.getName().equals(idx))
347                     scope.setSortBy(so.getNumber());
348             }
349             scope.setOrder(SortOption.DESCENDING);
350     		scope.setResultsPerPage(itemCount);
351     		
352             // gather & add items to the feed.
353     		BrowseEngine be = new BrowseEngine(context);
354     		BrowseInfo bi = be.browseMini(scope);
355     		Item[] results = bi.getItemResults(context);
356 
357             if (includeAll)
358             {
359                 return results;
360             }
361             else
362                 {
363                     // Check to see if we can include this item
364                 //Group[] authorizedGroups = AuthorizeManager.getAuthorizedGroups(context, results[i], Constants.READ);
365                 //boolean added = false;
366                 List<Item> items = new ArrayList<Item>();
367                 for (Item result : results)
368                     {
369                 checkAccess:
370                     for (Group group : AuthorizeManager.getAuthorizedGroups(context, result, Constants.READ))
371                         {
372                         if ((group.getID() == 0))
373                         {
374                             items.add(result);
375                             break checkAccess;
376                         }
377                     }
378                 }
379                 return items.toArray(new Item[items.size()]);
380                 }
381             }
382         catch (SortException se)
383         {
384             log.error("caught exception: ", se);
385             throw new IOException(se.getMessage());
386         }
387     	catch (BrowseException e)
388     	{
389     		log.error("caught exception: ", e);
390     		throw new IOException(e.getMessage());
391     	}

398 	/**
399      * Add a feed to the cache - reducing the size of the cache by 1 to make room if
400      * necessary. The removed entry has an access count equal to the minumum in the cache.
401      * @param feedKey
402      *            The cache key for the feed
403      * @param newFeed
404      *            The CacheFeed feed to be cached
405      */ 
406     private static void cache(String feedKey, CacheFeed newFeed)
407     {
408 		// remove older feed to make room if cache full
409 		if (feedCache.size() >= cacheSize)
410 		{
411 	    	// cache profiling data
412 	    	int total = 0;
413 	    	String minKey = null;
414 	    	CacheFeed minFeed = null;
415 	    	CacheFeed maxFeed = null;
416 	    	
417 	    	Iterator iter = feedCache.keySet().iterator();
418 	    	while (iter.hasNext())
419 	    	{
420 	    		String key = (String)iter.next();
421 	    		CacheFeed feed = (CacheFeed)feedCache.get(key);
422 	    		if (minKey != null)
423 	    		{
424 	    			if (feed.hits < minFeed.hits)
425 	    			{
426 	    				minKey = key;
427 	    				minFeed = feed;
428 	    			}
429 	    			if (feed.hits >= maxFeed.hits)
430 	    			{
431 	    				maxFeed = feed;
432 	    			}
433 	    		}
434 	    		else
435 	    		{
436 	    			minKey = key;
437 	    			minFeed = maxFeed = feed;
438 	    		}
439 	    		total += feed.hits;
440 	    	}
441 	    	// log a profile of the cache to assist administrator in tuning it
442 	    	int avg = total / feedCache.size();
443 	    	String logMsg = "feedCache() - size: " + feedCache.size() +
444 	    	                " Hits - total: " + total + " avg: " + avg +
445 	    	                " max: " + maxFeed.hits + " min: " + minFeed.hits;
446 	    	log.info(logMsg);
447 	    	// remove minimum hits entry
448 	    	feedCache.remove(minKey);
449 		}
450     	// add feed to cache
451 		feedCache.put(feedKey, newFeed);
452     }

454     /**
455      * Class to instrument accesses & currency of a given feed in cache
456      */  
457     private class CacheFeed
458 	{
459     	// currency timestamp
460     	public long timeStamp = 0L;
461     	// access count
462     	public int hits = 0;
463     	// the feed
464         private SyndicationFeed feed = null;
465     	
466         public CacheFeed(SyndicationFeed feed)
467     	{
468     		this.feed = feed;
469     		timeStamp = System.currentTimeMillis();
470     	}
471     	    	
472         public SyndicationFeed access()
473     	{
474     		++hits;
475     		return feed;
476     	}
