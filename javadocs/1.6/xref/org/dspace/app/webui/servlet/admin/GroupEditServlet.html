78      protected void doDSPost(Context c, HttpServletRequest request,
79              HttpServletResponse response) throws ServletException, IOException,
80              SQLException, AuthorizeException
81      {
82          // Find out if there's a group parameter
83          int groupID = UIUtil.getIntParameter(request, "group_id");
84          Group group = null;
85  
86          if (groupID >= 0)
87          {
88              group = Group.find(c, groupID);
89          }
90  
91          // group is set
92          if (group != null)
93          {
94              // is this user authorized to edit this group?
95              AuthorizeManager.authorizeAction(c, group, Constants.ADD);
96  
97              boolean submit_edit = (request.getParameter("submit_edit") != null);
98              boolean submit_group_update = (request.getParameter("submit_group_update") != null);
99              boolean submit_group_delete = (request.getParameter("submit_group_delete") != null);
100             boolean submit_confirm_delete = (request.getParameter("submit_confirm_delete") != null);
101             boolean submit_cancel_delete = (request.getParameter("submit_cancel_delete") != null);
102 
103 
104             // just chosen a group to edit - get group and pass it to
105             // group-edit.jsp
106             if (submit_edit && !submit_group_update && !submit_group_delete)
107             {
108                 request.setAttribute("group", group);
109                 request.setAttribute("members", group.getMembers());
110                 request.setAttribute("membergroups", group.getMemberGroups());
111 
112                 JSPManager.showJSP(request, response, "/tools/group-edit.jsp");
113             }
114             // update the members of the group
115             else if (submit_group_update)
116             {
117                 // first off, did we change the group name?
118                 String newName = request.getParameter("group_name");
119 
120                 if (!newName.equals(group.getName()))
121                 {
122                     group.setName(newName);
123                     group.update();
124                 }
125 
126                 int[] eperson_ids = UIUtil.getIntParameters(request,
127                         "eperson_id");
128                 int[] group_ids = UIUtil.getIntParameters(request, "group_ids");
129 
130                 // now get members, and add new ones and remove missing ones
131                 EPerson[] members = group.getMembers();
132                 Group[] membergroups = group.getMemberGroups();
133 
134                 if (eperson_ids != null)
135                 {
136                     // some epeople were listed, now make group's epeople match
137                     // given epeople
138                     Set memberSet = new HashSet();
139                     Set epersonIDSet = new HashSet();
140 
141                     // add all members to a set
142                     for (int x = 0; x < members.length; x++)
143                     {
144                         Integer epersonID = new Integer(members[x].getID());
145                         memberSet.add(epersonID);
146                     }
147 
148                     // now all eperson_ids are put in a set
149                     for (int x = 0; x < eperson_ids.length; x++)
150                     {
151                         epersonIDSet.add(new Integer(eperson_ids[x]));
152                     }
153 
154                     // process eperson_ids, adding those to group not already
155                     // members
156                     Iterator i = epersonIDSet.iterator();
157 
158                     while (i.hasNext())
159                     {
160                         Integer currentID = (Integer) i.next();
161 
162                         if (!memberSet.contains(currentID))
163                         {
164                             group.addMember(EPerson.find(c, currentID
165                                     .intValue()));
166                         }
167                     }
168 
169                     // process members, removing any that aren't in eperson_ids
170                     for (int x = 0; x < members.length; x++)
171                     {
172                         EPerson e = members[x];
173 
174                         if (!epersonIDSet.contains(new Integer(e.getID())))
175                         {
176                             group.removeMember(e);
177                         }
178                     }
179                 }
180                 else
181                 {
182                     // no members found (ids == null), remove them all!
183 
184                     for (int y = 0; y < members.length; y++)
185                     {
186                         group.removeMember(members[y]);
187                     }
188                 }
189 
190                 if (group_ids != null)
191                 {
192                     // some groups were listed, now make group's member groups
193                     // match given group IDs
194                     Set memberSet = new HashSet();
195                     Set groupIDSet = new HashSet();
196 
197                     // add all members to a set
198                     for (int x = 0; x < membergroups.length; x++)
199                     {
200                         Integer myID = new Integer(membergroups[x].getID());
201                         memberSet.add(myID);
202                     }
203 
204                     // now all eperson_ids are put in a set
205                     for (int x = 0; x < group_ids.length; x++)
206                     {
207                         groupIDSet.add(new Integer(group_ids[x]));
208                     }
209 
210                     // process group_ids, adding those to group not already
211                     // members
212                     Iterator i = groupIDSet.iterator();
213 
214                     while (i.hasNext())
215                     {
216                         Integer currentID = (Integer) i.next();
217 
218                         if (!memberSet.contains(currentID))
219                         {
220                             group
221                                     .addMember(Group.find(c, currentID
222                                             .intValue()));
223                         }
224                     }
225 
226                     // process members, removing any that aren't in eperson_ids
227                     for (int x = 0; x < membergroups.length; x++)
228                     {
229                         Group g = membergroups[x];
230 
231                         if (!groupIDSet.contains(new Integer(g.getID())))
232                         {
233                             group.removeMember(g);
234                         }
235                     }
236 
237                 }
238                 else
239                 {
240                     // no members found (ids == null), remove them all!
241                     for (int y = 0; y < membergroups.length; y++)
242                     {
243                         group.removeMember(membergroups[y]);
244                     }
245                 }
246 
247                 group.update();
248 
249                 request.setAttribute("group", group);
250                 request.setAttribute("members", group.getMembers());
251                 request.setAttribute("membergroups", group.getMemberGroups());
252 
253                 JSPManager.showJSP(request, response, "/tools/group-edit.jsp");
254                 c.complete();
255             }
256             else if (submit_group_delete)
257             {
258                 // direct to a confirmation step
259                 request.setAttribute("group", group);
260                 JSPManager.showJSP(request, response, "/dspace-admin/group-confirm-delete.jsp");
261             }
262             else if (submit_confirm_delete)
263             {
264                 // phony authorize, only admins can do this
265                 AuthorizeManager.authorizeAction(c, group, Constants.WRITE);
266 
267                 // delete group, return to group-list.jsp
268                 group.delete();
269 
270                 showMainPage(c, request, response);
271             }
272             else if (submit_cancel_delete)
273             {
274                 // show group list
275                 showMainPage(c, request, response);
276             }
277             else
278             {
279                 // unknown action, show edit page
280                 request.setAttribute("group", group);
281                 request.setAttribute("members", group.getMembers());
282                 request.setAttribute("membergroups", group.getMemberGroups());
283 
284                 JSPManager.showJSP(request, response, "/tools/group-edit.jsp");
285             }
286         }
287         else
288         // no group set
289         {
290             // want to add a group - create a blank one, and pass to
291             // group_edit.jsp
292             String button = UIUtil.getSubmitButton(request, "submit");
293 
294             if (button.equals("submit_add"))
295             {
296                 group = Group.create(c);
297 
298                 group.setName("new group" + group.getID());
299                 group.update();
300 
301                 request.setAttribute("group", group);
302                 request.setAttribute("members", group.getMembers());
303                 request.setAttribute("membergroups", group.getMemberGroups());
304 
305                 JSPManager.showJSP(request, response, "/tools/group-edit.jsp");
306                 c.complete();
307             }
308             else
309             {
310                 // show the main page (select groups)
311                 showMainPage(c, request, response);
312             }
313         }
314     }

316     private void showMainPage(Context c, HttpServletRequest request,
317             HttpServletResponse response) throws ServletException, IOException,
318             SQLException, AuthorizeException
319     {
320         Group[] groups = Group.findAll(c, Group.NAME);
321 
322         // if( groups == null ) { System.out.println("groups are null"); }
323         // else System.out.println("# of groups: " + groups.length);
324         request.setAttribute("groups", groups);
325 
326         JSPManager.showJSP(request, response, "/tools/group-list.jsp");
327         c.complete();
328     }
