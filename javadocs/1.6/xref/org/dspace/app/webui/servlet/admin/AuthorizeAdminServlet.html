525             int group_id = UIUtil.getIntParameter(request, "group_id");

539 
540             if (collection_id != -1)
541             {

543 
544                 // modify the policy
545                 policy.setAction(action_id);
546                 policy.setGroup(group);
547                 policy.update();
548 
549                 // if it is a read, policy, modify the logo policy to match
550                 if (action_id == Constants.READ)
551                 {
552                     // first get a list of READ policies from collection
553                     List rps = AuthorizeManager.getPoliciesActionFilter(c,
554                             collection, Constants.READ);
555 
556                     // remove all bitstream policies, then add READs
557                     Bitstream bs = collection.getLogo();
558 
559                     if (bs != null)
560                     {

563                     }
564                 }
565 
566                 // set up page attributes
567                 request.setAttribute("collection", collection);
568                 request.setAttribute("policies", AuthorizeManager.getPolicies(
569                         c, collection));
570                 display_page = "/dspace-admin/authorize-collection-edit.jsp";
571             }
572             else if (community_id != -1)
573             {
574                 community = Community.find(c, community_id);
575 
576                 // modify the policy
577                 policy.setAction(action_id);
578                 policy.setGroup(group);

588                     // remove all bitstream policies, then add READs

590 
591                     if (bs != null)
592                     {
593                         AuthorizeManager.removeAllPolicies(c, bs);
594                         AuthorizeManager.addPolicies(c, rps, bs);
595                     }
596                 }
597 
598                 // set up page attributes
599                 request.setAttribute("community", community);
600                 request.setAttribute("policies", AuthorizeManager.getPolicies(
601                         c, community));
602                 display_page = "/dspace-admin/authorize-community-edit.jsp";
603             }
604             else if (item_id != -1)
605             {
606                 item = Item.find(c, item_id);
607 
608                 // modify the policy
609                 policy.setAction(action_id);
610                 policy.setGroup(group);
611                 policy.update();
612 
613                 // show edit form!
614                 prepItemEditForm(c, request, item);
615 
616                 display_page = "/dspace-admin/authorize-item-edit.jsp";
