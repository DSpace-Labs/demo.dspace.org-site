64  /**
65   * Servlet for handling user registration and forgotten passwords.
66   * <P>
67   * This servlet handles both forgotten passwords and initial registration of
68   * users. Which it handles depends on the initialisation parameter "register" -
69   * if it's "true", it is treated as an initial registration and the user is
70   * asked to input their personal information.
71   * <P>
72   * The sequence of events is this: The user clicks on "register" or "I forgot my
73   * password." This servlet then displays the relevant "enter your e-mail" form.
74   * An e-mail address is POSTed back, and if this is valid, a token is created
75   * and e-mailed, otherwise an error is displayed, with another "enter your
76   * e-mail" form.
77   * <P>
78   * When the user clicks on the token URL mailed to them, this servlet receives a
79   * GET with the token as the parameter "KEY". If this is a valid token, the
80   * servlet then displays the "edit profile" or "edit password" screen as
81   * appropriate.
82   */

109     protected void doDSGet(Context context, HttpServletRequest request,
110             HttpServletResponse response) throws ServletException, IOException,
111             SQLException, AuthorizeException
112     {
113         /*
114          * Respond to GETs. A simple GET with no parameters will display the
115          * relevant "type in your e-mail" form. A GET with a "token" parameter
116          * will go to the "enter personal info" or "enter new password" page as
117          * appropriate.
118          */
119         boolean updated = false;
120 
121         // Get the token
122         String token = request.getParameter("token");
123 
124         if (token == null)
125         {
126             // Simple "enter your e-mail" page
127             if (registering)
128             {
129                 // Registering a new user
130                 if (ldap_enabled) JSPManager.showJSP(request, response, "/register/new-ldap-user.jsp");
131                 JSPManager.showJSP(request, response, "/register/new-user.jsp");
132             }
133             else
134             {
135                 // User forgot their password
136                 JSPManager.showJSP(request, response,
137                         "/register/forgot-password.jsp");
138             }
139         }
140         else
141         {
142             // We have a token. Find out who the it's for
143             String email = AccountManager.getEmail(context, token);
144 
145             EPerson eperson = null;
146 
147             if (email != null)
148             {
149                 eperson = EPerson.findByEmail(context, email);
150             }
151 
152             // Both forms need an EPerson object (if any)
153             request.setAttribute("eperson", eperson);
154 
155             // And the token
156             request.setAttribute("token", token);
157 
158             if (registering && (email != null))
159             {
160                 // Indicate if user can set password
161                 boolean setPassword =
162                     AuthenticationManager.allowSetPassword(context, request, email);
163                 request.setAttribute("set.password", new Boolean(setPassword));
164 
165                 // Forward to "personal info page"
166                 JSPManager.showJSP(request, response,
167                         "/register/registration-form.jsp");
168             }
169             else if (!registering && (eperson != null))
170             {
171                 // Token relates to user who's forgotten password
172                 JSPManager.showJSP(request, response,
173                         "/register/new-password.jsp");
174             }
175             else
176             {
177                 // Duff token!
178                 JSPManager.showJSP(request, response,
179                         "/register/invalid-token.jsp");
180 
181                 return;
182             }
183         }
184     }

186     protected void doDSPost(Context context, HttpServletRequest request,
187             HttpServletResponse response) throws ServletException, IOException,
188             SQLException, AuthorizeException
189     {
190         /*
191          * POSTs are the result of entering an e-mail in the "forgot my
192          * password" or "new user" forms, or the "enter profile information" or
193          * "enter new password" forms.
194          */
195 
196         // First get the step
197         int step = UIUtil.getIntParameter(request, "step");
198 
199         switch (step)
200         {
201         case ENTER_EMAIL_PAGE:
202             processEnterEmail(context, request, response);
203 
204             break;
205 
206         case PERSONAL_INFO_PAGE:
207             processPersonalInfo(context, request, response);
208 
209             break;
210 
211         case NEW_PASSWORD_PAGE:
212             processNewPassword(context, request, response);
213 
214             break;
215 
216         default:
217             log.warn(LogManager.getHeader(context, "integrity_error", UIUtil
218                     .getRequestLogInfo(request)));
219             JSPManager.showIntegrityError(request, response);
220         }
221     }

223     /**
224      * Process information from the "enter e-mail" page. If the e-mail
225      * corresponds to a valid user of the system, a token is generated and sent
226      * to that user.
227      * 
228      * @param context
229      *            current DSpace context
230      * @param request
231      *            current servlet request object
232      * @param response
233      *            current servlet response object
234      */
235     private void processEnterEmail(Context context, HttpServletRequest request,
236             HttpServletResponse response) throws ServletException, IOException,
237             SQLException, AuthorizeException
238     {
239         String email = request.getParameter("email");
240         if (email == null || email.length() > 64)
241         {
242         	// Malformed request or entered value is too long.
243         	email = "";
244         }
245         else
246         {
247         	email = email.toLowerCase().trim();
248         }
249         
250         String netid = request.getParameter("netid");
251         String password = request.getParameter("password");
252         EPerson eperson = EPerson.findByEmail(context, email);
253         EPerson eperson2 = null;
254         if (netid!=null) eperson2 = EPerson.findByNetid(context, netid.toLowerCase());
255 
256         try
257         {
258             if (registering)
259             {
260                 // If an already-active user is trying to register, inform them so
261                 if ((eperson != null && eperson.canLogIn()) || (eperson2 != null && eperson2.canLogIn()))
262                 {
263                     log.info(LogManager.getHeader(context,
264                             "already_registered", "email=" + email));
265 
266                     JSPManager.showJSP(request, response,
267                             "/register/already-registered.jsp");
268                 }
269                 else
270                 {
271                     // Find out from site authenticator whether this email can
272                     // self-register
273                     boolean canRegister =
274                         AuthenticationManager.canSelfRegister(context, request, email);
275 
276                     if (canRegister)
277                     {
278                         //-- registering by email
279                         if ((!ldap_enabled)||(netid==null)||(netid.trim().equals("")))
280                         {
281                             // OK to register.  Send token.
282                             log.info(LogManager.getHeader(context,
283                                 "sendtoken_register", "email=" + email));
284 
285                             try
286                             {
287                                 AccountManager.sendRegistrationInfo(context, email);
288                             }
289                             catch (javax.mail.SendFailedException e)
290                             {
291                             	if (e.getNextException() instanceof SMTPAddressFailedException)
292                             	{
293                                     // If we reach here, the email is email is invalid for the SMTP server (i.e. fbotelho).
294                                     log.info(LogManager.getHeader(context,
295                                         "invalid_email",
296                                         "email=" + email));
297                                     request.setAttribute("retry", new Boolean(true));
298                                     JSPManager.showJSP(request, response, "/register/new-user.jsp");
299                                     return;
300                             	}
301                             	else
302                             	{
303                             		throw e;
304                             	}
305                             }
306                             JSPManager.showJSP(request, response,
307                                 "/register/registration-sent.jsp");
308 
309                             // Context needs completing to write registration data
310                             context.complete();
311                         }
312                         //-- registering by netid
313                         else 
314                         {
315                             //--------- START LDAP AUTH SECTION -------------
316                             if (password!=null && !password.equals("")) 
317                             {
318                                 String ldap_provider_url = ConfigurationManager.getProperty("ldap.provider_url");
319                                 String ldap_id_field = ConfigurationManager.getProperty("ldap.id_field");
320                                 String ldap_search_context = ConfigurationManager.getProperty("ldap.search_context");
321                            
322                                 // Set up environment for creating initial context
323                                 Hashtable env = new Hashtable(11);
324                                 env.put(javax.naming.Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
325                                 env.put(javax.naming.Context.PROVIDER_URL, ldap_provider_url);
326                         
327                                 // Authenticate 
328                                 env.put(javax.naming.Context.SECURITY_AUTHENTICATION, "simple");
329                                 env.put(javax.naming.Context.SECURITY_PRINCIPAL, ldap_id_field+"="+netid+","+ldap_search_context);
330                                 env.put(javax.naming.Context.SECURITY_CREDENTIALS, password);
331                         
332                                 try {
333                                    // Create initial context
334                                    DirContext ctx = new InitialDirContext(env);
335              
336                                    // Close the context when we're done
337                                    ctx.close();
338                                 } 
339                                 catch (NamingException e) 
340                                 {
341                                     // If we reach here, supplied email/password was duff.
342                                     log.info(LogManager.getHeader(context,
343                                         "failed_login",
344                                         "netid=" + netid + e));
345                                     JSPManager.showJSP(request, response, "/login/ldap-incorrect.jsp");
346                                     return;
347                                 }
348                             }
349                             //--------- END LDAP AUTH SECTION -------------
350                             // Forward to "personal info page"
351                             JSPManager.showJSP(request, response, "/register/registration-form.jsp");
352                         }
353                     }
354                     else
355                     {
356                         JSPManager.showJSP(request, response,
357                             "/register/cannot-register.jsp");
358                     }
359                 }
360             }
361             else
362             {
363                 if (eperson == null)
364                 {
365                     // Invalid email address
366                     log.info(LogManager.getHeader(context, "unknown_email",
367                             "email=" + email));
368 
369                     request.setAttribute("retry", new Boolean(true));
370 
371                     JSPManager.showJSP(request, response,
372                             "/register/forgot-password.jsp");
373                 }
374                 else if (!eperson.canLogIn())
375                 {
376                     // Can't give new password to inactive user
377                     log.info(LogManager.getHeader(context,
378                             "unregistered_forgot_password", "email=" + email));
379 
380                     JSPManager.showJSP(request, response,
381                             "/register/inactive-account.jsp");
382                 }
383                 else if (eperson.getRequireCertificate() && !registering)
384                 {
385                     // User that requires certificate can't get password
386                     log.info(LogManager.getHeader(context,
387                             "certificate_user_forgot_password", "email="
388                                     + email));
389 
390                     JSPManager.showJSP(request, response,
391                             "/error/require-certificate.jsp");
392                 }
393                 else
394                 {
395                     // OK to send forgot pw token.
396                     log.info(LogManager.getHeader(context,
397                             "sendtoken_forgotpw", "email=" + email));
398 
399                     AccountManager.sendForgotPasswordInfo(context, email);
400                     JSPManager.showJSP(request, response,
401                             "/register/password-token-sent.jsp");
402 
403                     // Context needs completing to write registration data
404                     context.complete();
405                 }
406             }
407         }
408         catch (AddressException ae)
409         {
410             // Malformed e-mail address
411             log.info(LogManager.getHeader(context, "bad_email", "email="
412                     + email));
413 
414             request.setAttribute("retry", new Boolean(true));
415 
416             if (registering)
417             {
418                 if (ldap_enabled) JSPManager.showJSP(request, response, "/register/new-ldap-user.jsp");
419                 else JSPManager.showJSP(request, response, "/register/new-user.jsp");
420             }
421             else
422             {
423                 JSPManager.showJSP(request, response,
424                         "/register/forgot-password.jsp");
425             }
426         }
427         catch (MessagingException me)
428         {
429             // Some other mailing error
430             log.info(LogManager.getHeader(context, "error_emailing", "email="
431                     + email), me);
432 
433             JSPManager.showInternalError(request, response);
434         }
435     }

437     /**
438      * Process information from "Personal information page"
439      * 
440      * @param context
441      *            current DSpace context
442      * @param request
443      *            current servlet request object
444      * @param response
445      *            current servlet response object
446      */
447     private void processPersonalInfo(Context context,
448             HttpServletRequest request, HttpServletResponse response)
449             throws ServletException, IOException, SQLException,
450             AuthorizeException
451     {
452         // Get the token
453         String token = request.getParameter("token");
454 
455         // Get the email address
456         String email = AccountManager.getEmail(context, token);
457         String netid = request.getParameter("netid");
458         if ((netid!=null)&&(email==null)) email = request.getParameter("email");
459         
460         // If the token isn't valid, show an error
461         if (email == null && netid==null)
462         {
463             log.info(LogManager.getHeader(context, "invalid_token", "token="
464                 + token));
465 
466             // Invalid token
467             JSPManager
468                 .showJSP(request, response, "/register/invalid-token.jsp");
469 
470             return;
471         }
472 
473         // If the token is valid, we create an eperson record if need be
474         EPerson eperson = null;
475         if (email!=null) eperson = EPerson.findByEmail(context, email);
476         EPerson eperson2 = null;
477         if (netid!=null) eperson2 = EPerson.findByNetid(context, netid.toLowerCase());
478         if (eperson2 !=null) eperson = eperson2;
479         
480         if (eperson == null)
481         {
482             // Need to create new eperson
483             // FIXME: TEMPORARILY need to turn off authentication, as usually
484             // only site admins can create e-people
485             context.setIgnoreAuthorization(true);
486             eperson = EPerson.create(context);
487             eperson.setEmail(email);
488             if (netid!=null) eperson.setNetid(netid.toLowerCase());
489             eperson.update();
490             context.setIgnoreAuthorization(false);
491         }
492 
493         // Now set the current user of the context
494         // to the user associated with the token, so they can update their
495         // info
496         context.setCurrentUser(eperson);
497 
498         // Set the user profile info
499         boolean infoOK = EditProfileServlet.updateUserProfile(eperson, request);
500 
501         eperson.setCanLogIn(true);
502         eperson.setSelfRegistered(true);
503 
504         // Give site auth a chance to set/override appropriate fields
505         AuthenticationManager.initEPerson(context, request, eperson);
506 
507         // If the user set a password, make sure it's OK
508         boolean passwordOK = true;
509         if (eperson.getRequireCertificate() == false && netid==null &&
510             AuthenticationManager.allowSetPassword(context, request,
511                 eperson.getEmail()))
512         {
513             passwordOK = EditProfileServlet.confirmAndSetPassword(eperson,
514                     request);
515         }
516 
517         if (infoOK && passwordOK)
518         {
519             // All registered OK.
520             log.info(LogManager.getHeader(context, "usedtoken_register",
521                     "email=" + eperson.getEmail()));
522 
523             // delete the token
524             if (token!=null) AccountManager.deleteToken(context, token);
525             
526             // Update user record
527             eperson.update();
528 
529             request.setAttribute("eperson", eperson);
530 
531             JSPManager.showJSP(request, response, "/register/registered.jsp");
532             context.complete();
533         }
534         else
535         {
536             request.setAttribute("token", token);
537             request.setAttribute("eperson", eperson);
538             request.setAttribute("netid", netid);
539             request.setAttribute("missing.fields", new Boolean(!infoOK));
540             request.setAttribute("password.problem", new Boolean(!passwordOK));
541 
542             // Indicate if user can set password
543             boolean setPassword = AuthenticationManager.allowSetPassword(
544                     context, request, email);
545             request.setAttribute("set.password", new Boolean(setPassword));
546 
547             JSPManager.showJSP(request, response,
548                     "/register/registration-form.jsp");
549 
550             // Changes to/creation of e-person in DB cancelled
551             context.abort();
552         }
553     }

555     /**
556      * Process information from "enter new password"
557      * 
558      * @param context
559      *            current DSpace context
560      * @param request
561      *            current servlet request object
562      * @param response
563      *            current servlet response object
564      */
565     private void processNewPassword(Context context,
566             HttpServletRequest request, HttpServletResponse response)
567             throws ServletException, IOException, SQLException,
568             AuthorizeException
569     {
570         // Get the token
571         String token = request.getParameter("token");
572 
573         // Get the eperson associated with the password change
574         EPerson eperson = AccountManager.getEPerson(context, token);
575 
576         // If the token isn't valid, show an error
577         if (eperson == null)
578         {
579             log.info(LogManager.getHeader(context, "invalid_token", "token="
580                     + token));
581 
582             // Invalid token
583             JSPManager
584                     .showJSP(request, response, "/register/invalid-token.jsp");
585 
586             return;
587         }
588 
589         // If the token is valid, we set the current user of the context
590         // to the user associated with the token, so they can update their
591         // info
592         context.setCurrentUser(eperson);
593 
594         // Confirm and set the password
595         boolean passwordOK = EditProfileServlet.confirmAndSetPassword(eperson,
596                 request);
597 
598         if (passwordOK)
599         {
600             log.info(LogManager.getHeader(context, "usedtoken_forgotpw",
601                     "email=" + eperson.getEmail()));
602 
603             eperson.update();
604             AccountManager.deleteToken(context, token);
605 
606             JSPManager.showJSP(request, response,
607                     "/register/password-changed.jsp");
608             context.complete();
609         }
610         else
611         {
612             request.setAttribute("password.problem", new Boolean(true));
613             request.setAttribute("token", token);
614             request.setAttribute("eperson", eperson);
615 
616             JSPManager.showJSP(request, response, "/register/new-password.jsp");
617         }
618     }
