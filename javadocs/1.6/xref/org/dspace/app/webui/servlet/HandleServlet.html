83  /**
84   * Servlet for handling requests within a community or collection. The Handle is
85   * extracted from the URL, e.g: <code>/community/1721.1/1234</code>. If there
86   * is anything after the Handle, the request is forwarded to the appropriate
87   * servlet. For example:
88   * <P>
89   * <code>/community/1721.1/1234/simple-search</code>
90   * <P>
91   * would be forwarded to <code>/simple-search</code>. If there is nothing
92   * after the Handle, the community or collection home page is shown.
93   * 
94   * @author Robert Tansley
95   * @version $Revision: 4430 $
96   */
97  public class HandleServlet extends DSpaceServlet

112     protected void doDSGet(Context context, HttpServletRequest request,
113             HttpServletResponse response) throws ServletException, IOException,
114             SQLException, AuthorizeException
115     {
116         String handle = null;
117         String extraPathInfo = null;
118         DSpaceObject dso = null;
119 
120         // Original path info, of the form "1721.x/1234"
121         // or "1721.x/1234/extra/stuff"
122         String path = request.getPathInfo();
123 
124         if (path != null)
125         {
126             // substring(1) is to remove initial '/'
127             path = path.substring(1);
128 
129             try
130             {
131                 // Extract the Handle
132                 int firstSlash = path.indexOf('/');
133                 int secondSlash = path.indexOf('/', firstSlash + 1);
134 
135                 if (secondSlash != -1)
136                 {
137                     // We have extra path info
138                     handle = path.substring(0, secondSlash);
139                     extraPathInfo = path.substring(secondSlash);
140                 }
141                 else
142                 {
143                     // The path is just the Handle
144                     handle = path;
145                 }
146             }
147             catch (NumberFormatException nfe)
148             {
149                 // Leave handle as null
150             }
151         }
152 
153         // Find out what the handle relates to
154         if (handle != null)
155         {
156             dso = HandleManager.resolveToObject(context, handle);
157         }
158 
159         if (dso == null)
160         {
161             log.info(LogManager
162                     .getHeader(context, "invalid_id", "path=" + path));
163             JSPManager.showInvalidIDError(request, response, StringEscapeUtils.escapeHtml(path), -1);
164 
165             return;
166         }
167 
168         // OK, we have a valid Handle. What is it?
169         if (dso.getType() == Constants.ITEM)
170         {
171             Item item = (Item) dso;
172 
173             new DSpace().getEventService().fireEvent(
174             		new UsageEvent(
175             				UsageEvent.Action.VIEW,
176             				request, 
177             				context, 
178             				dso));
179             
180             // Only use last-modified if this is an anonymous access
181             // - caching content that may be generated under authorisation
182             //   is a security problem
183             if (context.getCurrentUser() == null)
184             {
185                 response.setDateHeader("Last-Modified", item
186                         .getLastModified().getTime());
187 
188                 // Check for if-modified-since header
189                 long modSince = request.getDateHeader("If-Modified-Since");
190 
191                 if (modSince != -1 && item.getLastModified().getTime() < modSince)
192                 {
193                     // Item has not been modified since requested date,
194                     // hence bitstream has not; return 304
195                     response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
196                 }
197                 else
198                 {
199                     // Display the item page
200                     displayItem(context, request, response, item, handle);
201                 }
202             }
203             else
204             {
205                 // Display the item page
206                 displayItem(context, request, response, item, handle);
207             }
208         }
209         else if (dso.getType() == Constants.COLLECTION)
210         {
211             Collection c = (Collection) dso;
212 
213             new DSpace().getEventService().fireEvent(
214             		new UsageEvent(
215             				UsageEvent.Action.VIEW,
216             				request, 
217             				context, 
218             				dso));
219             
220             //UsageEvent ue = new UsageEvent();
221             //ue.fire(request, context, AbstractUsageEvent.VIEW,
222             //        Constants.COLLECTION, dso.getID());
223 
224             // Store collection location in request
225             request.setAttribute("dspace.collection", c);
226 
227             /*
228              * Find the "parent" community the collection, mainly for
229              * "breadcrumbs" FIXME: At the moment, just grab the first community
230              * the collection is in. This should probably be more context
231              * sensitive when we have multiple inclusion.
232              */
233             Community[] parents = c.getCommunities();
234             request.setAttribute("dspace.community", parents[0]);
235 
236             /*
237              * Find all the "parent" communities for the collection for
238              * "breadcrumbs"
239              */
240             request.setAttribute("dspace.communities", getParents(parents[0],
241                     true));
242 
243             // home page, or forward to another page?
244             if ((extraPathInfo == null) || (extraPathInfo.equals("/")))
245             {
246                 collectionHome(context, request, response, parents[0], c);
247             }
248             else
249             {
250                 // Forward to another servlet
251                 request.getRequestDispatcher(extraPathInfo).forward(request,
252                         response);
253             }
254         }
255         else if (dso.getType() == Constants.COMMUNITY)
256         {
257             Community c = (Community) dso;
258 
259             new DSpace().getEventService().fireEvent(
260             		new UsageEvent(
261             				UsageEvent.Action.VIEW,
262             				request, 
263             				context, 
264             				dso));
265             
266             //UsageEvent ue = new UsageEvent();
267             //ue.fire(request, context, AbstractUsageEvent.VIEW,
268             //        Constants.COMMUNITY, dso.getID());
269 
270             // Store collection location in request
271             request.setAttribute("dspace.community", c);
272 
273             /*
274              * Find all the "parent" communities for the community
275              */
276             request.setAttribute("dspace.communities", getParents(c, false));
277 
278             // home page, or forward to another page?
279             if ((extraPathInfo == null) || (extraPathInfo.equals("/")))
280             {
281                 communityHome(context, request, response, c);
282             }
283             else
284             {
285                 // Forward to another servlet
286                 request.getRequestDispatcher(extraPathInfo).forward(request,
287                         response);
288             }
289         }
290         else
291         {
292             // Shouldn't happen. Log and treat as invalid ID
293             log.info(LogManager.getHeader(context,
294                     "Handle not an item, collection or community", "handle="
295                             + handle));
296             JSPManager.showInvalidIDError(request, response, StringEscapeUtils.escapeHtml(path), -1);
297 
298             return;
299         }
300     }

302     /**
303      * Show an item page
304      * 
305      * @param context
306      *            Context object
307      * @param request
308      *            the HTTP request
309      * @param response
310      *            the HTTP response
311      * @param item
312      *            the item
313      * @param handle
314      *            the item's handle
315      */
316     private void displayItem(Context context, HttpServletRequest request,
317             HttpServletResponse response, Item item, String handle)
318             throws ServletException, IOException, SQLException,
319             AuthorizeException
320     {
321         // Tombstone?
322         if (item.isWithdrawn())
323         {
324             JSPManager.showJSP(request, response, "/tombstone.jsp");
325 
326             return;
327         }
328 
329         // Ensure the user has authorisation
330         AuthorizeManager.authorizeAction(context, item, Constants.READ);
331 
332         log
333                 .info(LogManager.getHeader(context, "view_item", "handle="
334                         + handle));
335 
336         // show edit link
337         if (item.canEdit())
338         {
339             // set a variable to create an edit button
340             request.setAttribute("admin_button", new Boolean(true));
341         }
342 
343         // Get the collections
344         Collection[] collections = item.getCollections();
345 
346         // For the breadcrumbs, get the first collection and the first community
347         // that is in. FIXME: Not multiple-inclusion friendly--should be
348         // smarter, context-sensitive
349         request.setAttribute("dspace.collection", item.getOwningCollection());
350 
351         Community[] comms = item.getOwningCollection().getCommunities();
352         request.setAttribute("dspace.community", comms[0]);
353 
354         /*
355          * Find all the "parent" communities for the collection
356          */
357         request.setAttribute("dspace.communities", getParents(comms[0], true));
358 
359         // Full or simple display?
360         boolean displayAll = false;
361         String modeParam = request.getParameter("mode");
362 
363         if ((modeParam != null) && modeParam.equalsIgnoreCase("full"))
364         {
365             displayAll = true;
366         }
367 
368         String headMetadata = "";
369 
370         // Produce <meta> elements for header from crosswalk
371         try
372         {
373             List l = xHTMLHeadCrosswalk.disseminateList(item);
374             StringWriter sw = new StringWriter();
375 
376             XMLOutputter xmlo = new XMLOutputter();
377             xmlo.output(new Text("\n"), sw);
378             for (int i = 0; i < l.size(); i++)
379             {
380                 Element e = (Element) l.get(i);
381                 // FIXME: we unset the Namespace so it's not printed.
382                 // This is fairly yucky, but means the same crosswalk should
383                 // work for Manakin as well as the JSP-based UI.
384                 e.setNamespace(null);
385                 xmlo.output(e, sw);
386                 xmlo.output(new Text("\n"), sw);
387             }
388             headMetadata = sw.toString();
389         }
390         catch (CrosswalkException ce)
391         {
392             throw new ServletException(ce);
393         }
394 
395         // Enable suggest link or not
396         boolean suggestEnable = false;
397         if (!ConfigurationManager.getBooleanProperty("webui.suggest.enable"))
398         {
399             // do nothing, the suggestLink is allready set to false 
400         }
401         else
402         {
403             // it is in general enabled
404             suggestEnable= true;
405             
406             // check for the enable only for logged in users option
407             if(!ConfigurationManager.getBooleanProperty("webui.suggest.loggedinusers.only"))
408             {
409                 // do nothing, the suggestLink stays as it is
410             }
411             else
412             {
413                 // check whether there is a logged in user
414                 suggestEnable = (context.getCurrentUser() == null ? false : true);
415             }
416         }
417         
418         // Set attributes and display
419         request.setAttribute("suggest.enable", new Boolean(suggestEnable));
420         request.setAttribute("display.all", new Boolean(displayAll));
421         request.setAttribute("item", item);
422         request.setAttribute("collections", collections);
423         request.setAttribute("dspace.layout.head", headMetadata);
424         JSPManager.showJSP(request, response, "/display-item.jsp");
425     }

427     /**
428      * Show a community home page, or deal with button press on home page
429      * 
430      * @param context
431      *            Context object
432      * @param request
433      *            the HTTP request
434      * @param response
435      *            the HTTP response
436      * @param community
437      *            the community
438      */
439     private void communityHome(Context context, HttpServletRequest request,
440             HttpServletResponse response, Community community)
441             throws ServletException, IOException, SQLException
442     {
443         // Handle click on a browse or search button
444         if (!handleButton(request, response, community.getHandle()))
445         {
446             // No button pressed, display community home page
447             log.info(LogManager.getHeader(context, "view_community",
448                     "community_id=" + community.getID()));
449 
450             // Get the collections within the community
451             Collection[] collections = community.getCollections();
452 
453             // get any subcommunities of the community
454             Community[] subcommunities = community.getSubcommunities();
455 
456             // perform any necessary pre-processing
457             preProcessCommunityHome(context, request, response, community);
458 
459             // is the user a COMMUNITY_EDITOR?
460             if (community.canEditBoolean())
461             {
462                 // set a variable to create an edit button
463                 request.setAttribute("editor_button", new Boolean(true));
464             }
465 
466             // can they add to this community?
467             if (AuthorizeManager.authorizeActionBoolean(context, community,
468                     Constants.ADD))
469             {
470                 // set a variable to create an edit button
471                 request.setAttribute("add_button", new Boolean(true));
472             }
473 
474             // can they remove from this community?
475             if (AuthorizeManager.authorizeActionBoolean(context, community,
476                     Constants.REMOVE))
477             {
478                 // set a variable to create an edit button
479                 request.setAttribute("remove_button", new Boolean(true));
480             }
481 
482             // Forward to community home page
483             request.setAttribute("community", community);
484             request.setAttribute("collections", collections);
485             request.setAttribute("subcommunities", subcommunities);
486             JSPManager.showJSP(request, response, "/community-home.jsp");
487         }
488     }

510     /**
511      * Show a collection home page, or deal with button press on home page
512      * 
513      * @param context
514      *            Context object
515      * @param request
516      *            the HTTP request
517      * @param response
518      *            the HTTP response
519      * @param community
520      *            the community
521      * @param collection
522      *            the collection
523      */
524     private void collectionHome(Context context, HttpServletRequest request,
525             HttpServletResponse response, Community community,
526             Collection collection) throws ServletException, IOException,
527             SQLException, AuthorizeException
528     {
529         // Handle click on a browse or search button
530         if (!handleButton(request, response, collection.getHandle()))
531         {
532             // Will need to know whether to commit to DB
533             boolean updated = false;
534 
535             // No search or browse button pressed, check for
536             if (request.getParameter("submit_subscribe") != null)
537             {
538                 // Subscribe button pressed.
539                 // Only registered can subscribe, so redirect unless logged in.
540                 if (context.getCurrentUser() == null &&
541                     !Authenticate
542                             .startAuthentication(context, request, response))
543 
544                     return;
545                 else
546                 {
547                     Subscribe.subscribe(context, context.getCurrentUser(),
548                             collection);
549                     updated = true;
550                 }
551             }
552             else if (request.getParameter("submit_unsubscribe") != null)
553             {
554                 Subscribe.unsubscribe(context, context.getCurrentUser(),
555                         collection);
556                 updated = true;
557             }
558 
559             // display collection home page
560             log.info(LogManager.getHeader(context, "view_collection",
561                     "collection_id=" + collection.getID()));
562 
563             // perform any necessary pre-processing
564             preProcessCollectionHome(context, request, response, collection);
565             
566             // Is the user logged in/subscribed?
567             EPerson e = context.getCurrentUser();
568             boolean subscribed = false;
569 
570             if (e != null)
571             {
572                 subscribed = Subscribe.isSubscribed(context, e, collection);
573 
574                 // is the user a COLLECTION_EDITOR?
575                 if (collection.canEditBoolean(true))
576                 {
577                     // set a variable to create an edit button
578                     request.setAttribute("editor_button", new Boolean(true));
579                 }
580 
581                 // can they admin this collection?
582                 if (AuthorizeManager.authorizeActionBoolean(context,
583                         collection, Constants.COLLECTION_ADMIN))
584                 {
585                     request.setAttribute("admin_button", new Boolean(true));
586 
587                     // give them a button to manage submitter list
588                     // what group is the submitter?
589                     Group group = collection.getSubmitters();
590 
591                     if (group != null)
592                     {
593                         request.setAttribute("submitters", group);
594                     }
595                 }
596 
597                 // can they submit to this collection?
598                 if (AuthorizeManager.authorizeActionBoolean(context,
599                         collection, Constants.ADD))
600                 {
601                     request
602                             .setAttribute("can_submit_button",
603                                     new Boolean(true));
604 
605                 }
606                 else
607                 {
608                     request.setAttribute("can_submit_button",
609                             new Boolean(false));
610                 }
611             }
612 
613             // Forward to collection home page
614             request.setAttribute("collection", collection);
615             request.setAttribute("community", community);
616             request.setAttribute("logged.in", new Boolean(e != null));
617             request.setAttribute("subscribed", new Boolean(subscribed));
618             JSPManager.showJSP(request, response, "/collection-home.jsp");
619 
620             if (updated)
621             {
622                 context.complete();
623             }
624         }
625     }

646     /**
647      * Check to see if a browse or search button has been pressed on a community
648      * or collection home page. If so, redirect to the appropriate URL.
649      * 
650      * @param request
651      *            HTTP request
652      * @param response
653      *            HTTP response
654      * @param handle
655      *            Handle of the community/collection home page
656      * 
657      * @return true if a browse/search button was pressed and the user was
658      *         redirected
659      */
660     private boolean handleButton(HttpServletRequest request,
661             HttpServletResponse response, String handle) throws IOException
662     {
663         String button = UIUtil.getSubmitButton(request, "");
664         String location = request.getParameter("location");
665         String prefix = "/";
666         String url = null;
667 
668         if (location == null)
669         {
670             return false;
671         }
672 
673         /*
674          * Work out the "prefix" to which to redirect If "/", scope is all of
675          * DSpace, so prefix is "/" If prefix is a handle, scope is a community
676          * or collection, so "/handle/1721.x/xxxx/" is the prefix.
677          */
678         if (!location.equals("/"))
679         {
680             prefix = "/handle/" + location + "/";
681         }
682 
683         if (button.equals("submit_search")
684                 || (request.getParameter("query") != null))
685         {
686             /*
687              * Have to check for search button and query - in some browsers,
688              * typing a query into the box and hitting return doesn't produce a
689              * submit button parameter. Redirect to appropriate search page
690              */
691             url = request.getContextPath()
692                     + prefix
693                     + "simple-search?query="
694                     + URLEncoder.encode(request.getParameter("query"),
695                             Constants.DEFAULT_ENCODING);
696         }
697 
698         // If a button was pressed, redirect to appropriate page
699         if (url != null)
700         {
701             response.sendRedirect(response.encodeRedirectURL(url));
702 
703             return true;
704         }
705 
706         return false;
707     }

709     /**
710      * Utility method to obtain the titles for the Items in the given list.
711      * 
712      * @param items List of Items
713      * @return array of corresponding titles
714      */
715     private String[] getItemTitles(List items)
716     {
717         String[] titles = new String[items.size()];
718 
719         for (int i = 0; i < items.size(); i++)
720         {
721             Item item = (Item) items.get(i);
722 
723             // FIXME: Should probably check for preferred language?
724             DCValue[] titlesForThis = item.getDC("title", null, Item.ANY);
725 
726             // Just use the first title, if any
727             if (titlesForThis.length == 0)
728             {
729                 // No title at all!
730                 titles[i] = null;
731             }
732             else
733             {
734                 // Use first title
735                 titles[i] = titlesForThis[0].value;
736             }
737         }
738 
739         return titles;
740     }

742     /**
743      * Utility method obtain URLs for the most recent items
744      * 
745      * @param context
746      *            DSpace context
747      * @param items
748      *            the items to get URLs for
749      * @return an array of URLs (in Strings) corresponding to those items
750      */
751     private String[] getItemURLs(Context context, List items)
752             throws SQLException
753     {
754         String[] urls = new String[items.size()];
755 
756         for (int i = 0; i < items.size(); i++)
757         {
758             Item item = (Item) items.get(i);
759             urls[i] = "/handle/" + item.getHandle();
760         }
761 
762         return urls;
763     }

766      * Utility method to produce a list of parent communities for a given
