66  /**
67   * Servlet for browsing through indices, as they are defined in
68   * the configuration.  This class can take a wide variety of inputs from
69   * the user interface:
70   *
71   * - type:  the type of browse (index name) being performed
72   * - order: (ASC | DESC) the direction for result sorting
73   * - value: A specific value to find items around.  For example the author name or subject
74   * - month: integer specification of the month of a date browse
75   * - year: integer specification of the year of a date browse
76   * - starts_with: string value at which to start browsing
77   * - vfocus: start browsing with a value of this string
78   * - focus: integer id of the item at which to start browsing
79   * - rpp: integer number of results per page to display
80   * - sort_by: integer specification of the field to search on
81   * - etal: integer number to limit multiple value items specified in config to
82   *
83   * @author Richard Jones
84   * @version $Revision: 4365 $
85   */
86  public abstract class AbstractBrowserServlet extends DSpaceServlet
87  {

89      private static Logger log = Logger.getLogger(AbstractBrowserServlet.class);

96      /**
97       * Create a BrowserScope from the current request
98       *
99       * @param context The database context
100      * @param request The servlet request
101      * @param response The servlet response
102      * @return A BrowserScope for the current parameters
103      * @throws ServletException
104      * @throws IOException
105      * @throws SQLException
106      * @throws AuthorizeException
107      */
108     protected BrowserScope getBrowserScopeForRequest(Context context, HttpServletRequest request, HttpServletResponse response)
109             throws ServletException, IOException, SQLException, AuthorizeException
110     {
111         try
112         {
113             // first, lift all the stuff out of the request that we might need
114             String type = request.getParameter("type");
115             String order = request.getParameter("order");
116             String value = request.getParameter("value");
117             String valueLang = request.getParameter("value_lang");
118             String month = request.getParameter("month");
119             String year = request.getParameter("year");
120             String startsWith = request.getParameter("starts_with");
121             String valueFocus = request.getParameter("vfocus");
122             String valueFocusLang = request.getParameter("vfocus_lang");
123             String authority = request.getParameter("authority");
124             int focus = UIUtil.getIntParameter(request, "focus");
125             int offset = UIUtil.getIntParameter(request, "offset");
126             int resultsperpage = UIUtil.getIntParameter(request, "rpp");
127             int sortBy = UIUtil.getIntParameter(request, "sort_by");
128             int etAl = UIUtil.getIntParameter(request, "etal");
129 
130             // get the community or collection location for the browse request
131             // Note that we are only interested in getting the "smallest" container,
132             // so if we find a collection, we don't bother looking up the community
133             Collection collection = null;
134             Community community = null;
135             collection = UIUtil.getCollectionLocation(request);
136             if (collection == null)
137             {
138                 community = UIUtil.getCommunityLocation(request);
139             }
140 
141             // process the input, performing some inline validation
142             BrowseIndex bi = null;
143             if (type != null && !"".equals(type))
144             {
145                 bi = BrowseIndex.getBrowseIndex(type);
146             }
147 
148             if (bi == null)
149             {
150                 if (sortBy > 0)
151                     bi = BrowseIndex.getBrowseIndex(SortOption.getSortOption(sortBy));
152                 else
153                     bi = BrowseIndex.getBrowseIndex(SortOption.getDefaultSortOption());
154             }
155 
156             // If we don't have a sort column
157             if (bi != null && sortBy == -1)
158             {
159                 // Get the default one
160                 SortOption so = bi.getSortOption();
161                 if (so != null)
162                 {
163                     sortBy = so.getNumber();
164                 }
165             }
166             else if (bi != null && bi.isItemIndex() && !bi.isInternalIndex())
167             {
168                 // If a default sort option is specified by the index, but it isn't
169                 // the same as sort option requested, attempt to find an index that
170                 // is configured to use that sort by default
171                 // This is so that we can then highlight the correct option in the navigation
172                 SortOption bso = bi.getSortOption();
173                 SortOption so = SortOption.getSortOption(sortBy);
174                 if ( bso != null && bso != so)
175                 {
176                     BrowseIndex newBi = BrowseIndex.getBrowseIndex(so);
177                     if (newBi != null)
178                     {
179                         bi   = newBi;
180                         type = bi.getName();
181                     }
182                 }
183             }
184 
185             if (order == null && bi != null)
186             {
187                 order = bi.getDefaultOrder();
188             }
189 
190             // If the offset is invalid, reset to 0
191             if (offset < 0)
192             {
193                 offset = 0;
194             }
195 
196             // if no resultsperpage set, default to 20
197             if (resultsperpage < 0)
198             {
199                 resultsperpage = 20;
200             }
201 
202             // if year and perhaps month have been selected, we translate these into "startsWith"
203             // if startsWith has already been defined then it is overwritten
204             if (year != null && !"".equals(year) && !"-1".equals(year))
205             {
206                 startsWith = year;
207                 if ((month != null) && !"-1".equals(month) && !"".equals(month))
208                 {
209                     // subtract 1 from the month, so the match works appropriately
210                     if ("ASC".equals(order))
211                     {
212                         month = Integer.toString((Integer.parseInt(month) - 1));
213                     }
214 
215                     // They've selected a month as well
216                     if (month.length() == 1)
217                     {
218                         // Ensure double-digit month number
219                         month = "0" + month;
220                     }
221 
222                     startsWith = year + "-" + month;
223 
224                     if ("ASC".equals(order))
225                     {
226                         startsWith = startsWith + "-32";
227                     }
228                 }
229             }
230 
231             // determine which level of the browse we are at: 0 for top, 1 for second
232             int level = 0;
233             if (value != null || authority != null)
234             {
235                 level = 1;
236             }
237 
238             // if sortBy is still not set, set it to 0, which is default to use the primary index value
239             if (sortBy == -1)
240             {
241                 sortBy = 0;
242             }
243 
244             // figure out the setting for author list truncation
245             if (etAl == -1)     // there is no limit, or the UI says to use the default
246             {
247                 int limitLine = ConfigurationManager.getIntProperty("webui.browse.author-limit");
248                 if (limitLine != 0)
249                 {
250                     etAl = limitLine;
251                 }
252             }
253             else  // if the user has set a limit
254             {
255                 if (etAl == 0)  // 0 is the user setting for unlimited
256                 {
257                     etAl = -1;  // but -1 is the application setting for unlimited
258                 }
259             }
260 
261             // log the request
262             String comHandle = "n/a";
263             if (community != null)
264             {
265                 comHandle = community.getHandle();
266             }
267             String colHandle = "n/a";
268             if (collection != null)
269             {
270                 colHandle = collection.getHandle();
271             }
272 
273             String arguments = "type=" + type + ",order=" + order + ",value=" + value +
274                 ",month=" + month + ",year=" + year + ",starts_with=" + startsWith +
275                 ",vfocus=" + valueFocus + ",focus=" + focus + ",rpp=" + resultsperpage +
276                 ",sort_by=" + sortBy + ",community=" + comHandle + ",collection=" + colHandle +
277                 ",level=" + level + ",etal=" + etAl;
278 
279             log.info(LogManager.getHeader(context, "browse", arguments));
280 
281             // set up a BrowseScope and start loading the values into it
282             BrowserScope scope = new BrowserScope(context);
283             scope.setBrowseIndex(bi);
284             scope.setOrder(order);
285             scope.setFilterValue(value != null?value:authority);
286             scope.setFilterValueLang(valueLang);
287             scope.setJumpToItem(focus);
288             scope.setJumpToValue(valueFocus);
289             scope.setJumpToValueLang(valueFocusLang);
290             scope.setStartsWith(startsWith);
291             scope.setOffset(offset);
292             scope.setResultsPerPage(resultsperpage);
293             scope.setSortBy(sortBy);
294             scope.setBrowseLevel(level);
295             scope.setEtAl(etAl);
296             scope.setAuthorityValue(authority);
297 
298             // assign the scope of either Community or Collection if necessary
299             if (community != null)
300             {
301                 scope.setBrowseContainer(community);
302             }
303             else if (collection != null)
304             {
305                 scope.setBrowseContainer(collection);
306             }
307 
308             // For second level browses on metadata indexes, we need to adjust the default sorting
309             if (bi != null && bi.isMetadataIndex() && scope.isSecondLevel() && scope.getSortBy() <= 0)
310             {
311                 scope.setSortBy(1);
312             }
313 
314             return scope;
315         }
316         catch (SortException se)
317         {
318             log.error("caught exception: ", se);
319             throw new ServletException(se);
320         }
321         catch (BrowseException e)
322         {
323             log.error("caught exception: ", e);
324             throw new ServletException(e);
325         }
326     }

328     /**
329      * Do the usual DSpace GET method.  You will notice that browse does not currently
330      * respond to POST requests.
331      */
332     protected void processBrowse(Context context, BrowserScope scope, HttpServletRequest request, HttpServletResponse response)
333             throws ServletException, IOException, SQLException,
334             AuthorizeException
335     {
336         try
337         {
338             BrowseIndex bi = scope.getBrowseIndex();
339 
340             // now start up a browse engine and get it to do the work for us
341             BrowseEngine be = new BrowseEngine(context);
342             BrowseInfo binfo = be.browse(scope);
343             
344             request.setAttribute("browse.info", binfo);
345 
346             if (AuthorizeManager.isAdmin(context))
347             {
348                 // Set a variable to create admin buttons
349                 request.setAttribute("admin_button", new Boolean(true));
350             }
351 
352             if (binfo.hasResults())
353             {
354                 if (bi.isMetadataIndex() && !scope.isSecondLevel())
355                 {
356                     showSinglePage(context, request, response);
357                 }
358                 else
359                 {
360                     showFullPage(context, request, response);
361                 }
362             }
363             else
364             {
365                 showNoResultsPage(context, request, response);
366             }
367         }
368         catch (BrowseException e)
369         {
370             log.error("caught exception: ", e);
371             throw new ServletException(e);
372         }
373     }
