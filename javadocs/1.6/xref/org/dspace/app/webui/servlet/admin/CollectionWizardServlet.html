131     protected void doDSPost(Context context, HttpServletRequest request,
132             HttpServletResponse response) throws ServletException, IOException,
133             SQLException, AuthorizeException
134     {
135         /*
136          * For POST, we expect from the form:
137          * 
138          * community_id DB ID if it was a 'create a new collection' button press
139          * 
140          * OR
141          * 
142          * collection_id DB ID of collection we're dealing with stage Stage
143          * we're at (from constants above)
144          */
145 
146         // First, see if we have a multipart request
147         // (the 'basic info' page which might include uploading a logo)
148         String contentType = request.getContentType();
149 
150         if ((contentType != null)
151                 && (contentType.indexOf("multipart/form-data") != -1))
152         {
153             // This is a multipart request, so it's a file upload
154             processBasicInfo(context, request, response);
155 
156             return;
157         }
158 
159         int communityID = UIUtil.getIntParameter(request, "community_id");
160 
161         if (communityID > -1)
162         {
163             // We have a community ID, "create new collection" button pressed
164             Community c = Community.find(context, communityID);
165 
166             if (c == null)
167             {
168                 log.warn(LogManager.getHeader(context, "integrity_error",
169                         UIUtil.getRequestLogInfo(request)));
170                 JSPManager.showIntegrityError(request, response);
171 
172                 return;
173             }
174 
175             // Create the collection
176             Collection newCollection = c.createCollection();
177             request.setAttribute("collection", newCollection);
178 
179             if (AuthorizeManager.isAdmin(context))
180             {
181                 // set a variable to show all buttons
182                 request.setAttribute("sysadmin_button", new Boolean(true));
183             }
184             
185             try 
186             {
187                 AuthorizeUtil.authorizeManageAdminGroup(context, newCollection);                
188                 request.setAttribute("admin_create_button", new Boolean(true));
189             }
190             catch (AuthorizeException authex) {
191                 request.setAttribute("admin_create_button", new Boolean(false));
192             }
193             
194             try 
195             {
196                 AuthorizeUtil.authorizeManageSubmittersGroup(context, newCollection);                
197                 request.setAttribute("submitters_button", new Boolean(true));
198             }
199             catch (AuthorizeException authex) {
200                 request.setAttribute("submitters_button", new Boolean(false));
201             }
202             
203             try 
204             {
205                 AuthorizeUtil.authorizeManageWorkflowsGroup(context, newCollection);                
206                 request.setAttribute("workflows_button", new Boolean(true));
207             }
208             catch (AuthorizeException authex) {
209                 request.setAttribute("workflows_button", new Boolean(false));
210             }
211             
212             try 
213             {
214                 AuthorizeUtil.authorizeManageTemplateItem(context, newCollection);                
215                 request.setAttribute("template_button", new Boolean(true));
216             }
217             catch (AuthorizeException authex) {
218                 request.setAttribute("template_button", new Boolean(false));
219             }
220             
221             JSPManager.showJSP(request, response,
222                     "/dspace-admin/wizard-questions.jsp");
223             context.complete();
224         }
225         else
226         {
227             // Collection already created, dealing with one of the wizard pages
228             int collectionID = UIUtil.getIntParameter(request, "collection_id");
229             int stage = UIUtil.getIntParameter(request, "stage");
230 
231             // Get the collection
232             Collection collection = Collection.find(context, collectionID);
233 
234             // Put it in request attributes, as most JSPs will need it
235             request.setAttribute("collection", collection);
236 
237             if (collection == null)
238             {
239                 log.warn(LogManager.getHeader(context, "integrity_error",
240                         UIUtil.getRequestLogInfo(request)));
241                 JSPManager.showIntegrityError(request, response);
242 
243                 return;
244             }
245 
246             // All pages will need this attribute
247             request.setAttribute("collection.id", String.valueOf(collection
248                     .getID()));
249 
250             switch (stage)
251             {
252             case INITIAL_QUESTIONS:
253                 processInitialQuestions(context, request, response, collection);
254 
255                 break;
256 
257             case PERMISSIONS:
258                 processPermissions(context, request, response, collection);
259 
260                 break;
261 
262             case DEFAULT_ITEM:
263                 processDefaultItem(context, request, response, collection);
264 
265                 break;
266 
267             default:
268                 log.warn(LogManager.getHeader(context, "integrity_error",
269                         UIUtil.getRequestLogInfo(request)));
270                 JSPManager.showIntegrityError(request, response);
271             }
272         }
273     }

275     /**
276      * Process input from initial questions page
277      * 
278      * @param context
279      *            DSpace context
280      * @param request
281      *            HTTP request
282      * @param response
283      *            HTTP response
284      * @param collection
285      *            Collection we're editing
286      */
287     private void processInitialQuestions(Context context,
288             HttpServletRequest request, HttpServletResponse response,
289             Collection collection) throws SQLException, ServletException,
290             IOException, AuthorizeException
291     {
292         Group anonymousGroup = Group.find(context, 0);
293 
294         // "Public read" checkbox. Only need to do anything
295         // if it's not checked (only system admin can uncheck this!).
296         if (!UIUtil.getBoolParameter(request, "public_read")
297                 && AuthorizeManager.isAdmin(context))
298         {
299             // Remove anonymous default policies for new items
300             AuthorizeManager.removePoliciesActionFilter(context, collection,
301                     Constants.DEFAULT_ITEM_READ);
302             AuthorizeManager.removePoliciesActionFilter(context, collection,
303                     Constants.DEFAULT_BITSTREAM_READ);
304         }
305 
306         // Some people authorised to submit
307         if (UIUtil.getBoolParameter(request, "submitters"))
308         {
309             // Create submitters group
310             Group g = collection.createSubmitters();
311         }
312 
313         // Check for the workflow steps
314         for (int i = 1; i <= 3; i++)
315         {
316             if (UIUtil.getBoolParameter(request, "workflow" + i))
317             {
318                 // should have workflow step i
319                 Group g = collection.createWorkflowGroup(i);
320             }
321         }
322 
323         // Check for collection administrators
324         if (UIUtil.getBoolParameter(request, "admins"))
325         {
326             // Create administrators group
327             Group g = collection.createAdministrators();
328         }
329 
330         // Default item stuff?
331         if (UIUtil.getBoolParameter(request, "default.item"))
332         {
333             collection.createTemplateItem();
334         }
335 
336         // Need to set a name so that the indexer won't throw an exception
337         collection.setMetadata("name", "");
338         collection.update();
339 
340         // Now display "basic info" screen
341         JSPManager.showJSP(request, response,
342                 "/dspace-admin/wizard-basicinfo.jsp");
343         context.complete();
344     }

347      * Process input from one of the permissions pages
348      * 
349      * @param context
350      *            DSpace context
351      * @param request
352      *            HTTP request
353      * @param response
354      *            HTTP response

358     private void processPermissions(Context context,
359             HttpServletRequest request, HttpServletResponse response,
360             Collection collection) throws SQLException, ServletException,
361             IOException, AuthorizeException
362     {
363         // Which permission are we dealing with?
364         int permission = UIUtil.getIntParameter(request, "permission");
365 
366         // First, we deal with the special case of the MIT group...
367         if (UIUtil.getBoolParameter(request, "mitgroup"))
368         {
369             Group mitGroup = Group.findByName(context, "MIT Users");
370             int action;
371 
372             if (permission == PERM_READ)
373             {
374                 // assign default item and bitstream read to mitGroup
375                 AuthorizeManager.addPolicy(context, collection,
376                         Constants.DEFAULT_ITEM_READ, mitGroup);
377                 AuthorizeManager.addPolicy(context, collection,
378                         Constants.DEFAULT_BITSTREAM_READ, mitGroup);
379             }
380             else
381             {
382                 // Must be submit
383                 AuthorizeManager.addPolicy(context, collection, Constants.ADD,
384                         mitGroup);
385             }
386         }
387 
388         //We need to add the selected people to the group.
389         // First, get the relevant group
390         Group g = null;
391 
392         switch (permission)
393         {
394         case PERM_READ:
395 
396             // Actually need to create a group for this.
397             g = Group.create(context);
398 
399             // Name it according to our conventions
400             g
401                     .setName("COLLECTION_" + collection.getID()
402                             + "_DEFAULT_ITEM_READ");
403 
404             // Give it the needed permission
405             AuthorizeManager.addPolicy(context, collection,
406                     Constants.DEFAULT_ITEM_READ, g);
407             AuthorizeManager.addPolicy(context, collection,
408                     Constants.DEFAULT_BITSTREAM_READ, g);
409 
410             break;
411 
412         case PERM_SUBMIT:
413             g = collection.getSubmitters();
414 
415             break;
416 
417         case PERM_WF1:
418             g = collection.getWorkflowGroup(1);
419 
420             break;
421 
422         case PERM_WF2:
423             g = collection.getWorkflowGroup(2);
424 
425             break;
426 
427         case PERM_WF3:
428             g = collection.getWorkflowGroup(3);
429 
430             break;
431 
432         case PERM_ADMIN:
433             g = collection.getAdministrators();
434 
435             break;
436         }
437 
438         // Add people and groups from the form to the group
439         int[] eperson_ids = UIUtil.getIntParameters(request, "eperson_id");
440         int[] group_ids = UIUtil.getIntParameters(request, "group_ids");
441         
442         if (eperson_ids != null)
443         {
444             for (int i = 0; i < eperson_ids.length; i++)
445             {
446                 EPerson eperson = EPerson.find(context, eperson_ids[i]);
447 
448                 if (eperson != null)
449                 {
450                     g.addMember(eperson);
451                 }
452             }
453         }
454         
455         if (group_ids != null)
456         {
457             for (int i = 0; i < group_ids.length; i++)
458             {
459                 Group group = Group.find(context, group_ids[i]);
460             
461                 if (group != null)
462                 {
463                     g.addMember(group);
464                 }
465             }
466         }
467         
468 
469         // Update group
470         g.update();
471 
472         showNextPage(context, request, response, collection, permission);
473 
474         context.complete();
475     }

477     /**
478      * process input from basic info page
479      * 
480      * @param context
481      * @param request
482      * @param response
483      * @param collection
484      * @throws SQLException
485      * @throws ServletException
486      * @throws IOException
487      * @throws AuthorizeException
488      */
489     private void processBasicInfo(Context context, HttpServletRequest request,
490             HttpServletResponse response) throws SQLException,
491             ServletException, IOException, AuthorizeException
492     {
493         // Wrap multipart request to get the submission info
494         FileUploadRequest wrapper = new FileUploadRequest(request);
495 
496         Collection collection = Collection.find(context, UIUtil
497                 .getIntParameter(wrapper, "collection_id"));
498 
499         if (collection == null)
500         {
501             log.warn(LogManager.getHeader(context, "integrity_error", UIUtil
502                     .getRequestLogInfo(wrapper)));
503             JSPManager.showIntegrityError(request, response);
504 
505             return;
506         }
507 
508         // Get metadata
509         collection.setMetadata("name", wrapper.getParameter("name"));
510         collection.setMetadata("short_description", wrapper
511                 .getParameter("short_description"));
512         collection.setMetadata("introductory_text", wrapper
513                 .getParameter("introductory_text"));
514         collection.setMetadata("copyright_text", wrapper
515                 .getParameter("copyright_text"));
516         collection.setMetadata("side_bar_text", wrapper
517                 .getParameter("side_bar_text"));
518         collection.setMetadata("provenance_description", wrapper
519                 .getParameter("provenance_description"));
520 
521         // Need to be more careful about license -- make sure it's null if
522         // nothing was entered
523         String license = wrapper.getParameter("license");
524 
525         if ((license != null) || "".equals(license))
526         {
527             collection.setLicense(license);
528         }
529 
530         File temp = wrapper.getFile("file");
531 
532         if (temp != null)
533         {
534             // Read the temp file as logo
535             InputStream is = new BufferedInputStream(new FileInputStream(temp));
536             Bitstream logoBS = collection.setLogo(is);
537 
538             // Strip all but the last filename. It would be nice
539             // to know which OS the file came from.
540             String noPath = wrapper.getFilesystemName("file");
541 
542             while (noPath.indexOf('/') > -1)
543             {
544                 noPath = noPath.substring(noPath.indexOf('/') + 1);
545             }
546 
547             while (noPath.indexOf('\\') > -1)
548             {
549                 noPath = noPath.substring(noPath.indexOf('\\') + 1);
550             }
551 
552             logoBS.setName(noPath);
553             logoBS.setSource(wrapper.getFilesystemName("file"));
554 
555             // Identify the format
556             BitstreamFormat bf = FormatIdentifier.guessFormat(context, logoBS);
557             logoBS.setFormat(bf);
558             AuthorizeManager.addPolicy(context, logoBS, Constants.WRITE, context
559                     .getCurrentUser());
560             logoBS.update();
561 
562             // Remove temp file
563             temp.delete();
564         }
565 
566         collection.update();
567 
568         // Now work out what next page is
569         showNextPage(context, request, response, collection, BASIC_INFO);
570 
571         context.complete();
572     }

575      * Process input from default item page
576      * 
577      * @param context
578      *            DSpace context

586     private void processDefaultItem(Context context,
587             HttpServletRequest request, HttpServletResponse response,
588             Collection collection) throws SQLException, ServletException,
589             IOException, AuthorizeException
590     {
591         Item item = collection.getTemplateItem();
592 
593         for (int i = 0; i < 10; i++)
594         {
595             int dcTypeID = UIUtil.getIntParameter(request, "dctype_" + i);
596             String value = request.getParameter("value_" + i);
597             String lang = request.getParameter("lang_" + i);
598 
599             if ((dcTypeID != -1) && (value != null) && !value.equals(""))
600             {
601                 MetadataField field = MetadataField.find(context,dcTypeID);
602                 MetadataSchema schema = MetadataSchema.find(context,field.getSchemaID());
603                 item.addMetadata(schema.getName(),field.getElement(), field.getQualifier(), lang, value);
604             }
605         }
606 
607         item.update();
608 
609         // Now work out what next page is
610         showNextPage(context, request, response, collection, DEFAULT_ITEM);
611 
612         context.complete();
613     }

615     /**
616      * Work out which page to show next, and show it
617      * 
618      * @param context
619      * @param request
620      * @param response
621      * @param collection
622      * @param stage
623      *            the stage the user just finished, or if PERMISSIONS, the
624      *            particular permissions page
625      * @throws SQLException
626      * @throws ServletException
627      * @throws IOException
628      * @throws AuthorizeException
629      */
630     private void showNextPage(Context context, HttpServletRequest request,
631             HttpServletResponse response, Collection collection, int stage)
632             throws SQLException, ServletException, IOException,
633             AuthorizeException
634     {
635         // Put collection in request attributes, as most JSPs will need it
636         request.setAttribute("collection", collection);
637 
638         // FIXME: Not a nice hack -- do we show the MIT users checkbox?
639         if (Group.findByName(context, "MIT Users") != null)
640         {
641             request.setAttribute("mitgroup", new Boolean(true));
642         }
643 
644         log.debug(LogManager.getHeader(context, "nextpage", "stage=" + stage));
645 
646         switch (stage)
647         {
648         case BASIC_INFO:
649 
650             // Next page is 'permission to read' page iff ITEM_DEFAULT_READ
651             // for anonymous group is NOT there
652             List anonReadPols = AuthorizeManager.getPoliciesActionFilter(
653                     context, collection, Constants.DEFAULT_ITEM_READ);
654 
655             // At this stage, if there's any ITEM_DEFAULT_READ, it can only
656             // be an anonymous one.
657             if (anonReadPols.size() == 0)
658             {
659                 request.setAttribute("permission", new Integer(PERM_READ));
660                 JSPManager.showJSP(request, response,
661                         "/dspace-admin/wizard-permissions.jsp");
662 
663                 break;
664             }
665 
666         case PERM_READ:
667 
668             // Next page is 'permission to submit' iff there's a submit group
669             // defined
670             if (collection.getSubmitters() != null)
671             {
672                 request.setAttribute("permission", new Integer(PERM_SUBMIT));
673                 JSPManager.showJSP(request, response,
674                         "/dspace-admin/wizard-permissions.jsp");
675 
676                 break;
677             }
678 
679         case PERM_SUBMIT:
680 
681             // Next page is 'workflow step 1' iff there's a wf step 1 group
682             // defined
683             if (collection.getWorkflowGroup(1) != null)
684             {
685                 request.setAttribute("permission", new Integer(PERM_WF1));
686                 JSPManager.showJSP(request, response,
687                         "/dspace-admin/wizard-permissions.jsp");
688 
689                 break;
690             }
691 
692         case PERM_WF1:
693 
694             // Next page is 'workflow step 2' iff there's a wf step 2 group
695             // defined
696             if (collection.getWorkflowGroup(2) != null)
697             {
698                 request.setAttribute("permission", new Integer(PERM_WF2));
699                 JSPManager.showJSP(request, response,
700                         "/dspace-admin/wizard-permissions.jsp");
701 
702                 break;
703             }
704 
705         case PERM_WF2:
706 
707             // Next page is 'workflow step 3' iff there's a wf step 2 group
708             // defined
709             if (collection.getWorkflowGroup(3) != null)
710             {
711                 request.setAttribute("permission", new Integer(PERM_WF3));
712                 JSPManager.showJSP(request, response,
713                         "/dspace-admin/wizard-permissions.jsp");
714 
715                 break;
716             }
717 
718         case PERM_WF3:
719 
720             // Next page is 'collection administrator' iff there's a collection
721             // administrator group
722             if (collection.getAdministrators() != null)
723             {
724                 request.setAttribute("permission", new Integer(PERM_ADMIN));
725                 JSPManager.showJSP(request, response,
726                         "/dspace-admin/wizard-permissions.jsp");
727 
728                 break;
729             }
730 
731         case PERM_ADMIN:
732 
733             // Next page is 'default item' iff there's a default item
734             if (collection.getTemplateItem() != null)
735             {
736                 MetadataField[] types = MetadataField.findAll(context);
737                 request.setAttribute("dctypes", types);
738                 JSPManager.showJSP(request, response,
739                         "/dspace-admin/wizard-default-item.jsp");
740 
741                 break;
742             }
743 
744         case DEFAULT_ITEM:
745 
746             // Next page is 'summary page (the last page)
747             // sort of a hack to pass the community ID to the edit collection
748             // page,
749             // which needs it in other contexts
750             if (collection != null)
751             {
752                 Community[] communities = collection.getCommunities();
753                 request.setAttribute("community", communities[0]);
754 
755                 EditCommunitiesServlet.storeAuthorizeAttributeCollectionEdit(context, request, collection);
756             }
757 
758             JSPManager.showJSP(request, response, "/tools/edit-collection.jsp");
759 
760             break;
761         }
762     }
