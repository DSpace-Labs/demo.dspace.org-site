76  /**
77   * Upload step for DSpace JSP-UI. Handles the pages that revolve around uploading files
78   * (and verifying a successful upload) for an item being submitted into DSpace.
79   * <P>
80   * This JSPStep class works with the SubmissionController servlet
81   * for the JSP-UI
82   * <P>
83   * The following methods are called in this order:
84   * <ul>
85   * <li>Call doPreProcessing() method</li>
86   * <li>If showJSP() was specified from doPreProcessing(), then the JSP
87   * specified will be displayed</li>
88   * <li>If showJSP() was not specified from doPreProcessing(), then the
89   * doProcessing() method is called an the step completes immediately</li>
90   * <li>Call doProcessing() method on appropriate AbstractProcessingStep after the user returns from the JSP, in order
91   * to process the user input</li>
92   * <li>Call doPostProcessing() method to determine if more user interaction is
93   * required, and if further JSPs need to be called.</li>
94   * <li>If there are more "pages" in this step then, the process begins again
95   * (for the new page).</li>
96   * <li>Once all pages are complete, control is forwarded back to the
97   * SubmissionController, and the next step is called.</li>
98   * </ul>
99   *
100  * @see org.dspace.app.webui.servlet.SubmissionController
101  * @see org.dspace.app.webui.submit.JSPStep
102  * @see org.dspace.submit.step.UploadStep
103  *
104  * @author Tim Donohue
105  * @version $Revision: 4365 $
106  */

133     /**
134      * Do any pre-processing to determine which JSP (if any) is used to generate
135      * the UI for this step. This method should include the gathering and
136      * validating of all data required by the JSP. In addition, if the JSP
137      * requires any variable to passed to it on the Request, this method should
138      * set those variables.
139      * <P>
140      * If this step requires user interaction, then this method must call the
141      * JSP to display, using the "showJSP()" method of the JSPStepManager class.
142      * <P>
143      * If this step doesn't require user interaction OR you are solely using
144      * Manakin for your user interface, then this method may be left EMPTY,
145      * since all step processing should occur in the doProcessing() method.
146      *
147      * @param context
148      *            current DSpace context
149      * @param request
150      *            current servlet request object
151      * @param response
152      *            current servlet response object
153      * @param subInfo
154      *            submission info object
155      */
156     public void doPreProcessing(Context context, HttpServletRequest request,
157             HttpServletResponse response, SubmissionInfo subInfo)
158             throws ServletException, IOException, SQLException,
159             AuthorizeException
160     {
161         // pass on the fileupload setting
162         if (subInfo != null)
163         {
164             Collection c = subInfo.getSubmissionItem().getCollection();
165             try
166             {
167                 DCInputsReader inputsReader = new DCInputsReader();
168                 request.setAttribute("submission.inputs", inputsReader.getInputs(c
169                         .getHandle()));
170             }
171             catch (DCInputsReaderException e)
172             {
173                 throw new ServletException(e);
174             }
175         }
176              
177         // show whichever upload page is appropriate
178         // (based on if this item already has files or not)
179         showUploadPage(context, request, response, subInfo, false);
180     }

182     /**
183      * Do any post-processing after the step's backend processing occurred (in
184      * the doProcessing() method).
185      * <P>
186      * It is this method's job to determine whether processing completed
187      * successfully, or display another JSP informing the users of any potential
188      * problems/errors.
189      * <P>
190      * If this step doesn't require user interaction OR you are solely using
191      * Manakin for your user interface, then this method may be left EMPTY,
192      * since all step processing should occur in the doProcessing() method.
193      *
194      * @param context
195      *            current DSpace context
196      * @param request
197      *            current servlet request object
198      * @param response
199      *            current servlet response object
200      * @param subInfo
201      *            submission info object
202      * @param status
203      *            any status/errors reported by doProcessing() method
204      */
205     public void doPostProcessing(Context context, HttpServletRequest request,
206             HttpServletResponse response, SubmissionInfo subInfo, int status)
207             throws ServletException, IOException, SQLException,
208             AuthorizeException
209     {
210         String buttonPressed = UIUtil.getSubmitButton(request, UploadStep.NEXT_BUTTON);
211 
212         // Do we need to skip the upload entirely?
213         boolean fileRequired = ConfigurationManager.getBooleanProperty("webui.submit.upload.required", true);
214         if (buttonPressed.equalsIgnoreCase(UploadStep.SUBMIT_SKIP_BUTTON) ||
215             (buttonPressed.equalsIgnoreCase(UploadStep.SUBMIT_UPLOAD_BUTTON) && !fileRequired))
216         {
217             Bundle[] bundles = subInfo.getSubmissionItem().getItem()
218                     .getBundles("ORIGINAL");
219 
220             // if user already has uploaded at least one file
221             if (bundles.length > 0)
222             {
223                 // return to list of uploaded files
224                 showUploadFileList(context, request, response, subInfo, true,
225                         false);
226             }
227             
228             return; // return immediately, since we are skipping upload
229         }
230         
231         //If upload failed in JSPUI (just came from upload-error.jsp), user can retry the upload
232         if(buttonPressed.equalsIgnoreCase("submit_retry"))
233         {
234             showUploadPage(context, request, response, subInfo, false);
235         }
236         
237         // ------------------------------
238         // Check for Errors!
239         // ------------------------------
240         // if an error or message was passed back, determine what to do!
241         if (status != UploadStep.STATUS_COMPLETE)
242         {
243             if (status == UploadStep.STATUS_INTEGRITY_ERROR)
244             {
245                 // Some type of integrity error occurred
246                 log.warn(LogManager.getHeader(context, "integrity_error",
247                         UIUtil.getRequestLogInfo(request)));
248                 JSPManager.showIntegrityError(request, response);
249             }
250             else if (status == UploadStep.STATUS_UPLOAD_ERROR || status == UploadStep.STATUS_NO_FILES_ERROR)
251             {
252                 // There was a problem uploading the file!
253 
254                 //First, check if we just removed our uploaded file
255                 if(buttonPressed.startsWith("submit_remove_"))
256                 {
257                     //if file was just removed, go back to upload page
258                     showUploadPage(context, request, response, subInfo, false);
259                 }
260                 else
261                 {
262                     // We need to show the file upload error page
263                     if (subInfo != null)
264                     {
265                         try
266                         {
267                             Collection c = subInfo.getSubmissionItem().getCollection();
268                             DCInputsReader inputsReader = new DCInputsReader();
269                             request.setAttribute("submission.inputs", inputsReader
270                                     .getInputs(c.getHandle()));
271                         }
272                         catch (DCInputsReaderException e)
273                         {
274                             throw new ServletException(e);
275                         }
276 
277                     }
278                     JSPStepManager.showJSP(request, response, subInfo, UPLOAD_ERROR_JSP);
279                 }
280             }
281             else if (status == UploadStep.STATUS_UNKNOWN_FORMAT)
282             {
283                 // user uploaded a file where the format is unknown to DSpace
284 
285                 // forward user to page to request the file format
286                 showGetFileFormat(context, request, response, subInfo);
287             }
288         }
289         
290         // As long as there are no errors, clicking Next
291         // should immediately send them to the next step
292         if (status == UploadStep.STATUS_COMPLETE && buttonPressed.equals(UploadStep.NEXT_BUTTON))
293         {
294             // just return, so user will continue on to next step!
295             return;
296         }
297 
298         // ------------------------------
299         // Check for specific buttons
300         // ------------------------------
301         if (buttonPressed.equals(UploadStep.SUBMIT_MORE_BUTTON))
302         {
303             // Upload another file (i.e. show the Choose File jsp again)
304             showChooseFile(context, request, response, subInfo);
305         }
306         else if (buttonPressed.equals("submit_show_checksums"))
307         {
308             // Show the checksums
309             showUploadFileList(context, request, response, subInfo, false, true);
310         }
311         else if (buttonPressed.startsWith("submit_describe_"))
312         {
313             // Change the description of a bitstream
314             Bitstream bitstream;
315 
316             // Which bitstream does the user want to describe?
317             try
318             {
319                 int id = Integer.parseInt(buttonPressed.substring(16));
320                 bitstream = Bitstream.find(context, id);
321             }
322             catch (NumberFormatException nfe)
323             {
324                 bitstream = null;
325             }
326 
327             if (bitstream == null)
328             {
329                 // Invalid or mangled bitstream ID
330                 // throw an error and return immediately
331                 log.warn(LogManager.getHeader(context, "integrity_error",
332                         UIUtil.getRequestLogInfo(request)));
333                 JSPManager.showIntegrityError(request, response);
334             }
335 
336             // save the bitstream
337             subInfo.setBitstream(bitstream);
338 
339             // load the change file description page
340             showFileDescription(context, request, response, subInfo);
341         }
342         else if (buttonPressed.startsWith("submit_format_"))
343         {
344             // A "format is wrong" button must have been pressed
345             Bitstream bitstream;
346 
347             // Which bitstream does the user want to describe?
348             try
349             {
350                 int id = Integer.parseInt(buttonPressed.substring(14));
351                 bitstream = Bitstream.find(context, id);
352             }
353             catch (NumberFormatException nfe)
354             {
355                 bitstream = null;
356             }
357 
358             if (bitstream == null)
359             {
360                 // Invalid or mangled bitstream ID
361                 // throw an error and return immediately
362                 log.warn(LogManager.getHeader(context, "integrity_error",
363                         UIUtil.getRequestLogInfo(request)));
364                 JSPManager.showIntegrityError(request, response);
365             }
366 
367             subInfo.setBitstream(bitstream);
368             showGetFileFormat(context, request, response, subInfo);
369         }
370         else
371         {
372             // BY DEFAULT: just display either the first upload page or the
373             // upload file listing
374             String contentType = request.getContentType();
375             boolean fileUpload = false;
376 
377             // if multipart form, then we just finished a file upload
378             if ((contentType != null)
379                     && (contentType.indexOf("multipart/form-data") != -1))
380                 fileUpload = true;
381 
382             // show the appropriate upload page
383             // (based on if a file has just been uploaded or not)
384             showUploadPage(context, request, response, subInfo, fileUpload);
385         }
386     }

388     /**
389      * Display the appropriate upload page in the file upload sequence. Which
390      * page this is depends on whether the user has uploaded any files in this
391      * item already.
392      *
393      * @param context
394      *            the DSpace context object
395      * @param request

397      * @param response

404     private void showUploadPage(Context context, HttpServletRequest request,
405             HttpServletResponse response, SubmissionInfo subInfo,
406             boolean justUploaded) throws SQLException, ServletException,
407             IOException
408     {
409         Bundle[] bundles = subInfo.getSubmissionItem().getItem().getBundles(
410                 "ORIGINAL");
411 
412         if (justUploaded || bundles.length > 0)
413         {
414             // The item already has files associated with it.
415             showUploadFileList(context, request, response, subInfo,
416                     justUploaded, false);
417         }
418         else
419         {
420             // show the page to choose a file to upload
421             showChooseFile(context, request, response, subInfo);
422         }

489     /**
490      * Show the page which allows the user to change the format of the file that
491      * was just uploaded
492      *
493      * @param context
494      *            context object
495      * @param request
496      *            the request object
497      * @param response
498      *            the response object
499      * @param subInfo
500      *            the SubmissionInfo object
501      */
502     private void showGetFileFormat(Context context, HttpServletRequest request,
503             HttpServletResponse response, SubmissionInfo subInfo)
504             throws SQLException, ServletException, IOException
505     {
506         if (subInfo == null || subInfo.getBitstream() == null)
507         {
508             // We have an integrity error, since we seem to have lost
509             // which bitstream was just uploaded
510             log.warn(LogManager.getHeader(context, "integrity_error", UIUtil
511                     .getRequestLogInfo(request)));
512             JSPManager.showIntegrityError(request, response);
513         }
514 
515         BitstreamFormat[] formats = BitstreamFormat.findNonInternal(context);
516 
517         request.setAttribute("bitstream.formats", formats);
518 
519         // What does the system think it is?
520         BitstreamFormat guess = FormatIdentifier.guessFormat(context, subInfo
521                 .getBitstream());
522 
523         request.setAttribute("guessed.format", guess);
524 
525         // display choose file format JSP next
526         JSPStepManager.showJSP(request, response, subInfo, FILE_FORMAT_JSP);
527     }
