68  /**
69   * License step for DSpace JSP-UI. Presents the user with license information
70   * required for all items submitted into DSpace.
71   * <P>
72   * This JSPStep class works with the SubmissionController servlet
73   * for the JSP-UI
74   * <P>
75   * The following methods are called in this order:
76   * <ul>
77   * <li>Call doPreProcessing() method</li>
78   * <li>If showJSP() was specified from doPreProcessing(), then the JSP
79   * specified will be displayed</li>
80   * <li>If showJSP() was not specified from doPreProcessing(), then the
81   * doProcessing() method is called an the step completes immediately</li>
82   * <li>Call doProcessing() method on appropriate AbstractProcessingStep after the user returns from the JSP, in order
83   * to process the user input</li>
84   * <li>Call doPostProcessing() method to determine if more user interaction is
85   * required, and if further JSPs need to be called.</li>
86   * <li>If there are more "pages" in this step then, the process begins again
87   * (for the new page).</li>
88   * <li>Once all pages are complete, control is forwarded back to the
89   * SubmissionController, and the next step is called.</li>
90   * </ul>
91   * 
92   * @see org.dspace.app.webui.servlet.SubmissionController
93   * @see org.dspace.app.webui.submit.JSPStep
94   * @see org.dspace.submit.step.LicenseStep
95   * 
96   * @author Tim Donohue
97   * @version $Revision: 3705 $
98   */

111     private static Logger log = Logger.getLogger(JSPLicenseStep.class);

113     /**
114      * Do any pre-processing to determine which JSP (if any) is used to generate
115      * the UI for this step. This method should include the gathering and
116      * validating of all data required by the JSP. In addition, if the JSP
117      * requires any variable to passed to it on the Request, this method should
118      * set those variables.
119      * <P>
120      * If this step requires user interaction, then this method must call the
121      * JSP to display, using the "showJSP()" method of the JSPStepManager class.
122      * <P>
123      * If this step doesn't require user interaction OR you are solely using
124      * Manakin for your user interface, then this method may be left EMPTY,
125      * since all step processing should occur in the doProcessing() method.
126      * 
127      * @param context
128      *            current DSpace context
129      * @param request
130      *            current servlet request object
131      * @param response
132      *            current servlet response object
133      * @param subInfo
134      *            submission info object
135      */
136     public void doPreProcessing(Context context, HttpServletRequest request,
137             HttpServletResponse response, SubmissionInfo subInfo)
138             throws ServletException, IOException, SQLException,
139             AuthorizeException
140     {
141         // if creative commons licensing is enabled, then it is page #1
142         if (CreativeCommons.isEnabled() && LicenseStep.getCurrentPage(request) == 1)
143         {
144             showCCLicense(context, request, response, subInfo);
145         }
146         else
147         // otherwise load default DSpace license agreement
148         {
149             showLicense(context, request, response, subInfo);
150         }
151     }

153     /**
154      * Do any post-processing after the step's backend processing occurred (in
155      * the doProcessing() method).
156      * <P>
157      * It is this method's job to determine whether processing completed
158      * successfully, or display another JSP informing the users of any potential
159      * problems/errors.
160      * <P>
161      * If this step doesn't require user interaction OR you are solely using
162      * Manakin for your user interface, then this method may be left EMPTY,
163      * since all step processing should occur in the doProcessing() method.
164      * 
165      * @param context
166      *            current DSpace context
167      * @param request
168      *            current servlet request object
169      * @param response
170      *            current servlet response object
171      * @param subInfo
172      *            submission info object
173      * @param status
174      *            any status/errors reported by doProcessing() method
175      */
176     public void doPostProcessing(Context context, HttpServletRequest request,
177             HttpServletResponse response, SubmissionInfo subInfo, int status)
178             throws ServletException, IOException, SQLException,
179             AuthorizeException
180     {
181         String buttonPressed = Util.getSubmitButton(request, LicenseStep.CANCEL_BUTTON);
182 
183         //  JSP-UI Specific (only JSP UI has a "reject" button): 
184         //  License was explicitly rejected
185         if (buttonPressed.equals("submit_reject"))
186         {
187             // User has rejected license.
188             log.info(LogManager.getHeader(context, "reject_license",
189                     subInfo.getSubmissionLogInfo()));
190 
191             // If the license page was the 1st page in the Submission process,
192             // then delete the submission if they reject the license!
193             if (!subInfo.isInWorkflow()
194                     && (SubmissionController.getStepReached(subInfo) <= SubmissionController.FIRST_STEP))
195             {
196                 WorkspaceItem wi = (WorkspaceItem) subInfo.getSubmissionItem();
197                 wi.deleteAll();
198 
199                 // commit changes
200                 context.commit();
201             }
202 
203             // Show the license rejected page
204             JSPManager.showJSP(request, response, LICENSE_REJECT_JSP);
205         }
206 
207     }

210     /**
211      * Show the DSpace license page to the user
212      * 
213      * @param context
214      *            current DSpace context
215      * @param request
216      *            the request object
217      * @param response
218      *            the response object
219      * @param subInfo
220      *            the SubmissionInfo object
221      */
222     private void showLicense(Context context, HttpServletRequest request,
223             HttpServletResponse response, SubmissionInfo subInfo)
224             throws SQLException, ServletException, IOException
225     {
226         // determine collection & get license
227         Collection c = subInfo.getSubmissionItem().getCollection();
228         String license = c.getLicenseCollection();
229 
230         if ((license == null) || license.equals(""))
231         {
232             // Fallback to site-wide default
233             license = ConfigurationManager.getLicenseText(I18nUtil.getDefaultLicense(context));
234         }
235         request.setAttribute("license", license);
236 
237         JSPStepManager.showJSP(request, response, subInfo, LICENSE_JSP);
238     }

240     /**
241      * Show the Creative Commons license page to the user
242      * 
243      * @param context
244      *            current DSpace context
245      * @param request
246      *            the request object
247      * @param response
248      *            the response object
249      * @param subInfo
250      *            the SubmissionInfo object
251      */
252     private void showCCLicense(Context context, HttpServletRequest request,
253             HttpServletResponse response, SubmissionInfo subInfo)
254             throws SQLException, ServletException, IOException
255     {
256         // Do we already have a CC license?
257         Item item = subInfo.getSubmissionItem().getItem();
258         boolean exists = CreativeCommons.hasLicense(context, item);
259         request.setAttribute("cclicense.exists", new Boolean(exists));
260 
261         JSPStepManager.showJSP(request, response, subInfo, CC_LICENSE_JSP);
262     }
