98      /**
99       * Initialize the current JSPStepManager object, by loading the 
100      * specified step class.
101      * 
102      * @param stepConfig
103      *            the SubmissionStepConfig object which describes
104      *            this step's configuration in the item-submission.xml
105      *            
106      * @throws Exception
107      *             if the JSPStep cannot be loaded or the class
108      *             specified doesn't implement the JSPStep interface
109      */
110     public static JSPStepManager loadStep(SubmissionStepConfig stepConfig) throws Exception
111     {
112         
113         JSPStepManager stepManager = new JSPStepManager();
114         
115         //save step configuration 
116         stepManager.stepConfig = stepConfig;
117         
118         
119         /*
120          * First, load the step processing class (using the current class loader)
121          */
122         ClassLoader loader = stepManager.getClass().getClassLoader();
123         Class stepClass = loader
124                 .loadClass(stepConfig.getProcessingClassName());
125 
126         Object stepInstance =  stepClass.newInstance();
127         
128         if(stepInstance instanceof AbstractProcessingStep)
129         {
130             // load the JSPStep interface for this step
131             stepManager.stepProcessing = (AbstractProcessingStep) stepClass.newInstance();
132         }
133         else
134         {
135             throw new Exception("The submission step class specified by '" + stepConfig.getProcessingClassName() + 
136                     "' does not extend the class org.dspace.submit.AbstractProcessingStep!" +
137                     " Therefore it cannot be used by the Configurable Submission as the <processing-class>!");
138         }
139         
140         
141         /*
142          * Next, load the step's JSPUI binding class (using the current class loader)
143          * (Only load JSPUI binding class if specified...otherwise this is a non-interactive step)
144          */
145         if(stepConfig.getJSPUIClassName()!=null && stepConfig.getJSPUIClassName().length()>0)
146         {
147         	stepClass = loader
148                 	.loadClass(stepConfig.getJSPUIClassName());
149 
150         	stepInstance =  stepClass.newInstance();
151         
152 	        if(stepInstance instanceof JSPStep)
153 	        {
154 	            // load the JSPStep interface for this step
155 	            stepManager.stepJSPUI = (JSPStep) stepClass.newInstance();
156 	        }
157 	        else
158 	        {
159 	            throw new Exception("The submission step class specified by '" + stepConfig.getJSPUIClassName() + 
160 	                    "' does not extend the class org.dspace.app.webui.JSPStep!" +
161 	                    " Therefore it cannot be used by the Configurable Submission for the JSP user interface!");
162 	        }
163         }
164         return stepManager;
165     }

168     /**
169      * Initialize the current JSPStepManager object, to prepare for processing
170      * of this step. This method is called by the SubmissionController, and its
171      * job is to determine which "page" of the step the current user is on.
172      * <P>
173      * Once the page has been determined, this method also determines whether
174      * the user is completing this page (i.e. there is user input data that
175      * needs to be saved to the database) or beginning this page (i.e. the user
176      * needs to be sent to the JSP for this page).
177      * 
178      * @param context
179      *            a DSpace Context object
180      * @param request
181      *            the HTTP request
182      * @param response
183      *            the HTTP response
184      * @param subInfo
185      *            submission info object
186      * 
187      * @return true if step is completed (successfully), false otherwise
188      * 
189      * @throws ServletException
190      *             if a general error occurs
191      * @throws IOException
192      *             if a i/o error occurs
193      * @throws SQLException
194      *             if a database error occurs
195      * @throws AuthorizeException
196      *             if some authorization error occurs
197      */
198     public final boolean processStep(Context context,
199             HttpServletRequest request, HttpServletResponse response,
200             SubmissionInfo subInfo) throws ServletException, IOException,
201             SQLException, AuthorizeException
202     {
203         /*
204          * This method SHOULD NOT BE OVERRIDDEN, unless it's absolutely
205          * necessary. If you override this method, make sure you call the
206          * "doStepStart()" and "doStepEnd()" methods at the appropriate time in
207          * your processStep() method.
208          * 
209          */
210 
211         /*
212          * Figure out Current Page in this Step
213          */
214         int currentPage = -1;
215         // if user came to this Step by pressing "Previous" button
216         // from the following step, then we will have to calculate
217         // the correct page to go to.
218         if ((request.getAttribute("step.backwards") != null)
219                 && ((Boolean) request.getAttribute("step.backwards"))
220                         .booleanValue())
221         {
222             // current page should be the LAST page in this step
223             currentPage = getNumPagesInProgressBar(subInfo, this.stepConfig
224                     .getStepNumber());
225 
226             AbstractProcessingStep.setCurrentPage(request, currentPage);
227         }
228         else
229         {
230             // retrieve current page number from request
231             currentPage = AbstractProcessingStep.getCurrentPage(request);
232         }
233 
234         /*
235          * Update Last page reached (if necessary)
236          */
237         int pageReached = getPageReached(subInfo);
238 
239         // TD: Special Case, where an item was just rejected & returned to
240         // Workspace. In this case, we have an invalid "pageReached" in
241         // the database which needs updating
242         if (pageReached == AbstractProcessingStep.LAST_PAGE_REACHED)
243         {
244             // the first time this flag is encountered, we have to update
245             // the Database with the number of the last page in the current
246             // StepReached
247 
248             // get the number of pages in the last step reached
249             int lastPageNum = getNumPagesInProgressBar(subInfo,
250                     SubmissionController.getStepReached(subInfo));
251 
252             // update database with the number of this last page
253             updatePageReached(subInfo, lastPageNum);
254             context.commit();
255         }
256         // otherwise, check if pageReached needs updating
257         // (it only needs updating if we're in the latest step reached,
258         // and we've made it to a later page in that step)
259         else if ((this.stepConfig.getStepNumber() == SubmissionController
260                 .getStepReached(subInfo))
261                 && (currentPage > pageReached))
262         {
263             // reset page number reached & commit to database
264             updatePageReached(subInfo, currentPage);
265             context.commit();
266         }
267 
268         /*
269          * Determine whether we are Starting or Finishing this Step
270          */
271         // check if we just started this step
272         boolean beginningOfStep = SubmissionController
273                 .isBeginningOfStep(request);
274 
275         // if this step has just been started, do beginning processing
276         if (beginningOfStep)
277         {
278             return doStepStart(context, request, response, subInfo);
279         }
280         else
281         // We just returned from a page (i.e. JSP), so we need to do any end
282         // processing
283         {
284             return doStepEnd(context, request, response, subInfo);
285         }
286     }

288     /**
289      * Do the beginning of a Step. First do pre-processing, then display the JSP
290      * (if there is one). If there's no JSP, just End the step immediately.
291      * 
292      * @param context
293      *            current DSpace context
294      * @param request
295      *            current servlet request object
296      * @param response
297      *            current servlet response object
298      * @param subInfo
299      *            submission info object
300      * 
301      * @return true if the step is completed (no JSP was loaded), false
302      *         otherwise
303      * 
304      */
305     private boolean doStepStart(Context context, HttpServletRequest request,
306             HttpServletResponse response, SubmissionInfo subInfo)
307             throws ServletException, IOException, SQLException,
308             AuthorizeException
309     {
310         log.debug("Doing pre-processing for step " + this.getClass().getName());
311 
312         // first, do any pre-processing and get the JSP to display
313         // (assuming that this step has an interface)
314         if(stepJSPUI!=null)
315         	stepJSPUI.doPreProcessing(context, request, response, subInfo);
316 
317         // Complete this step, if this response has not already
318         // been committed.
319         //
320         // Note: the response should only be "committed" if the user
321         // has be forwarded on to a JSP page (in which case the step
322         // is not finished!).
323         if (!response.isCommitted())
324         {
325             // Otherwise, this is a non-interactive step!
326             // There's no JSP to display, so only perform the processing
327             // and forward back to the Submission Controller servlet
328 
329             log.debug("Calling processing for step "
330                     + this.getClass().getName());
331             int errorFlag = stepProcessing.doProcessing(context, request, response, subInfo);
332 
333             // if it didn't complete successfully, try and log this error!
334             if (errorFlag != AbstractProcessingStep.STATUS_COMPLETE)
335             {
336                 // see if an error message was defined!
337                 String errorMessage = stepProcessing.getErrorMessage(errorFlag);
338 
339                 // if not defined, construct a dummy error
340                 if (errorMessage == null)
341                 {
342                     errorMessage = "The doProcessing() method for "
343                             + this.getClass().getName()
344                             + " returned an error flag = "
345                             + errorFlag
346                             + ". "
347                             + "It is recommended to define a custom error message for this error flag using the addErrorMessage() method!";
348                 }
349 
350                 log.error(errorMessage);
351             }
352 
353             return completeStep(context, request, response, subInfo);
354         }
355         else
356             return false; // step not completed
357     }

360      * This method actually displays a JSP page for this "step". This method can
361      * be called from "doPostProcessing()" to display an error page, etc.
362      * 
363      * @param request
364      *            current servlet request object
365      * @param response
366      *            current servlet response object
367      * @param subInfo

373     public static final void showJSP(HttpServletRequest request,
374             HttpServletResponse response, SubmissionInfo subInfo,
375             String pathToJSP) throws ServletException, IOException,
376             SQLException
377     {
378 
379         // As long as user is not currently cancelling
380         // or saving submission, show the JSP specified
381         if (!response.isCommitted()
382                 && !SubmissionController.isCancellationInProgress(request))
383         {
384             // set beginningOfStep flag to false
385             // (so that we know it will be time for "post-processing"
386             // after loading the JSP)
387             SubmissionController.setBeginningOfStep(request, false);
388 
389             // save our current Submission information into the Request
390             // object (so JSP has access to it)
391             SubmissionController.saveSubmissionInfo(request, subInfo);
392 
393             // save our current page information to Request object
394             //setCurrentPage(request, currentPage);
395 
396             // save the JSP we are displaying
397             setLastJSPDisplayed(request, pathToJSP);
398 
399             // load JSP
400             JSPManager.showJSP(request, response, pathToJSP);
401         }

404     /**
405      * Do the end of a Step. If "cancel" or "progress bar" was pressed, we need
406      * to forward back to the SubmissionManagerServlet immediately. Otherwise,
407      * do Post-processing, and figure out if this step is truly finished or not.
408      * 
409      * @param context
410      *            current DSpace context
411      * @param request
412      *            current servlet request object
413      * @param response
414      *            current servlet response object
415      * @param subInfo
416      *            submission info object
417      * @return true if the step is completed (successfully), false otherwise
418      * 
419      */
420     private boolean doStepEnd(Context context, HttpServletRequest request,
421             HttpServletResponse response, SubmissionInfo subInfo)
422             throws ServletException, IOException, SQLException,
423             AuthorizeException
424     {
425         // we've returned from the JSP page
426         // and need to do the processing for this step
427         log.debug("Calling processing for step " + this.getClass().getName());
428 
429         int status = stepProcessing.doProcessing(context, request, response, subInfo);
430 
431         log.debug("Calling post-processing for step "
432                 + this.getClass().getName());
433 
434         // After doing the processing, we have to do any post-processing
435         // of any potential error messages, in case we need to re-display
436         // the JSP
437         stepJSPUI.doPostProcessing(context, request, response, subInfo, status);
438 
439         int currentPage = AbstractProcessingStep.getCurrentPage(request);
440         
441         // Assuming step didn't forward back to a JSP during
442         // post-processing, then check to see if this is the last page!
443         if (!response.isCommitted())
444         {
445             // check if this step has more pages!
446             if (!hasMorePages(request, subInfo, currentPage))
447             {
448                 // this step is complete! (return completion
449                 // status to SubmissionController)
450                 return completeStep(context, request, response, subInfo);
451             }
452             else
453             {
454                 // otherwise, increment to the next page
455                 AbstractProcessingStep.setCurrentPage(request, currentPage + 1);
456 
457                 // reset to beginning of step (so that pre-processing is run
458                 // again)
459                 SubmissionController.setBeginningOfStep(request, true);
460 
461                 // recursively call processStep to reload this Step class for
462                 // the next page
463                 return processStep(context, request, response, subInfo);
464             }
465         }
466         else
467         {
468             // this is the case when a response was already given
469             // in doPostProcessing(), which means the user was
470             // forwarded to a JSP page.
471             return false;
472         }
473     }

475     /**
476      * This method completes the processing of this "step" and forwards the
477      * request back to the SubmissionController (so that the next step can be
478      * called).
479      * 
480      * @param context
481      *            current DSpace context
482      * @param request
483      *            current servlet request object
484      * @param response
485      *            current servlet response object
486      * @param subInfo
487      *            submission info object
488      * 
489      * @return true if step completed (successfully), false otherwise
490      */
491     protected final boolean completeStep(Context context,
492             HttpServletRequest request, HttpServletResponse response,
493             SubmissionInfo subInfo) throws ServletException, IOException,
494             SQLException, AuthorizeException
495     {
496         log.debug("Completing Step " + this.getClass().getName());
497 
498         // If a response has not already been sent to the web browser,
499         // then forward control back to the SubmissionController
500         // (and let the controller know this step is complete!
501         //
502         // Note: The only cases in which a response would have already
503         // been sent to the web browser is if an error occurred or the
504         // Submission Process had to immediately exit for some reason
505         // (e.g. user rejected license, or tried to submit a thesis while theses
506         // are blocked, etc.)
507         if (!response.isCommitted())
508         {
509             // save our current Submission information into the Request object
510             SubmissionController.saveSubmissionInfo(request, subInfo);
511 
512             // save the current step into the Request object
513             SubmissionController.saveCurrentStepConfig(request, stepConfig);
514 
515             // reset current page back to 1
516             AbstractProcessingStep.setCurrentPage(request, 1);
517 
518             // return success (true) to controller!
519             return true;
520         }
521         else
522             return false; // couldn't return to controller since response is
523                             // committed
524     }

590     private void updatePageReached(SubmissionInfo subInfo, int page)
591             throws SQLException, AuthorizeException, IOException
592     {

594         {
595             WorkspaceItem wi = (WorkspaceItem) subInfo.getSubmissionItem();

599                 wi.setPageReached(page);
600                 wi.update();
601             }

605     /**
606      * Return the number pages for the given step in the Submission Process
607      * according to the Progress Bar. So, if "stepNum" is 7, this will return
608      * the number of pages in step #7 (at least according to the progressBar)
609      * 
610      * @param subInfo
611      *            current Submission Information object
612      * @param stepNum
613      *            the number of the step
614      * 
615      * @return the number of pages in the step
616      */
617     private static int getNumPagesInProgressBar(SubmissionInfo subInfo,
618             int stepNum)
619     {
620         // get the keys of the progressBar information (key format:
621         // stepNum.pageNum)
622         Iterator keyIterator = subInfo.getProgressBarInfo().keySet().iterator();
623 
624         // default to last page being page #1
625         int lastPage = 1;
626 
627         while (keyIterator.hasNext())
628         {
629             // find step & page info (format: stepNum.pageNum)
630             String stepAndPage = (String) keyIterator.next();
631 
632             if (stepAndPage.startsWith(stepNum + "."))
633             {
634                 // split into stepNum and pageNum
635                 String[] fields = stepAndPage.split("\\."); // split on period
636                 int page = Integer.parseInt(fields[1]);
637 
638                 if (page > lastPage)
639                 {
640                     // update last page found for this step
641                     lastPage = page;
642                 }
643             }
644         }
645 
646         // return # of last page found for this step (which is also the number
647         // of pages)
648         return lastPage;
649     }
