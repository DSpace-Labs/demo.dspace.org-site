351     /*
352      * Resolve the ID into a collection and check to see if its harvesting options are set. If so, return
353      * the collection, if not, bail out. 
354      */
355     private Collection resolveCollection(String collectionID) {
356     	
357     	DSpaceObject dso;
358     	Collection targetCollection = null;
359     	
360     	try {
361 	    	// is the ID a handle?
362 	        if (collectionID.indexOf('/') != -1)
363 	        {
364 	            // string has a / so it must be a handle - try and resolve it
365 	            dso = HandleManager.resolveToObject(context, collectionID);
366 	
367 	            // resolved, now make sure it's a collection
368 	            if (dso == null || dso.getType() != Constants.COLLECTION)
369 	                targetCollection = null;
370 	            else
371 	            	targetCollection = (Collection)dso;
372 	        }
373 	        // not a handle, try and treat it as an integer collection
374 	        // database ID
375 	        else if (collectionID != null)
376 	        {
377 	        	System.out.println("Looking up by id: " + collectionID + ", parsed as '" + Integer.parseInt(collectionID) + "', " + "in context: " + context);
378 	            targetCollection = Collection.find(context, Integer.parseInt(collectionID));
379 	        }
380 	        // was the collection valid?
381 	        if (targetCollection == null)
382 	        {
383 	        	System.out.println("Cannot resolve " + collectionID + " to collection");
384 	            System.exit(1);
385 	        }
386     	}
387     	catch (SQLException se) {
388     		se.printStackTrace();
389     	}
390     	
391     	return targetCollection;
392     }

396     	System.out.println("Running: configure collection");
397    	
398     	Collection collection = resolveCollection(collectionID);

401     	try {
402     		HarvestedCollection hc = HarvestedCollection.find(context, collection.getID());
403         	if (hc == null) {
404         		hc = HarvestedCollection.create(context, collection.getID());

406

416     		e.printStackTrace();

425     /**
426      * Purges a collection of all harvest-related data and settings. All items in the collection will be deleted.
427      * 
428      * @param collectionID
429      * @param email
430      */
431     private void purgeCollection(String collectionID, String email) {
432     	System.out.println("Purging collection of all items and reseting last_harvested and harvest_message: " + collectionID);
433     	Collection collection = resolveCollection(collectionID);
434    	
435     	try 
436     	{
437     		EPerson eperson = EPerson.findByEmail(context, email);
438         	context.setCurrentUser(eperson);
439     		context.turnOffAuthorisationSystem();
440     		
441     		ItemIterator it = collection.getAllItems();
442     		IndexBrowse ib = new IndexBrowse(context);
443     		int i=0;
444     		while (it.hasNext()) {
445     			i++;
446     			Item item = it.next();
447     			System.out.println("Deleting: " + item.getHandle());
448     			ib.itemRemoved(item);
449     			collection.removeItem(item);
450     			// commit every 50 items
451     			if (i%50 == 0) {
452     				context.commit();
453     				i=0;
454     			}
455     		}
456     		
457     		HarvestedCollection hc = HarvestedCollection.find(context, collection.getID());
458     		if (hc != null) {
459 	    		hc.setHarvestResult(null,"");
460 	    		hc.setHarvestStatus(HarvestedCollection.STATUS_READY);
461 	    		hc.setHarvestStartTime(null);
462 	    		hc.update();
463     		}
464     		context.restoreAuthSystemState();    		
465     		context.commit();
466     	} 
467     	catch (Exception e) {
468     		System.out.println("Changes could not be committed");
469     		e.printStackTrace();
470     		System.exit(1);
471     	}
472     	finally {
473     		context.restoreAuthSystemState();
474     	}
475     }

478     /**
479      * Run a single harvest cycle on the specified collection under the authorization of the supplied EPerson 
480      */
481     private void runHarvest(String collectionID, String email) {
482     	System.out.println("Running: a harvest cycle on " + collectionID);
483     	
484     	System.out.print("Initializing the harvester... ");
485     	OAIHarvester harvester = null;
486     	try {
487     		Collection collection = resolveCollection(collectionID);
488         	HarvestedCollection hc = HarvestedCollection.find(context, collection.getID());
489     		harvester = new OAIHarvester(context, collection, hc);
490     		System.out.println("success. ");
491     	}
492     	catch (HarvestingException hex) {
493     		System.out.print("failed. ");
494     		System.out.println(hex.getMessage());
495     		System.exit(1);
496     	} catch (SQLException se) {
497 			// TODO Auto-generated catch block
498 			se.printStackTrace();
499 		}
500     	    	
501     	try {
502     		// Harvest will not work for an anonymous user
503         	EPerson eperson = EPerson.findByEmail(context, email);
504         	System.out.println("Harvest started... ");
505         	context.setCurrentUser(eperson);
506     		harvester.runHarvest();
507     		context.complete();
508     	}
509     	catch (Exception e) {
510 			// Not much else we can do at this point
511 			e.printStackTrace();
512 			System.exit(1);
513 		}
514     	System.out.println("Harvest complete. ");
515     }
