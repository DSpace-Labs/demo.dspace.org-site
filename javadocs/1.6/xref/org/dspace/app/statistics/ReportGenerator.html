179     /**
180      * main method to be run from command line.  See usage information for
181      * details as to how to use the command line flags
182      */
183     public static void main(String [] argv)
184         throws Exception, SQLException
185     {
186         // create context as super user
187         Context context = new Context();
188         context.setIgnoreAuthorization(true);
189         
190         String myFormat = null;
191         String myInput = null;
192         String myOutput = null;
193         String myMap = null;
194         
195         // read in our command line options
196         for (int i = 0; i < argv.length; i++)
197         {
198             if (argv[i].equals("-format"))
199             {
200                 myFormat = argv[i+1].toLowerCase();
201             }
202             
203             if (argv[i].equals("-in"))
204             {
205                 myInput = argv[i+1];
206             }
207             
208             if (argv[i].equals("-out"))
209             {
210                 myOutput = argv[i+1];
211             }
212             
213             if (argv[i].equals("-map"))
214             {
215                 myMap = argv[i+1];
216             }
217             
218             if (argv[i].equals("-help"))
219             {
220                 usage();
221                 System.exit(0);
222             }
223         }
224 
225         processReport(context, myFormat, myInput, myOutput, myMap);
226     }

228     /**
229      * using the pre-configuration information passed here, read in the 
230      * aggregation data and output a file containing the report in the
231      * requested format
232      *
233      * this method is retained for backwards compatibility, but delegates the actual
234      * wprk to a new method
235      *
236      * @param   context     the DSpace context in which this action is performed
237      * @param   myFormat    the desired output format (currently on HTML supported)
238      * @param   myInput     the aggregation file to be turned into a report
239      * @param   myOutput    the file into which to write the report
240      */
241     public static void processReport(Context context, String myFormat, 
242                                      String myInput, String myOutput,
243                                      String myMap)
244         throws Exception, SQLException
245     {
246         // create the relevant report type
247         // FIXME: at the moment we only support HTML report generation
248         Report report = null;
249         if (myFormat.equals("html"))
250         {
251             report = new HTMLReport();
252             ((HTMLReport)report).setOutput(myOutput);
253         }
254 
255         if (myMap != null)
256         {
257             map = myMap;
258         }
259 
260         ReportGenerator.processReport(context, report, myInput);
261     }

263     /**
264      * using the pre-configuration information passed here, read in the
265      * aggregation data and output a file containing the report in the
266      * requested format
267      */
268     public static void processReport(Context context, Report report,
269                                      String myInput)
270         throws Exception, SQLException
271     {
272         startTime = new GregorianCalendar();
273              
274         /** instantiate aggregators */
275         actionAggregator = new HashMap();
276         searchAggregator = new HashMap();
277         userAggregator = new HashMap();
278         itemAggregator = new HashMap();
279         archiveStats = new HashMap();
280         actionMap = new HashMap();
281         
282         /** instantite lists */
283         generalSummary = new ArrayList();
284                 
285         // set the parameters for this analysis
286         setParameters(myInput);
287         
288         // pre prepare our standard file readers and buffered readers
289         FileReader fr = null;
290         BufferedReader br = null;
291         
292         // read the input file
293         readInput(input);
294         
295         // load the log file action to human readable action map
296         readMap(map);
297         
298         report.setStartDate(startDate);
299         report.setEndDate(endDate);
300         report.setMainTitle(name, serverName);
301         
302         // define our standard variables for re-use
303         // FIXME: we probably don't need these once we've finished re-factoring
304         Iterator keys = null;
305         int i = 0;
306         String explanation = null;
307         int value;
308         
309         // FIXME: All of these sections should probably be buried in their own
310         // custom methods
311         
312         Statistics overview = new Statistics();
313         
314         overview.setSectionHeader("General Overview");
315         
316         Iterator summaryEntries = generalSummary.iterator();
317         while (summaryEntries.hasNext())
318         {
319             String entry = (String) summaryEntries.next();
320             if (actionAggregator.containsKey(entry))
321             {
322                 int count = Integer.parseInt((String) actionAggregator.get(entry));
323                 overview.add(new Stat(translate(entry), count));
324             }
325         }
326         
327         report.addBlock(overview);
328         
329         // prepare the archive statistics package
330         if (archiveStats.size() > 0)
331         {
332             Statistics archiveInfo = prepareStats(archiveStats, true, false);
333             archiveInfo.setSectionHeader("Archive Information");
334             archiveInfo.setStatName("Content Type");
335             archiveInfo.setResultName("Number of items");
336         
337             report.addBlock(archiveInfo);
338         }
339         
340         
341         
342         // process the items in preparation to be displayed.  This includes sorting
343         // by view number, building the links, and getting further info where
344         // necessary
345         Statistics viewedItems = new Statistics("Item/Handle", "Number of views", itemFloor);
346         viewedItems.setSectionHeader("Items Viewed");
347         
348         Stat[] items = new Stat[itemAggregator.size()];
349         
350         keys = itemAggregator.keySet().iterator();
351         i = 0;
352         while (keys.hasNext())
353         {
354             String key = (String) keys.next();
355             String link = url + "handle/" + key;
356             value = Integer.parseInt((String) itemAggregator.get(key));
357             items[i] = new Stat(key, value, link);
358             i++;
359         }
360         
361         Arrays.sort(items);
362         
363         String info = null;
364         for (i = 0; i < items.length; i++)
365         {
366             // Allow negative value to say that all items should be looked up
367             if (itemLookup < 0 || i < itemLookup)
368             {
369                 info = getItemInfo(context, items[i].getKey());
370             }
371                
372             // if we get something back from the db then set it as the key,
373             // else just use the link
374             if (info != null)
375             {
376                 items[i].setKey(info  + " (" + items[i].getKey() + ")");
377             }
378             else
379             {
380                 items[i].setKey(items[i].getReference());
381             }
382             
383             // reset the info register
384             info = null;
385         }
386         
387         viewedItems.add(items);
388         
389         report.addBlock(viewedItems);
390         
391         // prepare a report of the full action statistics
392         Statistics fullInfo = prepareStats(actionAggregator, true, true);
393         fullInfo.setSectionHeader("All Actions Performed");
394         fullInfo.setStatName("Action");
395         fullInfo.setResultName("Number of times");
396         
397         report.addBlock(fullInfo);
398         
399         // prepare the user login statistics package
400         if (!userEmail.equals("off"))
401         {
402             Statistics userLogins = prepareStats(userAggregator, true, false);
403             userLogins.setSectionHeader("User Logins");
404             userLogins.setStatName("User");
405             userLogins.setResultName("Number of logins");
406             if (userEmail.equals("alias"))
407             {
408                 explanation = "(distinct addresses)";
409                 userLogins.setExplanation(explanation);
410             }
411         
412             report.addBlock(userLogins);
413         }
414 
415         // prepare the search word statistics package
416         Statistics searchWords = prepareStats(searchAggregator, true, false);
417         searchWords.setSectionHeader("Words Searched");
418         searchWords.setStatName("Word");
419         searchWords.setResultName("Number of searches");
420         searchWords.setFloor(searchFloor);
421         
422         report.addBlock(searchWords);
423         
424         // FIXME: because this isn't an aggregator it can't be passed to
425         // prepareStats; should we overload this method for use with this kind
426         // of data?
427         // prepare the average item views statistics
428         if (avgItemViews > 0)
429         {
430             Statistics avg = new Statistics();
431             avg.setSectionHeader("Averaging Information");
432 
433             Stat[] average = new Stat[1];
434         
435             average[0] = new Stat("Average views per item", avgItemViews);
436             avg.add(average);
437             report.addBlock(avg);
438         }
439       
440         
441         // prepare the log line level statistics
442         // FIXME: at the moment we only know about warnings, but future versions
443         // should aggregate all log line levels and display here
444         Statistics levels = new Statistics("Level", "Number of lines");
445         levels.setSectionHeader("Log Level Information");
446         
447         Stat[] level = new Stat[1];
448         level[0] = new Stat("Warnings", warnings);
449         
450         levels.add(level);
451         
452         report.addBlock(levels);
453         
454         // get the display processing time information
455         Calendar endTime = new GregorianCalendar();
456         long timeInMillis = (endTime.getTimeInMillis() - startTime.getTimeInMillis());
457         int outputProcessTime = (new Long(timeInMillis).intValue() / 1000);
458         
459         // prepare the processing information statistics
460         Statistics process = new Statistics("Operation", "");
461         process.setSectionHeader("Processing Information");
462         
463         Stat[] proc = new Stat[3];
464         proc[0] = new Stat("Log Processing Time", processTime);
465         proc[0].setUnits("seconds");
466         proc[1] = new Stat("Output Processing Time", outputProcessTime);
467         proc[1].setUnits("seconds");
468         proc[2] = new Stat("Log File Lines Analysed", logLines);
469         proc[2].setUnits("lines");
470         
471         process.add(proc);
472         
473         report.addBlock(process);
474 
475         report.render();
476         
477         return;
478     }

481     /**
482      * a standard stats block preparation method for use when an aggregator
483      * has to be put out in its entirity.  This method will not be able to
484      * deal with complex cases, although it will perform sorting by value and
485      * translations as per the map file if requested
486      *
487      * @param   aggregator      the aggregator that should be converted
488      * @param   sort            should the resulting stats be sorted by value
489      * @param   translate       translate the stat name using the map file
490      *
491      * @return      a Statistics object containing all the relevant information
492      */
493     public static Statistics prepareStats(Map aggregator, boolean sort, boolean translate)
494     {
495         Stat[] stats = new Stat[aggregator.size()];
496         if (aggregator.size() > 0)
497         {
498             Iterator keys = aggregator.keySet().iterator();
499             int i = 0;
500             while (keys.hasNext())
501             {
502                 String key = (String) keys.next();
503                 int value = Integer.parseInt((String) aggregator.get(key));
504                 if (translate)
505                 {
506                     stats[i] = new Stat(translate(key), value);
507                 }
508                 else
509                 {
510                     stats[i] = new Stat(key, value);
511                 }
512                 i++;
513             }
514             
515             if (sort)
516             {
517                 Arrays.sort(stats);
518             }
519         }
520         
521         // add the results to the statistics object
522         Statistics statistics = new Statistics();
523         statistics.add(stats);
524         
525         return statistics;
526     }

531      * value if one exists.  If no translation exists the original text is

534      * @param   text    the text to be translated

536      * @return      a string containing either the translated text or the original

539     public static String translate(String text)
540     {
541         if (actionMap.containsKey(text))
542         {
543             return (String) actionMap.get(text);
544         }
545         else
546         {
547             return text;
548         }
549     }

552     /**
553      * read in the action map file which converts log file line actions into
554      * actions which are more understandable to humans
555      *
556      * @param   map     the map file
557      */
558     public static void readMap(String map)
559         throws IOException
560     {
561         FileReader fr = null;
562         BufferedReader br = null;
563 
564         try
565         {
566             // read in the map file, printing a warning if none is found
567             String record = null;
568             try
569             {
570                 fr = new FileReader(map);
571                 br = new BufferedReader(fr);
572             }
573             catch (IOException e)
574             {
575                 System.err.println("Failed to read map file: log file actions will be displayed without translation");
576                 return;
577             }
578 
579             // loop through the map file and read in the values
580             while ((record = br.readLine()) != null)
581             {
582                 Matcher matchReal = real.matcher(record);
583 
584                 // if the line is real then read it in
585                 if (matchReal.matches())
586                 {
587                     actionMap.put(matchReal.group(1).trim(), matchReal.group(2).trim());
588                 }
589             }
590         }
591         finally
592         {
593             if (br != null)
594                 try { br.close(); } catch (IOException ioe) { }
595 
596             if (fr != null)
597                 try { fr.close(); } catch (IOException ioe) { }
598         }
599     }

621     /**
622      * read the input file and populate all the class globals with the contents
623      * The values that come from this file form the basis of the analysis report
624      *
625      * @param   input   the aggregator file
626      */
627     public static void readInput(String input)
628         throws IOException, ParseException
629     {
630         FileReader fr = null;
631         BufferedReader br = null;
632         
633         // read in the analysis information, throwing an error if we fail to open
634         // the given file
635         String record = null;
636         try 
637         {  
638             fr = new FileReader(input);
639             br = new BufferedReader(fr);
640         } 
641         catch (IOException e) 
642         {  
643             System.out.println("Failed to read input file: " + input);
644             return;
645         } 
646         
647         // first initialise a date format object to do our date processing
648         // if necessary
649         SimpleDateFormat sdf = new SimpleDateFormat("dd'/'MM'/'yyyy");
650 
651         // FIXME: although this works, it is not very elegant
652         // loop through the aggregator file and read in the values
653         while ((record = br.readLine()) != null) 
654         {
655             // match real lines
656             Matcher matchReal = real.matcher(record);
657             
658             // pre-prepare our input strings
659             String section = null;
660             String key = null;
661             String value = null;
662             
663             // temporary string to hold the left hand side of the equation
664             String left = null;
665             
666             // match the line or skip this record
667             if (matchReal.matches())
668             {
669                 // lift the values out of the matcher's result groups
670                 left = matchReal.group(1).trim();
671                 value = matchReal.group(2).trim();
672                 
673                 // now analyse the left hand side, splitting by ".", taking the
674                 // first token as the section and the remainder of the string
675                 // as they key if it exists
676                 StringTokenizer tokens = new StringTokenizer(left, ".");
677                 int numTokens = tokens.countTokens();
678                 if (tokens.hasMoreTokens())
679                 {
680                     section = tokens.nextToken();
681                     if (numTokens > 1)
682                     {
683                         key = left.substring(section.length() + 1);
684                     }
685                     else
686                     {
687                         key = "";
688                     }
689                 }
690             }
691             else
692             {
693                 continue;
694             }
695             
696             // if the line is real, then we carry on
697             // read the analysis contents in
698             if (section.equals("archive"))
699             {
700                 archiveStats.put(key, value);
701             }
702             else if (section.equals("action"))
703             {
704                 actionAggregator.put(key, value);
705             }
706             else if (section.equals("user"))
707             {
708                 userAggregator.put(key, value);
709             }
710             else if (section.equals("search"))
711             {
712                 searchAggregator.put(key, value);
713             }
714             else if (section.equals("item"))
715             {
716                 itemAggregator.put(key, value);
717             }
718             else if (section.equals("user_email"))
719             {
720                 userEmail = value;
721             }
722             else if (section.equals("item_floor"))
723             {
724                 itemFloor = Integer.parseInt(value);
725             }
726             else if (section.equals("search_floor"))
727             {
728                 searchFloor = Integer.parseInt(value);
729             }
730             else if (section.equals("host_url"))
731             {
732                 url = value;
733             }
734             else if (section.equals("item_lookup"))
735             {
736                 itemLookup = Integer.parseInt(value);
737             }
738             else if (section.equals("avg_item_views"))
739             {
740                 try 
741                 {
742                     avgItemViews = Integer.parseInt(value);
743                 }
744                 catch (NumberFormatException e)
745                 {
746                     avgItemViews = 0;
747                 }
748             }
749             else if (section.equals("server_name"))
750             {
751                 serverName = value;
752             }
753             else if (section.equals("service_name"))
754             {
755                 name = value;
756             }
757             else if (section.equals("start_date"))
758             {
759                 startDate = sdf.parse(value);
760             }
761             else if (section.equals("end_date"))
762             {
763                 endDate = sdf.parse(value);
764             }
765             else if (section.equals("analysis_process_time"))
766             {
767                 processTime = Integer.parseInt(value);
768             }
769             else if (section.equals("general_summary"))
770             {
771                 generalSummary.add(value);
772             }
773             else if (section.equals("log_lines"))
774             {
775                 logLines = Integer.parseInt(value);
776             }
777             else if (section.equals("warnings"))
778             {
779                 warnings = Integer.parseInt(value);
780             }
781         }
782 
783         // close the inputs
784         br.close();
785         fr.close();
786     }

788     /**
789      * get the information for the item with the given handle
790      *
791      * @param   context     the DSpace context we are operating under
792      * @param   handle      the handle of the item being looked up, in the form
793      *                      1234/567 and so forth
794      *
795      * @return      a string containing a reference (almost citation) to the
796      *              article
797      */
798     public static String getItemInfo(Context context, String handle)
799         throws SQLException
800     {
801         Item item = null;
802         
803         // ensure that the handle exists
804         try 
805         {
806             item = (Item) HandleManager.resolveToObject(context, handle);
807         } 
808         catch (Exception e)
809         {
810             return null;
811         }
812         
813         // if no handle that matches is found then also return null
814         if (item == null)
815         {
816             return null;
817         }
818         
819         // build the referece
820         // FIXME: here we have blurred the line between content and presentation
821         // and it should probably be un-blurred
822         DCValue[] title = item.getDC("title", null, Item.ANY);
