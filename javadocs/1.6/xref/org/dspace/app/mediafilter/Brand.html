52  /**
53   * Class to attach a footer to an image using ImageMagick.
54   * Thanks Ninh Nguyen from the National Library of Australia for providing the source code.
55   * This version of the code is basically Ninh's but reorganised a little. Used with permission.
56   */
57  
58  public class Brand
59  {
60  	private int brandWidth;
61  	private int brandHeight;
62  	private Font font;
63  	private int xOffset;
64  
65  	/**
66  	 * Constructor to set up footer image attributes.
67  	 *
68  	 * @param footerWidth length of the footer in pixels
69  	 * @param footerHeight height of the footer in pixels
70  	 * @param font font to use for text on the footer
71  	 * @param xOffset number of pixels text should be indented from left-hand side of footer
72  	 *
73  	 */
74  	public Brand(int brandWidth,
75  		      int brandHeight,
76  		      Font font,
77  		      int xOffset)
78  	{
79  		this.brandWidth = brandWidth;
80  		this.brandHeight = brandHeight;
81  		this.font = font;
82  		this.xOffset = xOffset;
83  	}
84  
85  	/**
86  	 * Create the brand image
87  	 *
88  	 * @param brandLeftText text that should appear in the bottom left of the image
89  	 * @param shortLeftText abbreviated form of brandLeftText that will be substituted if
90  	 *	the image is resized such that brandLeftText will not fit. <code>null</code> if not
91  	 *	required
92  	 * @param brandRightText text that should appear in the bottom right of the image
93  	 *
94  	 * @return BufferedImage a BufferedImage object describing the brand image file
95  	 */
96  	public BufferedImage create(String brandLeftText,
97  			   String shortLeftText,
98  			   String brandRightText)
99  	{
100 		BrandText[] allBrandText = null;
101 
102 		BufferedImage brandImage =
103 			new BufferedImage(brandWidth, brandHeight, BufferedImage.TYPE_INT_RGB);
104 
105 		if (brandWidth >= 350)
106 		{
107 			allBrandText = new BrandText[]
108 			{
109 				new BrandText(BrandText.BL, brandLeftText),
110 				new BrandText(BrandText.BR, brandRightText)
111 			};
112 		} 
113 		else if (brandWidth >= 190)
114 		{
115 			allBrandText = new BrandText[]
116 			{
117 				new BrandText(BrandText.BL, shortLeftText),
118 				new BrandText(BrandText.BR, brandRightText)
119 			};
120 		}
121 		else
122 		{
123 			allBrandText = new BrandText[]
124 			{
125 				new BrandText(BrandText.BR, brandRightText)
126 			};
127 		}
128 
129 		if (allBrandText != null && allBrandText.length > 0)
130 		{
131       		for (int i = 0; i < allBrandText.length; ++i)
132 			{
133         		drawImage(brandImage, allBrandText[i]);
134       		}
135     	}
136 
137 		return brandImage;
138 	}
139 
140 
141 	/**
142 	 * do the text placements and preparatory work for the brand image generation
143 	 *
144 	 * @param brandImage a BufferedImage object where the image is created
145 	 * @param identifier and Identifier object describing what text is to be placed in what
146 	 *	position within the brand
147 	 */
148 	private void drawImage(BufferedImage brandImage,
149 			BrandText brandText)
150 	{
151 		int imgWidth = brandImage.getWidth();
152 		int imgHeight = brandImage.getHeight();
153 
154 		int bx, by, tx, ty, bWidth, bHeight;
155 
156 		Graphics2D g2 = brandImage.createGraphics();
157 		g2.setFont(font);
158 		FontMetrics fm = g2.getFontMetrics();
159 
160 
161 		bWidth = fm.stringWidth(brandText.getText()) + xOffset * 2 + 1;
162 		bHeight = fm.getHeight();
163 
164 		bx = by = 0;
165 
166 		if (brandText.getLocation().equals(BrandText.TL))
167 		{
168 			bx = 0;
169 			by = 0;
170 		}
171 		else if (brandText.getLocation().equals(BrandText.TR))
172 		{
173 			bx = imgWidth - bWidth;
174 			by = 0;
175 		}
176 		else if (brandText.getLocation().equals(BrandText.BL))
177 		{
178 			bx = 0;
179 			by = imgHeight - bHeight;
180 		}
181 		else if (brandText.getLocation().equals(BrandText.BR))
182 		{
183 			bx = imgWidth - bWidth;
184 			by = imgHeight - bHeight;
185 		}
186 
187 		Rectangle box = new Rectangle(bx, by, bWidth, bHeight);
188 		tx = bx + xOffset;
189 		ty = by + fm.getAscent();
190 
191     	g2.setColor(Color.black);
192 		g2.fill(box);
193 		g2.setColor(Color.white);
194 		g2.drawString(brandText.getText(), tx, ty);
195 	}
