38  package org.dspace.app.mediafilter;

40  import org.dspace.content.Bitstream;
41  import org.dspace.content.Item;
42  import org.dspace.core.Context;

45  /**
46   * Abstract class which defines the default settings for a *simple* Media or Format Filter. 
47   * This class may be extended by any class which wishes to define a simple filter to be run 
48   * by the MediaFilterManager.  More complex filters should likely implement the FormatFilter
49   * interface directly, so that they can define their own pre/postProcessing methods.
50   */
51  public abstract class MediaFilter implements FormatFilter
52  {
53  	/**
54       * Perform any pre-processing of the source bitstream *before* the actual 
55       * filtering takes place in MediaFilterManager.processBitstream().
56       * <p>
57       * Return true if pre-processing is successful (or no pre-processing
58       * is necessary).  Return false if bitstream should be skipped
59       * for any reason.
60       * 
61       * 
62       * @param c
63       *            context
64       * @param item
65       *            item containing bitstream to process
66       * @param source
67       *            source bitstream to be processed
68       * 
69       * @return true if bitstream processing should continue, 
70       *          false if this bitstream should be skipped
71       */
72      public boolean preProcessBitstream(Context c, Item item, Bitstream source)
73              throws Exception
74      {
75          return true;  //default to no pre-processing
76      }
77       
78      /**
79       * Perform any post-processing of the generated bitstream *after* this
80       * filter has already been run.
81       * <p>
82       * Return true if pre-processing is successful (or no pre-processing
83       * is necessary).  Return false if bitstream should be skipped
84       * for some reason.
85       * 
86       * 
87       * @param c
88       *            context
89       * @param item
90       *            item containing bitstream to process
91       * @param generatedBitstream
92       *            the bitstream which was generated by
93       *            this filter.
94       */
95      public void postProcessBitstream(Context c, Item item, Bitstream generatedBitstream)
96              throws Exception
97      {
98          //default to no post-processing necessary
99      }
