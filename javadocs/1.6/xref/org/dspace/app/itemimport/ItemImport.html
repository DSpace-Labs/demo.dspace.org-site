83  /**
84   * Import items into DSpace. The conventional use is upload files by copying
85   * them. DSpace writes the item's bitstreams into its assetstore. Metadata is
86   * also loaded to the DSpace database.
87   * <P>
88   * A second use assumes the bitstream files already exist in a storage
89   * resource accessible to DSpace. In this case the bitstreams are 'registered'.
90   * That is, the metadata is loaded to the DSpace database and DSpace is given
91   * the location of the file which is subsumed into DSpace.
92   * <P>
93   * The distinction is controlled by the format of lines in the 'contents' file.
94   * See comments in processContentsFile() below.
95   * <P>
96   * Modified by David Little, UCSD Libraries 12/21/04 to
97   * allow the registration of files (bitstreams) into DSpace.
98   */

535         }
536 
537         if (mapOut != null)
538         {
539             mapOut.close();
540         }
541

545         }

549     private void addItems(Context c, Collection[] mycollections,
550             String sourceDir, String mapFile, boolean template) throws Exception
551     {
552         Map skipItems = new HashMap(); // set of items to skip if in 'resume'
553         // mode
554 
555         System.out.println("Adding items from directory: " + sourceDir);
556         System.out.println("Generating mapfile: " + mapFile);
557 
558         // create the mapfile
559         File outFile = null;
560 
561         if (!isTest)
562         {
563             // get the directory names of items to skip (will be in keys of
564             // hash)
565             if (isResume)
566             {
567                 skipItems = readMapFile(mapFile);
568             }
569 
570             // sneaky isResume == true means open file in append mode
571             outFile = new File(mapFile);
572             mapOut = new PrintWriter(new FileWriter(outFile, isResume));
573 
574             if (mapOut == null)
575             {
576                 throw new Exception("can't open mapfile: " + mapFile);
577             }
578         }
579 
580         // open and process the source directory
581         File d = new java.io.File(sourceDir);
582 
583         if (d == null)
584         {
585             System.out.println("Error, cannot open source directory "
586                     + sourceDir);
587             System.exit(1);
588         }
589 
590         String[] dircontents = d.list(directoryFilter);
591 
592         for (int i = 0; i < dircontents.length; i++)
593         {
594             if (skipItems.containsKey(dircontents[i]))
595             {
596                 System.out.println("Skipping import of " + dircontents[i]);
597             }
598             else
599             {
600                 addItem(c, mycollections, sourceDir, dircontents[i], mapOut, template);
601                 System.out.println(i + " " + dircontents[i]);
602             }
603         }
604     }

606     private void replaceItems(Context c, Collection[] mycollections,
607             String sourceDir, String mapFile, boolean template) throws Exception
608     {
609         // verify the source directory
610         File d = new java.io.File(sourceDir);
611 
612         if (d == null)
613         {
614             System.out.println("Error, cannot open source directory "
615                     + sourceDir);
616             System.exit(1);
617         }
618 
619         // read in HashMap first, to get list of handles & source dirs
620         Map myhash = readMapFile(mapFile);
621 
622         // for each handle, re-import the item, discard the new handle
623         // and re-assign the old handle
624         Iterator i = myhash.keySet().iterator();
625         ArrayList itemsToDelete = new ArrayList();
626 
627         while (i.hasNext())
628         {
629             // get the old handle
630             String newItemName = (String) i.next();
631             String oldHandle = (String) myhash.get(newItemName);
632 
633             Item oldItem = null;
634             Item newItem = null;
635 
636             if (oldHandle.indexOf('/') != -1)
637             {
638                 System.out.println("\tReplacing:  " + oldHandle);
639 
640                 // add new item, locate old one
641                 oldItem = (Item) HandleManager.resolveToObject(c, oldHandle);
642             }
643             else
644             {
645                 oldItem = Item.find(c, Integer.parseInt(oldHandle));
646             }
647             
648             /* Rather than exposing public item methods to change handles -- 
649              * two handles can't exist at the same time due to key constraints
650              * so would require temp handle being stored, old being copied to new and
651              * new being copied to old, all a bit messy -- a handle file is written to
652              * the import directory containing the old handle, the existing item is 
653              * deleted and then the import runs as though it were loading an item which 
654              * had already been assigned a handle (so a new handle is not even assigned).
655              * As a commit does not occur until after a successful add, it is safe to 
656              * do a delete as any error results in an aborted transaction without harming
657              * the original item */
658             File handleFile = new File(sourceDir + File.separatorChar + newItemName + File.separatorChar + "handle");
659             PrintWriter handleOut = new PrintWriter(new FileWriter(handleFile, true));
660 
661             if (handleOut == null)
662             {
663                 throw new Exception("can't open handle file: " + handleFile.getCanonicalPath());
664             }
665             
666             handleOut.println(oldHandle);
667             handleOut.close();
668             
669             deleteItem(c, oldItem);
670             
671             newItem = addItem(c, mycollections, sourceDir, newItemName, null, template);
672         }
673     }

675     private void deleteItems(Context c, String mapFile) throws Exception
676     {
677         System.out.println("Deleting items listed in mapfile: " + mapFile);
678 
679         // read in the mapfile
680         Map myhash = readMapFile(mapFile);
681 
682         // now delete everything that appeared in the mapFile
683         Iterator i = myhash.keySet().iterator();
684 
685         while (i.hasNext())
686         {
687             String itemID = (String) myhash.get(i.next());
688 
689             if (itemID.indexOf('/') != -1)
690             {
691                 String myhandle = itemID;
692                 System.out.println("Deleting item " + myhandle);
693                 deleteItem(c, myhandle);
694             }
695             else
696             {
697                 // it's an ID
698                 Item myitem = Item.find(c, Integer.parseInt(itemID));
699                 System.out.println("Deleting item " + itemID);
700                 deleteItem(c, myitem);
701             }
702         }
703     }

705     /**
706      * item? try and add it to the archive c mycollection path itemname handle -
707      * non-null means we have a pre-defined handle already mapOut - mapfile
708      * we're writing
709      */
710     private Item addItem(Context c, Collection[] mycollections, String path,
711             String itemname, PrintWriter mapOut, boolean template) throws Exception
712     {
713         String mapOutput = null;
714 
715         System.out.println("Adding item from directory " + itemname);
716 
717         // create workspace item
718         Item myitem = null;
719         WorkspaceItem wi = null;
720 
721         if (!isTest)
722         {
723             wi = WorkspaceItem.create(c, mycollections[0], template);
724             myitem = wi.getItem();
725         }
726 
727         // now fill out dublin core for item
728         loadMetadata(c, myitem, path + File.separatorChar + itemname
729                 + File.separatorChar);
730 
731         // and the bitstreams from the contents file
732         // process contents file, add bistreams and bundles, return any
733         // non-standard permissions
734         Vector options = processContentsFile(c, myitem, path
735                 + File.separatorChar + itemname, "contents");
736 
737         if (useWorkflow)
738         {
739             // don't process handle file
740             // start up a workflow
741             if (!isTest)
742             {
743                 WorkflowManager.startWithoutNotify(c, wi);
744 
745                 // send ID to the mapfile
746                 mapOutput = itemname + " " + myitem.getID();
747             }
748         }
749         else
750         {
751             // only process handle file if not using workflow system
752             String myhandle = processHandleFile(c, myitem, path
753                     + File.separatorChar + itemname, "handle");
754 
755             // put item in system
756             if (!isTest)
757             {
758                 InstallItem.installItem(c, wi, myhandle);
759 
760                 // find the handle, and output to map file
761                 myhandle = HandleManager.findHandle(c, myitem);
762 
763                 mapOutput = itemname + " " + myhandle;
764             }
765 
766             // set permissions if specified in contents file
767             if (options.size() > 0)
768             {
769                 System.out.println("Processing options");
770                 processOptions(c, myitem, options);
771             }
772         }
773 
774         // now add to multiple collections if requested
775         if (mycollections.length > 1)
776         {
777             for (int i = 1; i < mycollections.length; i++)
778             {
779                 if (!isTest)
780                 {
781                     mycollections[i].addItem(myitem);
782                 }
783             }
784         }
785 
786         // made it this far, everything is fine, commit transaction
787         if (mapOut != null)
788         {
789             mapOut.println(mapOutput);
790         }
791 
792         c.commit();
793 
794         return myitem;
795     }

1395      * 
1396      * Process the Options to apply to the Item. The options are tab delimited
1397      * 
1398      * Options:
1399      *      48217870-MIT.pdf        permissions: -r 'MIT Users'     description: Full printable version (MIT only)
1400      *      permissions:[r|w]-['group name']
1401      *      description: 'the description of the file'
1402      *      
1403      *      where:
1404      *          [r|w] (meaning: read|write)
1405      *          ['MIT Users'] (the group name)
1406      *          
1407      * @param c
1408      * @param myItem
1409      * @param options
1410      * @throws SQLException
1411      * @throws AuthorizeException
1412      */
1413     private void processOptions(Context c, Item myItem, Vector options)
1414             throws SQLException, AuthorizeException
1415     {
1416         for (int i = 0; i < options.size(); i++)
1417         {
1418             String line = options.elementAt(i).toString();
1419 
1420             System.out.println("\tprocessing " + line);
1421 
1422             boolean permissionsExist = false;
1423             boolean descriptionExists = false;
1424 
1425             String permissionsMarker = "\tpermissions:";
1426             int pMarkerIndex = line.indexOf(permissionsMarker);
1427             int pEndIndex = 0;
1428             if (pMarkerIndex > 0)
1429             {
1430                 pEndIndex = line.indexOf("\t", pMarkerIndex + 1);
1431                 if (pEndIndex == -1)
1432                 {
1433                     pEndIndex = line.length();
1434                 }
1435                 permissionsExist = true;
1436             }
1437 
1438             String descriptionMarker = "\tdescription:";
1439             int dMarkerIndex = line.indexOf(descriptionMarker);
1440             int dEndIndex = 0;
1441             if (dMarkerIndex > 0)
1442             {
1443                 dEndIndex = line.indexOf("\t", dMarkerIndex + 1);
1444                 if (dEndIndex == -1)
1445                 {
1446                     dEndIndex = line.length();
1447                 }
1448                 descriptionExists = true;
1449             }
1450 
1451             int bsEndIndex = line.indexOf("\t");
1452             String bitstreamName = line.substring(0, bsEndIndex);
1453 
1454             int actionID = -1;
1455             String groupName = "";
1456             Group myGroup = null;
1457             if (permissionsExist)
1458             {
1459                 String thisPermission = line.substring(pMarkerIndex
1460                         + permissionsMarker.length(), pEndIndex);
1461 
1462                 // get permission type ("read" or "write")
1463                 int pTypeIndex = thisPermission.indexOf("-");
1464 
1465                 // get permission group (should be in single quotes)
1466                 int groupIndex = thisPermission.indexOf("'", pTypeIndex);
1467                 int groupEndIndex = thisPermission.indexOf("'", groupIndex + 1);
1468 
1469                 // if not in single quotes, assume everything after type flag is
1470                 // group name
1471                 if (groupIndex == -1)
1472                 {
1473                     groupIndex = thisPermission.indexOf(" ", pTypeIndex);
1474                     groupEndIndex = thisPermission.length();
1475                 }
1476 
1477                 groupName = thisPermission.substring(groupIndex + 1,
1478                         groupEndIndex);
1479 
1480                 if (thisPermission.toLowerCase().charAt(pTypeIndex + 1) == 'r')
1481                 {
1482                     actionID = Constants.READ;
1483                 }
1484                 else if (thisPermission.toLowerCase().charAt(pTypeIndex + 1) == 'w')
1485                 {
1486                     actionID = Constants.WRITE;
1487                 }
1488 
1489                 try
1490                 {
1491                     myGroup = Group.findByName(c, groupName);
1492                 }
1493                 catch (SQLException sqle)
1494                 {
1495                     System.out.println("SQL Exception finding group name: "
1496                             + groupName);
1497                     // do nothing, will check for null group later
1498                 }
1499             }
1500 
1501             String thisDescription = "";
1502             if (descriptionExists)
1503             {
1504                 thisDescription = line.substring(
1505                         dMarkerIndex + descriptionMarker.length(), dEndIndex)
1506                         .trim();
1507             }

1511             if (!isTest)

1515                 for (int j = 0; j < bitstreams.length && notfound; j++)
