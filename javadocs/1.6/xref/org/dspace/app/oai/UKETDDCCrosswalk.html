55  /**
56   * A Crosswalk implementation that extracts qualified Dublin Core from
57   * DSpace items into the uketd_dc format.
58   *
59   * It supports the writing of UKETD_DC metadata
60   * in a METS document and to make the schema URIs available for
61   * inclusion in such a METS document. For this reason, the writing
62   * of the metadata itself has been separated from the writing
63   * of the schemas.
64   * This version places the writing of the header and metadata
65   * in its own method called by createMetadata so the headers are
66   * included in the UKETD_METS that also uses those methods.
67   * This allows the writeMetadata method to remain unchanged,
68   * with no header information included. It is therefore consistent with
69   * other DSpace crosswalks.
70   *
71   * @author Paul Needham (Cranfield University)
72   * @author Jon Bell & Stuart Lewis (Aberystwyth University)
73   */
74  public class UKETDDCCrosswalk extends Crosswalk

245     /**
246      * Creates the metadata necessary for UKTEDDC crosswalk.
247      * Adds the name space details and schemas to the metadata itself.
248      * It therefore creates a complete OAI-PMH record that matches
249      * the UKETD_DC metadata prefix.
250      *
251      * @return The OAI-PMH xml
252      */
253     public String createMetadata (Object nativeItem)
254                                 throws CannotDisseminateFormatException
255     {
256         // Get the Item
257         Item item = ((HarvestedItemInfo) nativeItem).item;
258 
259         // Write the record out
260         return writeMetadataWithSchema(item);
261     }

263    /**
264      * Write the item's metadata, headed by the schema namespace
265      * details. Separated from createMetadata, so UKETD_METS can
266      * use the method with an Item , not an Object (nativeItem).
267      *
268      * @param item The org.dspace.content.Item
269      * @return a String, the item's metadata in UKETD_DC format.
270      * @throws SQLException
271      */
272     public String writeMetadataWithSchema (Item item)
273     {
274         StringBuffer metadata = new StringBuffer ();
275         metadata.append (uketdIn + " ");
276         metadata.append ("xmlns:" + uketdNs + "=\"" + uketdUri + "\" ");
277         metadata.append ("xmlns:" + dcNs + "=\"" + dcUri + "\" ");
278         metadata.append ("xmlns:" + dcTermsNs + "=\"" + dcTermsUri + "\" ");
279         metadata.append ("xmlns:" + uketdTermsNs + "=\"" + uketdTermsUri + "\" ");
280         metadata.append (xsi + " ");
281         metadata.append (schemaLoc + "=\"" + uketdSchemaLocNs + " ");
282         metadata.append (uketdSchemaLocUri + "\">\n");
283         metadata.append (writeMetadata (item));
284         metadata.append ("</uketd_dc:uketddc>\n");
285         return metadata.toString ( );
286     }

289     /**
290      * Writes the UKETD_DC metadata for the specified item.
291      * It simply gets hold of the Dublin Core for an Item
292      * and converts it to UKEDT_DC, including the splitting
293      * of the Dublin Core publisher and type fields.
294      * The metadata is identical to that returned by
295      * the original version's create metadata method,
296      * without the schema information.
297      * This method does no checking of the correctness of the
298      * metadata format, nor does it throw any exception.
299      *
300      * @param item a org.dspace.content.Item
301      * @return a String, the item's metadata in UKETD_DC xml.
302      */
303     public String writeMetadata(Item item)
304     {
305         // The string we are constructing
306         StringBuffer metadata = new StringBuffer();
307 
308         // Get all the DC
309         DCValue[] allDC = item.getMetadata(MetadataSchema.DC_SCHEMA, Item.ANY, Item.ANY, Item.ANY);
310 
311         // Get the handle of the item
312         String itemhandle = item.getHandle();
313 
314         for (int i = 0; i < allDC.length; i++)
315         {
316             // Get the element, qualifier and value
317             String element = allDC[i].element;
318             String qualifier = allDC[i].qualifier;
319             String value = Utils.addEntities(allDC[i].value);
320 
321             // title
322             if (allDC[i].element.equals("title"))
323             {
324                 if (allDC[i].qualifier != null) {
325                     if (allDC[i].qualifier.equals("alternative"))
326                     {
327                         // title.alternative exposed as 'dcterms:alternative'
328                         this.makeDCTermsElement(qualifier, null, value, metadata);
329                     }
330                 } else
331                 {
332                     this.makeDCElement(element, null, value, metadata);
333                 }
334             }
335 
336             // contributor
337             if (allDC[i].element.equals("contributor"))
338             {
339                 if (allDC[i].qualifier != null) {
340                     if (allDC[i].qualifier.equals("author"))
341                     {
342                         this.makeDCElement("creator", null, value, metadata);
343                     } else if ((allDC[i].qualifier.equals("advisor")) ||
344                                (allDC[i].qualifier.equals("sponsor")))
345                     {
346                         // contributor.qualifier exposed as 'uketdterms:qualifier'
347                         this.makeUKDCTermsElement(qualifier, null, value, metadata);
348                     } else if (allDC[i].qualifier.equals("funder"))
349                     {
350 										    // contributor.qualifier exposed as 'uketdterms:qualifier'
351                         this.makeUKDCTermsElement("sponsor", null, value, metadata);
352                     } else
353                     {
354                         // contributor.qualifier exposed as 'dcterms:qualifier'
355                         this.makeDCTermsElement(qualifier, null, value, metadata);
356                     }
357                 } else {
358                     this.makeDCElement(element, null, value, metadata);
359                 }
360             }
361 
362             // subject
363             if (allDC[i].element.equals("subject"))
364             {
365                 if (allDC[i].qualifier != null) {
366                     boolean ddc = allDC[i].qualifier.equals("ddc");
367                     boolean lcc = allDC[i].qualifier.equals("lcc");
368                     boolean lcsh = allDC[i].qualifier.equals("lcsh");
369                     boolean mesh = allDC[i].qualifier.equals("mesh");
370                     boolean udc = allDC[i].qualifier.equals("udc");
371                     if (ddc || lcc || lcsh || mesh || udc)
372                     {
373                         // subject.qualifier exposed as 'dc:element xsi:type="dcterms:qualifier"'
374                         qualifier = qualifier.toUpperCase();
375                         this.makeDCElement(element, qualifier, value, metadata);
376                     } else
377                     {
378                         this.makeDCElement(element, null, value, metadata);
379                     }
380                 } else
381                 {
382                     this.makeDCElement(element, null, value, metadata);
383                 }
384             }
385 
386             // description
387             if (allDC[i].element.equals("description"))
388             {
389                 if (allDC[i].qualifier != null)
390                 {
391                     if (allDC[i].qualifier.equals("abstract"))
392                     {
393                         // e.g. description.abstract exposed as 'dcterms:abstract'
394                         this.makeDCTermsElement(qualifier, null, value, metadata);
395                     } else if  (allDC[i].qualifier.equals("sponsorship"))
396                     {
397                         // description.sponsorship exposed as 'uketdterms:sponsor"'
398                         this.makeUKDCTermsElement("sponsor", null, value, metadata);
399                     }
400                 } else {
401                     this.makeDCElement(element, null, value, metadata);
402                 }
403             }
404 
405             // publisher
406             if (allDC[i].element.equals("publisher"))
407             {
408                 if (allDC[i].qualifier != null) {
409                     if ((allDC[i].qualifier.equals("department")) ||
410                     (allDC[i].qualifier.equals("commercial")))
411                     {
412                         this.makeUKDCTermsElement(qualifier, null, value, metadata);
413                     }
414                 } else {
415                     String[] pubParts = value.split("(?<!(&[0-9a-zA-Z#]{2,4}));");
416                     this.makeUKDCTermsElement("institution", null,
417                                               pubParts[0], metadata);
418                     StringBuffer dept = new StringBuffer();
419                     if ((pubParts.length > 1) && (pubParts[1] != null)) {
420                         dept.append(pubParts[1] + ";");
421                     }
422                     if ((pubParts.length > 2) && (pubParts[2] != null)) {
423                         dept.append(" " + pubParts[2]);
424                     }
425                     if (dept.toString().length() > 0) {
426                             this.makeUKDCTermsElement("department", null,
427                                                       dept.toString(), metadata);
428                     }
429                 }
430 
431             }
432 
433             // date
434             if (allDC[i].element.equals("date"))
435             {
436                 if (allDC[i].qualifier != null)
437                 {
438                     if (allDC[i].qualifier.equals("issued"))
439                     {
440                         this.makeDCTermsElement(qualifier, null, value, metadata);
441                     } else
442                     {
443                         this.makeDCElement(element, null, value, metadata);
444                     }
445                 } else
446                 {
447                     this.makeDCElement(element, null, value, metadata);
448                 }
449             }
450 
451             // type
452             if (allDC[i].element.equals("type"))
453             {
454                 if (allDC[i].qualifier != null)
455                 {
456                     if ((allDC[i].qualifier.equals("qualificationlevel")) ||
457                         (allDC[i].qualifier.equals("qualificationname")))
458                     {
459                         this.makeUKDCTermsElement(qualifier, null, value, metadata);
460                     }
461                 } else {
462                     String[] Typepart = value.split("[;]");
463                     this.makeDCElement(element, null, Typepart[0], metadata);
464                     if ((Typepart.length > 1) && (Typepart[1] != null))
465                     {
466                         this.makeUKDCTermsElement("qualificationlevel", null,
467                                                   Typepart[1], metadata);
468                     } if ((Typepart.length > 2) && (Typepart[2] != null))
469                     {
470                         this.makeUKDCTermsElement("qualificationname", null,
471                                                   Typepart[2], metadata);
472                     }
473                 }
474             }
475 
476             // language
477             if (allDC[i].element.equals("language"))
478             {
479                 if (allDC[i].qualifier != null) {
480                     if (allDC[i].qualifier.equals("iso"))
481                     {
482                         // language.iso exposed as 'dc:element xsi:type="dcterms:qualifier"'
483                         this.makeDCElement(element, "ISO639-2", value, metadata);
484                     } else
485                     {
486                         this.makeDCElement(element, null, value, metadata);
487                     }
488                 } else
489                 {
490                     this.makeDCElement(element, null, value, metadata);
491                 }
492             }
493 
494             // relation
495             if (allDC[i].element.equals("relation"))
496             {
497                 if (allDC[i].qualifier != null) {
498                     if (allDC[i].qualifier.equals("hasversion"))
499                     {
500                         // relation.hasversion exposed as 'dcterms:qualifier'
501                         this.makeDCElement("hasVersion", null, value, metadata);
502                     } else if ((allDC[i].qualifier.equals("references")) ||
503                                (allDC[i].qualifier.equals("requires")))
504                     {
505                         // relation.references exposed as 'dcterms:qualifier'
506                         this.makeDCTermsElement(qualifier, null, value, metadata);
507                     } else
508                     {
509                         this.makeDCElement(element, null, value, metadata);
510                     }
511                 } else
512                 {
513                     this.makeDCElement(element, null, value, metadata);
514                 }
515             }
516 
517             // format
518             if (allDC[i].element.equals("format"))
519             {
520                 if (allDC[i].qualifier != null)
521                 {
522                     if (allDC[i].qualifier.equals("extent"))
523                     {
524                         // format exposed as 'dcterms:qualifier'
525                         this.makeDCTermsElement(qualifier, null, value, metadata);
526                     } else if (allDC[i].qualifier.equals("mimetype"))
527                     {
528                         this.makeDCElement(element, "IMT", value, metadata);
529                     }
530                 } else
531                 {
532                     // format exposed as 'dc:element'
533                     this.makeDCElement(element, null, value, metadata);
534                 }
535             }
536 
537             // identifier
538             if (allDC[i].element.equals("identifier"))
539             {
540                 if (allDC[i].qualifier != null)
541                 {
542                     if (allDC[i].qualifier.equals("uri"))
543                     {
544                         this.makeDCTermsElement("isReferencedBy", "URI", value, metadata);
545                     } else if (allDC[i].qualifier.equals("citation"))
546                     {
547                         this.makeDCTermsElement("hasVersion", null, value, metadata);
548                     } else if (allDC[i].qualifier.equals("grantnumber"))
549                     {
550                         this.makeUKDCTermsElement(qualifier, null, value, metadata);
551                     }
552                 } else
553                 {
554                     // identifier exposed as 'dc:element'
555                     this.makeDCElement(element, null, value, metadata);
556                 }
557             }
558 
559             // rights
560             if (allDC[i].element.equals("rights"))
561             {
562                 if (allDC[i].qualifier != null)
563                 {
564                     if ((allDC[i].qualifier.equals("embargodate")) ||
565                         (allDC[i].qualifier.equals("embargoreason")))
566                     {
567                         this.makeUKDCTermsElement(qualifier, null, value, metadata);
568                     } else
569                     {
570                         // rights exposed as 'dc:element'
571                         this.makeDCElement(element, null, value, metadata);
572                     }
573                 } else
574                 {
575                     // rights exposed as 'dc:element'
576                     this.makeDCElement(element, null, value, metadata);
577                 }
578             }
579         }
580 
581         // Generate bitstream URIs
582         Bundle[] bundles = {};
583         try
584         {
585             bundles = item.getBundles("ORIGINAL");
586             String url;
587             if (bundles.length > 0)
588             {
589                 // Itterate through each bundle
590                 for (int i = 0; i < bundles.length; i++)
591                 {
592                     // Itterate through each bitstream
593                     Bitstream[] bitstreams = bundles[i].getBitstreams();
594                     for (int k = 0; k < bitstreams.length ; k++)
595                     {
596                         // Skip internal types
597                         if (!bitstreams[k].getFormat().isInternal())
598                         {
599                             url = ConfigurationManager.getProperty("dspace.url") +
600                                   "/bitstream/" + itemhandle + "/" +
601                                   bitstreams[k].getSequenceID() + "/" +
602                                   bitstreams[k].getName();
603                             this.makeDCElement("identifier", "URI", url, metadata);
604                             this.makeUKDCTermsElement("checksum",
605                                                       bitstreams[k].getChecksumAlgorithm(),
606                                                       bitstreams[k].getChecksum(), metadata);
607                         }
608                     }
609                 }
610             }
611         } catch (SQLException sqle)
612         {
613             // Nothing we can do
614         }
615 
616         // Return the metadata - all done!
617         return metadata.toString();
618     }

668     /**
669      * Private wrapper method to create an element.
670      *
671      * @param element The element name
672      * @param qualifier The qualifier name (or null)
673      * @param value The value of the element
674      * @param buffer The buffer to add the element to
675      * @param terms The namespace of the term
676      * @return The buffer with the new element appended to
677      */
678 
679     private StringBuffer makeTermsElement(String element, String qualifier,
680                                           String value, StringBuffer buffer,
681                                           String namespace, String terms)
682     {
683         // Escape XML chars <, > and &
684         // Also replace all invalid characters with ' '
685         if (value != null)
686         {
687           	StringBuffer valueBuf = new StringBuffer(value.length());
688            	Matcher xmlMatcher = invalidXmlPattern.matcher(value.trim());
689            	while (xmlMatcher.find())
690            	{
691            		String group = xmlMatcher.group();
692 
693            		// group will either contain a character that we need to encode for xml
694            		// (ie. <, > or &), or it will be an invalid character
695            		// test the contents and replace appropriately
696            		if (group.equals("&"))
697            			xmlMatcher.appendReplacement(valueBuf, "&amp;");
698            		else if (group.equals("<"))
699            			xmlMatcher.appendReplacement(valueBuf, "&lt;");
