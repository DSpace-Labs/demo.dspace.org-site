120     /**
121      * Loads all known submission information based on the given in progress
122      * submission and request object.
123      * <P>
124      * If subItem is null, then just loads the default submission information
125      * for a new submission.
126      * 
127      * @param request
128      *            The HTTP Servlet Request object          
129      * @param subItem
130      *            The in-progress submission we are loading information for
131      * 
132      * @return a SubmissionInfo object
133      * 
134      * @throws ServletException
135      *             if an error occurs
136      */
137     public static SubmissionInfo load(HttpServletRequest request, InProgressSubmission subItem) throws ServletException
138     {
139         boolean forceReload = false;
140     	SubmissionInfo subInfo = new SubmissionInfo();
141         
142         // load SubmissionConfigReader only the first time
143         // or if we're using a different UI now.
144         if (submissionConfigReader == null)
145         {
146             submissionConfigReader = new SubmissionConfigReader();
147             forceReload=true;
148         }
149 
150         // save the item which is going through the submission process
151         subInfo.setSubmissionItem(subItem);
152 
153         // Only if the submission item is created can we set its collection
154         String collectionHandle = SubmissionConfigReader.DEFAULT_COLLECTION;
155         if (subItem != null)
156         {
157             collectionHandle = subItem.getCollection().getHandle();
158         }
159 
160         // save this collection handle to this submission info object
161         subInfo.setCollectionHandle(collectionHandle);
162 
163         // load Submission Process config for this item's collection
164         // (Note: this also loads the Progress Bar info, since it is
165         // dependent on the Submission config)
166         loadSubmissionConfig(request, subInfo, forceReload);
167 
168         return subInfo;
169     }

460     /**
461      * Load necessary information to build the Progress Bar for the Item
462      * Submission Progress.
463      * 
464      * This information is returned in the form of a HashMap (which is then
465      * stored as a part of the SubmissionInfo). The HashMap takes the following
466      * form:
467      * 
468      * Keys - the number of the step, followed by the number of the page within
469      * the step (e.g. "2.1" = The first page of Step 2) (e.g. "5.2" = The second
470      * page of Step 5)
471      * 
472      * Values - the headings to display for each step (e.g. "Describe",
473      * "Verify")
474      * 
475      * @param request
476      *            The HTTP Servlet Request object
477      * @param subInfo
478      *            the SubmissionInfo object we are loading into
479      * @param forceReload
480      *            If true, this method reloads from stratch (and overwrites
481      *            cached progress bar info)
482      * 
483      */
484     private static void loadProgressBar(HttpServletRequest request,
485             SubmissionInfo subInfo, boolean forceReload)
486     {
487         LinkedHashMap progressBarInfo = null;
488 
489         log.debug("Loading Progress Bar Info");
490 
491         if (!forceReload)
492         {
493             // first, attempt to load from cache
494             progressBarInfo = (LinkedHashMap) loadProgressBarFromCache(request
495                     .getSession());
496         }
497 
498         if (progressBarInfo != null && log.isDebugEnabled())
499         {
500             log.debug("Found Progress Bar Info in cache: "
501                     + progressBarInfo.size()
502                     + " pages to display in progress bar");
503         }
504         // if unable to load from cache, must load from scratch
505         else
506         {
507             progressBarInfo = new LinkedHashMap();
508 
509             // loop through all steps
510             for (int i = 0; i < subInfo.submissionConfig.getNumberOfSteps(); i++)
511             {
512                 // get the current step info
513                 SubmissionStepConfig currentStep = subInfo.submissionConfig
514                         .getStep(i);
515                 String stepNumber = Integer.toString(currentStep
516                         .getStepNumber());
517                 String stepHeading = currentStep.getHeading();
518 
519                 // as long as this step is visible, include it in
520                 // the Progress Bar
521                 if (currentStep.isVisible())
522                 {
523                     // default to just one page in this step
524                     int numPages = 1;
525 
526                     try
527                     {
528                         // load the processing class for this step
529                         ClassLoader loader = subInfo.getClass()
530                                 .getClassLoader();
531                         Class stepClass = loader.loadClass(currentStep
532                                 .getProcessingClassName());
533 
534                         // call the "getNumberOfPages()" method of the class
535                         // to get it's number of pages
536                         AbstractProcessingStep step = (AbstractProcessingStep) stepClass
537                                 .newInstance();
538 
539                         // get number of pages from servlet
540                         numPages = step.getNumberOfPages(request, subInfo);
541                     }
542                     catch (Exception e)
543                     {
544                         log.error(
545                                 "Error loading progress bar information from Step Class '"
546                                         + currentStep.getProcessingClassName()
547                                         + "' Error:", e);
548                     }
549 
550                     // save each of the step's pages to the progress bar
551                     for (int j = 1; j <= numPages; j++)
552                     {
553                         String pageNumber = Integer.toString(j);
554 
555                         // store ("stepNumber.pageNumber", Heading) for each
556                         // page in the step
557                         progressBarInfo.put(stepNumber + "." + pageNumber,
558                                 stepHeading);
559                     }// end for each page
560                 }
561             }// end for each step
562 
563             log.debug("Loaded Progress Bar Info from scratch: "
564                     + progressBarInfo.size()
565                     + " pages to display in progress bar");
566 
567             // cache this new progress bar
568             saveProgressBarToCache(request.getSession(), progressBarInfo);
569         }// end if null
570 
571         // save progressBarInfo to submission Info
572         subInfo.progressBar = progressBarInfo;
573     }

609     /**
610      * Loads SubmissionConfig object for the given submission info object. If a
611      * SubmissionConfig object cannot be loaded, a Servlet Error is thrown.
612      * <p>
613      * This method just loads this SubmissionConfig object internally, so that
614      * it is available via a call to "getSubmissionConfig()"
615      * 
616      * @param request
617      *            The HTTP Servlet Request object
618      * @param subInfo
619      *            the SubmissionInfo object we are loading into
620      * @param forceReload
621      *            If true, this method reloads from stratch (and overwrites
622      *            cached SubmissionConfig)
623      * 
624      */
625     private static void loadSubmissionConfig(HttpServletRequest request,
626             SubmissionInfo subInfo, boolean forceReload)
627             throws ServletException
628     {
629 
630         log.debug("Loading Submission Config information");
631 
632         if (!forceReload)
633         {
634             // first, try to load from cache
635             subInfo.submissionConfig = loadSubmissionConfigFromCache(request
636                     .getSession(), subInfo.getCollectionHandle(), subInfo
637                     .isInWorkflow());
638         }
639 
640         if (subInfo.submissionConfig == null || forceReload)
641         {
642             // reload the proper Submission process config
643             // (by reading the XML config file)
644             subInfo.submissionConfig = submissionConfigReader
645                     .getSubmissionConfig(subInfo.getCollectionHandle(), subInfo
646                             .isInWorkflow());
647 
648             // cache this new submission process configuration
649             saveSubmissionConfigToCache(request.getSession(),
650                     subInfo.submissionConfig, subInfo.getCollectionHandle(),
651                     subInfo.isInWorkflow());
652 
653             // also must force reload Progress Bar info,
654             // since it's based on the Submission config
655             loadProgressBar(request, subInfo, true);
656         }
657         else
658         {
659             log.debug("Found Submission Config in session cache!");
660 
661             // try and reload progress bar from cache
662             loadProgressBar(request, subInfo, false);
663         }
664     }

693     /**
694      * Loads SubmissionConfig object from session cache for the given
695      * Collection. If a SubmissionConfig object cannot be found, null is
696      * returned.
697      * 
698      * @param session
699      *            The HTTP Session object
700      * @param collectionHandle
701      *            The Collection handle of the SubmissionConfig to load
702      * @param isWorkflow
703      *            whether or not we loading the Submission process for a
704      *            workflow item
705      * 
706      * @return The cached SubmissionConfig for this collection
707      */
708     private static SubmissionConfig loadSubmissionConfigFromCache(
709             HttpSession session, String collectionHandle, boolean isWorkflow)
710     {
711         // attempt to load submission process config
712         // from cache for the current collection
713         String cachedHandle = (String) session
714                 .getAttribute("submission.config.collection");
715 
716         Boolean cachedIsWorkflow = (Boolean) session
717                 .getAttribute("submission.config.isWorkflow");
718 
719         // only load from cache if the collection handle and
720         // workflow item status both match!
721         if (collectionHandle.equals(cachedHandle)
722                 && isWorkflow == cachedIsWorkflow.booleanValue())
723 
724         {
725             return (SubmissionConfig) session.getAttribute("submission.config");
726         }
727         else
728         {
729             return null;
730         }
731     }
