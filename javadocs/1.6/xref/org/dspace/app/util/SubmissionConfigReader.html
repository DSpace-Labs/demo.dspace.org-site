55  /**
56   * Item Submission configuration generator for DSpace. Reads and parses the
57   * installed submission process configuration file, item-submission.xml, from
58   * the configuration directory. This submission process definiton details the
59   * ordering of the steps (and number of steps) that occur during the Item
60   * Submission Process. There may be multiple Item Submission processes defined,
61   * where each definition is assigned a unique name.
62   * 
63   * The file also specifies which collections use which Item Submission process.
64   * At a minimum, the definitions file must define a default mapping from the
65   * placeholder collection # to the distinguished submission process 'default'.
66   * Any collections that use a custom submission process are listed paired with
67   * the name of the item submission process they use.
68   * 
69   * @see org.dspace.app.util.SubmissionConfig
70   * @see org.dspace.app.util.SubmissionStepConfig
71   * 
72   * @author Tim Donohue based on DCInputsReader by Brian S. Hughes
73   * @version $Revision: 3734 $
74   */

131     /**
132      * Parse an XML encoded item submission configuration file.
133      * <P>
134      * Creates two main hashmaps:
135      * <ul>
136      * <li>Hashmap of Collection to Submission definition mappings -
137      * defines which Submission process a particular collection uses
138      * <li>Hashmap of all Submission definitions.  List of all valid
139      * Submision Processes by name.
140      * </ul>
141      */
142     private void buildInputs(String fileName) throws ServletException
143     {
144         collectionToSubmissionConfig = new HashMap();
145         submitDefns = new HashMap();
146 
147         String uri = "file:" + new File(fileName).getAbsolutePath();
148 
149         try
150         {
151             DocumentBuilderFactory factory = DocumentBuilderFactory
152                     .newInstance();
153             factory.setValidating(false);
154             factory.setIgnoringComments(true);
155             factory.setIgnoringElementContentWhitespace(true);
156 
157             DocumentBuilder db = factory.newDocumentBuilder();
158             Document doc = db.parse(uri);
159             doNodes(doc);
160         }
161         catch (FactoryConfigurationError fe)
162         {
163             throw new ServletException(
164                     "Cannot create Item Submission Configuration parser", fe);
165         }
166         catch (Exception e)
167         {
168             throw new ServletException(
169                     "Error creating Item Submission Configuration: " + e);
170         }
171     }

173     /**
174      * Returns the Item Submission process config used for a particular
175      * collection, or the default if none is defined for the collection
176      * 
177      * @param collectionHandle
178      *            collection's unique Handle
179      * @param isWorkflow
180      *            whether or not we are loading the submission process for a
181      *            workflow
182      * @return the SubmissionConfig representing the item submission config
183      * 
184      * @throws ServletException
185      *             if no default submission process configuration defined
186      */
187     public SubmissionConfig getSubmissionConfig(String collectionHandle,
188             boolean isWorkflow) throws ServletException
189     {
190         // get the name of the submission process config for this collection
191         String submitName = (String) collectionToSubmissionConfig
192                 .get(collectionHandle);
193         if (submitName == null)
194         {
195             submitName = (String) collectionToSubmissionConfig
196                     .get(DEFAULT_COLLECTION);
197         }
198         if (submitName == null)
199         {
200             throw new ServletException(
201                     "No item submission process configuration designated as 'default' in 'submission-map' section of 'item-submission.xml'.");
202         }
203 
204         log.debug("Loading submission process config named '" + submitName
205                 + "'");
206 
207         // check mini-cache, and return if match
208         if (lastSubmissionConfig != null
209                 && lastSubmissionConfig.getSubmissionName().equals(submitName)
210                 && lastSubmissionConfig.isWorkflow() == isWorkflow)
211         {
212             log.debug("Found submission process config '" + submitName
213                     + "' in cache.");
214 
215             return lastSubmissionConfig;
216         }
217 
218         // cache miss - construct new SubmissionConfig
219         Vector steps = (Vector) submitDefns.get(submitName);
220 
221         if (steps == null)
222         {
223             throw new ServletException(
224                     "Missing the Item Submission process config '" + submitName
225                             + "' (or unable to load) from 'item-submission.xml'.");
226         }
227 
228         log.debug("Submission process config '" + submitName
229                 + "' not in cache. Reloading from scratch.");
230 
231         lastSubmissionConfig = new SubmissionConfig(submitName, steps,
232                 isWorkflow);
233 
234         log.debug("Submission process config has "
235                 + lastSubmissionConfig.getNumberOfSteps() + " steps listed.");
236 
237         return lastSubmissionConfig;
238     }

270     /**
271      * Process the top level child nodes in the passed top-level node. These
272      * should correspond to the collection-form maps, the form definitions, and
273      * the display/storage word pairs.
274      */
275     private void doNodes(Node n) throws SAXException, ServletException
276     {
277         if (n == null)
278         {
279             return;
280         }
281         Node e = getElement(n);
282         NodeList nl = e.getChildNodes();
283         int len = nl.getLength();
284         boolean foundMap = false;
285         boolean foundStepDefs = false;
286         boolean foundSubmitDefs = false;
287         for (int i = 0; i < len; i++)
288         {
289             Node nd = nl.item(i);
290             if ((nd == null) || isEmptyTextNode(nd))
291             {
292                 continue;
293             }
294             String tagName = nd.getNodeName();
295             if (tagName.equals("submission-map"))
296             {
297                 processMap(nd);
298                 foundMap = true;
299             }
300             else if (tagName.equals("step-definitions"))
301             {
302                 processStepDefinition(nd);
303                 foundStepDefs = true;
304             }
305             else if (tagName.equals("submission-definitions"))
306             {
307                 processSubmissionDefinition(nd);
308                 foundSubmitDefs = true;
309             }
310             // Ignore unknown nodes
311         }
312         if (!foundMap)
313         {
314             throw new ServletException(
315                     "No collection to item submission map ('submission-map') found in 'item-submission.xml'");
316         }
317         if (!foundStepDefs)
318         {
319             throw new ServletException("No 'step-definitions' section found in 'item-submission.xml'");
320         }
321         if (!foundSubmitDefs)
322         {
323             throw new ServletException(
324                     "No 'submission-definitions' section found in 'item-submission.xml'");
325         }
326     }

328     /**
329      * Process the submission-map section of the XML file. Each element looks
330      * like: <name-map collection-handle="hdl" submission-name="name" /> Extract
331      * the collection handle and item submission name, put name in hashmap keyed
332      * by the collection handle.
333      */
334     private void processMap(Node e) throws SAXException
335     {
336         NodeList nl = e.getChildNodes();
337         int len = nl.getLength();
338         for (int i = 0; i < len; i++)
339         {
340             Node nd = nl.item(i);
341             if (nd.getNodeName().equals("name-map"))
342             {
343                 String id = getAttribute(nd, "collection-handle");
344                 String value = getAttribute(nd, "submission-name");
345                 String content = getValue(nd);
346                 if (id == null)
347                 {
348                     throw new SAXException(
349                             "name-map element is missing collection-handle attribute in 'item-submission.xml'");
350                 }
351                 if (value == null)
352                 {
353                     throw new SAXException(
354                             "name-map element is missing submission-name attribute in 'item-submission.xml'");
355                 }
356                 if (content != null && content.length() > 0)
357                 {
358                     throw new SAXException(
359                             "name-map element has content in 'item-submission.xml', it should be empty.");
360                 }
361                 collectionToSubmissionConfig.put(id, value);
362             } // ignore any child node that isn't a "name-map"
363         }
364     }

366     /**
367      * Process the "step-definition" section of the XML file. Each element is
368      * formed thusly: <step id="unique-id"> ...step_fields... </step> The valid
369      * step_fields are: heading, processing-servlet.
370      * <P>
371      * Extract the step information (from the step_fields) and place in a
372      * HashMap whose key is the step's unique id.
373      */
374     private void processStepDefinition(Node e) throws SAXException,
375             ServletException
376     {
377         int numStepDefns = 0;
378         stepDefns = new HashMap();
379 
380         NodeList nl = e.getChildNodes();
381         int len = nl.getLength();
382         for (int i = 0; i < len; i++)
383         {
384             Node nd = nl.item(i);
385             // process each step definition
386             if (nd.getNodeName().equals("step"))
387             {
388                 numStepDefns++;
389                 String stepID = getAttribute(nd, "id");
390                 if (stepID == null)
391                 {
392                     throw new SAXException(
393                             "step element has no 'id' attribute in 'item-submission.xml', which is required in the 'step-definitions' section");
394                 }
395                 else if (stepDefns.containsKey(stepID))
396                 {
397                     throw new SAXException(
398                             "There are two step elements with the id '" + stepID + "' in 'item-submission.xml'");
399                 }
400 
401                 HashMap stepInfo = processStepChildNodes("step-definition", nd);
402 
403                 stepDefns.put(stepID, stepInfo);
404             } // ignore any child that is not a 'step'
405         }
406 
407         // Sanity check number of step definitions
408         if (stepDefns.size() < 1)
409         {
410             throw new ServletException(
411                     "step-definition section has no steps! A step with id='collection' is required in 'item-submission.xml'!");
412         }
413 
414         // Sanity check to see that the required "collection" step is defined
415         if (!stepDefns.containsKey(SubmissionStepConfig.SELECT_COLLECTION_STEP))
416         {
417             throw new ServletException(
418                     "The step-definition section is REQUIRED to have a step with id='"
419                             + SubmissionStepConfig.SELECT_COLLECTION_STEP
420                             + "' in 'item-submission.xml'!  This step is used to ensure that a new item submission is assigned to a collection.");
421         }
422 
423         // Sanity check to see that the required "complete" step is defined
424         if (!stepDefns.containsKey(SubmissionStepConfig.COMPLETE_STEP))
425         {
426             throw new ServletException(
427                     "The step-definition section is REQUIRED to have a step with id='"
428                             + SubmissionStepConfig.COMPLETE_STEP
429                             + "' in 'item-submission.xml'!  This step is used to perform all processing necessary at the completion of the submission (e.g. starting workflow).");
430         }
431     }

433     /**
434      * Process the "submission-definition" section of the XML file. Each element
435      * is formed thusly: <submission-process name="submitName">...steps...</submit-process>
436      * Each step subsection is formed: <step> ...step_fields... </step> (with
437      * optional "id" attribute, to reference a step from the <step-definition>
438      * section). The valid step_fields are: heading, class-name.
439      * <P>
440      * Extract the submission-process name and steps and place in a HashMap
441      * whose key is the submission-process's unique name.
442      */
443     private void processSubmissionDefinition(Node e) throws SAXException,
444             ServletException
445     {
446         int numSubmitProcesses = 0;
447         Vector submitNames = new Vector();
448 
449         // find all child nodes of the 'submission-definition' node and loop
450         // through
451         NodeList nl = e.getChildNodes();
452         int len = nl.getLength();
453         for (int i = 0; i < len; i++)
454         {
455             Node nd = nl.item(i);
456 
457             // process each 'submission-process' node
458             if (nd.getNodeName().equals("submission-process"))
459             {
460                 numSubmitProcesses++;
461                 String submitName = getAttribute(nd, "name");
462                 if (submitName == null)
463                 {
464                     throw new SAXException(
465                             "'submission-process' element has no 'name' attribute in 'item-submission.xml'");
466                 }
467                 else if (submitNames.contains(submitName))
468                 {
469                     throw new SAXException(
470                             "There are two 'submission-process' elements with the name '"
471                                     + submitName + "' in 'item-submission.xml'.");
472                 }
473                 submitNames.add(submitName);
474 
475                 // the 'submission-process' definition contains steps
476                 Vector steps = new Vector();
477                 submitDefns.put(submitName, steps);
478 
479                 // loop through all the 'step' nodes of the 'submission-process'
480                 int stepNum = 0;
481                 NodeList pl = nd.getChildNodes();
482                 int lenStep = pl.getLength();
483                 for (int j = 0; j < lenStep; j++)
484                 {
485                     Node nStep = pl.item(j);
486                     stepNum++;
487 
488                     // process each 'step' definition
489                     if (nStep.getNodeName().equals("step"))
490                     {
491                         // check for an 'id' attribute
492                         String stepID = getAttribute(nStep, "id");
493 
494                         HashMap stepInfo;
495 
496                         // if this step has an id, load its information from the
497                         // step-definition section
498                         if ((stepID != null) && (stepID.length() > 0))
499                         {
500                             if (stepDefns.containsKey(stepID))
501                             {
502                                 // load the step information from the
503                                 // step-definition
504                                 stepInfo = (HashMap) stepDefns.get(stepID);
505                             }
506                             else
507                             {
508                                 throw new SAXException(
509                                         "The Submission process config named "
510                                                 + submitName
511                                                 + " contains a step with id="
512                                                 + stepID
513                                                 + ".  There is no step with this 'id' defined in the 'step-definition' section of 'item-submission.xml'.");
514                             }
515 
516                             // Ignore all children of a step element with an
517                             // "id"
518                         }
519                         else
520                         {
521                             // get information about step from its children
522                             // nodes
523                             stepInfo = processStepChildNodes(
524                                     "submission-process", nStep);
525                         }
526 
527                         steps.add(stepInfo);
528 
529                     } // ignore any child that is not a 'step'
530                 }
531 
532                 // sanity check number of steps
533                 if (steps.size() < 1)
534                 {
535                     throw new ServletException(
536                             "Item Submission process config named "
537                                     + submitName + " has no steps defined in 'item-submission.xml'");
538                 }
539 
540                 // ALL Item Submission processes MUST BEGIN with selecting a
541                 // Collection. So, automatically insert in the "collection" step
542                 // (from the 'step-definition' section)
543                 // Note: we already did a sanity check that this "collection"
544                 // step exists.
545                 steps.add(0, (HashMap) stepDefns
546                         .get(SubmissionStepConfig.SELECT_COLLECTION_STEP));
547 
548                 // ALL Item Submission processes MUST END with the
549                 // "Complete" processing step.
550                 // So, automatically append in the "complete" step
551                 // (from the 'step-definition' section)
552                 // Note: we already did a sanity check that this "complete"
553                 // step exists.
554                 steps.add((HashMap) stepDefns
555                         .get(SubmissionStepConfig.COMPLETE_STEP));
556 
557             }
558         }
559         if (numSubmitProcesses == 0)
560         {
561             throw new ServletException(
562                     "No 'submission-process' elements/definitions found in 'item-submission.xml'");
563         }
564     }

566     /**
567      * Process the children of the "step" tag of the XML file. Returns a HashMap
568      * of all the fields under that "step" tag, where the key is the field name,
569      * and the value is the field value.
570      * 
571      */
572     private HashMap processStepChildNodes(String configSection, Node nStep)
573             throws SAXException, ServletException
574     {
575         // initialize the HashMap of step Info
576         HashMap stepInfo = new HashMap();
577 
578         NodeList flds = nStep.getChildNodes();
579         int lenflds = flds.getLength();
580         for (int k = 0; k < lenflds; k++)
581         {
582             // process each child node of a <step> tag
583             Node nfld = flds.item(k);
584 
585             if (!isEmptyTextNode(nfld))
586             {
587                 String tagName = nfld.getNodeName();
588                 String value = getValue(nfld);
589                 stepInfo.put(tagName, value);
590             }
591         }// end for each field
592 
593         // check for ID attribute & save to step info
594         String stepID = getAttribute(nStep, "id");
595         if (stepID != null && stepID.length() > 0)
596         {
597             stepInfo.put("id", stepID);
598         }
599 
600         // look for REQUIRED 'step' information
601         String missing = null;
602         if (stepInfo.get("processing-class") == null)
603         {
604             missing = "'processing-class'";
605         }
606         if (missing != null)
607         {
608             String msg = "Required field " + missing
609                     + " missing in a 'step' in the " + configSection
610                     + " of the item submission configuration file ('item-submission.xml')";
611             throw new SAXException(msg);
612         }
613 
614         return stepInfo;
615     }

617     private Node getElement(Node nd)
618     {
619         NodeList nl = nd.getChildNodes();
620         int len = nl.getLength();
621         for (int i = 0; i < len; i++)
622         {
623             Node n = nl.item(i);
624             if (n.getNodeType() == Node.ELEMENT_NODE)
625             {
626                 return n;
627             }
628         }
629         return null;
630     }

632     private boolean isEmptyTextNode(Node nd)
633     {
634         boolean isEmpty = false;
635         if (nd.getNodeType() == Node.TEXT_NODE)
636         {
637             String text = nd.getNodeValue().trim();
638             if (text.length() == 0)
639             {
640                 isEmpty = true;
641             }
642         }
643         return isEmpty;
644     }

646     /**
647      * Returns the value of the node's attribute named <name>
648      */
649     private String getAttribute(Node e, String name)
650     {
651         NamedNodeMap attrs = e.getAttributes();
652         int len = attrs.getLength();
653         if (len > 0)
654         {
655             int i;
656             for (i = 0; i < len; i++)
657             {
658                 Node attr = attrs.item(i);
659                 if (name.equals(attr.getNodeName()))
660                 {
661                     return attr.getNodeValue().trim();
662                 }
663             }
664         }
665         // no such attribute
666         return null;
667     }

669     /**
670      * Returns the value found in the Text node (if any) in the node list that's
671      * passed in.
672      */
673     private String getValue(Node nd)
674     {
675         NodeList nl = nd.getChildNodes();
676         int len = nl.getLength();
677         for (int i = 0; i < len; i++)
678         {
679             Node n = nl.item(i);
680             short type = n.getNodeType();
681             if (type == Node.TEXT_NODE)
682             {
683                 return n.getNodeValue().trim();
684             }
685         }
686         // Didn't find a text node
687         return null;
688     }
