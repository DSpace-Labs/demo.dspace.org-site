91  /**
92   * Utility Class with static methods for producing OpenSearch-compliant search results,
93   * and the OpenSearch description document.
94   * <p>
95   * OpenSearch is a specification for describing and advertising search-engines
96   * and their result formats. Commonly, RSS and Atom formats are used, which
97   * the current implementation supports, as is HTML (used directly in browsers).
98   * NB: this is baseline OpenSearch, no extensions currently supported.
99   * </p>
100  * <p>
101  * The value of the "scope" parameter should either be absent (which means no
102  * scope restriction), or the handle of a community or collection.
103  * </p>
104  * 
105  * @author Richard Rodgers
106  *
107  */
108 public class OpenSearch

247     private static SyndicationFeed getResults(String format, String query, QueryResults qResults,
248                                           DSpaceObject scope, DSpaceObject[] results, Map<String, String> labels)
249     {
250         // Encode results in requested format
251         if ("rss".equals(format))
252             format = "rss_2.0";
253         else if ("atom".equals(format))
254             format = "atom_1.0";
255     	
256         SyndicationFeed feed = new SyndicationFeed(labels.get(SyndicationFeed.MSG_UITYPE));
257         feed.populate(null, scope, results, labels);
258         feed.setType(format);
259         feed.addModule(openSearchMarkup(query, qResults));
260     	return feed;
261 	}

263     /*
264      * Generates the OpenSearch elements which are added to the RSS or Atom feeds as foreign markup
265      * wrapped in a module
266      * 
267      * @param query the search query
268      * @param qRes the search results
269      * @return module
270      */
271     private static OpenSearchModule openSearchMarkup(String query, QueryResults qRes)
272     { 
273     	OpenSearchModule osMod = new OpenSearchModuleImpl();
274     	osMod.setTotalResults(qRes.getHitCount());
275     	osMod.setStartIndex(qRes.getStart());
276     	osMod.setItemsPerPage(qRes.getPageSize());
277     	OSQuery osq = new OSQuery();
278     	osq.setRole("request");
279         try
280         {
281             osq.setSearchTerms(URLEncoder.encode(query, "UTF-8"));
282         	}
283         catch(UnsupportedEncodingException e)
284         	{
285             log.error(e);
286         	}
287         osq.setStartPage(1 + (qRes.getStart() / qRes.getPageSize()));
288         osMod.addQuery(osq);
289         return osMod;
290         	}

292     /**
293      * Returns as a document the OpenSearch service document
294      * 
295      * @param scope - null for the entire repository, or a collection/community handle
296      * @return Service Document
297      */
298     private static org.jdom.Document getServiceDocument(String scope)
299     {
300     	Namespace ns = Namespace.getNamespace(osNs);
301         Element root = new Element("OpenSearchDescription", ns);
302         root.addContent(new Element("ShortName", ns).setText(ConfigurationManager.getProperty("websvc.opensearch.shortname")));
303         root.addContent(new Element("LongName", ns).setText(ConfigurationManager.getProperty("websvc.opensearch.longname")));
304         root.addContent(new Element("Description", ns).setText(ConfigurationManager.getProperty("websvc.opensearch.description")));
305         root.addContent(new Element("InputEncoding", ns).setText("UTF-8"));
306         root.addContent(new Element("OutputEncoding", ns).setText("UTF-8"));
307         // optional elements
308         String sample = ConfigurationManager.getProperty("websvc.opensearch.samplequery");
309         if (sample != null && sample.length() > 0)
310         {
311         	Element sq = new Element("Query", ns).setAttribute("role", "example");
312         	root.addContent(sq.setAttribute("searchTerms", sample));
313         }
314         String tags = ConfigurationManager.getProperty("websvc.opensearch.tags");
315         if (tags != null && tags.length() > 0)
316         {
317         	root.addContent(new Element("Tags", ns).setText(tags));
318         }
319         String contact = ConfigurationManager.getProperty("mail.admin");
320         if (contact != null && contact.length() > 0)
321         {
322         	root.addContent(new Element("Contact", ns).setText(contact));
323         }
324         String faviconUrl = ConfigurationManager.getProperty("websvc.opensearch.faviconurl");
325         if (faviconUrl != null && faviconUrl.length() > 0)
326         {
327         	String dim = String.valueOf(16);
328         	String type = faviconUrl.endsWith("ico") ? "image/vnd.microsoft.icon" : "image/png";
329         	Element fav = new Element("Image", ns).setAttribute("height", dim).setAttribute("width", dim).
330         	                setAttribute("type", type).setText(faviconUrl);
331         	root.addContent(fav);
332         }
333         // service URLs
334         for (String format : formats)
335         {
336         	Element url = new Element("Url", ns).setAttribute("type", getContentType(format));
337         	StringBuffer template = new StringBuffer();
338         	if ("html".equals(format))
339         	{
340         		template.append(uiUrl);
341         	}
342         	else
343         	{
344         		template.append(svcUrl);
345         	}
346             template.append("?query={searchTerms}");
347         	if(! "html".equals(format))
348         	{
349                	template.append("&start={startIndex?}&rpp={count?}&format=");
350                	template.append(format);
351         	}
352         	if (scope != null)
353         	{
354         		template.append("&scope=");
355         		template.append(scope);
356         	}
357         	url.setAttribute("template", template.toString());
358         	root.addContent(url);
359         }
360         return new org.jdom.Document(root);
361     }
