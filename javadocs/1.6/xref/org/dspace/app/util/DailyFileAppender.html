50  /**
51   * Special log appender for log4j. Adds the current date (ie. year-mon) to
52   * the end of the file name, so that rolling on to the next log is simply
53   * a case of starting a new one - no renaming of old logs.
54   *
55   * This is advisable if you are using Windows, and have multiple applications
56   * (ie. dspace, dspace-oai, dspace-sword) that all want to write to the same log file,
57   * as each would otherwise try to rename the old files during rollover.
58   *
59   * An example log4j.properties (one log per month, retains three months of logs)
60   *
61   * log4j.rootCategory=INFO, A1
62   * log4j.appender.A1=org.dspace.app.util.DailyFileAppender
63   * log4j.appender.A1.File=@@log.dir@@/dspace.log
64   * log4j.appender.A1.DatePattern=yyyy-MM
65   * log4j.appender.A1.MaxLogs=3
66   * log4j.appender.A1.layout=org.apache.log4j.PatternLayout
67   * log4j.appender.A1.layout.ConversionPattern=%d %-5p %c @ %m%n
68   *
69   */

181     /* (non-Javadoc)
182      * @see org.apache.log4j.WriterAppender#subAppend(org.apache.log4j.spi.LoggingEvent)
183      */
184     protected void subAppend(LoggingEvent pobjEvent)
185     {
186         Date   dtNow = new Date(System.currentTimeMillis());
187 
188         boolean rollover = false;
189 
190         if (mMonthOnly)
191         {
192             Calendar now = Calendar.getInstance();
193             Calendar cur = Calendar.getInstance();
194             now.setTime(dtNow);
195             cur.setTime(mstrDate);
196             rollover = !(now.get(Calendar.YEAR) == cur.get(Calendar.YEAR) && now.get(Calendar.MONTH) == cur.get(Calendar.MONTH));
197         }
198         else
199         {
200             rollover = !(DateUtils.isSameDay(dtNow, mstrDate));
201         }
202 
203         if (rollover)
204         {
205             try
206             {
207                 rollOver(dtNow);
208             }
209             catch (IOException IOEx)
210             {
211                 LogLog.error("rollOver() failed!", IOEx);
212             }
213         }
214 
215         super.subAppend(pobjEvent);
216     }

221     /**
222      * The helper function to vaildate the DatePattern.
223      * @param pstrPattern The DatePattern to be validated.
224      * @return The validated date pattern or defautlt DATE_PATTERN
225      */
226     private String checkPattern(String pstrPattern)
227     {
228         String strRet = null;
229         SimpleDateFormat objFmt = new SimpleDateFormat(DATE_PATTERN);
230 
231         try
232         {
233             this.mobjSDF = new SimpleDateFormat(pstrPattern);
234             strRet = pstrPattern;
235         }
236         catch (NullPointerException NPExIgnore)
237         {
238             LogLog.error("Invalid DatePattern " + pstrPattern, NPExIgnore);
239             this.mobjSDF = objFmt;
240             strRet = DATE_PATTERN;
241         }
242         catch (IllegalArgumentException IlArgExIgnore)
243         {
244             LogLog.error("Invalid DatePattern " + pstrPattern, IlArgExIgnore);
245             this.mobjSDF = objFmt;
246             strRet = DATE_PATTERN;
247         }
248         finally
249         {
250             objFmt = null;
251         }
252         return strRet;
253     }

255     /**
256      * This function is responsible for performing the actual file rollover.
257      * @param pstrName The name of the new folder based on current system date.
258      * @throws IOException
259      */
260     static private boolean deletingFiles = false;
261     private void cleanupOldFiles()
262     {
263         // If we need to delete log files
264         if (mMaxLogs > 0 && !deletingFiles)
265         {
266 			deletingFiles = true;
267 
268 			// Determine the final file extension with the hostname
269     		String hostFileExt = null;
270             try
271             {
272                 hostFileExt = "." + java.net.InetAddress.getLocalHost().getHostName();
273             }
274             catch (UnknownHostException e)
275             {
276                 LogLog.error("Unable to retrieve host name");
277             }
278 
279 			try
280 			{
281 				// Array to hold the logs we are going to keep
282 				File[] logsToKeep = new File[mMaxLogs];
283 
284 				// Get a 'master' file handle, and the parent directory from it
285 		        File logMaster = new File(mstrFileName);
286 		        File logDir = logMaster.getParentFile();
287 		        if (logDir.isDirectory())
288 		        {
289 		        	// Iterate all the files in that directory
290 		        	File[] logArr = logDir.listFiles();
291 		        	for (File curLog : logArr)
292 		        	{
293         				log.info("Comparing '" + curLog.getAbsolutePath() + "' to '" + mstrFileName + "'");
294 		        		String name = curLog.getAbsolutePath();
295 
296 		        		// First, see if we are not using hostname, or the log file ends with this host
297 		                if (!mWithHostName || (hostFileExt != null && name.endsWith(hostFileExt)))
298 		                {
299 		                	// Check that the file is indeed one we want (contains the master file name)
300 			        		if (name.contains(mstrFileName))
301 			        		{
302 			        			// Iterate through the array of logs we are keeping
303 			        			for (int i = 0; curLog != null && i < logsToKeep.length; i++)
304 			        			{
305 			        				// Have we exhausted the 'to keep' array?
306 			        				if (logsToKeep[i] == null)
307 			        				{
308 				        				// Empty space, retain this log file
309 			        					logsToKeep[i] = curLog;
310 			        					curLog = null;
311 			        				}
312 			        				// If the 'kept' file is older than the current one
313 			        				else if (logsToKeep[i].getName().compareTo(curLog.getName()) < 0)
314 			        				{
315 			        					// Replace tested entry with current file
316 			        					File temp = logsToKeep[i];
317 			        					logsToKeep[i] = curLog;
318 			        					curLog = temp;
319 			        				}
320 			        			}
321 
322 			        			// If we have a 'current' entry at this point, it's a log we don't want
323 			        			if (curLog != null)
324 			        			{
325 			        				log.info("Deleting log " + curLog.getName());
326 		        					curLog.delete();
327 			        			}
328 			        		}
329 		                }
330 		        	}
331 		        }
332 			}
333 			catch (Exception e)
334 			{
335 				// Don't worry about exceptions
336 			}
337 			finally
338 			{
339 				deletingFiles = false;
340 			}
341         }
342     }

344     private void rollOver(Date dtNow) throws IOException
345     {
346         mstrDate = dtNow;
347         setFileName();
348         this.setFile(fileName, true, bufferedIO, bufferSize);
349 
350         cleanupOldFiles();
351     }
352 
353     private void setFileName()
354     {
355         fileName = mstrFileName + "." + mobjSDF.format(mstrDate);
356 
357         if (mWithHostName)
358         {
359             try
360             {
361                 fileName += "." + java.net.InetAddress.getLocalHost().getHostName();
362             }
363             catch (UnknownHostException e)
364             {
365                 LogLog.error("Unable to retrieve host name");
366             }
367         }
368     }
