62  	/**
63  	 * Utility method to convert spaces in a string to HTML non-break space
64  	 * elements.
65  	 * 
66  	 * @param s
67  	 *            string to change spaces in
68  	 * @return the string passed in with spaces converted to HTML non-break
69  	 *         spaces
70  	 */
71  	public static String nonBreakSpace(String s) {
72  	    StringBuffer newString = new StringBuffer();
73  	
74  	    for (int i = 0; i < s.length(); i++)
75  	    {
76  	        char ch = s.charAt(i);
77  	
78  	        if (ch == ' ')
79  	        {
80  	            newString.append("&nbsp;");
81  	        }
82  	        else
83  	        {
84  	            newString.append(ch);
85  	        }
86  	    }
87  	
88  	    return newString.toString();
89  	}

91  	/**
92  	 * Encode a bitstream name for inclusion in a URL in an HTML document. This
93  	 * differs from the usual URL-encoding, since we want pathname separators to
94  	 * be passed through verbatim; this is required so that relative paths in
95  	 * bitstream names and HTML references work correctly.
96  	 * <P>
97  	 * If the link to a bitstream is generated with the pathname separators
98  	 * escaped (e.g. "%2F" instead of "/") then the Web user agent perceives it
99  	 * to be one pathname element, and relative URI paths within that document
100 	 * containing ".." elements will be handled incorrectly.
101 	 * <P>
102 	 * 
103 	 * @param stringIn
104 	 *            input string to encode

106 	 *            character encoding, e.g. UTF-8
107 	 * @return the encoded string
108 	 */

111 	    StringBuffer out = new StringBuffer();

113 	    final String[] pctEncoding = { "%00", "%01", "%02", "%03", "%04",
114 	            "%05", "%06", "%07", "%08", "%09", "%0a", "%0b", "%0c", "%0d",
115 	            "%0e", "%0f", "%10", "%11", "%12", "%13", "%14", "%15", "%16",
116 	            "%17", "%18", "%19", "%1a", "%1b", "%1c", "%1d", "%1e", "%1f",
117 	            "%20", "%21", "%22", "%23", "%24", "%25", "%26", "%27", "%28",
118 	            "%29", "%2a", "%2b", "%2c", "%2d", "%2e", "%2f", "%30", "%31",
119 	            "%32", "%33", "%34", "%35", "%36", "%37", "%38", "%39", "%3a",
120 	            "%3b", "%3c", "%3d", "%3e", "%3f", "%40", "%41", "%42", "%43",
121 	            "%44", "%45", "%46", "%47", "%48", "%49", "%4a", "%4b", "%4c",
122 	            "%4d", "%4e", "%4f", "%50", "%51", "%52", "%53", "%54", "%55",
123 	            "%56", "%57", "%58", "%59", "%5a", "%5b", "%5c", "%5d", "%5e",
124 	            "%5f", "%60", "%61", "%62", "%63", "%64", "%65", "%66", "%67",
125 	            "%68", "%69", "%6a", "%6b", "%6c", "%6d", "%6e", "%6f", "%70",
126 	            "%71", "%72", "%73", "%74", "%75", "%76", "%77", "%78", "%79",
127 	            "%7a", "%7b", "%7c", "%7d", "%7e", "%7f", "%80", "%81", "%82",
128 	            "%83", "%84", "%85", "%86", "%87", "%88", "%89", "%8a", "%8b",
129 	            "%8c", "%8d", "%8e", "%8f", "%90", "%91", "%92", "%93", "%94",
130 	            "%95", "%96", "%97", "%98", "%99", "%9a", "%9b", "%9c", "%9d",
131 	            "%9e", "%9f", "%a0", "%a1", "%a2", "%a3", "%a4", "%a5", "%a6",
132 	            "%a7", "%a8", "%a9", "%aa", "%ab", "%ac", "%ad", "%ae", "%af",
133 	            "%b0", "%b1", "%b2", "%b3", "%b4", "%b5", "%b6", "%b7", "%b8",
134 	            "%b9", "%ba", "%bb", "%bc", "%bd", "%be", "%bf", "%c0", "%c1",
135 	            "%c2", "%c3", "%c4", "%c5", "%c6", "%c7", "%c8", "%c9", "%ca",
136 	            "%cb", "%cc", "%cd", "%ce", "%cf", "%d0", "%d1", "%d2", "%d3",
137 	            "%d4", "%d5", "%d6", "%d7", "%d8", "%d9", "%da", "%db", "%dc",
138 	            "%dd", "%de", "%df", "%e0", "%e1", "%e2", "%e3", "%e4", "%e5",
139 	            "%e6", "%e7", "%e8", "%e9", "%ea", "%eb", "%ec", "%ed", "%ee",
140 	            "%ef", "%f0", "%f1", "%f2", "%f3", "%f4", "%f5", "%f6", "%f7",
141 	            "%f8", "%f9", "%fa", "%fb", "%fc", "%fd", "%fe", "%ff" };
142 	
143 	    byte[] bytes = stringIn.getBytes(encoding);
144 	
145 	    for (int i = 0; i < bytes.length; i++)
146 	    {
147 	        // Any unreserved char or "/" goes through unencoded
148 	        if ((bytes[i] >= 'A' && bytes[i] <= 'Z')
149 	                || (bytes[i] >= 'a' && bytes[i] <= 'z')
150 	                || (bytes[i] >= '0' && bytes[i] <= '9') || bytes[i] == '-'
151 	                || bytes[i] == '.' || bytes[i] == '_' || bytes[i] == '~'
152 	                || bytes[i] == '/')
153 	        {
154 	            out.append((char) bytes[i]);
155 	        }
156 	        else if (bytes[i] >= 0)
157 	        {
158 	            // encode other chars (byte code < 128)
159 	            out.append(pctEncoding[bytes[i]]);
160 	        }
161 	        else
162 	        {
163 	            // encode other chars (byte code > 127, so it appears as
164 	            // negative in Java signed byte data type)
165 	            out.append(pctEncoding[256 + bytes[i]]);
166 	        }
167 	    }
168 	    log.debug("encoded \"" + stringIn + "\" to \"" + out.toString() + "\"");
169 	
170 	    return out.toString();
171 	}

184 	  * Formats the file size. Examples:
185 	  * 
186 	  *  - 50 = 50B
187 	  *  - 1024 = 1KB
188 	  *  - 1,024,000 = 1MB etc
189 	  *  
190 	  *  The numbers are formatted using java Locales
191 	  *  
192 	  * @param in The number to covnert
193 	  * @return the file size as a String
194 	  */
195 	public static String formatFileSize(double in) {
196 	     // Work out the size of the file, and format appropriatly
197 	     // FIXME: When full i18n support is available, use the user's Locale
198 	     // rather than the default Locale.
199 	     NumberFormat nf = NumberFormat.getNumberInstance(Locale.getDefault());
200 	     DecimalFormat df = (DecimalFormat)nf;
201 	     df.applyPattern("###,###.##");
202 	     if (in < 1024)
203 	     {
204 	         df.applyPattern("0");
205 	         return df.format(in) +  " " + "B";
206 	     }
207 	     else if (in < 1024000)
208 	     {
209 	         in = in / 1024;
210 	         return df.format(in) + " " + "kB";
211 	     }
212 	     else if (in < 1024000000)
213 	     {
214 	         in = in / 1024000;
215 	         return df.format(in) + " " + "MB";
216 	     }
217 	     else
218 	     {
219 	         in = in / 1024000000;
220 	         return df.format(in) + " " + "GB";
221 	     }
222 	 }

224     /**
225      * Obtain a parameter from the given request as an int. <code>-1</code> is
226      * returned if the parameter is garbled or does not exist.
227      * 
228      * @param request
229      *            the HTTP request
230      * @param param
231      *            the name of the parameter
232      * 
233      * @return the integer value of the parameter, or -1
234      */
235     public static int getIntParameter(HttpServletRequest request, String param)
236     {
237         String val = request.getParameter(param);
238 
239         try
240         {
241             return Integer.parseInt(val.trim());
242         }
243         catch (Exception e)
244         {
245             // Problem with parameter
246             return -1;
247         }
248     }

250     /**
251      * Obtain an array of int parameters from the given request as an int. null
252      * is returned if parameter doesn't exist. <code>-1</code> is returned in
253      * array locations if that particular value is garbled.
254      * 
255      * @param request
256      *            the HTTP request
257      * @param param
258      *            the name of the parameter
259      * 
260      * @return array of integers or null
261      */
262     public static int[] getIntParameters(HttpServletRequest request,
263             String param)
264     {
265         String[] request_values = request.getParameterValues(param);
266 
267         if (request_values == null)
268         {
269             return null;
270         }
271 
272         int[] return_values = new int[request_values.length];
273 
274         for (int x = 0; x < return_values.length; x++)
275         {
276             try
277             {
278                 return_values[x] = Integer.parseInt(request_values[x]);
279             }
280             catch (Exception e)
281             {
282                 // Problem with parameter, stuff -1 in this slot
283                 return_values[x] = -1;
284             }
285         }
286 
287         return return_values;
288     }

309     /**
310      * Get the button the user pressed on a submitted form. All buttons should
311      * start with the text <code>submit</code> for this to work. A default
312      * should be supplied, since often the browser will submit a form with no
313      * submit button pressed if the user presses enter.
314      * 
315      * @param request
316      *            the HTTP request
317      * @param def
318      *            the default button
319      * 
320      * @return the button pressed
321      */
322     public static String getSubmitButton(HttpServletRequest request, String def)
323     {
324         Enumeration e = request.getParameterNames();
325 
326         while (e.hasMoreElements())
327         {
328             String parameterName = (String) e.nextElement();
329 
330             if (parameterName.startsWith("submit"))
331             {
332                 return parameterName;
333             }
334         }
335 
336         return def;
337     }
