71      public int authenticate(Context context, String username, String password,
72              String realm, HttpServletRequest request) throws SQLException
73      {
74          log.info("Shibboleth login started...");
75  
76          java.util.Enumeration names = request.getHeaderNames();
77          String name;
78          while (names.hasMoreElements())
79              log.debug("header:" + (name = names.nextElement().toString()) + "="
80                      + request.getHeader(name));
81  
82          boolean isUsingTomcatUser = ConfigurationManager
83                  .getBooleanProperty("authentication.shib.email-use-tomcat-remote-user");
84          String emailHeader = ConfigurationManager
85                  .getProperty("authentication.shib.email-header");
86          String fnameHeader = ConfigurationManager
87                  .getProperty("authentication.shib.firstname-header");
88          String lnameHeader = ConfigurationManager
89                  .getProperty("authentication.shib.lastname-header");
90  
91          String email = null;
92          String fname = null;
93          String lname = null;
94  
95          if (emailHeader != null)
96          {
97              // try to grab email from the header
98              email = request.getHeader(emailHeader);
99  
100             // fail, try lower case
101             if (email == null)
102                 email = request.getHeader(emailHeader.toLowerCase());
103         }
104 
105         // try to pull the "REMOTE_USER" info instead of the header
106         if (email == null && isUsingTomcatUser)
107         {
108             email = request.getRemoteUser();
109             log.info("RemoteUser identified as: " + email);
110         }
111 
112         // No email address, perhaps the eperson has been setup, better check it
113         if (email == null)
114         {
115             EPerson p = context.getCurrentUser();
116             if (p != null)
117                 email = p.getEmail();
118         }
119 
120         if (email == null)
121         {
122             log
123                     .error("No email is given, you're denied access by Shib, please release email address");
124             return AuthenticationMethod.BAD_ARGS;
125         }
126 
127         email = email.toLowerCase();
128 
129         if (fnameHeader != null)
130         {
131             // try to grab name from the header
132             fname = request.getHeader(fnameHeader);
133 
134             // fail, try lower case
135             if (fname == null)
136                 fname = request.getHeader(fnameHeader.toLowerCase());
137         }
138         if (lnameHeader != null)
139         {
140             // try to grab name from the header
141             lname = request.getHeader(lnameHeader);
142 
143             // fail, try lower case
144             if (lname == null)
145                 lname = request.getHeader(lnameHeader.toLowerCase());
146         }
147 
148         // future version can offer auto-update feature, this needs testing
149         // before inclusion to core code
150 
151         EPerson eperson = null;
152         try
153         {
154             eperson = EPerson.findByEmail(context, email);
155             context.setCurrentUser(eperson);
156         }
157         catch (AuthorizeException e)
158         {
159             log.warn("Fail to locate user with email:" + email, e);
160             eperson = null;
161         }
162 
163         // auto create user if needed
164         if (eperson == null
165                 && ConfigurationManager
166                         .getBooleanProperty("authentication.shib.autoregister"))
167         {
168             log.info(LogManager.getHeader(context, "autoregister", "email="
169                     + email));
170 
171             // TEMPORARILY turn off authorisation
172             context.setIgnoreAuthorization(true);
173             try
174             {
175                 eperson = EPerson.create(context);
176                 eperson.setEmail(email);
177                 if (fname != null)
178                     eperson.setFirstName(fname);
179                 if (lname != null)
180                     eperson.setLastName(lname);
181                 eperson.setCanLogIn(true);
182                 AuthenticationManager.initEPerson(context, request, eperson);
183                 eperson.update();
184                 context.commit();
185                 context.setCurrentUser(eperson);
186             }
187             catch (AuthorizeException e)
188             {
189                 log.warn("Fail to authorize user with email:" + email, e);
190                 eperson = null;
191             }
192             finally
193             {
194                 context.setIgnoreAuthorization(false);
195             }
196         }
197 
198         if (eperson == null)
199         {
200             return AuthenticationMethod.NO_SUCH_USER;
201         }
202         else
203         {
204             // the person exists, just return ok
205             context.setCurrentUser(eperson);
206             request.getSession().setAttribute("shib.authenticated",
207                     new Boolean("true"));
208         }
209 
210         return AuthenticationMethod.SUCCESS;
211     }

213     /**
214      * Grab the special groups to be automatically provisioned for the current
215      * user. Currently the mapping for the groups is done one-to-one, future
216      * version can consider the usage of regex for such mapping.
217      */
218     public int[] getSpecialGroups(Context context, HttpServletRequest request)
219     {
220         // no user logged in or user not logged from shibboleth
221         if (context.getCurrentUser() == null
222                 || request.getSession().getAttribute("shib.authenticated") == null)
223         {
224             return new int[0];
225         }
226                 
227         if (request.getSession().getAttribute("shib.specialgroup") != null)
228         {
229             return (int[]) request.getSession().getAttribute(
230                     "shib.specialgroup");
231         }
232 
233         java.util.Set groups = new java.util.HashSet();
234         String roleHeader = ConfigurationManager
235                 .getProperty("authentication.shib.role-header");
236         boolean roleHeader_ignoreScope = ConfigurationManager
237                 .getBooleanProperty("authentication.shib.role-header.ignore-scope");
238         if (roleHeader == null || roleHeader.trim().length() == 0)
239             roleHeader = "Shib-EP-UnscopedAffiliation"; // fall back to default
240         String affiliations = request.getHeader(roleHeader);
241 
242         // try again with all lower case...maybe has better luck
243         if (affiliations == null)
244             affiliations = request.getHeader(roleHeader.toLowerCase());
245 
246         // default role when fully authN but not releasing any roles?
247         String defaultRoles = ConfigurationManager
248                 .getProperty("authentication.shib.default-roles");
249         if (affiliations == null && defaultRoles != null)
250         {
251             affiliations = defaultRoles;
252         }
253 
254         if (affiliations != null)
255         {
256             java.util.StringTokenizer st = new java.util.StringTokenizer(
257                     affiliations, ";,");
258             // do the mapping here
259             while (st.hasMoreTokens())
260             {
261                 String affiliation = st.nextToken().trim();
262 
263                 // strip scope if present and roleHeader_ignoreScope
264                 if (roleHeader_ignoreScope) 
265                 {
266                         int index = affiliation.indexOf("@");
267                         if (index != -1) affiliation = affiliation.substring(0,index);
268                 }
269 
270                 // perform mapping here if necessary
271                 String groupLabels = ConfigurationManager
272                         .getProperty("authentication.shib.role." + affiliation);
273                 if (groupLabels == null || groupLabels.trim().length() == 0)
274                     groupLabels = ConfigurationManager
275                             .getProperty("authentication.shib.role."
276                                     + affiliation.toLowerCase());
277 
278                 // revert back to original entry when no mapping is provided
279                 if (groupLabels == null)
280                     groupLabels = affiliation;
281 
282                 String[] labels = groupLabels.split(",");
283                 for (int i = 0; i < labels.length; i++)
284                     addGroup(groups, context, labels[i].trim());
285             }
286         }
287 
288         int ids[] = new int[groups.size()];
289         java.util.Iterator it = groups.iterator();
290         for (int i = 0; it.hasNext(); i++)
291             ids[i] = ((Integer) it.next()).intValue();
292 
293         // store the special group, if already transformed from headers
294         // since subsequent header may not have the values anymore
295         if (ids.length != 0)
296         {
297             request.getSession().setAttribute("shib.specialgroup", ids);
298         }
299 
300         return ids;
301     }

303     /** Find dspaceGroup in DSpace database, if found, include it into groups */
304     private void addGroup(Collection groups, Context context, String dspaceGroup)
305     {
306         try
307         {
308             Group g = Group.findByName(context, dspaceGroup);
309             if (g == null)
310             {
311                 // oops - no group defined
312                 log.warn(LogManager.getHeader(context, dspaceGroup
313                         + " group is not found!! Admin needs to create one!",
314                         "requiredGroup=" + dspaceGroup));
315                 groups.add(new Integer(0));
316             }
317             else
318             {
319                 groups.add(new Integer(g.getID()));
320             }
321             log.info("Mapping group: " + dspaceGroup + " to groupID: "
322                     + (g == null ? 0 : g.getID()));
323         }
324         catch (SQLException e)
325         {
326             log.error("Mapping group:" + dspaceGroup + " failed with error", e);
327         }
328     }
