72  /**
73   * Implicit authentication method that gets credentials from the X.509 client
74   * certificate supplied by the HTTPS client when connecting to this server. The
75   * email address in that certificate is taken as the authenticated user name
76   * with no further checking, so be sure your HTTP server (e.g. Tomcat) is
77   * configured correctly to accept only client certificates it can validate.
78   * <p>
79   * See the <code>AuthenticationMethod</code> interface for more details.
80   * <p>
81   * <b>Configuration:</b>
82   * 
83   * <pre>
84   *   authentication.x509.keystore.path =
85   * <em>
86   * path to Java keystore file
87   * </em>
88   *   authentication.x509.keystore.password =
89   * <em>
90   * password to access the keystore
91   * </em>
92   *   authentication.x509.ca.cert =
93   * <em>
94   * path to certificate file for CA whose client certs to accept.
95   * </em>
96   *   authentication.x509.autoregister =
97   * <em>
98   * &quot;true&quot; if E-Person is created automatically for unknown new users.
99   * </em>
100  *   authentication.x509.groups = 
101  * <em>
102  * comma-delimited list of special groups to add user to if authenticated.
103  * </em>
104  *   authentication.x509.emaildomain = 
105  * <em>
106  * email address domain (after the 'at' symbol) to match before allowing 
107  * membership in special groups.
108  * </em>
109  * </pre>
110  * 
111  * Only one of the "<code>keystore.path</code>" or "<code>ca.cert</code>"
112  * options is required. If you supply a keystore, then all of the "trusted"
113  * certificates in the keystore represent CAs whose client certificates will be
114  * accepted. The <code>ca.cert</code> option only allows a single CA to be
115  * named.
116  * <p>
117  * You can configure <em>both</em> a keystore and a CA cert, and both will be
118  * used.
119  * <p>
120  * The <code>autoregister</code> configuration parameter determines what the
121  * <code>canSelfRegister()</code> method returns. It also allows an EPerson
122  * record to be created automatically when the presented certificate is
123  * acceptable but there is no corresponding EPerson.
124  * 
125  * @author Larry Stone
126  * @version $Revision: 3735 $
127  */
128 public class X509Authentication implements AuthenticationMethod

144     /**
145      * Initialization: Set caPublicKey and/or keystore. This loads the
146      * information needed to check if a client cert presented is valid and
147      * acceptable.
148      */
149     static
150     {
151         /*
152          * allow identification of alternative entry points for certificate
153          * authentication when selected by the user rather than implicitly.
154          */
155         loginPageTitle = ConfigurationManager
156                 .getProperty("authentication.x509.chooser.title.key");
157         loginPageURL = ConfigurationManager
158                 .getProperty("authentication.x509.chooser.uri");
159 
160         String keystorePath = ConfigurationManager
161                 .getProperty("authentication.x509.keystore.path");
162         String keystorePassword = ConfigurationManager
163                 .getProperty("authentication.x509.keystore.password");
164         String caCertPath = ConfigurationManager
165                 .getProperty("authentication.x509.ca.cert");
166 
167         // backward-compatible kludge
168         if (caCertPath == null)
169             caCertPath = ConfigurationManager.getProperty("webui.cert.ca");
170 
171         // First look for keystore full of trusted certs.
172         if (keystorePath != null)
173         {
174             FileInputStream fis = null;
175             if (keystorePassword == null)
176                 keystorePassword = "";
177             try
178             {
179                 KeyStore ks = KeyStore.getInstance("JKS");
180                 fis = new FileInputStream(keystorePath);
181                 ks.load(fis, keystorePassword.toCharArray());
182                 caCertKeyStore = ks;
183             }
184             catch (IOException e)
185             {
186                 log
187                         .error("X509Authentication: Failed to load CA keystore, file="
188                                 + keystorePath + ", error=" + e.toString());
189             }
190             catch (GeneralSecurityException e)
191             {
192                 log
193                         .error("X509Authentication: Failed to extract CA keystore, file="
194                                 + keystorePath + ", error=" + e.toString());
195             }
196             finally
197             {
198                 if (fis != null)
199                     try
200                     {
201                         fis.close();
202                     }
203                     catch (IOException ioe)
204                     {
205                     }
206             }
207         }
208 
209         // Second, try getting public key out of CA cert, if that's configured.
210         if (caCertPath != null)
211         {
212             InputStream is = null;
213             FileInputStream fis = null;
214             try
215             {
216                 fis = new FileInputStream(caCertPath);
217                 is = new BufferedInputStream(fis);
218                 X509Certificate cert = (X509Certificate) CertificateFactory
219                         .getInstance("X.509").generateCertificate(is);
220                 if (cert != null)
221                     caPublicKey = cert.getPublicKey();
222             }
223             catch (IOException e)
224             {
225                 log.error("X509Authentication: Failed to load CA cert, file="
226                         + caCertPath + ", error=" + e.toString());
227             }
228             catch (CertificateException e)
229             {
230                 log
231                         .error("X509Authentication: Failed to extract CA cert, file="
232                                 + caCertPath + ", error=" + e.toString());
233             }
234             finally
235             {
236                 if (is != null)
237                     try
238                     {
239                         is.close();
240                     }
241                     catch (IOException ioe)
242                     {
243                     }
244 
245                 if (fis != null)
246                     try
247                     {
248                         fis.close();
249                     }
250                     catch (IOException ioe)
251                     {
252                     }
253             }
254         }

257     /**
258      * Return the email address from <code>certificate</code>, or null if an
259      * email address cannot be found in the certificate.
260      * <p>
261      * Note that the certificate parsing has only been tested with certificates
262      * granted by the MIT Certification Authority, and may not work elsewhere.
263      * 
264      * @param certificate -
265      *            An X509 certificate object
266      * @return - The email address found in certificate, or null if an email
267      *         address cannot be found in the certificate.
268      */
269     private static String getEmail(X509Certificate certificate)
270             throws SQLException
271     {
272         Principal principal = certificate.getSubjectDN();
273 
274         if (principal == null)
275             return null;
276 
277         String dn = principal.getName();
278         if (dn == null)
279             return null;
280 
281         StringTokenizer tokenizer = new StringTokenizer(dn, ",");
282         String token = null;
283         while (tokenizer.hasMoreTokens())
284         {
285             int len = "emailaddress=".length();
286 
287             token = (String) tokenizer.nextToken();
288 
289             if (token.toLowerCase().startsWith("emailaddress="))
290             {
291                 // Make sure the token actually contains something
292                 if (token.length() <= len)
293                     return null;
294 
295                 return token.substring(len).toLowerCase();
296             }
297         }
298 
299         return null;
300     }

302     /**
303      * Verify CERTIFICATE against KEY. Return true if and only if CERTIFICATE is
304      * valid and can be verified against KEY.
305      * 
306      * @param certificate -
307      *            An X509 certificate object
308      * @param key -
309      *            PublicKey to check the certificate against.
310      * @return - True if CERTIFICATE is valid and can be verified against KEY,
311      *         false otherwise.
312      */
313     private static boolean isValid(Context context, X509Certificate certificate)
314     {
315         if (certificate == null)
316             return false;
317 
318         // This checks that current time is within cert's validity window:
319         try
320         {
321             certificate.checkValidity();
322         }
323         catch (CertificateException e)
324         {
325             log.info(LogManager.getHeader(context, "authentication",
326                     "X.509 Certificate is EXPIRED or PREMATURE: "
327                             + e.toString()));
328             return false;
329         }
330 
331         // Try CA public key, if available.
332         if (caPublicKey != null)
333         {
334             try
335             {
336                 certificate.verify(caPublicKey);
337                 return true;
338             }
339             catch (GeneralSecurityException e)
340             {
341                 log.info(LogManager.getHeader(context, "authentication",
342                         "X.509 Certificate FAILED SIGNATURE check: "
343                                 + e.toString()));
344             }
345         }
346 
347         // Try it with keystore, if available.
348         if (caCertKeyStore != null)
349         {
350             try
351             {
352                 Enumeration ke = caCertKeyStore.aliases();
353 
354                 while (ke.hasMoreElements())
355                 {
356                     String alias = (String) ke.nextElement();
357                     if (caCertKeyStore.isCertificateEntry(alias))
358                     {
359                         Certificate ca = caCertKeyStore.getCertificate(alias);
360                         try
361                         {
362                             certificate.verify(ca.getPublicKey());
363                             return true;
364                         }
365                         catch (CertificateException ce)
366                         {
367                         }
368                     }
369                 }
370                 log
371                         .info(LogManager
372                                 .getHeader(context, "authentication",
373                                         "Keystore method FAILED SIGNATURE check on client cert."));
374             }
375             catch (GeneralSecurityException e)
376             {
377                 log.info(LogManager.getHeader(context, "authentication",
378                         "X.509 Certificate FAILED SIGNATURE check: "
379                                 + e.toString()));
380             }
381 
382         }
383         return false;
384     }

424     /**
425      * Returns a list of group names that the user should be added to upon
426      * successful authentication, configured in dspace.cfg.
427      * 
428      * @return List<String> of special groups configured for this authenticator
429      */
430     private List<String> getX509Groups()
431     {
432         List<String> groupNames = new ArrayList<String>();
433 
434         String x509GroupConfig = null;
435         x509GroupConfig = ConfigurationManager
436                 .getProperty("authentication.x509.groups");
437 
438         if (null != x509GroupConfig && !x509GroupConfig.equals(""))
439         {
440             String[] groups = x509GroupConfig.split("\\s*,\\s*");
441 
442             for (int i = 0; i < groups.length; i++)
443             {
444                 groupNames.add(groups[i].trim());
445             }
446         }
447 
448         return groupNames;
449     }

451     /**
452      * Checks for configured email domain required to grant special groups
453      * membership. If no email domain is configured to verify, special group
454      * membership is simply granted.
455      * 
456      * @param request -
457      *            The current request object
458      * @param email -
459      *            The email address from the x509 certificate
460      */
461     private void setSpecialGroupsFlag(HttpServletRequest request, String email)
462     {
463         String emailDomain = null;
464         emailDomain = (String) request
465                 .getAttribute("authentication.x509.emaildomain");
466 
467         HttpSession session = request.getSession(true);
468 
469         if (null != emailDomain && !emailDomain.equals(""))
470         {
471             if (email.substring(email.length() - emailDomain.length()).equals(
472                     emailDomain))
473             {
474                 session.setAttribute("x509Auth", new Boolean(true));
475             }
476         }
477         else
478         {
479             // No configured email domain to verify. Just flag
480             // as authenticated so special groups are granted.
481             session.setAttribute("x509Auth", new Boolean(true));
482         }
483     }

485     /**
486      * Return special groups configured in dspace.cfg for X509 certificate
487      * authentication.
488      * 
489      * @param Context
490      * @param HttpServletRequest
491      *            object potentially containing the cert
492      * 
493      * @return An int array of group IDs
494      * 
495      */
496     public int[] getSpecialGroups(Context context, HttpServletRequest request)
497             throws SQLException
498     {
499 
500         Boolean authenticated = false;
501         HttpSession session = request.getSession(false);
502         authenticated = (Boolean) session.getAttribute("x509Auth");
503         authenticated = (null == authenticated) ? false : authenticated;
504 
505         if (authenticated)
506         {
507             List<String> groupNames = new ArrayList<String>();
508             List<Integer> groupIDs = new ArrayList<Integer>();
509 
510             groupNames = getX509Groups();
511 
512             for (String groupName : groupNames)
513             {
514                 if (groupName != null)
515                 {
516                     Group group = Group.findByName(context, groupName);
517                     if (group != null)
518                     {
519                         groupIDs.add(new Integer(group.getID()));
520                     }
521                     else
522                     {
523                         log.warn(LogManager.getHeader(context,
524                                 "configuration_error", "unknown_group="
525                                         + groupName));
526                     }
527                 }
528             }
529 
530             int[] results = new int[groupIDs.size()];
531             for (int i = 0; i < groupIDs.size(); i++)
532             {
533                 results[i] = (groupIDs.get(i)).intValue();
534             }
535 
536             if (log.isDebugEnabled())
537             {
538                 StringBuffer gsb = new StringBuffer();
539 
540                 for (int i = 0; i < results.length; i++)
541                 {
542                     if (i > 0)
543                         gsb.append(",");
544                     gsb.append(results[i]);
545                 }
546 
547                 log.debug(LogManager.getHeader(context, "authenticated",
548                         "special_groups=" + gsb.toString()));
549             }
550 
551             return results;
552         }
553 
554         return new int[0];
555     }

557     /**
558      * X509 certificate authentication. The client certificate is obtained from
559      * the <code>ServletRequest</code> object.
560      * <ul>
561      * <li>If the certificate is valid, and corresponds to an existing EPerson,
562      * and the user is allowed to login, return success.</li>
563      * <li>If the user is matched but is not allowed to login, it fails.</li>
564      * <li>If the certificate is valid, but there is no corresponding EPerson,
565      * the <code>"authentication.x509.autoregister"</code> configuration
566      * parameter is checked (via <code>canSelfRegister()</code>)
567      * <ul>
568      * <li>If it's true, a new EPerson record is created for the certificate,
569      * and the result is success.</li>
570      * <li>If it's false, return that the user was unknown.</li>
571      * </ul>
572      * </li>
573      * </ul>
574      * 
575      * @return One of: SUCCESS, BAD_CREDENTIALS, NO_SUCH_USER, BAD_ARGS
576      */
577     public int authenticate(Context context, String username, String password,
578             String realm, HttpServletRequest request) throws SQLException
579     {
580         // Obtain the certificate from the request, if any
581         X509Certificate[] certs = null;
582         if (request != null)
583             certs = (X509Certificate[]) request
584                     .getAttribute("javax.servlet.request.X509Certificate");
585 
586         if ((certs == null) || (certs.length == 0))
587             return BAD_ARGS;
588         else
589         {
590             // We have a cert -- check it and get username from it.
591             try
592             {
593                 if (!isValid(context, certs[0]))
594                 {
595                     log
596                             .warn(LogManager
597                                     .getHeader(context, "authenticate",
598                                             "type=x509certificate, status=BAD_CREDENTIALS (not valid)"));
599                     return BAD_CREDENTIALS;
600                 }
601 
602                 // And it's valid - try and get an e-person
603                 String email = getEmail(certs[0]);
604                 EPerson eperson = null;
605                 if (email != null)
606                     eperson = EPerson.findByEmail(context, email);
607                 if (eperson == null)
608                 {
609                     // Cert is valid, but no record.
610                     if (email != null
611                             && canSelfRegister(context, request, null))
612                     {
613                         // Register the new user automatically
614                         log.info(LogManager.getHeader(context, "autoregister",
615                                 "from=x.509, email=" + email));
616 
617                         // TEMPORARILY turn off authorisation
618                         context.setIgnoreAuthorization(true);
619                         eperson = EPerson.create(context);
620                         eperson.setEmail(email);
621                         eperson.setCanLogIn(true);
622                         AuthenticationManager.initEPerson(context, request,
623                                 eperson);
624                         eperson.update();
625                         context.commit();
626                         context.setIgnoreAuthorization(false);
627                         context.setCurrentUser(eperson);
628                         setSpecialGroupsFlag(request, email);
629                         return SUCCESS;
630                     }
631                     else
632                     {
633                         // No auto-registration for valid certs
634                         log
635                                 .warn(LogManager
636                                         .getHeader(context, "authenticate",
637                                                 "type=cert_but_no_record, cannot auto-register"));
638                         return NO_SUCH_USER;
639                     }
640                 }
641 
642                 // make sure this is a login account
643                 else if (!eperson.canLogIn())
644                 {
645                     log.warn(LogManager.getHeader(context, "authenticate",
646                             "type=x509certificate, email=" + email
647                                     + ", canLogIn=false, rejecting."));
648                     return BAD_ARGS;
649                 }
650 
651                 else
652                 {
653                     log.info(LogManager.getHeader(context, "login",
654                             "type=x509certificate"));
655                     context.setCurrentUser(eperson);
656                     setSpecialGroupsFlag(request, email);
657                     return SUCCESS;
658                 }
659             }
660             catch (AuthorizeException ce)
661             {
662                 log.warn(LogManager.getHeader(context, "authorize_exception",
663                         ""), ce);
664             }
665 
666             return BAD_ARGS;
667         }
668     }
