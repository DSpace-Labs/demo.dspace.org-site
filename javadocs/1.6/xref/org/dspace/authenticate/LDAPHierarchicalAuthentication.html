57  /**
58   * This LDAP authentication method is more complex than the simple 'LDAPAuthentication'
59   * in that it allows authentication against structured heirarchical LDAP trees of
60   * users. An initial bind is required using a user name and password in order to
61   * searchthe tree and find the DN of the user. A second bind is then required to
62   * chack the credentials of the user by binding directly to their DN.
63   *
64   * @author Stuart Lewis, Chris Yates, Alex Barbieri, Flavio Botelho, Reuben Pasquini
65   * @version $Revision: 3735 $
66   */
67  public class LDAPHierarchicalAuthentication
68      implements AuthenticationMethod {

116     /*
117      * Add authenticated users to the group defined in dspace.cfg by
118      * the ldap.login.specialgroup key.
119      */
120     public int[] getSpecialGroups(Context context, HttpServletRequest request)
121     {
122 		// Prevents anonymous users from being added to this group, and the second check
123 		// ensures they are LDAP users
124 		try
125 		{
126 			if (!context.getCurrentUser().getNetid().equals(""))
127 			{
128 				String groupName = ConfigurationManager.getProperty("ldap.login.specialgroup");
129 				if ((groupName != null) && (!groupName.trim().equals("")))
130 				{
131 					Group ldapGroup = Group.findByName(context, groupName);
132 					if (ldapGroup == null)
133 					{
134 						// Oops - the group isn't there.
135 						log.warn(LogManager.getHeader(context,
136 								"ldap_specialgroup",
137 								"Group defined in ldap.login.specialgroup does not exist"));
138 						return new int[0];
139 					} else
140 					{
141 						return new int[] { ldapGroup.getID() };
142 					}
143 				}
144 			}
145 		}
146 		catch (Exception npe) {
147 			// The user is not an LDAP user, so we don't need to worry about them
148 		}
149 		return new int[0];
150     }

152     /*
153      * Authenticate the given credentials.
154      * This is the heart of the authentication method: test the
155      * credentials for authenticity, and if accepted, attempt to match
156      * (or optionally, create) an <code>EPerson</code>.  If an <code>EPerson</code> is found it is
157      * set in the <code>Context</code> that was passed.
158      *
159      * @param context
160      *  DSpace context, will be modified (ePerson set) upon success.
161      *
162      * @param username
163      *  Username (or email address) when method is explicit. Use null for
164      *  implicit method.
165      *
166      * @param password
167      *  Password for explicit auth, or null for implicit method.
168      *
169      * @param realm
170      *  Realm is an extra parameter used by some authentication methods, leave null if
171      *  not applicable.
172      *
173      * @param request
174      *  The HTTP request that started this operation, or null if not applicable.
175      *
176      * @return One of:
177      *   SUCCESS, BAD_CREDENTIALS, CERT_REQUIRED, NO_SUCH_USER, BAD_ARGS
178      * <p>Meaning:
179      * <br>SUCCESS         - authenticated OK.
180      * <br>BAD_CREDENTIALS - user exists, but credentials (e.g. passwd) don't match
181      * <br>CERT_REQUIRED   - not allowed to login this way without X.509 cert.
182      * <br>NO_SUCH_USER    - user not found using this method.
183      * <br>BAD_ARGS        - user/pw not appropriate for this method
184      */
185     public int authenticate(Context context,
186                             String netid,
187                             String password,
188                             String realm,
189                             HttpServletRequest request)
190         throws SQLException
191     {
192         log.info(LogManager.getHeader(context, "auth", "attempting trivial auth of user="+netid));
193 
194         // Skip out when no netid or password is given.
195         if (netid == null || password == null)
196         	return BAD_ARGS;
197 
198         // Locate the eperson
199         EPerson eperson = null;
200         try
201         {
202         		eperson = EPerson.findByNetid(context, netid.toLowerCase());
203         }
204         catch (SQLException e)
205         {
206         }
207         SpeakerToLDAP ldap = new SpeakerToLDAP(log);
208 
209 		// Get the DN of the user
210 		String adminUser = ConfigurationManager.getProperty("ldap.search.user");
211 		String adminPassword = ConfigurationManager.getProperty("ldap.search.password");
212 		String dn = ldap.getDNOfUser(adminUser, adminPassword, context, netid);
213 
214 		// Check a DN was found
215 		if ((dn == null) || (dn.trim().equals("")))
216 		{
217 			log.info(LogManager
218 				.getHeader(context, "failed_login", "no DN found for user " + netid));
219 			return BAD_CREDENTIALS;
220 		}
221 
222 		// if they entered a netid that matches an eperson
223         if (eperson != null)
224         {
225             // e-mail address corresponds to active account
226             if (eperson.getRequireCertificate())
227                 return CERT_REQUIRED;
228             else if (!eperson.canLogIn())
229                 return BAD_ARGS;
230             {
231                 if (ldap.ldapAuthenticate(dn, password, context))
232                 {
233                     context.setCurrentUser(eperson);
234                     log.info(LogManager
235                         .getHeader(context, "authenticate", "type=ldap"));
236                     return SUCCESS;
237                 }
238                 else
239                    return BAD_CREDENTIALS;
240             }
241         }
242 
243         // the user does not already exist so try and authenticate them
244         // with ldap and create an eperson for them
245         else
246         {
247             if (ldap.ldapAuthenticate(dn, password, context))
248             {
249                 // Register the new user automatically
250                 log.info(LogManager.getHeader(context,
251                                 "autoregister", "netid=" + netid));
252 
253 				if ((ldap.ldapEmail!=null)&&(!ldap.ldapEmail.equals("")))
254 				{
255 					try
256 					{
257 						eperson = EPerson.findByEmail(context, ldap.ldapEmail);
258 						if (eperson!=null)
259 						{
260 							log.info(LogManager.getHeader(context,
261 									"type=ldap-login", "type=ldap_but_already_email"));
262 							context.setIgnoreAuthorization(true);
263 							eperson.setNetid(netid.toLowerCase());
264 							eperson.update();
265 							context.commit();
266 							context.setIgnoreAuthorization(false);
267 							context.setCurrentUser(eperson);
268 							return SUCCESS;
269 						}
270 						else
271 						{
272 							if (canSelfRegister(context, request, netid))
273 							{
274 								// TEMPORARILY turn off authorisation
275 								try
276 								{
277 									context.setIgnoreAuthorization(true);
278 									eperson = EPerson.create(context);
279 									if ((ldap.ldapEmail != null) && (!ldap.ldapEmail.equals("")))
280 									{
281 										eperson.setEmail(ldap.ldapEmail);
282 									}
283 									else
284 									{
285 										eperson.setEmail(netid + ConfigurationManager.getProperty("ldap.netid_email_domain"));
286 									}
287 									if ((ldap.ldapGivenName!=null) && (!ldap.ldapGivenName.equals("")))
288 									{
289 										eperson.setFirstName(ldap.ldapGivenName);
290 									}
291 									if ((ldap.ldapSurname!=null) && (!ldap.ldapSurname.equals("")))
292 									{
293 										eperson.setLastName(ldap.ldapSurname);
294 									}
295 									if ((ldap.ldapPhone!=null)&&(!ldap.ldapPhone.equals("")))
296 									{
297 										eperson.setMetadata("phone", ldap.ldapPhone);
298 									}
299 									eperson.setNetid(netid.toLowerCase());
300 									eperson.setCanLogIn(true);
301 									AuthenticationManager.initEPerson(context, request, eperson);
302 									eperson.update();
303 									context.commit();
304 									context.setCurrentUser(eperson);
305 								}
306 								catch (AuthorizeException e)
307 								{
308 									return NO_SUCH_USER;
309 								}
310 								finally
311 								{
312 									context.setIgnoreAuthorization(false);
313 								}
314 
315 								log.info(LogManager.getHeader(context, "authenticate",
316 											"type=ldap-login, created ePerson"));
317 								return SUCCESS;
318 							}
319 							else
320 							{
321 								// No auto-registration for valid certs
322 								log.info(LogManager.getHeader(context,
323 												"failed_login", "type=ldap_but_no_record"));
324 								return NO_SUCH_USER;
325 							}
326 						}
327 					}
328 					catch (AuthorizeException e)
329 					{
330 						eperson = null;
331 					}
332 					finally
333 					{
334 						context.setIgnoreAuthorization(false);
335 					}
336 				}
337 			}
338         }
339         return BAD_ARGS;
340     }

342     /**
343      * Internal class to manage LDAP query and results, mainly
344      * because there are multiple values to return.
345      */
346     public class SpeakerToLDAP {
347 
348         private Logger log = null;
349 
350         protected String ldapEmail = null;
351         protected String ldapGivenName = null;
352         protected String ldapSurname = null;
353         protected String ldapPhone = null;
354 
355 		/** LDAP settings */
356 		String ldap_provider_url = ConfigurationManager.getProperty("ldap.provider_url");
357 		String ldap_id_field = ConfigurationManager.getProperty("ldap.id_field");
358 		String ldap_search_context = ConfigurationManager.getProperty("ldap.search_context");
359 		String ldap_object_context = ConfigurationManager.getProperty("ldap.object_context");
360 		String ldap_search_scope = ConfigurationManager.getProperty("ldap.search_scope");
361 
362 		String ldap_email_field = ConfigurationManager.getProperty("ldap.email_field");
363 		String ldap_givenname_field = ConfigurationManager.getProperty("ldap.givenname_field");
364 		String ldap_surname_field = ConfigurationManager.getProperty("ldap.surname_field");
365 		String ldap_phone_field = ConfigurationManager.getProperty("ldap.phone_field");
366 
367 		SpeakerToLDAP(Logger thelog)
368         {
369             log = thelog;
370         }
371 
372 		protected String getDNOfUser(String adminUser, String adminPassword, Context context, String netid)
373 		{
374 			// The resultant DN
375 			String resultDN;
376 
377 			// The search scope to use (default to 0)
378 			int ldap_search_scope_value = 0;
379 			try
380 			{
381 				ldap_search_scope_value = Integer.parseInt(ldap_search_scope.trim());
382 			}
383 			catch (NumberFormatException e)
384 			{
385 				// Log the error if it has been set but is invalid
386 				if (ldap_search_scope != null)
387 				{
388 					log.warn(LogManager.getHeader(context,
389 							"ldap_authentication", "invalid search scope: " + ldap_search_scope));
390 				}
391 			}
392 
393 			// Set up environment for creating initial context
394 			Hashtable env = new Hashtable(11);
395 			env.put(javax.naming.Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
396 			env.put(javax.naming.Context.PROVIDER_URL, ldap_provider_url);
397 
398             if ((adminUser != null) && (!adminUser.trim().equals("")) &&
399                 (adminPassword != null) && (!adminPassword.trim().equals("")))
400             {
401                 // Use admin credencials for search// Authenticate
402                 env.put(javax.naming.Context.SECURITY_AUTHENTICATION, "simple");
403                 env.put(javax.naming.Context.SECURITY_PRINCIPAL, adminUser);
404                 env.put(javax.naming.Context.SECURITY_CREDENTIALS, adminPassword);
405             }
406             else
407             {
408                 // Use anonymous authentication
409                 env.put(javax.naming.Context.SECURITY_AUTHENTICATION, "none");
410             }
411 
412 			DirContext ctx = null;
413 			try
414 			{
415 				// Create initial context
416 				ctx = new InitialDirContext(env);
417 
418 				Attributes matchAttrs = new BasicAttributes(true);
419 				matchAttrs.put(new BasicAttribute(ldap_id_field, netid));
420 
421 				// look up attributes
422 				try
423 				{
424 					SearchControls ctrls = new SearchControls();
425 					ctrls.setSearchScope(ldap_search_scope_value);
426 
427 					NamingEnumeration<SearchResult> answer = ctx.search(
428 							ldap_provider_url + ldap_search_context,
429 							"(&({0}={1}))", new Object[] { ldap_id_field,
430 									netid }, ctrls);
431 
432 					while (answer.hasMoreElements()) {
433 						SearchResult sr = answer.next();
434 						resultDN = (sr.getName() + "," + ldap_search_context);
435 						String attlist[] = {ldap_email_field, ldap_givenname_field,
436 								            ldap_surname_field, ldap_phone_field};
437 						Attributes atts = sr.getAttributes();
438 						Attribute att;
439 
440 						if (attlist[0] != null) {
441 							att = atts.get(attlist[0]);
442 							if (att != null)
443 								ldapEmail = (String) att.get();
444 						}
445 
446 						if (attlist[1] != null) {
447 							att = atts.get(attlist[1]);
448 							if (att != null)
449 								ldapGivenName = (String) att.get();
450 						}
451 
452 						if (attlist[2] != null) {
453 							att = atts.get(attlist[2]);
454 							if (att != null)
455 								ldapSurname = (String) att.get();
456 						}
457 
458 						if (attlist[3] != null) {
459 							att = atts.get(attlist[3]);
460 							if (att != null)
461 								ldapPhone = (String) att.get();
462 						}
463 
464 						if (answer.hasMoreElements()) {
465 							// Oh dear - more than one match
466 							// Ambiguous user, can't continue
467 
468 						} else {
469 							log.debug(LogManager.getHeader(context, "got DN", resultDN));
470 							return resultDN;
471 						}
472 					}
473 				}
474 				catch (NamingException e)
475 				{
476 					// if the lookup fails go ahead and create a new record for them because the authentication
477 					// succeeded
478 					log.warn(LogManager.getHeader(context,
479 								"ldap_attribute_lookup", "type=failed_search "
480 										+ e));
481 				}
482 			}
483 			catch (NamingException e)
484 			{
485 				log.warn(LogManager.getHeader(context,
486 							"ldap_authentication", "type=failed_auth " + e));
487 			}
488 			finally
489 			{
490 				// Close the context when we're done
491 				try
492 				{
493 					if (ctx != null)
494 						ctx.close();
495 				}
496 				catch (NamingException e)
497 				{
498 				}
499 			}
500 
501 			// No DN match found
502 			return null;
503 		}

505 		/**
506          * contact the ldap server and attempt to authenticate
507          */
508 		protected boolean ldapAuthenticate(String netid, String password,
509 						Context context) {
510 			if (!password.equals("")) {
511 				// Set up environment for creating initial context
512 				Hashtable<String, String> env = new Hashtable<String, String>();
513 				env.put(javax.naming.Context.INITIAL_CONTEXT_FACTORY,
514 						"com.sun.jndi.ldap.LdapCtxFactory");
515 				env.put(javax.naming.Context.PROVIDER_URL, ldap_provider_url);
516 
517 				// Authenticate
518 				env.put(javax.naming.Context.SECURITY_AUTHENTICATION, "Simple");
519 				env.put(javax.naming.Context.SECURITY_PRINCIPAL, netid);
520 				env.put(javax.naming.Context.SECURITY_CREDENTIALS, password);
521 				env.put(javax.naming.Context.AUTHORITATIVE, "true");
522 				env.put(javax.naming.Context.REFERRAL, "follow");
523 
524 				DirContext ctx = null;
525 				try {
526 					// Try to bind
527 					ctx = new InitialDirContext(env);
528 				} catch (NamingException e) {
529 					log.warn(LogManager.getHeader(context,
530 							"ldap_authentication", "type=failed_auth " + e));
531 					return false;
532 				} finally {
533 					// Close the context when we're done
534 					try {
535 						if (ctx != null)
536 							ctx.close();
537 					} catch (NamingException e) {
538 					}
539 				}
540 			} else {
541 				return false;
542 			}
543 
544 			return true;
545 		}
