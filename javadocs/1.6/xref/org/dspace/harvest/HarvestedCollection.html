135     /** Returns whether the specified collection is harvestable, i.e. whether its harvesting 
136      * options are set up correctly. This is distinct from "ready", since this collection may
137      * be in process of being harvested.
138      */
139     public static boolean isHarvestable(Context c, int collectionId) throws SQLException 
140     {
141     	HarvestedCollection hc = HarvestedCollection.find(c, collectionId); 
142     	if (hc != null && hc.getHarvestType() > 0 && hc.getOaiSource() != null && hc.getOaiSetId() != null && 
143     			hc.getHarvestStatus() != HarvestedCollection.STATUS_UNKNOWN_ERROR) {
144     		return true;
145     	}
146     	return false;   
147     }

172     {
173     	if (this.isHarvestable() &&	(this.getHarvestStatus() == HarvestedCollection.STATUS_READY || this.getHarvestStatus() == HarvestedCollection.STATUS_OAI_ERROR))

200     /** Find all collections that are ready for harvesting 
201      * 
202      * return: list of collection id's
203      * @throws SQLException 
204      */
205     public static List<Integer> findReady(Context c) throws SQLException 
206     {
207     	int harvestInterval = ConfigurationManager.getIntProperty("harvester.harvestFrequency");
208     	if (harvestInterval == 0) harvestInterval = 720;
209     	
210     	int expirationInterval = ConfigurationManager.getIntProperty("harvester.threadTimeout");
211     	if (expirationInterval == 0) expirationInterval = 24;
212 
213     	Date startTime;
214         Date expirationTime;
215     	
216     	Calendar calendar = Calendar.getInstance();
217 		calendar.setTime(new Date());
218 		calendar.add(Calendar.MINUTE, -1 * harvestInterval);
219 		startTime = calendar.getTime();
220 		
221 		calendar.setTime(startTime);
222 		calendar.add(Calendar.HOUR, -2 * expirationInterval);
223 		expirationTime = calendar.getTime();
224     	
225     	/* Select all collections whose last_harvest is before our start time, whose harvest_type *is not* 0 and whose status *is* 0 (available) or 3 (OAI Error). */
226     	TableRowIterator tri = DatabaseManager.queryTable(c, "harvested_collection",
227         	"SELECT * FROM harvested_collection WHERE (last_harvested < ? or last_harvested is null) and harvest_type > ? and (harvest_status = ? or harvest_status = ? or (harvest_status=? and harvest_start_time < ?)) ORDER BY last_harvested",
228         	new java.sql.Timestamp(startTime.getTime()), 0, HarvestedCollection.STATUS_READY, HarvestedCollection.STATUS_OAI_ERROR, HarvestedCollection.STATUS_BUSY, new java.sql.Timestamp(expirationTime.getTime()));
229     	
230     	List<Integer> collectionIds = new ArrayList<Integer>();
231 
232     	while (tri.hasNext())
233     	{
234     		TableRow row = tri.next();
235     		collectionIds.add(row.getIntColumn("collection_id"));
236     	}
237     	
238     	return collectionIds;
239     }
