60  /**
61   * The Plugin Manager is a very simple component container.  It creates and
62   * organizes components (plugins), and helps select a plugin in the cases
63   * where there are many possible choices.  It also gives some limited
64   * control over the lifecycle of a plugin.  It manages three different types
65   * (usage patterns) of plugins:
66   * <p>
67   * <ol><li> Singleton Plugin
68   * <br>  There is only one implementation class for the plugin.  It is indicated
69   *   in the configuration.  This type of plugin chooses an implementations of
70   *   a service, for the entire system, at configuration time.  Your
71   *   application just fetches the plugin for that interface and gets the
72   *   configured-in choice.
73   *
74   * <p><li> Sequence Plugins
75   *  <br> You need a sequence or series of plugins, to implement a mechanism like
76   *   StackableAuthenticationMethods or a pipeline, where each plugin is
77   *   called in order to contribute its implementation of a process to the
78   *   whole.
79   *  <p><li> Named Plugins
80   *  <br> Use a named plugin when the application has to choose one plugin
81   *   implementation out of many available ones.  Each implementation is bound
82   *   to one or more names (symbolic identifiers) in the configuration.
83   *  </ol><p>
84   *  The name is just a <code>String</code> to be associated with the
85   *  combination of implementation class and interface.  It may contain
86   *  any characters except for comma (,) and equals (=).  It may contain
87   *  embedded spaces.  Comma is a special character used to separate
88   *  names in the configuration entry.
89   *
90   * @author Larry Stone
91   * @version $Revision: 3762 $
92   * @see SelfNamedPlugin
93   */
94  public class PluginManager

113     // Map of plugin class to "reusable" metric (as Boolean, must be Object)
114     // Key is Class, value is Boolean (true by default).
115     private static HashMap cacheMeCache = new HashMap();
116 
117     // Predicate -- whether or not to cache this class.  Ironically,
118     // the cacheability information is itself cached.
119     private static boolean cacheMe(Class implClass)
120     {
121         if (cacheMeCache.containsKey(implClass))
122         {
123             return ((Boolean)cacheMeCache.get(implClass)).booleanValue();
124         }
125         else
126         {
127         	String key = REUSABLE_PREFIX+implClass.getName();
128             boolean reusable = ConfigurationManager.getBooleanProperty(key, true);
129             cacheMeCache.put(implClass, new Boolean(reusable));
130             return reusable;
131         }
132     }

134     /**
135      * Returns an instance of the singleton (single) plugin implementing
136      * the given interface.  There must be exactly one single plugin
137      * configured for this interface, otherwise the
138      * <code>PluginConfigurationError</code> is thrown.
139      * <p>
140      * Note that this is the only "get plugin" method which throws an
141      * exception.  It is typically used at initialization time to set up
142      * a permanent part of the system so any failure is fatal.
143      *
144      * @param interfaceClass interface Class object
145      * @return instance of plugin
146      * @throws PluginConfigurationError
147      */
148     public static Object getSinglePlugin(Class interfaceClass)
149         throws PluginConfigurationError, PluginInstantiationException
150     {
151         String iname = interfaceClass.getName();
152 
153         // configuration format is  prefix.<interface> = <classname>
154         String classname = ConfigurationManager.getProperty(SINGLE_PREFIX+iname);
155         if (classname != null)
156             return getAnonymousPlugin(classname.trim());
157         else
158             throw new PluginConfigurationError("No Single Plugin configured for interface \""+iname+"\"");
159     }

166     /**
167      * Returns instances of all plugins that implement the interface
168      * intface, in an Array.  Returns an empty array if no there are no
169      * matching plugins.
170      * <p>
171      * The order of the plugins in the array is the same as their class
172      * names in the configuration's value field.
173      *
174      * @param intfc interface for which to find plugins.
175      * @return an array of plugin instances; if none are
176      *   available an empty array is returned.
177      */
178     public static Object[] getPluginSequence(Class intfc)
179         throws PluginInstantiationException
180     {
181         // cache the configuration for this interface after grovelling it once:
182         // format is  prefix.<interface> = <classname>
183         String iname = intfc.getName();
184         String classname[] = null;
185         if (!sequenceConfig.containsKey(iname))
186         {
187             String val = ConfigurationManager.getProperty(SEQUENCE_PREFIX+iname);
188             if (val == null)
189             {
190                 log.warn("No Configuration entry found for Sequence Plugin interface="+iname);
191                 return new Object[0];
192             }
193             classname = val.trim().split("\\s*,\\s*");
194             sequenceConfig.put(iname, classname);
195         }
196         else
197             classname = (String[])sequenceConfig.get(iname);
198 
199         Object result[] = (Object[])Array.newInstance(intfc, classname.length);
200         for (int i = 0; i < classname.length; ++i)
201         {
202             log.debug("Adding Sequence plugin for interface= "+iname+", class="+classname[i]);
203             result[i] = getAnonymousPlugin(classname[i]);
204         }
205         return result;
206     }

211     // Get possibly-cached plugin instance for un-named plugin,
212     // this is shared by Single and Sequence plugins.
213     private static Object getAnonymousPlugin(String classname)
214         throws PluginInstantiationException
215     {
216         try
217         {
218             Class pluginClass = Class.forName(classname);
219             if (cacheMe(pluginClass))
220             {
221                 Object cached = anonymousInstanceCache.get(pluginClass);
222                 if (cached == null)
223                 {
224                     cached = pluginClass.newInstance();
225                     anonymousInstanceCache.put(pluginClass, cached);
226                 }
227                 return cached;
228             }
229             else
230                 return pluginClass.newInstance();
231         }
232         catch (ClassNotFoundException e)
233         {
234             throw new PluginInstantiationException("Cannot load plugin class: " +
235             		                               e.toString(), e);
236         }
237         catch (InstantiationException e)
238         {
239             throw new PluginInstantiationException(e);
240         }
241         catch (IllegalAccessException e)
242         {
243             throw new PluginInstantiationException(e);
244         }
245     }

254     // load and cache configuration data for the given interface.
255     private static void configureNamedPlugin(String iname)
256         throws ClassNotFoundException
257     {
258         int found = 0;
259 
260         /**
261          * First load the class map for this interface (if not done yet):
262          * key is [intfc,name], value is class.
263          * There is ALSO a "marker key" of "intfc" by itself to show we
264          * loaded this intfc's configuration.
265          */
266         if (!namedPluginClasses.containsKey(iname))
267         {
268             // 1. Get classes named by the configuration. format is:
269             //    plugin.named.<INTF> = <CLASS> = <name>, <name> [,] \
270             //                        <CLASS> = <name>, <name> [ ... ]
271             String namedVal = ConfigurationManager.getProperty(NAMED_PREFIX+iname);
272             if (namedVal != null)
273             {
274                 namedVal = namedVal.trim();
275                 log.debug("Got Named configuration for interface="+iname+", config="+namedVal);
276 
277                 // match  "<classname> ="
278                 Pattern classnameEqual = Pattern.compile("([\\w\\p{Sc}\\.]+)\\s*\\=");
279 
280                 int prevEnd = -1;
281                 String prevClassName = null;
282                 Matcher classMatcher = classnameEqual.matcher(namedVal);
283                 while (classMatcher.find())
284                 {
285                     if (prevClassName != null)
286                         found += installNamedConfigs(iname, prevClassName,
287                                    namedVal.substring(prevEnd, classMatcher.start()).trim().split("\\s*,\\s*"));
288                     prevClassName = classMatcher.group(1);
289                     prevEnd = classMatcher.end();
290                 }
291                 if (prevClassName != null)
292                     found += installNamedConfigs(iname, prevClassName,
293                                namedVal.substring(prevEnd).trim().split("\\s*,\\s*"));
294             }
295 
296             // 2. Get Self-named config entries:
297             // format is plugin.selfnamed.<INTF> = <CLASS> , <CLASS> ..
298             String selfNamedVal = ConfigurationManager.getProperty(SELFNAMED_PREFIX+iname);
299             if (selfNamedVal != null)
300             {
301                 String classnames[] = selfNamedVal.trim().split("\\s*,\\s*");
302                 for (int i = 0; i < classnames.length; ++i)
303                 {
304                     try
305                     {
306                     	Class pluginClass = Class.forName(classnames[i]);
307                         String names[] = (String[])pluginClass.getMethod("getPluginNames").
308                                                    invoke(null);
309                         if (names == null || names.length == 0)
310                             log.error("Self-named plugin class \""+classnames[i]+"\" returned null or empty name list!");
311                         else
312                             found += installNamedConfigs(iname, classnames[i], names);
313                     }
314                     catch (NoSuchMethodException e)
315                     {
316                         log.error("Implementation Class \""+classnames[i]+"\" is not a subclass of SelfNamedPlugin, it has no getPluginNames() method.");
317                     }
318                     catch (Exception e)
319                     {
320                         log.error("While configuring self-named plugin: " + e.toString());
321                     }
322                 }
323             }
324             namedPluginClasses.put(iname, "org.dspace.core.marker");
325             if (found == 0)
326                 log.error("No named plugins found for interface="+iname);
327         }
328     }

330     // add info for a named plugin to cache, under all its names.
331     private static int installNamedConfigs(String iname, String classname, String names[])
332         throws ClassNotFoundException
333     {
334         int found = 0;
335         for (int i = 0; i < names.length; ++i)
336         {
337             String key = iname+SEP+names[i];
338             if (namedPluginClasses.containsKey(key))
339                 log.error("Name collision in named plugin, implementation class=\""+classname+
340                             "\", name=\""+names[i]+"\"");
341             else
342                 namedPluginClasses.put(key, classname);
343             log.debug("Got Named Plugin, intfc="+iname+", name="+names[i]+", class="+classname);
344             ++found;
345         }
346         return found;
347     }

349     /**
350      * Returns an instance of a plugin that implements the interface
351      * intface and is bound to a name matching name.  If there is no
352      * matching plugin, it returns null.  The names are matched by
353      * String.equals().
354      *
355      * @param intfc the interface class of the plugin
356      * @param name under which the plugin implementation is configured.
357      * @return instance of plugin implementation, or null if there is no match or an error.
358      */
359     public static Object getNamedPlugin(Class intfc, String name)
360          throws PluginInstantiationException
361     {
362         try
363         {
364             String iname = intfc.getName();
365             configureNamedPlugin(iname);
366             String key = iname + SEP + name;
367             String cname = (String)namedPluginClasses.get(key);
368             if (cname == null)
369                 log.warn("Cannot find named plugin for interface="+iname+", name=\""+name+"\"");
370             else
371             {
372                 Class pluginClass = Class.forName(cname);
373                 if (cacheMe(pluginClass))
374                 {
375                     String nkey = pluginClass.getName() + SEP + name;
376                     Object cached = namedInstanceCache.get(nkey);
377                     if (cached == null)
378                     {
379                     	log.debug("Creating cached instance of: " + cname + 
380                     			  " for interface=" + iname + 
381                     			  " pluginName=" + name );
382                         cached = pluginClass.newInstance();
383                         if (cached instanceof SelfNamedPlugin)
384                             ((SelfNamedPlugin)cached).setPluginInstanceName(name);
385                         namedInstanceCache.put(nkey, cached);
386                     }
387                     return cached;
388                 }
389                 else
390                 {
391                 	log.debug("Creating UNcached instance of: " + cname + 
392               			  " for interface=" + iname + 
393               			  " pluginName=" + name );
394                     Object result = pluginClass.newInstance();
395                     if (result instanceof SelfNamedPlugin)
396                         ((SelfNamedPlugin)result).setPluginInstanceName(name);
397                     return result;
398                 }
399             }
400         }
401         catch (ClassNotFoundException e)
402         {
403             throw new PluginInstantiationException("Cannot load plugin class: " +
404             		                               e.toString(), e);
405         }
406         catch (InstantiationException e)
407         {
408             throw new PluginInstantiationException(e);
409         }
410         catch (IllegalAccessException e)
411         {
412             throw new PluginInstantiationException(e);
413         }
414 
415         return null;
416     }

419     /**
420      * Returns all of the names under which a named plugin implementing
421      * the interface intface can be requested (with getNamedPlugin()).
422      * The array is empty if there are no matches.  Use this to populate
423      * a menu of plugins for interactive selection, or to document what
424      * the possible choices are.
425      * <p>
426      * NOTE: The names are NOT returned in any deterministic order.
427      *
428      * @param intfc plugin interface for which to return names.
429      * @return an array of strings with every name; if none are
430      *   available an empty array is returned.
431      */
432     public static String[] getAllPluginNames(Class intfc)
433     {
434         try
435         {
436             String iname = intfc.getName();
437             configureNamedPlugin(iname);
438             String prefix = iname + SEP;
439             ArrayList result = new ArrayList();
440 
441             Iterator ki = namedPluginClasses.keySet().iterator();
442             while (ki.hasNext())
443             {
444                 String key = (String)ki.next();
445                 if (key.startsWith(prefix))
446                     result.add(key.substring(prefix.length()));
447             }
448             if (result.size() == 0)
449                 log.error("Cannot find any names for named plugin, interface="+iname);
450 
451             return (String[])result.toArray(new String[result.size()]);
452         }
453         catch (ClassNotFoundException e)
454         {
455             return new String[0];
456         }
457     }

459     /**
460      * Tells the Plugin Manager to let go of any references to a
461      * reusable plugin, to prevent it from being given out again and to
462      * allow the object to be garbage-collected.  Call this when a
463      * plugin instance must be taken out of circulation.
464      *
465      * @param plugin the object to release, must have been created by
466      *   <code>getNamedPlugin</code> etc.
467      */
468     public static void releasePlugin(Object plugin)
469     {
470         forgetInstance(plugin, namedInstanceCache);
471         forgetInstance(plugin, anonymousInstanceCache);
472     }
473 
474     private static void forgetInstance(Object plugin, Map cacheMap)
475     {
476         Collection values = cacheMap.values();
477         Iterator ci = values.iterator();
478         while (ci.hasNext())
479         {
480             Object val = ci.next();
481             if (val == plugin)
482                 values.remove(val);
483         }
484     }

506     // true if classname is loadable AND is subclass of SelfNamedPlugin

521     // recursively climb superclass stack until we find SelfNamedPlugin
522     private static boolean checkSelfNamed(Class cls)
523     {
524         Class sup = cls.getSuperclass();
525         if (sup == null)
526             return false;
527         else if (sup.equals(SelfNamedPlugin.class))
528             return true;
529         else
530             return checkSelfNamed(sup);
531     }

547     /**
548      * Validate the entries in the DSpace Configuration relevant to
549      * PluginManager.  Look for inconsistencies, illegal syntax, etc.
550      * Announce violations with "log.error" so they appear in the log
551      * or in the standard error stream if this is run interactively.
552      * <ul>
553      * <li>Look for duplicate keys (by parsing the config file)
554      * <li>Interface in plugin.single, plugin.sequence, plugin.named, plugin.selfnamed is valid.
555      * <li>Classname in plugin.reusable exists and matches a plugin config.
556      * <li>Classnames in config values exist.
557      * <li>Classnames in plugin.selfnamed loads and is subclass of <code>SelfNamedPlugin</code>
558      * <li>Implementations of named plugin have no name collisions.
559      * <li>Named plugin entries lacking names.
560      * </ul>
561      */
562     public static void checkConfiguration()
563         throws IOException
564     {
565         FileReader fr = null;
566         BufferedReader cr = null;
567 
568         /*  XXX TODO:  (maybe) test that implementation class is really a
569          *  subclass or impl of the plugin "interface"
570          */
571          
572         // tables of config keys for each type of config line:
573         Map singleKey = new HashMap();
574         Map sequenceKey = new HashMap();
575         Map namedKey = new HashMap();
576         Map selfnamedKey = new HashMap();
577         Map reusableKey = new HashMap();
578         HashMap keyMap = new HashMap();
579 
580         // 1. First pass -- grovel the actual config file to check for
581         //    duplicate keys, since Properties class hides them from us.
582         //    Also build lists of each type of key, check for misspellings.
583         File config = ConfigurationManager.getConfigurationFile();
584         try
585         {
586             fr = new FileReader(config);
587             cr = new BufferedReader(fr);
588             String line = null;
589             boolean continued = false;
590             Pattern keyPattern = Pattern.compile("([^\\s\\=\\:]+)");
591             while ((line = cr.readLine()) != null)
592             {
593                 line = line.trim();
594                 if (line.startsWith("!") || line.startsWith("#"))
595                     continued = false;
596                 else
597                 {
598                     if (!continued && line.startsWith("plugin."))
599                     {
600                         Matcher km = keyPattern.matcher(line);
601                         if (km.find())
602                         {
603                             String key = line.substring(0, km.end(1));
604                             if (keyMap.containsKey(key))
605                                 log.error("Duplicate key \""+key+"\" in DSpace configuration file="+config.toString());
606                             else
607                                 keyMap.put(key, key);
608 
609                             if (key.startsWith(SINGLE_PREFIX))
610                                 singleKey.put(key.substring(SINGLE_PREFIX.length()), key);
611                             else if (key.startsWith(SEQUENCE_PREFIX))
612                                 sequenceKey.put(key.substring(SEQUENCE_PREFIX.length()), key);
613                             else if (key.startsWith(NAMED_PREFIX))
614                                 namedKey.put(key.substring(NAMED_PREFIX.length()), key);
615                             else if (key.startsWith(SELFNAMED_PREFIX))
616                                 selfnamedKey.put(key.substring(SELFNAMED_PREFIX.length()), key);
617                             else if (key.startsWith(REUSABLE_PREFIX))
618                                 reusableKey.put(key.substring(REUSABLE_PREFIX.length()), key);
619                             else
620                                 log.error("Key with unknown prefix \""+key+"\" in DSpace configuration file="+config.toString());
621                         }
622                     }
623                     continued = line.length() > 0 && line.charAt(line.length()-1) == '\\';
624                 }
625             }
626         }
627         finally
628         {
629             if (cr != null)
630                 try { cr.close(); } catch (IOException ioe) { }
631 
632             if (fr != null)
633                 try { fr.close(); } catch (IOException ioe) { }
634         }
635 
636         // 1.1 Sanity check, make sure keyMap == set of keys from Configuration
637         Enumeration pne = ConfigurationManager.propertyNames();
638         HashSet pn = new HashSet();
639         while (pne.hasMoreElements())
640         {
641             String nk = (String)pne.nextElement();
642             if (nk.startsWith("plugin."))
643             {
644                 pn.add(nk);
645                 if (!keyMap.containsKey(nk))
646                     log.error("Key is in ConfigurationManager.propertyNames() but NOT text crawl: \""+nk+"\"");
647             }
648         }
649         Iterator pi = keyMap.keySet().iterator();
650         while (pi.hasNext())
651         {
652             String key = (String)pi.next();
653             if (!pn.contains(key))
654                 log.error("Key is in text crawl but NOT ConfigurationManager.propertyNames(): \""+key+"\"");
655         }
656 
657         // 2. Build up list of all interfaces and test that they are loadable.
658         // don't bother testing that they are "interface" rather than "class"
659         // since either one will work for the Plugin Manager.
660         ArrayList allInterfaces = new ArrayList();
661         allInterfaces.addAll(singleKey.keySet());
662         allInterfaces.addAll(sequenceKey .keySet());
663         allInterfaces.addAll(namedKey.keySet());
664         allInterfaces.addAll(selfnamedKey.keySet());
665         allInterfaces.addAll(reusableKey.keySet());
666         Iterator ii = allInterfaces.iterator();
667         while (ii.hasNext())
668             checkClassname((String)ii.next(), "key interface or class");
669 
670         // Check implementation classes:
671         //  - each class is loadable.
672         //  - plugin.selfnamed values are each  subclass of SelfNamedPlugin
673         //  - save classname in allImpls
674         Map allImpls = new HashMap();
675          
676         // single plugins - just check that it has a valid impl. class
677         ii = singleKey.keySet().iterator();
678         while (ii.hasNext())
679         {
680             String key = (String)ii.next();
681             String val = ConfigurationManager.getProperty(SINGLE_PREFIX+key);
682             if (val == null)
683                 log.error("Single plugin config not found for: "+SINGLE_PREFIX+key);
684             else
685             {
686                 val = val.trim();
687                 if (checkClassname(val, "implementation class"))
688                     allImpls.put(val, val);
689             }
690         }
691          
692         // sequence plugins - all values must be classes
693         ii = sequenceKey.keySet().iterator();
694         while (ii.hasNext())
695         {
696             String key = (String)ii.next();
697             String val = ConfigurationManager.getProperty(SEQUENCE_PREFIX+key);
698             if (val == null)
699                 log.error("Sequence plugin config not found for: "+SEQUENCE_PREFIX+key);
700             else
701             {
702                 val = val.trim();
703                 String classname[] = val.split("\\s*,\\s*");
704                 for (int i = 0; i < classname.length; ++i)
705                     if (checkClassname(classname[i], "implementation class"))
706                         allImpls.put(classname[i], classname[i]);
707             }
708         }
709 
710         // 3. self-named plugins - grab and check all values
711         //   then make sure it is a subclass of SelfNamedPlugin
712         ii = selfnamedKey.keySet().iterator();
713         while (ii.hasNext())
714         {
715             String key = (String)ii.next();
716             String val = ConfigurationManager.getProperty(SELFNAMED_PREFIX+key);
717             if (val == null)
718                 log.error("Selfnamed plugin config not found for: "+SELFNAMED_PREFIX+key);
719             else
720             {
721                 val = val.trim();
722                 String classname[] = val.split("\\s*,\\s*");
723                 for (int i = 0; i < classname.length; ++i)
724                     if (checkClassname(classname[i], "selfnamed implementation class"))
725                     {
726                         allImpls.put(classname[i], classname[i]);
727                         checkSelfNamed(classname[i]);
728                     }
729                 checkNames(key);
730             }
731         }
732 
733         // 4. named plugins - extract the classnames and treat same as sequence.
734         // use named plugin config mechanism to test for duplicates, unnamed.
735         ii = namedKey.keySet().iterator();
736         Pattern classnameEqual = Pattern.compile("([\\w\\p{Sc}\\.]+)\\s*\\=");
737         while (ii.hasNext())
738         {
739             String key = (String)ii.next();
740             String val = ConfigurationManager.getProperty(NAMED_PREFIX+key);
741             if (val == null)
742                 log.error("Named plugin config not found for: "+NAMED_PREFIX+key);
743             else
744             {
745                 checkNames(key);
746                 val = val.trim();
747                 Matcher classMatcher = classnameEqual.matcher(val);
748                 while (classMatcher.find())
749                 {
750                     String classname = classMatcher.group(1);
751 
752                     if (checkClassname(classname, "implementation class"))
753                         allImpls.put(classname, classname);
754                 }
755             }
756         }
757 
758         // 5. all classes named in Reusable config lines must be other classes.
759         Iterator ri = reusableKey.keySet().iterator();
760         while (ri.hasNext())
761         {
762             String rk = (String)ri.next();
763             if (!(allImpls.containsKey(rk)))
764                 log.error("In plugin.reusable configuration, class \""+rk+"\" is NOT a plugin implementation class.");
765         }
766     }
