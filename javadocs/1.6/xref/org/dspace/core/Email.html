67  /**
68   * Class representing an e-mail message, also used to send e-mails.
69   * <P>
70   * Typical use:
71   * <P>
72   * <code>Email email = ConfigurationManager.getEmail(name);</code><br>
73   * <code>email.addRecipient("foo@bar.com");</code><br>
74   * <code>email.addArgument("John");</code><br>
75   * <code>email.addArgument("On the Testing of DSpace");</code><br>
76   * <code>email.send();</code><br>
77   * <P>
78   * <code>name</code> is the name of an email template in
79   * <code>dspace-dir/config/emails/</code> (which also includes the subject.)
80   * <code>arg0</code> and <code>arg1</code> are arguments to fill out the
81   * message with.
82   * <P>
83   * Emails are formatted using <code>java.text.MessageFormat.</code>
84   * Additionally, comment lines (starting with '#') are stripped, and if a line
85   * starts with "Subject:" the text on the right of the colon is used for the
86   * subject line. For example:
87   * <P>
88   * 
89   * <pre>
90   *    
91   *     # This is a comment line which is stripped
92   *     #
93   *     # Parameters:   {0}  is a person's name
94   *     #               {1}  is the name of a submission
95   *     #
96   *     Subject: Example e-mail
97   *    
98   *     Dear {0},
99   *    
100  *     Thank you for sending us your submission &quot;{1}&quot;.
101  *     
102  * </pre>
103  * 
104  * <P>
105  * If the example code above was used to send this mail, the resulting mail
106  * would have the subject <code>Example e-mail</code> and the body would be:
107  * <P>
108  * 
109  * <pre>
110  *    
111  *    
112  *     Dear John,
113  *    
114  *     Thank you for sending us your submission &quot;On the Testing of DSpace&quot;.
115  *     
116  * </pre>
117  * 
118  * <P>
119  * Note that parameters like <code>{0}</code> cannot be placed in the subject
120  * of the e-mail; they won't get filled out.
121  * 
122  * 
123  * @author Robert Tansley
124  * @author Jim Downing - added attachment handling code
125  * @version $Revision: 4287 $
126  */

256     /**
257      * Sends the email.
258      * 
259      * @throws MessagingException
260      *             if there was a problem sending the mail.
261      */
262     public void send() throws MessagingException
263     {
264         // Get the mail configuration properties
265         String server = ConfigurationManager.getProperty("mail.server");
266         String from = ConfigurationManager.getProperty("mail.from.address");
267         boolean disabled = ConfigurationManager.getBooleanProperty("mail.server.disabled", false);
268 
269         if (disabled) {
270             log.info("message not sent due to mail.server.disabled: " + subject);
271             return;
272         }
273 
274         // Set up properties for mail session
275         Properties props = System.getProperties();
276         props.put("mail.smtp.host", server);
277 
278         // Set the port number for the mail server
279         String portNo = ConfigurationManager.getProperty("mail.server.port");
280         if (portNo == null)
281         {
282         	portNo = "25";
283         }
284         props.put("mail.smtp.port", portNo.trim());
285 
286         // If no character set specified, attempt to retrieve a default
287         if (charset == null)
288         {
289             charset = ConfigurationManager.getProperty("mail.charset");    
290         }
291 
292         // Get session
293         Session session;
294         
295         // Get the SMTP server authentication information
296         String username = ConfigurationManager.getProperty("mail.server.username");
297         String password = ConfigurationManager.getProperty("mail.server.password");
298         
299         if (username != null)
300         {
301             props.put("mail.smtp.auth", "true");
302             SMTPAuthenticator smtpAuthenticator = new SMTPAuthenticator(
303                     username, password);
304             session = Session.getDefaultInstance(props, smtpAuthenticator);
305         }
306         else
307         {
308             session = Session.getDefaultInstance(props);
309         }
310 
311         // Set extra configuration properties
312         String extras = ConfigurationManager.getProperty("mail.extraproperties");
313         if ((extras != null) && (!"".equals(extras.trim())))
314         {
315             String arguments[] = extras.split(",");
316             String key, value;
317             for (String argument : arguments)
318             {
319                 key = argument.substring(0, argument.indexOf('=')).trim();
320                 value = argument.substring(argument.indexOf('=') + 1).trim();
321                 props.put(key, value);
322             }
323         }
324 
325         // Create message
326         MimeMessage message = new MimeMessage(session);
327 
328         // Set the recipients of the message
329         Iterator i = recipients.iterator();
330 
331         while (i.hasNext())
332         {
333             message.addRecipient(Message.RecipientType.TO, new InternetAddress(
334                     (String) i.next()));
335         }
336 
337         // Format the mail message
338         Object[] args = arguments.toArray();
339         String fullMessage = MessageFormat.format(content, args);
340         Date date = new Date();
341 
342         message.setSentDate(date);
343         message.setFrom(new InternetAddress(from));
344 
345         // Set the subject of the email (may contain parameters)
346         String fullSubject = MessageFormat.format(subject, args);
347         if (charset != null)
348         {
349             message.setSubject(fullSubject, charset);
350         }
351         else
352         {
353             message.setSubject(fullSubject);
354         }
355         
356         // Add attachments
357         if (attachments.isEmpty())
358         {
359             // If a character set has been specified, or a default exists
360             if (charset != null)
361             {
362                 message.setText(fullMessage, charset);
363             }
364             else
365             {
366                 message.setText(fullMessage);
367             }
368         }
369         else
370         {
371             Multipart multipart = new MimeMultipart();
372             // create the first part of the email
373             BodyPart messageBodyPart = new MimeBodyPart();
374             messageBodyPart.setText(fullMessage);
375             multipart.addBodyPart(messageBodyPart);
376 
377             for (Iterator iter = attachments.iterator(); iter.hasNext();)
378             {
379                 FileAttachment f = (FileAttachment) iter.next();
380                 // add the file
381                 messageBodyPart = new MimeBodyPart();
382                 messageBodyPart.setDataHandler(new DataHandler(
383                         new FileDataSource(f.file)));
384                 messageBodyPart.setFileName(f.name);
385                 multipart.addBodyPart(messageBodyPart);
386             }
387             message.setContent(multipart);
388         }
389 
390         if (replyTo != null)
391         {
392             Address[] replyToAddr = new Address[1];
393             replyToAddr[0] = new InternetAddress(replyTo);
394             message.setReplyTo(replyToAddr);
395         }
396 
397         Transport.send(message);
398     }
