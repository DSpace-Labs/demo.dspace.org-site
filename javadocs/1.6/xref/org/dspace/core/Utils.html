154     public static byte[] getMD5Bytes(byte[] data)
155     {
156         try
157         {
158             MessageDigest digest = MessageDigest.getInstance("MD5");
159 
160             return digest.digest(data);
161         }
162         catch (NoSuchAlgorithmException nsae)
163         {
164         }
165 
166         // Should never happen
167         return null;

170     /**
171      * Return a hex representation of the byte array
172      * 
173      * @param data
174      *            The data to transform.
175      * @return A hex representation of the data.
176      */
177     public static String toHex(byte[] data)
178     {
179         if ((data == null) || (data.length == 0))
180         {
181             return null;
182         }
183 
184         StringBuffer result = new StringBuffer();
185 
186         // This is far from the most efficient way to do things...
187         for (int i = 0; i < data.length; i++)
188         {
189             int low = (int) (data[i] & 0x0F);
190             int high = (int) (data[i] & 0xF0);
191 
192             result.append(Integer.toHexString(high).substring(0, 1));
193             result.append(Integer.toHexString(low));
194         }
195 
196         return result.toString();
197     }

240     /**
241      * Copy stream-data from source to destination. This method does not buffer,
242      * flush or close the streams, as to do so would require making non-portable
243      * assumptions about the streams' origin and further use. If you wish to
244      * perform a buffered copy, use {@link #bufferedCopy}.
245      * 
246      * @param input
247      *            The InputStream to obtain data from.
248      * @param output
249      *            The OutputStream to copy data to.
250      */
251     public static void copy(final InputStream input, final OutputStream output)
252             throws IOException
253     {
254         final int BUFFER_SIZE = 1024 * 4;
255         final byte[] buffer = new byte[BUFFER_SIZE];
256 
257         while (true)
258         {
259             final int count = input.read(buffer, 0, BUFFER_SIZE);
260 
261             if (-1 == count)
262             {
263                 break;
264             }
265 
266             // write out those same bytes
267             output.write(buffer, 0, count);
268         }
269 
270         // needed to flush cache
271         // output.flush();
272     }

274     /**
275      * Copy stream-data from source to destination, with buffering. This is
276      * equivalent to passing {@link #copy}a
277      * <code>java.io.BufferedInputStream</code> and
278      * <code>java.io.BufferedOuputStream</code> to {@link #copy}, and
279      * flushing the output stream afterwards. The streams are not closed after
280      * the copy.
281      * 
282      * @param source
283      *            The InputStream to obtain data from.
284      * @param destination
285      *            The OutputStream to copy data to.
286      */
287     public static void bufferedCopy(final InputStream source,
288             final OutputStream destination) throws IOException
289     {
290         final BufferedInputStream input = new BufferedInputStream(source);
291         final BufferedOutputStream output = new BufferedOutputStream(
292                 destination);
293         copy(input, output);
294         output.flush();
295     }

297     /**
298      * Replace characters that could be interpreted as HTML codes with symbolic
299      * references (entities). This function should be called before displaying
300      * any metadata fields that could contain the characters " <", ">", "&",
301      * "'", and double quotation marks. This will effectively disable HTML links
302      * in metadata.
303      * 
304      * @param value
305      *            the metadata value to be scrubbed for display
306      * 
307      * @return the passed-in string, with html special characters replaced with
308      *         entities.
309      */
310     public static String addEntities(String value)
311     {
312         if (value==null || value.length() == 0)
313             return value;
314         
315         value = value.replaceAll("&", "&amp;");
316         value = value.replaceAll("\"", "&quot;");
317 
318         // actually, &apos; is an XML entity, not in HTML.
319         // that's why it's commented out.
320         // value = value.replaceAll("'", "&apos;");
321         value = value.replaceAll("<", "&lt;");
322         value = value.replaceAll(">", "&gt;");
323 
324         return value;
325     }

327     /**
328      * Utility method to parse durations defined as \d+[smhdwy] (seconds,
329      * minutes, hours, days, weeks, years)
330      * 
331      * @param duration
332      *            specified duration
333      * 
334      * @return number of milliseconds equivalent to duration.
335      * 
336      * @throws ParseException
337      *             if the duration is of incorrect format
338      */
339     public static long parseDuration(String duration) throws ParseException
340     {
341         Matcher m = DURATION_PATTERN.matcher(duration.trim());
342         if (!m.matches())
343         {
344             throw new ParseException("'" + duration
345                     + "' is not a valid duration definition", 0);
346         }
347 
348         String units = m.group(2);
349         long multiplier = MS_IN_SECOND;
350 
351         if ("s".equals(units))
352         {
353             multiplier = MS_IN_SECOND;
354         }
355         else if ("m".equals(units))
356         {
357             multiplier = MS_IN_MINUTE;
358         }
359         else if ("h".equals(units))
360         {
361             multiplier = MS_IN_HOUR;
362         }
363         else if ("d".equals(units))
364         {
365             multiplier = MS_IN_DAY;
366         }
367         else if ("w".equals(units))
368         {
369             multiplier = MS_IN_WEEK;
370         }
371         else if ("y".equals(units))
372         {
373             multiplier = MS_IN_YEAR;
374         }
375         else
376         {
377             throw new ParseException(units
378                     + " is not a valid time unit (must be 'y', "
379                     + "'w', 'd', 'h', 'm' or 's')", duration.indexOf(units));
380         }
381 
382         long qint = Long.parseLong(m.group(1));
383 
384         return qint * multiplier;
385     }

387     /**
388      * Translates timestamp from an ISO 8601-standard format, which
389      * is commonly used in XML and RDF documents.
390      * This method is synchronized because it depends on a non-reentrant
391      * static DateFormat (more efficient than creating a new one each call).
392      *
393      * @param s the input string
394      * @return Date object, or null if there is a problem translating.
395      */
396     public static synchronized Date parseISO8601Date(String s)
397     {
398         // attempt to normalize the timezone to something we can parse;
399         // SimpleDateFormat can't handle "Z"
400         char tzSign = s.charAt(s.length()-6);
401         if (s.endsWith("Z"))
402             s = s.substring(0, s.length()-1) + "GMT+00:00";
403 
404         // check for trailing timezone
405         else if (tzSign == '-' || tzSign == '+')
406             s = s.substring(0, s.length()-6) + "GMT" + s.substring(s.length()-6);
407 
408         // try to parse without millseconds
409         ParseException lastError = null;
410         for (int i = 0; i < parseFmt.length; ++i)
411         {
412             try
413             {
414                 return parseFmt[i].parse(s);
415             }
416             catch (ParseException e)
417             {
418                 lastError = e;
419             }
420         }
421         if (lastError != null)
422             log.error("Error parsing date:", lastError);
423         return null;
424     }

426     /**
427      * Convert a Date to String in the ISO 8601 standard format.
428      * The RFC822 timezone is almost right, still need to insert ":".
429      * This method is synchronized because it depends on a non-reentrant
430      * static DateFormat (more efficient than creating a new one each call).
431      *
432      * @param d the input Date
433      * @return String containing formatted date.
434      */
435     public static synchronized String formatISO8601Date(Date d)
436     {
437         String result;
438         outCal.setTime(d);
439         if (outCal.get(Calendar.MILLISECOND) == 0)
440             result = outFmtSecond.format(d);
441         else
442             result = outFmtMillisec.format(d);
443         int rl = result.length();
444         return result.substring(0, rl-2) + ":" + result.substring(rl-2);
445     }
