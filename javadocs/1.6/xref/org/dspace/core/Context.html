59  /**
60   * Class representing the context of a particular DSpace operation. This stores
61   * information such as the current authenticated user and the database
62   * connection being used.
63   * <P>
64   * Typical use of the context object will involve constructing one, and setting
65   * the current user if one is authenticated. Several operations may be performed
66   * using the context object. If all goes well, <code>complete</code> is called
67   * to commit the changes and free up any resources used by the context. If
68   * anything has gone wrong, <code>abort</code> is called to roll back any
69   * changes and free up the resources.
70   * <P>
71   * The context object is also used as a cache for CM API objects.
72   * 
73   * 
74   * @version $Revision: 3985 $
75   */

207     /**
208      * Turn Off the Authorisation System for this context and store this change
209      * in a history for future use.
210      */
211     public void turnOffAuthorisationSystem()
212     {
213         authStateChangeHistory.push(ignoreAuth);
214         if (log.isDebugEnabled())
215         {
216             Thread currThread = Thread.currentThread();
217             StackTraceElement[] stackTrace = currThread.getStackTrace();
218             String caller = stackTrace[stackTrace.length - 1].getClassName();
219 
220             authStateClassCallHistory.push(caller);
221         }
222         ignoreAuth = true;
223     }

225     /**
226      * Restore the previous Authorisation System State. If the state was not
227      * changed by the current caller a warning will be displayed in log. Use:
228      * <code>
229      *     mycontext.turnOffAuthorisationSystem();
230      *     some java code that require no authorisation check
231      *     mycontext.restoreAuthSystemState(); 
232          * </code> If Context debug is enabled, the correct sequence calling will be
233      * checked and a warning will be displayed if not.
234      */
235     public void restoreAuthSystemState()
236     {
237         Boolean previousState;
238         try
239         {
240             previousState = authStateChangeHistory.pop();
241         }
242         catch (EmptyStackException ex)
243         {
244             log.warn(LogManager.getHeader(this, "restore_auth_sys_state",
245                     "not previous state info available "
246                             + ex.getLocalizedMessage()));
247             previousState = new Boolean(false);
248         }
249         if (log.isDebugEnabled())
250         {
251             Thread currThread = Thread.currentThread();
252             StackTraceElement[] stackTrace = currThread.getStackTrace();
253             String caller = stackTrace[stackTrace.length - 1].getClassName();
254 
255             String previousCaller = (String) authStateClassCallHistory.pop();
256 
257             // if previousCaller is not the current caller *only* log a warning
258             if (!previousCaller.equals(caller))
259             {
260                 log
261                         .warn(LogManager
262                                 .getHeader(
263                                         this,
264                                         "restore_auth_sys_state",
265                                         "Class: "
266                                                 + caller
267                                                 + " call restore but previous state change made by "
268                                                 + previousCaller));
269             }
270         }
271         ignoreAuth = previousState.booleanValue();
272     }

315     /**
316      * Close the context object after all of the operations performed in the
317      * context have completed succesfully. Any transaction with the database is
318      * committed.
319      * 
320      * @exception SQLException
321      *                if there was an error completing the database transaction
322      *                or closing the connection
323      */
324     public void complete() throws SQLException
325     {
326         // FIXME: Might be good not to do a commit() if nothing has actually
327         // been written using this connection
328         try
329         {
330             // Commit any changes made as part of the transaction
331             commit();
332         }
333         finally
334         {
335             // Free the connection
336             DatabaseManager.freeConnection(connection);
337             connection = null;
338         }
339     }
340 
341     /**
342      * Commit any transaction that is currently in progress, but do not close
343      * the context.
344      * 
345      * @exception SQLException
346      *                if there was an error completing the database transaction
347      *                or closing the connection
348      */
349     public void commit() throws SQLException
350     {
351         // Commit any changes made as part of the transaction
352         Dispatcher dispatcher = null;
353 
354         try
355         {
356             if (events != null)
357             {
358 
359                 if (dispName == null)
360                 {
361                     dispName = EventManager.DEFAULT_DISPATCHER;
362                 }
363 
364                 dispatcher = EventManager.getDispatcher(dispName);
365                 connection.commit();
366                 dispatcher.dispatch(this);
367             }
368             else
369             {
370                 connection.commit();
371             }
372 
373         }
374         finally
375         {
376             events = null;
377             if (dispatcher != null)
378             {
379                 EventManager.returnDispatcher(dispName, dispatcher);
380             }
381         }
382 
383     }

425     /**
426      * Close the context, without committing any of the changes performed using
427      * this context. The database connection is freed. No exception is thrown if
428      * there is an error freeing the database connection, since this method may
429      * be called as part of an error-handling routine where an SQLException has
430      * already been thrown.
431      */
432     public void abort()
433     {
434         try
435         {
436             if (!connection.isClosed())
437                 connection.rollback();
438         }
439         catch (SQLException se)
440         {
441             log.error(se.getMessage());
442             se.printStackTrace();
443         }
444         finally
445         {
446             try
447             {
448                 if (!connection.isClosed())
449                     DatabaseManager.freeConnection(connection);
450             }
451             catch (Exception ex)
452             {
453                 ex.printStackTrace();
454             }
455             connection = null;
456             events = null;
457         }
458     }

575     /**
576      * gets an array of all of the special groups that current user is a member
577      * of
578      * 
579      * @return
580      * @throws SQLException
581      */
582     public Group[] getSpecialGroups() throws SQLException
583     {
584         List myGroups = new ArrayList();
585 
586         Iterator i = specialGroups.iterator();
587 
588         while (i.hasNext())
589         {
590             myGroups.add(Group.find(this, ((Integer) i.next()).intValue()));
591         }
592 
593         return (Group[]) myGroups.toArray(new Group[0]);
594     }
