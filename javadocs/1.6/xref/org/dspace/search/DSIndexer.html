95  /**
96   * DSIndexer contains the methods that index Items and their metadata,
97   * collections, communities, etc. It is meant to either be invoked from the
98   * command line (see dspace/bin/index-all) or via the indexContent() methods
99   * within DSpace.
100  * 
101  * As of 1.4.2 this class has new incremental update of index functionality
102  * and better detection of locked state thanks to Lucene 2.1 moving write.lock.
103  * It will attempt to attain a lock on the index in the event that an update
104  * is requested and will wait a maximum of 30 seconds (a worst case scenario)
105  * to attain the lock before giving up and logging the failure to log4j and 
106  * to the DSpace administrator email account. 
107  * 
108  * The Administrator can choose to run DSIndexer in a cron that
109  * repeats regularly, a failed attempt to index from the UI will be "caught" up
110  * on in that cron.
111  * 
112  * @author Mark Diggory
113  * @author Graham Triggs
114  */
115 public class DSIndexer

171     static {
172     	
173     	// calculate maxfieldlength
174     	if (ConfigurationManager.getProperty("search.maxfieldlength") != null)
175         {
176             maxfieldlength = ConfigurationManager.getIntProperty("search.maxfieldlength");
177         }
178     	
179         // read in indexes from the config
180         ArrayList<String> indexConfigList = new ArrayList<String>();
181 
182         // read in search.index.1, search.index.2....
183         for (int i = 1; ConfigurationManager.getProperty("search.index." + i) != null; i++)
184         {
185             indexConfigList.add(ConfigurationManager.getProperty("search.index." + i));
186         }
187     
188         if (indexConfigList.size() > 0)
189         {
190             indexConfigArr = new IndexConfig[indexConfigList.size()];
191             
192             for (int i = 0; i < indexConfigList.size(); i++)
193             {
194                 indexConfigArr[i] = new IndexConfig();
195                 String index = indexConfigList.get(i);
196     
197                 String[] configLine = index.split(":");
198                 
199                 indexConfigArr[i].indexName = configLine[0];
200     
201                 // Get the schema, element and qualifier for the index
202                 // TODO: Should check valid schema, element, qualifier?
203                 String[] parts = configLine[1].split("\\.");
204     
205                 switch (parts.length)
206                 {
207                 case 3:
208                     indexConfigArr[i].qualifier = parts[2];
209                 case 2:
210                     indexConfigArr[i].schema  = parts[0];
211                     indexConfigArr[i].element = parts[1];
212                     break;
213                 default:
214                     log.warn("Malformed configuration line: search.index." + i);
215                     // FIXME: Can't proceed here, no suitable exception to throw
216                     throw new RuntimeException(
217                             "Malformed configuration line: search.index." + i);
218                 }
219 
220                 if (configLine.length > 2)
221                 {
222                     indexConfigArr[i].type = configLine[2];
223                 }
224             }
225         }
226         
227         /*
228          * Increase the default write lock so that Indexing can be interupted.
229          */
230         IndexWriter.setDefaultWriteLockTimeout(WRITE_LOCK_TIMEOUT);
231     
232         /*
233          * Create the index directory if it doesn't already exist.
234          */
235         if(!IndexReader.indexExists(index_directory))
236     	{
237     		try {
238     			new File(index_directory).mkdirs();
239 				openIndex(true).close();
240 			} catch (IOException e) {
241                 throw new RuntimeException(
242                         "Could not create search index: " + e.getMessage(),e);
243                }
244     	}

263     /**
264      * If the handle for the "dso" already exists in the index, and
265      * the "dso" has a lastModified timestamp that is newer than 
266      * the document in the index then it is updated, otherwise a 
267      * new document is added.
268      * 
269      * @param context Users Context
270      * @param dso DSpace Object (Item, Collection or Community
271      * @param force Force update even if not stale.
272      * @throws SQLException
273      * @throws IOException
274      */
275     public static void indexContent(Context context, DSpaceObject dso, boolean force)
276     throws SQLException
277     {
278     	
279     	String handle = dso.getHandle();
280     	
281     	if(handle == null)
282     	{
283     		handle = HandleManager.findHandle(context, dso);   
284     	}
285 
286 		Term t = new Term("handle", handle);
287 
288         try
289         {
290             switch (dso.getType())
291             {
292             case Constants.ITEM :
293                 Item item = (Item)dso;
294                 if (item.isArchived() && !item.isWithdrawn())
295                 {
296                     /** If the item is in the repository now, add it to the index*/
297                     if (requiresIndexing(t, ((Item)dso).getLastModified()) || force)
298                     {
299                         buildDocument(context, (Item) dso, t);                       
300                     }
301                 }       
302                 else
303                 {
304                     /** 
305                      * Make sure the item is not in the index if it is not in archive.
306                      * TODO: Someday DSIndexer should block withdrawn
307                      * content on search/retrieval and allow admins the ablitity to
308                      * still search for withdrawn Items.
309                      */
310                     DSIndexer.unIndexContent(context, handle);
311                     log.info("Removed Item: " + handle + " from Index");
312                 }
313                 break;
314                 
315             case Constants.COLLECTION :
316             	buildDocument(context, (Collection) dso, t);
317             	log.info("Wrote Collection: " + handle + " to Index");
318             	break;
319 
320             case Constants.COMMUNITY :
321             	buildDocument(context, (Community) dso, t);
322                 log.info("Wrote Community: " + handle + " to Index");
323                 break;
324                 
325             default :
326                 log.error("Only Items, Collections and Communities can be Indexed");
327             }
328 
329         } catch (Exception e) {
330 			log.error(e.getMessage(), e);
331 		}
332     }

454     /**
455      * When invoked as a command-line tool, creates, updates, removes 
456      * content from the whole index
457      *
458      * @param args
459      *            the command-line arguments, none used
460      * @throws IOException 
461      * @throws SQLException 
462      */
463     public static void main(String[] args) throws SQLException, IOException
464     {
465 
466     	Context context = new Context();
467         context.setIgnoreAuthorization(true);
468         
469         String usage = "org.dspace.search.DSIndexer [-cbhof[r <item handle>]] or nothing to update/clean an existing index.";
470         Options options = new Options();
471         HelpFormatter formatter = new HelpFormatter();
472         CommandLine line = null;
473 
474         options.addOption(OptionBuilder
475                         .withArgName("item handle")
476                         .hasArg(true)
477                         .withDescription(
478                                 "remove an Item, Collection or Community from index based on its handle")
479                         .create("r"));
480 
481         options.addOption(OptionBuilder.isRequired(false).withDescription(
482                 "optimize existing index").create("o"));
483 
484         options.addOption(OptionBuilder
485                         .isRequired(false)
486                         .withDescription(
487                                 "clean existing index removing any documents that no longer exist in the db")
488                         .create("c"));
489 
490         options.addOption(OptionBuilder.isRequired(false).withDescription(
491                 "(re)build index, wiping out current one if it exists").create(
492                 "b"));
493 
494         options.addOption(OptionBuilder
495                         .isRequired(false)
496                         .withDescription(
497                                 "if updating existing index, force each handle to be reindexed even if uptodate")
498                         .create("f"));
499 
500         options.addOption(OptionBuilder.isRequired(false).withDescription(
501                 "print this help message").create("h"));
502 
503         try
504         {
505             line = new PosixParser().parse(options, args);
506         }
507         catch (Exception e)
508         {
509             // automatically generate the help statement
510             formatter.printHelp(usage, e.getMessage(), options, "");
511             System.exit(1);
512         }
513 
514         if (line.hasOption("h"))
515         {
516             // automatically generate the help statement
517             formatter.printHelp(usage, options);
518             System.exit(1);
519         }
520 
521         if (line.hasOption("r"))
522         {
523             log.info("Removing " + line.getOptionValue("r") + " from Index");
524             unIndexContent(context, line.getOptionValue("r"));
525         }
526         else if (line.hasOption("o"))
527         {
528             log.info("Optimizing Index");
529             optimizeIndex(context);
530         }
531         else if (line.hasOption("c"))
532         {
533             log.info("Cleaning Index");
534             cleanIndex(context);
535         }
536         else if (line.hasOption("b"))
537         {
538             log.info("(Re)building index from scratch.");
539             createIndex(context);
540         }
541         else
542         {
543             log.info("Updating and Cleaning Index");
544             cleanIndex(context);
545             updateIndex(context, line.hasOption("f"));
546         }
547 
548         log.info("Done with indexing");
549     }

562     /**
563      * Iterates over all Items, Collections and Communities. And updates
564      * them in the index. Uses decaching to control memory footprint.
565      * Uses indexContent and isStale ot check state of item in index.
566      * 
567      * At first it may appear counterintuitive to have an IndexWriter/Reader
568      * opened and closed on each DSO. But this allows the UI processes
569      * to step in and attain a lock and write to the index even if other
570      * processes/jvms are running a reindex.
571      * 
572      * @param context
573      * @param force 
574      */
575     public static void updateIndex(Context context, boolean force) {
576     		try
577     		{
578                 ItemIterator items = null;
579                 try
580                 {
581                     for(items = Item.findAll(context);items.hasNext();)
582                     {
583                         Item item = (Item) items.next();
584                         indexContent(context,item,force);
585                         item.decache();
586                     }
587                 }
588                 finally
589                 {
590                     if (items != null)
591                         items.close();
592                 }
593 
594                 Collection[] collections = Collection.findAll(context);
595     	        for (int i = 0; i < collections.length; i++)
596     	        {
597     	            indexContent(context,collections[i],force);
598     	            context.removeCached(collections[i], collections[i].getID());
599     	            
600     	        }
601     		
602     			Community[] communities = Community.findAll(context);
603     	        for (int i = 0; i < communities.length; i++)
604     	        {
605     	            indexContent(context,communities[i],force);
606     	            context.removeCached(communities[i], communities[i].getID());
607     	        }
608     			
609     	        optimizeIndex(context);
610     			
611     		}
612     		catch(Exception e)
613     		{
614     			log.error(e.getMessage(), e);
615     		}
616     }

618     /**
619      * Iterates over all documents in the Lucene index and verifies they 
620      * are in database, if not, they are removed.
621      * 
622      * @param context
623      * @throws IOException 
624      * @throws SQLException 
625      */
626     public static void cleanIndex(Context context) throws IOException, SQLException {
627 
628     	IndexReader reader = DSQuery.getIndexReader();
629     	
630     	for(int i = 0 ; i < reader.numDocs(); i++)
631     	{
632     		if(!reader.isDeleted(i))
633     		{
634     			Document doc = reader.document(i);
635         		String handle = doc.get("handle");
636         		
637         		DSpaceObject o = HandleManager.resolveToObject(context, handle);
638 
639                 if (o == null)
640                 {
641                 	log.info("Deleting: " + handle);
642                 	/* Use IndexWriter to delete, its easier to manage write.lock */
643                 	DSIndexer.unIndexContent(context, handle);
644                 }
645                 else
646                 {
647                 	context.removeCached(o, o.getID());
648                 	log.debug("Keeping: " + handle);
649                 }
650     		}
651     		else
652     		{
653     			log.debug("Encountered deleted doc: " + i);
654     		}
655     	}
656 	}

658 	/**
659      * Get the Lucene analyzer to use according to current configuration (or
660      * default). TODO: Should have multiple analyzers (and maybe indices?) for
661      * multi-lingual DSpaces.
662      *
663      * @return <code>Analyzer</code> to use
664      * @throws IllegalStateException
665      *             if the configured analyzer can't be instantiated
666      */
667     static Analyzer getAnalyzer() throws IllegalStateException
668     {
669         if (analyzer == null)
670         {
671             // We need to find the analyzer class from the configuration
672             String analyzerClassName = ConfigurationManager.getProperty("search.analyzer");
673 
674             if (analyzerClassName == null)
675             {
676                 // Use default
677                 analyzerClassName = "org.dspace.search.DSAnalyzer";
678             }
679 
680             try
681             {
682                 Class analyzerClass = Class.forName(analyzerClassName);
683                 analyzer = (Analyzer) analyzerClass.newInstance();
684             }
685             catch (Exception e)
686             {
687                 log.fatal(LogManager.getHeader(null, "no_search_analyzer",
688                         "search.analyzer=" + analyzerClassName), e);
689 
690                 throw new IllegalStateException(e.toString());
691             }
692         }
693 
694         return analyzer;
695     }

702     private static void emailException(Exception exception) {
703 		// Also email an alert, system admin may need to check for stale lock
704 		try {
705 			String recipient = ConfigurationManager
706 					.getProperty("alert.recipient");
707 
708 			if (recipient != null) {
709 				Email email = ConfigurationManager.getEmail(I18nUtil.getEmailFilename(Locale.getDefault(), "internal_error"));
710 				email.addRecipient(recipient);
711 				email.addArgument(ConfigurationManager
712 						.getProperty("dspace.url"));
713 				email.addArgument(new Date());
714 
715 				String stackTrace;
716 
717 				if (exception != null) {
718 					StringWriter sw = new StringWriter();
719 					PrintWriter pw = new PrintWriter(sw);
720 					exception.printStackTrace(pw);
721 					pw.flush();
722 					stackTrace = sw.toString();
723 				} else {
724 					stackTrace = "No exception";
725 				}
726 
727 				email.addArgument(stackTrace);
728 				email.send();
729 			}
730 		} catch (Exception e) {
731 			// Not much we can do here!
732 			log.warn("Unable to send email alert", e);
733 		}
734 
735 	}

737     /**
738 	 * Is stale checks the lastModified time stamp in the database and the index
739 	 * to determine if the index is stale.
740 	 * 
741 	 * @param handle
742 	 * @param lastModified
743 	 * @throws SQLException
744 	 * @throws IOException
745 	 */
746     private static boolean requiresIndexing(Term t, Date lastModified)
747     throws SQLException, IOException
748     {
749 		
750 		boolean reindexItem = false;
751 		boolean inIndex = false;
752 		
753 		IndexReader ir = DSQuery.getIndexReader();
754

1210      *  
1211      * @param t
1212      * @return
1213      */
1214     private static Date toDate(String t)
1215     {
1216         SimpleDateFormat[] dfArr;
1217 
1218         // Choose the likely date formats based on string length
1219         switch (t.length())
1220         {
1221             case 4:
1222                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyy") };
1223                 break;
1224             case 6:
1225                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyyMM") };
1226                 break;
1227             case 7:
1228                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyy-MM") };
1229                 break;
1230             case 8:
1231                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyyMMdd"), new SimpleDateFormat("yyyy MMM") };
1232                 break;
1233             case 10:
1234                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyy-MM-dd") };
1235                 break;
1236             case 11:
1237                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyy MMM dd") };
1238                 break;
1239             case 20:
1240                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'") };
1241                 break;
1242             default:
1243                 dfArr = new SimpleDateFormat[] { new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSS'Z'") };
1244                 break;
1245         }
1246 
1247 
1248         for (SimpleDateFormat df : dfArr)
1249         {
1250             try
1251             {
1252                 // Parse the date
1253                 df.setCalendar(Calendar.getInstance(TimeZone.getTimeZone("UTC")));
1254                 df.setLenient(false);
1255                 return df.parse(t);
1256             }
1257             catch (ParseException pe)
