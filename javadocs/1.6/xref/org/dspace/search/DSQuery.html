113     /**
114      * Do a query, returning a QueryResults object
115      *
116      * @param c  context
117      * @param args query arguments in QueryArgs object
118      * 
119      * @return query results QueryResults
120      */
121     public static QueryResults doQuery(Context c, QueryArgs args)
122             throws IOException
123     {
124         String querystring = args.getQuery();
125         QueryResults qr = new QueryResults();
126         List hitHandles = new ArrayList();
127         List hitIds     = new ArrayList();
128         List hitTypes   = new ArrayList();
129 
130         // set up the QueryResults object
131         qr.setHitHandles(hitHandles);
132         qr.setHitIds(hitIds);
133         qr.setHitTypes(hitTypes);
134         qr.setStart(args.getStart());
135         qr.setPageSize(args.getPageSize());
136         qr.setEtAl(args.getEtAl());
137 
138         // massage the query string a bit
139         querystring = checkEmptyQuery(querystring); // change nulls to an empty string
140         // We no longer need to work around the Lucene bug with recent versions
141         //querystring = workAroundLuceneBug(querystring); // logicals changed to && ||, etc.
142         querystring = stripHandles(querystring); // remove handles from query string
143         querystring = stripAsterisk(querystring); // remove asterisk from beginning of string
144 
145         try
146         {
147             // grab a searcher, and do the search
148             Searcher searcher = getSearcher(c);
149 
150             QueryParser qp = new QueryParser("default", DSIndexer.getAnalyzer());
151             log.debug("Final query string: " + querystring);
152             
153             if (operator == null || operator.equals("OR"))
154             {
155             	qp.setDefaultOperator(QueryParser.OR_OPERATOR);
156             }
157             else
158             {
159             	qp.setDefaultOperator(QueryParser.AND_OPERATOR);
160             }
161             
162             Query myquery = qp.parse(querystring);
163             Hits hits = null;
164 
165             try
166             {
167                 if (args.getSortOption() == null)
168                 {
169                     SortField[] sortFields = new SortField[] {
170                             new SortField("search.resourcetype", true),
171                             new SortField(null, SortField.SCORE, SortOption.ASCENDING.equals(args.getSortOrder()))
172                         };
173                     hits = searcher.search(myquery, new Sort(sortFields));
174                 }
175                 else
176                 {
177                     SortField[] sortFields = new SortField[] {
178                             new SortField("search.resourcetype", true),
179                             new SortField("sort_" + args.getSortOption().getName(), SortOption.DESCENDING.equals(args.getSortOrder())),
180                             SortField.FIELD_SCORE
181                         };
182                     hits = searcher.search(myquery, new Sort(sortFields));
183                 }
184             }
185             catch (Exception e)
186             {
187                 // Lucene can throw an exception if it is unable to determine a sort time from the specified field
188                 // Provide a fall back that just works on relevancy.
189                 log.error("Unable to use speficied sort option: " + (args.getSortOption() == null ? "type/relevance": args.getSortOption().getName()));
190                 hits = searcher.search(myquery, new Sort(SortField.FIELD_SCORE));
191             }
192             
193             // set total number of hits
194             qr.setHitCount(hits.length());
195 
196             // We now have a bunch of hits - snip out a 'window'
197             // defined in start, count and return the handles
198             // from that window
199             // first, are there enough hits?
200             if (args.getStart() < hits.length())
201             {
202                 // get as many as we can, up to the window size
203                 // how many are available after snipping off at offset 'start'?
204                 int hitsRemaining = hits.length() - args.getStart();
205 
206                 int hitsToProcess = (hitsRemaining < args.getPageSize()) ? hitsRemaining
207                         : args.getPageSize();
208 
209                 for (int i = args.getStart(); i < (args.getStart() + hitsToProcess); i++)
210                 {
211                     Document d = hits.doc(i);
212 
213                     String resourceId   = d.get("search.resourceid");
214                     String resourceType = d.get("search.resourcetype");
215 
216                     String handleText = d.get("handle");
217                     String handleType = d.get("type");
218 
219                     switch (Integer.parseInt( resourceType != null ? resourceType : handleType))
220                     {
221                         case Constants.ITEM:
222                             hitTypes.add(new Integer(Constants.ITEM));
223                             break;
224 
225                         case Constants.COLLECTION:
226                             hitTypes.add(new Integer(Constants.COLLECTION));
227                             break;
228 
229                         case Constants.COMMUNITY:
230                             hitTypes.add(new Integer(Constants.COMMUNITY));
231                             break;
232                     }
233 
234                     hitHandles.add( handleText );
235                     hitIds.add( resourceId == null ? null: Integer.parseInt(resourceId) );
236                 }
237             }
238         }
239         catch (NumberFormatException e)
240         {
241             log.warn(LogManager.getHeader(c, "Number format exception", "" + e));
242             qr.setErrorMsg("number-format-exception");
243         }
244         catch (ParseException e)
245         {
246             // a parse exception - log and return null results
247             log.warn(LogManager.getHeader(c, "Invalid search string", "" + e));
248             qr.setErrorMsg("invalid-search-string");
249         }
250         catch (TokenMgrError tme)
251         {
252             // Similar to parse exception
253             log.warn(LogManager.getHeader(c, "Invalid search string", "" + tme));
254             qr.setErrorMsg("invalid-search-string");
255         }
256         catch(BooleanQuery.TooManyClauses e)
257         {
258             log.warn(LogManager.getHeader(c, "Query too broad", e.toString()));
259             qr.setErrorMsg("query-too-broad");
260         }
261 
262         return qr;
263     }

265     static String checkEmptyQuery(String myquery)
266     {
267         if (myquery == null || myquery.equals("()") || myquery.equals(""))
268         {
269             myquery = "empty_query_string";
270         }
271 
272         return myquery;

275     /**
276      * Workaround Lucene bug that breaks wildcard searching.
277      * This is no longer required with Lucene upgrades.
278      * 
279      * @param myquery
280      * @return
281      * @deprecated
282      */
283     static String workAroundLuceneBug(String myquery)
284     {
285         // Lucene currently has a bug which breaks wildcard
286         // searching when you have uppercase characters.
287         // Here we substitute the boolean operators -- which
288         // have to be uppercase -- before tranforming the
289         // query string to lowercase.
290         return myquery.replaceAll(" AND ", " && ")
291                       .replaceAll(" OR ", " || ")
292                       .replaceAll(" NOT ", " ! ")
293                       .toLowerCase();
294     }

296     static String stripHandles(String myquery)
297     {
298         // Drop beginning pieces of full handle strings
299         return myquery.replaceAll("^\\s*http://hdl\\.handle\\.net/", "")
300                       .replaceAll("^\\s*hdl:", "");
301     }
302 
303     static String stripAsterisk(String myquery)
304     {
305         // query strings (or words) begining with "*" cause a null pointer error
306         return myquery.replaceAll("^\\*", "")
307                       .replaceAll("\\s\\*", " ")
308                       .replaceAll("\\(\\*", "(")
309                       .replaceAll(":\\*", ":");
310     }

312     /**
313      * Do a query, restricted to a collection
314      * 
315      * @param c
316      *            context
317      * @param args
318      *            query args
319      * @param coll
320      *            collection to restrict to
321      * 
322      * @return QueryResults same results as doQuery, restricted to a collection
323      */
324     public static QueryResults doQuery(Context c, QueryArgs args,
325             Collection coll) throws IOException
326     {
327         String querystring = args.getQuery();
328 
329         querystring = checkEmptyQuery(querystring);
330 
331         String location = "l" + (coll.getID());
332 
333         String newquery = new String("+(" + querystring + ") +location:\""
334                 + location + "\"");
335 
336         args.setQuery(newquery);
337 
338         return doQuery(c, args);
339     }

371     /**
372      * Do a query, printing results to stdout largely for testing, but it is
373      * useful
374      */
375     public static void doCMDLineQuery(String query)
376     {
377         System.out.println("Command line query: " + query);
378         System.out.println("Only reporting default-sized results list");
379 
380         try
381         {
382             Context c = new Context();
383 
384             QueryArgs args = new QueryArgs();
385             args.setQuery(query);
386 
387             QueryResults results = doQuery(c, args);
388 
389             Iterator i = results.getHitHandles().iterator();
390             Iterator j = results.getHitTypes().iterator();
391 
392             while (i.hasNext())
393             {
394                 String thisHandle = (String) i.next();
395                 Integer thisType = (Integer) j.next();
396                 String type = Constants.typeText[thisType.intValue()];
397 
398                 // also look up type
399                 System.out.println(type + "\t" + thisHandle);
400             }
401         }
402         catch (Exception e)
403         {
404             System.out.println("Exception caught: " + e);
405         }
406     }

447     /**
448      * get an IndexSearcher, hopefully a cached one (gives much better
449      * performance.) checks to see if the index has been modified - if so, it
450      * creates a new IndexSearcher
451      */
452     protected static synchronized IndexSearcher getSearcher(Context c)
453             throws IOException
454     {
455        
456         // If we have already opened a searcher, check to see if the index has been updated
457         // If it has, we need to close the existing searcher - we will open a new one later
458         if (searcher != null && lastModified != IndexReader.getCurrentVersion(indexDir))
459         {
460             try
461             {
462                 // Close the cached IndexSearcher
463                 searcher.close();
464             }
465             catch (IOException ioe)
466             {
467                 // Index is probably corrupt. Log the error, but continue to either:
468                 // 1) Return existing searcher (may yet throw exception, no worse than throwing here)
469                 log.warn("DSQuery: Unable to check for updated index", ioe);
470             }
471             finally
472             {
473             	searcher = null;
474             }
475         }
476 
477         // There is no existing searcher - either this is the first execution,
478         // or the index has been updated and we closed the old index.
479         if (searcher == null)
480         {
481             // So, open a new searcher
482             lastModified = IndexReader.getCurrentVersion(indexDir);
483             searcher = new IndexSearcher(indexDir){
484             	/* 
485             	 * TODO: Has Lucene fixed this bug yet?
486             	 * Lucene doesn't release read locks in 
487             	 * windows properly on finalize. Our hack
488             	 * extend IndexSearcher to force close().
489             	 */
490                 protected void finalize() throws Throwable {
491             		this.close();
492             		super.finalize();
493             	}
494             };
495         }
496 
497         return searcher;
498     }
