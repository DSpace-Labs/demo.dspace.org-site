69  /**
70   * Public interface to the embargo subsystem.
71   *
72   * Configuration properties: (with examples)
73   *   # DC metadata field to hold the user-supplied embargo terms
74   *   embargo.field.terms = dc.embargo.terms
75   *   # DC metadata field to hold computed "lift date" of embargo
76   *   embargo.field.lift = dc.date.available
77   *   # String to indicate indefinite (forever) embargo in terms
78   *   embargo.terms.open = Indefinite
79   *   # implementation of embargo setter plugin
80   *   plugin.single.org.dspace.embargo.EmbargoSetter = edu.my.Setter
81   *   # implementation of embargo lifter plugin
82   *   plugin.single.org.dspace.embargo.EmbargoLifter = edu.my.Lifter

84   * @author Larry Stone
85   * @author Richard Rodgers

114     /**
115      * Put an Item under embargo until the specified lift date.
116      * Calls EmbargoSetter plugin to adjust Item access control policies.
117      *
118      * @param context the DSpace context
119      * @param item the item to embargo
120      * @param lift date on which the embargo is to be lifted.
121      */
122     public static void setEmbargo(Context context, Item item, DCDate lift)
123         throws SQLException, AuthorizeException, IOException
124     {
125         init();
126         String slift = lift.toString();
127         boolean ignoreAuth = context.ignoreAuthorization();
128         try
129         {
130             context.setIgnoreAuthorization(true);
131             item.clearMetadata(lift_schema, lift_element, lift_qualifier, Item.ANY);
132             item.addMetadata(lift_schema, lift_element, lift_qualifier, null, slift);
133             log.info("Set embargo on Item "+item.getHandle()+", expires on: "+slift);
134             setter.setEmbargo(context, item);
135             item.update();
136         }
137         finally
138         {
139             context.setIgnoreAuthorization(ignoreAuth);
140         }
141     }

143     /**
144      * Get the embargo lift date for an Item, if any.  This looks for the
145      * metadata field configured to hold embargo terms, and gives it
146      * to the EmbargoSetter plugin's method to interpret it into
147      * an absolute timestamp.  This is intended to be called at the time
148      * the Item is installed into the archive.
149      * <p>
150      * Note that the plugin is *always* called, in case it gets its cue for
151      * the embargo date from sources other than, or in addition to, the
152      * specified field.
153      *
154      * @param context the DSpace context
155      * @param item the item to embargo
156      * @return lift date on which the embargo is to be lifted, or null if none
157      */
158     public static DCDate getEmbargoDate(Context context, Item item)
159         throws SQLException, AuthorizeException, IOException
160     {
161         init();
162         DCValue terms[] = item.getMetadata(terms_schema, terms_element, terms_qualifier, Item.ANY);
163         DCDate result = setter.parseTerms(context, item, terms.length > 0 ? terms[0].value : null);
164 
165         // sanity check: do not allow an embargo lift date in the past.
166         if (result != null && result.toDate().before(new Date()))
167         {
168             throw new IllegalArgumentException("Embargo lift date must be in the future, but this is in the past: "+result.toString());
169         }
170         return result;
171     }

173     /**
174      * Lift the embargo on an item which is assumed to be under embargo.
175      * Call the plugin to manage permissions in its own way, then delete
176      * the administrative metadata fields that dictated embargo date.
177      *
178      * @param context the DSpace context
179      * @param item the item on which to lift the embargo
180      */
181     public static void liftEmbargo(Context context, Item item)
182         throws SQLException, AuthorizeException, IOException
183     {
184         init();
185         lifter.liftEmbargo(context, item);
186         item.clearMetadata(lift_schema, lift_element, lift_qualifier, Item.ANY);
187 
188         // set the dc.date.available value to right now
189         item.clearMetadata(MetadataSchema.DC_SCHEMA, "date", "available", Item.ANY);
190         item.addMetadata(MetadataSchema.DC_SCHEMA, "date", "available", null, DCDate.getCurrent().toString());
191 
192         log.info("Lifting embargo on Item "+item.getHandle());
193         item.update();
194     }

196     /**
197      * Command-line service to scan for every Items with an expired embargo,
198      * and then lift that embargo.
199      *
200      * Options:
201      *   -c,--check         Function: ONLY check the state of embargoed Items, do
202      *                      NOT lift any embargoes.
203      *   -h,--help          help
204      *   -i,--identifier    Process ONLY this Handle identifier(s), which must be
205      *                      an Item.  Can be repeated.
206      *   -l,--lift          Function: ONLY lift embargoes, do NOT check the state
207      *                      of any embargoed Items.
208      *   -n,--dryrun        Do not change anything in the data model, print
209      *                      message instead.
210      *   -v,--verbose       Print a line describing action taken for each
211      *                      embargoed Item found.
212      *   -q,--quiet         No output except upon error
213      */
214     public static void main(String argv[])
215     {
216         init();
217         int status = 0;
218 
219         Options options = new Options();
220         options.addOption("v", "verbose", false,
221                 "Print a line describing action taken for each embargoed Item found.");
222         options.addOption("q", "quiet", false,
223                 "Do not print anything except for errors.");
224         options.addOption("n", "dryrun", false,
225                 "Do not change anything in the data model, print message instead.");
226         options.addOption("i", "identifier", true,
227                         "Process ONLY this Handle identifier(s), which must be an Item.  Can be repeated.");
228         options.addOption("c", "check", false,
229                         "Function: ONLY check the state of embargoed Items, do NOT lift any embargoes.");
230         options.addOption("l", "lift", false,
231                         "Function: ONLY lift embargoes, do NOT check the state of any embargoed Items.");
232         options.addOption("h", "help", false, "help");
233         CommandLine line = null;
234         try
235         {
236             line = new PosixParser().parse(options, argv);
237         }
238         catch(ParseException e)
239         {
240             System.err.println("Command error: " + e.getMessage());
241             new HelpFormatter().printHelp(EmbargoManager.class.getName(), options);
242             System.exit(1);
243         }
244 
245         if (line.hasOption('h'))
246         {
247             new HelpFormatter().printHelp(EmbargoManager.class.getName(), options);
248             System.exit(0);
249         }
250 
251         // sanity check, --lift and --check are mutually exclusive:
252         if (line.hasOption('l') && line.hasOption('c'))
253         {
254             System.err.println("Command error: --lift and --check are mutually exclusive, try --help for assistance.");
255             System.exit(1);
256         }
257 
258         Context context = null;
259         try
260         {
261             context = new Context();
262             context.setIgnoreAuthorization(true);
263             Date now = new Date();
264              
265             // scan items under embargo
266             if (line.hasOption('i'))
267             {
268                 for (String handle : line.getOptionValues('i'))
269                 {
270                     DSpaceObject dso = HandleManager.resolveToObject(context, handle);
271                     if (dso == null)
272                     {
273                         System.err.println("Error, cannot resolve handle="+handle+" to a DSpace Item.");
274                         status = 1;
275                     }
276                     else if (dso.getType() != Constants.ITEM)
277                     {
278                         System.err.println("Error, the handle="+handle+" is not a DSpace Item.");
279                         status = 1;
280                     }
281                     else
282                     {
283                         if (processOneItem(context, (Item)dso, line, now))
284                             status = 1;
285                     }
286                 }
287             }
288             else
289             {
290                 ItemIterator ii = Item.findByMetadataField(context, lift_schema, lift_element, lift_qualifier, Item.ANY);
291                 while (ii.hasNext())
292                 {
293                     if (processOneItem(context, ii.next(), line, now))
294                         status = 1;
295                 }
296             }
297             log.debug("Cache size at end = "+context.getCacheSize());
298             context.complete();
299             context = null;
300         }
301         catch (Exception e)
302         {
303             System.err.println("ERROR, got exception: "+e);
304             e.printStackTrace();
305             status = 1;
306         }
307         finally
308         {
309             if (context != null)
310             {
311                 try
312                 {
313                     context.abort();
314                 }
315                 catch (Exception e)
316                 {
317                 }
318             }
319         }
320         System.exit(status);
321     }

323     // lift or check embargo on one Item, handle exceptions
324     // return flase on success, true if there was fatal exception.
325     private static boolean processOneItem(Context context, Item item, CommandLine line, Date now)
326         throws Exception
327     {
328         boolean status = false;
329         DCValue lift[] = item.getMetadata(lift_schema, lift_element, lift_qualifier, Item.ANY);
330 
331         if (lift.length > 0)
332         {
333             // need to survive any failure on a single item, go on to process the rest.
334             try
335             {
336                 DCDate liftDate = new DCDate(lift[0].value);
337                 log.debug("Testing embargo on item="+item.getHandle()+", date="+liftDate.toString());
338                 if (liftDate.toDate().before(now))
339                 {
340                     if (line.hasOption('v'))
341                         System.err.println("Lifting embargo from Item handle="+item.getHandle()+", lift date="+lift[0].value);
342                     if (line.hasOption('n'))
343                     {
344                         if (!line.hasOption('q'))
345                             System.err.println("DRY RUN: would have lifted embargo from Item handle="+item.getHandle()+", lift date="+lift[0].value);
346                     }
347                     else if (!line.hasOption('c'))
348                         liftEmbargo(context, item);
349                 }
350                 else if (!line.hasOption('l'))
351                 {
352                     if (line.hasOption('v'))
353                         System.err.println("Checking current embargo on Item handle="+item.getHandle()+", lift date="+lift[0].value);
354                     setter.checkEmbargo(context, item);
355                 }
356             }
357             catch (Exception e)
358             {
359                 log.error("Failed attempting to lift embargo, item="+item.getHandle()+": ", e);
360                 System.err.println("Failed attempting to lift embargo, item="+item.getHandle()+": "+ e);
361                 status = true;
362             }
363         }
364         context.removeCached(item, item.getID());
365         return status;
366     }

368     // initialize - get plugins and MD field settings from config
369     private static void init()
370     {
371         if (terms_schema == null)
372         {
373             String terms = ConfigurationManager.getProperty("embargo.field.terms");
374             String lift = ConfigurationManager.getProperty("embargo.field.lift");
375             if (terms == null || lift == null)
376                 throw new IllegalStateException("Missing one or more of the required DSpace configuration properties for EmbargoManager, check your configuration file.");
377             terms_schema = getSchemaOf(terms);
378             terms_element = getElementOf(terms);
379             terms_qualifier = getQualifierOf(terms);
380             lift_schema = getSchemaOf(lift);
381             lift_element = getElementOf(lift);
382             lift_qualifier = getQualifierOf(lift);
383 
384             setter = (EmbargoSetter)PluginManager.getSinglePlugin(EmbargoSetter.class);
385             if (setter == null)
386                 throw new IllegalStateException("The EmbargoSetter plugin was not defined in DSpace configuration.");
387             lifter = (EmbargoLifter)PluginManager.getSinglePlugin(EmbargoLifter.class);
388             if (lifter == null)
389                 throw new IllegalStateException("The EmbargoLifter plugin was not defined in DSpace configuration.");
390 
391         }
392     }

394     // return the schema part of "schema.element.qualifier" metadata field
395     private static String getSchemaOf(String field)
396     {
397         String sa[] = field.split("\\.", 3);
398         return sa[0];
399     }
400 
401     // return the schema part of "schema.element.qualifier" metadata field
402     private static String getElementOf(String field)
403     {
404         String sa[] = field.split("\\.", 3);
405         return sa.length > 0 ? sa[1] : null;
406     }
407 
408     // return the schema part of "schema.element.qualifier" metadata field
409     private static String getQualifierOf(String field)
410     {
411         String sa[] = field.split("\\.", 3);
412         return sa.length > 1 ? sa[2] : null;
413     }
