123     /**
124      * Populate Group with eperson and group objects
125      * 
126      * @throws SQLException
127      */
128     public void loadData()
129     {
130         // only populate if not already populated
131         if (!isDataLoaded)
132         {
133             // naughty thing to do - swallowing SQL exception and throwing it as
134             // a RuntimeException - a hack to avoid changing the API all over
135             // the place
136             try
137             {
138                 // get epeople objects
139                 TableRowIterator tri = DatabaseManager.queryTable(myContext,"eperson",
140                                 "SELECT eperson.* FROM eperson, epersongroup2eperson WHERE " +
141                                 "epersongroup2eperson.eperson_id=eperson.eperson_id AND " +
142                                 "epersongroup2eperson.eperson_group_id= ?",
143                                 myRow.getIntColumn("eperson_group_id"));
144 
145                 try
146                 {
147                     while (tri.hasNext())
148                     {
149                         TableRow r = (TableRow) tri.next();
150 
151                         // First check the cache
152                         EPerson fromCache = (EPerson) myContext.fromCache(
153                                 EPerson.class, r.getIntColumn("eperson_id"));
154 
155                         if (fromCache != null)
156                         {
157                             epeople.add(fromCache);
158                         }
159                         else
160                         {
161                             epeople.add(new EPerson(myContext, r));
162                         }
163                     }
164                 }
165                 finally
166                 {
167                     // close the TableRowIterator to free up resources
168                     if (tri != null)
169                         tri.close();
170                 }
171 
172                 // now get Group objects
173                 tri = DatabaseManager.queryTable(myContext,"epersongroup",
174                                 "SELECT epersongroup.* FROM epersongroup, group2group WHERE " +
175                                 "group2group.child_id=epersongroup.eperson_group_id AND "+
176                                 "group2group.parent_id= ? ",
177                                 myRow.getIntColumn("eperson_group_id"));
178 
179                 try
180                 {
181                     while (tri.hasNext())
182                     {
183                         TableRow r = (TableRow) tri.next();
184 
185                         // First check the cache
186                         Group fromCache = (Group) myContext.fromCache(Group.class,
187                                 r.getIntColumn("eperson_group_id"));
188 
189                         if (fromCache != null)
190                         {
191                             groups.add(fromCache);
192                         }
193                         else
194                         {
195                             groups.add(new Group(myContext, r));
196                         }
197                     }
198                 }
199                 finally
200                 {
201                     // close the TableRowIterator to free up resources
202                     if (tri != null)
203                         tri.close();
204                 }
205 
206             }
207             catch (Exception e)
208             {
209                 throw new RuntimeException(e);
210             }
211             isDataLoaded = true;
212         }
213     }

299      * add group to this group
300      * 
301      * @param g
302      */
303     public void addMember(Group g)
304     {
305         loadData(); // make sure Group has data loaded
306 
307         // don't add if it's already a member
308         // and don't add itself
309         if (isMember(g) || getID()==g.getID())

312         }
313 
314         groups.add(g);
315         groupsChanged = true;
316 
317         myContext.addEvent(new Event(Event.ADD, Constants.GROUP, getID(), Constants.GROUP, g.getID(), g.getName()));
318     }

401         if (groupid == 0)

412      * Get all of the groups that an eperson is a member of
413      * 
414      * @param c
415      * @param e
416      * @return
417      * @throws SQLException
418      */
419     public static Group[] allMemberGroups(Context c, EPerson e)
420             throws SQLException
421     {
422         List<Group> groupList = new ArrayList<Group>();
423 
424         Set<Integer> myGroups = allMemberGroupIDs(c, e);
425         // now convert those Integers to Groups
426         Iterator i = myGroups.iterator();
427 
428         while (i.hasNext())
429         {
430             groupList.add(Group.find(c, ((Integer) i.next()).intValue()));
431         }
432 
433         return (Group[]) groupList.toArray(new Group[0]);
434     }

436     /**
437      * get Set of Integers all of the group memberships for an eperson
438      * 
439      * @param c
440      * @param e
441      * @return Set of Integer groupIDs
442      * @throws SQLException
443      */
444     public static Set<Integer> allMemberGroupIDs(Context c, EPerson e)
445             throws SQLException
446     {
447         Set<Integer> groupIDs = new HashSet<Integer>();
448         
449         if (e != null)
450         {
451             // two queries - first to get groups eperson is a member of
452             // second query gets parent groups for groups eperson is a member of
453 
454             TableRowIterator tri = DatabaseManager.queryTable(c,
455                     "epersongroup2eperson",
456                     "SELECT * FROM epersongroup2eperson WHERE eperson_id= ?", e
457                             .getID());
458 
459             try
460             {
461                 while (tri.hasNext())
462                 {
463                     TableRow row = tri.next();
464 
465                     int childID = row.getIntColumn("eperson_group_id");
466 
467                     groupIDs.add(new Integer(childID));
468                 }
469             }
470             finally
471             {
472                 // close the TableRowIterator to free up resources
473                 if (tri != null)
474                     tri.close();
475             }
476         }
477         // Also need to get all "Special Groups" user is a member of!
478         // Otherwise, you're ignoring the user's membership to these groups!
479         // However, we only do this is we are looking up the special groups
480         // of the current user, as we cannot look up the special groups
481         // of a user who is not logged in.
482         if ((c.getCurrentUser() != null) && (c.getCurrentUser().getID() == e.getID()))
483         {
484             Group[] specialGroups = c.getSpecialGroups();
485             for(Group special : specialGroups)
486             {
487                 groupIDs.add(new Integer(special.getID()));
488             }
489         }
490 
491         // all the users are members of the anonymous group 
492         groupIDs.add(new Integer(0));
493         
494         // now we have all owning groups, also grab all parents of owning groups
495         // yes, I know this could have been done as one big query and a union,
496         // but doing the Oracle port taught me to keep to simple SQL!
497 
498         String groupQuery = "";
499 
500         Iterator i = groupIDs.iterator();
501 
502         // Build a list of query parameters
503         Object[] parameters = new Object[groupIDs.size()];
504         int idx = 0;
505         while (i.hasNext())
506         {
507             int groupID = ((Integer) i.next()).intValue();
508 
509             parameters[idx++] = new Integer(groupID);
510             
511             groupQuery += "child_id= ? ";
512             if (i.hasNext())
513                 groupQuery += " OR ";
514         }
515 
516         // was member of at least one group
517         // NOTE: even through the query is built dynamicaly all data is seperated into the
518         // the parameters array.
519         TableRowIterator tri = DatabaseManager.queryTable(c, "group2groupcache",
520                 "SELECT * FROM group2groupcache WHERE " + groupQuery,
521                 parameters);
522 
523         try
524         {
525             while (tri.hasNext())
526             {
527                 TableRow row = tri.next();
528 
529                 int parentID = row.getIntColumn("parent_id");
530 
531                 groupIDs.add(new Integer(parentID));
532             }
533         }
534         finally
535         {
536             // close the TableRowIterator to free up resources
537             if (tri != null)
538                 tri.close();
539         }
540 
541         return groupIDs;
542     }

574     /**
575      * Get Set of all Integers all of the epeople
576      * members for a group
577      * 
578      * @param c
579      *          DSpace context
580      * @param g
581      *          Group object
582      * @return Set of Integer epersonIDs
583      * @throws SQLException
584      */
585     public static Set<Integer> allMemberIDs(Context c, Group g)
586             throws SQLException
587     {
588         // two queries - first to get all groups which are a member of this group
589         // second query gets all members of each group in the first query
590         Set<Integer> epeopleIDs = new HashSet<Integer>();
591         
592         // Get all groups which are a member of this group
593         TableRowIterator tri = DatabaseManager.queryTable(c, "group2groupcache",
594                 "SELECT * FROM group2groupcache WHERE parent_id= ? ",
595                 g.getID());
596         
597         Set<Integer> groupIDs = new HashSet<Integer>();
598 
599         try
600         {
601             while (tri.hasNext())
602             {
603                 TableRow row = tri.next();
604 
605                 int childID = row.getIntColumn("child_id");
606 
607                 groupIDs.add(new Integer(childID));
608             }
609         }
610         finally
611         {
612             // close the TableRowIterator to free up resources
613             if (tri != null)
614                 tri.close();
615         }
616 
617         // now we have all the groups (including this one)
618         // it is time to find all the EPeople who belong to those groups
619         // and filter out all duplicates
620 
621         Object[] parameters = new Object[groupIDs.size()+1];
622         int idx = 0;
623         Iterator i = groupIDs.iterator();
624 
625         // don't forget to add the current group to this query!
626         parameters[idx++] = new Integer(g.getID());
627         String epersonQuery = "eperson_group_id= ? ";
628         if (i.hasNext())
629             epersonQuery += " OR ";
630         
631         while (i.hasNext())
632         {
633             int groupID = ((Integer) i.next()).intValue();
634             parameters[idx++] = new Integer(groupID);
635             
636             epersonQuery += "eperson_group_id= ? ";
637             if (i.hasNext())
638                 epersonQuery += " OR ";
639         }
640 
641         //get all the EPerson IDs
642         // Note: even through the query is dynamicaly built all data is seperated
643         // into the parameters array.
644         tri = DatabaseManager.queryTable(c, "epersongroup2eperson",
645                 "SELECT * FROM epersongroup2eperson WHERE " + epersonQuery,
646                 parameters);
647 
648         try
649         {
650             while (tri.hasNext())
651             {
652                 TableRow row = tri.next();
653 
654                 int epersonID = row.getIntColumn("eperson_id");
655 
656                 epeopleIDs.add(new Integer(epersonID));
657             }
658         }
659         finally
660         {
661             // close the TableRowIterator to free up resources
662             if (tri != null)
663                 tri.close();
664         }
665 
666         return epeopleIDs;
667     }

740     /**
741      * Finds all groups in the site
742      * 
743      * @param context
744      *            DSpace context
745      * @param sortField
746      *            field to sort by -- Group.ID or Group.NAME
747      * 
748      * @return array of all groups in the site
749      */
750     public static Group[] findAll(Context context, int sortField)
751             throws SQLException
752     {
753         String s;
754 
755         switch (sortField)
756         {
757         case ID:
758             s = "eperson_group_id";
759 
760             break;
761 
762         case NAME:
763             s = "name";
764 
765             break;
766 
767         default:
768             s = "name";
769         }
770 
771         // NOTE: The use of 's' in the order by clause can not cause an sql 
772         // injection because the string is derived from constant values above.
773         TableRowIterator rows = DatabaseManager.queryTable(
774         		context, "epersongroup",
775                 "SELECT * FROM epersongroup ORDER BY "+s);
776 
777         try
778         {
779             List gRows = rows.toList();
780 
781             Group[] groups = new Group[gRows.size()];
782 
783             for (int i = 0; i < gRows.size(); i++)
784             {
785                 TableRow row = (TableRow) gRows.get(i);
786 
787                 // First check the cache
788                 Group fromCache = (Group) context.fromCache(Group.class, row
789                         .getIntColumn("eperson_group_id"));
790 
791                 if (fromCache != null)
792                 {
793                     groups[i] = fromCache;
794                 }
795                 else
796                 {
797                     groups[i] = new Group(context, row);
798                 }
799             }
800 
801             return groups;
802         }
803         finally
804         {
805             if (rows != null)
806                 rows.close();
807         }
808     }

827     /**
828      * Find the groups that match the search query across eperson_group_id or name
829      * 
830      * @param context
831      *            DSpace context
832      * @param query
833      *            The search string
834      * @param offset
835      *            Inclusive offset 
836      * @param limit
837      *            Maximum number of matches returned
838      * 
839      * @return array of Group objects
840      */
841     public static Group[] search(Context context, String query, int offset, int limit)
842     		throws SQLException
843 	{
844 		String params = "%"+query.toLowerCase()+"%";

1192      * Regenerate the group cache AKA the group2groupcache table in the database -
1193      * meant to be called when a group is added or removed from another group
1194      * 
1195      */
1196     private void rethinkGroupCache() throws SQLException
1197     {
1198         // read in the group2group table
1199         TableRowIterator tri = DatabaseManager.queryTable(myContext, "group2group",
1200                 "SELECT * FROM group2group");
1201 
1202         Map<Integer,Set<Integer>> parents = new HashMap<Integer,Set<Integer>>();
1203 
1204         try
1205         {
1206             while (tri.hasNext())
1207             {
1208                 TableRow row = (TableRow) tri.next();
1209 
1210                 Integer parentID = new Integer(row.getIntColumn("parent_id"));
1211                 Integer childID = new Integer(row.getIntColumn("child_id"));
1212 
1213                 // if parent doesn't have an entry, create one
1214                 if (!parents.containsKey(parentID))
1215                 {
1216                     Set<Integer> children = new HashSet<Integer>();
1217 
1218                     // add child id to the list
1219                     children.add(childID);
1220                     parents.put(parentID, children);
1221                 }
1222                 else
1223                 {
1224                     // parent has an entry, now add the child to the parent's record
1225                     // of children
1226                     Set<Integer> children =  parents.get(parentID);
1227                     children.add(childID);
1228                 }
1229             }
1230         }
1231         finally
1232         {
1233             // close the TableRowIterator to free up resources
1234             if (tri != null)
1235                 tri.close();
1236         }
1237 
1238         // now parents is a hash of all of the IDs of groups that are parents
1239         // and each hash entry is a hash of all of the IDs of children of those
1240         // parent groups
1241         // so now to establish all parent,child relationships we can iterate
1242         // through the parents hash
1243 
1244         Iterator i = parents.keySet().iterator();
1245 
1246         while (i.hasNext())
1247         {
1248             Integer parentID = (Integer) i.next();
1249 
1250             Set<Integer> myChildren = getChildren(parents, parentID);
1251 
1252             Iterator j = myChildren.iterator();
1253 
1254             while (j.hasNext())
1255             {
1256                 // child of a parent
1257                 Integer childID = (Integer) j.next();
1258 
1259                 ((Set<Integer>) parents.get(parentID)).add(childID);
1260             }
1261         }
1262 
1263         // empty out group2groupcache table
1264         DatabaseManager.updateQuery(myContext,
1265                 "DELETE FROM group2groupcache WHERE id >= 0");
1266

1268         Iterator pi = parents.keySet().iterator(); // parent iterator
1269 
1270         while (pi.hasNext())
1271         {
1272             Integer parent = (Integer) pi.next();
1273 
1274             Set<Integer> children =  parents.get(parent);
1275             Iterator ci = children.iterator(); // child iterator
1276 
1277             while (ci.hasNext())
1278             {
1279                 Integer child = (Integer) ci.next();
1280 
1281                 TableRow row = DatabaseManager.create(myContext,
1282                         "group2groupcache");
1283

1285                 int childID = child.intValue();
1286

1295     /**
1296      * Used recursively to generate a map of ALL of the children of the given
1297      * parent
1298      * 
1299      * @param parents
1300      *            Map of parent,child relationships
1301      * @param parent
1302      *            the parent you're interested in
1303      * @return Map whose keys are all of the children of a parent
1304      */
1305     private Set<Integer> getChildren(Map<Integer,Set<Integer>> parents, Integer parent)
1306     {
1307         Set<Integer> myChildren = new HashSet<Integer>();
1308 
1309         // degenerate case, this parent has no children
1310         if (!parents.containsKey(parent))
1311             return myChildren;
1312 
1313         // got this far, so we must have children
1314         Set<Integer> children =  parents.get(parent);
1315 
1316         // now iterate over all of the children
1317         Iterator i = children.iterator();
1318 
1319         while (i.hasNext())
1320         {
1321             Integer childID = (Integer) i.next();
1322 
1323             // add this child's ID to our return set
1324             myChildren.add(childID);
1325 
1326             // and now its children
1327             myChildren.addAll(getChildren(parents, childID));
1328         }
1329 
1330         return myChildren;
1331     }
