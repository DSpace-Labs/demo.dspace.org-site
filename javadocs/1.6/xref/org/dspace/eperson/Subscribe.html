89      /**
90       * Subscribe an e-person to a collection. An e-mail will be sent every day a
91       * new item appears in the collection.
92       * 
93       * @param context
94       *            DSpace context
95       * @param eperson
96       *            EPerson to subscribe
97       * @param collection
98       *            Collection to subscribe to
99       */
100     public static void subscribe(Context context, EPerson eperson,
101             Collection collection) throws SQLException, AuthorizeException
102     {
103         // Check authorisation. Must be administrator, or the eperson.
104         if (AuthorizeManager.isAdmin(context)
105                 || ((context.getCurrentUser() != null) && (context
106                         .getCurrentUser().getID() == eperson.getID())))
107         {
108             // already subscribed?          
109             TableRowIterator r = DatabaseManager.query(context,
110                     "SELECT * FROM subscription WHERE eperson_id= ? " +
111                     " AND collection_id= ? ", 
112                     eperson.getID(),collection.getID());
113 
114             try
115             {
116                 if (!r.hasNext())
117                 {
118                     // Not subscribed, so add them
119                     TableRow row = DatabaseManager.create(context, "subscription");
120                     row.setColumn("eperson_id", eperson.getID());
121                     row.setColumn("collection_id", collection.getID());
122                     DatabaseManager.update(context, row);
123 
124                     log.info(LogManager.getHeader(context, "subscribe",
125                             "eperson_id=" + eperson.getID() + ",collection_id="
126                                     + collection.getID()));
127                 }
128             }
129             finally
130             {
131                 // close the TableRowIterator to free up resources
132                 if (r != null)
133                     r.close();
134             }
135         }
136         else
137         {
138             throw new AuthorizeException(
139                     "Only admin or e-person themselves can subscribe");
140         }
141     }

143     /**
144      * Unsubscribe an e-person to a collection. Passing in <code>null</code>
145      * for the collection unsubscribes the e-person from all collections they
146      * are subscribed to.
147      * 
148      * @param context
149      *            DSpace context
150      * @param eperson
151      *            EPerson to unsubscribe
152      * @param collection
153      *            Collection to unsubscribe from
154      */
155     public static void unsubscribe(Context context, EPerson eperson,
156             Collection collection) throws SQLException, AuthorizeException
157     {
158         // Check authorisation. Must be administrator, or the eperson.
159         if (AuthorizeManager.isAdmin(context)
160                 || ((context.getCurrentUser() != null) && (context
161                         .getCurrentUser().getID() == eperson.getID())))
162         {
163             if (collection == null)
164             {
165                 // Unsubscribe from all
166                 DatabaseManager.updateQuery(context,
167                         "DELETE FROM subscription WHERE eperson_id= ? ",
168                         eperson.getID());
169             }
170             else
171             {       
172                 DatabaseManager.updateQuery(context,
173                         "DELETE FROM subscription WHERE eperson_id= ? " +
174                         "AND collection_id= ? ", 
175                         eperson.getID(),collection.getID());
176 
177                 log.info(LogManager.getHeader(context, "unsubscribe",
178                         "eperson_id=" + eperson.getID() + ",collection_id="
179                                 + collection.getID()));
180             }
181         }
182         else
183         {
184             throw new AuthorizeException(
185                     "Only admin or e-person themselves can unsubscribe");
186         }
187     }

196      * @return array of collections e-person is subscribed to

198     public static Collection[] getSubscriptions(Context context, EPerson eperson)
199             throws SQLException
200     {
201         TableRowIterator tri = DatabaseManager.query(context,
202                 "SELECT collection_id FROM subscription WHERE eperson_id= ? ",
203                 eperson.getID());
204 
205         List collections = new ArrayList();
206 
207         try
208         {
209             while (tri.hasNext())
210             {
211                 TableRow row = tri.next();
212 
213                 collections.add(Collection.find(context, row
214                         .getIntColumn("collection_id")));
215             }
216         }
217         finally
218         {
219             // close the TableRowIterator to free up resources
220             if (tri != null)
221                 tri.close();
222         }
223         
224         Collection[] collArray = new Collection[collections.size()];
225 
226         return (Collection[]) collections.toArray(collArray);
227     }

260     /**
261      * Process subscriptions. This must be invoked only once a day. Messages are
262      * only sent out when a collection has actually received new items, so that
263      * people's mailboxes are not clogged with many "no new items" mails.
264      * <P>
265      * Yesterday's newly available items are included. If this is run at for
266      * example midday, any items that have been made available during the
267      * current day will not be included, but will be included in the next day's
268      * run.
269      * <P>
270      * For example, if today's date is 2002-10-10 (in UTC) items made available
271      * during 2002-10-09 (UTC) will be included.
272      * 
273      * @param context
274      *            DSpace context object
275      * @param test 
276      */
277     public static void processDaily(Context context, boolean test) throws SQLException,
278             IOException
279     {
280         // Grab the subscriptions
281         TableRowIterator tri = DatabaseManager.query(context,
282                 "SELECT * FROM subscription ORDER BY eperson_id");
283 
284         EPerson currentEPerson = null;
285         List collections = null; // List of Collections
286 
287         try
288         {
289             // Go through the list collating subscriptions for each e-person
290             while (tri.hasNext())
291             {
292                 TableRow row = tri.next();
293 
294                 // Does this row relate to the same e-person as the last?
295                 if ((currentEPerson == null)
296                         || (row.getIntColumn("eperson_id") != currentEPerson
297                                 .getID()))
298                 {
299                     // New e-person. Send mail for previous e-person
300                     if (currentEPerson != null)
301                     {
302 
303                         try
304                         {
305                             sendEmail(context, currentEPerson, collections, test);
306                         }
307                         catch (MessagingException me)
308                         {
309                             log.error("Failed to send subscription to eperson_id="
310                                     + currentEPerson.getID());
311                             log.error(me);
312                         }
313                     }
314 
315                     currentEPerson = EPerson.find(context, row
316                             .getIntColumn("eperson_id"));
317                     collections = new ArrayList();
318                 }
319 
320                 collections.add(Collection.find(context, row
321                         .getIntColumn("collection_id")));
322             }
323         }
324         finally
325         {
326             // close the TableRowIterator to free up resources
327             if (tri != null)
328                 tri.close();
329         }
330         
331         // Process the last person
332         if (currentEPerson != null)
333         {
334             try
335             {
336                 sendEmail(context, currentEPerson, collections, test);
337             }
338             catch (MessagingException me)
339             {
340                 log.error("Failed to send subscription to eperson_id="
341                         + currentEPerson.getID());
342                 log.error(me);
343             }
344         }
345     }

347     /**
348      * Sends an email to the given e-person with details of new items in the
349      * given collections, items that appeared yesterday. No e-mail is sent if
350      * there aren't any new items in any of the collections.
351      * 
352      * @param context
353      *            DSpace context object
354      * @param eperson
355      *            eperson to send to
356      * @param collections
357      *            List of collection IDs (Integers)
358      * @param test 
359      */
360     public static void sendEmail(Context context, EPerson eperson,
361             List collections, boolean test) throws IOException, MessagingException,
362             SQLException
363     {
364         // Get a resource bundle according to the eperson language preferences
365         Locale supportedLocale = I18nUtil.getEPersonLocale(eperson); 
366         ResourceBundle labels =  ResourceBundle.getBundle("Messages", supportedLocale);
367         
368         // Get the start and end dates for yesterday
369         Date thisTimeYesterday = new Date(System.currentTimeMillis()
370                 - (24 * 60 * 60 * 1000));
371 
372         DCDate dcDateYesterday = new DCDate(thisTimeYesterday);
373 
374         // this time yesterday in ISO 8601, stripping the time
375         String isoDateYesterday = dcDateYesterday.toString().substring(0, 10);
376 
377         String startDate = isoDateYesterday;
378 
379         // FIXME: text of email should be more configurable from an
380         // i18n viewpoint
381         StringBuffer emailText = new StringBuffer();
382         boolean isFirst = true;
383 
384         for (int i = 0; i < collections.size(); i++)
385         {
386             Collection c = (Collection) collections.get(i);
387 
388             try {
389                 boolean includeAll = ConfigurationManager.getBooleanProperty("harvest.includerestricted.subscription", true);
390                 
391                 // we harvest all the changed item from yesterday until now
392                 List itemInfos = Harvest.harvest(context, c, startDate, null, 0, // Limit
393                                                                                     // and
394                                                                                     // offset
395                                                                                     // zero,
396                                                                                     // get
397                                                                                     // everything
398                         0, true, // Need item objects
399                         false, // But not containers
400                         false, // Or withdrawals
401                         includeAll);
402     
403                 if (ConfigurationManager.getBooleanProperty("eperson.subscription.onlynew", false))
404                 {
405                     // get only the items archived yesterday
406                     itemInfos = filterOutModified(itemInfos);
407                 }
408                 else
409                 {
410                     // strip out the item archived today or 
411                     // not archived yesterday and modified today
412                     itemInfos = filterOutToday(itemInfos);
413                 }
414 
415                 // Only add to buffer if there are new items
416                 if (itemInfos.size() > 0)
417                 {
418                     if (!isFirst)
419                     {
420                         emailText
421                                 .append("\n---------------------------------------\n");
422                     }
423                     else
424                     {
425                         isFirst = false;
426                     }
427     
428                     emailText.append(labels.getString("org.dspace.eperson.Subscribe.new-items")).append(" ").append(
429                             c.getMetadata("name")).append(": ").append(
430                             itemInfos.size()).append("\n\n");
431     
432                     for (int j = 0; j < itemInfos.size(); j++)
433                     {
434                         HarvestedItemInfo hii = (HarvestedItemInfo) itemInfos
435                                 .get(j);
436     
437                         DCValue[] titles = hii.item.getDC("title", null, Item.ANY);
438                         emailText.append("      ").append(labels.getString("org.dspace.eperson.Subscribe.title")).append(" ");
439     
440                         if (titles.length > 0)
441                         {
442                             emailText.append(titles[0].value);
443                         }
444                         else
445                         {
446                             emailText.append(labels.getString("org.dspace.eperson.Subscribe.untitled"));
447                         }
448     
449                         DCValue[] authors = hii.item.getDC("contributor", Item.ANY,
450                                 Item.ANY);
451     
452                         if (authors.length > 0)
453                         {
454                             emailText.append("\n    ").append(labels.getString("org.dspace.eperson.Subscribe.authors")).append(" ").append(
455                                     authors[0].value);
456     
457                             for (int k = 1; k < authors.length; k++)
458                             {
459                                 emailText.append("\n             ").append(
460                                         authors[k].value);
461                             }
462                         }
463     
464                         emailText.append("\n         ").append(labels.getString("org.dspace.eperson.Subscribe.id")).append(" ").append(
465                                 HandleManager.getCanonicalForm(hii.handle)).append(
466                                 "\n\n");
467                     }
468                 }
469             }
470             catch (ParseException pe)
471             {
472                 // This should never get thrown as the Dates are auto-generated
473             }
474         }
475 
476         // Send an e-mail if there were any new items
477         if (emailText.length() > 0)
478         {
479             
480             if(test)
481             {
482                 log.info(LogManager.getHeader(context, "subscription:", "eperson=" + eperson.getEmail() ));
483                 log.info(LogManager.getHeader(context, "subscription:", "text=" + emailText.toString() ));
484 
485             } else {
486                 
487                 Email email = ConfigurationManager.getEmail(I18nUtil.getEmailFilename(supportedLocale, "subscription"));
488                 email.addRecipient(eperson.getEmail());
489                 email.addArgument(emailText.toString());
490                 email.send();
491                 
492                 log.info(LogManager.getHeader(context, "sent_subscription", "eperson_id=" + eperson.getID() ));
493                 
494             }
495 
496             
497         }
498     }

501      * Method for invoking subscriptions via the command line
502      * 
503      * @param argv
504      *            command-line arguments, none used yet

506     public static void main(String[] argv) 
507     {
508         String usage = "org.dspace.eperson.Subscribe [-t] or nothing to send out subscriptions.";
509         
510         Options options = new Options();
511         HelpFormatter formatter = new HelpFormatter();
512         CommandLine line = null;
513         
514         {
515             Option opt = new Option("t", "test", false, "Run test session");
516             opt.setRequired(false);
517             options.addOption(opt);
518         }
519         
520         {
521             Option opt = new Option("h", "help", false, "Print this help message");
522             opt.setRequired(false);
523             options.addOption(opt);
524         }
525 
526         try
527         {
528             line = new PosixParser().parse(options, argv);
529         }
530         catch (Exception e)
531         {
532             // automatically generate the help statement
533             formatter.printHelp(usage, e.getMessage(), options, "");
534             System.exit(1);
535         }
536 
537         if (line.hasOption("h"))
538         {
539             // automatically generate the help statement
540             formatter.printHelp(usage, options);
541             System.exit(1);
542         }
543         
544         boolean test = line.hasOption("t");
545 
546         if(test)
547             log.setLevel(Level.DEBUG);
548         
549         Context context = null;
550 
551         try
552         {
553             context = new Context();
554             processDaily(context, test);
555             context.complete();
556         }
557         catch( Exception e )
558         {
559             log.fatal(e);
560         }
561         finally
562         {
563             if( context != null && context.isValid() )
564             {
565                 // Nothing is actually written
566                 context.abort();
567             }
568         }
569     }

571     private static List filterOutToday(List completeList)
572     {
573         log.debug("Filtering out all today item to leave new items list size="
574                 + completeList.size());
575         List filteredList = new ArrayList();
576 
577         SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
578         String today = sdf.format(new Date());
579         // Get the start and end dates for yesterday
580         Date thisTimeYesterday = new Date(System.currentTimeMillis()
581                 - (24 * 60 * 60 * 1000));
582         String yesterday = sdf.format(thisTimeYesterday);
583 
584         for (Iterator iter = completeList.iterator(); iter.hasNext();)
585         {
586             HarvestedItemInfo infoObject = (HarvestedItemInfo) iter.next();
587 
588             Date lastUpdate = infoObject.item.getLastModified();
589             String lastUpdateStr = sdf.format(lastUpdate);
590 
591             // has the item modified today?
592             if (lastUpdateStr.equals(today))
593             {
594                 DCValue[] dateAccArr = infoObject.item.getMetadata("dc",
595                         "date", "accessioned", Item.ANY);
596                 // we need only the item archived yesterday
597                 if (dateAccArr != null && dateAccArr.length > 0)
598                 {
599                     for (DCValue date : dateAccArr)
600                     {
601                         if (date != null && date.value != null)
602                         {
603                             // if it hasn't been archived today
604                             if (date.value.startsWith(yesterday))
605                             {
606                                 filteredList.add(infoObject);
607                                 log.debug("adding : " + dateAccArr[0].value
608                                         + " : " + today + " : "
609                                         + infoObject.handle);
610                                 break;
611                             }
612                             else
613                             {
614                                 log.debug("ignoring : " + dateAccArr[0].value
615                                         + " : " + today + " : "
616                                         + infoObject.handle);
617                             }
618                         }
619                     }
620                 }
621                 else
622                 {
623                     log.debug("no date accessioned, adding  : "
624                             + infoObject.handle);
625                     filteredList.add(infoObject);
626                 }
627             }
628             else
629             {
630                 // the item has been modified yesterday... 
631                 filteredList.add(infoObject);
632             }
633         }
634 
635         return filteredList;
636     }
