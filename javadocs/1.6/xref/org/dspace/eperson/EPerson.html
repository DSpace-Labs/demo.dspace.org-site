115     /**
116      * Get an EPerson from the database.
117      * 
118      * @param context
119      *            DSpace context object
120      * @param id
121      *            ID of the EPerson
122      * 
123      * @return the EPerson format, or null if the ID is invalid.
124      */
125     public static EPerson find(Context context, int id) throws SQLException
126     {
127         // First check the cache
128         EPerson fromCache = (EPerson) context.fromCache(EPerson.class, id);
129 
130         if (fromCache != null)
131         {
132             return fromCache;
133         }
134 
135         TableRow row = DatabaseManager.find(context, "eperson", id);
136 
137         if (row == null)
138         {
139             return null;
140         }
141         else
142         {
143             return new EPerson(context, row);
144         }
145     }

235     /**
236      * Find the epeople that match the search query across firstname, lastname or email. 
237      * This method also allows offsets and limits for pagination purposes. 
238      * 
239      * @param context
240      *            DSpace context
241      * @param query
242      *            The search string
243      * @param offset
244      *            Inclusive offset 
245      * @param limit
246      *            Maximum number of matches returned
247      * 
248      * @return array of EPerson objects
249      */
250     public static EPerson[] search(Context context, String query, int offset, int limit) 
251     		throws SQLException
252 	{
253 		String params = "%"+query.toLowerCase()+"%";
254         StringBuffer queryBuf = new StringBuffer();
255         queryBuf.append("SELECT * FROM eperson WHERE eperson_id = ? OR ");
256         queryBuf.append("LOWER(firstname) LIKE LOWER(?) OR LOWER(lastname) LIKE LOWER(?) OR LOWER(email) LIKE LOWER(?) ORDER BY lastname, firstname ASC ");
257 
258         // Add offset and limit restrictions - Oracle requires special code
259         if ("oracle".equals(ConfigurationManager.getProperty("db.name")))
260         {
261             // First prepare the query to generate row numbers
262             if (limit > 0 || offset > 0)
263             {
264                 queryBuf.insert(0, "SELECT /*+ FIRST_ROWS(n) */ rec.*, ROWNUM rnum  FROM (");
265                 queryBuf.append(") ");
266             }
267 
268             // Restrict the number of rows returned based on the limit
269             if (limit > 0)
270             {
271                 queryBuf.append("rec WHERE rownum<=? ");
272                 // If we also have an offset, then convert the limit into the maximum row number
273                 if (offset > 0)
274                     limit += offset;
275             }
276 
277             // Return only the records after the specified offset (row number)
278             if (offset > 0)
279             {
280                 queryBuf.insert(0, "SELECT * FROM (");
281                 queryBuf.append(") WHERE rnum>?");
282             }
283         }
284         else
285         {
286             if (limit > 0)
287                 queryBuf.append(" LIMIT ? ");
288 
289             if (offset > 0)
290                 queryBuf.append(" OFFSET ? ");
291         }
292 
293         String dbquery = queryBuf.toString();
294 
295         // When checking against the eperson-id, make sure the query can be made into a number
296 		Integer int_param;
297 		try {
298 			int_param = Integer.valueOf(query);
299 		}
300 		catch (NumberFormatException e) {
301 			int_param = new Integer(-1);
302 		}
303 
304         // Create the parameter array, including limit and offset if part of the query
305         Object[] paramArr = new Object[] {int_param,params,params,params};
306         if (limit > 0 && offset > 0)
307             paramArr = new Object[] {int_param,params,params,params,limit,offset};
308         else if (limit > 0)
309             paramArr = new Object[] {int_param,params,params,params,limit};
310         else if (offset > 0)
311             paramArr = new Object[] {int_param,params,params,params,offset};
312 
313         // Get all the epeople that match the query
314 		TableRowIterator rows = DatabaseManager.query(context, dbquery, paramArr);
315 		try
316         {
317             List epeopleRows = rows.toList();
318             EPerson[] epeople = new EPerson[epeopleRows.size()];
319 
320             for (int i = 0; i < epeopleRows.size(); i++)
321             {
322                 TableRow row = (TableRow) epeopleRows.get(i);
323 
324                 // First check the cache
325                 EPerson fromCache = (EPerson) context.fromCache(EPerson.class, row
326                         .getIntColumn("eperson_id"));
327 
328                 if (fromCache != null)
329                 {
330                     epeople[i] = fromCache;
331                 }
332                 else
333                 {
334                     epeople[i] = new EPerson(context, row);
335                 }
336             }
337 
338             return epeople;
339         }
340         finally
341         {
342             if (rows != null)
343                 rows.close();
344         }
345     }

347     /**
348      * Returns the total number of epeople returned by a specific query, without the overhead 
349      * of creating the EPerson objects to store the results.
350      * 
351      * @param context
352      *            DSpace context
353      * @param query
354      *            The search string
355      * 
356      * @return the number of epeople mathching the query
357      */
358     public static int searchResultCount(Context context, String query)
359     	throws SQLException
360 	{
361 		String dbquery = "%"+query.toLowerCase()+"%";
362 		Long count;
363 		
364 		// When checking against the eperson-id, make sure the query can be made into a number
365 		Integer int_param;
366 		try {
367 			int_param = Integer.valueOf(query);
368 		}
369 		catch (NumberFormatException e) {
370 			int_param = new Integer(-1);
371 		}
372 		
373 		// Get all the epeople that match the query
374 		TableRow row = DatabaseManager.querySingle(context,
375 		        "SELECT count(*) as epcount FROM eperson WHERE eperson_id = ? OR " +
376 		        "LOWER(firstname) LIKE LOWER(?) OR LOWER(lastname) LIKE LOWER(?) OR LOWER(email) LIKE LOWER(?)",
377 		        new Object[] {int_param,dbquery,dbquery,dbquery});
378 				
379 		// use getIntColumn for Oracle count data
380         if ("oracle".equals(ConfigurationManager.getProperty("db.name")))
381         {
382             count = new Long(row.getIntColumn("epcount"));
383         }
384         else  //getLongColumn works for postgres
385         {
386             count = new Long(row.getLongColumn("epcount"));
387         }
388         
389 		return count.intValue();
390 	}

394     /**
395      * Find all the epeople that match a particular query
396      * <ul>
397      * <li><code>ID</code></li>
398      * <li><code>LASTNAME</code></li>
399      * <li><code>EMAIL</code></li>
400      * <li><code>NETID</code></li>
401      * </ul>
402      * 
403      * @return array of EPerson objects
404      */
405     public static EPerson[] findAll(Context context, int sortField)
406             throws SQLException
407     {
408         String s;
409 
410         switch (sortField)
411         {
412         case ID:
413             s = "eperson_id";
414             break;
415 
416         case EMAIL:
417             s = "email";
418             break;
419 
420         case LANGUAGE:
421             s = "language";
422             break;
423         case NETID:
424             s = "netid";
425             break;
426 
427         default:
428             s = "lastname";
429         }
430 
431         // NOTE: The use of 's' in the order by clause can not cause an sql 
432         // injection because the string is derived from constant values above.
433         TableRowIterator rows = DatabaseManager.query(context,
434                 "SELECT * FROM eperson ORDER BY "+s);
435 
436         try
437         {
438             List epeopleRows = rows.toList();
439 
440             EPerson[] epeople = new EPerson[epeopleRows.size()];
441 
442             for (int i = 0; i < epeopleRows.size(); i++)
443             {
444                 TableRow row = (TableRow) epeopleRows.get(i);
445 
446                 // First check the cache
447                 EPerson fromCache = (EPerson) context.fromCache(EPerson.class, row
448                         .getIntColumn("eperson_id"));
449 
450                 if (fromCache != null)
451                 {
452                     epeople[i] = fromCache;
453                 }
454                 else
455                 {
456                     epeople[i] = new EPerson(context, row);
457                 }
458             }
459 
460             return epeople;
461         }
462         finally
463         {
464             if (rows != null)
465                 rows.close();
466         }
467     }

498     /**
499      * Delete an eperson
500      * 
501      */
502     public void delete() throws SQLException, AuthorizeException,
503             EPersonDeletionException
504     {
505         // authorized?
506         if (!AuthorizeManager.isAdmin(myContext))
507         {
508             throw new AuthorizeException(
509                     "You must be an admin to delete an EPerson");
510         }
511 
512         // check for presence of eperson in tables that
513         // have constraints on eperson_id
514         Vector constraintList = getDeleteConstraints();
515 
516         // if eperson exists in tables that have constraints
517         // on eperson, throw an exception
518         if (constraintList.size() > 0)
519         {
520             throw new EPersonDeletionException(constraintList);
521         }
522 
523         myContext.addEvent(new Event(Event.DELETE, Constants.EPERSON, getID(), getEmail()));
524 
525         // Remove from cache
526         myContext.removeCached(this, getID());
527 
528         // XXX FIXME: This sidesteps the object model code so it won't
529         // generate  REMOVE events on the affected Groups.
530 
531         // Remove any group memberships first
532         DatabaseManager.updateQuery(myContext,
533                 "DELETE FROM EPersonGroup2EPerson WHERE eperson_id= ? ",
534                 getID());
535 
536         // Remove any subscriptions
537         DatabaseManager.updateQuery(myContext,
538                 "DELETE FROM subscription WHERE eperson_id= ? ",
539                 getID());
540 
541         // Remove ourself
542         DatabaseManager.delete(myContext, myRow);
543 
544         log.info(LogManager.getHeader(myContext, "delete_eperson",
545                 "eperson_id=" + getID()));
546     }

646     {
647         String f = myRow.getStringColumn("firstname");
648         String l = myRow.getStringColumn("lastname");
649 
650         if ((l == null) && (f == null))
651         {
652             return getEmail();
653         }
654         else if (f == null)
655         {
656             return l;
657         }
658         else
659         {
660             return (f + " " + l);

837     /**
838      * Update the EPerson
839      */
840     public void update() throws SQLException, AuthorizeException
841     {
842         // Check authorisation - if you're not the eperson
843         // see if the authorization system says you can
844         if (!myContext.ignoreAuthorization()
845                 && ((myContext.getCurrentUser() == null) || (getID() != myContext
846                         .getCurrentUser().getID())))
847         {
848             AuthorizeManager.authorizeAction(myContext, this, Constants.WRITE);
849         }
850 
851         DatabaseManager.update(myContext, myRow);
852 
853         log.info(LogManager.getHeader(myContext, "update_eperson",
854                 "eperson_id=" + getID()));
855 
856         if (modified)
857         {
858             myContext.addEvent(new Event(Event.MODIFY, Constants.EPERSON, getID(), null));
859             modified = false;
860         }
