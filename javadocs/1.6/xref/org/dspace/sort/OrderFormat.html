51  /**
52   * Class implementing static helpers for anywhere that interacts with the sort columns
53   * (ie. ItemsByAuthor.sort_author, ItemsByTitle.sort_title)
54   *
55   * This class maps index 'types' to delegates that implement the sort string creation
56   * 
57   * Types can be defined or configured using the plugin manager:
58   * 
59   * plugin.named.org.dspace.sort.OrderFormatDelegate=
60   * 		org.dspace.sort.OrderFormatTitleMarc21=title
61   * 		org.dspace.sort.OrderFormatAuthor=author
62   * 
63   * The following standard types have been defined by default, but can be reconfigured
64   * via the plugin manager:
65   * 
66   * author	= org.dspace.sort.OrderFormatAuthor
67   * title	= org.dspace.sort.OrderFormatTitle
68   * text 	= org.dspace.sort.OrderFormatText
69   * 
70   * IMPORTANT - If you change any of the orderings, you need to rebuild the browse sort columns
71   * (ie. run 'index-all', or 'dsrun org.dspace.browse.InitializeBrowse')
72   * 
73   * @author Graham Triggs
74   * @version $Revision: 4497 $
75   */
76  public class OrderFormat
77  {
78  	private final static Logger log = LogManager.getLogger(OrderFormat.class);
79  
80  	public final static String AUTHOR = "author";
81  	public final static String TITLE  = "title";
82  	public final static String TEXT   = "text";
83  	public final static String DATE   = "date";
84          public final static String AUTHORITY = "authority";
85  	
86  	// Array of all available order delegates - avoids excessive calls to plugin manager
87  	private final static String[] delegates = PluginManager.getAllPluginNames(OrderFormatDelegate.class);
88  
89      private final static OrderFormatDelegate authorDelegate = new OrderFormatAuthor();
90      private final static OrderFormatDelegate titleDelegate  = new OrderFormatTitle();
91      private final static OrderFormatDelegate textDelegate   = new OrderFormatText();
92      private final static OrderFormatDelegate dateDelegate   = new OrderFormatDate();
93      private final static OrderFormatDelegate authorityDelegate = new OrderFormatText();
94      
95      /**
96       * Generate a sort string for the given DC metadata
97       */
98      public static String makeSortString(String value, String language, String type)
99      {
100     	OrderFormatDelegate delegate = null;
101     	
102         // If there is no value, return null
103         if (value == null)
104             return null;
105 
106     	// If a named index has been supplied
107     	if (type != null && type.length() > 0)
108     	{
109     		// Use a delegate if one is configured
110         	if ((delegate = OrderFormat.getDelegate(type)) != null)
111         	{
112         		return delegate.makeSortString(value, language);
113         	}
114 
115             // No delegates found, so apply defaults
116             if (type.equalsIgnoreCase(OrderFormat.AUTHOR) && authorDelegate != null)
117             {
118               return authorDelegate.makeSortString(value, language);
119             }
120 
121             if (type.equalsIgnoreCase(OrderFormat.TITLE) && titleDelegate != null)
122             {
123               return titleDelegate.makeSortString(value, language);
124             }
125 
126             if (type.equalsIgnoreCase(OrderFormat.TEXT) && textDelegate != null)
127             {
128               return textDelegate.makeSortString(value, language);
129             }
130             
131             if (type.equalsIgnoreCase(OrderFormat.DATE) && dateDelegate != null)
132             {
133               return dateDelegate.makeSortString(value, language);
134             }
135 
136             if (type.equalsIgnoreCase(OrderFormat.AUTHORITY) && authorityDelegate != null)
137             {
138               return authorityDelegate.makeSortString(value, language);
139             }
140     	}
141 
142     	return value;
143     }
144 
145     /**
146      * Retrieve the named delegate
147      */
148     private static OrderFormatDelegate getDelegate(String name)
149     {
150    		if (name != null && name.length() > 0)
151    		{
152    			// Check the cached array of names to see if the delegate has been configured
153 	   		for (int idx = 0; idx < delegates.length; idx++)
154 	   		{
155 	   			if (delegates[idx].equals(name))
156 	   			{
157 	   				return (OrderFormatDelegate)PluginManager.getNamedPlugin(OrderFormatDelegate.class, name);
158 	   			}
159 	   		}
160    		}
161    		
162     	return null;
163     }
