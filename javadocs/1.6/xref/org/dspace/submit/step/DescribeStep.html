121     /**
122      * Do any processing of the information input by the user, and/or perform
123      * step processing (if no user interaction required)
124      * <P>
125      * It is this method's job to save any data to the underlying database, as
126      * necessary, and return error messages (if any) which can then be processed
127      * by the appropriate user interface (JSP-UI or XML-UI)
128      * <P>
129      * NOTE: If this step is a non-interactive step (i.e. requires no UI), then
130      * it should perform *all* of its processing in this method!
131      *
132      * @param context
133      *            current DSpace context
134      * @param request
135      *            current servlet request object
136      * @param response
137      *            current servlet response object
138      * @param subInfo
139      *            submission info object
140      * @return Status or error flag which will be processed by
141      *         doPostProcessing() below! (if STATUS_COMPLETE or 0 is returned,
142      *         no errors occurred!)
143      */
144     public int doProcessing(Context context, HttpServletRequest request,
145             HttpServletResponse response, SubmissionInfo subInfo)
146             throws ServletException, IOException, SQLException,
147             AuthorizeException
148     {
149         // check what submit button was pressed in User Interface
150         String buttonPressed = Util.getSubmitButton(request, NEXT_BUTTON);
151 
152         // get the item and current page
153         Item item = subInfo.getSubmissionItem().getItem();
154         int currentPage = getCurrentPage(request);
155 
156         // lookup applicable inputs
157         Collection c = subInfo.getSubmissionItem().getCollection();
158         DCInput[] inputs = null;
159         try
160         {
161             inputs = inputsReader.getInputs(c.getHandle()).getPageRows(
162                     currentPage - 1,
163                     subInfo.getSubmissionItem().hasMultipleTitles(),
164                     subInfo.getSubmissionItem().isPublishedBefore());
165         }
166         catch (DCInputsReaderException e)
167         {
168             throw new ServletException(e);
169         }
170 
171         // Step 1:
172         // clear out all item metadata defined on this page
173         for (int i = 0; i < inputs.length; i++)
174         {
175             if (!inputs[i]
176                     .isVisible(subInfo.isInWorkflow() ? DCInput.WORKFLOW_SCOPE
177                             : DCInput.SUBMISSION_SCOPE))
178             {
179                 continue;
180             }
181             String qualifier = inputs[i].getQualifier();
182             if (qualifier == null
183                     && inputs[i].getInputType().equals("qualdrop_value"))
184             {
185                 qualifier = Item.ANY;
186             }
187             item.clearMetadata(inputs[i].getSchema(), inputs[i].getElement(),
188                     qualifier, Item.ANY);
189         }
190 
191         // Clear required-field errors first since missing authority
192         // values can add them too.
193         clearErrorFields(request);
194 
195         // Step 2:
196         // now update the item metadata.
197         String fieldName;
198         boolean moreInput = false;
199         for (int j = 0; j < inputs.length; j++)
200         {
201             if (!inputs[j]
202                         .isVisible(subInfo.isInWorkflow() ? DCInput.WORKFLOW_SCOPE
203                                 : DCInput.SUBMISSION_SCOPE))
204             {
205                 continue;
206             }
207             String element = inputs[j].getElement();
208             String qualifier = inputs[j].getQualifier();
209             String schema = inputs[j].getSchema();
210             if (qualifier != null && !qualifier.equals(Item.ANY))
211             {
212                 fieldName = schema + "_" + element + '_' + qualifier;
213             }
214             else
215             {
216                 fieldName = schema + "_" + element;
217             }
218 
219             String fieldKey = MetadataAuthorityManager.makeFieldKey(schema, element, qualifier);
220             ChoiceAuthorityManager cmgr = ChoiceAuthorityManager.getManager();
221             String inputType = inputs[j].getInputType();
222             if (inputType.equals("name"))
223             {
224                 readNames(request, item, schema, element, qualifier, inputs[j]
225                         .getRepeatable());
226             }
227             else if (inputType.equals("date"))
228             {
229                 readDate(request, item, schema, element, qualifier);
230             }
231             // choice-controlled input with "select" presentation type is
232             // always rendered as a dropdown menu
233             else if (inputType.equals("dropdown") || inputType.equals("list") ||
234                      (cmgr.isChoicesConfigured(fieldKey) &&
235                       "select".equals(cmgr.getPresentation(fieldKey))))
236             {
237                 String[] vals = request.getParameterValues(fieldName);
238                 if (vals != null)
239                 {
240                     for (int z = 0; z < vals.length; z++)
241                     {
242                         if (!vals[z].equals(""))
243                         {
244                             item.addMetadata(schema, element, qualifier, LANGUAGE_QUALIFIER,
245                                     vals[z]);
246                         }
247                     }
248                 }
249             }
250             else if (inputType.equals("series"))
251             {
252                 readSeriesNumbers(request, item, schema, element, qualifier,
253                         inputs[j].getRepeatable());
254             }
255             else if (inputType.equals("qualdrop_value"))
256             {
257                 List quals = getRepeatedParameter(request, schema + "_"
258                         + element, schema + "_" + element + "_qualifier");
259                 List vals = getRepeatedParameter(request, schema + "_"
260                         + element, schema + "_" + element + "_value");
261                 for (int z = 0; z < vals.size(); z++)
262                 {
263                     String thisQual = (String) quals.get(z);
264                     if ("".equals(thisQual))
265                     {
266                         thisQual = null;
267                     }
268                     String thisVal = (String) vals.get(z);
269                     if (!buttonPressed.equals("submit_" + schema + "_"
270                             + element + "_remove_" + z)
271                             && !thisVal.equals(""))
272                     {
273                         item.addMetadata(schema, element, thisQual, null,
274                                 thisVal);
275                     }
276                 }
277             }
278             else if ((inputType.equals("onebox"))
279                     || (inputType.equals("twobox"))
280                     || (inputType.equals("textarea")))
281             {
282                 readText(request, item, schema, element, qualifier, inputs[j]
283                         .getRepeatable(), LANGUAGE_QUALIFIER);
284             }
285             else
286             {
287                 throw new ServletException("Field " + fieldName
288                         + " has an unknown input type: " + inputType);
289             }
290 
291             // determine if more input fields were requested
292             if (!moreInput
293                     && buttonPressed.equals("submit_" + fieldName + "_add"))
294             {
295                 subInfo.setMoreBoxesFor(fieldName);
296                 subInfo.setJumpToField(fieldName);
297                 moreInput = true;
298             }
299             // was XMLUI's "remove" button pushed?
300             else if (buttonPressed.equals("submit_" + fieldName + "_delete"))
301             {
302                 subInfo.setJumpToField(fieldName);
303             }
304         }
305 
306         // Step 3:
307         // Check to see if any fields are missing
308         // Only check for required fields if user clicked the "next", the "previous" or the "progress bar" button
309         if (buttonPressed.equals(NEXT_BUTTON)
310                 || buttonPressed.startsWith(PROGRESS_BAR_PREFIX)
311                 || buttonPressed.equals(PREVIOUS_BUTTON)
312                 || buttonPressed.equals(CANCEL_BUTTON))
313         {
314             for (int i = 0; i < inputs.length; i++)
315             {
316                 DCValue[] values = item.getMetadata(inputs[i].getSchema(),
317                         inputs[i].getElement(), inputs[i].getQualifier(), Item.ANY);
318 
319                 if (inputs[i].isRequired() && values.length == 0)
320                 {
321                     // since this field is missing add to list of error fields
322                     addErrorField(request, getFieldName(inputs[i]));
323                 }
324             }
325         }
326 
327         // Step 4:
328         // Save changes to database
329         subInfo.getSubmissionItem().update();
330 
331         // commit changes
332         context.commit();
333 
334         // check for request for more input fields, first
335         if (moreInput)
336         {
337             return STATUS_MORE_INPUT_REQUESTED;
338         }
339         // if one or more fields errored out, return
340         else if (getErrorFields(request) != null && getErrorFields(request).size() > 0)
341         {
342             return STATUS_MISSING_REQUIRED_FIELDS;
343         }
344 
345         // completed without errors
346         return STATUS_COMPLETE;
347     }

351     /**
352      * Retrieves the number of pages that this "step" extends over. This method
353      * is used to build the progress bar.
354      * <P>
355      * This method may just return 1 for most steps (since most steps consist of
356      * a single page). But, it should return a number greater than 1 for any
357      * "step" which spans across a number of HTML pages. For example, the
358      * configurable "Describe" step (configured using input-forms.xml) overrides
359      * this method to return the number of pages that are defined by its
360      * configuration file.
361      * <P>
362      * Steps which are non-interactive (i.e. they do not display an interface to
363      * the user) should return a value of 1, so that they are only processed
364      * once!
365      *
366      * @param request
367      *            The HTTP Request
368      * @param subInfo
369      *            The current submission information object
370      *
371      * @return the number of pages in this step
372      */
373     public int getNumberOfPages(HttpServletRequest request,
374             SubmissionInfo subInfo) throws ServletException
375     {
376         // by default, use the "default" collection handle
377         String collectionHandle = DCInputsReader.DEFAULT_COLLECTION;
378 
379         if (subInfo.getSubmissionItem() != null)
380         {
381             collectionHandle = subInfo.getSubmissionItem().getCollection()
382                     .getHandle();
383         }
384 
385         // get number of input pages (i.e. "Describe" pages)
386         try
387         {
388             return getInputsReader().getNumberInputPages(collectionHandle);
389         }
390         catch (DCInputsReaderException e)
391         {
392             throw new ServletException(e);
393         }
394     }

396     /**
397      *
398      * @return the current DCInputsReader
399      */
400     public static DCInputsReader getInputsReader() throws ServletException
401     {
402         // load inputsReader only the first time
403         if (inputsReader == null)
404         {
405             // read configurable submissions forms data
406             try
407             {
408                 inputsReader = new DCInputsReader();
409             }
410             catch (DCInputsReaderException e)
411             {
412                 throw new ServletException(e);
413             }
414         }
415         
416         return inputsReader;
417     }

458     /**
459      * Set relevant metadata fields in an item from name values in the form.
460      * Some fields are repeatable in the form. If this is the case, and the
461      * field is "dc.contributor.author", the names in the request will be from
462      * the fields as follows:
463      *
464      * dc_contributor_author_last -> last name of first author
465      * dc_contributor_author_first -> first name(s) of first author
466      * dc_contributor_author_last_1 -> last name of second author
467      * dc_contributor_author_first_1 -> first name(s) of second author
468      *
469      * and so on. If the field is unqualified:
470      *
471      * dc_contributor_last -> last name of first contributor
472      * dc_contributor_first -> first name(s) of first contributor
473      *
474      * If the parameter "submit_dc_contributor_author_remove_n" is set, that
475      * value is removed.
476      *
477      * Otherwise the parameters are of the form:
478      *
479      * dc_contributor_author_last dc_contributor_author_first
480      *
481      * The values will be put in separate DCValues, in the form "last name,
482      * first name(s)", ordered as they appear in the list. These will replace
483      * any existing values.
484      *
485      * @param request
486      *            the request object
487      * @param item
488      *            the item to update
489      * @param schema
490      *            the metadata schema
491      * @param element
492      *            the metadata element
493      * @param qualifier
494      *            the metadata qualifier, or null if unqualified
495      * @param repeated
496      *            set to true if the field is repeatable on the form
497      */
498     protected void readNames(HttpServletRequest request, Item item,
499             String schema, String element, String qualifier, boolean repeated)
500     {
501         String metadataField = MetadataField
502                 .formKey(schema, element, qualifier);
503 
504         String fieldKey = MetadataAuthorityManager.makeFieldKey(schema, element, qualifier);
505         boolean isAuthorityControlled = MetadataAuthorityManager.getManager().isAuthorityControlled(fieldKey);
506 
507         // Names to add
508         List firsts = new LinkedList();
509         List lasts = new LinkedList();
510         List auths = new LinkedList();
511         List confs = new LinkedList();
512 
513         if (repeated)
514         {
515             firsts = getRepeatedParameter(request, metadataField, metadataField
516                     + "_first");
517             lasts = getRepeatedParameter(request, metadataField, metadataField
518                     + "_last");
519             auths = getRepeatedParameter(request, metadataField, metadataField
520                     + "_authority");
521             confs = getRepeatedParameter(request, metadataField, metadataField
522                     + "_confidence");
523 
524             // Find out if the relevant "remove" button was pressed
525             // TODO: These separate remove buttons are only relevant
526             // for DSpace JSP UI, and the code below can be removed
527             // once the DSpace JSP UI is obsolete!
528             String buttonPressed = Util.getSubmitButton(request, "");
529             String removeButton = "submit_" + metadataField + "_remove_";
530 
531             if (buttonPressed.startsWith(removeButton))
532             {
533                 int valToRemove = Integer.parseInt(buttonPressed
534                         .substring(removeButton.length()));
535 
536                 firsts.remove(valToRemove);
537                 lasts.remove(valToRemove);
538                 auths.remove(valToRemove);
539                 confs.remove(valToRemove);
540             }
541         }
542         else
543         {
544             // Just a single name
545             String lastName = request.getParameter(metadataField + "_last");
546             String firstNames = request.getParameter(metadataField + "_first");
547             String authority = request.getParameter(metadataField + "_authority");
548             String confidence = request.getParameter(metadataField + "_confidence");
549 
550             if (lastName != null)
551                 lasts.add(lastName);
552             if (firstNames != null)
553                 firsts.add(firstNames);
554             auths.add(authority == null ? "" : authority);
555             confs.add(confidence == null ? "" : confidence);
556         }
557 
558         // Remove existing values, already done in doProcessing see also bug DS-203
559         // item.clearMetadata(schema, element, qualifier, Item.ANY);
560 
561         // Put the names in the correct form
562         for (int i = 0; i < lasts.size(); i++)
563         {
564             String f = (String) firsts.get(i);
565             String l = (String) lasts.get(i);
566 
567             // only add if lastname is non-empty
568             if ((l != null) && !((l.trim()).equals("")))
569             {
570                 // Ensure first name non-null
571                 if (f == null)
572                 {
573                     f = "";
574                 }
575 
576                 // If there is a comma in the last name, we take everything
577                 // after that comma, and add it to the right of the
578                 // first name
579                 int comma = l.indexOf(',');
580 
581                 if (comma >= 0)
582                 {
583                     f = f + l.substring(comma + 1);
584                     l = l.substring(0, comma);
585 
586                     // Remove leading whitespace from first name
587                     while (f.startsWith(" "))
588                     {
589                         f = f.substring(1);
590                     }
591                 }
592 
593                 // Add to the database -- unless required authority is missing
594                 if (isAuthorityControlled)
595                 {
596                     String authKey = auths.size() > i ? (String)auths.get(i) : null;
597                     String sconf = (authKey != null && confs.size() > i) ? (String)confs.get(i) : null;
598                     if (MetadataAuthorityManager.getManager().isAuthorityRequired(fieldKey) &&
599                         (authKey == null || authKey.length() == 0))
600                     {
601                         log.warn("Skipping value of "+metadataField+" because the required Authority key is missing or empty.");
602                         addErrorField(request, metadataField);
603                     }
604                     else
605                         item.addMetadata(schema, element, qualifier, null,
606                                 new DCPersonName(l, f).toString(), authKey,
607                                 (sconf != null && sconf.length() > 0) ?
608                                     Choices.getConfidenceValue(sconf) : Choices.CF_ACCEPTED);
609                 }
610                 else
611                     item.addMetadata(schema, element, qualifier, null,
612                         new DCPersonName(l, f).toString());
613             }
614         }
615     }

617     /**
618      * Fill out an item's metadata values from a plain standard text field. If
619      * the field isn't repeatable, the input field name is called:
620      *
621      * element_qualifier
622      *
623      * or for an unqualified element:
624      *
625      * element
626      *
627      * Repeated elements are appended with an underscore then an integer. e.g.:
628      *
629      * dc_title_alternative dc_title_alternative_1
630      *
631      * The values will be put in separate DCValues, ordered as they appear in
632      * the list. These will replace any existing values.
633      *
634      * @param request
635      *            the request object
636      * @param item
637      *            the item to update
638      * @param schema
639      *            the short schema name
640      * @param element
641      *            the metadata element
642      * @param qualifier
643      *            the metadata qualifier, or null if unqualified
644      * @param repeated
645      *            set to true if the field is repeatable on the form
646      * @param lang
647      *            language to set (ISO code)
648      */
649     protected void readText(HttpServletRequest request, Item item, String schema,
650             String element, String qualifier, boolean repeated, String lang)
651     {
652         // FIXME: Of course, language should be part of form, or determined
653         // some other way
654         String metadataField = MetadataField
655                 .formKey(schema, element, qualifier);
656 
657         String fieldKey = MetadataAuthorityManager.makeFieldKey(schema, element, qualifier);
658         boolean isAuthorityControlled = MetadataAuthorityManager.getManager().isAuthorityControlled(fieldKey);
659 
660         // Values to add
661         List vals = null;
662         List auths = null;
663         List confs = null;
664 
665         if (repeated)
666         {
667             vals = getRepeatedParameter(request, metadataField, metadataField);
668             if (isAuthorityControlled)
669             {
670                 auths = getRepeatedParameter(request, metadataField, metadataField+"_authority");
671                 confs = getRepeatedParameter(request, metadataField, metadataField+"_confidence");
672             }
673 
674             // Find out if the relevant "remove" button was pressed
675             // TODO: These separate remove buttons are only relevant
676             // for DSpace JSP UI, and the code below can be removed
677             // once the DSpace JSP UI is obsolete!
678             String buttonPressed = Util.getSubmitButton(request, "");
679             String removeButton = "submit_" + metadataField + "_remove_";
680 
681             if (buttonPressed.startsWith(removeButton))
682             {
683                 int valToRemove = Integer.parseInt(buttonPressed
684                         .substring(removeButton.length()));
685 
686                 vals.remove(valToRemove);
687             }
688         }
689         else
690         {
691             // Just a single name
692             vals = new LinkedList();
693             String value = request.getParameter(metadataField);
694             if (value != null)
695                 vals.add(value.trim());
696             if (isAuthorityControlled)
697             {
698                 auths = new LinkedList();
699                 confs = new LinkedList();
700                 String av = request.getParameter(metadataField+"_authority");
701                 String cv = request.getParameter(metadataField+"_confidence");
702                 auths.add(av == null ? "":av.trim());
703                 confs.add(cv == null ? "":cv.trim());
704         }
705         }
706 
707         // Remove existing values, already done in doProcessing see also bug DS-203
708         // item.clearMetadata(schema, element, qualifier, Item.ANY);
709 
710         // Put the names in the correct form
711         for (int i = 0; i < vals.size(); i++)

885     /**
886      * Get repeated values from a form. If "foo" is passed in as the parameter,
887      * values in the form of parameters "foo", "foo_1", "foo_2", etc. are
888      * returned.
889      * <P>
890      * This method can also handle "composite fields" (metadata fields which may
891      * require multiple params, etc. a first name and last name).
892      *
893      * @param request
894      *            the HTTP request containing the form information
895      * @param metadataField
896      *            the metadata field which can store repeated values
897      * @param param
898      *            the repeated parameter on the page (used to fill out the
899      *            metadataField)
900      *
901      * @return a List of Strings
902      */
903     protected List getRepeatedParameter(HttpServletRequest request,
904             String metadataField, String param)
905     {
906         List vals = new LinkedList();
907 
908         int i = 1;    //start index at the first of the previously entered values
909         boolean foundLast = false;
910 
911         // Iterate through the values in the form.
912         while (!foundLast)
913         {
914             String s = null;
915 
916             //First, add the previously entered values.
917             // This ensures we preserve the order that these values were entered
918             s = request.getParameter(param + "_" + i);
919 
920             // If there are no more previously entered values,
921             // see if there's a new value entered in textbox
922             if (s==null)
923             {
924                 s = request.getParameter(param);
925                 //this will be the last value added
926                 foundLast = true;
927             }
928 
929             // We're only going to add non-null values
930             if (s != null)
931             {
932                 boolean addValue = true;
933 
934                 // Check to make sure that this value was not selected to be
935                 // removed.
936                 // (This is for the "remove multiple" option available in
937                 // Manakin)
938                 String[] selected = request.getParameterValues(metadataField
939                         + "_selected");
940 
941                 if (selected != null)
942                 {
943                     for (int j = 0; j < selected.length; j++)
944                     {
945                         if (selected[j].equals(metadataField + "_" + i))
946                         {
947                             addValue = false;
948                         }
949                     }
950                 }
951 
952                 if (addValue)
953                   vals.add(s.trim());
954             }
955 
956             i++;
957         }
958 
959         log.debug("getRepeatedParameter: metadataField=" + metadataField
960                 + " param=" + metadataField + ", return count = "+vals.size());
961 
962         return vals;
963     }
