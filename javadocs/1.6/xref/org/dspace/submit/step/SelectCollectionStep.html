57  /**
58   * SelectCollection Step which processes the collection that the user selected
59   * in that step of the DSpace Submission process
60   * <P>
61   * This class performs all the behind-the-scenes processing that
62   * this particular step requires.  This class's methods are utilized 
63   * by both the JSP-UI and the Manakin XML-UI
64   * 
65   * @see org.dspace.app.util.SubmissionConfig
66   * @see org.dspace.app.util.SubmissionStepConfig
67   * @see org.dspace.submit.AbstractProcessingStep
68   * 
69   * @author Tim Donohue
70   * @version $Revision: 3738 $
71   */
72  public class SelectCollectionStep extends AbstractProcessingStep
73  {
74      /***************************************************************************
75       * STATUS / ERROR FLAGS (returned by doProcessing() if an error occurs or
76       * additional user interaction may be required)
77       * 
78       * (Do NOT use status of 0, since it corresponds to STATUS_COMPLETE flag
79       * defined in the JSPStepManager class)
80       **************************************************************************/
81      // no collection was selected
82      public static final int STATUS_NO_COLLECTION = 1;
83  
84      // invalid collection or error finding collection
85      public static final int STATUS_INVALID_COLLECTION = 2;
86  
87      /** log4j logger */
88      private static Logger log = Logger.getLogger(SelectCollectionStep.class);
89  
90      /**
91       * Do any processing of the information input by the user, and/or perform
92       * step processing (if no user interaction required)
93       * <P>
94       * It is this method's job to save any data to the underlying database, as
95       * necessary, and return error messages (if any) which can then be processed
96       * by the appropriate user interface (JSP-UI or XML-UI)
97       * <P>
98       * NOTE: If this step is a non-interactive step (i.e. requires no UI), then
99       * it should perform *all* of its processing in this method!
100      * 
101      * @param context
102      *            current DSpace context
103      * @param request
104      *            current servlet request object
105      * @param response
106      *            current servlet response object
107      * @param subInfo
108      *            submission info object
109      * @return Status or error flag which will be processed by
110      *         doPostProcessing() below! (if STATUS_COMPLETE or 0 is returned,
111      *         no errors occurred!)
112      */
113     public int doProcessing(Context context, HttpServletRequest request,
114             HttpServletResponse response, SubmissionInfo subInfo)
115             throws ServletException, IOException, SQLException,
116             AuthorizeException
117     {
118         // First we find the collection which was selected
119         int id = Util.getIntParameter(request, "collection");
120 
121         // if the user didn't select a collection,
122         // send him/her back to "select a collection" page
123         if (id < 0)
124         {
125             return STATUS_NO_COLLECTION;
126         }
127 
128         // try to load the collection
129         Collection col = Collection.find(context, id);
130 
131         // Show an error if the collection is invalid
132         if (col == null)
133         {
134             return STATUS_INVALID_COLLECTION;
135         }
136         else
137         {
138             // create our new Workspace Item
139             WorkspaceItem wi = WorkspaceItem.create(context, col, true);
140 
141             // update Submission Information with this Workspace Item
142             subInfo.setSubmissionItem(wi);
143 
144             // commit changes to database
145             context.commit();
146 
147             // need to reload current submission process config,
148             // since it is based on the Collection selected
149             subInfo.reloadSubmissionConfig(request);
150         }
151 
152         // no errors occurred
153         return STATUS_COMPLETE;
154     }

156     /**
157      * Retrieves the number of pages that this "step" extends over. This method
158      * is used to build the progress bar.
159      * <P>
160      * This method may just return 1 for most steps (since most steps consist of
161      * a single page). But, it should return a number greater than 1 for any
162      * "step" which spans across a number of HTML pages. For example, the
163      * configurable "Describe" step (configured using input-forms.xml) overrides
164      * this method to return the number of pages that are defined by its
165      * configuration file.
166      * <P>
167      * Steps which are non-interactive (i.e. they do not display an interface to
168      * the user) should return a value of 1, so that they are only processed
169      * once!
170      * 
171      * @param request
172      *            The HTTP Request
173      * @param subInfo
174      *            The current submission information object
175      * 
176      * @return the number of pages in this step
177      */
178     public int getNumberOfPages(HttpServletRequest request,
179             SubmissionInfo subInfo) throws ServletException
180     {
181         // there is always just one page in the "select a collection" step!
182         return 1;
183     }
