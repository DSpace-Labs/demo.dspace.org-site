123     /**
124      * Do any processing of the information input by the user, and/or perform
125      * step processing (if no user interaction required)
126      * <P>
127      * It is this method's job to save any data to the underlying database, as
128      * necessary, and return error messages (if any) which can then be processed
129      * by the appropriate user interface (JSP-UI or XML-UI)
130      * <P>
131      * NOTE: If this step is a non-interactive step (i.e. requires no UI), then
132      * it should perform *all* of its processing in this method!
133      * 
134      * @param context
135      *            current DSpace context
136      * @param request
137      *            current servlet request object
138      * @param response
139      *            current servlet response object
140      * @param subInfo
141      *            submission info object
142      * @return Status or error flag which will be processed by
143      *         doPostProcessing() below! (if STATUS_COMPLETE or 0 is returned,
144      *         no errors occurred!)
145      */
146     public int doProcessing(Context context, HttpServletRequest request,
147             HttpServletResponse response, SubmissionInfo subInfo)
148             throws ServletException, IOException, SQLException,
149             AuthorizeException
150     {
151         // get button user pressed
152         String buttonPressed = Util.getSubmitButton(request, NEXT_BUTTON);
153 
154         // get reference to item
155         Item item = subInfo.getSubmissionItem().getItem();
156 
157         // -----------------------------------
158         // Step #0: Upload new files (if any)
159         // -----------------------------------
160         String contentType = request.getContentType();
161 
162         // if multipart form, then we are uploading a file
163         if ((contentType != null)
164                 && (contentType.indexOf("multipart/form-data") != -1))
165         {
166             // This is a multipart request, so it's a file upload
167             // (return any status messages or errors reported)
168             int status = processUploadFile(context, request, response, subInfo);
169 
170             // if error occurred, return immediately
171             if (status != STATUS_COMPLETE)
172                 return status;
173         }
174             
175         // if user pressed jump-to button in process bar,
176         // return success (so that jump will occur)
177         if (buttonPressed.startsWith(PROGRESS_BAR_PREFIX))
178         {
179             // check if a file is required to be uploaded
180             if (fileRequired && !item.hasUploadedFiles())
181             {
182                 return STATUS_NO_FILES_ERROR;
183             }
184             else
185             {
186                 return STATUS_COMPLETE;
187             }
188         }
189 
190         // ---------------------------------------------
191         // Step #1: Check if this was just a request to
192         // edit file information.
193         // (or canceled editing information)
194         // ---------------------------------------------
195         // check if we're already editing a specific bitstream
196         if (request.getParameter("bitstream_id") != null)
197         {
198             if (buttonPressed.equals(CANCEL_EDIT_BUTTON))
199             {
200                 // canceled an edit bitstream request
201                 subInfo.setBitstream(null);
202 
203                 // this flag will just return us to the normal upload screen
204                 return STATUS_EDIT_COMPLETE;
205             }
206             else
207             {
208                 // load info for bitstream we are editing
209                 Bitstream b = Bitstream.find(context, Integer.parseInt(request
210                         .getParameter("bitstream_id")));
211 
212                 // save bitstream to submission info
213                 subInfo.setBitstream(b);
214             }
215         }
216         else if (buttonPressed.startsWith("submit_edit_"))
217         {
218             // get ID of bitstream that was requested for editing
219             String bitstreamID = buttonPressed.substring("submit_edit_"
220                     .length());
221 
222             Bitstream b = Bitstream
223                     .find(context, Integer.parseInt(bitstreamID));
224 
225             // save bitstream to submission info
226             subInfo.setBitstream(b);
227 
228             // return appropriate status flag to say we are now editing the
229             // bitstream
230             return STATUS_EDIT_BITSTREAM;
231         }
232 
233         // ---------------------------------------------
234         // Step #2: Process any remove file request(s)
235         // ---------------------------------------------
236         // Remove-selected requests come from Manakin
237         if (buttonPressed.equalsIgnoreCase("submit_remove_selected"))
238         {
239             // this is a remove multiple request!
240 
241             if (request.getParameter("remove") != null)
242             {
243                 // get all files to be removed
244                 String[] removeIDs = request.getParameterValues("remove");
245 
246                 // remove each file in the list
247                 for (int i = 0; i < removeIDs.length; i++)
248                 {
249                     int id = Integer.parseInt(removeIDs[i]);
250 
251                     int status = processRemoveFile(context, item, id);
252 
253                     // if error occurred, return immediately
254                     if (status != STATUS_COMPLETE)
255                         return status;
256                 }
257 
258                 // remove current bitstream from Submission Info
259                 subInfo.setBitstream(null);
260             }
261         }
262         else if (buttonPressed.startsWith("submit_remove_"))
263         {
264             // A single file "remove" button must have been pressed
265 
266             int id = Integer.parseInt(buttonPressed.substring(14));
267             int status = processRemoveFile(context, item, id);
268 
269             // if error occurred, return immediately
270             if (status != STATUS_COMPLETE)
271                 return status;
272 
273             // remove current bitstream from Submission Info
274             subInfo.setBitstream(null);
275         }
276 
277         // -------------------------------------------------
278         // Step #3: Check for a change in file description
279         // -------------------------------------------------
280         String fileDescription = request.getParameter("description");
281 
282         if (fileDescription != null && fileDescription.length() > 0)
283         {
284             // save this file description
285             int status = processSaveFileDescription(context, request, response,
286                     subInfo);
287 
288             // if error occurred, return immediately
289             if (status != STATUS_COMPLETE)
290                 return status;
291         }
292 
293         // ------------------------------------------
294         // Step #4: Check for a file format change
295         // (if user had to manually specify format)
296         // ------------------------------------------
297         int formatTypeID = Util.getIntParameter(request, "format");
298         String formatDesc = request.getParameter("format_description");
299 
300         // if a format id or description was found, then save this format!
301         if (formatTypeID >= 0
302                 || (formatDesc != null && formatDesc.length() > 0))
303         {
304             // save this specified format
305             int status = processSaveFileFormat(context, request, response,
306                     subInfo);
307 
308             // if error occurred, return immediately
309             if (status != STATUS_COMPLETE)
310                 return status;
311         }
312 
313         // ---------------------------------------------------
314         // Step #5: Check if primary bitstream has changed
315         // -------------------------------------------------
316         if (request.getParameter("primary_bitstream_id") != null)
317         {
318             Bundle[] bundles = item.getBundles("ORIGINAL");
319             bundles[0].setPrimaryBitstreamID(new Integer(request
320                     .getParameter("primary_bitstream_id")).intValue());
321             bundles[0].update();
322         }
323 
324         // ---------------------------------------------------
325         // Step #6: Determine if there is an error because no
326         // files have been uploaded.
327         // ---------------------------------------------------
328         //check if a file is required to be uploaded
329         if (fileRequired && !item.hasUploadedFiles())
330         {
331             return STATUS_NO_FILES_ERROR;
332         }
333 
334         // commit all changes to database
335         context.commit();
336 
337         return STATUS_COMPLETE;
338     }

367         return 1;

376     /**
377      * Remove a file from an item
378      * 
379      * @param context
380      *            current DSpace context
381      * @param item
382      *            Item where file should be removed from
383      * @param bitstreamID
384      *            The id of bitstream representing the file to remove
385      * @return Status or error flag which will be processed by
386      *         UI-related code! (if STATUS_COMPLETE or 0 is returned,
387      *         no errors occurred!)
388      */
389     protected int processRemoveFile(Context context, Item item, int bitstreamID)
390             throws IOException, SQLException, AuthorizeException
391     {
392         Bitstream bitstream;
393 
394         // Try to find bitstream
395         try
396         {
397             bitstream = Bitstream.find(context, bitstreamID);
398         }
399         catch (NumberFormatException nfe)
400         {
401             bitstream = null;
402         }
403 
404         if (bitstream == null)
405         {
406             // Invalid or mangled bitstream ID
407             // throw an error and return immediately
408             return STATUS_INTEGRITY_ERROR;
409         }
410 
411         // remove bitstream from bundle..
412         // delete bundle if it's now empty
413         Bundle[] bundles = bitstream.getBundles();
414 
415         bundles[0].removeBitstream(bitstream);
416 
417         Bitstream[] bitstreams = bundles[0].getBitstreams();
418 
419         // remove bundle if it's now empty
420         if (bitstreams.length < 1)
421         {
422             item.removeBundle(bundles[0]);
423             item.update();
424         }
425 
426         // no errors occurred
427         return STATUS_COMPLETE;
428     }

430     /**
431      * Process the upload of a new file!
432      * 
433      * @param context
434      *            current DSpace context
435      * @param request
436      *            current servlet request object
437      * @param response
438      *            current servlet response object
439      * @param subInfo
440      *            submission info object
441      * 
442      * @return Status or error flag which will be processed by
443      *         UI-related code! (if STATUS_COMPLETE or 0 is returned,
444      *         no errors occurred!)
445      */
446     protected int processUploadFile(Context context, HttpServletRequest request,
447             HttpServletResponse response, SubmissionInfo subInfo)
448             throws ServletException, IOException, SQLException,
449             AuthorizeException
450     {
451         boolean formatKnown = true;
452         boolean fileOK = false;
453         BitstreamFormat bf = null;
454         Bitstream b = null;
455  
456         //NOTE: File should already be uploaded. 
457         //Manakin does this automatically via Cocoon.
458         //For JSP-UI, the SubmissionController.uploadFiles() does the actual upload
459 
460         Enumeration attNames = request.getAttributeNames();
461         
462         //loop through our request attributes
463         while(attNames.hasMoreElements())
464         {
465             String attr = (String) attNames.nextElement();
466             
467             //if this ends with "-path", this attribute
468             //represents a newly uploaded file
469             if(attr.endsWith("-path"))
470             {
471                 //strip off the -path to get the actual parameter 
472                 //that the file was uploaded as
473                 String param = attr.replace("-path", "");
474                 
475                 // Load the file's path and input stream and description
476                 String filePath = (String) request.getAttribute(param + "-path");
477                 InputStream fileInputStream = (InputStream) request
478                                     .getAttribute(param + "-inputstream");
479                 
480                 //attempt to get description from attribute first, then direct from a parameter
481                 String fileDescription =  (String) request
482                                     .getAttribute(param + "-description");
483                 if(fileDescription==null ||fileDescription.length()==0)
484                     request.getParameter("description");
485                 
486                 // if information wasn't passed by User Interface, we had a problem
487                 // with the upload
488                 if (filePath == null || fileInputStream == null)
489                     return STATUS_UPLOAD_ERROR;
490                 
491                 if (subInfo != null)
492                 {
493                     // Create the bitstream
494                     Item item = subInfo.getSubmissionItem().getItem();
495         
496                     // do we already have a bundle?
497                     Bundle[] bundles = item.getBundles("ORIGINAL");
498         
499                     if (bundles.length < 1)
500                     {
501                         // set bundle's name to ORIGINAL
502                         b = item.createSingleBitstream(fileInputStream, "ORIGINAL");
503                     }
504                     else
505                     {
506                         // we have a bundle already, just add bitstream
507                         b = bundles[0].createBitstream(fileInputStream);
508                     }
509         
510                     // Strip all but the last filename. It would be nice
511                     // to know which OS the file came from.
512                     String noPath = filePath;
513         
514                     while (noPath.indexOf('/') > -1)
515                     {
516                         noPath = noPath.substring(noPath.indexOf('/') + 1);
517                     }
518         
519                     while (noPath.indexOf('\\') > -1)
520                     {
521                         noPath = noPath.substring(noPath.indexOf('\\') + 1);
522                     }
523         
524                     b.setName(noPath);
525                     b.setSource(filePath);
526                     b.setDescription(fileDescription);
527         
528                     // Identify the format
529                     bf = FormatIdentifier.guessFormat(context, b);
530                     b.setFormat(bf);
531         
532                     // Update to DB
533                     b.update();
534                     item.update();
535         
536                     if (bf == null || !bf.isInternal())
537                     {
538                         fileOK = true;
539                     }
540                     else
541                     {
542                         log.warn("Attempt to upload file format marked as internal system use only");
543                         
544                         // remove bitstream from bundle..
545                         // delete bundle if it's now empty
546                         Bundle[] bnd = b.getBundles();
547 
548                         bnd[0].removeBitstream(b);
549 
550                         Bitstream[] bitstreams = bnd[0].getBitstreams();
551 
552                         // remove bundle if it's now empty
553                         if (bitstreams.length < 1)
554                         {
555                             item.removeBundle(bnd[0]);
556                             item.update();
557                         }
558 
559                         subInfo.setBitstream(null);
560                     }
561                 }// if subInfo not null
562                 else
563                 {
564                     // In any event, if we don't have the submission info, the request
565                     // was malformed
566                     return STATUS_INTEGRITY_ERROR;
567                 }
568         
569                 // as long as everything completed ok, commit changes. Otherwise show
570                 // error page.
571                 if (fileOK)
572                 {
573                     context.commit();
574         
575                     // save this bitstream to the submission info, as the
576                     // bitstream we're currently working with
577                     subInfo.setBitstream(b);
578         
579                     //if format was not identified
580                     if (bf == null)
581                     {
582                         // the bitstream format is unknown!
583                         formatKnown=false;
584                     }
585                 }
586                 else
587                 {
588                     // if we get here there was a problem uploading the file!
589                     return STATUS_UPLOAD_ERROR;
590                 }
591             }//end if attribute ends with "-path"
592         }//end while
593         
594         if(!formatKnown)
595         {
596             //return that the bitstream format is unknown!
597             return STATUS_UNKNOWN_FORMAT;
598         }
599         else
600             return STATUS_COMPLETE;
601               
602     }

604     /**
605      * Process input from get file type page
606      * 
607      * @param context
608      *            current DSpace context
609      * @param request
610      *            current servlet request object
611      * @param response
612      *            current servlet response object
613      * @param subInfo
614      *            submission info object
615      * 
616      * @return Status or error flag which will be processed by
617      *         UI-related code! (if STATUS_COMPLETE or 0 is returned,
618      *         no errors occurred!)
619      */
620     protected int processSaveFileFormat(Context context,
621             HttpServletRequest request, HttpServletResponse response,
622             SubmissionInfo subInfo) throws ServletException, IOException,
623             SQLException, AuthorizeException
624     {
625         if (subInfo.getBitstream() != null)
626         {
627             // Did the user select a format?
628             int typeID = Util.getIntParameter(request, "format");
629 
630             BitstreamFormat format = BitstreamFormat.find(context, typeID);
631 
632             if (format != null)
633             {
634                 subInfo.getBitstream().setFormat(format);
635             }
636             else
637             {
638                 String userDesc = request.getParameter("format_description");
639 
640                 subInfo.getBitstream().setUserFormatDescription(userDesc);
641             }
642 
643             // update database
644             subInfo.getBitstream().update();
645         }
646         else
647         {
648             return STATUS_INTEGRITY_ERROR;
649         }
650 
651         return STATUS_COMPLETE;
652     }
