56  /**
57   * Abstract processing class for DSpace Submission Steps. This defines the base
58   * methods which are required for any Step processing class.
59   * <P>
60   * This abstract class defines the base methods which are used by both the
61   * Manakin XML UI and the JSP UI to perform submission step processing.
62   * <P>
63   * This includes the following methods:
64   * <ul>
65   * <li>doProcessing() method - called to perform any step processing</li>
66   * <li>getErrorFields() method - called to determine the fields which errored
67   * out during processing</li>
68   * <li>getErrorMessage() method - called to determine any error message
69   * returned after processing</li>
70   * </ul>
71   * <P>
72   * If you are using the JSP UI (with the SubmissionController servlet) you
73   * should extend the org.dspace.submit.SubmissionStep class, which defines
74   * additional methods used to maintain the context of the submission within a
75   * JSP environment!
76   * 
77   * @see org.dspace.app.webui.submit.JSPStepManager
78   * @see org.dspace.app.webui.servlet.SubmissionController
79   * @see org.dspace.app.util.SubmissionConfig
80   * @see org.dspace.app.util.SubmissionStepConfig
81   * 
82   * @author Tim Donohue
83   * @version $Revision: 3738 $
84   */

243     /**
244      * Return the text of an error message based on the passed in error flag.
245      * These error messages are used for non-interactive steps (so that they can
246      * log something more specific than just an error flag)
247      * <P>
248      * Since each step can define its own error messages and flags, this method
249      * depends on all the error messages being initialized by using the
250      * "addErrorMessage()" method within the constructor for the step class!
251      * 
252      * @param errorFlag
253      *            The error flag defined in this step which represents an error
254      *            message.
255      * @return String which contains the text of the error message, or null if
256      *         error message not found
257      */
258     public final String getErrorMessage(int errorFlag)
259     {
260         if (this.errorMessages == null || this.errorMessages.size() == 0)
261             return null;
262         else
263             return (String) this.errorMessages.get(Integer.valueOf(errorFlag));
264     }

311     /**
312      * Find out which page a user is currently viewing
313      * 
314      * @param request
315      *            HTTP request
316      * 
317      * @return current page
318      */
319     public static final int getCurrentPage(HttpServletRequest request)
320     {
321         int pageNum = -1;
322 
323         // try to retrieve cached page from request attribute
324         Integer currentPage = (Integer) request.getAttribute("submission.page");
325 
326         if (currentPage == null)
327         {
328             // try and get it as a 'page' parameter
329             String val = request.getParameter("page");
330 
331             try
332             {
333                 pageNum =  Integer.parseInt(val.trim());
334             }
335             catch (Exception e)
336             {
337                 // Problem with parameter
338                 pageNum = -1;
339             }
340             
341             // if couldn't find page in request parameter
342             if (pageNum < 0)
343             {
344                 // default to page #1, since no other optionsc
345                 pageNum = 1;
346 
347                 setCurrentPage(request, pageNum);
348             }
349             else
350             {
351                 // save to request attribute
352                 setCurrentPage(request, pageNum);
353             }
354         }
355         else
356         {
357             pageNum = currentPage.intValue();
358         }
359 
360         return pageNum;
361     }
