54  /**
55   * An Event object represents a single action that changed one object in the
56   * DSpace data model. An "atomic" action at the application or business-logic
57   * API level may spawn many of these events.
58   * <p>
59   * This class includes tools to help set and use the contents of the event. Note
60   * that it describes DSpace data object types in two ways: by the type
61   * identifiers in the Constants class, and also by an Event-specific bitmask
62   * (used by its internal filters). All public API calls use the Constants
63   * version of the data model types.
64   * <p>
65   * Note that the type of the event itself is actually descriptive of the
66   * <em>action</em> it performs: ADD, MODIFY, etc. The most significant
67   * elements of the event are:
68   * <p>
69   * <br> - (Action) Type <br> - Subject -- DSpace object to which the action
70   * applies, e.g. the Collection to which an ADD adds a member. <br> - Object --
71   * optional, when present it is the other object effected by an action, e.g. the
72   * Item ADDed to a Collection by an ADD. <br> - detail -- a textual summary of
73   * what changed, content and its significance varies by the combination of
74   * action and subject type. <br> - timestamp -- exact millisecond timestamp at
75   * which event was logged.
76   * 
77   * @version $Revision: 3762 $
78   */

263     /**
264      * Compare two events. Ignore any difference in the timestamps. Also ignore
265      * transactionID since that is not always set initially.
266      * 
267      * @param other
268      *            the event to compare this one to
269      * @returns true if events are "equal", false otherwise.
270      */
271     public boolean equals(Event other)
272     {
273         return (this.detail == null ? other.detail == null : this.detail
274                 .equals(other.detail))
275                 && this.eventType == other.eventType
276                 && this.subjectType == other.subjectType
277                 && this.subjectID == other.subjectID
278                 && this.objectType == other.objectType
279                 && this.objectID == other.objectID;
280     }

396     /**
397      * Translate a textual DSpace Object type name into an event subject-type
398      * mask. NOTE: This returns a BIT-MASK, not a numeric type value; the mask
399      * is only used within the event system.
400      * 
401      * @param s
402      *            text name of object type.
403      * @returns numeric value of object type or 0 for error.
404      */
405     public static int parseObjectType(String s)
406     {
407         if ("*".equals(s) || "all".equalsIgnoreCase(s))
408             return ALL_OBJECTS_MASK;
409         else
410         {
411             int id = Constants.getTypeID(s.toUpperCase());
412             if (id >= 0)
413                 return 1 << id;
414         }
415         return 0;
416     }

442     /**
443      * Interpret named event type.
444      * 
445      * @param text
446      *            name of event type.
447      * @returns numeric value of event type or 0 for error.
448      */
449     public static int parseEventType(String s)
450     {
451         if ("*".equals(s) || "all".equalsIgnoreCase(s))
452         {
453             int result = 0;
454             for (int i = 0; i < eventTypeText.length; ++i)
455                 result |= (1 << i);
456             return result;
457         }
458 
459         for (int i = 0; i < eventTypeText.length; ++i)
460             if (eventTypeText[i].equalsIgnoreCase(s))
461                 return 1 << i;
462         return 0;
463     }

530     /**
531      * @param filters
532      *            list of filter masks; each one is an Array of two ints.
533      * @returns true if this event would be passed through the given filter
534      *          list.
535      */
536     public boolean pass(List filters)
537     {
538         boolean result = false;
539 
540         for (Iterator fi = filters.iterator(); fi.hasNext();)
541         {
542             int filter[] = (int[]) fi.next();
543             if ((subjectType & filter[SUBJECT_MASK]) != 0
544                     && (eventType & filter[EVENT_MASK]) != 0)
545                 result = true;
546         }
547 
548         if (log.isDebugEnabled())
549             log.debug("Filtering event: " + "eventType="
550                     + String.valueOf(eventType) + ", subjectType="
551                     + String.valueOf(subjectType) + ", result="
552                     + String.valueOf(result));
553 
554         return result;
555     }

557     // dumb integer "log base 2", returns -1 if there are no 1's in number.
558     private static int log2(int n)
559     {
560         for (int i = 0; i < 32; ++i)
561             if (n == 1)
562                 return i;
563             else
564                 n = n >> 1;
565         return -1;
566     }
