52  /**
53   * Implements MARC 21 standards to disregard initial
54   * definite or indefinite article in sorting.
55   * 
56   * Note: This only works for languages defined with IANA code entries.
57   * 
58   * @author Graham Triggs
59   */
60  public class MARC21InitialArticleWord extends InitialArticleWord
61  {
62      public MARC21InitialArticleWord()
63      {
64          // Default behaviour is to strip the initial word completely
65          super(true);
66      }
67  
68      public MARC21InitialArticleWord(boolean stripWord)
69      {
70          super(stripWord);
71      }
72  
73      /**
74       * Return the list of definite and indefinite article codes
75       * for this language.
76       */
77      protected String[] getArticleWords(String lang)
78      {
79          // No language - no words
80          if (StringUtils.isEmpty(lang))
81              return defaultWords;
82  
83          Language l = Language.getLanguage(lang);
84          
85          // Is the lanugage in our map?
86          if (l != null && ianaArticleMap.containsKey(l.IANA))
87          {
88              // Get the list of words for this language
89              ArticlesForLang articles = (ArticlesForLang)ianaArticleMap.get(l.IANA);
90              
91              if (articles != null)
92                  return articles.words;
93          }
94          
95          return null;
96      }
97      
98      // Mapping of IANA codes to article word lists
99      private static Map ianaArticleMap = new HashMap();

254         // Initialize the lang -> article map
255         ianaArticleMap = new HashMap();
256 
257         int wordIdx = 0;
258         int langIdx = 0;
259 
260         // Iterate through word/language array
261         // Generate temporary language map
262         Map langWordMap = new HashMap();
263         for (wordIdx = 0; wordIdx < articleWordArray.length; wordIdx++)
264         {
265             for (langIdx = 1; langIdx < articleWordArray[wordIdx].length; langIdx++)
266             {
267                 Language lang = (Language)articleWordArray[wordIdx][langIdx];
268 
269                 if (lang != null && lang.IANA.length() > 0)
270                 {
271                     List words = (List)langWordMap.get(lang);
272                     
273                     if (words == null)
274                     {
275                         words = new ArrayList();
276                         langWordMap.put(lang, words);
277                     }
278                     
279                     // Add language to list if we haven't done so already
280                     if (!words.contains(articleWordArray[wordIdx][0]))
281                         words.add(articleWordArray[wordIdx][0]);
282                 }
283             }
284         }
285         
286         // Iterate through languages
287         Iterator langIter = langWordMap.keySet().iterator();
288         while (langIter.hasNext())
289         {
290             Language lang = (Language)langIter.next();
291             List wordList = (List)langWordMap.get(lang);
292 
293             // Convert the list into an array of strings
294             String[] words = new String[wordList.size()];
295             
296             for (int idx = 0; idx < wordList.size(); idx++)
297                 words[idx] = (String)wordList.get(idx);
298 
299             // Sort the array into length order - longest to shortest
300             // This ensures maximal matching on the article words
301             Arrays.sort(words, new MARC21InitialArticleWord.InverseLengthComparator() );
302             
303             // Add language/article entry to map
304             ianaArticleMap.put(lang.IANA, new MARC21InitialArticleWord.ArticlesForLang(lang, words));
305         }
306 
307         // Setup default stop words for null languages
308         String defaultLangs = ConfigurationManager.getProperty("marc21wordfilter.defaultlang");
309         if (!StringUtils.isEmpty(defaultLangs))
310         {
311             String[] langArr = defaultLangs.split("[, ]+");
312             if (langArr != null && langArr.length > 0)
313             {
314                 int wordCount = 0;
315                 ArticlesForLang[] afl = new ArticlesForLang[langArr.length];
316 
317                 for (int idx = 0; idx < afl.length; idx++)
318                 {
319                     Language l = Language.getLanguage(langArr[idx]);
320                     if (l != null && ianaArticleMap.containsKey(l.IANA))
321                     {
322                         afl[idx] = (ArticlesForLang)ianaArticleMap.get(l.IANA);
323                         if (afl[idx] != null)
324                         {
325                             wordCount += afl[idx].words.length;
326                         }
327                     }
328                 }
329 
330                 if (wordCount > 0)
331                 {
332                     int destPos = 0;
333                     defaultWords = new String[wordCount];
334                     for (int idx = 0; idx < afl.length; idx++)
335                     {
336                         if (afl[idx] != null)
337                         {
338                             System.arraycopy(afl[idx].words, 0, defaultWords, destPos, afl[idx].words.length);
339                             destPos += afl[idx].words.length;
340                         }
341                     }
342                 }
343             }
344         }
345     }

347     // Wrapper class for inserting word arrays into a map
348     private static class ArticlesForLang
349     {
350         final Language lang;
351         final String[] words;
352         
353         ArticlesForLang(Language lang, String[] words)
354         {
355            this.lang  = lang;
356            this.words = words;
357         }
358     }
359     
360     // Compare strings according to their length - longest to shortest
361     private static class InverseLengthComparator implements Comparator
362     {
363         public int compare(Object arg0, Object arg1)
364         {
365             return ((String)arg1).length() - ((String)arg0).length(); 
366         };
367         
368     };
