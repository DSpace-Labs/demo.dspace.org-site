115     private static synchronized Language create(String iana, String iso639_1, String iso639_2)
116     {
117         Language lang = null;
118         
119         lang = (lang != null ? lang : LanguageMaps.getLanguageForIANA(iana));
120         lang = (lang != null ? lang : LanguageMaps.getLanguageForISO639_1(iso639_1));
121         lang = (lang != null ? lang : LanguageMaps.getLanguageForISO639_2(iso639_2));
122         
123         return (lang != null ? lang : new Language(iana, iso639_1, iso639_2));
124     }
125     
126     private static class LanguageMaps
127     {
128         private static final Map langMapIANA     = new HashMap();
129         private static final Map langMapISO639_1 = new HashMap();
130         private static final Map langMapISO639_2 = new HashMap();
131 
132         static void add(Language l)
133         {
134             if (l.IANA != null && l.IANA.length() > 0 && !langMapIANA.containsKey(l.IANA))
135                 langMapIANA.put(l.IANA, l);
136             
137             if (l.ISO639_1 != null && l.ISO639_1.length() > 0 && !langMapISO639_1.containsKey(l.ISO639_1))
138                 langMapISO639_1.put(l.ISO639_1, l);
139 
140             if (l.ISO639_2 != null && l.ISO639_2.length() > 0 && !langMapISO639_2.containsKey(l.ISO639_2))
141                 langMapISO639_2.put(l.ISO639_2, l);
142         }
143 
144         public static Language getLanguage(String lang)
145         {
146             if (langMapIANA.containsKey(lang))
147                 return (Language)langMapIANA.get(lang);
148 
149             return (Language)langMapISO639_2.get(lang);
150         }
151 
152         public static Language getLanguageForIANA(String iana)
153         {
154             return (Language)langMapIANA.get(iana);
155         }
156         
157         public static Language getLanguageForISO639_1(String iso)
158         {
159             return (Language)langMapISO639_1.get(iso);
160         }
161 
162         public static Language getLanguageForISO639_2(String iso)
163         {
164             return (Language)langMapISO639_2.get(iso);
165         }
166     }
