39  package org.dspace.text.filter;
40  
41  /**
42   * Abstract class for implementing initial article word filters
43   * Allows you to create new classes with their own rules for mapping
44   * languages to article word lists.
45   * 
46   * @author Graham Triggs
47   */
48  public abstract class InitialArticleWord implements TextFilter
49  {
50      /**
51       * When no language is passed, use null and let implementation decide what to do
52       */
53      public String filter(String str)
54      {
55          return filter(str, null);
56      }
57      
58      /**
59       * Do an initial definite/indefinite article filter on the passed string.
60       * On matching an initial word, can strip or move to the end, depending on the 
61       * configuration of the implementing class.
62       * 
63       * @param str  The string to parse
64       * @param lang The language of the passed string
65       * @return String The filtered string
66       */
67      public String filter(String str, String lang)
68      {
69          // Get the list of article words for this language
70          String[] articleWordArr = getArticleWords(lang);
71  
72          // If we have an article word array, process the string
73          if (articleWordArr != null && articleWordArr.length > 0)
74          {
75              String initialArticleWord = null;
76              int curPos =  0;
77              int initialStart = -1;
78              int initialEnd   = -1;
79              
80              // Iterate through the characters until we find something significant, or hit the end
81              while (initialEnd < 0 && curPos < str.length())
82              {
83                  // Have we found a significant character
84                  if (Character.isLetterOrDigit(str.charAt(curPos)))
85                  {
86                      // Mark this as the cut point for the initial word
87                      initialStart = curPos;
88                      
89                      // Loop through the article words looking for a match
90                      for (int idx = 0; initialEnd < 0 && idx < articleWordArr.length; idx++)
91                      {
92                          // Extract a fragment from the string to test
93                          // Must be same length as the article word
94                          if (idx > 1 && initialArticleWord != null)
95                          {
96                              // Only need to do so if we haven't already got one
97                              // of the right length
98                              if (initialArticleWord.length() != articleWordArr[idx].length())
99                                  initialArticleWord = extractText(str, curPos, articleWordArr[idx].length());
100                         }
101                         else
102                         {
103                             initialArticleWord = extractText(str, curPos, articleWordArr[idx].length());
104                         }
105 
106                         // Does the fragment match an article word?
107                         if (initialArticleWord!= null && initialArticleWord.equalsIgnoreCase(articleWordArr[idx]))
108                         {
109                             // Check to see if the next character in the source
110                             // is a whitespace
111                             boolean isNextWhitespace = Character.isWhitespace(
112                                     str.charAt(curPos + articleWordArr[idx].length())
113                                 );
114                             
115                             // Check to see if the last character of the article word is a letter or digit
116                             boolean endsLetterOrDigit = Character.isLetterOrDigit(initialArticleWord.charAt(initialArticleWord.length() - 1));
117                          
118                             // If the last character of the article word is  a letter or digit,
119                             // then it must be followed by whitespace, if not, it can be anything
120                             // Setting endPos signifies that we have found an article word
121                             if (endsLetterOrDigit && isNextWhitespace)
122                                 initialEnd = curPos + initialArticleWord.length();
123                             else if (!endsLetterOrDigit)
124                                 initialEnd = curPos + initialArticleWord.length();
125                         }
126                     }
127 
128                     // Quit the loop, as we have a significant character
129                     break;
130                 }
131                 
132                 // Keep going
133                 curPos++;
134             }
135             
136             // If endPos is positive, then we've found an article word
137             if (initialEnd > 0)
138             {
139                 // Find a cut point in the source string, removing any whitespace after the article word
140                 int cutPos = initialEnd;
141                 while (cutPos < str.length() && Character.isWhitespace(str.charAt(cutPos)))
142                     cutPos++;
143                 
144                 // Are we stripping the article word?
145                 if (stripInitialArticle)
146                 {
147                     // Yes, simply return everything after the cut
148                     return str.substring(cutPos);
149                 }
150                 else
151                 {
152                     // No - move the initial article word to the end
153                     return new StringBuffer(str.substring(cutPos))
154                                         .append(wordSeperator)
155                                         .append(str.substring(initialStart, initialEnd))
156                                         .toString();
157                 }
158             }
159         }
160         
161         // Didn't do any processing, or didn't find an initial article word
162         // Return the original string
163         return str;
164     }
165     
166     protected InitialArticleWord(boolean stripWord)
167     {
168         stripInitialArticle = stripWord;
169     }
170     
171     protected InitialArticleWord()
172     {
173         stripInitialArticle = false;
174     }
175 
176     /**
177      * Abstract method to get the list of words to use in the initial word filter
178      * 
179      * @param lang The language to retrieve article words for
180      * @return An array of definite/indefinite article words
181      */
182     protected abstract String[] getArticleWords(String lang);
183 
184     // Seperator to use when appending article to end
185     private String wordSeperator = ", ";
186 
187     // Flag to signify initial article word should be removed
188     // If false, then the initial article word is appended to the end
189     private boolean stripInitialArticle = false;
190     
191     /**
192      * Helper method to extract text from a string.
193      * Ensures that there is significant data (ie. non-whitespace)
194      * after the segment requested.
195      * 
196      * @param str
197      * @param pos
198      * @param len
199      * @return
200      */
201     private String extractText(String str, int pos, int len)
202     {
203         int testPos = pos + len;
204         while (testPos < str.length() && Character.isWhitespace(str.charAt(testPos)))
205             testPos++;
206         
207         if (testPos < str.length())
208             return str.substring(pos, pos + len);
209         
210         return null;
211     }
