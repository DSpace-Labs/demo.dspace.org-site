157     /**
158      * Execute command line. See Usage string for options and arguments.
159      * 
160      * @param argv the argv
161      * 
162      * @throws Exception the exception
163      */
164     public static void main(String[] argv) throws Exception
165     {
166         Options options = new Options();
167 
168         OptionGroup func = new OptionGroup();
169         func.addOption(new Option("c", "copy", true,
170                 "copy <Item> to -C <Collection>"));
171         func.addOption(new Option("s", "submit", true,
172                 "submit <collection> -P <packager> -i <file>"));
173         func.addOption(new Option("d", "disseminate", true,
174                 "disseminate <item> -P <packager> -o <file>"));
175         func.addOption(new Option("f", "propfind", true,
176                 "propfind of all properties or -N <propname>"));
177         func.addOption(new Option("r", "rpropfind", true,
178                 "recursive propfind, only collections"));
179         func.addOption(new Option("n", "names", true,
180                 "list all property names on resource"));
181         func.addOption(new Option("p", "proppatch", true,
182                 "set property: <handle> -N <property> -V <newvalue>"));
183         func.setRequired(true);
184         options.addOptionGroup(func);
185 
186         options.addOption("h", "help", false, "show help message");
187         options
188                 .addOption("e", "endpoint", true,
189                         "SOAP endpoint URL (REQUIRED)");
190         options.addOption("P", "packager", true,
191                 "Packager to use to import/export a package.");
192         options.addOption("C", "collection", true,
193                 "Target collection of -c copy");
194         options
195                 .addOption("o", "output", true,
196                         "file to create for new package");
197         options.addOption("i", "input", true,
198                 "file containing package to submit");
199         options.addOption("N", "name", true, "name of property to query/set");
200         options.addOption("V", "value", true,
201                 "new value for property being set");
202 
203         try
204         {
205             CommandLine line = (new PosixParser()).parse(options, argv);
206             if (line.hasOption("h"))
207             {
208                 Usage(options, 0, null);
209             }
210 
211             // get SOAP client connection, using the endpoint URL
212             String endpoint = line.getOptionValue("e");
213             if (endpoint == null)
214             {
215                 Usage(options, 2, "Missing the required -e endpoint argument");
216             }
217             LNISoapServletServiceLocator loc = new LNISoapServletServiceLocator();
218             LNISoapServlet lni = loc.getDSpaceLNI(new URL(endpoint));
219 
220             // propfind - with optional single-property Name
221             if (line.hasOption("f"))
222             {
223                 String pfXml = (line.hasOption("N")) ? specificPropPrefix
224                         + line.getOptionValue("N") + specificPropSuffix
225                         : allProp;
226                 doPropfind(lni, line.getOptionValue("f"), pfXml, 0, null);
227             }
228 
229             // recursive propfind limited to collection, community objects
230             else if (line.hasOption("r"))
231             {
232                 doPropfind(lni, line.getOptionValue("r"), someProp, -1,
233                         "collection,community");
234             }
235             else if (line.hasOption("n"))
236             {
237                 doPropfind(lni, line.getOptionValue("n"), nameProp, 0, null);
238             }
239             else if (line.hasOption("p"))
240             {
241                 if (line.hasOption("N") && line.hasOption("V"))
242                 {
243                     doProppatch(lni, line.getOptionValue("p"), line
244                             .getOptionValue("N"), line.getOptionValue("V"));
245                 }
246                 else
247                 {
248                     Usage(options, 13,
249                             "Missing required args: -N <name> -V <value>n");
250                 }
251             }
252 
253             // submit a package
254             else if (line.hasOption("s"))
255             {
256                 if (line.hasOption("P") && line.hasOption("i"))
257                 {
258                     doPut(lni, line.getOptionValue("s"), line
259                             .getOptionValue("P"), line.getOptionValue("i"),
260                             endpoint);
261                 }
262                 else
263                 {
264                     Usage(options, 13,
265                             "Missing required args after -s: -P <packager> -i <file>");
266                 }
267             }
268 
269             // Disseminate (GET) item as package
270             else if (line.hasOption("d"))
271             {
272                 if (line.hasOption("P") && line.hasOption("o"))
273                 {
274                     doGet(lni, line.getOptionValue("d"), line
275                             .getOptionValue("P"), line.getOptionValue("o"),
276                             endpoint);
277                 }
278                 else
279                 {
280                     Usage(options, 13,
281                             "Missing required args after -d: -P <packager> -o <file>");
282                 }
283             }
284 
285             // copy from src to dst
286             else if (line.hasOption("c"))
287             {
288                 if (line.hasOption("C"))
289                 {
290                     doCopy(lni, line.getOptionValue("c"), line
291                             .getOptionValue("C"));
292                 }
293                 else
294                 {
295                     Usage(options, 13,
296                             "Missing required args after -c: -C <collection>\n");
297                 }
298             }
299             else
300             {
301                 Usage(options, 14, "Missing command option.\n");
302             }
303 
304         }
305         catch (ParseException pe)
306         {
307             Usage(options, 1, "Error in arguments: " + pe.toString());
308             
309         }
310         catch (java.rmi.RemoteException de)
311         {
312             System.out.println("ERROR, got RemoteException, message="
313                     + de.getMessage());
314             
315             de.printStackTrace();
316             
317             die(1, "  Exception class=" + de.getClass().getName());
318         }

333     // Like LNI lookup(), but dies on error so we don't have to check result.
334     // Also interprets the special format "handle bitstream" (with space
335     // separating the words) as the ``handle'' of a bitstream -- a combination
336     // of Item handle and bitstream sequence-ID.
337     // On success it returns a DAV resource URI, relative to the
338     // root of the DAV resource hierarchy, e.g. "/dso_12345.678$123"
339     /**
340      * Do lookup.
341      * 
342      * @param lni the lni
343      * @param handle the handle
344      * @param bitstream the bitstream
345      * 
346      * @return the string
347      * 
348      * @throws RemoteException the remote exception
349      */
350     private static String doLookup(LNISoapServlet lni, String handle,
351             String bitstream) throws java.rmi.RemoteException
352     {
353         // hack: if "handle" starts with '/' and there is no bitstream
354         // assume it is URI passed in and just return that.
355         if (handle.startsWith("/") && bitstream == null)
356         {
357             return handle;
358         }
359 
360         // hack: parse "handle bitstream" syntax of handle.
361         if (handle.indexOf(" ") >= 0 && bitstream == null)
362         {
363             String h[] = handle.split("\\s+");
364             handle = h[0];
365             bitstream = h[1];
366         }
367 
368         String uri = lni.lookup(handle, bitstream);
369         if (uri == null)
370         {
371             die(2, "ERROR, got null from lookup(\"" + handle + "\")");
372         }
373         System.out.println("DEBUG: lookup returns: \"" + uri + "\"");
374         return uri;
375     }

377     /**
378      * Do propfind.
379      * 
380      * @param lni the lni
381      * @param handle the handle
382      * @param pf the pf
383      * @param depth the depth
384      * @param types the types
385      * 
386      * @throws RemoteException the remote exception
387      * @throws IOException Signals that an I/O exception has occurred.
388      */
389     private static void doPropfind(LNISoapServlet lni, String handle,
390             String pf, int depth, String types)
391             throws java.rmi.RemoteException, java.io.IOException
392     {
393         String uri = doLookup(lni, handle, null);
394         String result = lni.propfind(uri, pf, depth, types);
395         try
396         {
397             SAXBuilder builder = new SAXBuilder();
398             Document msDoc = builder.build(new java.io.StringReader(result));
399             Element ms = msDoc.getRootElement();
400             ListIterator ri = ms.getChildren("response", NS_DAV).listIterator();
401             while (ri.hasNext())
402             {
403                 Element resp = (Element) ri.next();
404                 String href = resp.getChildText("href", NS_DAV);
405                 System.out.println("Resource = " + href);
406                 ListIterator pi = resp.getChildren("propstat", NS_DAV)
407                         .listIterator();
408                 while (pi.hasNext())
409                 {
410                     Element ps = (Element) pi.next();
411                     String status = ps.getChildText("status", NS_DAV);
412                     if (status.indexOf("200") >= 0)
413                     {
414                         System.out
415                                 .println("  === PROPERTIES Successfully returned:");
416                     }
417                     else
418                     {
419                         System.out.println("  === PROPERTIES with Status="
420                                 + status);
421                     }
422                     // print properties and values
423                     Element prop = ps.getChild("prop", NS_DAV);
424                     ListIterator ppi = prop.getChildren().listIterator();
425                     while (ppi.hasNext())
426                     {
427                         Element e = (Element) ppi.next();
428                         String value = e.getTextTrim();
429                         if (value.equals(""))
430                         {
431                             List kids = e.getChildren();
432                             if (kids.size() > 0)
433                             {
434                                 value = outputPretty.outputString(kids);
435                             }
436                             if (value.indexOf("\n") >= 0)
437                             {
438                                 value = "\n" + value;
439                             }
440                         }
441                         else
442                         {
443                             value = "\"" + value + "\"";
444                         }
445                         String equals = value.equals("") ? "" : " = ";
446                         System.out.println("    " + e.getQualifiedName()
447                                 + equals + value);
448                     }
449                 }
450             }
451 
452         }
453         catch (JDOMParseException je)
454         {
455             je.printStackTrace();
456             
457             die(3, "ERROR: " + je.toString());
458         }
459         catch (JDOMException je)
460         {
461             je.printStackTrace();
462             
463             die(4, "ERROR: " + je.toString());
464         }
465 
466     }

517     /**
518      * Implement WebDAV PUT http request.
519      * 
520      * This might be simpler with a real HTTP client library, but
521      * java.net.HttpURLConnection is part of the standard SDK and it
522      * demonstrates the concepts.
523      * 
524      * @param lni the lni
525      * @param collHandle the coll handle
526      * @param packager the packager
527      * @param source the source
528      * @param endpoint the endpoint
529      * 
530      * @throws RemoteException the remote exception
531      * @throws ProtocolException the protocol exception
532      * @throws IOException Signals that an I/O exception has occurred.
533      * @throws FileNotFoundException the file not found exception
534      */
535     private static void doPut(LNISoapServlet lni, String collHandle,
536             String packager, String source, String endpoint)
537             throws java.rmi.RemoteException, ProtocolException, IOException,
538             FileNotFoundException
539     {
540         // assemble URL from chopped endpoint-URL and relative URI
541         String collURI = doLookup(lni, collHandle, null);
542         URL url = LNIClientUtils.makeDAVURL(endpoint, collURI, packager);
543         System.err.println("DEBUG: PUT file=" + source + " to URL="
544                 + url.toString());
545 
546         // connect with PUT method, then copy file over.
547         HttpURLConnection conn = (HttpURLConnection) url.openConnection();
548         conn.setRequestMethod("PUT");
549         conn.setDoOutput(true);
550         fixBasicAuth(url, conn);
551         conn.connect();
552 
553         InputStream in = new FileInputStream(source);
554         OutputStream out = conn.getOutputStream();
555         try {
556             copyStream(in, out);
557         } finally {
558             in.close();
559             out.close();
560         }
561 
562         int status = conn.getResponseCode();
563         if (status < 200 || status >= 300)
564         {
565             die(status, "HTTP error, status=" + String.valueOf(status)
566                     + ", message=" + conn.getResponseMessage());
567         }
568 
569         // diagnostics, and get resulting new item's location if avail.
570         System.err.println("DEBUG: sent " + source);
571         System.err.println("RESULT: Status="
572                 + String.valueOf(conn.getResponseCode()) + " "
573                 + conn.getResponseMessage());
574         String loc = conn.getHeaderField("Location");
575         System.err.println("RESULT: Location="
576                 + ((loc == null) ? "NULL!" : loc));
577     }

620         out.close();

623         System.err.println("RESULT: Status="
624                 + String.valueOf(conn.getResponseCode()) + " "
625                 + conn.getResponseMessage());

649     /**
650      * Copy stream. copy from one stream to another
651      * 
652      * @param input the input
653      * @param output the output
654      * 
655      * @throws IOException Signals that an I/O exception has occurred.
656      */
657     private static void copyStream(final InputStream input,
658             final OutputStream output) throws IOException
659     {
660         final int BUFFER_SIZE = 1024 * 4;
661         final byte[] buffer = new byte[BUFFER_SIZE];
662 
663         while (true)
664         {
665             final int count = input.read(buffer, 0, BUFFER_SIZE);
666             if (-1 == count)
667             {
668                 break;
669             }
670             output.write(buffer, 0, count);
671         }
672     }
