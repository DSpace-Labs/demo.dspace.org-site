12  /***
13   * Allow a service to be notified when the service manager has started
14   * up all known services (including activators).  This is useful if a
15   * service wants to do lookups of all known services or providers of a 
16   * certain type when the system startup is complete.  It can also be 
17   * used as a way to execute some code when the service manager startup
18   * is complete.
19   * 
20   * @author Aaron Zeckoski (azeckoski @ gmail.com)
21   */
22  public interface ServiceManagerReadyAware {
23  
24      /***
25       * This is called when the startup of all core services and 
26       * activators is complete.
27       * <p>
28       * WARNING: This will never be called for providers which are 
29       * registered with a service manager which is already running!
30       * 
31       * @param serviceManager the completely started service manager
32       */
33      public void serviceManagerReady(ServiceManager serviceManager);
34  
35  }
