8   package org.dspace.kernel;
9   
10  /***
11   * Beans that have a lifecycle and can be controlled via their lifecycle implement this interface.
12   * Based on the Sakai K2 lifecycle interface -AZ
13   * 
14   * @param <T> the type of object managed by this lifecycle.
15   */
16  public interface CommonLifecycle<T> {
17  
18      /***
19       * Starts the bean. This initializes and causes the object to begin functioning.
20       * Should not happen automatically when the object is created.
21       */
22      public void start();
23  
24      /***
25       * Stops the bean. This turns the object off and causes related things to be shutdown.
26       * Object should be able to be started again.
27       */
28      public void stop();
29  
30      /***
31       * Gets a reference to the bean that is being managed inside this lifecycle.
32       * @return the managed object
33       */
34      public T getManagedBean();
35  
36      /***
37       * Destroy the managed bean entirely. It will be stopped first if not stopped and cannot be
38       * started again afterwards.
39       */
40      public void destroy();
41  
42  }
